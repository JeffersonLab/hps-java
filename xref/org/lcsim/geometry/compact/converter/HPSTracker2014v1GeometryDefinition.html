16      public static class LongAxialSlotHalfModule extends HPSTracker2014GeometryDefinition.LongAxialSlotHalfModuleBase {
17  
18          public LongAxialSlotHalfModule(String name, SurveyVolume mother, AlignmentCorrection alignmentCorrection,
19                  int layer, String half) {
20              super(name, mother, alignmentCorrection, layer, half);
21              init();
22          }
23  
24          @Override
25          protected void applyGenericCoordinateSystemCorrections() {
26  
27              super.applyGenericCoordinateSystemCorrections();
28  
29              // apply 180 degree rotation around w to get hybrid on the correct side
30  
31              if (debug) {
32                  System.out.printf("%s: Coord before corrections\n%s\n", getClass().getSimpleName(), getCoord()
33                          .toString());
34                  System.out.printf("%s: box center before corrections\n%s\n", getClass().getSimpleName(), getBoxDim()
35                          .toString());
36              }
37              getCoord().rotateApache(getSlotRotation());
38  
39              if (debug) {
40                  System.out.printf("%s: Coord after corrections\n%s\n", getClass().getSimpleName(), getCoord()
41                          .toString());
42                  System.out.printf("%s: box center after corrections\n%s\n", getClass().getSimpleName(), getBoxDim()
43                          .toString());
44              }
45          }
46      }

81      @Override
82      protected LongHalfModule createLongAxialSlotHalfModule(String name, SurveyVolume mother,
83              AlignmentCorrection alignmentCorrection, int layer, String half) {
84          return new LongAxialSlotHalfModule(name, mother, alignmentCorrection, layer, half);
85      }
86  
87      @Override
88      protected LongHalfModule createLongStereoSlotHalfModule(String name, SurveyVolume mother,
89              AlignmentCorrection alignmentCorrection, int layer, String half) {
90  
91          return new LongStereoSlotHalfModule(name, mother, alignmentCorrection, layer, half);
92      }

94      /**
95       * PI rotation around generic z-axis
96       * 
97       * @return
98       */
