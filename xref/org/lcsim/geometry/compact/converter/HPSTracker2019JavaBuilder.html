10  public class HPSTracker2019JavaBuilder extends HPSTracker2014v1JavaBuilder {
11  
12      public HPSTracker2019JavaBuilder(boolean debugFlag, Element node) {
13          super(debugFlag, node);
14      }
15  
16      @Override
17      public HPSTrackerGeometryDefinition createGeometryDefinition(boolean debug, Element node) {
18          return new HPSTracker2019GeometryDefinition(debug, node);
19      }
20  
21      protected void addModule(BaseModuleBundle bundle, JavaSurveyVolume mother) {
22          if (bundle instanceof ShortModuleBundleOneSensor) {
23              addShortModuleOneSensor((ShortModuleBundleOneSensor) bundle, mother);
24          } else if (bundle instanceof ShortModuleBundle) {
25              addShortModule((ShortModuleBundle) bundle, mother);
26          } else if (bundle instanceof TestRunModuleBundle) {
27              addTestRunModule((TestRunModuleBundle) bundle, mother);
28          } else if (bundle instanceof LongModuleBundle) {
29              addLongModule((LongModuleBundle) bundle, mother);
30          } else {
31              throw new RuntimeException("The bundle is of unknown class type!");
32          }
33      }
34  
35      /**
36       * Rules for adding the LCDD module geometry.
37       * 
38       * @param bundle - module to be added
39       * @param mother - mother LCDD geometry object
40       */
41      protected void addShortModule(ShortModuleBundle bundle, JavaSurveyVolume mother) {
42          // This could perhaps be fixed if there is a relation with daughters in
43          // geometry definition?
44          // create the module
45          JavaSurveyVolume lcddM = new JavaGhostSurveyVolume(bundle.module, mother);
46          add(lcddM);
47          if (bundle.halfModuleAxialHole != null)
48              addLongHalfModule(bundle.halfModuleAxialHole, lcddM);
49          //if (bundle.halfModuleAxialSlot != null)
50          //    addLongHalfModule(bundle.halfModuleAxialSlot, lcddM);
51          // if(bundle.coldBlock!=null) add(new LCDDSurveyVolume(bundle.coldBlock,
52          // lcdd, lcddM));
53          if (bundle.halfModuleStereoHole != null)
54              addLongHalfModule(bundle.halfModuleStereoHole, lcddM);
55          //if (bundle.halfModuleStereoSlot != null)
56          //    addLongHalfModule(bundle.halfModuleStereoSlot, lcddM);
57      }
58      
59      protected void addShortModuleOneSensor(ShortModuleBundleOneSensor bundle, JavaSurveyVolume mother) {
60          // This could perhaps be fixed if there is a relation with daughters in
61          // geometry definition?
62          // create the module
63          JavaSurveyVolume lcddM = new JavaGhostSurveyVolume(bundle.module, mother);
64          add(lcddM);
65          if (bundle.halfModuleAxial != null)
66              addLongHalfModule(bundle.halfModuleAxial, lcddM);
67          if (bundle.halfModuleStereo != null)
68              addLongHalfModule(bundle.halfModuleStereo, lcddM);
69      }
