38          Element eNameTop = node.getChild(xmlTagNameTop);
39          if (eNameTop == null) {

119 
120         if (debug)
121             System.out.printf("%s: getCoordFromCompact for %s from %s\n", getClass().getSimpleName(), getName(),

123

127 
128         Element eNameTop = node.getChild(xmlTagNameTop);
129         if (eNameTop == null) {
130             throw new RuntimeException("no eName for " + xmlTagNameTop + " found in compact file");
131         }
132 
133         List<Element> eNames = eNameTop.getChildren(xmlTagName);
134         if (eNames == null) {
135             throw new RuntimeException("no eNames for " + xmlTagName + " found in compact file");
136         }
137 
138         for (Element eName : eNames) {
139 
140             if (eName.getAttributeValue("name").compareTo(getName()) == 0) {
141 
142                 Element eCoord = eName.getChild("SurveyCoord");
143 
144                 if (eCoord == null) {

146                 }
147 
148                 Element eOrigin = eCoord.getChild("origin");
