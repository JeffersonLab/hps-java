32      public void build(Volume worldVolume) {
33  
34          // set and build geometry
35          setBuilder(createGeometryDefinition(_debug, node));
36  
37          if (_builder == null)
38              throw new RuntimeException("need to set builder class before calling build!");
39  
40          if (isDebug())
41              System.out.printf("%s: build the base geometry objects\n", getClass().getSimpleName());
42  
43          _builder.build();
44  
45          if (isDebug())
46              System.out.printf("%s: DONE build the base geometry objects\n", getClass().getSimpleName());
47  
48          if (isDebug())
49              System.out.printf("%s: build the LCDD geometry objects\n", getClass().getSimpleName());
50  
51          // Go through the list of volumes to build that is created in the generic builder class
52          // TODO this is manual now since I don't have a way of knowing in the generic builder class what is a ghost volume at this point.
53          LCDDSurveyVolume trackingGeometry = new LCDDSurveyVolume(
54                  _builder.getSurveyVolume(org.lcsim.geometry.compact.converter.HPSTrackerGeometryDefinition.TrackingVolume.class),
55                  worldVolume);
56          add(trackingGeometry);
57          baseSurveyVolume = new LCDDSurveyVolume(_builder.getSurveyVolume(TrackerEnvelope.class), lcdd, trackingGeometry);
58          add(baseSurveyVolume);
59          LCDDSurveyVolume basePlateGeometry = new LCDDSurveyVolume(_builder.getSurveyVolume(BasePlate.class), lcdd,
60                  baseSurveyVolume);
61          add(basePlateGeometry);
62          // TODO I don't think this c-support has any use at all since the coordinates of it has been already used in the builder. Should remove?
63          LCDDSurveyVolume cSupportGeometry = new LCDDGhostSurveyVolume(_builder.getSurveyVolume(CSupport.class),
64                  baseSurveyVolume);
65          add(cSupportGeometry);
66          LCDDSurveyVolume supportBottomGeometry = new LCDDGhostSurveyVolume(
67                  _builder.getSurveyVolume(SupportBottom.class), baseSurveyVolume);
68          add(supportBottomGeometry);
69          LCDDSurveyVolume supportPlateBottomGeometry = new LCDDSurveyVolume(
70                  _builder.getSurveyVolume(SupportPlateBottom.class), lcdd, baseSurveyVolume);
71          add(supportPlateBottomGeometry);
72          LCDDSurveyVolume supportTopGeometry = new LCDDGhostSurveyVolume(_builder.getSurveyVolume(SupportTop.class),
73                  baseSurveyVolume);
74          add(supportTopGeometry);
75          LCDDSurveyVolume supportPlateTopGeometry = new LCDDSurveyVolume(
76                  _builder.getSurveyVolume(SupportPlateTop.class), lcdd, baseSurveyVolume);
77          add(supportPlateTopGeometry);
78  
79          // build modules
80  
81          if (isDebug())
82              System.out.printf("%s: build modules\n", getClass().getSimpleName());
83  
84          // Loop over all modules created
85          for (BaseModuleBundle mod : _builder.modules) {
86              TestRunModuleBundle m = (TestRunModuleBundle) mod;
87              if (isDebug()) {
88                  System.out.printf("%s: module layer %d half %s\n", getClass().getSimpleName(), m.getLayer(),
89                          m.getHalf());
90                  m.print();
91              }
92  
93              // Find the mother among the LCDD objects using its name, should probably have a better way...
94              String name_mother = m.getMother().getName();
95              LCDDSurveyVolume mother = null;
96              for (LCDDSurveyVolume g : lcddSurveyVolumes) {
97                  if (g.getName().equals(name_mother)) {
98                      mother = g;
99                      break;
100                 }
101             }
102             // Check that it had a mother
103             if (mother == null)
104                 throw new RuntimeException("Cound't find mother to module layer " + m.getLayer() + " half "
105                         + m.getHalf());
106 
107             if (isDebug())
108                 System.out.printf("%s: found mother %s for module layer %d half %s\n", getClass().getSimpleName(),
109                         mother.getName(), m.getLayer(), m.getHalf());
110 
111             // add the module to the list of objects that will be added to LCDD
112             addTestRunModule(m, mother);
113 
114         }
115 
116         if (isDebug()) {
117             System.out.printf("%s: DONE building the LCDD geometry objects\n", getClass().getSimpleName());
118             System.out.printf("%s: List of all %d LCDD geometry objects built\n", this.getClass().getSimpleName(),
119                     lcddSurveyVolumes.size());
120             for (SurveyVolumeImpl bg : lcddSurveyVolumes) {
121                 System.out.printf("-------\n%s\n", bg.toString());
122             }
123         }
124 
125         // Set visualization features
126         setVisualization();
127 
128     }

130     /**
131      * Rules for adding the LCDD module geometry.
132      * 
133      * @param bundle - module to be added
134      * @param mother - mother LCDD geometry object
135      */
136     protected void addTestRunModule(TestRunModuleBundle bundle, LCDDSurveyVolume mother) {
137         // This could perhaps be fixed if there is a relation with daughters in geometry definition?
138         // create the module
139         LCDDSurveyVolume lcddM = new LCDDSurveyVolume(bundle.module, lcdd, mother);
140         add(lcddM);
141         if (bundle.halfModuleAxial != null)
142             addTestRunHalfModule(bundle.halfModuleAxial, lcddM);
143         if (bundle.coldBlock != null)
144             add(new LCDDSurveyVolume(bundle.coldBlock, lcdd, lcddM));
145         if (bundle.halfModuleStereo != null)
146             addTestRunHalfModule((TestRunHalfModuleBundle) bundle.halfModuleStereo, lcddM);
147         // if(bundle.halfModuleAxial!=null) addHalfModule((TestRunHalfModuleBundle)bundle.halfModuleAxial,lcddM);
148         // if(bundle.coldBlock!=null) add(new LCDDSurveyVolume(bundle.coldBlock, lcdd, lcddM));
149         // if(bundle.halfModuleStereo!=null) addHalfModule((TestRunHalfModuleBundle)bundle.halfModuleStereo,lcddM);
150 
151     }
152 
153     /**
154      * Rules for adding the LCDD half module geometry.
155      * 
156      * @param bundle2 - module to be added
157      * @param mother - mother LCDD geometry object
158      */
159     protected void addTestRunHalfModule(HalfModuleBundle bundle2, LCDDSurveyVolume mother) {
160         // This could perhaps be fixed if there is a relation with daughters in geometry definition?
161         TestRunHalfModuleBundle bundle = (TestRunHalfModuleBundle) bundle2;
162 
163         // create the half-module
164         LCDDSurveyVolume lcddHM = new LCDDSurveyVolume(bundle.halfModule, lcdd, mother);
165         add(lcddHM);
166         // create the sensor
167         LCDDSurveyVolume lcddS = new LCDDSurveyVolume(bundle.sensor, lcdd, lcddHM);
168         add(lcddS);
169         // create the active sensor
170         LCDDSurveyVolume lcddAS = new LCDDSurveyVolume(bundle.activeSensor, lcdd, lcddS);
171         add(lcddAS);
172         // create the lamination
173         LCDDSurveyVolume lcddL = new LCDDSurveyVolume(bundle.lamination, lcdd, lcddHM);
174         add(lcddL);
175         // create the carbon fiber frame
176         LCDDSurveyVolume lcddCF = new LCDDSurveyVolume(bundle.carbonFiber, lcdd, lcddHM);
177         add(lcddCF);
178         // create the hybrid frame
179         LCDDSurveyVolume lcddH = new LCDDSurveyVolume(bundle.hybrid, lcdd, lcddHM);
180         add(lcddH);
181 
182     }
