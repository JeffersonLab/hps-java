22  /**
23   * LCDD model for the HPS inner ECal.
24   * 
25   * Coordinate System defined as follows:<br>
26   * 
27   * <ul>
28   * <li>Beam travels in +X direction.</li>
29   * <li>Magnetic field is in -Z.</li>
30   * <li>Y is the beam bend plane.</li>
31   * </ul>
32   * 
33   * The <dimensions> element defines the crystal HALF dimensions: x1, x2, y1, y2, and z.<br>
34   * 
35   * The <layout> element defines the number of crystals and (optionally) crystals to be left out.<br>
36   * 
37   * <ul>
38   * <li><b>beamgap</b> - offset from the beamline in the Y coordinate of the top and bottom halves</li>
39   * <li><b>beamgapTop</b> - offset from the beamline in the Y coordinate for the top half (defaults to <b>beamgap</b>)</li>
40   * <li><b>beamgapBottom</b> - offset from the beamline in the Y coordinate for the bottom half (defaults to <b>beamgap</b>)</li>
41   * <li><b>nx</b> - number of crystals in X</li>
42   * <li><b>ny</b> - number of crystals in Y along each side of beam</li>
43   * <li><b>dface</b> - distance from origin to the face of the calorimeter along Z</li>
44   * </ul>
45   * 
46   * Under the layout element, <remove> tags can be included to exclude crystal placement by range. This element
47   * has the following parameters.<br>
48   * 
49   * <ul>
50   * <li><b>ixmin</b> - minimum x index to exclude (inclusive)</li>
51   * <li><b>ixmax</b> - maximum x index to exclude (inclusive)</li>
52   * <li><b>iymin</b> - minimum y index to exclude (inclusive)</li>
53   * <li><b>iymax</b> - maximum y index to exclude (inclusive)</li>
54   * </ul>
55   * 
56   * To be excluded, a crystal's ID must pass all four of these min/max checks.<br>
57   */
58  public class HPSEcal3 extends LCDDSubdetector {
59      // Tolerance factor for moving crystals to appropriate place in mom volume.
60      static final double tolerance = 0.0;
61  
62      // Tolerance factor for separating crystals to avoid overlaps.
63      static final double crystalToleranceY = 0.35;
64      static final double crystalToleranceX = 0.2;
65  
66      // Margin for mother volume.
67      static final double margin = 1.1;
68  
69      List<CrystalRange> ranges = new ArrayList<CrystalRange>();
70  
71      private static class CrystalRange {
72          int ixmin;
73          int ixmax;
74          int iymin;
75          int iymax;
76  
77          CrystalRange(Element elem) throws Exception {
78              ixmin = ixmax = iymin = iymax = 0;
79  
80              if (elem.getAttribute("ixmin") != null) {
81                  ixmin = elem.getAttribute("ixmin").getIntValue();
82              } else {
83                  throw new RuntimeException("Missing ixmin parameter.");
84              }
85  
86              if (elem.getAttribute("ixmax") != null) {
87                  ixmax = elem.getAttribute("ixmax").getIntValue();
88              } else {
89                  throw new RuntimeException("Missing ixmax parameter.");
90              }
91  
92              if (elem.getAttribute("iymin") != null) {
93                  iymin = elem.getAttribute("iymin").getIntValue();
94              } else {
95                  throw new RuntimeException("Missing ixmax parameter.");
96              }
97  
98              if (elem.getAttribute("iymax") != null) {
99                  iymax = elem.getAttribute("iymax").getIntValue();
100             } else {
101                 throw new RuntimeException("Missing iymax parameter.");
102             }
103         }
104     }
105 
106     private boolean checkRange(int ix, int iy, List<CrystalRange> ranges) {
107         if (ranges.size() == 0)
108             return true;
109         for (CrystalRange range : ranges) {
110             if ((ix >= range.ixmin && ix <= range.ixmax) && ((iy >= range.iymin) && (iy <= range.iymax))) {
111                 return false;
112             }
113 
114         }
115         return true;
116     }

122     void addToLCDD(LCDD lcdd, SensitiveDetector sens) throws JDOMException {
123         if (sens == null)
124             throw new RuntimeException("SensitiveDetector parameter points to null.");
125 
126         // Crystal dimensions.
127         Element dimensions = node.getChild("dimensions");
128         double dx1 = dimensions.getAttribute("x1").getDoubleValue();
129         double dx2 = dimensions.getAttribute("x2").getDoubleValue();
130         double dy1 = dimensions.getAttribute("y1").getDoubleValue();
131         double dy2 = dimensions.getAttribute("y2").getDoubleValue();
132         double dz = dimensions.getAttribute("z").getDoubleValue();
133 
134         int system = this.getSystemID();
135 
136         // Crystal material.
137         Element mat = node.getChild("material");
138         String materialName = mat.getAttributeValue("name");
139 
140         // Layout parameters.
141         Element layout = node.getChild("layout");
142         double beamgap = 0;
143         if (layout.getAttribute("beamgap") != null) {
144             beamgap = layout.getAttribute("beamgap").getDoubleValue();
145         } else {
146             if (layout.getAttribute("beamgapTop") == null || layout.getAttribute("beamgapBottom") == null) {
147                 throw new RuntimeException("Missing beamgap parameter in layout element, and beamgapTop or beamgapBottom was not provided.");
148             }
149         }
150         double beamgapTop = beamgap;
151         if (layout.getAttribute("beamgapTop") != null) {
152             beamgapTop = layout.getAttribute("beamgapTop").getDoubleValue();
153         }
154         double beamgapBottom = beamgap;
155         if (layout.getAttribute("beamgapBottom") != null) {
156             beamgapBottom = layout.getAttribute("beamgapBottom").getDoubleValue();
157         } 
158         int nx = layout.getAttribute("nx").getIntValue();
159         int ny = layout.getAttribute("ny").getIntValue();
160         double dface = layout.getAttribute("dface").getDoubleValue();
161         
162         double tdx, tdy, tdz;        
163         double bdx, bdy, bdz;
164         tdx = tdy = tdz = bdx = bdy = bdz = 0.;
165         Element topElement = layout.getChild("top");
166         Element bottomElement = layout.getChild("bottom");
167         if (topElement != null) {
168             if (topElement.getAttribute("dx") != null)
169                 tdx = topElement.getAttribute("dx").getDoubleValue();
170             if (topElement.getAttribute("dy") != null)
171                 tdy = topElement.getAttribute("dy").getDoubleValue();
172             if (topElement.getAttribute("dz") != null)
173                 tdz = topElement.getAttribute("dz").getDoubleValue();
174         }
175         if (bottomElement != null) {
176             if (bottomElement.getAttribute("dx") != null)
177                 bdx = bottomElement.getAttribute("dx").getDoubleValue();
178             if (bottomElement.getAttribute("dy") != null)
179                 bdy = bottomElement.getAttribute("dy").getDoubleValue();
180             if (bottomElement.getAttribute("dz") != null)
181                 bdz = bottomElement.getAttribute("dz").getDoubleValue();
182         }
183 
184         // Setup range of indices to be skipped.
185         for (Object obj : layout.getChildren("remove")) {
186             Element remove = (Element) obj;
187             try {
188                 ranges.add(new CrystalRange(remove));
189             } catch (Exception x) {
190                 throw new RuntimeException(x);
191             }
192         }
193 
194         // Setup crystal logical volume.
195         Trapezoid crystalTrap = new Trapezoid("crystal_trap", dx1, dx2, dy1, dy2, dz);
196         Volume crystalLogVol = new Volume("crystal_volume", crystalTrap, lcdd.getMaterial(materialName));
197         crystalLogVol.setSensitiveDetector(sens);
198 
199         // Set vis attributes on crystal log vol.
200         setVisAttributes(lcdd, this.getNode(), crystalLogVol);
201 
202         // Add shape and log vol to lcdd.
203         lcdd.add(crystalTrap);
204         lcdd.add(crystalLogVol);
205 
206         // Place crystals in world volume.
207         Volume world = lcdd.pickMotherVolume(this);
208 
209         //
210         // Now we calculate parameters for crystal placement...
211         //
212 
213         // Slope of the trapezoid side in X.
214         double sx = (dx2 - dx1) / (2 * dz);
215 
216         // Angle of the side of the trapezoid w.r.t. center line in X. Rotation about Y axis.
217         double dthetay = atan(sx);
218 
219         // Slope of the trapezoid side in Y.
220         double sy = (dy2 - dy1) / (2 * dz);
221 
222         // Angle of the side of the trapezoid w.r.t. center line in Y. Rotation about X axis.
223         double dthetax = atan(sx);
224 
225         // Distance between (virtual) angular origin and center of trapezoid in X.
226         double z0x = dx1 / sx + dz;
227 
228         // Distance between (virtual) angular origin and center of trapezoid in Y.
229         double z0y = dy1 / sy + dz;
230 
231         // Odd or even number of crystals in X.
232         boolean oddx = (nx % 2 != 0);
233 
234         // Calculate number of X for loop.
235         if (oddx) {
236             nx -= 1;
237             nx /= 2;
238         } else {
239             nx /= 2;
240         }
241 
242         double ycorrtot = 0;
243         double zcorrtoty = 0;
244 
245         Define define = lcdd.getDefine();
246 
247         for (int iy = 1; iy <= ny; iy++) {
248             double zcorrtotx = 0;
249             double xcorrtot = 0;
250 
251             int coeffy = 2 * iy - 1;
252             double thetax = coeffy * dthetax;
253             double zcorry = dy1 * (2 * sin(coeffy * dthetax));
254             double ycorr = zcorry * tan((coeffy - 1) * dthetax);
255             double ycenter = z0y * sin(coeffy * dthetax) + ycorr + ycorrtot + (crystalToleranceY * iy);
256             double thetaz = 0;
257 
258             for (int ix = 0; ix <= nx; ix++) {
259                 // Coefficient for even/odd crystal
260                 int coeffx = 2 * ix;
261                 if (!oddx) {
262                     coeffx -= 1;
263                     // For even number of crystals, the 0th is skipped.
264                     if (ix == 0)
265                         continue;
266                 }
267 
268                 // Set parameters for next crystal placement.
269                 double thetay = coeffx * dthetay;
270                 double zcorrx = dx1 * (2 * sin(coeffx * dthetay));
271                 double xcorr = zcorrx * tan((coeffx - 1) * dthetay);
272                 double xcenter = z0x * sin(coeffx * dthetay) + xcorr + xcorrtot + (crystalToleranceX * ix);
273                 double zcenter = z0y * (cos(coeffy * dthetax) - 1) + z0x * (cos(coeffx * dthetay) - 1) + zcorrx + zcorrtotx + zcorry + zcorrtoty;
274                 zcenter += dz;
275 
276                 String baseName = "crystal" + ix + "-" + iy;
277 
278                 //
279                 // Bottom section.
280                 //
281 
282                 if (checkRange(ix, -iy, ranges)) {
283                     // Transform of positive bottom crystal.
284                     Position iposBot = new Position(baseName + "_pos_pos_bot", xcenter + bdx, -(beamgapBottom + ycenter + tolerance) + bdy, zcenter + tolerance + dface + bdz);
285                     
286                     //System.out.println("iposBot = " + iposBot.x() + ", " + iposBot.y() + " , " + iposBot.z() + " --> " + ix + ", " + -iy);
287                     Rotation irotBot = new Rotation(baseName + "_rot_pos_bot", -thetax, -thetay, thetaz);
288                     define.addPosition(iposBot);
289                     define.addRotation(irotBot);
290 
291                     // Place positive crystal.
292                     PhysVol posCrystalPlacementBot = new PhysVol(crystalLogVol, world, iposBot, irotBot);
293 
294                     // Add volume IDs.
295                     posCrystalPlacementBot.addPhysVolID("system", system);
296                     posCrystalPlacementBot.addPhysVolID("ix", ix);
297                     posCrystalPlacementBot.addPhysVolID("iy", -iy);
298                 }
299 
300                 // Reflection to negative.
301                 if (ix != 0) {
302                     if (checkRange(-ix, -iy, ranges)) {
303                         // Transform of negative.
304                         Position iposnegBot = new Position(baseName + "_pos_neg_bot", -xcenter + bdx, -(beamgapBottom + ycenter + tolerance) + bdy, zcenter + tolerance + dface + bdz);
305                         //System.out.println("iposnegBot = " + iposnegBot.x() + ", " + iposnegBot.y() + " , " + iposnegBot.z() + " --> " + -ix + ", " + -iy);
306                         Rotation irotnegBot = new Rotation(baseName + "_rot_neg_bot", -thetax, thetay, thetaz);
307 
308                         define.addPosition(iposnegBot);
309                         define.addRotation(irotnegBot);
310 
311                         // Place negative crystal.
312                         PhysVol negCrystalPlacementBot = new PhysVol(crystalLogVol, world, iposnegBot, irotnegBot);
313 
314                         // Add volume IDs.
315                         negCrystalPlacementBot.addPhysVolID("system", system);
316                         negCrystalPlacementBot.addPhysVolID("ix", -ix);
317                         negCrystalPlacementBot.addPhysVolID("iy", -iy);
318                     }
319                 }
320 
321                 if (checkRange(ix, iy, ranges)) {
322                     // Transform of positive top crystal.
323                     Position iposTop = new Position(baseName + "_pos_pos_top", xcenter + tdx, beamgapTop + ycenter + tolerance + tdy, zcenter + tolerance + dface + tdz);
324                     //System.out.println("iposTop = " + iposTop.x() + ", " + iposTop.y() + " , " + iposTop.z() + " --> " + ix + ", " + iy);
325                     Rotation irotTop = new Rotation(baseName + "_rot_pos_top", thetax, -thetay, thetaz);
326                     define.addPosition(iposTop);
327                     define.addRotation(irotTop);
328 
329                     // Place positive top crystal.
330                     PhysVol posCrystalPlacementTop = new PhysVol(crystalLogVol, world, iposTop, irotTop);
331 
332                     // Add volume IDs.
333                     posCrystalPlacementTop.addPhysVolID("system", system);
334                     posCrystalPlacementTop.addPhysVolID("ix", ix);
335                     posCrystalPlacementTop.addPhysVolID("iy", iy);
336                 }
337 
338                 // Reflection to negative.
339                 if (ix != 0) {
340                     if (checkRange(-ix, iy, ranges)) {
341                         // Transform of negative.
342                         Position iposnegTop = new Position(baseName + "_pos_neg_top", -xcenter + tdx, beamgapTop + ycenter + tolerance + tdy, zcenter + tolerance + dface + tdz);
343                         //System.out.println("iposTop = " + iposnegTop.x() + ", " + iposnegTop.y() + " , " + iposnegTop.z() + " --> " + -ix + ", " + iy);
344                         Rotation irotnegTop = new Rotation(baseName + "_rot_neg_top", thetax, thetay, thetaz);
345 
346                         define.addPosition(iposnegTop);
347                         define.addRotation(irotnegTop);
348 
349                         // Place negative crystal.
350                         PhysVol negCrystalPlacementTop = new PhysVol(crystalLogVol, world, iposnegTop, irotnegTop);
351 
352                         // Add volume IDs.
353                         negCrystalPlacementTop.addPhysVolID("system", system);
354                         negCrystalPlacementTop.addPhysVolID("ix", -ix);
355                         negCrystalPlacementTop.addPhysVolID("iy", iy);
356                     }
357                 }
358 
359                 // Increment running X and Z totals and include tolerance to avoid overlaps.
360                 xcorrtot += xcorr;
361                 zcorrtotx += zcorrx;
362             }
363 
364             // Increment running Y totals.
365             ycorrtot += ycorr;
366             zcorrtoty += zcorry;
367         }
368     }
