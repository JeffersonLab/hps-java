66      /**
67       * Build the LCDD for the subdetector.
68       * 
69       * @param lcdd - the LCDD file being created.
70       * @param sens - the SD for this subdetector.
71       */
72  
73      public void addToLCDD(LCDD lcdd, SensitiveDetector sens) throws JDOMException {
74  
75          // / General comments
76          // Roll: rotation around x
77          // pitch: rotation around y
78          // yaw: rotation around z
79  
80          // kinematic mounts:
81          // ball (constraints x,y,z)
82          // vee (constraints pitch & yaw)
83          // flat (constraints roll)
84          // /
85          
86          /**
87           * The sensitive detector for reading out hits from the inactive region of the
88           * sensor is setup here, if necessary.  --JM
89           */
90          if (this.inactiveSiReadoutName != null) {            
91              Tracker tracker = new Tracker(this.inactiveSiReadoutName + "_det");
92              tracker.setHitsCollection(this.inactiveSiReadoutName);
93              Element element = new Element("idspecref");
94              element.setAttribute("ref", this.inactiveSiReadoutName);
95              tracker.addContent(element);
96              lcdd.addSensitiveDetector(tracker);
97              this.inactiveSiTracker = tracker;
98          }
99  
100         // ID of the detector.
101         int id = node.getAttribute("id").getIntValue();
102 
103         // Name of the detector.
104         String detector_name = node.getAttributeValue("name");
105 
106         if (_debug)
107             System.out.printf("%s: detector id %d name %s", getClass().getSimpleName(), id, detector_name);
108 
109         // Pick the mother volume (tracking volume).
110         Volume trackingVolume = lcdd.pickMotherVolume(this);
111 
112         if (_debug)
113             System.out.printf("%s: setup and build the LCDD geometry\n", getClass().getSimpleName());
114 
115         // setup and build the LCDD geometry
116         builder = initializeBuilder(lcdd, sens);
117         // builder = new HPSTestRunTracker2014LCDDBuilder(_debug,node,lcdd,sens);
118 
119         // builder.setLCDD(lcdd);
120         // builder.setSensitiveDetector(sens);
121         builder.build(trackingVolume);
122 
123         if (_debug)
124             System.out.printf("%s: DONE setup and build the LCDD geometry\n", getClass().getSimpleName());
125 
126         if (buildBeamPlane) {
127             makeBeamPlane(trackingVolume, lcdd, sens);
128         }
129 
130         // Actually build the LCDD
131         setupPhysicalVolumes();
132 
133     }

135     /**
136      * Top function to add objects to the LCDD file using the geometry builder class.
137      */
138     protected void setupPhysicalVolumes() {
139 
140         if (_debug)
141             System.out.printf("%s: buildLCDD\n", getClass().getSimpleName());
142 
143         // Get a reference to the LCDD
144         LCDD lcdd = builder.getLCDD();
145         SensitiveDetector sd = builder.getSensitiveDetector();
146 
147         // Reference to the top level object in the builder class
148         // In this case it is the base volume holding the entire tracker
149         LCDDSurveyVolume lcddObj = (LCDDSurveyVolume) builder.getBaseLCDD();
150 
151         // Add the base volume and all its daughters to the LCDD
152         setupPhysicalVolumes(lcddObj, lcdd, sd);
153 
154         if (_debug)
155             System.out.printf("%s: buildLCDD DONE\n", getClass().getSimpleName());
156 
157     }

159     /**
160      * Add a @LCDDBaseGeom geometry object to the LCDD file.
161      * 
162      * @param lcddObj to add
163      * @param lcdd file
164      */
165     private void setupPhysicalVolumes(LCDDSurveyVolume lcddObj, LCDD lcdd, SensitiveDetector sd) {
166 
167         if (_debug)
168             System.out.printf("%s: adding %s to LCDD\n", getClass().getSimpleName(), lcddObj.getName());
169 
170         boolean validLCDD = true;
171 
172         if (lcddObj instanceof LCDDGhostSurveyVolume) {
173 
174             if (_debug)
175                 System.out.printf("%s: %s is a ghost volume, don't add to LCDD\n", getClass().getSimpleName(),
176                         lcddObj.getName());
177             validLCDD = false;
178 
179         } else {
180 
181             // Special case for top level volume which is a non-ghost but is already there.
182             if (lcddObj.getName().contains("tracking")) {
183 
184                 if (_debug)
185                     System.out.printf("%s: %s is the tracking volume, don't add to LCDD\n", getClass().getSimpleName(),
186                             lcddObj.getName());
187                 validLCDD = false;
188 
189             } else {
190 
191                 // X-check
192                 if (lcddObj instanceof LCDDGhostSurveyVolume)
193                     throw new RuntimeException("trying to add a ghost volume (" + lcddObj.getName() + ") to LCDD!?");
194 
195                 // add box, pos, rotation and create phys volume
196                 lcdd.add(lcddObj.getBox());
197                 lcdd.add(lcddObj.getPos());
198                 lcdd.add(lcddObj.getRot());
199                 lcddObj.buildPhysVolume();
200 
201                 // setup the properties of the phys volume
202                 try {
203                     setPhysicalVolumeProperties(lcddObj, sd);
204                 } catch (DataConversionException e) {
205                     e.printStackTrace();
206                 }
207             }
208         }
209 
210         // add daughters to this volume
211 
212         if (_debug)
213             System.out.printf("%s: add %d daughters to %s\n", getClass().getSimpleName(),
214                     lcddObj.getDaughters().size(), lcddObj.getName());
215 
216         for (LCDDSurveyVolume daughter : lcddObj.getDaughters()) {
217             setupPhysicalVolumes(daughter, lcdd, sd);
218         }
219 
220         // finally add volume
221         if (validLCDD) {
222             if (_debug)
223                 System.out.printf("%s: adding volume %s\n", getClass().getSimpleName(), lcddObj.getName());
224             if (!lcddObj.getVisName().isEmpty()) {
225                 if (_debug)
226                     System.out.printf("%s: set vis %s for volume %s\n", getClass().getSimpleName(),
227                             lcddObj.getVisName(), lcddObj.getName());
228                 lcddObj.getVolume().setVisAttributes(lcdd.getVisAttributes(lcddObj.getVisName()));
229                 if (_debug)
230                     if (lcdd.getVisAttributes(lcddObj.getVisName()) != null) {
231                         System.out.printf("%s: found vis %s\n", getClass().getSimpleName(),
232                                 lcdd.getVisAttributes(lcddObj.getVisName()).getAttributeValue("name"));
233                     } else {
234                         System.out.printf("%s: vis not found\n", getClass().getSimpleName());
235                     }
236             }
237             lcdd.add(lcddObj.getVolume());
238         } else {
239             if (_debug)
240                 System.out.printf("%s: don't add volume %s\n", getClass().getSimpleName(), lcddObj.getName());
241         }
242         if (_debug)
243             System.out.printf("%s: DONE adding %s\n", getClass().getSimpleName(), lcddObj.getName());
244     }

246     /**
247      * Set properties of the physical volume.
248      * 
249      * @param surveyVolume
250      */
251     private void setPhysicalVolumeProperties(LCDDSurveyVolume surveyVolume, SensitiveDetector sd)
252             throws DataConversionException {
253                        
254         if (_debug)
255             System.out.printf("%s: setPhysVolumeProperties for name %s\n", getClass().getSimpleName(),
256                     surveyVolume.getName());                    
257 
258         String name = surveyVolume.getName();
259         if (HPSTrackerBuilder.isHalfModule(surveyVolume.getName())) {
260             setHalfModulePhysicalVolumeProperties(surveyVolume);            
261         } else if (HPSTrackerBuilder.isActiveSensor(surveyVolume.getName())) {
262             setActiveSensorPhysicalVolumeProperties(surveyVolume, sd);
263         } else if (HPSTrackerBuilder.isSensor(surveyVolume.getName())) {
264             setSensorPhysicalVolumeProperties(surveyVolume);
265         } else if (name.endsWith("lamination")) {
266             surveyVolume.getPhysVolume().addPhysVolID("component", 2);
267         } else if (name.endsWith("cf")) {
268             surveyVolume.getPhysVolume().addPhysVolID("component", 1);
269         } else if (name.endsWith("hybrid")) {
270             surveyVolume.getPhysVolume().addPhysVolID("component", 3);
271         }
272 
273         if (_debug) {
274             System.out.printf("%s: %d physvolid's\n", getClass().getSimpleName(), surveyVolume.getPhysVolume()
275                     .getChildren("physvolid").size());
276             // geomObj.getPhysVolume().getChildren("physvolid field_name="sensor" value="0"")
277             for (Iterator i = surveyVolume.getPhysVolume().getChildren("physvolid").iterator(); i.hasNext();) {
278                 Element e = (Element) i.next();
279                 System.out.printf("%s: %s %d\n", getClass().getSimpleName(), e.getAttributeValue("field_name"), e
280                         .getAttribute("value").getIntValue());
281             }
282 
283             if (_debug)
284                 System.out.printf("%s: DONE setPhysVolumeProperties for name %s\n", getClass().getSimpleName(),
285                         surveyVolume.getName());
286 
287         }
288 
289     }

291     private void setSensorPhysicalVolumeProperties(LCDDSurveyVolume surveyVolume) {
292         surveyVolume.getPhysVolume().addPhysVolID("component", 0);
293     }

295     private void setActiveSensorPhysicalVolumeProperties(LCDDSurveyVolume surveyVolume, SensitiveDetector sd) {
296         surveyVolume.getPhysVolume().addPhysVolID("sensor", 0);
297         surveyVolume.getVolume().setSensitiveDetector(sd);
298         
299         /**
300          * Assign inactive Si detector to half-module, which must happen after the active sensor has been processed so
301          * that XML elements appear in the correct order defined by the schema.  --JM
302          */ 
303         if (this.inactiveSiTracker != null) {
304             //System.out.println("Setting sens det '" + this.inactiveSiReadoutName 
305             //        + "' on inactive Si volume '" + surveyVolume.getVolume().getName() + "'");
306             surveyVolume.getPhysMother().getVolume().setSensitiveDetector(this.inactiveSiTracker);
307         }

312     private void setHalfModulePhysicalVolumeProperties(LCDDSurveyVolume surveyVolume) throws DataConversionException {
313         PhysVol physVol = surveyVolume.getPhysVolume();
314         int sysId = node.getAttribute("id").getIntValue();
315 
316         // use the old definition of layer number to be consistent
317         int layer = builder._builder.getOldGeomDefLayerFromVolumeName(surveyVolume.getName());
318         if (_debug)
319             System.out.printf("%s: physVolId layer = %d (compare with new layer %d)\n", getClass().getSimpleName(),
320                     layer, HPSTrackerBuilder.getLayerFromVolumeName(surveyVolume.getName()));
321 
322         // Find the module number
323         int moduleNumber = getModuleNumber(surveyVolume.getName());
324 
325         physVol.addPhysVolID("system", sysId);
326         physVol.addPhysVolID("barrel", 0);
327         surveyVolume.getPhysVolume().addPhysVolID("layer", layer);
328         surveyVolume.getPhysVolume().addPhysVolID("module", moduleNumber);
329     }

331     protected void makeBeamPlane(Volume motherVolume, LCDD lcdd, SensitiveDetector sens) throws JDOMException {
332         Hep3Vector ball_pos_beamplane = new BasicHep3Vector(-1.0 * beamPlaneWidth / 2.0, 0.0, beamPlaneLength / 2.0);
333         Hep3Vector vee_pos_beamplane = new BasicHep3Vector(ball_pos_beamplane.x() + beamPlaneWidth,
334                 ball_pos_beamplane.y(), ball_pos_beamplane.z());
335         Hep3Vector flat_pos_beamplane = new BasicHep3Vector(ball_pos_beamplane.x(), ball_pos_beamplane.y(),
336                 ball_pos_beamplane.z() - beamPlaneLength / 2.0);
337         makeBeamPlane(motherVolume, ball_pos_beamplane, vee_pos_beamplane, flat_pos_beamplane, lcdd, sens);
338     }

340     protected void makeBeamPlane(Volume motherVolume, Hep3Vector ball_pos_base_plate, Hep3Vector vee_pos_base_plate,
341             Hep3Vector flat_pos_base_plate, LCDD lcdd, SensitiveDetector sens) throws JDOMException {
342 
343         if (_debug) {
344             System.out.println("--- makeBeamPlane ----");
345 
346         }
347 
348         // create the coordinate system of the beam plane in the tracking volume
349         // since this is a dummy volume it is based on the position of the base plate coordinate system
350         // width - u
351         // length - v
352         // thickness - w
353         Hep3Vector ball_pos_beamplane = ball_pos_base_plate;
354         Hep3Vector vee_pos_beamplane = vee_pos_base_plate;
355         Hep3Vector flat_pos_beamplane = flat_pos_base_plate;
356         SurveyCoordinateSystem beamplane_coord = new SurveyCoordinateSystem(ball_pos_beamplane, vee_pos_beamplane,
357                 flat_pos_beamplane);
358         Transform3D trans_beamplane_to_tracking = beamplane_coord.getTransformation();
359 
360         String volName = "beamPlaneVol";
361         // Box box = new Box(volName + "Box", HPSTestRunTracker2014GeometryDefinition.TrackerEnvelope.base_width,
362         // HPSTestRunTracker2014GeometryDefinition.TrackerEnvelope.base_length*2, beamPlaneThickness);
363         Box box = new Box(volName + "Box", beamPlaneWidth, beamPlaneLength * 2, beamPlaneThickness);
364         lcdd.add(box);
365         Volume volume = new Volume(volName + "_volume", box, lcdd.getMaterial("Vacuum"));
366 
367         if (_debug) {
368             System.out.println(String.format("ball_pos_beamplane %s", ball_pos_beamplane.toString()));
369             System.out.println(String.format("vee_pos_beamplane %s", vee_pos_beamplane.toString()));
370             System.out.println(String.format("flat_pos_beamplane %s", flat_pos_beamplane.toString()));
371             System.out.println(String.format("beamplane_coord:\n%s", beamplane_coord.toString()));
372         }
373 
374         // Find distance to center in the local coordinate system
375         Hep3Vector box_center_base_local = new BasicHep3Vector(beamPlaneWidth / 2.0, beamPlaneLength / 2.0,
376                 beamPlaneThickness / 2.0);
377 
378         // translate to the mother coordinate system
379         Hep3Vector box_center_base = trans_beamplane_to_tracking.transformed(box_center_base_local);
380 
381         if (_debug) {
382             System.out.println(String.format("box_center_base_local  %s", box_center_base_local.toString()));
383             System.out.println(String.format("box_center_base        %s", box_center_base.toString()));
384         }
385 
386         // Create the LCDD position
387         Position pos = new Position(volName + "_position", box_center_base.x(), box_center_base.y(),
388                 box_center_base.z());
389 
390         // Find LCDD Euler rotation angles from coordinate system unit vectors
391         Hep3Vector lcdd_rot_angles = TransformationUtils.getCardanAngles(beamplane_coord.v(), beamplane_coord.w(),
392                 new BasicHep3Vector(0, 1, 0), new BasicHep3Vector(0, 0, 1));
393         Rotation rot = new Rotation(volName + "_rotation", lcdd_rot_angles.x(), lcdd_rot_angles.y(),
394                 lcdd_rot_angles.z());
395         lcdd.add(pos);
396         lcdd.add(rot);
397 
398         // Create the physical volume
399         PhysVol basePV = new PhysVol(volume, motherVolume, pos, rot);
400         if (_debug) {
401             System.out.println("Created physical vomume " + basePV.getName());
402         }
403 
404         volume.setVisAttributes(lcdd.getVisAttributes("BeamPlaneVis"));
405 
406         lcdd.add(volume);
407     }
