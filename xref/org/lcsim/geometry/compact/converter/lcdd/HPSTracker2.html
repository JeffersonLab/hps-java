37      void addToLCDD(LCDD lcdd, SensitiveDetector sd) throws JDOMException {
38  
39          // Get parameters.
40          int sysId = node.getAttribute("id").getIntValue();
41          String subdetName = node.getAttributeValue("name");
42          vacuum = lcdd.getMaterial("Vacuum");
43  
44          // Create module logical volumes.
45          createModules(lcdd, sd);
46  
47          // Create module placements in tracking volume.
48          createModulePlacements(lcdd, sysId, subdetName);
49      }

51      // Place modules within the tracking volume.
52      private void createModulePlacements(LCDD lcdd, int sysId, String subdetName) throws DataConversionException {
53          // Volume trackingVolume = lcdd.getTrackingVolume();
54          Volume momVolume = lcdd.pickMotherVolume(this);
55          // Loop over layers.
56          for (Iterator i = node.getChildren("layer").iterator(); i.hasNext();) {
57              Element layerElement = (Element) i.next();
58              int layerNumber = layerElement.getAttribute("id").getIntValue();
59              // Loop over modules within layer.
60              for (Iterator j = layerElement.getChildren("module_placement").iterator(); j.hasNext();) {
61  
62                  Element modulePlacementElement = (Element) j.next();
63                  String moduleName = modulePlacementElement.getAttributeValue("name");
64                  int moduleNumber = modulePlacementElement.getAttribute("id").getIntValue();
65  
66                  // Get the position and rotation parameters.
67                  double x, y, z, rx, ry, rz;
68                  x = y = z = rx = ry = rz = 0;
69  
70                  // If not specified, default value of zero will be used for each parameter.
71                  if (modulePlacementElement.getAttribute("x") != null) {
72                      x = modulePlacementElement.getAttribute("x").getDoubleValue();
73                  }
74                  if (modulePlacementElement.getAttribute("y") != null) {
75                      y = modulePlacementElement.getAttribute("y").getDoubleValue();
76                  }
77                  if (modulePlacementElement.getAttribute("z") != null) {
78                      z = modulePlacementElement.getAttribute("z").getDoubleValue();
79                  }
80                  if (modulePlacementElement.getAttribute("rx") != null) {
81                      rx = modulePlacementElement.getAttribute("rx").getDoubleValue();
82                  }
83                  if (modulePlacementElement.getAttribute("ry") != null) {
84                      ry = modulePlacementElement.getAttribute("ry").getDoubleValue();
85                  }
86                  if (modulePlacementElement.getAttribute("rz") != null) {
87                      rz = modulePlacementElement.getAttribute("rz").getDoubleValue();
88                  }
89  
90                  // Place the module with position and rotation from above.
91                  String modulePlacementName = subdetName + "_" + moduleName + "_layer" + layerNumber + "_module"
92                          + moduleNumber;
93                  Position p = new Position(modulePlacementName + "_position", x, y, z);
94                  Rotation r = new Rotation(modulePlacementName + "_rotation", rx, ry, rz);
95                  lcdd.add(p);
96                  lcdd.add(r);
97                  PhysVol modulePhysVol = new PhysVol(modules.get(moduleName), momVolume, p, r);
98  
99                  // Add identifier values to the placement volume.
100                 modulePhysVol.addPhysVolID("system", sysId);
101                 modulePhysVol.addPhysVolID("barrel", 0);
102                 modulePhysVol.addPhysVolID("layer", layerNumber);
103                 modulePhysVol.addPhysVolID("module", moduleNumber);
104             }
105         }
106     }

108     // Create the module logical volumes.
109     private void createModules(LCDD lcdd, SensitiveDetector sd) {
110         for (Iterator i = node.getChildren("module").iterator(); i.hasNext();) {
111             Element module = (Element) i.next();
112             String moduleName = module.getAttributeValue("name");
113             moduleParameters.put(moduleName, new ModuleParameters(module));
114             modules.put(moduleName, makeModule(moduleParameters.get(moduleName), sd, lcdd));
115         }
116     }

118     private Volume makeModule(ModuleParameters params, SensitiveDetector sd, LCDD lcdd) {
119         double thickness = params.getThickness();
120         double x, y;
121         // x = params.getDimension(0);
122         // y = params.getDimension(1);
123         y = params.getDimension(0); // Y is in X plane in world coordinates.
124         x = params.getDimension(1); // X is in Y plane in world coordinates.
125         // System.out.println("making module with x = " + x + " and y = " + y);
126         Box box = new Box(params.getName() + "Box", x, y, thickness);
127         lcdd.add(box);
128 
129         Volume moduleVolume = new Volume(params.getName() + "Volume", box, vacuum);
130         makeModuleComponents(moduleVolume, params, sd, lcdd);
131         lcdd.add(moduleVolume);
132 
133         if (params.getVis() != null) {
134             moduleVolume.setVisAttributes(lcdd.getVisAttributes(params.getVis()));
135         }
136 
137         return moduleVolume;
138     }

140     private void makeModuleComponents(Volume moduleVolume, ModuleParameters moduleParameters, SensitiveDetector sd,
141             LCDD lcdd) {
142         Box envelope = (Box) lcdd.getSolid(moduleVolume.getSolidRef());
143 
144         double moduleX = envelope.getX();
145         double moduleY = envelope.getY();
146 
147         double posZ = -moduleParameters.getThickness() / 2;
148 
149         String moduleName = moduleVolume.getVolumeName();
150 
151         int sensor = 0;
152         for (ModuleComponentParameters component : moduleParameters) {
153 
154             double thickness = component.getThickness();
155 
156             Material material = null;
157             try {
158                 material = lcdd.getMaterial(component.getMaterialName());
159             } catch (JDOMException except) {
160                 throw new RuntimeException(except);
161             }
162             boolean sensitive = component.isSensitive();
163             int componentNumber = component.getComponentNumber();
164 
165             posZ += thickness / 2;
166 
167             String componentName = moduleName + "_component" + componentNumber;
168 
169             Box componentBox = new Box(componentName + "Box", moduleX, moduleY, thickness);
170             lcdd.add(componentBox);
171 
172             Volume componentVolume = new Volume(componentName, componentBox, material);
173 
174             Position position = new Position(componentName + "_position", 0., 0., posZ);
175             lcdd.add(position);
176             Rotation rotation = new Rotation(componentName + "_rotation", 0., 0., 0.);
177             lcdd.add(rotation);
178 
179             PhysVol pv = new PhysVol(componentVolume, moduleVolume, position, rotation);
180             pv.addPhysVolID("component", componentNumber);
181 
182             if (sensitive) {
183                 if (sensor > 1) {
184                     throw new RuntimeException("Maximum of 2 sensors per module.");
185                 }
186 
187                 // Build a child sensor volume to allow dead areas.
188 
189                 String sensorName = componentName + "Sensor" + sensor;
190 
191                 // Flipped these around!!!
192                 double sensorX = component.getDimensionY();
193                 double sensorY = component.getDimensionX();
194 
195                 Box sensorBox = new Box(sensorName + "Box", sensorX, sensorY, thickness);
196                 lcdd.add(sensorBox);
197 
198                 Volume sensorVol = new Volume(sensorName, sensorBox, material);
199                 sensorVol.setSensitiveDetector(sd);
200                 lcdd.add(sensorVol);
201 
202                 Position sensorPosition = new Position(sensorName + "Position", 0, 0, 0);
203                 lcdd.add(sensorPosition);
204                 Rotation sensorRotation = new Rotation(sensorName + "Rotation", 0, 0, 0);
205                 lcdd.add(sensorRotation);
206 
207                 PhysVol sensorPhysVol = new PhysVol(sensorVol, componentVolume, sensorPosition, sensorRotation);
208                 sensorPhysVol.addPhysVolID("sensor", sensor);
209 
210                 ++sensor;
211             }
212 
213             // Add component volume after (possible) sensor child volume.
214             lcdd.add(componentVolume);
215 
216             // Set vis attributes of component.
217             if (component.getVis() != null) {
218                 componentVolume.setVisAttributes(lcdd.getVisAttributes(component.getVis()));
219             }
220 
221             // Step to next component placement position.
222             posZ += thickness / 2;
223         }
224     }

275     private static class ModuleParameters extends ArrayList<ModuleComponentParameters> {
276 
277         double thickness;
278         String name;
279         double dimensions[] = new double[3];
280         String vis;
281 
282         public ModuleParameters(Element element) {
283             name = element.getAttributeValue("name");
284 
285             if (element.getAttribute("vis") != null)
286                 this.vis = element.getAttribute("vis").getValue();
287 
288             // Optional dimension parameters (not always present).
289             if (element.getChild("trd") != null) {
290                 Element trd = element.getChild("trd");
291                 try {
292                     dimensions[0] = trd.getAttribute("x1").getDoubleValue();
293                     dimensions[1] = trd.getAttribute("x2").getDoubleValue();
294                     dimensions[2] = trd.getAttribute("z").getDoubleValue();
295                 } catch (DataConversionException x) {
296                     throw new RuntimeException(x);
297                 }
298             } else if (element.getChild("box") != null) {
299                 Element box = element.getChild("box");
300                 try {
301                     dimensions[0] = box.getAttribute("x").getDoubleValue();
302                     dimensions[1] = box.getAttribute("y").getDoubleValue();
303                 } catch (DataConversionException x) {
304                     throw new RuntimeException(x);
305                 }
306             }
307 
308             int cntr = 0;
309             for (Object o : element.getChildren("module_component")) {
310                 try {
311 
312                     Element e = (Element) o;
313 
314                     double thickness = e.getAttribute("thickness").getDoubleValue();
315 
316                     String materialName = e.getAttributeValue("material");
317 
318                     boolean sensitive = false;
319                     if (e.getAttribute("sensitive") != null)
320                         sensitive = e.getAttribute("sensitive").getBooleanValue();
321                     String componentVis = null;
322                     if (e.getAttribute("vis") != null)
323                         componentVis = e.getAttribute("vis").getValue();
324 
325                     // Sensors may have reduced dimensions for dead area.
326                     double x = dimensions[0]; // default
327                     double y = dimensions[1]; // default
328                     if (sensitive && e.getChild("dimensions") != null) {
329                         Element dimensions = e.getChild("dimensions");
330                         x = dimensions.getAttribute("x").getDoubleValue();
331                         y = dimensions.getAttribute("y").getDoubleValue();
332                         // System.out.println("x,y="+x+","+y);
333                     }
334                     add(new ModuleComponentParameters(x, y, thickness, materialName, cntr, sensitive, componentVis));
335                 } catch (JDOMException x) {
336                     throw new RuntimeException(x);
337                 }
338                 ++cntr;
339             }
340 
341             calculateThickness();
342         }
343 
344         void calculateThickness() {
345             thickness = 0.; // reset thickness
346             for (ModuleComponentParameters p : this) {
347                 thickness += p.getThickness();
348             }
349         }
350 
351         double getThickness() {
352             return thickness;
353         }
354 
355         String getName() {
356             return name;
357         }
358 
359         double getDimension(int i) {
360             if (i > (dimensions.length - 1) || i < 0)
361                 throw new RuntimeException("Invalid dimensions index: " + i);
362             return dimensions[i];
363         }
364 
365         String getVis() {
366             return vis;
367         }
368     }
