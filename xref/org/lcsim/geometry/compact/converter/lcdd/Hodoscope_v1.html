201         params.scintillatorDepth = getDoubleVariable(node, "scintillator_depth");
202         if(Double.isNaN(params.scintillatorDepth)) { params.scintillatorDepth = PIXEL_DEPTH; }
203         params.coverDepth = getDoubleVariable(node, "cover_depth");

348     /**
349      * Gets the value specified from the variable element of the
350      * detector compact.xml description. Note that it is required for
351      * the variable to declared in the format <br/><br/>
352      * <code>&lt;VAR_NAME value="DOUBLE_VALUE"/&gt;</code>
353      * <br/><br/>
354      * If the specified variable node does not exist as a child of
355      * the parent node <code>root</code>, a value of {@link
356      * java.lang.Double.NaN Double.NaN} is returned instead.
357      * @param root - The detector XML node which serves as a parent
358      * to the variable nodes.
359      * @param varName - The {@link java.lang.String String} name of
360      * the variables.
361      * @return Returns the variable, if the node exists. Otherwise, a
362      * value of {@link java.lang.Double.NaN Double.NaN} is returned.
363      * @throws DataConversionException Occurs if the value defined is
364      * not convertible to a <code>double</code> primitive.
365      * @throws RuntimeException Occurs if the variable node exists,
366      * but is missing the necessary value attribute.
367      */
368     private static final double getDoubleVariable(Element root, String varName) throws DataConversionException, RuntimeException {
369         // Get the value node. If it exists, attempt to access the
370         // variable. Otherwise, just return Double.NaN.
371         Element valueNode = root.getChild(varName);
372         if(valueNode != null) {
373             // Attempt to obtain the variable attribute. If it does
374             // not exist, there is a formatting problem with the
375             // detector declaration in the compact.xml. Produce an
376             // exception and alert the user.
377             Attribute valueAttribute = valueNode.getAttribute("value");
378             if(valueAttribute == null) {
379                 throw new RuntimeException(Hodoscope_v1.class.getSimpleName() + ": Node \""
380                         + varName + "\" is missing attribute \"value\".");
381             }
382             
383             // Otherwise, parse the value and store it.
384             return valueNode.getAttribute("value").getDoubleValue();
385         } else {
386             return Double.NaN;
387         }
388     }

390     private static final double[] getDoubleArrayVariable(Element root, String varName)
391             throws IllegalArgumentException, NumberFormatException {
392         // Get the value node. If it exists, attempt to access the
393         // variable. Otherwise, just return null.
394         Element valueNode = root.getChild(varName);
395         if(valueNode != null) {
396             // Attempt to obtain the variable attribute. If it does
397             // not exist, there is a formatting problem with the
398             // detector declaration in the compact.xml. Produce an
399             // exception and alert the user.
400             Attribute valueAttribute = valueNode.getAttribute("value");
401             if(valueAttribute == null) {
402                 throw new RuntimeException(Hodoscope_v1.class.getSimpleName() + ": Node \""
403                         + varName + "\" is missing attribute \"value\".");
404             }
405             
406             // The value should take the form of a comma-delimited
407             // string of doubles. First, sanitize the string to
408             // remove any whitespace. Also ensure that it contains
409             // only numbers, decimal points, and commas.
410             int entries = 1;
411             StringBuffer sanitizationBuffer = new StringBuffer();
412             for(char c : valueAttribute.getValue().toCharArray()) {
413                 if(Character.isDigit(c) || c == ',' || Character.isWhitespace(c) || c == '.') {
414                     if(c == ',') { entries++; }
415                     if(!Character.isWhitespace(c)) {
416                         sanitizationBuffer.append(c);
417                     }
418                 } else {
419                     throw new IllegalArgumentException(Hodoscope_v1.class.getSimpleName() + ": Numeric array variable \""
420                             + varName + "\" contains unsupported character \'" + Character.toString(c) + "\'.");
421                 }
422             }
423             
424             // If the input string is sanitized, prepare it for
425             // parsing.
426             String input = sanitizationBuffer.toString();
427             
428             // If the input is an empty string, return a size zero
429             // array of type double.
430             if(input.length() == 0) {
431                 return new double[0];
432             }
433             
434             // Extract all double values from the input.
435             int curIndex = 0;
436             double[] values = new double[entries];
437             StringBuffer valueBuffer = new StringBuffer();
438             for(char c : input.toCharArray()) {
439                 // All digits aside from the delimiter should be
440                 // added to the buffer for later parsing.
441                 if(c != ',') {
442                     valueBuffer.append(c);
443                 }
444                 
445                 // If the delimiter has been seen, parse the digit.
446                 else {
447                     // If there is no content in the buffer, it is
448                     // an error.
449                     if(valueBuffer.length() == 0) {
450                         throw new IllegalArgumentException(Hodoscope_v1.class.getSimpleName() + ": Numeric array variable \""
451                                 + varName + "\" is missing one or more value declarations.");
452                     }
453                     
454                     // Otherwise, attempt to parse the digit.
455                     values[curIndex] = Double.parseDouble(valueBuffer.toString());
456                     
457                     // Reset the buffer an increment the index.
458                     valueBuffer = new StringBuffer();
459                     curIndex++;
460                 }
461             }
462             
463             // There should be one digit left in the buffer at the
464             // end of the loop.
465             if(valueBuffer.length() == 0) {
466                 throw new IllegalArgumentException(Hodoscope_v1.class.getSimpleName() + ": Numeric array variable \""
467                         + varName + "\" is missing one or more value declarations.");
468             }
469             values[curIndex] = Double.parseDouble(valueBuffer.toString());
470             valueBuffer = new StringBuffer();
471             curIndex++;
472             
473             // Return the result.
474             return values;
475         } else {
476             return null;
477         }

480     /**
481      * Reads the name of a material from the specified variable node
482      * in the parent node <code>root</code> and attempts to obtain
483      * the material definition from the LCDD file <code>lcdd</code>.
484      * If it exists, the material object is returned. Otherwise, a
485      * value of <code>null</code> is returned.
486      * @param lcdd - The LCDD file object, in which the material data
487      * definition may be found.
488      * @param root - The detector XML node which serves as a parent
489      * to the variable nodes.
490      * @param varName - The {@link java.lang.String String} name of
491      * the variables.
492      * @return Returns a {@link
493      * org.lcsim.geometry.compact.converter.lcdd.util.Material
494      * Material} object if the variable requested exists. Otherwise,
495      * <code>null</code> will be returned.
496      * @throws JDOMException Occurs if there is an error accessing
497      * the material definition from the LCDD file.
498      * @throws RuntimeException  Occurs if the material declaration
499      * attribute in the XML is missing.
500      */
501     private static final Material getMaterialVariable(LCDD lcdd, Element root, String varName) throws JDOMException, RuntimeException {
502         // Get the value node. If it exists, attempt to access the
503         // variable. Otherwise, just return null.
504         Element valueNode = root.getChild(varName);
505         if(valueNode != null) {
506             // Attempt to obtain the variable attribute. If it does
507             // not exist, there is a formatting problem with the
508             // detector declaration in the compact.xml. Produce an
509             // exception and alert the user.
510             Attribute valueAttribute = valueNode.getAttribute("value");
511             if(valueAttribute == null) {
512                 throw new RuntimeException(Hodoscope_v1.class.getSimpleName() + ": Node \""
513                         + varName + "\" is missing attribute \"value\".");
514             }
515             
516             // Otherwise, parse the value and store it.
517             return lcdd.getMaterial(valueAttribute.getValue());
518         } else {
519             return null;
520         }

536      * org.lcsim.geometry.compact.converter.lcdd.Hodoscope_v1#TOP
537      * TOP} or {@link
538      * org.lcsim.geometry.compact.converter.lcdd.Hodoscope_v1#BOTTOM
539      * BOTTOM}.
540      * @param ix - The x-index for the pixel. This must be an integer
541      * value that is non-negative.
542      * @return Returns a unique identifier string.
543      */
544     private static final String getName(int layer, int topBot, int ix) {
545         return String.format("L" + (layer + 1) + (topBot == TOP ? 'T' : 'B') + 'P' + ix);
546     }
547     
548     private final void makePixel(LCDD lcdd, SensitiveDetector sens, PixelParameters params,
549             int layer, int topBot, int ix, double pixelWidth, double xShift) {
550         // Get a unique string that represents this pixel.
551         String uid = getName(layer, topBot, ix);
552         
553         // Create the geometric shapes that define the pixel. These
554         // are always rectangular prisms. The prism is then used to
555         // define a volume, which sets the material. Lastly, it is
556         // assigned the display properties for the hodoscope and is
557         // attached to the hodoscope's sensitive detector.
558         Box pixelShape = new Box("hodo_pixel_" + uid, pixelWidth, params.scintillatorHeight, params.scintillatorDepth);
559         Volume pixelVolume = new Volume("hodo_vol_" + uid, pixelShape, params.scintillatorMaterial);
560         pixelVolume.setSensitiveDetector(sens);
561         setVisAttributes(lcdd, getNode(), pixelVolume);
562         lcdd.add(pixelShape);
563         lcdd.add(pixelVolume);
564         
565         // Create the geometric shapes that define the scintillator
566         // covers. These are rectangular prisms with the same x- and
567         // y-dimensions as the scintillator.
568         Box coverShape = new Box("hodo_cover_" + uid, pixelWidth, params.scintillatorHeight, params.coverDepth);
569         Volume coverVolume = new Volume("hodo_cover_vol_" + uid, coverShape, params.coverMaterial);
570         setVisAttributes(lcdd, getNode(), coverVolume);
571         lcdd.add(coverShape);
572         lcdd.add(coverVolume);
573         
574         // Create the geometric shapes that define the scintillator
575         // side reflectors. These have a depth and height equal to
576         // the scintillator, but define their own width.
577         Box sideReflectorShape = new Box("hodo_siderefl_" + uid, params.reflectorDepth, params.scintillatorHeight,
578                 params.scintillatorDepth + (2 * params.coverDepth));
579         Volume sideReflectorVolume = new Volume("hodo_siderefl_vol_" + uid, sideReflectorShape, params.reflectorMaterial);
580         setVisAttributes(lcdd, getNode(), sideReflectorVolume);
581         lcdd.add(sideReflectorShape);
582         lcdd.add(sideReflectorVolume);
583         
584         // Create the geometric shapes that define the scintillator
585         // top reflectors. These have a depth and width equal to the
586         // scintillator, but define their own height.
587         Box topReflectorShape = new Box("hodo_toprefl_" + uid,
588                 pixelWidth + (2 * params.reflectorDepth),
589                 params.reflectorDepth,
590                 params.scintillatorDepth + (2 * params.coverDepth));
591         Volume topReflectorVolume = new Volume("hodo_toprefl_vol_" + uid, topReflectorShape, params.reflectorMaterial);
592         setVisAttributes(lcdd, getNode(), topReflectorVolume);
593         lcdd.add(topReflectorShape);
594         lcdd.add(topReflectorVolume);
595         
596         // Define the position of the crystal. Note that the position
597         // of a volume is defined as the position of its centerpoint,
598         // so it must shifted by half its width and height to account
599         // for this and place its edge at the correct position. After
600         // this, the position and default rotation defines should be
601         // must be added.
602         Position scinPos = new Position("hodo_pos_" + uid,
603                 X_SHIFT + xShift + positionValues[layer][topBot][X] + sideReflectorShape.getX() + (pixelShape.getX() / 2),
604                 (topBot == TOP ? 1 : -1) * (positionValues[layer][topBot][Y] + topReflectorShape.getY() + (pixelShape.getY() / 2)),
605                 positionValues[layer][topBot][Z] + coverShape.getZ() + (pixelShape.getZ() / 2));
606         lcdd.getDefine().addPosition(scinPos);
607         
608         // Create the physical object representing the pixel. This
609         // can also have a number of additional properties that can
610         // be attached to it.
611         PhysVol physvol = new PhysVol(pixelVolume, lcdd.pickMotherVolume(this), scinPos, PIXEL_ROTATION);
612         physvol.addPhysVolID("system", getSystemID());
613         physvol.addPhysVolID("barrel", 1);
614         physvol.addPhysVolID("layer", layer == LAYER1 ? 0 : 1);
615         physvol.addPhysVolID("ix", ix);
616         physvol.addPhysVolID("iy", topBot == TOP ? 1 : -1);
617 
618         int layerId = (layer == LAYER1 ? 1 : 2);
619         int iy = (topBot == TOP ? 1 : -1);
620         System.out.printf("%s (system/layer/ix/iy): %d/%d/%d/%d%n", uid, getSystemID(), layerId, ix, iy);
621         
622         // Define the positions of the scintillator covers. These sit
623         // both in front of and behind the scintillator.
624         Position[] coverPos = {
625             new Position("hodo_forecover_pos_" + uid, scinPos.x(), scinPos.y(),
626                     positionValues[layer][topBot][Z] + (coverShape.getZ() / 2)),
627             new Position("hodo_postcover_pos_" + uid, scinPos.x(), scinPos.y(),
628                     positionValues[layer][topBot][Z] + coverShape.getZ() + pixelShape.getZ() + (coverShape.getZ() / 2))
629         };
630         
631         // Create the physical cover objects.
632         for(Position pos : coverPos) {
633             lcdd.getDefine().addPosition(pos);
634             new PhysVol(coverVolume, lcdd.pickMotherVolume(this), pos, PIXEL_ROTATION);
635         }
636         
637         // Define the positions of the scintillator side reflectors.
638         // These sit on either side of a scintillator crystal.
639         Position[] sideReflectorPos = {
640                 new Position("hodo_sidereflR_pos_" + uid,
641                         X_SHIFT + xShift + positionValues[layer][topBot][X] + (sideReflectorShape.getX() / 2),
642                         scinPos.y(), scinPos.z()),
643                 new Position("hodo_siderefl2L_pos_" + uid,
644                         X_SHIFT + xShift + positionValues[layer][topBot][X] + sideReflectorShape.getX()
645                         + pixelShape.getX() + (sideReflectorShape.getX() / 2),
646                         scinPos.y(), scinPos.z())
647         };
648         
649         // Create the physical side reflector objects.
650         for(Position pos : sideReflectorPos) {
651             lcdd.getDefine().addPosition(pos);
652             new PhysVol(sideReflectorVolume, lcdd.pickMotherVolume(this), pos, PIXEL_ROTATION);
653         }
654         
655         // Define the positions of the scintillator top reflectors.
656         // These sit on the top and bottom of a scintillator crystal.
657         Position[] topReflectorPos = {
658                 new Position("hodo_topreflT_pos_" + uid,
659                         scinPos.x(),
660                         scinPos.y() + (params.scintillatorHeight / 2) + (params.reflectorDepth / 2),
661                         scinPos.z()),
662                 new Position("hodo_toprefl2B_pos_" + uid,
663                         scinPos.x(),
664                         scinPos.y() - (params.scintillatorHeight / 2) - (params.reflectorDepth / 2),
665                         scinPos.z())
666         };
667         
668         // Create the physical top reflector objects.
669         for(Position pos : topReflectorPos) {
670             lcdd.getDefine().addPosition(pos);
671             new PhysVol(topReflectorVolume, lcdd.pickMotherVolume(this), pos, PIXEL_ROTATION);
672         }
673     }
