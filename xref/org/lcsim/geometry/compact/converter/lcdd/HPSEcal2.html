20  /**
21   * LCDD model for the HPS inner ECal. Coordinate System Notes: Beam travels in +X direction. Magnetic field is -Z (????). Y is the beam bend plane. The dimensions element defines the crystal HALF dimensions: x1, x2, y1, y2, and z. The layout element defines the placement of the crystals.
22   * <ul>
23   * <li>beamgap - offset from the beamline in the Y coordinate</li>
24   * <li>nx - number of crystals in X</li>
25   * <li>ny - number of crystals in Y along each side of beam</li>
26   * <li>dface - distance from origin to the face of the calorimeter along X</li>
27   * </ul>
28   */
29  public class HPSEcal2 extends LCDDSubdetector {
30  
31      // Tolerance factor for moving crystals to appropriate place in mom volume.
32      static final double tolerance = 0.1;
33  
34      // Tolerance factor for separating crystals to avoid overlaps.
35      static final double crystalTolerance = 0.1;
36  
37      HPSEcal2(Element node) throws JDOMException {
38          super(node);
39      }
40  
41      void addToLCDD(LCDD lcdd, SensitiveDetector sens) throws JDOMException {
42          if (sens == null)
43              throw new RuntimeException("SensitiveDetector is null!");
44  
45          // Crystal dimensions.
46          Element dimensions = node.getChild("dimensions");
47          double dx1 = dimensions.getAttribute("x1").getDoubleValue();
48          double dx2 = dimensions.getAttribute("x2").getDoubleValue();
49          double dy1 = dimensions.getAttribute("y1").getDoubleValue();
50          double dy2 = dimensions.getAttribute("y2").getDoubleValue();
51          double dz = dimensions.getAttribute("z").getDoubleValue();
52  
53          // Crystal material.
54          Element mat = node.getChild("material");
55          String materialName = mat.getAttributeValue("name");
56  
57          // Layout parameters.
58          Element layout = node.getChild("layout");
59          double beamgap = layout.getAttribute("beamgap").getDoubleValue();
60          int nx = layout.getAttribute("nx").getIntValue();
61          int ny = layout.getAttribute("ny").getIntValue();
62          double dface = layout.getAttribute("dface").getDoubleValue();
63  
64          // Setup crystal logical volume.
65          Trapezoid crystalTrap = new Trapezoid("crystal_trap", dx1, dx2, dy1, dy2, dz);
66          Volume crystalLogVol = new Volume("crystal_volume", crystalTrap, lcdd.getMaterial(materialName));
67          crystalLogVol.setSensitiveDetector(sens);
68  
69          lcdd.add(crystalTrap);
70          lcdd.add(crystalLogVol);
71  
72          // Mother volume dimensions.
73          double mx, my, mz;
74          double margin = 1.1;
75          mx = nx * Math.max(dx2, dx1) * margin;
76          my = ny * Math.max(dy2, dy1) * margin;
77          mz = dz * margin;
78  
79          // Envelope box and logical volume for one section of ECal.
80          Box momBox = new Box("ecal_env_box", mx * 2, my * 2, mz * 2);
81          Volume momVol = new Volume("ecal_env_volume", momBox, lcdd.getMaterial("Air"));
82  
83          lcdd.add(momBox);
84          lcdd.add(momVol);
85  
86          // Slope of the trapezoid side in X.
87          double sx = (dx2 - dx1) / (2 * dz);
88  
89          // Angle of the side of the trapezoid w.r.t. center line in X. Rotation about Y axis.
90          double dthetay = atan(sx);
91  
92          // Slope of the trapezoid side in Y.
93          double sy = (dy2 - dy1) / (2 * dz);
94  
95          // Angle of the side of the trapezoid w.r.t. center line in Y. Rotation about X axis.
96          double dthetax = atan(sx);
97  
98          // Distance between (virtual) angular origin and center of trapezoid in X.
99          double z0x = dx1 / sx + dz;
100 
101         // Distance between (virtual) angular origin and center of trapezoid in Y.
102         double z0y = dy1 / sy + dz;
103 
104         // Odd or even number of crystals in X.
105         boolean oddx = (nx % 2 != 0);
106 
107         // Calculate number of X for loop.
108         if (oddx) {
109             nx -= 1;
110             nx /= 2;
111         } else {
112             nx /= 2;
113         }
114 
115         double ycorrtot = 0;
116         double zcorrtoty = 0;
117 
118         Define define = lcdd.getDefine();
119 
120         for (int iy = 1; iy <= ny; iy++) {
121             double zcorrtotx = 0;
122             double xcorrtot = 0;
123 
124             int coeffy = 2 * iy - 1;
125             double thetax = coeffy * dthetax;
126             double zcorry = dy1 * (2 * sin(coeffy * dthetax));
127             double ycorr = zcorry * tan((coeffy - 1) * dthetax);
128             double ycenter = z0y * sin(coeffy * dthetax) + ycorr + ycorrtot + (crystalTolerance * iy);
129 
130             for (int ix = 0; ix <= nx; ix++) {
131                 // Coefficient for even/odd crystal
132                 int coeffx = 2 * ix;
133                 if (!oddx) {
134                     coeffx -= 1;
135                     // For even number of crystals, the 0th is skipped.
136                     if (ix == 0)
137                         continue;
138                 }
139 
140                 // Set parameters for next crystal placement.
141                 // FIXME Need documentation here.
142                 double thetay = coeffx * dthetay;
143                 double zcorrx = dx1 * (2 * sin(coeffx * dthetay));
144                 double xcorr = zcorrx * tan((coeffx - 1) * dthetay);
145                 double xcenter = z0x * sin(coeffx * dthetay) + xcorr + xcorrtot + (crystalTolerance * ix);
146                 double zcenter = z0y * (cos(coeffy * dthetax) - 1) + z0x * (cos(coeffx * dthetay) - 1) + zcorrx
147                         + zcorrtotx + zcorry + zcorrtoty;
148                 zcenter += dz;
149 
150                 double thetaz = 0;
151 
152                 String baseName = "crystal" + ix + "-" + iy;
153 
154                 // Transform of positive.
155                 Position ipos = new Position(baseName + "_pos_pos", xcenter, ycenter - my + tolerance, zcenter - mz
156                         + tolerance);
157                 Rotation irot = new Rotation(baseName + "_rot_pos", thetax, -thetay, thetaz);
158 
159                 define.addPosition(ipos);
160                 define.addRotation(irot);
161 
162                 // Place positive crystal.
163                 PhysVol posCrystalPlacement = new PhysVol(crystalLogVol, momVol, ipos, irot);
164 
165                 // Add volume IDs.
166                 posCrystalPlacement.addPhysVolID("ix", ix);
167                 posCrystalPlacement.addPhysVolID("iy", iy);
168 
169                 // Reflection to negative.
170                 if (ix != 0) {
171                     // Transform of negative.
172                     Position iposneg = new Position(baseName + "_pos_neg", -xcenter, ycenter - my + tolerance, zcenter
173                             - mz + tolerance);
174                     Rotation irotneg = new Rotation(baseName + "_rot_neg", thetax, thetay, thetaz);
175 
176                     define.addPosition(iposneg);
177                     define.addRotation(irotneg);
178 
179                     // Place negative crystal.
180                     PhysVol negCrystalPlacement = new PhysVol(crystalLogVol, momVol, iposneg, irotneg);
181 
182                     // Add volume IDs.
183                     negCrystalPlacement.addPhysVolID("ix", -ix);
184                     negCrystalPlacement.addPhysVolID("iy", iy);
185                 }
186 
187                 // Increment running X and Z totals and include tolerance to avoid overlaps.
188                 xcorrtot += xcorr;
189                 zcorrtotx += zcorrx;
190             }
191 
192             // Increment running Y totals.
193             ycorrtot += ycorr;
194             zcorrtoty += zcorry;
195         }
196 
197         // Get mother volume for envelope.
198         Volume world = lcdd.pickMotherVolume(this);
199 
200         // Place the top section.
201         Position mpostop = new Position(momVol.getVolumeName() + "_top_pos", 0, my + beamgap, dface + mz);
202         // Rotation mrottop = new Rotation(momVol.getVolumeName() + "_top_rot", 0, -Math.PI/2, (3*Math.PI)/2);
203         Rotation mrottop = new Rotation(momVol.getVolumeName() + "_top_rot", 0, 0, 0);
204         define.addPosition(mpostop);
205         define.addRotation(mrottop);
206         PhysVol topSide = new PhysVol(momVol, world, mpostop, mrottop);
207         topSide.addPhysVolID("system", this.getSystemID());
208         topSide.addPhysVolID("side", 1);
209 
210         // Place the bottom section.
211         Position mposbot = new Position(momVol.getVolumeName() + "_bot_pos", 0, -my - beamgap, dface + mz);
212         Rotation mrotbot = new Rotation(momVol.getVolumeName() + "_bot_rot", 0, 0, Math.PI);
213         define.addPosition(mposbot);
214         define.addRotation(mrotbot);
215         PhysVol botSide = new PhysVol(momVol, world, mposbot, mrotbot);
216         botSide.addPhysVolID("system", this.getSystemID());
217         botSide.addPhysVolID("side", -1);
218 
219         // Make the section box invisible.
220         momVol.setVisAttributes(lcdd.getVisAttributes("InvisibleWithDaughters"));
221     }
222 
223     public boolean isCalorimeter() {
224         return true;
225     }
