11  /**
12   * Convert the HPS Test run tracker 2014 to the LCDD format.
13   */
14  public class HPSTracker2014 extends HPSTracker2014Base {
15  
16      public HPSTracker2014(Element node) throws JDOMException {
17          super(node);
18      }
19  
20      /*
21       * (non-Javadoc)
22       * @see org.lcsim.geometry.compact.converter.lcdd.HPSTracker2014Base#initializeBuilder(org.lcsim.geometry.compact.converter.lcdd.util.LCDD, org.lcsim.geometry.compact.converter.lcdd.util.SensitiveDetector)
23       */
24      protected HPSTrackerLCDDBuilder initializeBuilder(LCDD lcdd, SensitiveDetector sens) {
25          HPSTrackerLCDDBuilder b = new HPSTracker2014LCDDBuilder(_debug, node, lcdd, sens);
26          return b;
27      }
28  
29      protected int getModuleNumber(String surveyVolume) {
30          boolean isTopLayer = HPSTrackerBuilder.getHalfFromName(surveyVolume).equals("top") ? true : false;
31          int layer = HPSTrackerBuilder.getLayerFromVolumeName(surveyVolume);
32          int moduleNumber = -1;
33          if (isTopLayer) {
34              if (layer < 4) {
35                  moduleNumber = 0;
36              } else {
37                  if (HPSTrackerBuilder.isHoleFromName(surveyVolume)) {
38                      moduleNumber = 2;
39                  } else {
40                      moduleNumber = 0;
41                  }
42              }
43          } else {
44              if (layer < 4) {
45                  moduleNumber = 1;
46              } else {
47                  if (HPSTrackerBuilder.isHoleFromName(surveyVolume)) {
48                      moduleNumber = 1;
49                  } else {
50                      moduleNumber = 3;
51                  }
52              }
53          }
54  
55          if (moduleNumber < 0)
56              throw new RuntimeException("Invalid module nr found for " + surveyVolume);
57  
58          return moduleNumber;
59      }
60  
61  }
