11  public class HPSTracker2017 extends HPSTracker2014v1 {
12      public HPSTracker2017(Element node) throws JDOMException {
13          super(node);
14      }
15  
16      /*
17       * (non-Javadoc)
18       * 
19       * @see org.lcsim.geometry.compact.converter.lcdd.HPSTracker2014Base#
20       * initializeBuilder(org.lcsim.geometry.compact.converter.lcdd.util.LCDD,
21       * org.lcsim.geometry.compact.converter.lcdd.util.SensitiveDetector)
22       */
23      protected HPSTrackerLCDDBuilder initializeBuilder(LCDD lcdd,
24              SensitiveDetector sens) {
25          return new HPSTracker2017LCDDBuilder(_debug, node, lcdd, sens);
26      }
27  
28      /*
29       * Override this to handle different layer numbering.
30       * 
31       * (non-Javadoc)
32       * 
33       * @see
34       * org.lcsim.geometry.compact.converter.lcdd.HPSTracker2014#getModuleNumber
35       * (java.lang.String)
36       */
37      protected int getModuleNumber(String surveyVolume) {
38          boolean isTopLayer = HPSTrackerBuilder.getHalfFromName(surveyVolume)
39                  .equals("top") ? true : false;
40          int layer = HPSTrackerBuilder.getLayerFromVolumeName(surveyVolume);
41          int moduleNumber = -1;
42          if (isTopLayer) {
43              if (layer == 1 || layer > 4) {
44                  if (HPSTrackerBuilder.isHoleFromName(surveyVolume)) {
45                      moduleNumber = 2;
46                  } else {
47                      moduleNumber = 0;
48                  }
49              } else {
50                  moduleNumber = 0;
51              }
52          } else {
53              if (layer == 1 || layer > 4) {
54                  if (HPSTrackerBuilder.isHoleFromName(surveyVolume)) {
55                      moduleNumber = 1;
56                  } else {
57                      moduleNumber = 3;
58                  }
59              } else {
60                  moduleNumber = 1;
61              }
62          }
63  
64          if (moduleNumber < 0)
65              throw new RuntimeException("Invalid module nr found for "
66                      + surveyVolume);
67  
68          return moduleNumber;
69      }
