89      // This finds specific type. I would like to use the ID for this but can't, I think.
90      // TODO there must be a factory instance to do this
91      public SiTrackerModule getModuleDetectorElement(SiTrackerModule testElement) {
92          if (isDebug())
93              System.out.printf("%s: getModuleDetectorElement for module  %s path: \"%s\"\n", this.getClass()
94                      .getSimpleName(), testElement.getName(), testElement.getGeometry().getPathString());
95          SiTrackerModule element = null;
96          for (IDetectorElement e : moduleDetectorElements) {
97              SiTrackerModule m = (SiTrackerModule) e;
98              if (isDebug())
99                  System.out.printf("%s: compare with module  %s path: %s\"%s\" \n", this.getClass().getSimpleName(),
100                         m.getName(), m.getGeometry().getPathString());
101             if (m.getGeometry().getPathString().equals(testElement.getGeometry().getPathString())) {
102                 if (element != null)
103                     throw new RuntimeException("two DE sharing extended ID?");
104                 if (isDebug())
105                     System.out.printf("%s: found it\n", this.getClass().getSimpleName());
106                 element = m;
107             }
108         }
109         return element;
110     }

115     public IDetectorElement getSupportDetectorMother(String daughterName, String volume, int layer) {
116         IDetectorElement element = null;
117                 
118         //Loop on the layerDetectorElements
119         for (IDetectorElement e : layerDetectorElements) {
120             //System.out.printf("%s::getSupportDetecDetector: Checking element %s \n", this.getClass().getSimpleName(), e.getName());
121         
122             //Check for volume 
123             if (e.getName().contains(volume)) {
124                 if (e.getName().contains("supportRing") && (layer < 5) ) 
125                     return e;
126                 if (e.getName().contains("UChannel") && (layer >=5) )
127                     return e;
128             }
129             
130         }    
131                     
132         return null;
133     }

146             if (eId.equals(expId)) { // TODO order matters as expId is an interface without that function!?

165         IExpandedIdentifier expId = e.getExpandedIdentifier();
