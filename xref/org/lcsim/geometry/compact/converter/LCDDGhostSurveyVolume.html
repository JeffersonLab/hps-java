3   /**
4    * Interface to the LCDD converter geometry for the geometry definition. No volume is built but it can be used as reference in building the geometry.
5    */
6   public class LCDDGhostSurveyVolume extends LCDDSurveyVolume {
7   
8       /**
9        * Initialize with base and mother. This is typically for a reference geometry object that is used for referencing coordinate systems but that doesn't have a volume itself.
10       * 
11       * @param base - object used to get geometry definitions
12       * @param mother - mother LCDD object
13       */
14      public LCDDGhostSurveyVolume(SurveyVolume base, LCDDSurveyVolume mother) {
15          super(base);
16          if (isDebug())
17              System.out.printf("%s: constructing LCDD ghost object %s with mother %s\n",
18                      this.getClass().getSimpleName(), base.getName(), mother == null ? "null" : mother.getName());
19          setMother(mother);
20          mother.addDaughter(this);
21          if (isDebug())
22              System.out
23                      .printf("%s: DONE constructing LCDD object %s\n", this.getClass().getSimpleName(), base.getName());
24      }
