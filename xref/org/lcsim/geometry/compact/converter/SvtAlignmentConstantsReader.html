1   package org.lcsim.geometry.compact.converter;

3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.hps.conditions.database.DatabaseConditionsManager;
7   import org.hps.conditions.svt.SvtAlignmentConstant;
8   import org.hps.conditions.svt.SvtAlignmentConstant.SvtAlignmentConstantCollection;
9   
10  /**
11   * Reads in SVT alignment constants from the database and converts them to the {@link MilleParameter} class expected by the detector model.
12   */
13  public class SvtAlignmentConstantsReader {
14  
15      private SvtAlignmentConstantsReader() {
16      }
17  
18      /**
19       * Read SVT alignment constants from the conditions database table <i>svt_alignments</i> and create a list of <code>MilleParameter</code> objects from it.
20       * 
21       * @return the Millepede parameter list
22       */
23      static List<MilleParameter> readMilleParameters() {
24  
25          final DatabaseConditionsManager manager = DatabaseConditionsManager.getInstance();
26  
27          final List<MilleParameter> milleParameters = new ArrayList<MilleParameter>();
28  
29          final SvtAlignmentConstantCollection alignmentConstants = manager.getCachedConditions(
30                  SvtAlignmentConstantCollection.class, "svt_alignments").getCachedData();
31  
32          for (final SvtAlignmentConstant constant : alignmentConstants) {
33              final MilleParameter p = new MilleParameter(constant.getParameter(), constant.getValue(), 0.0);
34              milleParameters.add(p);
35          }
36  
37          return milleParameters;
38      }
39  }
