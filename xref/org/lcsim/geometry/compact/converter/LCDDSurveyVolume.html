47       * Initialize this object with a known volume and no mother. Typically the world volume would use this.
48       * 
49       * @param surveyVolume - core geometry definitions
50       * @param volume - given volume
51       */
52      public LCDDSurveyVolume(SurveyVolume surveyVolume, Volume volume) {
53          super(surveyVolume);
54          if (isDebug())
55              System.out.printf("%s: constructing LCDD object %s with volume name %s\n", this.getClass().getSimpleName(),
56                      surveyVolume.getName(), volume.getName());
57          setVolume(volume);
58          if (isDebug())
59              System.out.printf("%s: DONE constructing LCDD object %s\n", this.getClass().getSimpleName(),
60                      surveyVolume.getName());
61          Hep3Vector lcdd_rot_angles = TransformationUtils.getCardanAngles(surveyVolume.getCoord().v(), surveyVolume
62                  .getCoord().w(), new BasicHep3Vector(0, 1, 0), new BasicHep3Vector(0, 0, 1));
63          setPos(new Position(getName() + "_position", 0, 0, 0));
64          setRot(new Rotation(getName() + "_rotation", lcdd_rot_angles.x(), lcdd_rot_angles.y(), lcdd_rot_angles.z()));
65          if (isDebug())
66              System.out.printf("%s: DONE  %s\n", this.getClass().getSimpleName(), surveyVolume.getName());
67      }

70       * Interface to the LCDD converter geometry for the geometry definition.
71       * 
72       * @param surveyVolume - core geometry definition
73       * @param lcdd - lcdd file
74       * @param mother - reference to mother LCDD definition

76      public LCDDSurveyVolume(SurveyVolume surveyVolume, LCDD lcdd, LCDDSurveyVolume mother) {
77          super(surveyVolume);
78          if (isDebug())
79              System.out.printf("%s: constructing LCDD object %s with mother %s\n", this.getClass().getSimpleName(),
80                      surveyVolume.getName(), mother == null ? "null" : mother.getName());
81          this.lcdd = lcdd;
82          setMother(mother);
83          mother.addDaughter(this);
84          buildBox();
85          buildVolume();
86          setPositionAndRotation(surveyVolume);
87          // buildPhysVolume(mother);
88          if (isDebug())
89              System.out.printf("%s: DONE constructing LCDD object %s\n", this.getClass().getSimpleName(),
90                      surveyVolume.getName());
91      }

110     public void buildVolume() {
111         if (isDebug())
112             System.out.printf("%s: build volume for %s with material %s\n", this.getClass().getSimpleName(), getName(),
113                     getMaterial());
114         try {
115             Material mat = lcdd.getMaterial(getMaterial());

122     public void setPositionAndRotation(SurveyVolume base) {
123         if (isDebug())
124             System.out.printf("%s: set position and rotation for volume %s\n", this.getClass().getSimpleName(),
125                     getName());
126 
127         // NOTE:
128         // This sets position and reference w.r.t. mother coordinate system.
129         // If I'm not building that volume this will be wrong.
130         // TODO Similar to in the JAVA converter this should be something like the physical mother.
131 
132         if (base.getMother() == null)
133             throw new RuntimeException("trying to set coordinates w/o mother defined for " + base.getName());
134 
135         // Vector from origin to center of box locally
136         Hep3Vector box_center_base_local = base.getCenter();
137 
138         // translate to the mother coordinate system
139         LCDDSurveyVolume physMother = getPhysMother();
140         if (isDebug())
141             System.out.printf("%s: physical mother to transform to is %s; find the transform to it\n", this.getClass()
142                     .getSimpleName(), physMother.getName());
143         Transform3D trf = HPSTrackerBuilder.getTransform(base.getCoord().getTransformation(), base.getMother(),
144                 physMother.getName());
145         if (isDebug())
146             System.out.printf("%s: found transform to physical mother \n%s\n\n", this.getClass().getSimpleName(),
147                     trf.toString());
148 
149         // find the position of the center in the physical mother coord
150         Hep3Vector box_center_base = trf.transformed(box_center_base_local);
151 
152         // find the position of the center of the box in the mother coordinate system, make sure to use the physical mother coordinates
153         if (isDebug())
154             System.out.printf("%s: find center of box in physical mother coord %s \n", this.getClass().getSimpleName(),
155                     physMother.getName());
156         // hack since my getTransform function needs a mother TODO Fix this!
157         SurveyVolume gm = base;
158         if (isDebug())
159             System.out.printf("%s: look for physical mother %s starting from mother %s \n", this.getClass()
160                     .getSimpleName(), physMother.getName(), gm.getMother() != null ? gm.getMother().getName()
161                     : "-- no mother --");
162         while ((gm = gm.getMother()).getName() != physMother.getName()) {
163             if (isDebug())
164                 System.out.printf("%s: gm is %s \n", this.getClass().getSimpleName(), gm.getName());
165             // gm = gm.getMother();
166         }
167         if (isDebug())
168             System.out.printf("%s: found physical mother %s with center at %s \n", this.getClass().getSimpleName(),
169                     gm.getName(), gm.getCenter());
170 
171         Hep3Vector mother_center = gm.getCenter();
172 
173         // find the position of the center in the mother coord
174         Hep3Vector box_center = VecOp.sub(box_center_base, mother_center);
175 
176         // Find LCDD Euler rotation angles from coordinate system unit vectors
177         // Note that this has to be rotation wrt to physical mother and not just mother as normally is the case
178         if (isDebug())
179             System.out.printf(
180                     "%s: find LCDD Cardan rotation angles - need to find mother to physical mother transform \n", this
181                             .getClass().getSimpleName(), physMother.getName());
182         Hep3Vector base_u = base.getCoord().u();
183         Hep3Vector base_v = base.getCoord().v();
184         Hep3Vector base_w = base.getCoord().w();
185         if (isDebug())
186             System.out.printf("%s: unit vectors in mother coord: %s, %s, %s\n", this.getClass().getSimpleName(),
187                     base_u.toString(), base_v.toString(), base_w.toString());
188         Hep3Vector unit_u = new BasicHep3Vector(1, 0, 0);
189         Hep3Vector unit_v = new BasicHep3Vector(0, 1, 0);
190         Hep3Vector unit_w = new BasicHep3Vector(0, 0, 1);
191         if (!base.getMother().getName().equals(physMother.getName())) {
192             if (isDebug())
193                 System.out.printf("%s: Need to get unit vectors in physical mother %s coord system\n", this.getClass()
194                         .getSimpleName(), physMother.getName());
195             Transform3D trf_mother = HPSTrackerBuilder.getTransform(base.getMother().getCoord().getTransformation(),
196                     base.getMother().getMother(), physMother.getName());
197             if (isDebug())
198                 System.out.printf("%s: found transform from mother to physical mother \n%s\n", this.getClass()
199                         .getSimpleName(), trf_mother.toString());
200             // unit_u = VecOp.unit(trf_mother.rotated(unit_u));
201             // unit_v = VecOp.unit(trf_mother.rotated(unit_v));
202             // unit_w = VecOp.unit(trf_mother.rotated(unit_w));
203             base_u = VecOp.unit(trf_mother.rotated(base_u));
204             base_v = VecOp.unit(trf_mother.rotated(base_v));
205             base_w = VecOp.unit(trf_mother.rotated(base_w));
206 
207         } else {
208             if (isDebug())
209                 System.out.printf("%s: mother and physical mother is the same so unit vectors didn't change\n",
210                         getClass().getSimpleName());
211         }
212 
213         if (isDebug()) {
214             if (isDebug())
215                 System.out.printf("%s: final unit vectors to get Cardan angles from : \n%s, %s, %s -> %s, %s, %s \n",
216                         this.getClass().getSimpleName(), base_u.toString(), base_v.toString(), base_w.toString(),
217                         unit_u.toString(), unit_v.toString(), unit_w.toString());
218             // System.out.printf("%s: unit vectors u %s v %s w %s\n", this.getClass().getSimpleName(),base.getCoord().u().toString(),base.getCoord().v().toString(),base.getCoord().w().toString());
219         }
220         Hep3Vector lcdd_rot_angles = TransformationUtils
221                 .getCardanAngles(base_u, base_v, base_w, unit_u, unit_v, unit_w);
222 
223         // Create the LCDD position and rotation
224         setPos(new Position(getName() + "_position", box_center.x(), box_center.y(), box_center.z()));
225         setRot(new Rotation(getName() + "_rotation", lcdd_rot_angles.x(), lcdd_rot_angles.y(), lcdd_rot_angles.z()));
226 
227         if (isDebug()) {
228             System.out
229                     .printf("%s: SurveyVolume information for %s:\n", this.getClass().getSimpleName(), base.getName());
230             System.out.printf("%s: box_center_base_local  %s\n", this.getClass().getSimpleName(),
231                     box_center_base_local.toString());
232             System.out.printf("%s: box_center_base        %s\n", this.getClass().getSimpleName(),
233                     box_center_base.toString());
234             System.out.printf("%s: mother center          %s\n", this.getClass().getSimpleName(),
235                     mother_center.toString());
236             System.out
237                     .printf("%s: box_center             %s\n", this.getClass().getSimpleName(), box_center.toString());
238             System.out.printf("%s: pos                    %s\n", this.getClass().getSimpleName(), getPos().toString());
239             System.out.printf("%s: euler                  %s\n", this.getClass().getSimpleName(),
240                     lcdd_rot_angles.toString());
241             System.out.printf("%s: rot                    %s\n", this.getClass().getSimpleName(), getRot().toString());
242 
243             // calculate the position in tracking volume separately as a xcheck
244             Hep3Vector box_center_tracking_xcheck = HPSTrackerBuilder.transformToTracking(box_center_base_local, base);
245             System.out.printf("%s: box_center_tracking_xcheck  %s (for %s)\n", this.getClass().getSimpleName(),
246                     box_center_tracking_xcheck.toString(), base.getName());
247         }
248

251     /**
252      * Find the first non-ghost volume among parents.
253      * 
254      * @return mother object
255      */
256     public LCDDSurveyVolume getPhysMother() {
257         // if(isDebug()) System.out.printf("%s: finding physical mother to %s\n", this.getClass().getSimpleName(), getName());
258         if (mother == null)
259             throw new RuntimeException("Trying to get phys mother but there is no mother!");
260         if (mother instanceof LCDDGhostSurveyVolume) {
261             return mother.getPhysMother();
262         } else {
263             // if(isDebug()) System.out.printf("%s: found a non-ghost volume: %s\n", this.getClass().getSimpleName(), mother.getName());
264             return mother;
265         }
266     }

324     public String toString() {
325         String s = getClass().getSimpleName() + ": " + getName() + "\n";
326         if (getPos() != null && getRot() != null) {
327             double x = Double.valueOf(getPos().getAttributeValue("x"));
328             double y = Double.valueOf(getPos().getAttributeValue("y"));
329             double z = Double.valueOf(getPos().getAttributeValue("z"));
330             s += "Position: " + String.format("(%.4f %.4f %.4f)\n", x, y, z);
331             x = Double.valueOf(getRot().getAttributeValue("x"));
332             y = Double.valueOf(getRot().getAttributeValue("y"));
333             z = Double.valueOf(getRot().getAttributeValue("z"));
334             s += "Rotation: " + String.format("(%.4f %.4f %.4f)\n", x, y, z);
335         } else {
336             s += " - no position/rotation info -\n";
337         }
338         return s;
