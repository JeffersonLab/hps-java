64      /**
65       * Create the half-module.
66       * 
67       * @param side - stereo or axial
68       * @param mother to the half-module
69       */
70      protected void makeHalfModule(String side, BaseModule mother) {
71  
72          String moduleName = mother.getName();
73  
74          if (isDebug())
75              System.out.printf("%s: makeHalfModule for %s %s \n", this.getClass().getSimpleName(), moduleName, side);
76  
77          String volName = moduleName + "_halfmodule_" + side;
78  
79          // top or bottom?
80          String half = mother.getHalf();
81          boolean isTopLayer = !mother.isBottom();
82  
83          // find layer
84          int layer = mother.getLayer();
85  
86          // axial or stereo
87          boolean isAxial = isAxialFromName(volName);
88  
89          // find layer according to old definition
90          int millepedeLayer = getMillepedeLayer(volName);
91  
92          // if(isDebug()) System.out.printf("%s: half? %s layer %d oldlayer %d axial? %s\n",
93          // this.getClass().getSimpleName(), isTopLayer?"top":"bottom", layer,oldLayer,isAxial?"yes":"no");
94  
95          // find alignment correction to this volume
96          AlignmentCorrection alignmentCorrection = getHalfModuleAlignmentCorrection(isTopLayer, millepedeLayer);
97          alignmentCorrection.setNode(node);
98  
99          // find the module bundle that it will be added to
100         // TestRunModuleBundle bundle = (TestRunModuleBundle)getModuleBundle(mother);
101         // TestRunHalfModuleBundle halfModuleBundle;
102         TestRunModuleBundle bundle = (TestRunModuleBundle) getModuleBundle(mother);
103         if (bundle == null) {
104             throw new RuntimeException("Couldn't find bundle for " + volName + " from mother " + mother.getName());
105         }
106 
107         // Build the half-module bundle and add the half-module to it
108         HalfModuleBundle halfModuleBundle;
109         TestRunHalfModule halfModule;
110         if (isAxial) {
111             halfModule = createTestRunHalfModuleAxial(volName, mother, alignmentCorrection, layer, half);
112             halfModuleBundle = new TestRunHalfModuleBundle(halfModule);
113             bundle.halfModuleAxial = halfModuleBundle;
114         } else {
115             halfModule = createTestRunHalfModuleStereo(volName, mother, alignmentCorrection, layer, half);
116             halfModuleBundle = new TestRunHalfModuleBundle(halfModule);
117             bundle.halfModuleStereo = halfModuleBundle;
118         }
119 
120         // create the half module components
121 
122         makeHalfModuleComponentSensor(halfModule);
123 
124         makeHalfModuleComponentKapton(halfModule);
125 
126         makeHalfModuleComponentCF(halfModule);
127 
128         makeHalfModuleComponentHybrid(halfModule);
129 
130     }

132     void makeHalfModuleComponentHybrid(TestRunHalfModule mother) {
133 
134         if (isDebug())
135             System.out.printf("%s: makeHalfModuleComponentHybrid for %s \n", this.getClass().getSimpleName(),
136                     mother.getName());
137 
138         String volName = mother.getName() + "_hybrid";
139 
140         // Build the half-module
141 
142         // id is hard coded
143         int component_number = 3;
144 
145         Hybrid hybrid = new Hybrid(volName, mother, component_number);
146         hybrid.setMaterial("G10");
147 
148         TestRunHalfModuleBundle hm = (TestRunHalfModuleBundle) getHalfModuleBundle((BaseModule) mother.getMother(),
149                 mother.getName());
150         hm.hybrid = hybrid;
151 
152         if (isDebug())
153             System.out.printf("%s: added hybrid to half-module with name %s \n", this.getClass().getSimpleName(),
154                     hm.halfModule.getName());
155 
156     }
