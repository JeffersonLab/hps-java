3   /**
4    * Interface to the JAVA converter geometry for the geometry definition. In this case no volume is built 
5    * but can be used as reference in building the geometry.
6    */
7   public class JavaGhostSurveyVolume extends JavaSurveyVolume {
8   
9       /**
10       * Initialize with base and mother. This is typically for a reference geometry object that is used for referencing coordinate systems but that doesn't have a volume itself.
11       * 
12  
13       * @param surveyVolume - object used to get geometry definitions
14       * @param mother - mother object
15       */
16      public JavaGhostSurveyVolume(SurveyVolume surveyVolume, JavaSurveyVolume mother) {
17          super(surveyVolume);
18          if (isDebug())
19              System.out
20                      .printf("%s: constructing JAVA ghost object %s with mother %s\n", this.getClass().getSimpleName(),
21                              surveyVolume.getName(), mother == null ? "null" : mother.getName());
22          setMother(mother);
23          mother.addDaughter(this);
24          setPositionAndRotation(surveyVolume);
25          if (isDebug())
26              System.out.printf("%s: DONE constructing JAVA object %s\n", this.getClass().getSimpleName(),
27                      surveyVolume.getName());
28      }
29  
30  }
