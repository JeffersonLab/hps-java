86      private static class CrystalRange {
87          int ixmin;
88          int ixmax;
89          int iymin;
90          int iymax;
91  
92          CrystalRange(Element elem) throws Exception {
93              ixmin = ixmax = iymin = iymax = 0;
94  
95              if (elem.getAttribute("ixmin") != null) {
96                  ixmin = elem.getAttribute("ixmin").getIntValue();
97              } else {
98                  throw new RuntimeException("Missing ixmin parameter.");
99              }
100 
101             if (elem.getAttribute("ixmax") != null) {
102                 ixmax = elem.getAttribute("ixmax").getIntValue();
103             } else {
104                 throw new RuntimeException("Missing ixmax parameter.");
105             }
106 
107             if (elem.getAttribute("iymin") != null) {
108                 iymin = elem.getAttribute("iymin").getIntValue();
109             } else {
110                 throw new RuntimeException("Missing ixmax parameter.");
111             }
112 
113             if (elem.getAttribute("iymax") != null) {
114                 iymax = elem.getAttribute("iymax").getIntValue();
115             } else {
116                 throw new RuntimeException("Missing iymax parameter.");
117             }
118         }

121     private boolean isValidXY(int ix, int iy) {
122         if (!isValidX(ix))
123             return false;
124         if (!isValidY(iy))
125             return false;
126         return checkRange(ix, iy, this.removeCrystals);
127     }
128 
129     private boolean checkRange(int ix, int iy, List<CrystalRange> ranges) {
130         if (ranges.size() == 0)
131             return true;
132         for (CrystalRange range : ranges) {
133             if ((ix >= range.ixmin && ix <= range.ixmax) && ((iy >= range.iymin) && (iy <= range.iymax))) {
134                 return false;
135             }
136 
137         }
138         return true;
139     }

199     /**
200      * Get the neighbors for a given cell ID. Each crystal not on an edge has 8 neighbors. Edge crystals have fewer. 
201      * @param id The cell ID.
202      * @return A <code>Set</code> containing the cell's neighbors.
203      */
204     Set<Long> getNeighbors(Long id) {
205         // Get the IDDecoder.
206         IDDecoder dec = getIDDecoder();
207 
208         // Set the ID.
209         dec.setID(id);
210 
211         // Get ID field values.
212         int x = dec.getValue("ix");
213         int y = dec.getValue("iy");
214 
215         // Get field indices.
216         int ix = dec.getFieldIndex("ix");
217         int iy = dec.getFieldIndex("iy");
218 
219         // Get X, Y, & side neighbor data for this crystal.
220         Set<XY> neighbors = getNeighbors(x, y);
221 
222         // Get buffer with values from current ID.
223         int[] buffer = new int[dec.getFieldCount()];
224         dec.getValues(buffer);
225 
226         // Create an encoder to make neighbor IDs.
227         IDEncoder enc = new IDEncoder(dec.getIDDescription());
228 
229         // Set to hold neighbor IDs.
230         Set<Long> ids = new HashSet<Long>();
231 
232         // Loop over neighbor objects to make IDs.
233         for (XY xyside : neighbors) {
234             buffer[ix] = xyside.x;
235             buffer[iy] = xyside.y;
236             long nId = enc.setValues(buffer);
237             ids.add(nId);
238         }
239 
240         return ids;
241     }

243     Set<XY> getNeighbors(int ix, int iy) {
244         Set<Integer> xneighbors = getXNeighbors(ix);
245         Set<Integer> yneighbors = getYNeighbors(iy);
246 
247         Set<XY> neighbors = new HashSet<XY>();
248 
249         for (Integer jx : xneighbors) {
250             for (Integer jy : yneighbors) {
251                 // Filter out self.
252                 if (jx == ix && jy == iy) {
253                     continue;
254                 }
255 
256                 // Check for valid neighbor.
257                 // FIXME: Duplication of isValidX + isValidY.
258                 if (!isValidXY(jx, jy))
259                     continue;
260 
261                 neighbors.add(new XY(jx, jy));
262             }
263         }
264 
265         return neighbors;
266     }

268     Set<Integer> getXNeighbors(int ix) {
269         Set<Integer> neighbors = new HashSet<Integer>();
270 
271         // Add self.
272         neighbors.add(ix);
273 
274         // Left neighbor.
275         if (isValidX(ix - 1)) {
276             neighbors.add(ix - 1);
277         } else if (isValidX(ix - 2)) {
278             neighbors.add(ix - 2);
279         }
280 
281         // Right neighbor.
282         if (isValidX(ix + 1)) {
283             neighbors.add(ix + 1);
284         } else if (isValidX(ix + 2)) {
285             neighbors.add(ix + 2);
286         }
287 
288         return neighbors;
289     }
290 
291     Set<Integer> getYNeighbors(int iy) {
292         Set<Integer> neighbors = new HashSet<Integer>();
293 
294         // Add self.
295         neighbors.add(iy);
296 
297         // Lower neighbor.
298         if (isValidY(iy - 1)) {
299             neighbors.add(iy - 1);
300         }
301 
302         // Upper neighbor.
303         if (isValidY(iy + 1)) {
304             neighbors.add(iy + 1);
305         }
306 
307         return neighbors;
308     }

310     boolean isValidY(int iy) {
311         // Zero is not valid because ID scheme goes from 1.
312         return iy >= -ny && iy <= ny && iy != 0;
313     }
314 
315     boolean isValidX(int ix) {
316         // Even case.
317         if (!oddX) {
318             return ix >= -nx / 2 && ix <= nx / 2 && ix != 0;
319         }
320         // Odd case.
321         else {
322             return ix >= (-nx - 1) / 2 && ix <= (nx + 1) / 2;
323         }
324     }

345     private void instantiateMaps() {
346         // Setup the private instance of the map.
347         neighborMap = new NeighborMap(this.getDetectorElement().getIdentifierHelper());
348         cellIDMap = new HashMap<Long, Point>();
349 
350         IDDecoder dec = getIDDecoder();
351         IDEncoder enc = new IDEncoder(dec.getIDDescription());
352 
353         int nfields = dec.getFieldCount();
354         int[] vals = new int[nfields];
355 
356         vals[dec.getFieldIndex("system")] = getSystemID();
357 
358         int idxx = dec.getFieldIndex("ix");
359         int idxy = dec.getFieldIndex("iy");
360 
361         // Loop over y.
362         for (int iy = -ny; iy <= ny; iy++) {
363             int loopx = (int) Math.floor(nx / 2);
364             // Loop over x.
365             for (int ix = -loopx; ix <= loopx; ix++) {
366                 if (!isValidXY(ix, iy))
367                     continue;
368 
369                 vals[idxx] = ix;
370                 vals[idxy] = iy;
371 
372                 Long id = enc.setValues(vals);
373                 Set<Long> neighbors = getNeighbors(id);
374 
375                 neighborMap.put(id, neighbors);
376                 cellIDMap.put(id, new Point(ix, iy));
377             }
378         }
379     }
