40      private static Logger LOGGER = Logger.getLogger(HPSEcal3Converter.class.getPackage().getName());
41          
42      private static final double crystalToleranceX = 0.2;
43      // Tolerance factor for separating crystals to avoid overlaps.
44      private static final double crystalToleranceY = 0.35;
45  
46      // Margin for mother volume.
47      //private static final double margin = 1.1;
48  
49      // Tolerance factor for moving crystals to appropriate place in mom volume.
50      private static final double tolerance = 0.0;
51  
52      private IIdentifierDictionary dict;
53      private IIdentifierHelper helper;

55      private List<CrystalRange> ranges = new ArrayList<CrystalRange>();
56  
57      private boolean checkRange(final int ix, final int iy, final List<CrystalRange> ranges) {
58          if (ranges.size() == 0) {
59              return true;
60          }
61          for (final CrystalRange range : ranges) {
62              if (ix >= range.getXIndexMin() && ix <= range.getXIndexMax() && iy >= range.getYIndexMin() && iy <= range.getYIndexMax()) {
63                  return false;
64              }
65  
66          }
67          return true;
68      }

70      @Override
71      public void convert(final Subdetector subdet, final Detector detector) {
72          
73          LOGGER.info("converting subdetector " + subdet.getName());
74  
75          helper = subdet.getDetectorElement().getIdentifierHelper();
76          dict = helper.getIdentifierDictionary();
77  
78          // Crystal dimensions.
79          final Element dimensions = subdet.getNode().getChild("dimensions");
80  
81          double dx1, dx2, dy1, dy2, dz;
82          Element layout;
83          double beamgap = 0;
84          double beamgapTop, beamgapBottom;
85          int nx, ny;
86          double dface;
87          double tdx, tdy, tdz;
88          double bdx, bdy, bdz;
89          tdx = tdy = tdz = bdx = bdy = bdz = 0.;
90  
91          try {
92              dx1 = dimensions.getAttribute("x1").getDoubleValue();
93              dx2 = dimensions.getAttribute("x2").getDoubleValue();
94              dy1 = dimensions.getAttribute("y1").getDoubleValue();
95              dy2 = dimensions.getAttribute("y2").getDoubleValue();
96              dz = dimensions.getAttribute("z").getDoubleValue();
97  
98              // Layout parameters.
99              layout = subdet.getNode().getChild("layout");
100             if (layout.getAttribute("beamgap") != null) {
101                 beamgap = layout.getAttribute("beamgap").getDoubleValue();
102             } else {
103                 if (layout.getAttribute("beamgapTop") == null || layout.getAttribute("beamgapBottom") == null) {
104                     throw new RuntimeException(
105                             "Missing beamgap parameter in layout element, and beamgapTop or beamgapBottom was not provided.");
106                 }
107             }
108             beamgapTop = beamgap;
109             if (layout.getAttribute("beamgapTop") != null) {
110                 beamgapTop = layout.getAttribute("beamgapTop").getDoubleValue();
111             }
112             beamgapBottom = beamgap;
113             if (layout.getAttribute("beamgapBottom") != null) {
114                 beamgapBottom = layout.getAttribute("beamgapBottom").getDoubleValue();
115             }
116             nx = layout.getAttribute("nx").getIntValue();
117             ny = layout.getAttribute("ny").getIntValue();
118             dface = layout.getAttribute("dface").getDoubleValue();
119 
120             final Element topElement = layout.getChild("top");
121             final Element bottomElement = layout.getChild("bottom");
122             if (topElement != null) {
123                 if (topElement.getAttribute("dx") != null) {
124                     tdx = topElement.getAttribute("dx").getDoubleValue();
125                 }
126                 if (topElement.getAttribute("dy") != null) {
127                     tdy = topElement.getAttribute("dy").getDoubleValue();
128                 }
129                 if (topElement.getAttribute("dz") != null) {
130                     tdz = topElement.getAttribute("dz").getDoubleValue();
131                 }
132             }
133             if (bottomElement != null) {
134                 if (bottomElement.getAttribute("dx") != null) {
135                     bdx = bottomElement.getAttribute("dx").getDoubleValue();
136                 }
137                 if (bottomElement.getAttribute("dy") != null) {
138                     bdy = bottomElement.getAttribute("dy").getDoubleValue();
139                 }
140                 if (bottomElement.getAttribute("dz") != null) {
141                     bdz = bottomElement.getAttribute("dz").getDoubleValue();
142                 }
143             }
144         } catch (final DataConversionException e) {
145             throw new RuntimeException("Error converting HPSEcal3 from XML.", e);
146         }
147 
148         // Crystal material.
149         final Element mat = subdet.getNode().getChild("material");
150         final String materialName = mat.getAttributeValue("name");
151 
152         // Setup range of indices to be skipped.
153         for (final Object obj : layout.getChildren("remove")) {
154             final Element remove = (Element) obj;
155             try {
156                 ranges.add(new CrystalRange(remove));
157             } catch (final Exception x) {
158                 throw new RuntimeException(x);
159             }
160         }
161 
162         if (!ranges.isEmpty()) {
163             // FIXME: Assumes single range of removed crystals.
164             ((HPSEcalDetectorElement) subdet.getDetectorElement()).setBeamGapIndices(ranges.get(0));
165         }
166 
167         // Setup crystal logical volume.
168         final Trd crystalTrap = new Trd("crystal_trap", dx1, dx2, dy1, dy2, dz);
169         final ILogicalVolume crystalLogVol = new LogicalVolume("crystal_volume", crystalTrap, MaterialStore
170                 .getInstance().get(materialName));
171 
172         //
173         // Now we calculate parameters for crystal placement...
174         //
175 
176         // Slope of the trapezoid side in X.
177         final double sx = (dx2 - dx1) / (2 * dz);
178 
179         // Angle of the side of the trapezoid w.r.t. center line in X. Rotation
180         // about Y axis.
181         final double dthetay = atan(sx);
182 
183         // Slope of the trapezoid side in Y.
184         final double sy = (dy2 - dy1) / (2 * dz);
185 
186         // Angle of the side of the trapezoid w.r.t. center line in Y. Rotation
187         // about X axis.
188         final double dthetax = atan(sx);
189 
190         // Distance between (virtual) angular origin and center of trapezoid in
191         // X.
192         final double z0x = dx1 / sx + dz;
193 
194         // Distance between (virtual) angular origin and center of trapezoid in
195         // Y.
196         final double z0y = dy1 / sy + dz;
197 
198         // Odd or even number of crystals in X.
199         final boolean oddx = nx % 2 != 0;
200 
201         // Calculate number of X for loop.
202         if (oddx) {
203             nx -= 1;
204             nx /= 2;
205         } else {
206             nx /= 2;
207         }
208 
209         double ycorrtot = 0;
210         double zcorrtoty = 0;
211 
212         // Crystal sequence number used for unique volume names.
213         int crystaln = 1;
214 
215         // Base name for volume.
216         final String baseName = subdet.getName() + "_crystal";
217 
218         // World volume.
219         final ILogicalVolume mom = detector.getWorldVolume().getLogicalVolume();
220 
221         for (int iy = 1; iy <= ny; iy++) {
222             double zcorrtotx = 0;
223             double xcorrtot = 0;
224 
225             final int coeffy = 2 * iy - 1;
226             final double thetax = coeffy * dthetax;
227             final double zcorry = dy1 * (2 * sin(coeffy * dthetax));
228             final double ycorr = zcorry * tan((coeffy - 1) * dthetax);
229             final double ycenter = z0y * sin(coeffy * dthetax) + ycorr + ycorrtot + crystalToleranceY * iy;
230             final double thetaz = 0;
231 
232             for (int ix = 0; ix <= nx; ix++) {
233                 // Coefficient for even/odd crystal
234                 int coeffx = 2 * ix;
235                 if (!oddx) {
236                     coeffx -= 1;
237                     // For even number of crystals, the 0th is skipped.
238                     if (ix == 0) {
239                         continue;
240                     }
241                 }
242 
243                 // Set parameters for next crystal placement.
244                 final double thetay = coeffx * dthetay;
245                 final double zcorrx = dx1 * (2 * sin(coeffx * dthetay));
246                 final double xcorr = zcorrx * tan((coeffx - 1) * dthetay);
247                 final double xcenter = z0x * sin(coeffx * dthetay) + xcorr + xcorrtot + crystalToleranceX * ix;
248                 double zcenter = z0y * (cos(coeffy * dthetax) - 1) + z0x * (cos(coeffx * dthetay) - 1) + zcorrx
249                         + zcorrtotx + zcorry + zcorrtoty;
250                 zcenter += dz;
251 
252                 //
253                 // Bottom section.
254                 //
255 
256                 if (this.checkRange(ix, -iy, ranges)) {
257                     // Transform of positive bottom crystal.
258                     final ITranslation3D iposBot = new Translation3D(xcenter + bdx,
259                             -(beamgapBottom + ycenter + tolerance) + bdy, zcenter + tolerance + dface + bdz);
260                     final IRotation3D irotBot = new RotationGeant(-thetax, -thetay, thetaz);
261 
262                     // Place positive crystal.
263                     final IPhysicalVolume posCrystalPlacementBot = new PhysicalVolume(
264                             new Transform3D(iposBot, irotBot), baseName + crystaln, crystalLogVol, mom, crystaln);
265                     this.createDetectorElement(detector, subdet, posCrystalPlacementBot, ix, -iy);
266                     ++crystaln;
267                 }
268 
269                 // Reflection to negative.
270                 if (ix != 0) {
271                     if (this.checkRange(-ix, -iy, ranges)) {
272                         // Transform of negative.
273                         final ITranslation3D iposnegBot = new Translation3D(-xcenter + bdx,
274                                 -(beamgapBottom + ycenter + tolerance) + bdy, zcenter + tolerance + dface + bdz);
275                         final IRotation3D irotnegBot = new RotationGeant(-thetax, thetay, thetaz);
276 
277                         // Place negative crystal.
278                         final PhysicalVolume negCrystalPlacementBot = new PhysicalVolume(new Transform3D(iposnegBot,
279                                 irotnegBot), baseName + crystaln, crystalLogVol, detector.getWorldVolume()
280                                 .getLogicalVolume(), crystaln);
281                         this.createDetectorElement(detector, subdet, negCrystalPlacementBot, -ix, -iy);
282                         ++crystaln;
283                     }
284                 }
285 
286                 if (this.checkRange(ix, iy, ranges)) {
287                     // Transform of positive top crystal.
288                     final Translation3D iposTop = new Translation3D(xcenter + tdx, beamgapTop + ycenter + tolerance
289                             + tdy, zcenter + tolerance + dface + tdz);
290                     final IRotation3D irotTop = new RotationGeant(thetax, -thetay, thetaz);
291 
292                     // Place positive top crystal.
293                     final PhysicalVolume posCrystalPlacementTop = new PhysicalVolume(new Transform3D(iposTop, irotTop),
294                             baseName + crystaln, crystalLogVol, detector.getWorldVolume().getLogicalVolume(), crystaln);
295                     this.createDetectorElement(detector, subdet, posCrystalPlacementTop, ix, iy);
296                     ++crystaln;
297                 }
298 
299                 // Reflection to negative.
300                 if (ix != 0) {
301                     if (this.checkRange(-ix, iy, ranges)) {
302                         // Transform of negative.
303                         final ITranslation3D iposnegTop = new Translation3D(-xcenter + tdx, beamgapTop + ycenter
304                                 + tolerance + tdy, zcenter + tolerance + dface + tdz);
305                         final IRotation3D irotnegTop = new RotationGeant(thetax, thetay, thetaz);
306 
307                         // Place negative crystal.
308                         final PhysicalVolume negCrystalPlacementTop = new PhysicalVolume(new Transform3D(iposnegTop,
309                                 irotnegTop), baseName + crystaln, crystalLogVol, detector.getWorldVolume()
310                                 .getLogicalVolume(), crystaln);
311                         this.createDetectorElement(detector, subdet, negCrystalPlacementTop, -ix, iy);
312                         ++crystaln;
313                     }
314                 }
315 
316                 // Increment running X and Z totals and include tolerance to
317                 // avoid overlaps.
318                 xcorrtot += xcorr;
319                 zcorrtotx += zcorrx;
320             }
321 
322             // Increment running Y totals.
323             ycorrtot += ycorr;
324             zcorrtoty += zcorry;
325         }
326     }

328     /**
329      * Create a DetectorElement for an ECal crystal.
330      *
331      * @param detector The full detector.
332      * @param subdet The subdetector.
333      * @param crystal The crystal physical volume.
334      * @param ix The value of the ix field.
335      * @param iy The value of the iy field.
336      */
337     private final void createDetectorElement(final Detector detector, final Subdetector subdet,
338             final IPhysicalVolume crystal, final int ix, final int iy) {
339         final String path = "/" + crystal.getName();
340         final IExpandedIdentifier expId = new ExpandedIdentifier(helper.getIdentifierDictionary().getNumberOfFields());
341         expId.setValue(dict.getFieldIndex("system"), subdet.getSystemID());
342         expId.setValue(dict.getFieldIndex("ix"), ix);
343         expId.setValue(dict.getFieldIndex("iy"), iy);
344         final IIdentifier id = helper.pack(expId);
345         new EcalCrystal(subdet.getName() + "_crystal" + crystal.getCopyNumber(), subdet.getDetectorElement(), path, id);
346     }

353     @Override
354     public IDetectorElement makeSubdetectorDetectorElement(final Detector detector, final Subdetector subdetector) {
355         LOGGER.info("creating detector element for subdetector " + subdetector.getName());
356         final IDetectorElement subdetectorDE = new HPSEcalDetectorElement(subdetector.getName(),
357                 detector.getDetectorElement());
358         subdetector.setDetectorElement(subdetectorDE);
359         return subdetectorDE;
360     }
