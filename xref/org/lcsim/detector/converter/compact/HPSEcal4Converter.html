39      private static Logger LOGGER = Logger.getLogger(HPSEcal4Converter.class.getPackage().getName());
40  
41      private IIdentifierDictionary dict;
42      private IIdentifierHelper helper;
43  
44      @Override
45      public void convert(final Subdetector subdet, final Detector detector) {
46  
47          helper = subdet.getDetectorElement().getIdentifierHelper();
48          dict = helper.getIdentifierDictionary();
49          final Element dimensions = subdet.getNode().getChild("dimensions");
50          double dx1, dx2, dy1, dy2, dz;
51  
52          double trTop[] = new double[3];
53          double trBot[] = new double[3];
54          double rotTop[] = new double[3];
55          double rotBot[] = new double[3];
56          
57          try {
58              dx1 = dimensions.getAttribute("x1").getDoubleValue();
59              dx2 = dimensions.getAttribute("x2").getDoubleValue();
60              dy1 = dimensions.getAttribute("y1").getDoubleValue();
61              dy2 = dimensions.getAttribute("y2").getDoubleValue();
62              dz = dimensions.getAttribute("z").getDoubleValue();
63  
64              Element tra = subdet.getNode().getChild("translations");
65  
66              trTop[0] = tra.getAttribute("top_tr_x").getDoubleValue();
67              trTop[1] = tra.getAttribute("top_tr_y").getDoubleValue();
68              trTop[2] = tra.getAttribute("top_tr_z").getDoubleValue();
69  
70              trBot[0] = tra.getAttribute("bot_tr_x").getDoubleValue();
71              trBot[1] = tra.getAttribute("bot_tr_y").getDoubleValue();
72              trBot[2] = tra.getAttribute("bot_tr_z").getDoubleValue();
73  
74              Element rota = subdet.getNode().getChild("rotations");
75  
76              rotTop[0] = rota.getAttribute("top_rot_alpha").getDoubleValue();
77              rotTop[1] = rota.getAttribute("top_rot_beta").getDoubleValue();
78              rotTop[2] = rota.getAttribute("top_rot_gamma").getDoubleValue();
79  
80              rotBot[0] = rota.getAttribute("bot_rot_alpha").getDoubleValue();
81              rotBot[1] = rota.getAttribute("bot_rot_beta").getDoubleValue();
82              rotBot[2] = rota.getAttribute("bot_rot_gamma").getDoubleValue();
83  
84          } catch (final DataConversionException e) {
85              throw new RuntimeException("Error converting HPSEcal4 from XML.", e);
86          }
87          final Element mat = subdet.getNode().getChild("material");
88          final String materialName = mat.getAttributeValue("name");
89          final Trd crystalTrap = new Trd("crystal_trap", dx1, dx2, dy1, dy2, dz);
90          final ILogicalVolume crystalLogVol = new LogicalVolume("crystal_volume", crystalTrap, MaterialStore
91                  .getInstance().get(materialName));
92          Transformations transTop = new Transformations(trTop, rotTop);
93          Transformations transBot = new Transformations(trBot, rotBot);
94          final String baseName = subdet.getName() + "_crystal";
95          final ILogicalVolume mom = detector.getWorldVolume().getLogicalVolume();
96          
97          // Get database conditions.
98          DatabaseConditionsManager mgr = DatabaseConditionsManager.getInstance();
99          EcalCrystalPositionCollection positions = mgr.getCachedConditions(EcalCrystalPositionCollection.class,
100                 "ecal_crystal_positions").getCachedData();
101         EcalChannelCollection channels = mgr.getCachedConditions(EcalChannelCollection.class,
102                 "ecal_channels").getCachedData();
103         
104         // Loop over crystal positions.
105         for (EcalCrystalPosition position : positions) {
106 
107             // Get channel info.
108             int channelId = position.getChannelId();
109             EcalChannel channel = channels.findChannel((long) channelId);
110             int iy = channel.getY();
111             int ix = channel.getX();
112 
113             // z axis rotation parameter of the whole module
114             double zrotCry;
115             Transformations trans;
116             if (iy > 0) {
117                 zrotCry = rotTop[0];
118                 trans = transTop;
119             } else {
120                 zrotCry = rotBot[0];
121                 trans = transBot;
122             }
123             
124             Geant4Position g4pos = trans.transformToGeant4(position, channel);
125             double[] centerxyz = g4pos.getCenterArr();
126             double[] thetaxyz = g4pos.getTaitBryanAngles();
127             
128             final ITranslation3D iposBot = new Translation3D(centerxyz[0], centerxyz[1], centerxyz[2]);
129             final IRotation3D irotBot = new RotationGeant(thetaxyz[0], thetaxyz[1], thetaxyz[2] - zrotCry);
130 
131             // Place the crystal.
132             final IPhysicalVolume CrystalPlacement = new PhysicalVolume(
133                     new Transform3D(iposBot, irotBot), 
134                     baseName + channelId, 
135                     crystalLogVol, 
136                     mom, 
137                     channelId);
138             
139             // Create the DE pointing to the crystal vol.
140             this.createDetectorElement(detector, subdet, CrystalPlacement, ix, iy);
141         }       
142     }
143 
144     /**
145      * Create a DetectorElement for an ECal crystal.
146      *
147      * @param detector The full detector.
148      * @param subdet The sub detector.
149      * @param crystal The crystal physical volume.
150      * @param ix The value of the ix field.
151      * @param iy The value of the iy field.
152      */
153     private final void createDetectorElement(final Detector detector, final Subdetector subdet,
154             final IPhysicalVolume crystal, final int ix, final int iy) {
155         final String path = "/" + crystal.getName();
156         final IExpandedIdentifier expId = new ExpandedIdentifier(helper.getIdentifierDictionary().getNumberOfFields());
157         expId.setValue(dict.getFieldIndex("system"), subdet.getSystemID());
158         expId.setValue(dict.getFieldIndex("ix"), ix);
159         expId.setValue(dict.getFieldIndex("iy"), iy);
160         final IIdentifier id = helper.pack(expId);
161         new EcalCrystal(subdet.getName() + "_crystal" + crystal.getCopyNumber(), subdet.getDetectorElement(), path, id);
162     }

169     @Override

171         final IDetectorElement subdetectorDE = new HPSEcalDetectorElement(subdetector.getName(),
172                 detector.getDetectorElement());
173         subdetector.setDetectorElement(subdetectorDE);
174         return subdetectorDE;
175     }
