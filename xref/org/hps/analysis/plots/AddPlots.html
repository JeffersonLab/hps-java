18  import org.apache.commons.cli.PosixParser;
19  import org.apache.commons.cli.HelpFormatter;
20  import org.apache.commons.cli.Options;

23  /**
24   * Command line tool to add together histograms from multiple AIDA files.
25   */
26  public class AddPlots {
27      
28      private static Logger LOGGER = Logger.getLogger(AddPlots.class.getPackage().getName());
29          
30      private static final Options OPTIONS = new Options();
31      static {
32          OPTIONS.addOption("o", "output", true, "output file name");
33          OPTIONS.addOption("h", "help", false, "print help and exit");
34      }
35      
36      public static void printUsage() {
37          final HelpFormatter help = new HelpFormatter();
38          help.printHelp("AddPlots [-o outputFile] file1.aida file2.aida [...]", "", OPTIONS, "");
39          System.exit(1);
40      }
41      
42      public static void main(String[] args) throws Exception {
43          CommandLine cl = new PosixParser().parse(OPTIONS, args);        
44          if (cl.getArgList().isEmpty()) {
45              throw new RuntimeException("No input AIDA files to add.");
46          }        
47          if (cl.getArgList().size() < 2) {
48              throw new RuntimeException("Not enough AIDA input files.");
49          }
50          
51          String outputFile = "combined_plots.aida";
52          if (cl.hasOption("o")) {
53              outputFile = cl.getOptionValue("o");
54          }
55          
56          if (new File(outputFile).exists()) {
57              throw new RuntimeException("The output file already exists.");
58          }
59          
60          List<File> inputFiles = new ArrayList<File>();
61          for (String arg : cl.getArgList()) {
62              File inputFile = new File(arg);
63              if (!inputFile.exists()) {
64                  throw new RuntimeException("The input file " + inputFile.getPath() + " does not exist.");
65              }
66              inputFiles.add(inputFile);
67          }
68          
69          AIDA aida = AIDA.defaultInstance();
70          IAnalysisFactory af = aida.analysisFactory();
71          IHistogramFactory hf = aida.histogramFactory();
72          ITree tree = af.createTreeFactory().create(inputFiles.get(0).getAbsolutePath());
73                          
74          List<String> histogramNames = new ArrayList<String>();
75          
76          String[] objectTypes = tree.listObjectTypes("/", true);
77          String[] objectNames = tree.listObjectNames("/", true);
78          for (int pathIndex = 0; pathIndex < objectNames.length; pathIndex++) {
79              if (objectTypes[pathIndex].startsWith("IHistogram")) {
80                  histogramNames.add(objectNames[pathIndex]);
81              }
82              LOGGER.fine(objectNames[pathIndex] + ":" + objectTypes[pathIndex]);
83          }
84          
85          LOGGER.info("found " + histogramNames.size() + " histograms in " + inputFiles.get(0).getPath());
86  
87          for (int fileIndex = 1; fileIndex < inputFiles.size(); fileIndex++) {            
88              File file = inputFiles.get(fileIndex);
89              LOGGER.info("processing " + file.getPath());
90              ITree srcTree = af.createTreeFactory().create(file.getAbsolutePath());
91              for (String histogramName : histogramNames) {                
92                  String path = histogramName.substring(0, histogramName.lastIndexOf('/'));
93                  aida.tree().mkdirs(path);
94                  IManagedObject object = srcTree.find(histogramName);
95                  if (object != null) {
96                      IBaseHistogram src = (IBaseHistogram) srcTree.find(histogramName);
97                      IBaseHistogram target = (IBaseHistogram) tree.find(histogramName);
98                      LOGGER.fine("Adding " + histogramName + " from " + file.getPath());
99                      add(hf, histogramName, src, target);
100                 } else {
101                     LOGGER.warning("The object " + histogramName + " was not found in " + file.getPath() + ".");
102                 }
103             }
104         }
105 
106         if (inputFiles.size() == 1) {
107             aida.tree().mount("/", tree, "/");
108         }
109         
110         aida.saveAs(outputFile);
111         LOGGER.info("wrote plots to " + outputFile);
112     }
113     
114     private static void add(IHistogramFactory factory, String path, IBaseHistogram src, IBaseHistogram target) {
115         if (src instanceof IHistogram1D) {
116             factory.add(path, (IHistogram1D) src, (IHistogram1D) target);
117         } else if (src instanceof IHistogram2D) {
118             factory.add(path, (IHistogram2D) src, (IHistogram2D) target);
119         } else if (src instanceof IHistogram3D) {
120             factory.add(path, (IHistogram3D) src, (IHistogram3D) target);
121         }
122     }
123 }
