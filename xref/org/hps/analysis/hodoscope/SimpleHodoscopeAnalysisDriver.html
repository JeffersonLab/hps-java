1   package org.hps.analysis.hodoscope;

3   import java.util.List;
4   
5   import org.hps.detector.hodoscope.HodoscopeDetectorElement;
6   import org.lcsim.detector.IDetectorElement;
7   import org.lcsim.detector.IGeometryInfo;

9   import org.lcsim.detector.identifier.IIdentifierHelper;

12  import org.lcsim.event.SimTrackerHit;
13  import org.lcsim.geometry.Detector;
14  import org.lcsim.geometry.IDDecoder;

16  import org.lcsim.util.aida.AIDA;
17  
18  import hep.aida.ICloud1D;
19  import hep.aida.ICloud2D;
20  import hep.aida.IHistogram1D;
21  import hep.physics.vec.Hep3Vector;
22  
23  public class SimpleHodoscopeAnalysisDriver extends Driver {
24  
25      private AIDA aida = AIDA.defaultInstance();
26      private String hitCollName = "HodoscopeHits";
27  
28      private IHistogram1D simHitCountPlot = aida.histogram1D("Sim Hit Count", 100, 0., 100.);
29      private IHistogram1D simHitXPlot = aida.histogram1D("Sim Hit X", 200, 60., 260.);    
30      private ICloud1D simHitYPlot = aida.cloud1D("Sim Hit Y");
31      private ICloud1D simHitZPlot = aida.cloud1D("Sim Hit Z");
32      private ICloud2D simHitXYPlot = aida.cloud2D("Sim Hit X vs Y");
33      private ICloud1D simHitLenPlot = aida.cloud1D("Sim Hit Len");
34      private ICloud1D simHitLayerPlot = aida.cloud1D("Sim Hit Layer");
35      private ICloud1D simHitIdXPlot = aida.cloud1D("Sim Hit X ID");
36      private ICloud1D simHitIdYPlot = aida.cloud1D("Sim Hit Y ID");
37      private ICloud2D simHitIdXYPlot = aida.cloud2D("Sim Hit XY ID");
38      private ICloud1D simHitEnergyPlot = aida.cloud1D("Sim Hit Energy");
39      private ICloud1D simHitPathLenPlot = aida.cloud1D("Sim Hit Path Len");
40      private IHistogram1D simHitTimePlot = aida.histogram1D("Sim Hit Time", 1000, 0., 250.);
41      private ICloud2D simHitLenEnergyPlot = aida.cloud2D("Sim Hit Len vs Energy");
42      
43      private ICloud2D simHitPosPixelX = aida.cloud2D("Sim Hit Position X vs Pixel X");
44      private ICloud2D simHitPosPixelY = aida.cloud2D("Sim Hit Position Y vs Pixel Y");
45      private ICloud2D simHitPosPixelZ = aida.cloud2D("Sim Hit Position Z vs Pixel Z");
46      
47      private ICloud2D simHitPixelXY = aida.cloud2D("Sim Hit Pixel X vs Y");
48      
49      private HodoscopeDetectorElement hodoDetElem;
50      private IIdentifierHelper helper;
51      
52      protected void detectorChanged(Detector det) {
53          hodoDetElem = (HodoscopeDetectorElement) det.getSubdetector("Hodoscope").getDetectorElement();
54          helper = hodoDetElem.getIdentifierHelper();
55      }
56      
57      protected void startOfData() {
58      }    
59  
60      protected void endOfData() {
61      }
62  
63      protected void process(EventHeader event) {
64          List<SimTrackerHit> simHits = event.get(SimTrackerHit.class, hitCollName);
65          simHitCountPlot.fill(simHits.size());
66          LCMetaData meta = event.getMetaData(simHits);
67          IDDecoder dec = meta.getIDDecoder();       
68          for (SimTrackerHit simHit : simHits) {
69              double[] pos = simHit.getPosition();
70              simHitXPlot.fill(pos[0]);
71              simHitYPlot.fill(pos[1]);
72              simHitZPlot.fill(pos[2]);
73              simHitXYPlot.fill(pos[0], pos[1]);
74              Hep3Vector posVec = simHit.getPositionVec();
75              simHitLenPlot.fill(posVec.magnitude());
76              
77              dec.setID(simHit.getCellID64());
78              int layer = dec.getValue("layer");
79              int ix = dec.getValue("ix");
80              int iy = dec.getValue("iy");
81              simHitLayerPlot.fill(layer);
82              simHitIdXPlot.fill(ix);
83              simHitIdYPlot.fill(iy);
84              simHitIdXYPlot.fill(ix, iy);
85              
86              simHitEnergyPlot.fill(simHit.getdEdx());
87              
88              simHitPathLenPlot.fill(simHit.getPathLength());
89              
90              simHitTimePlot.fill(simHit.getTime());
91              
92              simHitLenEnergyPlot.fill(simHit.getPathLength(), simHit.getdEdx());
93              
94              IIdentifier hitId = simHit.getIdentifier();
95              IDetectorElement idDetElem = hodoDetElem.findDetectorElement(hitId).get(0);
96                          
97              IGeometryInfo geom = idDetElem.getGeometry();
98              simHitPosPixelX.fill(pos[0], geom.getPosition().x());
99              simHitPosPixelY.fill(pos[1], geom.getPosition().y());
100             simHitPosPixelZ.fill(pos[2], geom.getPosition().z());
101             
102             simHitPixelXY.fill(geom.getPosition().x(), geom.getPosition().y());            
103         }
104     }
105 }
