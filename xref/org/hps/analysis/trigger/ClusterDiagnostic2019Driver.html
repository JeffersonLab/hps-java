34  /**
35   * <code>ClusterDiagnostic2019Driver</code> performs diagnostic checks on run
36   * data to verify that clustering is being performed as expected and that there
37   * are no errors occurring. It takes VTP clusters created by the software
38   * simulation (simulated clusters) and compares them to VTP clusters reported by
39   * the hardware (hardware, or VTP, clusters). <br/>
40   * <br/>
41   * Simulated clusters are compared to hardware clusters to find matches.
42   * Clusters are considered to match if they share the same crystal indices and
43   * time-stamp. Clusters are then "verified" by comparing their energies and hit
44   * counts. The thresholds for verification are programmable. <br/>
45   * <br/>
46   * Statistics are produced for how well clusters matched and verified along with
47   * diagnostic plots. Additionally, two levels of verbose output may be enabled
48   * to provide detailed event readouts for the entire matching and verification
49   * process to aid in diagnosing issues. <br/>
50   * <br/>
51   * Certain DAQ settings for the run are required for the diagnostic process.
52   * These are obtained via the <code>ConfigurationManager2019</code> class
53   * automatically. Note that a driver to initialize this class must be present in
54   * the event chain in order for the diagnostics to initialize and function.
55   * 
56   * <code>ClusterDiagnostic2019Driver</code> is developed based on
57   * <code>ClusterDiagnosticDriver</code>
58   */

340     private static final Set<Point> getClusterHitIndices(Cluster cluster) {
341         // Store the possible hits in a set. The seed hit is excluded.
342         Set<Point> hitSet = new HashSet<Point>(8);
343 
344         // Iterate over the cluster hits and add any existing positions
345         // to the hit set.
346         for (CalorimeterHit hit : cluster.getCalorimeterHits()) {
347             if (hit != cluster.getCalorimeterHits().get(0))
348                 hitSet.add(new Point(hit.getIdentifierFieldValue("ix"), hit.getIdentifierFieldValue("iy")));
349         }
350 
351         // Return the set of constituent hits.
352         return hitSet;
353     }

355     private static final Set<Point> getClusterPossibleHitIndices(Cluster cluster) {
356         // Get the cluster seed position.
357         int ix = TriggerModule2019.getClusterXIndex(cluster);
358         int iy = TriggerModule2019.getClusterYIndex(cluster);
359 
360         // Store the possible hits in a set. The seed hit is excluded.
361         Set<Point> hitSet = new HashSet<Point>(8);
362 
363         // Get all eight adjacent hits.
364         xLoop: for (int xMod = -1; xMod <= 1; xMod++) {
365             // Get the modified x position.
366             int hix = ix + xMod;
367 
368             // Values where |x| > 23 do not exist.
369             if (Math.abs(hix) > 23) {
370                 continue xLoop;
371             }
372 
373             // Values of x = 0 do not exist. x = 1 and x = -1 are to
374             // be treated as adjacent.
375             if (hix == 0) {
376                 hix = ix == -1 ? 1 : -1;
377             }
378 
379             yLoop: for (int yMod = -1; yMod <= 1; yMod++) {
380                 // Get the modified y position.
381                 int hiy = iy + yMod;
382 
383                 // Values of y = 0 and |y| > 5 do not exist.
384                 if (hiy == 0 || Math.abs(hiy) > 5) {
385                     continue yLoop;
386                 }
387 
388                 // Add the potential hit position to the set.
389                 hitSet.add(new Point(hix, hiy));
390             }
391         }
392 
393         // Return the set of possible hit positions.
394         return hitSet;
395     }

397     private static final double getRatioError(double num, double sigmaNum, double den, double sigmaDen) {
398         double ratio = num / den;
399         return Math.abs(ratio) * Math.sqrt(Math.pow(sigmaNum / num, 2) + Math.pow(sigmaDen / den, 2));
400     }

1095                                     // If time of the hit (over seed threshold) is less than 32 ns,it could be
1096                                     // skipped in hardware either.
1097                                     if (hit.getTime() < 32) {
1098                                         failedPositionDeadtime++;
1099                                         continue matchedPairLoop;
1100                                     }
1101                                 }
1102                             }
1103                             // no other hit is over seed threshold, so no cluster is constructed in hardware
1104                             failedPositionDeadtime++;
1105                             continue matchedPairLoop;
1106                         }
1107                     }
1108 
1109                     // case 2: there is a hit at 0 ns, such hit is skipped in hardware, and another
1110                     // hit after the hit is recorded. The hit at 0 ns is not included by cluster
1111                     // since the time cluster is larger than "temporal window before".
1112                     if (reconstructedClusterTime < 32 + windowBefore && reconstructedClusterTime > windowBefore) {
1113                         Set<Point> possibleHits = getClusterPossibleHitIndices(reconstructedCluster);
1114                         for (CalorimeterHit hit : hits) {
1115                             if (hit.getTime() == 0) {
1116                                 Point ixy = new Point(hit.getIdentifierFieldValue("ix"),
1117                                         hit.getIdentifierFieldValue("iy"));
1118                                 if (possibleHits.contains(ixy)) {
1119                                     failedPositionDeadtime++;
1120                                     continue matchedPairLoop;
1121                                 }
1122                             }
1123                         }
1124                     }
1125                 }
1126 
1127                 // Check for more complicated fail state causes.
1128                 else {
1129                     // The "clone bug" occurs when the hardware reports
1130                     // a fake cluster at the same position as an existing
1131                     // cluster, but at the same time as a cluster which
1132                     // should exist, but is missed.
1133                     // Check if more than VTP cluster exists at the
1134                     // same position as another cluster, and one of them
1135                     // matches in time.
1136                     Map<Point, Integer> countMap = new HashMap<Point, Integer>();
1137                     Map<Point, VTPCluster> timeMap = new HashMap<Point, VTPCluster>();
1138                     for (VTPCluster hardwareCluster : hardwareClusters) {
1139                         Point ixy = new Point(TriggerModule2019.getClusterXIndex(hardwareCluster),
1140                                 TriggerModule2019.getClusterYIndex(hardwareCluster));
1141                         if (countMap.containsKey(ixy)) {
1142                             countMap.put(ixy, countMap.get(ixy) + 1);
1143                         } else {
1144                             countMap.put(ixy, new Integer(1));
1145                         }
1146                         if (TriggerModule2019.getClusterTime(hardwareCluster) == reconstructedClusterTime) {
1147                             timeMap.put(ixy, hardwareCluster);
1148                         }
1149                     }
1150 
1151                     // Check each cluster position. If there are more
1152                     // clusters than one at that position, and one of
1153                     // them has the same time as the position failure
1154                     // cluster, the clone bug could have occurred.
1155                     unmatchedLoop: for (Map.Entry<Point, Integer> entry : countMap.entrySet()) {
1156                         // Check that at least two clusters exist at
1157                         // this position and that one of them matched
1158                         // the fail state cluster in time.
1159                         if (entry.getValue() >= 2 && timeMap.containsKey(entry.getKey())) {
1160                             // Get the matching cluster.
1161                             VTPCluster cloneTestCluster = timeMap.get(entry.getKey());
1162 
1163                             // The cluster that matched the fail state
1164                             // cluster in time must be itself unmatched.
1165                             for (ClusterMatchedPair2019 match : matchedPairs) {
1166                                 // If the cluster matching the fail state
1167                                 // cluster in time is matched, this is
1168                                 // not the clone bug.
1169                                 if (match.getVTPCluster() == cloneTestCluster) {
1170                                     continue unmatchedLoop;
1171                                 }
1172                             }
1173 
1174                             // If not match was found for the cluster
1175                             // that matches the fail state cluster in
1176                             // time, this is probably the clone bug.
1177                             failedCloneBug++;
1178                             break unmatchedLoop;
1179                         }
1180                     }
1181                 }
1182             }
1183         }
1184 
1185         // Increment the statistics.
1186         this.matchedClusters += matchedClusters;
1187         localMatchedClusters += matchedClusters;
1188         this.failedMatchTime += failedMatchTime;
1189         this.failedMatchEnergy += failedMatchEnergy;
1190         this.failedMatchHitCount += failedMatchHitCount;
1191         this.failedMatchPosition += failedMatchPosition;
1192 
1193         // Output the results summary header.
1194         logger.printNewLine(2);
1195         logger.println("==== Cluster Verification Summary ====================================");
1196         logger.println("======================================================================");
1197 
1198         // Output the cluster pairs that successfully verified.
1199         logger.println();
1200         logger.println("Verified Simulated/Hardware Cluster Pairs:");
1201         if (matchedClusters != 0) {
1202             for (ClusterMatchedPair2019 pair : matchedPairs) {
1203                 if (pair.isMatch()) {
1204                     logger.printf("\t%s --> %s%n",
1205                             TriggerDiagnosticUtil.clusterToString(pair.getReconstructedCluster()),
1206                             TriggerDiagnosticUtil.clusterToString(pair.getVTPCluster()));
1207                 }
1208             }
1209         } else {
1210             logger.println("\tNone");
1211         }
1212 
1213         // Output the event statistics to the diagnostics logger.
1214         logger.println();
1215         logger.println("Event Statistics:");
1216         logger.printf("\tSafe Clusters      :: %d%n", goodSimulatedClusters.size());
1217         logger.printf("\tClusters Matched   :: %d%n", matchedClusters);
1218         logger.printf("\tFailed (Position)  :: %d%n", failedMatchPosition);
1219         logger.printf("\tFailed (Time)      :: %d%n", failedMatchTime);
1220         logger.printf("\tFailed (Energy)    :: %d%n", failedMatchEnergy);
1221         logger.printf("\tFailed (Hit Count) :: %d%n", failedMatchHitCount);
1222         if (goodSimulatedClusters.isEmpty()) {
1223             logger.printf("\tCluster Efficiency :: N/A %n", 100.0 * matchedClusters / goodSimulatedClusters.size());
1224         } else {
1225             logger.printf("\tCluster Efficiency :: %3.0f%%%n", 100.0 * matchedClusters / goodSimulatedClusters.size());
1226         }
1227 
1228         // The verification process is considered to fail when any not
1229         // pulse-clipped simulated cluster fails to verify. If it is
1230         // appropriate to the verbosity settings, print the event log.
1231         if (verbose || (printOnFail && matchedPairs.size() != matchedClusters)) {
1232             logger.printLog();
1233         }
1234 
1235         // If the local time window has elapsed, update the efficiency
1236         // plot values and reset the counters.
1237         if (localEndTime - localStartTime >= localWindowSize) {
1238             // Values are stored in an array within a list. Each array
1239             // index corresponds to specific value and each array entry
1240             // corresponds to an individual local window.
1241             double[] databank = new double[2];
1242             databank[0] = localGoodSimulatedClusterCount;
1243             databank[1] = localMatchedClusters;
1244 
1245             // The efficiency list also stores the time that corresponds
1246             // to the particular databank. This is defined as half way
1247             // between the start and end time of the current window.
1248             Long time = localEndTime - firstStartTime - ((localEndTime - localStartTime) / 2);
1249 
1250             // Store the databank in the list.
1251             efficiencyPlotEntries.add(new Pair<Long, double[]>(time, databank));
1252 
1253             // Reset all of the counters.
1254             localMatchedClusters = 0;
1255             localGoodSimulatedClusterCount = 0;
1256 
1257             // The new window start time is the current event time.
1258             localStartTime = event.getTimeStamp();
1259         }
1260     }

1437     /**
1438      * Performs cluster matching between a collection of simulated clusters and a
1439      * collection of hardware clusters using the strictly time-compliant algorithm.
1440      * Simulated clusters are matched with a hardware cluster by comparing the x-
1441      * and y-indices of the two clusters, as well as their time-stamps. If all of
1442      * these values match, the clusters are considered to be the same. The cluster
1443      * then undergoes verification by requiring that both the cluster energies and
1444      * hit counts are within a certain programmable range of one another. Matched
1445      * clusters are then stored along with a flag that indicates whether they were
1446      * properly verified or not. Simulated clusters that do not match any hardware
1447      * cluster in both position and time are treated as failing to have verified.
1448      * 
1449      * @param simulatedClusters - A collection of VTP clusters generated by the
1450      *                          software simulation.
1451      * @param hardwareClusters  - A collection of VTP clusters reported in the VTP
1452      *                          bank by the hardware.
1453      * @param energyWindow      - The window of allowed deviation between the
1454      *                          simulated cluster and hardware cluster energies.
1455      *                          Units are in GeV.
1456      * @param hitWindow         - The window of allowed deviation between the
1457      *                          simulated cluster and hardware cluster hit counts.
1458      * @param logger            - The logger object to which diagnostic text should
1459      *                          be written.
1460      * @return Returns a <code>List</code> containing all the matched
1461      *         simulated/hardware cluster pairs as well as their verification
1462      *         statuses.
1463      */
1464     private static final List<ClusterMatchedPair2019> matchSimulatedToHardware(Collection<Cluster> simulatedClusters,
1465             Collection<VTPCluster> hardwareClusters, double energyWindow, int hitWindow, LocalOutputLogger logger) {
1466         // Store the list of clusters, along with their matches (if
1467         // applicable) and their pair verification status.
1468         List<ClusterMatchedPair2019> pairList = new ArrayList<ClusterMatchedPair2019>();
1469 
1470         // Store the clusters which have been successfully paired.
1471         Set<VTPCluster> hardwareMatched = new HashSet<VTPCluster>(hardwareClusters.size());
1472 
1473         // Find simulated/hardware cluster matched pairs.
1474         softwareLoop: for (Cluster simCluster : simulatedClusters) {
1475             // Increment the plot counts for simulated clusters.
1476             AIDA.defaultInstance().histogram1D(softwareClustersTimePlot)
1477                     .fill(TriggerModule2019.getClusterTime(simCluster));
1478             AIDA.defaultInstance().histogram1D(softwareClustersEnergyPlot)
1479                     .fill(TriggerModule2019.getValueClusterTotalEnergy(simCluster));
1480 
1481             // Track whether a position-matched cluster was found.
1482             boolean matchedPosition = false;
1483 
1484             // VERBOSE :: Output the cluster being matched.
1485             logger.printf("Considering %s%n", TriggerDiagnosticUtil.clusterToString(simCluster));
1486 
1487             // Search through the hardware clusters for a match.
1488             hardwareLoop: for (VTPCluster hardwareCluster : hardwareClusters) {
1489                 // VERBOSE :: Output the hardware cluster being considered.
1490                 logger.printf("\t%s ", TriggerDiagnosticUtil.clusterToString(hardwareCluster));
1491                 
1492                 //Over Ecal readout time window
1493                 if (TriggerModule2019.getClusterTime(hardwareCluster) > windowWidth) {
1494                     logger.printf("[ %16s]%n", "", "Over window");
1495                     continue hardwareLoop;
1496                 }
1497 
1498                 // Clusters must be matched in a one-to-one relationship,
1499                 // so clusters that have already been matched should
1500                 // be skipped.
1501                 if (hardwareMatched.contains(hardwareCluster)) {
1502                     logger.printf("[ %7s; %9s ]%n", "fail", "matched");
1503                     continue hardwareLoop;
1504                 }
1505 
1506                 // If the clusters are the same, they must have the same
1507                 // x- and y-indices.
1508                 if (TriggerModule2019.getClusterXIndex(simCluster) != TriggerModule2019
1509                         .getClusterXIndex(hardwareCluster)
1510                         || TriggerModule2019.getClusterYIndex(simCluster) != TriggerModule2019
1511                                 .getClusterYIndex(hardwareCluster)) {
1512                     logger.printf("[ %7s; %9s ]%n", "fail", "position");
1513                     continue hardwareLoop;
1514                 }
1515 
1516                 // Note that the cluster matched another cluster in
1517                 // position. This is used to determine why a cluster
1518                 // failed to verify, if necessary.
1519                 matchedPosition = true;
1520 
1521                 // If the clusters are the same, they should occur at
1522                 // the same time as well.
1523                 if (TriggerModule2019.getClusterTime(simCluster) != TriggerModule2019.getClusterTime(hardwareCluster)) {
1524                     logger.printf("[ %7s; %9s ]%n", "fail", "time");
1525                     continue hardwareLoop;
1526                 }
1527 
1528                 // It is impossible for two clusters to exist at the
1529                 // same place and the same time, so clusters that pass
1530                 // both the time comparison and position comparison are
1531                 // assumed to be the same.
1532                 hardwareMatched.add(hardwareCluster);
1533 
1534                 // Plot a comparison of the two clusters' energies and
1535                 // hit counts.
1536                 AIDA.defaultInstance().histogram1D(matchedClustersHitDiffPlot)
1537                         .fill(Math.abs(TriggerModule2019.getClusterHitCount(simCluster)
1538                                 - TriggerModule2019.getClusterHitCount(hardwareCluster)));
1539                 AIDA.defaultInstance().histogram2D(matchedClusters2DHitDiffPlot).fill(
1540                         TriggerModule2019.getClusterHitCount(simCluster),
1541                         TriggerModule2019.getClusterHitCount(hardwareCluster));
1542                 AIDA.defaultInstance().histogram1D(matchedClustersEnergyDiffPlot)
1543                         .fill(Math.abs(TriggerModule2019.getValueClusterTotalEnergy(simCluster)
1544                                 - TriggerModule2019.getValueClusterTotalEnergy(hardwareCluster)));
1545                 AIDA.defaultInstance().histogram1D(matchedClustersEnergyDiffTightRangePlot)
1546                         .fill(Math.abs(TriggerModule2019.getValueClusterTotalEnergy(simCluster)
1547                                 - TriggerModule2019.getValueClusterTotalEnergy(hardwareCluster)));
1548                 AIDA.defaultInstance().histogram1D(matchedClustersEnergyDiffTighterRangePlot)
1549                         .fill(Math.abs(TriggerModule2019.getValueClusterTotalEnergy(simCluster)
1550                                 - TriggerModule2019.getValueClusterTotalEnergy(hardwareCluster)));
1551                 AIDA.defaultInstance().histogram2D(matchedClusters2DEnergyDiffPlot).fill(
1552                         TriggerModule2019.getValueClusterTotalEnergy(simCluster),
1553                         TriggerModule2019.getValueClusterTotalEnergy(hardwareCluster));
1554                 AIDA.defaultInstance().histogram2D(matchedClusters2DHitDiffEnergyDiffPlot)
1555                         .fill(Math.abs(TriggerModule2019.getClusterHitCount(simCluster)
1556                                 - TriggerModule2019.getClusterHitCount(hardwareCluster)),
1557                                 Math.abs(TriggerModule2019.getValueClusterTotalEnergy(simCluster)
1558                                         - TriggerModule2019.getValueClusterTotalEnergy(hardwareCluster)));
1559 
1560                 // While time and position matched clusters are considered
1561                 // to be the same cluster, the clusters must have similar
1562                 // energies and hit counts to be properly verified. First
1563                 // perform the hit count check. The hardware cluster must
1564                 // match the simulated cluster hit count to within a given
1565                 // bound.
1566                 if (isHitMatch(simCluster, hardwareCluster, hitWindow)) {
1567                     // Next, check that the hardware cluster matches the
1568                     // simulated cluster in energy to within a given
1569                     // bound.
1570                     if (TriggerModule2019.getValueClusterTotalEnergy(
1571                             hardwareCluster) >= TriggerModule2019.getValueClusterTotalEnergy(simCluster) - energyWindow
1572                             && TriggerModule2019.getValueClusterTotalEnergy(hardwareCluster) <= TriggerModule2019
1573                                     .getValueClusterTotalEnergy(simCluster) + energyWindow) {
1574                         // The cluster is a match.
1575                         pairList.add(new ClusterMatchedPair2019(simCluster, hardwareCluster,
1576                                 ClusterMatchedPair2019.CLUSTER_STATE_MATCHED));
1577                         logger.printf("[ %7s; %9s ]%n", "success", "matched");
1578 
1579                         // Increment the matched cluster plots.
1580                         AIDA.defaultInstance().histogram1D(matchedClustersTimePlot)
1581                                 .fill(TriggerModule2019.getClusterTime(simCluster));
1582                         AIDA.defaultInstance().histogram1D(matchedClustersEnergyPlot)
1583                                 .fill(TriggerModule2019.getValueClusterTotalEnergy(simCluster));
1584 
1585                         // Since the cluster is already matched, there
1586                         // is no need to perform further checks.
1587                         continue softwareLoop;
1588                     }
1589 
1590                     // If the energies of the two clusters are not
1591                     // sufficiently close, the clusters fail to verify.
1592                     else {
1593                         pairList.add(new ClusterMatchedPair2019(simCluster, hardwareCluster,
1594                                 ClusterMatchedPair2019.CLUSTER_STATE_FAIL_ENERGY));
1595                         logger.printf("[ %7s; %9s ]%n", "fail", "energy");
1596                         AIDA.defaultInstance().histogram1D(failedEnergyHitDiffPlot)
1597                                 .fill(Math.abs(TriggerModule2019.getClusterHitCount(simCluster)
1598                                         - TriggerModule2019.getClusterHitCount(hardwareCluster)));
1599                         AIDA.defaultInstance().histogram1D(failedEnergyEnergyDiffPlot)
1600                                 .fill(Math.abs(TriggerModule2019.getValueClusterTotalEnergy(simCluster)
1601                                         - TriggerModule2019.getValueClusterTotalEnergy(hardwareCluster)));
1602                         continue softwareLoop;
1603                     } // End energy check.
1604                 }
1605 
1606                 // If the hit counts of the two clusters are not
1607                 // sufficiently close, the clusters fail to verify.
1608                 else {
1609                     pairList.add(new ClusterMatchedPair2019(simCluster, hardwareCluster,
1610                             ClusterMatchedPair2019.CLUSTER_STATE_FAIL_HIT_COUNT));
1611                     logger.printf("[ %7s; %9s ]%n", "fail", "hit count");
1612                     AIDA.defaultInstance().histogram1D(failedHitCountHitDiffPlot)
1613                             .fill(Math.abs(TriggerModule2019.getClusterHitCount(simCluster)
1614                                     - TriggerModule2019.getClusterHitCount(hardwareCluster)));
1615                     AIDA.defaultInstance().histogram1D(failedHitCountEnergyDiffPlot)
1616                             .fill(Math.abs(TriggerModule2019.getValueClusterTotalEnergy(simCluster)
1617                                     - TriggerModule2019.getValueClusterTotalEnergy(hardwareCluster)));
1618                     AIDA.defaultInstance().histogram1D(failedHitCountSoftwareClustersTimePlot)
1619                             .fill(TriggerModule2019.getClusterTime(simCluster));
1620                     AIDA.defaultInstance().histogram2D(failedHitCount2DSoftwareTimeEnergyDiffPlot).fill(
1621                             TriggerModule2019.getClusterTime(simCluster),
1622                             Math.abs(TriggerModule2019.getValueClusterTotalEnergy(simCluster)
1623                                     - TriggerModule2019.getValueClusterTotalEnergy(hardwareCluster)));
1624 
1625                     continue softwareLoop;
1626                 } // End hit count check.
1627             } // End hardware loop.
1628 
1629             // This point may only be reached if a cluster failed to
1630             // match with another cluster in either time or position.
1631             // Check whether a cluster at the same position was found.
1632             // If it was not, then the cluster fails to verify by reason
1633             // of position.
1634             if (!matchedPosition) {
1635                 pairList.add(new ClusterMatchedPair2019(simCluster, null,
1636                         ClusterMatchedPair2019.CLUSTER_STATE_FAIL_POSITION));
1637             }
1638 
1639             // Otherwise, the cluster had a potential match found at
1640             // the same position, but the time-stamps did not align.
1641             else {
1642                 // Probably, a software cluster matches a hardware cluster in position, and
1643                 // matches with the other hardware in time.
1644                 // Such pair should be tagged as fail position
1645                 for (VTPCluster hardwareCluster : hardwareClusters) {
1646                     if (hardwareMatched.contains(hardwareCluster))
1647                         continue;
1648                     else {
1649                         if (TriggerModule2019.getClusterTime(simCluster) == TriggerModule2019
1650                                 .getClusterTime(hardwareCluster)) {
1651                             pairList.add(new ClusterMatchedPair2019(simCluster, null,
1652                                     ClusterMatchedPair2019.CLUSTER_STATE_FAIL_POSITION));
1653                             continue softwareLoop;
1654                         }
1655                     }
1656 
1657                 }
1658                 pairList.add(
1659                         new ClusterMatchedPair2019(simCluster, null, ClusterMatchedPair2019.CLUSTER_STATE_FAIL_TIME));
1660             }
1661         } // End sim loop.
1662 
1663         // Return the list of clusters, their matches, and their
1664         // verification states.
1665         return pairList;
1666     }

1669      * Converts a <code>CalorimeterHit</code> to a basic descriptive string.
1670      *

1679                     hit.getIdentifierFieldValue("iy"), hit.getCorrectedEnergy(), hit.getTime());
1680         }
1681     }
