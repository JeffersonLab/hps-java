34  /**
35   * Driver <code>TriggeDiagnosticDriver</code> performs a comparison
36   * between the trigger results reported by the hardware and those
37   * simulated by the LCSim software for a single trigger. <br/>
38   * <br/>
39   * The driver requires simulated trigger objects, which are produced
40   * separately by the driver <code>SimTriggerDriver</code>, in order
41   * to function. It also requires the presence of the runtime settings
42   * management driver, <code>DAQConfigurationDriver</code>. <br/>
43   * <br/>
44   * The driver works by taking trigger objects simulated by LCSim, using
45   * both hardware reported (SSP) clusters and software clusters, and
46   * comparing these to the hardware's reported triggers. Reported triggers
47   * include a trigger time and which cuts passed. The driver requires
48   * that there exists a hardware trigger that matches in each of these
49   * fields for every simulated trigger. Note that the reverse may not
50   * be true; pulse-clipping can occur for simulated clusters, resulting
51   * in a reduced cluster energy, that can affect whether a cluster or
52   * pair passes the trigger. This does not occur at the hardware level,
53   * though. As such, triggers within the "pulse-clipping" region are not
54   * necessarily comparable. The driver ignores these. <br/>
55   * <br/>
56   * Output consists primarily of text printed at the end of the run. The
57   * driver outputs the overall efficiency (defined as the number of
58   * simulated triggers matched versus the total number) for both cluster
59   * types. It also provides a more detailed table that further separates
60   * this into efficiency by cluster type and active TI bit. The other
61   * output is an efficiency over time plot that shows the efficiency for
62   * a programmable time frame throughout the run.
63   */
64  public class TriggerDiagnosticDriver extends Driver {

463     /**
464      * Outputs the global efficiency data.
465      */
466     @Override
467     public void endOfData() {
468         // Get the number of digits in the largest value that is to be
469         // displayed.
470         int largestValue = max(hardwareTriggerCount[ALL_TRIGGERS], simTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS],
471                 simTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS],
472                 matchedTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS],
473                 matchedTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS]);
474         int maxChars = TriggerDiagnosticUtil.getDigits(largestValue);
475         String charDisplay = "%" + maxChars + "d";
476 
477         // Calculate the efficiencies and determine the display value.
478         double[] efficiency = new double[2];
479         String[] efficiencyDisp = new String[2];
480         for (int i = SOURCE_SIM_CLUSTER; i <= SOURCE_SSP_CLUSTER; i++) {
481             // Calculate the efficiency. This is defined as the number
482             // triggers which were matched, versus the number that are
483             // expected to have matched.
484             efficiency[i] = 100.0 * matchedTriggerCount[i][ALL_TRIGGERS] / simTriggerCount[i][ALL_TRIGGERS];
485 
486             // If there were no triggers, then the efficiency is not
487             // defined. Display "N/A."
488             if (Double.isNaN(efficiency[i])) {
489                 efficiencyDisp[i] = "   N/A   ";
490             }
491 
492             // Otherwise, display the value as a percentage.
493             else {
494                 efficiencyDisp[i] = String.format("%7.3f%%", efficiency[i]);
495             }
496         }
497 
498         // Output the trigger efficiency statistics header.
499         System.out.println();
500         System.out.println();
501         System.out.println("======================================================================");
502         System.out.println("=== Trigger Efficiency - " + triggerType.toString() + " "
503                 + generateLine(44 - triggerType.toString().length()));
504         System.out.println("======================================================================");
505 
506         // Output the global trigger statistics.
507         System.out.println("Global Efficiency:");
508         System.out.printf("Total Hardware Triggers       :: " + charDisplay + "%n", hardwareTriggerCount[ALL_TRIGGERS]);
509         System.out.printf("Total Software Sim Triggers   :: " + charDisplay + "%n",
510                 simTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS]);
511         System.out.printf("Total Hardware Sim Triggers   :: " + charDisplay + "%n",
512                 simTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS]);
513         System.out.printf("Matched Software Sim Triggers :: " + charDisplay + "%n",
514                 matchedTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS]);
515         System.out.printf("Matched Hardware Sim Triggers :: " + charDisplay + "%n",
516                 matchedTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS]);
517 
518         System.out.printf("Software Sim Efficiency       :: " + charDisplay + " / " + charDisplay,
519                 simTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS],
520                 matchedTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS]);
521         if (simTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS] == 0
522                 || matchedTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS] == 0) {
523             System.out.printf(" (%7.3f%% ± %7.3f%%)%n", 0.0, 0.0);
524         } else {
525             System.out.printf(
526                     " (%7.3f%% ± %7.3f%%)%n",
527                     100.0 * matchedTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS]
528                             / simTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS],
529                     getRatioError(matchedTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS],
530                             Math.sqrt(matchedTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS]),
531                             simTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS],
532                             Math.sqrt(simTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS])));
533         }
534 
535         System.out.printf("Hardware Sim Efficiency       :: " + charDisplay + " / " + charDisplay,
536                 simTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS],
537                 matchedTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS]);
538         if (simTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS] == 0
539                 || matchedTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS] == 0) {
540             System.out.printf(" (%7.3f%% ± %7.3f%%)%n", 0.0, 0.0);
541         } else {
542             System.out.printf(
543                     " (%7.3f%% ± %7.3f%%)%n",
544                     100.0 * matchedTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS]
545                             / simTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS],
546                     getRatioError(matchedTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS],
547                             Math.sqrt(matchedTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS]),
548                             simTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS],
549                             Math.sqrt(simTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS])));
550         }
551 
552         // Get the largest number of spaces needed to display the TI
553         // bit specific values.
554         int tiMaxValue = Integer.MIN_VALUE;
555         for (TriggerType trigger : TriggerType.values()) {
556             tiMaxValue = max(tiMaxValue, simTriggerCount[SOURCE_SIM_CLUSTER][trigger.ordinal()],
557                     matchedTriggerCount[SOURCE_SIM_CLUSTER][trigger.ordinal()],
558                     simTriggerCount[SOURCE_SSP_CLUSTER][trigger.ordinal()],
559                     matchedTriggerCount[SOURCE_SSP_CLUSTER][trigger.ordinal()]);
560         }
561         int tiMaxChars = TriggerDiagnosticUtil.getDigits(tiMaxValue);
562 
563         // Define the column width and column headers for the TI-bit
564         // specific efficiencies.
565         int[] columnWidth = new int[3];
566         String[] header = {"TI Bit", "Software Sim Efficiency", "Hardware Sim Efficiency"};
567 
568         // Determine the width of the first column. This displays the
569         // name of the TI bit with respect to which the efficiency is
570         // displayed. It should be equal to either the longest trigger
571         // name or the header width, whichever is longer.
572         columnWidth[0] = header[0].length();
573         for (TriggerType trigger : TriggerType.values()) {
574             columnWidth[0] = max(columnWidth[0], trigger.toString().length());
575         }
576 
577         // The second and third columns display the total number of
578         // matched triggers versus the total number of expected triggers
579         // as well as the percentage of two. This takes the form of the
580         // string [MATCHED] / [EXPECTED] (NNN.NNN% + NNN.NNN%). The width needed
581         // to display this will vary based on the numbers of triggers.
582         // The width should be set to wither the largest size of the
583         // largest of these values, or to the width of the header text.
584         // whichever is larger.
585         columnWidth[1] = max(header[1].length(), tiMaxChars + tiMaxChars + 22);
586         columnWidth[2] = max(header[2].length(), tiMaxChars + tiMaxChars + 22);
587 
588         // Finally, define the column size strings and the individual
589         // value size strings.
590         String valueString = "%" + tiMaxChars + "d";
591         String efficiencyString = valueString + " / " + valueString + "  (%19s)";
592         String[] columnString = {"%-" + columnWidth[0] + "s", "%-" + columnWidth[1] + "s", "%-" + columnWidth[2] + "s"};
593 
594         // Output the efficiency as a function of active TI bit.
595         System.out.println("\n\nTI-Bit Efficiency:");
596         System.out.printf("\t" + columnString[0] + "   " + columnString[1] + "   " + columnString[2] + "%n", header[0],
597                 header[1], header[2]);
598         for (TriggerType trigger : TriggerType.values()) {
599             // Calculate the efficiencies for the second and third columns.
600             double softwareSimEfficiency = 100.0 * matchedTriggerCount[SOURCE_SIM_CLUSTER][trigger.ordinal()]
601                     / simTriggerCount[SOURCE_SIM_CLUSTER][trigger.ordinal()];
602             double hardwareSimEfficiency = 100.0 * matchedTriggerCount[SOURCE_SSP_CLUSTER][trigger.ordinal()]
603                     / simTriggerCount[SOURCE_SSP_CLUSTER][trigger.ordinal()];
604 
605             // The efficiency value strings are either the efficiency
606             // rounded to two decimal points, or "N/A" if there were
607             // no triggers found.
608             String softwareSimString = Double.isNaN(softwareSimEfficiency) ? String.format("%7.3f%% ± %7.3f%%", 0.0,
609                     0.0) : String.format(
610                     "%7.3f%% ± %7.3f%%",
611                     softwareSimEfficiency,
612                     getRatioError(matchedTriggerCount[SOURCE_SIM_CLUSTER][trigger.ordinal()],
613                             Math.sqrt(matchedTriggerCount[SOURCE_SIM_CLUSTER][trigger.ordinal()]),
614                             simTriggerCount[SOURCE_SIM_CLUSTER][trigger.ordinal()],
615                             Math.sqrt(simTriggerCount[SOURCE_SIM_CLUSTER][trigger.ordinal()])));
616             String hardwareSimString = Double.isNaN(hardwareSimEfficiency) ? String.format("%7.3f%% ± %7.3f%%", 0.0,
617                     0.0) : String.format(
618                     "%7.3f%% ± %7.3f%%",
619                     hardwareSimEfficiency,
620                     getRatioError(matchedTriggerCount[SOURCE_SSP_CLUSTER][trigger.ordinal()],
621                             Math.sqrt(matchedTriggerCount[SOURCE_SSP_CLUSTER][trigger.ordinal()]),
622                             simTriggerCount[SOURCE_SSP_CLUSTER][trigger.ordinal()],
623                             Math.sqrt(simTriggerCount[SOURCE_SSP_CLUSTER][trigger.ordinal()])));
624 
625             // The efficiency column strings take the form "[MATCHED] /
626             // [EXPECTED] (NNN.NN%)".
627             String softwareSimColumn = String.format(efficiencyString,
628                     matchedTriggerCount[SOURCE_SIM_CLUSTER][trigger.ordinal()],
629                     simTriggerCount[SOURCE_SIM_CLUSTER][trigger.ordinal()], softwareSimString);
630             String hardwareSimColumn = String.format(efficiencyString,
631                     matchedTriggerCount[SOURCE_SSP_CLUSTER][trigger.ordinal()],
632                     simTriggerCount[SOURCE_SSP_CLUSTER][trigger.ordinal()], hardwareSimString);
633 
634             // Create the efficiency string
635             System.out.printf("\t" + columnString[0] + "   " + columnString[1] + "   " + columnString[2] + "%n",
636                     trigger.toString(), softwareSimColumn, hardwareSimColumn);
637         }

640         AIDA.defaultInstance().cloud2D(moduleHeader + "Software Sim Trigger Efficiency", efficiencyPlotEntries.size());
641         AIDA.defaultInstance().cloud2D(moduleHeader + "Hardware Sim Trigger Efficiency", efficiencyPlotEntries.size());
642         for (Pair<Long, int[][]> entry : efficiencyPlotEntries) {
643             // Calculate the value for each type of efficiency.
644             double softwareEfficiency = 100.0 * entry.getSecondElement()[SOURCE_SIM_CLUSTER][LOCAL_MATCHED_TRIGGERS]
645                     / entry.getSecondElement()[SOURCE_SIM_CLUSTER][LOCAL_ALL_TRIGGERS];

647                     / entry.getSecondElement()[SOURCE_SSP_CLUSTER][LOCAL_ALL_TRIGGERS];

660             }

664         // trigger plots, as appropriate.
665         for (TriggerType trigger : triggerTypes) {
666             for (CutType cut : CutType.values()) {

673                 // Define the plot for the current TI-bit and cut.
674                 for (int type = SOURCE_SIM_CLUSTER; type <= SOURCE_SSP_CLUSTER; type++) {

676                             .histogramFactory()
677                             .divide(getPlotNameEfficiency(cut, trigger, type),
678                                     AIDA.defaultInstance().histogram1D(getPlotNameVerified(cut, trigger, type)),
679                                     AIDA.defaultInstance().histogram1D(getPlotNameTotal(cut, trigger, type)));
680                 }
681             }

685     /**
686      * Processes an event and performs trigger verification. Method
687      * will only run if the <code>ConfigurationManager</code> has been
688      * initialized. Handles testing for noise events and extracting
689      * basic event data for analysis, and also ensuring that everything
690      * necessary is present. Event trigger data is actually processed
691      * separately by the method <code>triggerVerification</code>. Method
692      * also handles logger output.
693      * 
694      * @param event - The data object containing all event data.
695      */
696     @Override
697     public void process(EventHeader event) {
698         // If the DAQ configuration manager is not yet initialized,
699         // it is not possible to analyze the event.
700         if (!ConfigurationManager.isInitialized()) {
701             return;
702         }
703 
704         // Clear the logger at the start of each event.
705         logger.clearLog();
706 
707         // If this is the first event, set the starting time stamp.
708         // The end time is always the time of the current event.
709         localEndTime = event.getTimeStamp();
710         if (localStartTime == Long.MIN_VALUE) {
711             localStartTime = event.getTimeStamp();
712             firstStartTime = event.getTimeStamp();
713         }
714 
715         // Output the trigger diagnostic header.
716         logger.printNewLine(2);
717         logger.println("======================================================================");
718         logger.println("=== Trigger Diagnostics - " + triggerType.toString() + " "
719                 + generateLine(43 - triggerType.toString().length()));
720         logger.println("======================================================================");
721 
722         // Output basic event information.
723         logger.printf("Event Number :: %d%n", event.getEventNumber());
724         logger.printf("Event Time   :: %d%n", event.getTimeStamp());
725 
726         // ==========================================================
727         // ==== Obtain calorimeter hits =============================
728         // ==========================================================
729 
730         // Calorimeter hits are only relevant if noise events are to
731         // be skipped. Otherwise, the calorimeter hits needn't be defined.
732         if (skipNoiseEvents) {
733             // Get the calorimeter hits.
734             List<CalorimeterHit> hits = getCalorimeterHits(event, hitCollectionName);
735 
736             // Noise culling can not be performed if there are no hits
737             // present in the event.
738             if (hits == null) {
739                 System.err.println("TriggerDiagnostics :: Skipping event; no hit data found.");
740                 if (verbose) {
741                     logger.printLog();
742                 }
743                 return;
744             }
745 
746             // If noise event culling is enabled, check if this is a noise
747             // event.
748             if (skipNoiseEvents) {
749                 if (hits.size() >= noiseEventThreshold) {
750                     logger.println("TriggerDiagnostics: Event exceeds the noise threshold for total number of hits and will be skipped.");
751                     if (verbose) {
752                         logger.printLog();
753                     }
754                     return;
755                 }
756             }
757         }
758 
759         // ==========================================================
760         // ==== Obtain SSP and TI Banks =============================
761         // ==========================================================
762 
763         // If there is no bank data, this event can not be analyzed.
764         if (!event.hasCollection(GenericObject.class, bankCollectionName)) {
765             System.err.println("TriggerDiagnostics :: Skipping event; no bank data found.");
766             if (verbose) {
767                 logger.printLog();
768             }
769             return;
770         }
771 
772         // Get the SSP and TI banks. The TI bank stores TI bits, which
773         // are used for trigger-type efficiency values. The SSP bank
774         // contains hardware triggers and hardware clusters.
775         Pair<TIData, SSPData> banks = getBanks(event, bankCollectionName);
776 
777         // Check that all of the required objects are present.
778         if (banks.getSecondElement() == null) {
779             System.err.println("TriggerDiagnostics: Skipping event; no SSP bank found for this event.");
780             if (verbose) {
781                 logger.printLog();
782             }
783             return;
784         }
785         if (banks.getFirstElement() == null) {
786             System.err.println("TriggerDiagnostics: Skipping event; no TI bank found for this event.");
787             if (verbose) {
788                 logger.printLog();
789             }
790             return;
791         }
792 
793         // Extract the active TI bits and make sure that at least one
794         // is set.
795         boolean activeBitRead = setTIFlags(banks.getFirstElement());
796         if (!activeBitRead) {
797             System.err.println("TriggerDiagnostics: Skipping event; no TI trigger bits are active.");
798             if (verbose) {
799                 logger.printLog();
800             }
801             return;
802         }
803 
804         // ==========================================================
805         // ==== Obtain Simulated Triggers ===========================
806         // ==========================================================
807 
808         // If the simulated trigger collection does not exist, analysis
809         // can not be performed.
810         if (!event.hasCollection(SimTriggerData.class, simTriggerCollectionName)) {
811             System.err.println("TriggerDiagnostics: Skipping event; no simulated triggers found.");
812             if (verbose) {
813                 logger.printLog();
814             }
815             return;
816         }
817 
818         // Get the simulated trigger module.
819         List<SimTriggerData> stdList = event.get(SimTriggerData.class, simTriggerCollectionName);
820         SimTriggerData triggerData = stdList.get(0);
821 
822         // Engage the trigger verification.
823         triggerVerification(triggerData, banks.getSecondElement(), triggerType);
824 
825         // If the local time window has elapsed, update the efficiency
826         // plot values and reset the counters.
827         if (localEndTime - localStartTime >= localWindowSize) {
828             // Values are stored in an array within a list. Each array
829             // index corresponds to specific value and each array entry
830             // corresponds to an individual local window.
831             int[][] databank = new int[2][2];
832             databank[SOURCE_SIM_CLUSTER][LOCAL_ALL_TRIGGERS] = simTriggerCount[SOURCE_SIM_CLUSTER][LOCAL_WINDOW_TRIGGERS];
833             databank[SOURCE_SSP_CLUSTER][LOCAL_ALL_TRIGGERS] = simTriggerCount[SOURCE_SSP_CLUSTER][LOCAL_WINDOW_TRIGGERS];
834             databank[SOURCE_SIM_CLUSTER][LOCAL_MATCHED_TRIGGERS] = matchedTriggerCount[SOURCE_SIM_CLUSTER][LOCAL_WINDOW_TRIGGERS];
835             databank[SOURCE_SSP_CLUSTER][LOCAL_MATCHED_TRIGGERS] = matchedTriggerCount[SOURCE_SSP_CLUSTER][LOCAL_WINDOW_TRIGGERS];
836 
837             // The efficiency list also stores the time that corresponds
838             // to the particular databank. This is defined as half way
839             // between the start and end time of the current window.
840             Long time = localEndTime - firstStartTime - ((localEndTime - localStartTime) / 2);
841 
842             // Store the databank in the list.
843             efficiencyPlotEntries.add(new Pair<Long, int[][]>(time, databank));
844 
845             // Reset all of the counters.
846             hardwareTriggerCount[LOCAL_WINDOW_TRIGGERS] = 0;
847             simTriggerCount[SOURCE_SIM_CLUSTER][LOCAL_WINDOW_TRIGGERS] = 0;
848             simTriggerCount[SOURCE_SSP_CLUSTER][LOCAL_WINDOW_TRIGGERS] = 0;
849             matchedTriggerCount[SOURCE_SIM_CLUSTER][LOCAL_WINDOW_TRIGGERS] = 0;
850             matchedTriggerCount[SOURCE_SSP_CLUSTER][LOCAL_WINDOW_TRIGGERS] = 0;
851 
852             // The new window start time is the current event time.
853             localStartTime = event.getTimeStamp();
854         }
855 
856         // If appropriate, output the logger text.
857         if (verbose || (printOnSoftwareSimFailure && softwareSimFailure)
858                 || (printOnHardwareSimFailure && hardwareSimFailure)) {
859             logger.printLog();
860         }
861 
862         // Reset the event failure flags.
863         softwareSimFailure = false;
864         hardwareSimFailure = false;
865     }

1059      */
1060     private int compareSimulatedToHardware(Collection<Trigger<?>> simTriggers,
1061             Collection<SSPNumberedTrigger> hardwareTriggers, Class<?> clusterType) {
1062         // Print out the appropriate sub-header.
1063         logger.printNewLine(2);
1064         if (clusterType == Cluster.class || clusterType == Cluster[].class) {
1065             logger.println("==== Simulated Sim Trigger to Hardware Trigger Verification ==========");
1066         } else {
1067             logger.println("==== Hardware Sim Trigger to Hardware Trigger Verification ===========");
1068         }
1069 
1070         // Trigger matches must be one-to-one. Track which hardware
1071         // triggers are already matched do that they are not matched
1072         // twice.
1073         Set<SSPNumberedTrigger> matchedTriggers = new HashSet<SSPNumberedTrigger>();
1074 
1075         // Iterate over each trigger simulated from hardware clusters.
1076         // It is expected that each of these triggers will correspond
1077         // to an existing hardware trigger, since, purportedly, these
1078         // are the same clusters from which the hardware triggers are
1079         // generated.
1080         simLoop: for (Trigger<?> simTrigger : simTriggers) {
1081             // Since hardware triggers do not retain the triggering
1082             // cluster, the only way to compare triggers is by the
1083             // time stamp and the cut results. In order to be verified,
1084             // a trigger must have all of these values match. The time
1085             // of a singles trigger is the time of the cluster which
1086             // created it.
1087             double simTime = getTriggerTime(simTrigger);
1088 
1089             // Output the current trigger that is being matched.
1090             logger.printf("Matching Trigger %s%n", getTriggerText(simTrigger, true));
1091 
1092             // Iterate over the hardware triggers and look for one that
1093             // matches all of the simulated trigger's values.
1094             hardwareLoop: for (SSPNumberedTrigger hardwareTrigger : hardwareTriggers) {
1095                 // Output the comparison hardware trigger.
1096                 logger.printf("\t%s", getTriggerText(hardwareTrigger));
1097 
1098                 // If the current trigger has already been matched,
1099                 // then skip over it.
1100                 if (matchedTriggers.contains(hardwareTrigger)) {
1101                     logger.printf(" [ fail; matched      ]%n");
1102                     continue hardwareLoop;
1103                 }
1104 
1105                 // The triggers must occur at the same time to classify
1106                 // as a match.
1107                 if (hardwareTrigger.getTime() != simTime) {
1108                     logger.printf(" [ fail; time         ]%n");
1109                     continue hardwareLoop;
1110                 }
1111 
1112                 // Cut comparisons are fundamentally different between
1113                 // singles and pair triggers, so these must be handled
1114                 // separately.
1115                 if (isSinglesTrigger(simTrigger) && hardwareTrigger instanceof SSPSinglesTrigger) {
1116                     // Cast both the triggers to pair trigger objects.
1117                     SinglesTrigger<?> simSinglesTrigger = (SinglesTrigger<?>) simTrigger;
1118                     SSPSinglesTrigger hardwareSinglesTrigger = (SSPSinglesTrigger) hardwareTrigger;
1119 
1120                     // Since there is no seed hit information for the
1121                     // hardware cluster, we just assume that this cut
1122                     // matches. Instead, move to the cluster energy
1123                     // lower bound cut.
1124                     if (hardwareSinglesTrigger.passCutEnergyMin() != simSinglesTrigger.getStateClusterEnergyLow()) {
1125                         logger.printf(" [ fail; energy low  ]%n");
1126                         continue hardwareLoop;
1127                     }
1128 
1129                     // Next, check the cluster energy upper bound cut.
1130                     if (hardwareSinglesTrigger.passCutEnergyMax() != simSinglesTrigger.getStateClusterEnergyHigh()) {
1131                         logger.printf(" [ fail; energy high ]%n");
1132                         continue hardwareLoop;
1133                     }
1134 
1135                     // Lastly, check if the cluster hit count cut matches.
1136                     if (hardwareSinglesTrigger.passCutHitCount() != simSinglesTrigger.getStateHitCount()) {
1137                         logger.printf(" [ fail; hit count   ]%n");
1138                         continue hardwareLoop;
1139                     }
1140                 } else if (isPairTrigger(simTrigger) && hardwareTrigger instanceof SSPPairTrigger) {
1141                     // Cast both the triggers to pair trigger objects.
1142                     PairTrigger<?> simPairTrigger = (PairTrigger<?>) simTrigger;
1143                     SSPPairTrigger hardwarePairTrigger = (SSPPairTrigger) hardwareTrigger;
1144 
1145                     // Since there is no singles cuts data for the
1146                     // hardware trigger, we just assume that these cuts
1147                     // match. Move to the pair energy sum cut.
1148                     if (hardwarePairTrigger.passCutEnergySum() != simPairTrigger.getStateEnergySum()) {
1149                         logger.printf(" [ fail; sum          ]%n");
1150                         continue hardwareLoop;
1151                     }
1152 
1153                     // Next, check the energy difference cut.
1154                     if (hardwarePairTrigger.passCutEnergyDifference() != simPairTrigger.getStateEnergyDifference()) {
1155                         logger.printf(" [ fail; difference   ]%n");
1156                         continue hardwareLoop;
1157                     }
1158 
1159                     // Next, check the energy slope cut.
1160                     if (hardwarePairTrigger.passCutEnergySlope() != simPairTrigger.getStateEnergySlope()) {
1161                         logger.printf(" [ fail; slope        ]%n");
1162                         continue hardwareLoop;
1163                     }
1164 
1165                     // Lastly, check the coplanarity cut.
1166                     if (hardwarePairTrigger.passCutCoplanarity() != simPairTrigger.getStateCoplanarity()) {
1167                         logger.printf(" [ fail; coplanarity  ]%n");
1168                         continue hardwareLoop;
1169                     }
1170                 } else {
1171                     throw new IllegalArgumentException("Trigger type is unrecongnized or simulated and "
1172                             + "hardware triggers are of different types.");
1173                 }
1174 
1175                 // If all three values match, then these triggers are
1176                 // considered a match and verified.
1177                 logger.printf(" [ trigger verified   ]%n");
1178                 matchedTriggers.add(hardwareTrigger);
1179 
1180                 // Plot the trigger for the verified plots.
1181                 plotTrigger(simTrigger, tiFlags, true);
1182 
1183                 // Update the verified count for each type of trigger
1184                 // for the local and global windows.
1185                 if (getTriggerTime(simTrigger) >= triggerWindowStart && getTriggerTime(simTrigger) <= triggerWindowEnd) {
1186                     if (clusterType == Cluster.class || clusterType == Cluster[].class) {
1187                         matchedTriggerCount[SOURCE_SIM_CLUSTER][ALL_TRIGGERS]++;
1188                         matchedTriggerCount[SOURCE_SIM_CLUSTER][LOCAL_WINDOW_TRIGGERS]++;
1189                     } else {
1190                         matchedTriggerCount[SOURCE_SSP_CLUSTER][ALL_TRIGGERS]++;
1191                         matchedTriggerCount[SOURCE_SSP_CLUSTER][LOCAL_WINDOW_TRIGGERS]++;
1192                     }
1193 
1194                     // Update the verified count for each active TI bit.
1195                     for (TriggerType trigger : TriggerType.values()) {
1196                         if (tiFlags[trigger.ordinal()]) {
1197                             if (clusterType == Cluster.class || clusterType == Cluster[].class) {
1198                                 matchedTriggerCount[SOURCE_SIM_CLUSTER][trigger.ordinal()]++;
1199                             } else {
1200                                 matchedTriggerCount[SOURCE_SSP_CLUSTER][trigger.ordinal()]++;
1201                             }
1202                         }
1203                     }
1204                 }
1205 
1206                 // A trigger can only be matched once, so no further
1207                 // matching is necessary.
1208                 continue simLoop;
1209             }
1210 
1211             // If this point is reached, all possible hardware triggers
1212             // have been checked and failed to match. This trigger then
1213             // fails to verify.
1214             logger.println("\t\tVerification failed!");
1215         }
1216 
1217         // The matched trigger set is equivalent in size to the number
1218         // of triggers that successfully passed verification.
1219         return matchedTriggers.size();
1220     }

1307     private static final double getPairTriggerTime(Trigger<?> trigger) throws IllegalArgumentException {
1308         // Get the cluster times and y positions as appropriate to the
1309         // cluster object type.
1310         int[] y = new int[2];
1311         double[] time = new double[2];
1312         if (trigger.getTriggerSource() instanceof SSPCluster[]
1313                 && ((SSPCluster[]) trigger.getTriggerSource()).length == 2) {
1314             SSPCluster[] pair = (SSPCluster[]) trigger.getTriggerSource();
1315             for (int i = 0; i < 2; i++) {
1316                 y[i] = TriggerModule.getClusterYIndex(pair[i]);
1317                 time[i] = TriggerModule.getClusterTime(pair[i]);
1318             }
1319         } else if (trigger.getTriggerSource() instanceof Cluster[]
1320                 && ((Cluster[]) trigger.getTriggerSource()).length == 2) {

1325             }
1326         } else {
1327             throw new IllegalArgumentException("Trigger source object type is not recognized.");
1328         }
1329 
1330         // The bottom cluster defines the trigger time.
1331         if (y[0] < 0 && y[1] > 0) {
1332             return time[0];
1333         } else if (y[1] < 0 && y[0] > 0) {
1334             return time[1];
1335         } else if (y[0] < 0 && y[1] < 0) {
1336             throw new IllegalArgumentException("Ambiguous cluster pair; both bottom clusters.");
1337         } else {
1338             throw new IllegalArgumentException("Ambiguous cluster pair; both top clusters.");
1339         }
1340     }
