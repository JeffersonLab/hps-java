14  public class TriggerDecisionCalculator {
15  
16      public enum TriggerType {
17          SINGLES0, SINGLES1, PAIR0, PAIR1, SINGLES1_SIM, PULSER, SINGLES1_RESULTS
18      }
19  
20      private List<TriggerType> passedTriggers = new ArrayList<TriggerType>();
21  
22      public TriggerDecisionCalculator(EventHeader event) {
23          process(event);
24      }
25  
26      public void add(TriggerType type) {
27          passedTriggers.add(type);
28      }
29  
30      public boolean passed(TriggerType type) {
31          for (TriggerType passed : passedTriggers) {
32              if (passed == type)
33                  return true;
34          }
35          return false;
36      }
37  
38      public void process(EventHeader event) {
39          List<GenericObject> triggerBanks = event.get(GenericObject.class, "TriggerBank");
40          for (GenericObject triggerBank : triggerBanks) {
41              if (AbstractIntData.getTag(triggerBank) == SSPData.BANK_TAG) {
42                  SSPData sspBank = new SSPData(triggerBank);
43  
44                  // recompute the decision for singles1
45                  List<SSPCluster> sspClusters = sspBank.getClusters();
46                  List<List<SinglesTrigger<SSPCluster>>> singleTriggers = constructSinglesTriggersFromSSP(sspClusters);
47                  if (singleTriggers.get(1).size() > 0)
48                      passedTriggers.add(TriggerType.SINGLES1_SIM);
49  
50                  // "use trigger results objects"
51                  List<SSPSinglesTrigger> triggerResults = sspBank.getSinglesTriggers();
52                  boolean passedSingles1 = false;
53                  resultLoop: for (SSPSinglesTrigger result : triggerResults) {
54                      // Check if this is a singles 1 trigger result.
55                      if (result.isSecondTrigger()) {
56                          // Check if the all the cuts passed.
57                          if (result.passCutEnergyMin() && result.passCutEnergyMax() && result.passCutHitCount()) {
58                              passedSingles1 = true;
59                              break resultLoop;
60                          }
61                      }
62                  }
63                  if (passedSingles1)
64                      passedTriggers.add(TriggerType.SINGLES1_RESULTS);
65              } else if (AbstractIntData.getTag(triggerBank) == TIData.BANK_TAG) {
66                  TIData tiData = new TIData(triggerBank);
67                  if (tiData.isSingle0Trigger())
68                      passedTriggers.add(TriggerType.SINGLES0);
69                  else if (tiData.isSingle1Trigger())
70                      passedTriggers.add(TriggerType.SINGLES1);
71                  else if (tiData.isPair0Trigger())
72                      passedTriggers.add(TriggerType.PAIR0);
73                  else if (tiData.isPair1Trigger())
74                      passedTriggers.add(TriggerType.PAIR1);
75                  else if (tiData.isPulserTrigger())
76                      passedTriggers.add(TriggerType.PULSER);
77              }
78          }
79      }

81      static public List<List<SinglesTrigger<SSPCluster>>> constructSinglesTriggersFromSSP(List<SSPCluster> clusters) {
82  
83          List<List<SinglesTrigger<SSPCluster>>> triggers = new ArrayList<List<SinglesTrigger<SSPCluster>>>(2);
84          // Instantiate the triggers lists.
85          for (int triggerNum = 0; triggerNum < 2; triggerNum++)
86              triggers.add(new ArrayList<SinglesTrigger<SSPCluster>>());
87  
88          final double ENERGY_CUT_LOW[] = {0.06, 0.4};
89          final double ENERGY_CUT_HIGH[] = {2.5, 1.1};
90          final int HIT_COUNT_CUT_LOW[] = {3, 3};
91          final boolean singlesCutsEnabled_ENERGY_MIN[] = {true, true};
92          final boolean singlesCutsEnabled_ENERGY_MAX[] = {true, true};
93          final boolean singlesCutsEnabled_HIT_COUNT[] = {true, true};
94  
95          for (SSPCluster cluster : clusters) {
96  
97              triggerLoop: for (int triggerNum = 0; triggerNum < 2; triggerNum++) {
98                  // For a cluster to have formed it is assumed to have
99                  // passed the cluster seed energy cuts. This can not
100                 // be verified since the SSP bank does not report
101                 // individual hit.
102                 boolean passSeedLow = true;
103                 boolean passSeedHigh = true;
104 
105                 // The remaining cuts may be acquired from trigger module.
106                 boolean passClusterLow = cluster.getEnergy() >= ENERGY_CUT_LOW[triggerNum] ? true : false;
107                 boolean passClusterHigh = cluster.getEnergy() <= ENERGY_CUT_HIGH[triggerNum] ? true : false;
108                 boolean passHitCount = cluster.getHitCount() >= HIT_COUNT_CUT_LOW[triggerNum] ? true : false;
109 
110                 // Make a trigger to store the results.
111                 SinglesTrigger<SSPCluster> trigger = new SinglesTrigger<SSPCluster>(cluster, triggerNum);
112                 trigger.setStateSeedEnergyLow(passSeedLow);
113                 trigger.setStateSeedEnergyHigh(passSeedHigh);
114                 trigger.setStateClusterEnergyLow(passClusterLow);
115                 trigger.setStateClusterEnergyHigh(passClusterHigh);
116                 trigger.setStateHitCount(passHitCount);
117 
118                 // A trigger will only be reported by the SSP if it
119                 // passes all of the enabled cuts for that trigger.
120                 // Check whether this trigger meets these conditions.
121                 // Set the singles cut statuses.
122                 if (singlesCutsEnabled_ENERGY_MIN[triggerNum] && !trigger.getStateClusterEnergyLow()) {
123                     continue triggerLoop;
124                 }
125                 if (singlesCutsEnabled_ENERGY_MAX[triggerNum] && !trigger.getStateClusterEnergyHigh()) {
126                     continue triggerLoop;
127                 }
128                 if (singlesCutsEnabled_HIT_COUNT[triggerNum] && !trigger.getStateHitCount()) {
129                     continue triggerLoop;
130                 }
131 
132                 // If all the necessary checks passed, store the new
133                 // trigger for verification.
134 
135                 triggers.get(triggerNum).add(trigger);
136             }
137         }
138         return triggers;
139     }
