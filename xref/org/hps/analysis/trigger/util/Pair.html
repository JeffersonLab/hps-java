3   /**
4    * Class <code>Pair</code> represents a pair of two objects.
5    * 
6    * @param <E> - The object type of the first element in the pair.
7    * @param <F> - The object type of the second element in the pair.
8    */
9   public class Pair<E, F> {
10      private final E firstObject;
11      private final F secondObject;
12      
13      /**
14       * Creates a pair of the two indicated objects.
15       * @param firstElement - The first object.
16       * @param secondElement - The second object.
17       */
18      public Pair(E firstElement, F secondElement) {
19          this.firstObject = firstElement;
20          this.secondObject = secondElement;
21      }
22      
23      /**
24       * Gets the first element of the pair.
25       * @return Returns the first element.
26       */
27      public E getFirstElement() {
28          return firstObject;
29      }
30      
31      /**
32       * Gets the second element of the pair.
33       * @return Returns the second element.
34       */
35      public F getSecondElement() {
36          return secondObject;
37      }
38  }
