1   package org.hps.analysis.trigger.util;
2   
3   import org.hps.record.triggerbank.TriggerModule;
4   
5   public class PairTrigger<E> extends SinglesTrigger<E> {
6       // Define the supported trigger cuts.
7       private static final String PAIR_ENERGY_SUM_LOW = TriggerModule.PAIR_ENERGY_SUM_LOW;
8       private static final String PAIR_ENERGY_SUM_HIGH = TriggerModule.PAIR_ENERGY_SUM_HIGH;
9       private static final String PAIR_ENERGY_DIFFERENCE_HIGH = TriggerModule.PAIR_ENERGY_DIFFERENCE_HIGH;
10      private static final String PAIR_ENERGY_SLOPE_LOW = TriggerModule.PAIR_ENERGY_SLOPE_LOW;
11      private static final String PAIR_COPLANARITY_HIGH = TriggerModule.PAIR_COPLANARITY_HIGH;
12      private static final String PAIR_TIME_COINCIDENCE = "pairTimeCoincidence";
13      
14      /**
15       * Instantiates a new <code>PairTrigger</code> with all cut
16       * states set to <code>false</code> and with the trigger source
17       * defined according to the specified object.
18       * @param source - The object from which the trigger cut states
19       * are derived.
20       */
21      public PairTrigger(E source, int triggerNum) {
22          // Instantiate the superclass.
23          super(source, triggerNum);
24          
25          // Add the supported cuts types.
26          addValidCut(PAIR_ENERGY_SUM_LOW);
27          addValidCut(PAIR_ENERGY_SUM_HIGH);
28          addValidCut(PAIR_ENERGY_DIFFERENCE_HIGH);
29          addValidCut(PAIR_ENERGY_SLOPE_LOW);
30          addValidCut(PAIR_COPLANARITY_HIGH);
31          addValidCut(PAIR_TIME_COINCIDENCE);
32      }
33      
34      /**
35       * Gets whether the pair energy sum lower bound cut was met.
36       * @return Returns <code>true</code> if the cut was met and
37       * <code>false</code> otherwise.
38       */
39      public boolean getStateEnergySumLow() {
40          return getCutState(PAIR_ENERGY_SUM_LOW);
41      }
42      
43      /**
44       * Gets whether the pair energy sum upper bound cut was met.
45       * @return Returns <code>true</code> if the cut was met and
46       * <code>false</code> otherwise.
47       */
48      public boolean getStateEnergySumHigh() {
49          return getCutState(PAIR_ENERGY_SUM_HIGH);
50      }
51      
52      /**
53       * Gets whether both the pair energy sum upper and lower bound cuts
54       * were met.
55       * @return Returns <code>true</code> if the cuts were met and
56       * <code>false</code> otherwise.
57       */
58      public boolean getStateEnergySum() {
59          return getCutState(PAIR_ENERGY_SUM_LOW) && getCutState(PAIR_ENERGY_SUM_HIGH);
60      }
61      
62      /**
63       * Gets whether the pair energy difference cut was met.
64       * @return Returns <code>true</code> if the cut was met and
65       * <code>false</code> otherwise.
66       */
67      public boolean getStateEnergyDifference() {
68          return getCutState(PAIR_ENERGY_DIFFERENCE_HIGH);
69      }
70      
71      /**
72       * Gets whether the pair energy slope cut was met.
73       * @return Returns <code>true</code> if the cut was met and
74       * <code>false</code> otherwise.
75       */
76      public boolean getStateEnergySlope() {
77          return getCutState(PAIR_ENERGY_SLOPE_LOW);
78      }
79      
80      /**
81       * Gets whether the pair coplanarity cut was met.
82       * @return Returns <code>true</code> if the cut was met and
83       * <code>false</code> otherwise.
84       */
85      public boolean getStateCoplanarity() {
86          return getCutState(PAIR_COPLANARITY_HIGH);
87      }
88      
89      /**
90       * Gets whether the time coincidence cut was met.
91       * @return Returns <code>true</code> if the cut was met and
92       * <code>false</code> otherwise.
93       */
94      public boolean getStateTimeCoincidence() {
95          return getCutState(PAIR_TIME_COINCIDENCE);
96      }
97      
98      /**
99       * Sets whether the conditions for the pair energy sum lower bound
100      * cut were met.
101      * @param state - <code>true</code> indicates that the cut conditions
102      * were met and <code>false</code> that they were not.
103      */
104     public void setStateEnergySumLow(boolean state) {
105         setCutState(PAIR_ENERGY_SUM_LOW, state);
106     }
107     
108     /**
109      * Sets whether the conditions for the pair energy sum upper bound
110      * cut were met.
111      * @param state - <code>true</code> indicates that the cut conditions
112      * were met and <code>false</code> that they were not.
113      */
114     public void setStateEnergySumHigh(boolean state) {
115         setCutState(PAIR_ENERGY_SUM_HIGH, state);
116     }
117     
118     /**
119      * Sets whether the conditions for the pair energy difference cut
120      * were met.
121      * @param state - <code>true</code> indicates that the cut conditions
122      * were met and <code>false</code> that they were not.
123      */
124     public void setStateEnergyDifference(boolean state) {
125         setCutState(PAIR_ENERGY_DIFFERENCE_HIGH, state);
126     }
127     
128     /**
129      * Sets whether the conditions for the pair energy slope cut were
130      * met.
131      * @param state - <code>true</code> indicates that the cut conditions
132      * were met and <code>false</code> that they were not.
133      */
134     public void setStateEnergySlope(boolean state) {
135         setCutState(PAIR_ENERGY_SLOPE_LOW, state);
136     }
137     
138     /**
139      * Sets whether the conditions for the pair coplanarity cut were
140      * met.
141      * @param state - <code>true</code> indicates that the cut conditions
142      * were met and <code>false</code> that they were not.
143      */
144     public void setStateCoplanarity(boolean state) {
145         setCutState(PAIR_COPLANARITY_HIGH, state);
146     }
147     
148     /**
149      * Sets whether the conditions for the time coincidence cut were
150      * met.
151      * @param state - <code>true</code> indicates that the cut conditions
152      * were met and <code>false</code> that they were not.
153      */
154     public void setStateTimeCoincidence(boolean state) {
155         setCutState(PAIR_TIME_COINCIDENCE, state);
156     }
157     
158     @Override
159     public String toString() {
160         return String.format("EClusterLow: %d; EClusterHigh %d; HitCount: %d; ESumLow: %d, ESumHigh: %d, EDiff: %d, ESlope: %d, Coplanarity: %d",
161                 getStateClusterEnergyLow() ? 1 : 0, getStateClusterEnergyHigh() ? 1 : 0,
162                 getStateHitCount() ? 1 : 0, getStateEnergySumLow() ? 1 : 0,
163                 getStateEnergySumHigh() ? 1 : 0, getStateEnergyDifference() ? 1 : 0,
164                 getStateEnergySlope() ? 1 : 0, getStateCoplanarity() ? 1 : 0);
165     }
166 }
