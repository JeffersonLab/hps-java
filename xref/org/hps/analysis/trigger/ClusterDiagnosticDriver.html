31  /**
32   * <code>ClusterDiagnosticDriver</code> performs diagnostic checks on
33   * run data to verify that clustering is being performed as expected
34   * and that there are no errors occurring. It takes GTP clusters created
35   * by the software simulation (simulated clusters) and compares them to
36   * GTP clusters reported by the hardware (hardware, or SSP, clusters).
37   * <br/><br/>
38   * Simulated clusters are compared to hardware clusters to find matches.
39   * Clusters are considered to match if they share the same crystal
40   * indices and time-stamp. Clusters are then "verified" by comparing
41   * their energies and hit counts. The thresholds for verification are
42   * programmable.
43   * <br/><br/>
44   * Statistics are produced for how well clusters matched and verified
45   * along with diagnostic plots. Additionally, two levels of verbose
46   * output may be enabled to provide detailed event readouts for the
47   * entire matching and verification process to aid in diagnosing issues.
48   * <br/><br/>
49   * Certain DAQ settings for the run are required for the diagnostic
50   * process. These are obtained via the <code>ConfigurationManager</code>
51   * class automatically. Note that a driver to initialize this class
52   * must be present in the event chain in order for the diagnostics to
53   * initialize and function.
54   */
55  public final class ClusterDiagnosticDriver extends Driver {

215     private static final Set<Point> getClusterHitIndices(Cluster cluster) {
216         // Store the possible hits in a set. The seed hit is excluded.
217         Set<Point> hitSet = new HashSet<Point>(8);
218         
219         // Iterate over the cluster hits and add any existing positions
220         // to the hit set.
221         for(CalorimeterHit hit : cluster.getCalorimeterHits()) {
222             hitSet.add(new Point(hit.getIdentifierFieldValue("ix"), hit.getIdentifierFieldValue("iy")));
223         }
224         
225         // Return the set of constituent hits.
226         return hitSet;
227     }
228     
229     private static final Set<Point> getClusterPossibleHitIndices(Cluster cluster) {
230         // Get the cluster seed position.
231         int ix = TriggerModule.getClusterXIndex(cluster);
232         int iy = TriggerModule.getClusterYIndex(cluster);
233         
234         // Store the possible hits in a set. The seed hit is excluded.
235         Set<Point> hitSet = new HashSet<Point>(8);
236         
237         // Get all eight adjacent hits.
238         xLoop:
239         for(int xMod = -1; xMod <= 2; xMod++) {
240             // Get the modified x position.
241             int hix = ix + xMod;
242             
243             // Values where |x| > 23 do not exist.
244             if(Math.abs(hix) > 23) { continue xLoop; }
245             
246             // Values of x = 0 do not exist. x = 1 and x = -1 are to
247             // be treated as adjacent.
248             if(hix == 0) { hix = ix == -1 ? 1 : -1; }
249             
250             yLoop:
251             for(int yMod = -1; yMod <= 2; yMod++) {
252                 // Get the modified y position.
253                 int hiy = iy + yMod;
254                 
255                 // Values of y = 0 and |y| > 5 do not exist.
256                 if(hiy == 0 || Math.abs(hiy) > 5) { continue yLoop; }
257                 
258                 // Add the potential hit position to the set.
259                 hitSet.add(new Point(hix, hiy));
260             }
261         }
262         
263         // Return the set of possible hit positions.
264         return hitSet;
265     }

267     private static final double getRatioError(double num, double sigmaNum, double den, double sigmaDen) {
268         double ratio = num / den;
269         return Math.abs(ratio) * Math.sqrt(Math.pow(sigmaNum / num, 2) + Math.pow(sigmaDen / den, 2));
270     }

272     /**
273      * Prints out final run verification statistics and generates
274      * efficiency plots.
275      */
276     @Override
277     public void endOfData() {
278         // Time and energy binned clustering efficiency plots are defined
279         // by the ratio of all verified software clusters to the verified
280         // software clusters that were matched.
281         AIDA.defaultInstance().histogramFactory().divide(matchedClustersTimeEfficiencyPlot,
282                 AIDA.defaultInstance().histogram1D(matchedClustersTimePlot), AIDA.defaultInstance().histogram1D(softwareClustersTimePlot));
283         AIDA.defaultInstance().histogramFactory().divide(matchedClustersEnergyEfficiencyPlot,
284                 AIDA.defaultInstance().histogram1D(matchedClustersEnergyPlot), AIDA.defaultInstance().histogram1D(softwareClustersEnergyPlot));
285         
286         // Calculate errors for efficiencies and failure rates.
287         double sigmaSimCount         = Math.sqrt(goodSimulatedClusterCount);
288         double sigmaMatched          = Math.sqrt(matchedClusters);
289         double sigmaFailedPosition   = Math.sqrt(failedMatchPosition);
290         double sigmaFailedTime       = Math.sqrt(failedMatchTime);
291         double sigmaFailedHitCount   = Math.sqrt(failedMatchHitCount);
292         double sigmaFailedEnergy     = Math.sqrt(failedMatchEnergy);
293         double sigmaFailedNearSeed   = Math.sqrt(failedNearSeedThreshold);
294         double sigmaFailedDeadtime   = Math.sqrt(failedPositionDeadtime);
295         double sigmaFailedNegative   = Math.sqrt(failedNegativeEnergyHit);
296         double sigmaFailedHCDeadtime = Math.sqrt(failedHitCountDeadtime);
297         double sigmaFailedCloneBug   = Math.sqrt(failedCloneBug);
298         
299         // Get the maximum number of digits needed to display the
300         // largest of the counts. This should always be either the
301         // total number hardware or simulated clusters.
302         String countDisp = "%" + TriggerDiagnosticUtil.getDigits(Math.max(simulatedClusterCount, goodSimulatedClusterCount)) + "d";
303         
304         // Print the global run statistics for cluster verification.
305         System.out.println("Cluster Verification:");
306         System.out.printf("\tSimulated Clusters     :: " + countDisp + "%n", simulatedClusterCount);
307         System.out.printf("\tUnclipped Sim Clusters :: " + countDisp + "%n", goodSimulatedClusterCount);
308         System.out.printf("\tHardware Clusters      :: " + countDisp + "%n", hardwareClusterCount);
309         System.out.printf("\tClusters Matched       :: " + countDisp + "%n", matchedClusters);
310         
311         System.out.printf("\tFailed (Position)      :: " + countDisp, failedMatchPosition);
312         if(failedMatchPosition == 0 || goodSimulatedClusterCount == 0) {
313             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 0.0, 0.0);
314         } else {
315             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 100.0 * failedMatchPosition / goodSimulatedClusterCount,
316                     getRatioError(failedMatchPosition, sigmaFailedPosition, goodSimulatedClusterCount, sigmaSimCount));
317         }
318         
319         System.out.printf("\t> Failed (Deadtime)    :: " + countDisp, failedPositionDeadtime);
320         if(failedPositionDeadtime == 0 || goodSimulatedClusterCount == 0) {
321             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 0.0, 0.0);
322         } else {
323             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 100.0 * failedPositionDeadtime / goodSimulatedClusterCount,
324                     getRatioError(failedPositionDeadtime, sigmaFailedDeadtime, goodSimulatedClusterCount, sigmaSimCount));
325         }
326         
327         System.out.printf("\t> Failed (Clone Bug)   :: " + countDisp, failedCloneBug);
328         if(failedPositionDeadtime == 0 || goodSimulatedClusterCount == 0) {
329             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 0.0, 0.0);
330         } else {
331             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 100.0 * failedCloneBug / goodSimulatedClusterCount,
332                     getRatioError(failedCloneBug, sigmaFailedCloneBug, goodSimulatedClusterCount, sigmaSimCount));
333         }
334         
335         System.out.printf("\t> Failed (Seed Thresh) :: " + countDisp, failedNearSeedThreshold);
336         if(failedNearSeedThreshold == 0 || goodSimulatedClusterCount == 0) {
337             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 0.0, 0.0);
338         } else {
339             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 100.0 * failedNearSeedThreshold / goodSimulatedClusterCount,
340                     getRatioError(failedNearSeedThreshold, sigmaFailedNearSeed, goodSimulatedClusterCount, sigmaSimCount));
341         }
342         
343         System.out.printf("\tFailed (Time)          :: " + countDisp, failedMatchTime);
344         if(failedMatchTime == 0 || goodSimulatedClusterCount == 0) {
345             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 0.0, 0.0);
346         } else {
347             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 100.0 * failedMatchTime / goodSimulatedClusterCount,
348                     getRatioError(failedMatchTime, sigmaFailedTime, goodSimulatedClusterCount, sigmaSimCount));
349         }
350         
351         System.out.printf("\tFailed (Hit Count)     :: " + countDisp, failedMatchHitCount);
352         if(failedMatchHitCount == 0 || goodSimulatedClusterCount == 0) {
353             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 0.0, 0.0);
354         } else {
355             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 100.0 * failedMatchHitCount / goodSimulatedClusterCount,
356                     getRatioError(failedMatchHitCount, sigmaFailedHitCount, goodSimulatedClusterCount, sigmaSimCount));
357         }
358         
359         System.out.printf("\t> Failed (Deadtime)    :: " + countDisp, failedHitCountDeadtime);
360         if(failedPositionDeadtime == 0 || goodSimulatedClusterCount == 0) {
361             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 0.0, 0.0);
362         } else {
363             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 100.0 * failedHitCountDeadtime / goodSimulatedClusterCount,
364                     getRatioError(failedHitCountDeadtime, sigmaFailedHCDeadtime, goodSimulatedClusterCount, sigmaSimCount));
365         }
366         
367         System.out.printf("\t> Failed (Negative En) :: " + countDisp, failedNegativeEnergyHit);
368         if(failedNegativeEnergyHit == 0 || goodSimulatedClusterCount == 0) {
369             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 0.0, 0.0);
370         } else {
371             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 100.0 * failedNegativeEnergyHit / goodSimulatedClusterCount,
372                     getRatioError(failedNegativeEnergyHit, sigmaFailedNegative, goodSimulatedClusterCount, sigmaSimCount));
373         }
374         
375         System.out.printf("\tFailed (Energy)        :: " + countDisp, failedMatchEnergy);
376         if(failedMatchEnergy == 0 || goodSimulatedClusterCount == 0) {
377             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 0.0, 0.0);
378         } else {
379             System.out.printf("   (%7.3f%%  %7.3f%%)%n", 100.0 * failedMatchEnergy / goodSimulatedClusterCount,
380                     getRatioError(failedMatchEnergy, sigmaFailedEnergy, goodSimulatedClusterCount, sigmaSimCount));
381         }
382         
383         if(simulatedClusterCount == 0 || goodSimulatedClusterCount == 0) {
384             System.out.printf("\tCluster Efficiency     :: %7.3f%%  %7.3f%%%n", 0.0, 0.0);
385         } else {
386             System.out.printf("\tCluster Efficiency     :: %7.3f%%  %7.3f%%%n", 100.0 * matchedClusters / goodSimulatedClusterCount,
387                     100.0 * getRatioError(matchedClusters, sigmaMatched, goodSimulatedClusterCount, sigmaSimCount));
388         }
389         
390         // Create and populate the efficiency over time plot.
391         AIDA.defaultInstance().cloud2D(runtimeEfficiencyPlot, efficiencyPlotEntries.size());
392         for(Pair<Long, double[]> entry : efficiencyPlotEntries) {
393             // Calculate the value for each type of efficiency.
394             double localEfficiency = 100.0 * entry.getSecondElement()[1] / entry.getSecondElement()[0];
395             
396             // Convert the time to units of seconds.
397             long time = Math.round(entry.getFirstElement() / 1000000.0);
398             
399             // If the value is properly defined, add it to the plot.
400             if(!Double.isNaN(localEfficiency)) {
401                 AIDA.defaultInstance().cloud2D(runtimeEfficiencyPlot).fill(time, localEfficiency);
402             }
403         }
404     }

406     /**
407      * Obtains all necessary collections, performs cluster verification,
408      * and outputs event statistics. Also handles logging.
409      */
410     @Override
411     public void process(EventHeader event) {
412         // If the DAQ configuration manager is not yet initialized,
413         // it is not possible to analyze the event.
414         if(!ConfigurationManager.isInitialized()) {
415             return;
416         }
417         
418         // Clear the logger at the start of each event.
419         logger.clearLog();
420         
421         // If this is the first event, set the starting time stamp.
422         // The end time is always the time of the current event.
423         localEndTime = event.getTimeStamp();
424         if(localStartTime == Long.MIN_VALUE) {
425             localStartTime = event.getTimeStamp();
426             firstStartTime = event.getTimeStamp();
427         }
428         
429         // Output the clustering diagnostic header.
430         logger.printNewLine(2);
431         logger.println("======================================================================");
432         logger.println("==== Clustering Diagnostics ==========================================");
433         logger.println("======================================================================");
434         logger.printf("Event Number :: %d%n", event.getEventNumber());
435         logger.printf("Event Time   :: %d%n", event.getTimeStamp());
436         
437         // Get the needed object collections.
438         List<CalorimeterHit> hits = getCalorimeterHits(event, hitCollectionName);
439         List<SSPCluster> hardwareClusters = getHardwareClusters(event, bankCollectionName);
440         List<Cluster> simulatedClusters = getSimulatedClusters(event, clusterCollectionName);
441         
442         // Verify that all of the collections are present.
443         logger.printNewLine(2);
444         logger.printf("Hit Collection Found               :: %b%n", hits != null);
445         logger.printf("Hardware Cluster Collection Found  :: %b%n", hardwareClusters != null);
446         logger.printf("Simulated Cluster Collection Found :: %b%n", simulatedClusters != null);
447         if(hits == null || hardwareClusters == null || simulatedClusters == null) {
448             logger.println("TriggerDiagnostics: One or more collections is missing from event. Verification can not proceed and event will be skipped.");
449             if(verbose) { logger.printLog(); }
450             return;
451         }
452         
453         // If noise event culling is enabled, check if this is a noise
454         // event.
455         if(skipNoiseEvents) {
456             if(hits.size() >= noiseEventThreshold) {
457                 logger.println("TriggerDiagnostics: Event exceeds the noise threshold for total number of hits and will be skipped.");
458                 if(verbose) { logger.printLog(); }
459                 return;
460             }
461         }
462         
463         // Output the FADC hits, hardware clusters, and simulated
464         // clusters to the diagnostic logger.
465         logger.printNewLine(2);
466         logger.println("==== Event Summary ===================================================");
467         logger.println("======================================================================");
468         
469         // Output the FADC hits.
470         logger.println("FADC Hits:");
471         for(CalorimeterHit hit : hits) { logger.println("\t" + toString(hit)); }
472         if(hits.isEmpty()) { logger.println("\tNone"); }
473         
474         // Output the simulated clusters from the software.
475         logger.printNewLine(1);
476         logger.println("Software Clusters:");
477         for(Cluster cluster : simulatedClusters) {
478             logger.printf("\t%s%n", TriggerDiagnosticUtil.clusterToString(cluster));
479             for(CalorimeterHit hit : cluster.getCalorimeterHits()) {
480                 logger.println("\t\t> " + toString(hit));
481             }
482         }
483         if(simulatedClusters.isEmpty()) { logger.println("\tNone"); }
484         
485         // Output the reported clusters from the hardware.
486         logger.printNewLine(1);
487         logger.println("Hardware Clusters:");
488         for(SSPCluster cluster : hardwareClusters) {
489             logger.printf("\t%s%n", TriggerDiagnosticUtil.clusterToString(cluster));
490         }
491         if(hardwareClusters.isEmpty()) { logger.println("\tNone"); }
492         
493         // When hits are written to data by the FADC, the pulse height
494         // is only written within the bounds of the event window. Thus,
495         // if a hit is too close to the beginning or the end of the
496         // event window, it will experience "pulse-clipping" where the
497         // hit loses a part of its energy. Clusters containing these
498         // hits will often fail verification because the reduced energy
499         // despite this not indicating an actual problem. To avoid
500         // this, simulated clusters that are at risk of pulse-clipping
501         // are excluded from cluster verification.
502         logger.printNewLine(2);
503         logger.println("==== Pulse-Clipping Verification =====================================");
504         logger.println("======================================================================");
505         
506         // Iterate through each simulated cluster and keep only the
507         // clusters safe from pulse-clipping.
508         List<Cluster> goodSimulatedClusters = new ArrayList<Cluster>();
509         logger.println("Simulated Cluster Pulse-Clipping Check:");
510         for(Cluster cluster : simulatedClusters) {
511             boolean isSafe = TriggerDiagnosticUtil.isVerifiable(cluster, nsa, nsb, windowWidth);
512             if(isSafe) { goodSimulatedClusters.add(cluster); }
513             logger.printf("\t%s [ %7s ]%n", TriggerDiagnosticUtil.clusterToString(cluster),
514                     isSafe ? "Safe" : "Clipped");
515         }
516         if(goodSimulatedClusters.isEmpty()) { logger.println("\tNone"); }
517         
518         // Print the header for cluster verification.
519         logger.printNewLine(2);
520         logger.println("==== Cluster Accuracy Verification ===================================");
521         logger.println("======================================================================");
522         
523         // Generate a list of matched simulated/hardware cluster pairs
524         // and the verification status of each pair.
525         List<ClusterMatchedPair> matchedPairs = matchSimulatedToHardware(goodSimulatedClusters, hardwareClusters,
526                 energyVerificationThreshold, hitVerificationThreshold, logger);
527         
528         // Increment the number of clusters of each type processed.
529         hardwareClusterCount += hardwareClusters.size();
530         simulatedClusterCount += simulatedClusters.size();
531         goodSimulatedClusterCount += goodSimulatedClusters.size();
532         localGoodSimulatedClusterCount += goodSimulatedClusters.size();
533         
534         // Iterate over the list of pairs and extract statistical data
535         // for this set of clusters.
536         int matchedClusters = 0;
537         int failedMatchTime = 0;
538         int failedMatchEnergy = 0;
539         int failedMatchHitCount = 0;
540         int failedMatchPosition = 0;
541         for(ClusterMatchedPair pair : matchedPairs) {
542             if(pair.isMatch()) { matchedClusters++; }
543             if(pair.isTimeFailState()) { failedMatchTime++; }
544             if(pair.isEnergyFailState()) { failedMatchEnergy++; }
545             if(pair.isHitCountFailState()) {
546                 // Increment the general hit count fail state count.
547                 failedMatchHitCount++;
548                 
549                 // If the simulated cluster is different in hit count,
550                 // but has the same energy, it is likely caused by a
551                 // near-zero energy hit that registered as negative to
552                 // the simulation, and was consequently ignored.
553                 if(Math.abs(TriggerModule.getValueClusterTotalEnergy(pair.getReconstructedCluster())
554                         - TriggerModule.getValueClusterTotalEnergy(pair.getSSPCluster())) <= energyVerificationThreshold) {
555                     failedNegativeEnergyHit++;
556                 }
557                 
558                 // Deadtime can also cause a failure in hit count. A
559                 // simple check of this is to see if either the cluster
560                 // occurred within 32 ns of the event start, or if any
561                 // channel that would be included by the cluster for
562                 // which there is no hit has a hit within this time
563                 // frame that is also within 32 ns of the cluster.
564                 else {
565                     // Check for the case that the cluster itself is
566                     // within the deadtime uncertainty region.
567                     if(TriggerModule.getClusterTime(pair.getReconstructedCluster()) < 32) {
568                         failedHitCountDeadtime++;
569                     }
570                     
571                     // Otherwise, determine which hits are not included
572                     // in the cluster and see if a hit exists which could
573                     // produce a deadtime error in that channel.
574                     else {
575                         // Get a list of the hits that could exist in
576                         // the cluster.
577                         Set<Point> actualHits = getClusterHitIndices(pair.getReconstructedCluster());
578                         Set<Point> possibleHits = getClusterPossibleHitIndices(pair.getReconstructedCluster());
579                         
580                         // Iterate over each hit and look for one that
581                         // could produce a deadtime error.
582                         hitLoop:
583                         for(CalorimeterHit hit : hits) {
584                             // The hit is required to be within the
585                             // first 32 of the event.
586                             if(hit.getTime() >= 32) { continue hitLoop; }
587                             
588                             // The hit must be within 32 ns of the
589                             // cluster  time and also occur before
590                             // the cluster time or simultaneously.
591                             if((TriggerModule.getClusterTime(pair.getReconstructedCluster()) < hit.getTime()) 
592                                     || (TriggerModule.getClusterTime(pair.getReconstructedCluster()) - hit.getTime() >= 32)) {
593                                 continue hitLoop;
594                             }
595                             
596                             // The hit must occur on a channel that
597                             // falls within the 3x3 cluster range, and
598                             // there must not already be a hit on that
599                             // channel in the cluster.
600                             Point ixy = new Point(hit.getIdentifierFieldValue("ix"), hit.getIdentifierFieldValue("iy"));
601                             if(possibleHits.contains(ixy) && !actualHits.contains(ixy)) {
602                                 failedHitCountDeadtime++;
603                                 break hitLoop;
604                             }
605                         }
606                     }
607                 }
608             }
609             if(pair.isPositionFailState()) {
610                 // Increment the general position fail state count.
611                 failedMatchPosition++;
612                 
613                 // If the simulated cluster is within the allowed energy
614                 // range of the seed energy threshold, it is likely to
615                 // have failed because of energy differences between the
616                 // hardware and the simulation. Track these.
617                 double energyDifference = Math.abs(TriggerModule.getValueClusterTotalEnergy(pair.getReconstructedCluster())
618                         - ConfigurationManager.getInstance().getGTPConfig().getSeedEnergyCutConfig().getLowerBound());
619                 if(energyDifference <= energyVerificationThreshold) {
620                     failedNearSeedThreshold++;
621                 }
622                 
623                 // If the simulated cluster has a seed hit which is
624                 // less than 32 ns in time, it is possible that it was
625                 // missed by the hardware due to a dead time issue.
626                 // This is most likely to the case for any cluster where
627                 // there exists a t = 0 cluster with a hit on the same
628                 // channel. Check for this case and track the instances
629                 // where it occurred.
630                 else if(TriggerModule.getClusterTime(pair.getReconstructedCluster()) < 32) {
631                     // Get the cluster channel.
632                     int ix = TriggerModule.getClusterXIndex(pair.getReconstructedCluster());
633                     int iy = TriggerModule.getClusterYIndex(pair.getReconstructedCluster());
634                     
635                     // Check for a t = 0 cluster with a hit on the same
636                     // channel.
637                     deadtimeLoop:
638                     for(Cluster cluster : simulatedClusters) {
639                         if(TriggerModule.getClusterTime(cluster) == 0) {
640                             if(TriggerModule.getClusterXIndex(cluster) == ix && TriggerModule.getClusterYIndex(cluster) == iy) {
641                                 failedPositionDeadtime++;
642                                 break deadtimeLoop;
643                             }
644                         }
645                     }
646                 }
647                 
648                 // Check for more complicated fail state causes.
649                 else {
650                     // The "clone bug" occurs when the hardware reports
651                     // a fake cluster at the same position as an existing
652                     // cluster, but at the same time as a cluster which
653                     // should exist, but is missed.
654                     // Check if more than SSP cluster exists at the
655                     // same position as another cluster, and one of them
656                     // matches in time.
657                     double time = TriggerModule.getClusterTime(pair.getReconstructedCluster());
658                     Map<Point, Integer> countMap = new HashMap<Point, Integer>();
659                     Map<Point, SSPCluster> timeMap = new HashMap<Point, SSPCluster>();
660                     for(SSPCluster sspCluster : hardwareClusters) {
661                         Point ixy = new Point(TriggerModule.getClusterXIndex(sspCluster), TriggerModule.getClusterYIndex(sspCluster));
662                         if(countMap.containsKey(ixy)) { countMap.put(ixy, countMap.get(ixy) + 1); }
663                         else { countMap.put(ixy, new Integer(1)); }
664                         if(TriggerModule.getClusterTime(sspCluster) == time) { timeMap.put(ixy, sspCluster); }
665                     }
666                     
667                     // Check each cluster position. If there are more
668                     // clusters than one at that position, and one of
669                     // them has the same time as the position failure
670                     // cluster, the clone bug could have occurred.
671                     unmatchedLoop:
672                     for(Map.Entry<Point, Integer> entry : countMap.entrySet()) {
673                         // Check that at least two clusters exist at
674                         // this position and that one of them matched
675                         // the fail state cluster in time.
676                         if(entry.getValue() >= 2 && timeMap.containsKey(entry.getKey())) {
677                             // Get the matching cluster.
678                             SSPCluster cloneTestCluster = timeMap.get(entry.getKey());
679                             
680                             // The cluster that matched the fail state
681                             // cluster in time must be itself unmatched.
682                             for(ClusterMatchedPair match : matchedPairs) {
683                                 // If the cluster matching the fail state
684                                 // cluster in time is matched, this is
685                                 // not the clone bug.
686                                 if(match.getSSPCluster() == cloneTestCluster) {
687                                     continue unmatchedLoop;
688                                 }
689                             }
690                             
691                             // If not match was found for the cluster
692                             // that matches the fail state cluster in
693                             // time, this is probably the clone bug.
694                             failedCloneBug++;
695                             break unmatchedLoop;
696                         }
697                     }
698                 }
699             }
700         }
701         
702         // Increment the statistics.
703         this.matchedClusters += matchedClusters;
704         localMatchedClusters += matchedClusters;
705         this.failedMatchTime += failedMatchTime;
706         this.failedMatchEnergy += failedMatchEnergy;
707         this.failedMatchHitCount += failedMatchHitCount;
708         this.failedMatchPosition += failedMatchPosition;
709         
710         // Output the results summary header.
711         logger.printNewLine(2);
712         logger.println("==== Cluster Verification Summary ====================================");
713         logger.println("======================================================================");
714         
715         // Output the cluster pairs that successfully verified.
716         logger.println();
717         logger.println("Verified Simulated/Hardware Cluster Pairs:");
718         if(matchedClusters != 0) {
719             for(ClusterMatchedPair pair : matchedPairs) {
720                 if(pair.isMatch()) {
721                     logger.printf("\t%s --> %s%n",
722                             TriggerDiagnosticUtil.clusterToString(pair.getReconstructedCluster()),
723                             TriggerDiagnosticUtil.clusterToString(pair.getSSPCluster()));
724                 }
725             }
726         } else { logger.println("\tNone"); }
727         
728         // Output the event statistics to the diagnostics logger.
729         logger.println();
730         logger.println("Event Statistics:");
731         logger.printf("\tSafe Clusters      :: %d%n", goodSimulatedClusters.size());
732         logger.printf("\tClusters Matched   :: %d%n", matchedClusters);
733         logger.printf("\tFailed (Position)  :: %d%n", failedMatchPosition);
734         logger.printf("\tFailed (Time)      :: %d%n", failedMatchTime);
735         logger.printf("\tFailed (Energy)    :: %d%n", failedMatchEnergy);
736         logger.printf("\tFailed (Hit Count) :: %d%n", failedMatchHitCount);
737         if(goodSimulatedClusters.isEmpty()) {
738             logger.printf("\tCluster Efficiency :: N/A %n", 100.0 * matchedClusters / goodSimulatedClusters.size());
739         } else {
740             logger.printf("\tCluster Efficiency :: %3.0f%%%n", 100.0 * matchedClusters / goodSimulatedClusters.size());
741         }
742         
743         // The verification process is considered to fail when any not
744         // pulse-clipped simulated cluster fails to verify. If it is
745         // appropriate to the verbosity settings, print the event log.
746         if(verbose || (printOnFail && matchedPairs.size() != matchedClusters)) {
747             logger.printLog();
748         }
749         
750         // If the local time window has elapsed, update the efficiency
751         // plot values and reset the counters.
752         if(localEndTime - localStartTime >= localWindowSize) {
753             // Values are stored in an array within a list. Each array
754             // index corresponds to specific value and each array entry
755             // corresponds to an individual local window.
756             double[] databank = new double[2];
757             databank[0] = localGoodSimulatedClusterCount;
758             databank[1] = localMatchedClusters;
759             
760             // The efficiency list also stores the time that corresponds
761             // to the particular databank. This is defined as half way
762             // between the start and end time of the current window.
763             Long time = localEndTime - firstStartTime - ((localEndTime - localStartTime) / 2);
764             
765             // Store the databank in the list.
766             efficiencyPlotEntries.add(new Pair<Long, double[]>(time, databank));
767             
768             // Reset all of the counters.
769             localMatchedClusters = 0;
770             localGoodSimulatedClusterCount = 0;
771             
772             // The new window start time is the current event time.
773             localStartTime = event.getTimeStamp();
774         }
775     }

886         // was located, then there are no SSP cluster present.

909     /**
910      * Performs the verification check for cluster hit counts.
911      * @param simCluster - The simulated cluster to check.
912      * @param sspCluster - The hardware cluster to check.
913      * @param hitWindow - The range by which hit counts are allowed to
914      * differ between the clusters.
915      * @return Returns <code>true</code> if the hit counts match to
916      * within threshold and <code>false</code> otherwise.
917      */
918     private static final boolean isHitMatch(Cluster simCluster, SSPCluster sspCluster, int hitWindow) {
919         // Get the hit counts for both clusters.
920         double simHitCount = TriggerModule.getClusterHitCount(simCluster);
921         double sspHitCount = TriggerModule.getClusterHitCount(sspCluster);
922         
923         // The hardware does not store cluster hit counts as higher than
924         // 7, so if the software hit count is 8 or 9, we must treat it
925         // as 7 instead to match this behavior.
926         if(simHitCount > 7) { simHitCount = 7; }
927         
928         // Perform the hit count check.
929         return (sspHitCount >= simHitCount - hitWindow && sspHitCount <= simHitCount + hitWindow);
930     }

932     /**
933      * Performs cluster matching between a collection of simulated
934      * clusters and a collection of hardware clusters using the strictly
935      * time-compliant algorithm. Simulated clusters are matched with
936      * a hardware cluster by comparing the x- and y-indices of the two
937      * clusters, as well as their time-stamps. If all of these values
938      * match, the clusters are considered to be the same. The cluster
939      * then undergoes verification by requiring that both the cluster
940      * energies and hit counts are within a certain programmable range
941      * of one another. Matched clusters are then stored along with a
942      * flag that indicates whether they were properly verified or not.
943      * Simulated clusters that do not match any hardware cluster in both
944      * position and time are treated as failing to have verified.
945      * @param simulatedClusters - A collection of GTP clusters generated
946      * by the software simulation.
947      * @param hardwareClusters - A collection of GTP clusters reported
948      * in the SSP bank by the hardware.
949      * @param energyWindow - The window of allowed deviation between
950      * the simulated cluster and hardware cluster energies. Units are
951      * in GeV.
952      * @param hitWindow - The window of allowed deviation between
953      * the simulated cluster and hardware cluster hit counts.
954      * @param logger - The logger object to which diagnostic text should
955      * be written.
956      * @return Returns a <code>List</code> containing all the matched
957      * simulated/hardware cluster pairs as well as their verification
958      * statuses.
959      */
960     private static final List<ClusterMatchedPair> matchSimulatedToHardware(Collection<Cluster> simulatedClusters,
961             Collection<SSPCluster> hardwareClusters, double energyWindow, int hitWindow, LocalOutputLogger logger) {
962         // Store the list of clusters, along with their matches (if
963         // applicable) and their pair verification status.
964         List<ClusterMatchedPair> pairList = new ArrayList<ClusterMatchedPair>();
965         
966         // Store the clusters which have been successfully paired.
967         Set<SSPCluster> hardwareMatched = new HashSet<SSPCluster>(hardwareClusters.size());
968         
969         // Find simulated/hardware cluster matched pairs.
970         softwareLoop:
971         for(Cluster simCluster : simulatedClusters) {
972             // Increment the plot counts for simulated clusters.
973             AIDA.defaultInstance().histogram1D(softwareClustersTimePlot).fill(TriggerModule.getClusterTime(simCluster));
974             AIDA.defaultInstance().histogram1D(softwareClustersEnergyPlot).fill(TriggerModule.getValueClusterTotalEnergy(simCluster));
975             
976             // Track whether a position-matched cluster was found.
977             boolean matchedPosition = false;
978             
979             // VERBOSE :: Output the cluster being matched.
980             logger.printf("Considering %s%n", TriggerDiagnosticUtil.clusterToString(simCluster));
981             
982             // Search through the hardware clusters for a match.
983             hardwareLoop:
984             for(SSPCluster hardwareCluster : hardwareClusters) {
985                 // VERBOSE :: Output the hardware cluster being considered.
986                 logger.printf("\t%s ", TriggerDiagnosticUtil.clusterToString(hardwareCluster));
987                 
988                 // Clusters must be matched in a one-to-one relationship,
989                 // so clusters that have already been matched should
990                 // be skipped.
991                 if(hardwareMatched.contains(hardwareCluster)) {
992                     logger.printf("[ %7s; %9s ]%n", "fail", "matched");
993                     continue hardwareLoop;
994                 }
995                 
996                 // If the clusters are the same, they must have the same
997                 // x- and y-indices. 
998                 if(TriggerModule.getClusterXIndex(simCluster) != TriggerModule.getClusterXIndex(hardwareCluster)
999                         || TriggerModule.getClusterYIndex(simCluster) != TriggerModule.getClusterYIndex(hardwareCluster)) {
1000                     logger.printf("[ %7s; %9s ]%n", "fail", "position");
1001                     continue hardwareLoop;
1002                 }
1003                 
1004                 // Note that the cluster matched another cluster in
1005                 // position. This is used to determine why a cluster
1006                 // failed to verify, if necessary.
1007                 matchedPosition = true;
1008                 
1009                 // If the clusters are the same, they should occur at
1010                 // the same time as well.
1011                 if(TriggerModule.getClusterTime(simCluster) != TriggerModule.getClusterTime(hardwareCluster)) {
1012                     logger.printf("[ %7s; %9s ]%n", "fail", "time");
1013                     continue hardwareLoop;
1014                 }
1015                 
1016                 // It is impossible for two clusters to exist at the
1017                 // same place and the same time, so clusters that pass
1018                 // both the time comparison and position comparison are
1019                 // assumed to be the same.
1020                 hardwareMatched.add(hardwareCluster);
1021                 
1022                 // Plot a comparison of the two clusters' energies and
1023                 // hit counts.
1024                 AIDA.defaultInstance().histogram1D(matchedClustersHitDiffPlot).fill(Math.abs(TriggerModule.getClusterHitCount(simCluster)
1025                         - TriggerModule.getClusterHitCount(hardwareCluster)));
1026                 AIDA.defaultInstance().histogram2D(matchedClusters2DHitDiffPlot).fill(TriggerModule.getClusterHitCount(simCluster),
1027                         TriggerModule.getClusterHitCount(hardwareCluster));
1028                 AIDA.defaultInstance().histogram1D(matchedClustersEnergyDiffPlot).fill(Math.abs(TriggerModule.getValueClusterTotalEnergy(simCluster)
1029                         - TriggerModule.getValueClusterTotalEnergy(hardwareCluster)));
1030                 AIDA.defaultInstance().histogram2D(matchedClusters2DEnergyDiffPlot).fill(TriggerModule.getValueClusterTotalEnergy(simCluster),
1031                         TriggerModule.getValueClusterTotalEnergy(hardwareCluster));
1032                 
1033                 // While time and position matched clusters are considered
1034                 // to be the same cluster, the clusters must have similar
1035                 // energies and hit counts to be properly verified. First
1036                 // perform the hit count check. The hardware cluster must
1037                 // match the simulated cluster hit count to within a given
1038                 // bound.
1039                 //if(TriggerModule.getClusterHitCount(hardwareCluster) >= TriggerModule.getClusterHitCount(simCluster) - hitWindow &&
1040                 //        TriggerModule.getClusterHitCount(hardwareCluster) <= TriggerModule.getClusterHitCount(simCluster) + hitWindow) {
1041                 if(isHitMatch(simCluster, hardwareCluster, hitWindow)) {
1042                     // Next, check that the hardware cluster matches the
1043                     // simulated cluster in energy to within a given
1044                     // bound.
1045                     if(TriggerModule.getValueClusterTotalEnergy(hardwareCluster) >= TriggerModule.getValueClusterTotalEnergy(simCluster) - energyWindow
1046                             && TriggerModule.getValueClusterTotalEnergy(hardwareCluster) <= TriggerModule.getValueClusterTotalEnergy(simCluster) + energyWindow) {
1047                         // The cluster is a match.
1048                         pairList.add(new ClusterMatchedPair(simCluster, hardwareCluster, ClusterMatchedPair.CLUSTER_STATE_MATCHED));
1049                         logger.printf("[ %7s; %9s ]%n", "success", "matched");
1050                         
1051                         // Increment the matched cluster plots.
1052                         AIDA.defaultInstance().histogram1D(matchedClustersTimePlot).fill(TriggerModule.getClusterTime(simCluster));
1053                         AIDA.defaultInstance().histogram1D(matchedClustersEnergyPlot).fill(TriggerModule.getValueClusterTotalEnergy(simCluster));
1054                         
1055                         // Since the cluster is already matched, there
1056                         // is no need to perform further checks.
1057                         continue softwareLoop;
1058                     }
1059                     
1060                     // If the energies of the two clusters are not
1061                     // sufficiently close, the clusters fail to verify.
1062                     else {
1063                         pairList.add(new ClusterMatchedPair(simCluster, hardwareCluster, ClusterMatchedPair.CLUSTER_STATE_FAIL_ENERGY));
1064                         logger.printf("[ %7s; %9s ]%n", "fail", "energy");
1065                         AIDA.defaultInstance().histogram1D(failedEnergyHitDiffPlot).fill(Math.abs(TriggerModule.getClusterHitCount(simCluster)
1066                                 - TriggerModule.getClusterHitCount(hardwareCluster)));
1067                         AIDA.defaultInstance().histogram1D(failedEnergyEnergyDiffPlot).fill(Math.abs(TriggerModule.getValueClusterTotalEnergy(simCluster)
1068                                 - TriggerModule.getValueClusterTotalEnergy(hardwareCluster)));
1069                         continue softwareLoop;
1070                     } // End energy check.
1071                 }
1072                 
1073                 // If the hit counts of the two clusters are not
1074                 // sufficiently close, the clusters fail to verify.
1075                 else {
1076                     pairList.add(new ClusterMatchedPair(simCluster, hardwareCluster, ClusterMatchedPair.CLUSTER_STATE_FAIL_HIT_COUNT));
1077                     logger.printf("[ %7s; %9s ]%n", "fail", "hit count");
1078                     AIDA.defaultInstance().histogram1D(failedHitCountHitDiffPlot).fill(Math.abs(TriggerModule.getClusterHitCount(simCluster)
1079                             - TriggerModule.getClusterHitCount(hardwareCluster)));
1080                     AIDA.defaultInstance().histogram1D(failedHitCountEnergyDiffPlot).fill(Math.abs(TriggerModule.getValueClusterTotalEnergy(simCluster)
1081                             - TriggerModule.getValueClusterTotalEnergy(hardwareCluster)));
1082                     continue softwareLoop;
1083                 } // End hit count check.
1084             } // End hardware loop.
1085             
1086             // This point may only be reached if a cluster failed to
1087             // match with another cluster in either time or position.
1088             // Check whether a cluster at the same position was found.
1089             // If it was not, then the cluster fails to verify by reason
1090             // of position.
1091             if(!matchedPosition) {
1092                 pairList.add(new ClusterMatchedPair(simCluster, null, ClusterMatchedPair.CLUSTER_STATE_FAIL_POSITION));
1093             }
1094             
1095             // Otherwise, the cluster had a potential match found at
1096             // the same position, but the time-stamps did not align.
1097             else {
1098                 pairList.add(new ClusterMatchedPair(simCluster, null, ClusterMatchedPair.CLUSTER_STATE_FAIL_TIME));
1099             }
1100         } // End sim loop.
1101         
1102         // Return the list of clusters, their matches, and their
1103         // verification states.
1104         return pairList;
1105     }

1107     /**
1108      * Converts a <code>CalorimeterHit</code> to a basic descriptive
1109      * string.
1110      * @param hit - The hit to be converted.
1111      * @return Returns the hit object as a <code>String</code>.
1112      */
1113     private static final String toString(CalorimeterHit hit) {
1114         if(hit == null) { return "Invalid hit."; }
1115         else {
1116             return String.format("Hit at (%3d, %3d) with %7.3f GeV at time %3.0f ns",
1117                     hit.getIdentifierFieldValue("ix"), hit.getIdentifierFieldValue("iy"),
1118                     hit.getCorrectedEnergy(), hit.getTime());
1119         }
1120     }
