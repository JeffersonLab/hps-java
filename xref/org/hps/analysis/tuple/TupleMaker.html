172             try {

174                         .getCachedConditions(BeamEnergy.BeamEnergyCollection.class, "beam_energies").getCachedData();
175                 ebeam = beamEnergyCollection.get(0).getBeamEnergy();
176             } catch (Exception e) {
177             }
178         }
179         setupVariables();
180         if (tupleFile != null) {
181             try {
182                 tupleWriter = new PrintWriter(tupleFile);

184                 tupleWriter = null;
185             }
186             tupleWriter.println(StringUtils.join(tupleVariables, ":"));
187         }
188         sensors = detector.getSubdetector(SUBDETECTOR_NAME).getDetectorElement().findDescendants(HpsSiSensor.class);
189         trackerSubdet = detector.getSubdetector(SUBDETECTOR_NAME);

192     protected boolean setupCollections(EventHeader event) {
193         String unconstrainedV0CandidatesColName = "Unconstrained" + CandidatesColName;
194         String unconstrainedV0VerticesColName = "Unconstrained" + VerticesColName;
195         String beamspotConstrainedV0CandidatesColName = "BeamspotConstrained" + CandidatesColName;
196         String targetConstrainedV0CandidatesColName = "TargetConstrained" + CandidatesColName;
197         
198         /*  make sure everything is there */
199         if (!event.hasCollection(ReconstructedParticle.class, unconstrainedV0CandidatesColName)) {
200             return false;
201         }
202         if (!event.hasCollection(ReconstructedParticle.class, beamspotConstrainedV0CandidatesColName)) {
203             beamspotConstrainedV0CandidatesColName = null;
204         }
205         if (!event.hasCollection(ReconstructedParticle.class, targetConstrainedV0CandidatesColName)) {
206             targetConstrainedV0CandidatesColName = null;
207         }
208         
209         unConstrainedV0List = event.get(ReconstructedParticle.class, unconstrainedV0CandidatesColName);
210 
211         if (unconstrainedV0VerticesColName != null) {
212             unConstrainedV0VerticeList = event.get(Vertex.class, unconstrainedV0VerticesColName);
213             cand2vert  = correlateCandidates(unConstrainedV0List, unConstrainedV0VerticeList);
214         }
215         if (beamspotConstrainedV0CandidatesColName != null) {
216             bsConstrainedV0List = event.get(ReconstructedParticle.class, beamspotConstrainedV0CandidatesColName);
217             unc2bsc = correlateCollections(unConstrainedV0List, bsConstrainedV0List);
218         }
219         if (targetConstrainedV0CandidatesColName != null) {
220             tarConstrainedV0List = event.get(ReconstructedParticle.class, targetConstrainedV0CandidatesColName);
221             unc2tar = correlateCollections(unConstrainedV0List, tarConstrainedV0List);
222         }
223         
224         triggerData = checkTrigger(event);
225         if (triggerData == null)
226             return false;
227         
228 
229         return true;
230     }

239     protected boolean matchTriggerType(TIData triggerData) {
240         if (triggerType.contentEquals("") || triggerType.contentEquals("all")) {
241             return true;
242         }
243         if (triggerData.isSingle0Trigger() && triggerType.contentEquals("singles0")) {
244             return true;
245         }
246         if (triggerData.isSingle1Trigger() && triggerType.contentEquals("singles1")) {
247             return true;
248         }
249         if (triggerData.isPair0Trigger() && triggerType.contentEquals("pairs0")) {
250             return true;
251         }
252         if (triggerData.isPair1Trigger() && triggerType.contentEquals("pairs1")) {
253             return true;
254         }
255         return false;
256     }

258     protected void writeTuple() {
259         for (String variable : tupleVariables) {
260             Double value = tupleMap.get(variable);
261             if (value == null || Double.isNaN(value)) {
262                 value = -9999.0;
263             }
264             if (variable.endsWith("/I") || variable.endsWith("/B")) {
265                 tupleWriter.format("%d\t", Math.round(value));
266             } else {
267                 tupleWriter.format("%g\t", value);
268             }
269         }
270         tupleWriter.println();
271         // tupleMap.clear();
272     }

282     public TIData checkTrigger(EventHeader event) {
283         TIData triggerData = null;
284         if (event.hasCollection(GenericObject.class, "TriggerBank")) {
285             for (GenericObject data : event.get(GenericObject.class, "TriggerBank")) {
286                 if (AbstractIntData.getTag(data) == TIData.BANK_TAG) {
287                     triggerData = new TIData(data);
288                 }
289             }
290         }
291         //check to see if this event is from the correct trigger (or "all");
292         if (triggerData != null && !matchTriggerType(triggerData)) {
293             return null;
294         }
295         return triggerData;
296     }

298     protected boolean fillBasicTuple(EventHeader event, TIData triggerData, ReconstructedParticle uncV0, boolean isMoller) {
299         boolean isOK = true;
300 
301         if (isGBL != TrackType.isGBL(uncV0.getType())) {
302             return false;
303         }
304         int runNumber = event.getRunNumber();
305         
306         fillEventVariables(event, triggerData);
307 
308         double minPositiveIso = 0;
309         double minNegativeIso = 0;
310         if (isMoller) {
311             ReconstructedParticle top = uncV0.getParticles().get(ReconParticleDriver.MOLLER_TOP);
312             ReconstructedParticle bot = uncV0.getParticles().get(ReconParticleDriver.MOLLER_BOT);
313             if (top.getCharge() != -1 || bot.getCharge() != -1) {
314                 throw new RuntimeException("incorrect charge on v0 daughters");
315             }
316             fillParticleVariables(event, top, "top");
317             fillParticleVariables(event, bot, "bot");
318             minPositiveIso = Math.min(tupleMap.get("topMinPositiveIso/D"), tupleMap.get("botMinPositiveIso/D"));
319             minNegativeIso = Math.min(Math.abs(tupleMap.get("topMinNegativeIso/D")), Math.abs(tupleMap.get("botMinNegativeIso/D")));
320             
321         }
322         else {
323             ReconstructedParticle electron = uncV0.getParticles().get(ReconParticleDriver.ELECTRON);
324             ReconstructedParticle positron = uncV0.getParticles().get(ReconParticleDriver.POSITRON);
325             if (electron.getCharge() != -1 || positron.getCharge() != 1) {
326                 throw new RuntimeException("incorrect charge on v0 daughters");
327             }
328             TrackState eleTSTweaked = fillParticleVariables(event, electron, "ele");
329             TrackState posTSTweaked = fillParticleVariables(event, positron, "pos");
330             minPositiveIso = Math.min(tupleMap.get("eleMinPositiveIso/D"), tupleMap.get("posMinPositiveIso/D"));
331             minNegativeIso = Math.min(Math.abs(tupleMap.get("eleMinNegativeIso/D")), Math.abs(tupleMap.get("posMinNegativeIso/D")));
332         }
333         double minIso = Math.min(minPositiveIso, minNegativeIso);
334         tupleMap.put("minPositiveIso/D", minPositiveIso);
335         tupleMap.put("minNegativeIso/D", minNegativeIso);
336         tupleMap.put("minIso/D", minIso);
337 
338         fillVertexVariables("unc", uncV0, false, runNumber);
339         if (unc2bsc != null) {
340             ReconstructedParticle temp = unc2bsc.get(uncV0);
341             if (temp == null)
342                 isOK = false;
343             else
344                 fillVertexVariables("bsc", temp, false, runNumber);
345         }
346         if (unc2bsc != null) {
347             ReconstructedParticle temp = unc2tar.get(uncV0);
348             if (temp == null)
349                 isOK = false;
350             fillVertexVariables("tar", temp, false, runNumber);
351         }
352 
353         return isOK;
354

542         if (event.hasCollection(GenericObject.class, "RFHits")) {
543             List<GenericObject> rfTimes = event.get(GenericObject.class, "RFHits");
544             if (rfTimes.size() > 0) {
545                 tupleMap.put("rfT1/D", rfTimes.get(0).getDoubleVal(0));
546                 tupleMap.put("rfT2/D", rfTimes.get(0).getDoubleVal(1));
547             }

550         List<LCRelation> fittedHits = event.get(LCRelation.class, "SVTFittedRawTrackerHits");
551         tupleMap.put("nSVTHits/I", (double) fittedHits.size());
552 
553         int[] nLhits = {0,0,0,0,0,0};
554         int[] nLbhits = {0,0,0,0,0,0};
555         List<RawTrackerHit> rawHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
556         for (RawTrackerHit rHit : rawHits) {
557             HpsSiSensor sensor = (HpsSiSensor) rHit.getDetectorElement();
558             int layer = sensor.getLayerNumber();
559             int i = ((sensor.getLayerNumber() + 1) / 2) - 1;
560             
561             if (layer % 2 == 0) {
562                 //bottom hit
563                 nLbhits[i]++;
564             }
565             else {
566                 nLhits[i]++;
567             }
568         }
569 
570         for (int k = 1; k<7; k++) {
571             String putMe = String.format("nSVTHitsL%d/I", k);
572             tupleMap.put(putMe, (double) nLhits[k-1]);
573             putMe = String.format("nSVTHitsL%db/I", k);
574             tupleMap.put(putMe, (double) nLbhits[k-1]);
575         }
576         
577         double topL1HitX = 9999;
578         double topL1HitY = 9999;
579         double botL1HitX = 9999;
580         double botL1HitY = -9999;
581 
582         // Get the collection of 3D hits from the event. This collection
583         // contains all 3D hits in the event and not just those associated
584         // with a track.
585         List<TrackerHit> hits = event.get(TrackerHit.class, "RotatedHelicalTrackHits");
586         tupleMap.put("n3DSVTHits/I", (double) hits.size());
587 
588         // Loop over the collection of 3D hits in the event and map them to
589         // their corresponding layer.
590         for (TrackerHit hit : hits) {
591             // Retrieve the sensor associated with one of the hits. This will
592             // be used to retrieve the layer number
593             HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) hit.getRawHits().get(0)).getDetectorElement();
594 
595             // Retrieve the layer number by using the sensor
596             int layer = (sensor.getLayerNumber() + 1) / 2;
597 
598             // If hit isn't in layer one, skip it.
599             // You can also create another list which contains just layer 1 hits ...
600             if (layer != 1)
601                 continue;
602 
603             if (sensor.isTopLayer() && topL1HitY > hit.getPosition()[2]) {
604                 topL1HitY = hit.getPosition()[2];
605                 topL1HitX = hit.getPosition()[1];
606 
607             }
608             if (sensor.isBottomLayer() && botL1HitY < hit.getPosition()[2]) {
609                 botL1HitY = hit.getPosition()[2];
610                 botL1HitX = hit.getPosition()[1];
611 
612             }
613         }
614         tupleMap.put("topL1HitX/D", topL1HitX);
615         tupleMap.put("topL1HitY/D", topL1HitY);
616         tupleMap.put("botL1HitX/D", botL1HitX);
617         tupleMap.put("botL1HitY/D", botL1HitY);
618     }
