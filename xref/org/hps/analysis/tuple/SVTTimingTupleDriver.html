233     @Override
234     public void process(EventHeader event) {
235 
236         if (!event.hasCollection(LCRelation.class, fittedTrackerHitCollectionName)) {
237             System.out.println("No fitted hits???");
238             return;
239         }
240 
241         List<LCRelation> fittedTrackerHits = event.get(LCRelation.class, fittedTrackerHitCollectionName);
242         RelationalTable rthtofit = new BaseRelationalTable(RelationalTable.Mode.ONE_TO_ONE, RelationalTable.Weighting.UNWEIGHTED);
243         for (LCRelation hit : fittedTrackerHits)
244             rthtofit.add(FittedRawTrackerHit.getRawTrackerHit(hit), FittedRawTrackerHit.getShapeFitParameters(hit));
245 
246         TIData triggerData = null;
247         if (event.hasCollection(GenericObject.class, "TriggerBank"))
248             for (GenericObject data : event.get(GenericObject.class, "TriggerBank"))
249                 if (AbstractIntData.getTag(data) == TIData.BANK_TAG)
250                     triggerData = new TIData(data);
251 
252         if (!event.hasCollection(TrackData.class, TrackData.TRACK_DATA_COLLECTION)
253                 || !event.hasCollection(LCRelation.class, TrackData.TRACK_DATA_RELATION_COLLECTION)) {
254             System.out.println("No TrackData");
255             return;
256         }
257 
258 //
259 //        List<TrackData> trkDataList = (List<TrackData>) event.get(TrackData.class, TrackData.TRACK_DATA_COLLECTION);
260 //        List<LCRelation> trkDataRelationList = (List<LCRelation>) event.get(LCRelation.class, TrackData.TRACK_DATA_RELATION_COLLECTION);
261 ////
262 //        if (!event.hasCollection(ReconstructedParticle.class, finalStateParticlesColName)) {
263 //            if (debug)
264 //                LOGGER.info(finalStateParticlesColName + " collection not found???");
265 //            return;
266 //        }
267 //
268 //        if (!event.hasCollection(MCParticle.class, mcParticlesColName)) {
269 //            if (debug)
270 //                LOGGER.info(mcParticlesColName + " collection not found???");
271 //            return;
272 //        }
273         //check to see if this event is from the correct trigger (or "all");
274         if (triggerData != null && !matchTriggerType(triggerData))
275             return;
276 
277         nRecoEvents++;
278         if (debug)
279             LOGGER.info("##########  Start of SVTTimngTuple   ##############");
280         List<ReconstructedParticle> finalStateParticles = event.get(ReconstructedParticle.class, finalStateParticlesColName);
281         if (debug)
282             LOGGER.info("This events has " + finalStateParticles.size() + " final state particles");
283         List<ReconstructedParticle> unconstrainedV0List = event.get(ReconstructedParticle.class, unconstrainedV0CandidatesColName);
284         if (debug)
285             LOGGER.info("This events has " + unconstrainedV0List.size() + " unconstrained V0s");
286         List<ReconstructedParticle> BSCV0List = event.get(ReconstructedParticle.class, beamConV0CandidatesColName);
287         if (debug)
288             LOGGER.info("This events has " + BSCV0List.size() + " BSC V0s");
289 
290         if (event.hasCollection(MCParticle.class, mcParticlesColName)) {
291             List<MCParticle> MCParticleList = event.get(MCParticle.class, mcParticlesColName);
292 
293             //find electron and positron MCParticles
294             Hep3Vector vertexPositionMC = null;
295             double apMassMC = -9999;
296             MCParticle ap = null;
297             MCParticle eleMC = null;
298             MCParticle posMC = null;
299 
300             for (MCParticle mcp : MCParticleList)
301                 if (mcp.getPDGID() == 622 && mcp.getDaughters().size() == 2) {
302                     ap = mcp;
303                     vertexPositionMC = mcp.getEndPoint();
304                     apMassMC = mcp.getMass();
305                     List<MCParticle> daugList = mcp.getDaughters();
306                     for (MCParticle daug : daugList)
307                         if (daug.getPDGID() == 11)
308                             eleMC = daug;
309                         else if (daug.getPDGID() == -11)
310                             posMC = daug;
311                 }
312         }
313 //        if (eleMC == null || posMC == null) {
314 //            if (debug)
315 //                LOGGER.info("Couldn't find the MC e+e- from A'?????  Quitting.");
316 //            return;
317 //        }
318         double bestMom = 99999;
319         int layer;
320         String layerLabel;
321         String typeLabel;
322         String axStLabel;
323         String tString;
324         System.out.println("Event has " + finalStateParticles.size() + " final state particles");
325         for (ReconstructedParticle fsp : finalStateParticles) {
326             tupleMap.clear();
327             //if (!TrackType.isGBL(fsp.getType())) // we only care about GBL vertices
328 
329             if (event.hasCollection(GenericObject.class, "TriggerBank")) {
330                 List<GenericObject> triggerList = event.get(GenericObject.class, "TriggerBank");
331                 for (GenericObject data : triggerList)
332                     if (AbstractIntData.getTag(data) == SSPData.BANK_TAG) {
333                         SSPData sspData = new SSPData(data);
334                         double trigTime = sspData.getTime();
335                         tupleMap.put("TrigTime/D", trigTime);
336                     }
337             }
338 
339             if (event.hasCollection(GenericObject.class, "RFHits")) {
340                 List<GenericObject> rfTimes = event.get(GenericObject.class, "RFHits");
341                 if (rfTimes.size() > 0) {
342                     tupleMap.put("rfT1/D", rfTimes.get(0).getDoubleVal(0));
343                     tupleMap.put("rfT2/D", rfTimes.get(0).getDoubleVal(1));
344                 }
345             }
346             //does the track have a matched cluster?              
347             System.out.println("# clusters = " + fsp.getClusters().size());
348             System.out.println("# tracks = " + fsp.getTracks().size());
349             if (fsp.getTracks().size() != 1)
350                 continue;
351 
352             if (!TrackType.isGBL(fsp.getTracks().get(0).getType())) // we only care about GBL tracks
353                 continue;
354             System.out.println("Found GBL Track");
355 //            System.out.println("Got a cluster and track");
356             Track trk = fsp.getTracks().get(0);
357             if (trk.getTrackerHits().size() != 6)
358                 continue;
359             System.out.println("Got 6-hit track");
360             if (fsp.getClusters().size() == 1) {
361                 Cluster clu = fsp.getClusters().get(0);
362                 double cltime = ClusterUtilities.findSeedHit(clu).getTime();
363                 tupleMap.put("ECalClTime", cltime);
364             }
365             //include track stuff
366             typeLabel = "Tr";
367             //   TrackData trkData = (TrackData) TrackData.getTrackData(event, trk);
368             double trkTime = TrackData.getTrackTime(TrackData.getTrackData(event, trk));
369             tString = typeLabel + "Time/D";
370             tupleMap.put(tString, trkTime);
371             double trkMom = fsp.getMomentum().z();
372             tString = typeLabel + "Pz/D";
373             tupleMap.put(tString, trkMom);
374             trkMom = fsp.getMomentum().x();
375             tString = typeLabel + "Px/D";
376             tupleMap.put(tString, trkMom);
377             trkMom = fsp.getMomentum().y();
378             tString = typeLabel + "Py/D";
379             tupleMap.put(tString, trkMom);
380             double trkCharge = fsp.getCharge();
381             tString = typeLabel + "Pz/I";
382             tupleMap.put(tString, trkCharge);
383 
384             List<TrackerHit> trHits = trk.getTrackerHits();
385             for (TrackerHit trHit : trHits) {
386 
387                 HelicalTrackCross hth = (HelicalTrackCross) trHit; //need to run full recon for this!!!
388                 //stuff with 3d hits
389                 typeLabel = "Cl3d";
390                 layer = moduleNumber(hth.Layer());
391                 layerLabel = layerMap.get(layer);
392                 double hthTime = hth.getTime();
393                 tString = typeLabel + layerLabel + "Time/D";
394                 tupleMap.put(tString, hthTime);
395                 //                
396                 List<HelicalTrackStrip> htsList = hth.getStrips();
397                 for (HelicalTrackStrip hts : htsList) {
398                     //do strip cluster stuff
399                     typeLabel = "Cl";
400                     layer = moduleNumber(hts.layer());
401                     layerLabel = layerMap.get(layer);
402 
403                     List<RawTrackerHit> rthList = hts.rawhits();
404                     RawTrackerHit rthDummy = (RawTrackerHit) hts.rawhits().get(0);
405                     HpsSiSensor sensor = ((HpsSiSensor) rthDummy.getDetectorElement());
406 
407                     axStLabel = "s";
408                     if (sensor.isAxial())
409                         axStLabel = "a";
410                     tString = typeLabel + layerLabel + axStLabel + "Time/D";
411                     double htsTime = hts.time();
412                     tupleMap.put(tString, htsTime);
413                     double htsNHit = hts.rawhits().size();
414                     tString = typeLabel + layerLabel + axStLabel + "NHit/D";
415                     tupleMap.put(tString, htsNHit);
416                     //  
417                     RawTrackerHit rthSeed = null;
418                     GenericObject parsSeed = null;
419                     double peak = -99999;
420                     for (RawTrackerHit rth : rthList) {
421                         GenericObject pars = (GenericObject) rthtofit.to(rth);
422                         // get the highest amplitude hit
423 
424                         double amp = ShapeFitParameters.getAmp(pars);
425                         if (amp > peak) {
426                             peak = amp;
427                             parsSeed = pars;
428                             rthSeed = rth;
429                         }
430                     }
431                     //found the seed; now do strip hit stuff
432                     typeLabel = "Str";
433                     double t0 = ShapeFitParameters.getT0(parsSeed);
434                     double amp = ShapeFitParameters.getAmp(parsSeed);
435                     double chiProb = ShapeFitParameters.getChiProb(parsSeed);
436                     tString = typeLabel + layerLabel + axStLabel + "Time/D";
437                     tupleMap.put(tString, t0);
438                     tString = typeLabel + layerLabel + axStLabel + "Amp/D";
439                     tupleMap.put(tString, amp);
440                     tString = typeLabel + layerLabel + axStLabel + "ChiSq/D";
441                     tupleMap.put(tString, chiProb);
442                 }
443             }
444 
445             if (event.hasCollection(TriggerTime.class, "TriggerTime")) {
446                 if (debug)
447                     System.out.println("Getting TriggerTime Object");
448                 List<TriggerTime> jitterList = event.get(TriggerTime.class, "TriggerTime");
449                 if (debug)
450                     System.out.println("TriggerTime List Size = " + jitterList.size());
451                 TriggerTime jitterObject = jitterList.get(0);
452                 double jitter = jitterObject.getDoubleVal();
453                 if (debug)
454                     System.out.println("RF time jitter " + jitter);
455                 tupleMap.put("RFJitter/D", jitter);
456 
457             } else
458                 System.out.println("Requested RF Time correction but TriggerTime Collection doesn't exist!!!");
459             if (tupleWriter != null) {
460                 System.out.println("writing tuple");
461                 writeTuple();
462             }
463         }
464
