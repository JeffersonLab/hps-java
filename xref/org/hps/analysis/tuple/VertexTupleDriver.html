246     @Override
247     public void process(EventHeader event) {
248 
249         TIData triggerData = null;
250         if (event.hasCollection(GenericObject.class, "TriggerBank"))
251             for (GenericObject data : event.get(GenericObject.class, "TriggerBank"))
252                 if (AbstractIntData.getTag(data) == TIData.BANK_TAG)
253                     triggerData = new TIData(data);
254 
255         if (!event.hasCollection(ReconstructedParticle.class, finalStateParticlesColName)) {
256             if (debug)
257                 LOGGER.info(finalStateParticlesColName + " collection not found???");
258             return;
259         }
260 
261         if (!event.hasCollection(MCParticle.class, mcParticlesColName)) {
262             if (debug)
263                 LOGGER.info(mcParticlesColName + " collection not found???");
264             return;
265         }
266 
267         //check to see if this event is from the correct trigger (or "all");
268         if (triggerData != null && !matchTriggerType(triggerData))
269             return;
270 
271         nRecoEvents++;
272         if (debug)
273             LOGGER.info("##########  Start of VertexDebugger   ##############");
274         List<ReconstructedParticle> finalStateParticles = event.get(ReconstructedParticle.class, finalStateParticlesColName);
275         if (debug)
276             LOGGER.info("This events has " + finalStateParticles.size() + " final state particles");
277         List<ReconstructedParticle> unconstrainedV0List = event.get(ReconstructedParticle.class, unconstrainedV0CandidatesColName);
278         if (debug)
279             LOGGER.info("This events has " + unconstrainedV0List.size() + " unconstrained V0s");
280         List<ReconstructedParticle> BSCV0List = event.get(ReconstructedParticle.class, beamConV0CandidatesColName);
281         if (debug)
282             LOGGER.info("This events has " + BSCV0List.size() + " BSC V0s");
283 
284         List<MCParticle> MCParticleList = event.get(MCParticle.class, mcParticlesColName);
285 
286         tupleMap.clear();
287         //find electron and positron MCParticles
288         Hep3Vector vertexPositionMC = null;
289         double apMassMC = -9999;
290         MCParticle ap = null;
291         MCParticle eleMC = null;
292         MCParticle posMC = null;
293 
294         for (MCParticle mcp : MCParticleList)
295             if (mcp.getPDGID() == 622 && mcp.getDaughters().size() == 2) {
296                 ap = mcp;
297                 vertexPositionMC = mcp.getEndPoint();
298                 apMassMC = mcp.getMass();
299                 List<MCParticle> daugList = mcp.getDaughters();
300                 for (MCParticle daug : daugList)
301                     if (daug.getPDGID() == 11)
302                         eleMC = daug;
303                     else if (daug.getPDGID() == -11)
304                         posMC = daug;
305             }
306         if (eleMC == null || posMC == null) {
307             if (debug)
308                 LOGGER.info("Couldn't find the MC e+e- from A'?????  Quitting.");
309             return;
310         }
311 
312         if (debug) {
313             LOGGER.info("Found A' MC with vertex at :" + vertexPositionMC.x() + "; " + vertexPositionMC.y() + "; " + vertexPositionMC.z());
314             LOGGER.info("Found A' MC with mass = " + apMassMC);
315             LOGGER.info("Found A' MC electron momentum = " + eleMC.getMomentum().x() + "; " + eleMC.getMomentum().y() + "; " + eleMC.getMomentum().z());
316             LOGGER.info("Found A' MC positron momentum = " + posMC.getMomentum().x() + "; " + posMC.getMomentum().y() + "; " + posMC.getMomentum().z());
317         }
318 
319         Hep3Vector pEleMC = eleMC.getMomentum();
320         Hep3Vector pPosMC = posMC.getMomentum();
321         Hep3Vector pApMC = ap.getMomentum();
322         Hep3Vector apStart = ap.getOrigin();
323         double[] apTarXY = getTargetXY(pApMC, vertexPositionMC);
324         tupleMap.put("apMass/D", apMassMC);
325         tupleMap.put("vtxXMC/D", vertexPositionMC.x());
326         tupleMap.put("vtxYMC/D", vertexPositionMC.y());
327         tupleMap.put("vtxZMC/D", vertexPositionMC.z());
328         tupleMap.put("pEleXMC/D", pEleMC.x());
329         tupleMap.put("pEleYMC/D", pEleMC.y());
330         tupleMap.put("pEleZMC/D", pEleMC.z());
331         tupleMap.put("pPosXMC/D", pPosMC.x());
332         tupleMap.put("pPosYMC/D", pPosMC.y());
333         tupleMap.put("pPosZMC/D", pPosMC.z());
334         tupleMap.put("pApXMC/D", pApMC.x());
335         tupleMap.put("pApYMC/D", pApMC.y());
336         tupleMap.put("pApZMC/D", pApMC.z());
337         tupleMap.put("apTarX/D", apTarXY[0]);
338         tupleMap.put("apTarY/D", apTarXY[1]);
339         tupleMap.put("apOrigX/D", apStart.x());  //these AP origin variables should be at Z~0 (the target) and XY with the width of beamspot
340         tupleMap.put("apOrigY/D", apStart.y());
341         tupleMap.put("apOrigZ/D", apStart.z());
342 
343 
344         fillEventVariables(event, triggerData);
345         fillMCTridentVariables(event);
346         int index=0;
347         boolean foundEventMatch=false;
348         for (ReconstructedParticle v0 : unconstrainedV0List) {
349             if(foundEventMatch)
350                 break;
351             
352             List<ReconstructedParticle> daug=v0.getParticles();
353           
354             if (daug.size() != 2) {
355                 System.out.println("Not 2 daughters in this V0???");
356                 continue;
357             }
358             Track electron = daug.get(0).getTracks().get(0);
359             Track positron = daug.get(1).getTracks().get(0);
360             if (electron.getCharge() > 1) {//wrong charge...swap them
361                 Track tmp = electron;
362                 electron = positron;
363                 positron = tmp;
364             }
365             double[] eleP = electron.getTrackStates().get(0).getMomentum();
366             double[] posP = positron.getTrackStates().get(0).getMomentum();
367             double elePMag = pMag(eleP);
368 
369             //  try to match reco'ed electron and positron to the v0
370             double matchPos = 0.0;
371             double matchEle = 0.0;
372             double eleMCPmag=eleMC.getMomentum().magnitude();
373             System.out.println("Positron Track P() = ("+posP[0]+","+posP[1]+","+posP[2]+")");
374             System.out.println("Positron MC    P() = ("+posMC.getPX()+","+posMC.getPY()+","+posMC.getPZ()+")");
375             System.out.println("Electron Track P() = ("+eleP[0]+","+eleP[1]+","+eleP[2]+")");
376             System.out.println("Electron MC    P() = ("+eleMC.getPX()+","+eleMC.getPY()+","+eleMC.getPZ()+")");
377             if (posP[2] * posMC.getPY() > 0) //dumb matching..make sure in same half
378                 matchPos = 1.0;
379             if (eleP[2] * eleMC.getPY() > 0 && Math.abs(eleMC.getMomentum().magnitude() - elePMag) < matchEleP) //dumb matching..make sure in same half and has reasonable momentum
380                 matchEle = 1.0;   
381             foundEventMatch=(matchPos==1.0)&&(matchEle==1.0);
382             if(foundEventMatch)
383                 System.out.println("Found an e+e- pair match"+elePMag+"; "+eleMCPmag);
384             if (!foundEventMatch) {
385                 System.out.println("This e+e- pair didn't match");
386                 if (index+1<unconstrainedV0List.size()) //if this is not the last V0 in the list, go to next v0 and try to find e+e- match; otherwise, go ahead and save info for this v0
387                     continue;   
388             }
389                       
390             tupleMap.put("matchEle/D", matchEle);
391             tupleMap.put("matchPos/D", matchPos);
392             BilliorVertex vtxUnc = (BilliorVertex) v0.getStartVertex();
393             Hep3Vector vtxPosUnc = vtxUnc.getPosition();
394             Hep3Vector pEleUnc = vtxUnc.getFittedMomentum(0);
395             Hep3Vector pPosUnc = vtxUnc.getFittedMomentum(1);
396             double mUncUnc = vtxUnc.getInvMass();
397             double chisqUnc = vtxUnc.getChi2();
398             Hep3Vector v0Mom = vtxUnc.getV0Momentum();
399             Hep3Vector v0MomError = vtxUnc.getV0MomentumError();
400             double[] v0AtTarget = vtxUnc.getV0TargetXY();
401             double[] v0AtTargetError = vtxUnc.getV0TargetXYError();
402             
403             tupleMap.put("mUnc/D", mUncUnc);
404             tupleMap.put("vtxXUnc/D", vtxPosUnc.x());
405             tupleMap.put("vtxYUnc/D", vtxPosUnc.y());
406             tupleMap.put("vtxZUnc/D", vtxPosUnc.z());
407             tupleMap.put("pEleXUnc/D", pEleUnc.x());
408             tupleMap.put("pEleYUnc/D", pEleUnc.y());
409             tupleMap.put("pEleZUnc/D", pEleUnc.z());
410             tupleMap.put("pPosXUnc/D", pPosUnc.x());
411             tupleMap.put("pPosYUnc/D", pPosUnc.y());
412             tupleMap.put("pPosZUnc/D", pPosUnc.z());
413             tupleMap.put("chisqUnc/D", chisqUnc);
414             tupleMap.put("probUnc/D",vtxUnc.getProbability());
415 
416             tupleMap.put("mErrUnc/D", vtxUnc.getInvMassError());
417             tupleMap.put("vtxXErrUnc/D", vtxUnc.getPositionError().x());
418             tupleMap.put("vtxYErrUnc/D", vtxUnc.getPositionError().y());
419             tupleMap.put("vtxZErrUnc/D", vtxUnc.getPositionError().z());
420             tupleMap.put("pEleXErrUnc/D", vtxUnc.getFittedMomentumError(0).x());
421             tupleMap.put("pEleYErrUnc/D", vtxUnc.getFittedMomentumError(0).y());
422             tupleMap.put("pEleZErrUnc/D", vtxUnc.getFittedMomentumError(0).z());
423             tupleMap.put("pPosXErrUnc/D", vtxUnc.getFittedMomentumError(1).x());
424             tupleMap.put("pPosYErrUnc/D", vtxUnc.getFittedMomentumError(1).y());
425             tupleMap.put("pPosZErrUnc/D", vtxUnc.getFittedMomentumError(1).z());
426             
427             tupleMap.put("v0MomXErrUnc/D", v0MomError.x());
428             tupleMap.put("v0MomYErrUnc/D", v0MomError.y());
429             tupleMap.put("v0MomZErrUnc/D", v0MomError.z());
430             tupleMap.put("v0XYTarXErrUnc/D", v0AtTargetError[0]);
431             tupleMap.put("v0XYTarYErrUnc/D", v0AtTargetError[1]);
432             tupleMap.put("v0MomXUnc/D", v0Mom.x());
433             tupleMap.put("v0MomYUnc/D", v0Mom.y());
434             tupleMap.put("v0MomZUnc/D", v0Mom.z());
435             //System.out.println("VertexDebugTuple::v0 projection X = " + v0AtTarget[0] + "; Y = " + v0AtTarget[1]);
436             tupleMap.put("v0XYTarXUnc/D", v0AtTarget[0]);
437             tupleMap.put("v0XYTarYUnc/D", v0AtTarget[1]);
438             
439             ReconstructedParticle  bscv0=BSCV0List.get(index);

498     private BilliorVertex fitVertex(VertexTupleDriver.Constraint constraint, BilliorTrack electron, BilliorTrack positron, Hep3Vector v0) {
499         // Create a vertex fitter from the magnetic field.
500 
501         BilliorVertexer vtxFitter = new BilliorVertexer(B_FIELD);
502         // TODO: The beam size should come from the conditions database.
503         vtxFitter.setBeamSize(beamSize);
504         vtxFitter.setDebug(false);
505 
506 //  mg: 5/11/2018  use ReferencePosition now instead of beam position...
507 //        if (v0 != null)
508 //            vtxFitter.setBeamPosition(v0.v());
509         if (v0 != null)
510             vtxFitter.setReferencePosition(v0.v());
511 
512         // Perform the vertexing based on the specified constraint.
513         switch (constraint) {
514             case UNCONSTRAINED:
515                 vtxFitter.doBeamSpotConstraint(false);
516                 break;
517             case BS_CONSTRAINED:
518                 vtxFitter.doBeamSpotConstraint(true);
519                 break;
520             case TARGET_CONSTRAINED:
521                 vtxFitter.doTargetConstraint(true);
522                 break;
523         }
524 
525         // Add the electron and positron tracks to a track list for
526         // the vertex fitter.
527         List<BilliorTrack> billiorTracks = new ArrayList<BilliorTrack>();
528         billiorTracks.add(electron);
529         billiorTracks.add(positron);
530 
531         // Find and return a vertex based on the tracks.
532         BilliorVertex vtx = vtxFitter.fitVertex(billiorTracks);
533 
534         return vtx;
535     }

593     // invMass  is probably defined in the code a hundred times...here it is again. 
594     private double invMass(Hep3Vector p1, Hep3Vector p2) {
595         double me2 = 0.000511 * 0.000511;
596         double esum = sqrt(p1.magnitudeSquared() + me2) + sqrt(p2.magnitudeSquared() + me2);
597         double pxsum = p1.x() + p2.x();
598         double pysum = p1.y() + p2.y();
599         double pzsum = p1.z() + p2.z();
600 
601         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
602         double evtmass = esum * esum - psum * psum;
603 
604         if (evtmass > 0)
605             return Math.sqrt(evtmass);
606         else
607             return -99;
608     }

610     private double getCosOpenX(Hep3Vector p1, Hep3Vector p2) {   // p vectors are in tracking frame...return open angle in X-detector!!!       
611         Hep3Vector pEleX = new BasicHep3Vector(p1.y(), 0.0, p1.x());
612         Hep3Vector pPosX = new BasicHep3Vector(p2.y(), 0.0, p2.x());
613         return Math.acos(VecOp.dot(pEleX, pPosX) / (pEleX.magnitude() * pPosX.magnitude()));
614     }
