263     protected void fillFullVertexTruth(EventHeader event, Track eleTrack, Track posTrack){
264         if (!event.hasCollection(SimTrackerHit.class, trackerHitsCollectionName)) {
265             return;
266         }
267         if (!event.hasCollection(SimTrackerHit.class, inactiveTrackerHitsCollectionName)) {
268             return;
269         }
270         if (!event.hasCollection(SimCalorimeterHit.class, ecalHitsCollectionName)) {
271             return;
272         }
273         if (!event.hasCollection(MCParticle.class, mcParticleCollectionName)) {
274             return;
275         }
276         MCFullDetectorTruth eleFullTruth = new MCFullDetectorTruth(event, eleTrack, bFieldMap, sensors, trackerSubdet);
277         MCFullDetectorTruth posFullTruth = new MCFullDetectorTruth(event, posTrack, bFieldMap, sensors, trackerSubdet);
278         
279         MCParticle ele = eleFullTruth.getMCParticle();
280         MCParticle pos = posFullTruth.getMCParticle();
281         
282         if(ele != null)
283             fillTruth("ele",eleFullTruth);
284         else
285             tupleMap.put("eleHasTruthMatch/I", (double) 0);
286         
287         if(pos != null)
288             fillTruth("pos",posFullTruth);
289         else
290             tupleMap.put("posHasTruthMatch/I", (double) 0);
291     }

293     public void fillTruth(String MCprefix,MCFullDetectorTruth partTruth){
294         fillMCParticleVariables(MCprefix, partTruth.getMCParticle());
295         tupleMap.put(MCprefix+"HasTruthMatch/I", (double) 1);
296         tupleMap.put(MCprefix+"NTruthHits/I", (double) partTruth.getNHits());
297         tupleMap.put(MCprefix+"NGoodTruthHits/I", (double) partTruth.getNBadHits());
298         tupleMap.put(MCprefix+"NBadTruthHits/I", (double) partTruth.getNBadHits());
299         tupleMap.put(MCprefix+"Purity/D", partTruth.getPurity());
300         String isTop = "t";
301         if(!partTruth.isTop())
302             isTop = "b";
303         for(int i = 0; i < nLay*2; i++){
304             int layer = i + 1;
305             tupleMap.put(MCprefix+"L"+Integer.toString(layer)+isTop+"NTruthParticles/I", (double) partTruth.getNumberOfMCParticles(layer));
306             if(partTruth.getHitList(layer) != null)
307                 tupleMap.put(MCprefix+"L"+Integer.toString(layer)+isTop+"IsGoodTruthHit/I", (double) ((partTruth.getHitList(layer)) ? 1 : 0));
308 
309             String prefix = MCprefix + "L" + Integer.toString(layer);
310             if(partTruth.isTop())
311                 prefix = prefix + "t";
312             else
313                 prefix = prefix + "b";
314             if(partTruth.getActiveHitPosition(layer) != null){
315                 tupleMap.put(prefix+"svthitX/D", partTruth.getActiveHitPosition(layer).x());
316                 tupleMap.put(prefix+"svthitY/D", partTruth.getActiveHitPosition(layer).y());
317                 tupleMap.put(prefix+"svthitZ/D", partTruth.getActiveHitPosition(layer).z());
318             }
319             if(partTruth.getActiveHitMomentum(layer) != null){
320                 tupleMap.put(prefix+"svthitPx/D", partTruth.getActiveHitMomentum(layer).x());
321                 tupleMap.put(prefix+"svthitPy/D", partTruth.getActiveHitMomentum(layer).y());
322                 tupleMap.put(prefix+"svthitPz/D", partTruth.getActiveHitMomentum(layer).z());
323             }
324             if(partTruth.getActiveHitScatter(layer) != null){
325                 tupleMap.put(prefix+"thetaX/D", partTruth.getActiveHitScatter(layer)[0]);
326                 tupleMap.put(prefix+"thetaY/D", partTruth.getActiveHitScatter(layer)[1]);
327             }
328             if(partTruth.getActiveHitResidual(layer) != null){
329                 tupleMap.put(prefix+"residualX/D", partTruth.getActiveHitResidual(layer)[0]);
330                 tupleMap.put(prefix+"residualY/D", partTruth.getActiveHitResidual(layer)[1]);
331             }
332             
333             if(partTruth.getInactiveHitPosition(layer) != null){
334                 tupleMap.put(prefix+"InsvthitX/D", partTruth.getInactiveHitPosition(layer).x());
335                 tupleMap.put(prefix+"InsvthitY/D", partTruth.getInactiveHitPosition(layer).y());
336                 tupleMap.put(prefix+"InsvthitZ/D", partTruth.getInactiveHitPosition(layer).z());
337             }
338             if(partTruth.getInactiveHitMomentum(layer) != null){
339                 tupleMap.put(prefix+"InsvthitPx/D", partTruth.getInactiveHitMomentum(layer).x());
340                 tupleMap.put(prefix+"InsvthitPy/D", partTruth.getInactiveHitMomentum(layer).y());
341                 tupleMap.put(prefix+"InsvthitPz/D", partTruth.getInactiveHitMomentum(layer).z());
342             }
343             if(partTruth.getInactiveHitScatter(layer) != null){
344                 tupleMap.put(prefix+"InthetaX/D", partTruth.getInactiveHitScatter(layer)[0]);
345                 tupleMap.put(prefix+"InthetaY/D", partTruth.getInactiveHitScatter(layer)[1]);
346             }
347             if(partTruth.getInactiveHitResidual(layer) != null){
348                 tupleMap.put(prefix+"InresidualX/D", partTruth.getInactiveHitResidual(layer)[0]);
349                 tupleMap.put(prefix+"InresidualY/D", partTruth.getInactiveHitResidual(layer)[1]);
350             }
351         }
352         
353         for (int i = 0; i < partTruth.getEcalNHits(); i++) {
354             String HitNum = Integer.toString(i);
355             String prefix = MCprefix + "Hit" + HitNum;
356             if(partTruth.getEcalHitIndex(i) != null){
357                 tupleMap.put(prefix+"ecalhitIx/I", (double) partTruth.getEcalHitIndex(i)[0]);
358                 tupleMap.put(prefix+"ecalhitIy/I", (double) partTruth.getEcalHitIndex(i)[1]);
359             }
360             if(partTruth.getEcalHitPosition(i) != null){
361                 tupleMap.put(prefix+"ecalhitX/D", partTruth.getEcalHitPosition(i).x());
362                 tupleMap.put(prefix+"ecalhitY/D", partTruth.getEcalHitPosition(i).y());
363                 tupleMap.put(prefix+"ecalhitZ/D", partTruth.getEcalHitPosition(i).z());
364             }
365             tupleMap.put(prefix+"ecalhitEnergy/D", partTruth.getEcalHitEnergy(i));
366         }
