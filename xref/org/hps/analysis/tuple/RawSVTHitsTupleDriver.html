205     @Override
206     public void process(EventHeader event) {
207 
208         if (!event.hasCollection(RawTrackerHit.class, rawTrackerHitCollectionName))
209 //            System.out.println("No raw tracker hits???");
210             return;
211 
212         List<RawTrackerHit> rawHits = event.get(RawTrackerHit.class, rawTrackerHitCollectionName);
213         if (rawHits == null)
214             throw new RuntimeException("Event is missing SVT hits collection!");
215         else
216             System.out.println("RawSVTHitsTupleDriver::process  number of raw hits = " + rawHits.size());
217         String axStLabel = "";
218         String tString = "";
219         String layerLabel = "";
220         Double hosl = 0.0;
221         Double axst = 0.0;
222         for (RawTrackerHit rth : rawHits) {
223             short[] samples = rth.getADCValues();
224             int layer = rth.getLayerNumber();
225             layerLabel = layerMap.get(layer);
226             HpsSiSensor sensor = ((HpsSiSensor) rth.getDetectorElement());
227             hosl = 0.0;
228             axst = 0.0;
229             if (sensor.getSide() == "POSITRON")
230                 hosl = 1.0;
231             if (sensor.isStereo())
232                 axst = 1.0;
233 //            if (sensor.())
234 //                axst=1.0;
235 //            System.out.println("RawSVTHitsTupleDriver::process  filling samples for l=" + layer + axStLabel);
236             tupleMap.put("L/D", layer / 1.0);
237             tupleMap.put("AxSt/D", axst);
238             for (int i = 0; i < 6; i++) {
239 //              tString = "L" + layerLabel + axStLabel + "S" + String.valueOf(i) + "/D";
240                 tString = "S" + String.valueOf(i + 1) + "/D";
241                 // System.out.println(tString+" "+(double) (samples[i] / 1.0));
242                 tupleMap.put(tString, (double) (samples[i] / 1.0));
243             }
244             if (tupleWriter != null) {
245 //                System.out.println("writing tuple");
246                 writeTuple();
247 
248             }
249         }     
250 
251     }
