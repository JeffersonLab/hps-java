302     @Override
303     public void process(EventHeader event) {
304 
305         TIData triggerData = null;
306         if (event.hasCollection(GenericObject.class, "TriggerBank"))
307             for (GenericObject data : event.get(GenericObject.class, "TriggerBank"))
308                 if (AbstractIntData.getTag(data) == TIData.BANK_TAG)
309                     triggerData = new TIData(data);
310 
311         if (!event.hasCollection(ReconstructedParticle.class, finalStateParticlesColName)) {
312             if (debug)
313                 LOGGER.info(finalStateParticlesColName + " collection not found???");
314             return;
315         }
316 
317         if (!event.hasCollection(MCParticle.class, mcParticlesColName))
318             if (debug)
319                 LOGGER.info(mcParticlesColName + " collection not found???"); //            return;
320 
321         //check to see if this event is from the correct trigger (or "all");
322         if (triggerData != null && !matchTriggerType(triggerData))
323             return;
324 
325         nRecoEvents++;
326         if (debug)
327             LOGGER.info("##########  Start of VertexDebugger   ##############");
328         List<ReconstructedParticle> finalStateParticles = event.get(ReconstructedParticle.class, finalStateParticlesColName);
329         if (debug)
330             LOGGER.info("This events has " + finalStateParticles.size() + " final state particles");
331         List<ReconstructedParticle> unconstrainedV0List = event.get(ReconstructedParticle.class, unconstrainedV0CandidatesColName);
332         if (debug)
333             LOGGER.info("This events has " + unconstrainedV0List.size() + " unconstrained V0s");
334         List<ReconstructedParticle> BSCV0List = event.get(ReconstructedParticle.class, beamConV0CandidatesColName);
335         if (debug)
336             LOGGER.info("This events has " + BSCV0List.size() + " BSC V0s");
337         List<ReconstructedParticle> TCV0List = event.get(ReconstructedParticle.class, targetV0ConCandidatesColName);
338         if (debug)
339             LOGGER.info("This events has " + TCV0List.size() + " Target Constrained V0s");
340 
341         //find electron and positron MCParticles
342         Hep3Vector vertexPositionMC = null;
343         double apMassMC = -9999;
344         MCParticle ap = null;
345         MCParticle eleMC = null;
346         MCParticle posMC = null;
347 
348         boolean isMC = event.hasCollection(MCParticle.class, mcParticlesColName);
349 
350         if (isMC) {
351             List<MCParticle> MCParticleList = event.get(MCParticle.class, mcParticlesColName);
352             for (MCParticle mcp : MCParticleList)
353                 if (mcp.getPDGID() == 622 && mcp.getDaughters().size() == 2) {
354                     ap = mcp;
355                     vertexPositionMC = mcp.getEndPoint();
356                     apMassMC = mcp.getMass();
357                     List<MCParticle> daugList = mcp.getDaughters();
358                     for (MCParticle daug : daugList)
359                         if (daug.getPDGID() == 11)
360                             eleMC = daug;
361                         else if (daug.getPDGID() == -11)
362                             posMC = daug;
363                 }
364             if (eleMC == null || posMC == null)
365                 if (debug)
366                     LOGGER.info("Couldn't find the MC e+e- from A'????? ");

1145     private double invMass(Hep3Vector p1, Hep3Vector p2) {
1146         double me2 = 0.000511 * 0.000511;
1147         double esum = sqrt(p1.magnitudeSquared() + me2) + sqrt(p2.magnitudeSquared() + me2);
1148         double pxsum = p1.x() + p2.x();
1149         double pysum = p1.y() + p2.y();
1150         double pzsum = p1.z() + p2.z();
1151 
1152         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
1153         double evtmass = esum * esum - psum * psum;
1154 
1155         if (evtmass > 0)
1156             return Math.sqrt(evtmass);
1157         else
1158             return -99;
1159     }
