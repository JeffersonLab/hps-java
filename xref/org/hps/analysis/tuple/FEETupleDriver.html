1   package org.hps.analysis.tuple;
2   
3   import java.util.List;
4   import org.hps.recon.tracking.TrackType;
5   import org.lcsim.event.EventHeader;
6   import org.lcsim.event.ReconstructedParticle;
7   
8   public class FEETupleDriver extends TupleMaker {
9   
10      private final String finalStateParticlesColName = "OtherElectrons";
11      private final double tupleTrkPCut = 0.7;
12  
13      @Override
14      protected void setupVariables() {
15          tupleVariables.clear();
16          addEventVariables();
17          addParticleVariables("fsp");
18      }
19  
20      @Override
21      public void process(EventHeader event) {
22          /*  make sure everything is there */
23          if (!event.hasCollection(ReconstructedParticle.class, finalStateParticlesColName)) {
24              return;
25          }
26          triggerData = checkTrigger(event);
27          if (triggerData == null)
28              return;
29  
30          List<ReconstructedParticle> fspList = event.get(ReconstructedParticle.class, finalStateParticlesColName);
31  
32          for (ReconstructedParticle fsp : fspList) {
33              if (isGBL != TrackType.isGBL(fsp.getType())) {
34                  continue;
35              }
36              tupleMap.clear();
37              fillEventVariables(event, triggerData);
38  
39              fillParticleVariables(event, fsp, "fsp", false, false, false);
40  
41              if (tupleWriter != null) {
42                  if (!cutTuple || (passesCuts())) {
43                      writeTuple();
44                  }
45              }
46          }
47      }
48  
49      @Override
50      boolean passesCuts() {
51          return tupleMap.get("fspP/D") > tupleTrkPCut * ebeam;
52      }
53  }
