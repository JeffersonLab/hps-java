32      protected void fillMCTridentVariables(EventHeader event) {
33          List<MCParticle> MCParticles = event.getMCParticles();
34  
35          MCParticle trident = null;
36  
37          MCParticle ele1 = null;// highest-energy electron daughter
38          MCParticle ele2 = null;// second-highest-energy electron daughter (if any)
39          MCParticle pos = null;// highest-energy positron daughter
40  
41          List<MCParticle> tridentParticles = null;
42  
43          for (MCParticle particle : MCParticles) {
44              if (particle.getPDGID() == 622) {
45                  trident = particle;
46                  tridentParticles = particle.getDaughters();
47                  break;
48              }
49          }
50          if (trident == null) {
51              return;
52          }
53  
54          fillMCParticleVariables("tri", trident);
55  
56          for (MCParticle particle : tridentParticles) {
57              switch (particle.getPDGID()) {
58                  case -11:
59                      if (pos == null || particle.getEnergy() > pos.getEnergy()) {
60                          pos = particle;
61                      }
62                      break;
63                  case 11:
64                      if (ele1 == null || particle.getEnergy() > ele1.getEnergy()) {
65                          ele2 = ele1;
66                          ele1 = particle;
67                      } else if (ele2 == null || particle.getEnergy() > ele2.getEnergy()) {
68                          ele2 = particle;
69                      }
70                      break;
71              }
72          }
73  
74          if (ele1 != null) {
75              fillMCParticleVariables("triEle1", ele1);
76          }
77          if (ele2 != null) {
78              fillMCParticleVariables("triEle2", ele2);
79          }
80          if (pos != null) {
81              fillMCParticleVariables("triPos", pos);
82          }
83  
84          if (pos != null && ele1 != null) {
85              fillMCPairVariables("triPair1", ele1, pos);
86              if (ele2 != null) {
87                  fillMCPairVariables("triPair2", ele2, pos);
88              }
89          }
90  
91      }

165     protected void fillMCParticleVariables(String prefix, MCParticle particle) {
166         // System.out.format("%d %x\n", particle.getGeneratorStatus(), particle.getSimulatorStatus().getValue());
167         Hep3Vector start = VecOp.mult(beamAxisRotation, particle.getOrigin());
168         Hep3Vector end;
169         MCParticle parent;
170         try {
171             end = VecOp.mult(beamAxisRotation, particle.getEndPoint());
172         } catch (RuntimeException e) {
173             end = null;
174         }
175         
176         try {
177             parent = particle.getParents().get(0);
178         } catch (RuntimeException e) {
179             parent = null;
180         }
181 
182         Hep3Vector p = VecOp.mult(beamAxisRotation, particle.getMomentum());
183 
184         tupleMap.put(prefix + "StartX/D", start.x());
185         tupleMap.put(prefix + "StartY/D", start.y());
186         tupleMap.put(prefix + "StartZ/D", start.z());
187         if (end != null) {
188             tupleMap.put(prefix + "EndX/D", end.x());
189             tupleMap.put(prefix + "EndY/D", end.y());
190             tupleMap.put(prefix + "EndZ/D", end.z());
191         }
192         tupleMap.put(prefix + "StartPX/D", p.x());
193         tupleMap.put(prefix + "StartPY/D", p.y());
194         tupleMap.put(prefix + "StartPZ/D", p.z());
195         tupleMap.put(prefix + "StartP/D", p.magnitude());
196         tupleMap.put(prefix + "M/D", particle.getMass());
197         tupleMap.put(prefix + "E/D", particle.getEnergy());
198         tupleMap.put(prefix + "pdgid/I", (double) particle.getPDGID());
199         if(parent != null){
200             tupleMap.put(prefix + "parentID/I", (double) parent.getPDGID());
201         }
202     }

204     protected void fillMCPairVariables(String prefix, MCParticle ele, MCParticle pos) {
205         HepLorentzVector vtx = VecOp.add(ele.asFourVector(), pos.asFourVector());
206         Hep3Vector vtxP = VecOp.mult(beamAxisRotation, vtx.v3());
207         tupleMap.put(prefix + "PX/D", vtxP.x());
208         tupleMap.put(prefix + "PY/D", vtxP.y());
209         tupleMap.put(prefix + "PZ/D", vtxP.z());
210         tupleMap.put(prefix + "P/D", vtxP.magnitude());
211         tupleMap.put(prefix + "M/D", vtx.magnitude());
212         tupleMap.put(prefix + "E/D", vtx.t());
213     }

216     protected void fillMCWabVariables(EventHeader event) {
217         List<MCParticle> MCParticles = event.getMCParticles();
218 
219         MCParticle wab = null;
220 
221         MCParticle ele1 = null;// conversion electron daughter
222         MCParticle ele2 = null;// recoil wab electron
223         MCParticle pos = null;// conversion positron daughter
224 
225         List<MCParticle> wabParticles = null;
226 
227         for (MCParticle particle : MCParticles) {
228             if (particle.getPDGID() == 22 && particle.getGeneratorStatus() == 1 && particle.getDaughters().size() == 2) {
229                 double wabEnergy = particle.getEnergy();
230                 for(MCParticle p : MCParticles){
231                     if(p.getPDGID() != 11 || !p.getParents().isEmpty()) continue;
232                     double eleEnergy = p.getEnergy();
233                     double energy = wabEnergy + eleEnergy;
234                     if(energy < 0.98 * ebeam || energy > 1.02 * ebeam) continue;
235                     ele2 = p;
236                     wab = particle;
237                     wabParticles = wab.getDaughters();
238                     break;
239                 }
240             }
241         }
242         if (wab == null) {
243             return;
244         }
245 
246         fillMCParticleVariables("wab", wab);
247 
248         for (MCParticle particle : wabParticles) {
249             if(particle.getPDGID() == 11){
250                 pos = particle;
251             }
252             if(particle.getPDGID() == -11){
253                 ele1 = particle;
254             }
255         }
256 
257         if (ele1 != null) {
258             fillMCParticleVariables("wabEle1", ele1);
259         }
260         if (ele2 != null) {
261             fillMCParticleVariables("wabEle2", ele2);
262         }
263         if (pos != null) {
264             fillMCParticleVariables("wabPos", pos);
265         }
266     }
