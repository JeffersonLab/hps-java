1   package org.hps.analysis.tuple;
2   
3   import java.util.Arrays;
4   
5   import org.lcsim.event.EventHeader;
6   import org.lcsim.event.ReconstructedParticle;
7   
8   public class TridentMCTupleDriver extends MCTupleMaker {
9       private final double tupleTrkPCut = 0.9;
10      private final double tupleMaxSumCut = 1.3;
11      
12      @Override
13      protected void setupVariables() {
14          tupleVariables.clear();
15          addEventVariables();
16          addVertexVariables();
17          addParticleVariables("ele");
18          addParticleVariables("pos");
19          addMCTridentVariables();
20  
21          String[] newVars = new String[]{"minPositiveIso/D", "minNegativeIso/D", "minIso/D"};
22          tupleVariables.addAll(Arrays.asList(newVars));
23  
24      }
25  
26      @Override
27      public void process(EventHeader event) {
28  
29          setupCollections(event);
30          
31          for (ReconstructedParticle uncV0 : unConstrainedV0List) {
32              tupleMap.clear();
33              boolean isOK = fillBasicTuple(event, triggerData, uncV0, false);
34              fillMCTridentVariables(event);
35              if (tupleWriter != null && isOK) {
36                  if (!cutTuple || (passesCuts())) {
37                      writeTuple();
38                  }
39              }
40          }
41  
42      }
43  
44      @Override
45      boolean passesCuts() {
46          boolean trkCut = tupleMap.get("eleP/D") < tupleTrkPCut * ebeam && tupleMap.get("posP/D") < tupleTrkPCut * ebeam;
47          boolean sumCut = tupleMap.get("eleP/D") + tupleMap.get("posP/D") < tupleMaxSumCut * ebeam;
48          return (trkCut && sumCut);
49      }
50  }
