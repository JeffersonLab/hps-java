23  /**
24   * Driver for analyzing monster events
25   */
26  public class MonsterEventAnalysis extends Driver {
27  
28      // Plotting
29      protected AIDA aida = AIDA.defaultInstance();
30      ITree tree; 
31      IHistogramFactory histogramFactory; 
32  
33      //List of Sensors
34      private List<HpsSiSensor> sensors = null;
35      
36      Map<String, IHistogram1D> nRawHits = new HashMap<String,IHistogram1D>();
37      Map<String, IHistogram2D> adc = new HashMap<String,IHistogram2D>();
38      Map<String, IHistogram2D> adcMonster = new HashMap<String,IHistogram2D>();
39      Map<String, IHistogram1D> channelMonster = new HashMap<String,IHistogram1D>();
40      Map<String, IHistogram1D> channelAboveThresh = new HashMap<String,IHistogram1D>();
41      Map<String, IHistogram1D> channelBelowThresh = new HashMap<String,IHistogram1D>();
42      
43      //Histogram Settings
44      double minX = 0;
45      double maxX = 10;
46      double minADC = -1000;
47      double maxADC = 5000;
48      int nBins = 120;
49      int nHitMax = 100;
50      
51      //Collection Strings
52      private String fittedHitsCollectionName = "SVTFittedRawTrackerHits";
53      String rawTrackerHitCollectionName = "SVTRawTrackerHits";
54     
55      private static final String SUBDETECTOR_NAME = "Tracker";
56      
57      //Beam Energy
58      double ebeam;
59      
60      public void detectorChanged(Detector detector){
61  
62          aida.tree().cd("/");
63          tree = aida.tree();
64          histogramFactory = IAnalysisFactory.create().createHistogramFactory(tree);
65      
66          //Set Beam Energy
67          BeamEnergyCollection beamEnergyCollection = 
68                  this.getConditionsManager().getCachedConditions(BeamEnergyCollection.class, "beam_energies").getCachedData();        
69          ebeam = beamEnergyCollection.get(0).getBeamEnergy();      
70          
71          // Get the HpsSiSensor objects from the tracker detector element
72          sensors = detector.getSubdetector(SUBDETECTOR_NAME)
73                            .getDetectorElement().findDescendants(HpsSiSensor.class);
74     
75          // If the detector element had no sensors associated with it, throw
76          // an exception
77          if (sensors.size() == 0) {
78              throw new RuntimeException("No sensors were found in this detector.");
79          }
80  
81          
82          for(HpsSiSensor sensor:sensors){
83              nRawHits.put(sensor.getName(), histogramFactory.createHistogram1D("Raw Hits " + sensor.getName(), 640, 0, 640));
84              adc.put(sensor.getName(), histogramFactory.createHistogram2D("ADC Counts " + sensor.getName(), 6, 0, 6, nBins, minADC, maxADC));
85              adcMonster.put(sensor.getName(), histogramFactory.createHistogram2D("ADC Counts Monster Events " + sensor.getName(), 6, 0, 6, nBins, minADC, maxADC));
86              channelMonster.put(sensor.getName(), histogramFactory.createHistogram1D("Channels Hit Monster Events " + sensor.getName(), 640, 0, 640));
87              channelAboveThresh.put(sensor.getName(), histogramFactory.createHistogram1D("Channels Hit Above Threshold " + sensor.getName(), 640, 0, 640));
88              channelBelowThresh.put(sensor.getName(), histogramFactory.createHistogram1D("Channels Hit Below Threshold " + sensor.getName(), 640, 0, 640));
89          }
90  
91      }
92  
93      public void process(EventHeader event){
94          aida.tree().cd("/");
95          
96          // Get the list of fitted hits from the event
97          List<LCRelation> fittedHits = event.get(LCRelation.class, fittedHitsCollectionName);
98           
99          // Map the fitted hits to their corresponding raw hits
100         Map<RawTrackerHit, LCRelation> fittedRawTrackerHitMap = new HashMap<RawTrackerHit, LCRelation>();
101 
102         List<RawTrackerHit> rawHits = event.get(RawTrackerHit.class, rawTrackerHitCollectionName);
103         
104         /*for (RawTrackerHit rawHit : rawHits) {
105              
106             // Access the sensor associated with the raw hit
107             HpsSiSensor sensor = (HpsSiSensor) rawHit.getDetectorElement();
108              
109             // Retrieve the channel ID of the raw hit
110             int channel = rawHit.getIdentifierFieldValue("strip");
111         } */
112         
113                  
114         for (LCRelation fittedHit : fittedHits) {
115             fittedRawTrackerHitMap.put(FittedRawTrackerHit.getRawTrackerHit(fittedHit), fittedHit);
116         }
117         Map<String,Integer> hits = new HashMap<String,Integer>();
118         for(HpsSiSensor sensor:sensors){
119             hits.put(sensor.getName(), 0);
120         }
121         
122         for (RawTrackerHit rawHit : rawHits) {
123             // Access the sensor associated with the raw hit
124             HpsSiSensor sensor = (HpsSiSensor) rawHit.getDetectorElement();
125             Integer nHits = hits.get(sensor.getName());
126             if (nHits == null)
127                 nHits = 0;
128             nHits++;
129             hits.put(rawHit.getDetectorElement().getName(), nHits);
130             
131             // Get the hit amplitude
132             double amplitude = FittedRawTrackerHit.getAmp(fittedRawTrackerHitMap.get(rawHit));
133              
134             // Get the t0 of the hit
135             double t0 = FittedRawTrackerHit.getT0(fittedRawTrackerHitMap.get(rawHit));
136             
137          // Retrieve the channel ID of the raw hit
138             int channel = rawHit.getIdentifierFieldValue("strip");
139             
140             short[] adcs = rawHit.getADCValues();
141             
142             double[] threshold = new double[6];
143 
144             for(int i = 0; i < adcs.length; i++){
145                 adc.get(sensor.getName()).fill(i,adcs[i] - sensor.getPedestal(channel, i));
146                 threshold[i] = adcs[i] - sensor.getPedestal(channel, i) - 2 * sensor.getNoise(channel, i);
147             }
148 
149             boolean sample1 = threshold[0] > 0 && threshold [1] > 0 && threshold[2] > 0;
150             boolean sample2 = threshold[1] > 0 && threshold [2] > 0 && threshold[3] > 0;
151             boolean sample3 = threshold[2] > 0 && threshold [3] > 0 && threshold[4] > 0;
152             boolean sample4 = threshold[3] > 0 && threshold [4] > 0 && threshold[5] > 0;
153             
154             if(sample1 || sample2 || sample3 || sample4){
155                 channelAboveThresh.get(sensor.getName()).fill(channel);
156             }
157             else{
158                 channelBelowThresh.get(sensor.getName()).fill(channel);
159             }
160             
161         }
162         for (HpsSiSensor sensor : sensors) {
163             Integer nHits = hits.get(sensor.getName());
164             if (nHits == null)
165                 nRawHits.get(sensor.getName()).fill(0);
166             else{
167                 nRawHits.get(sensor.getName()).fill(nHits);
168                 if(nHits > nHitMax){
169                     for (RawTrackerHit rawHit : rawHits) {
170                         if(!sensor.equals(rawHit.getDetectorElement())) continue;
171                         int channel = rawHit.getIdentifierFieldValue("strip");
172                         short[] adcs = rawHit.getADCValues();
173                         for(int i = 0; i < adcs.length; i++){
174                             adcMonster.get(sensor.getName()).fill(i,adcs[i] - sensor.getPedestal(channel, i));
175                         }
176                         channelMonster.get(sensor.getName()).fill(channel);
177                     }
178                 }
179             }
180         }
181     }
