33      /*
34       *  creates a new plotter with one region and puts the histogram in it
35       *  copied from org.hps.monitoring.drivers.ecal.EcalMonitoringUtilities.java
36       */
37      static IPlotter plot(IPlotterFactory plotterFactory, IBaseHistogram histogram, IPlotterStyle style, boolean show) {
38          if (style == null)
39              style = getPlotterStyle(histogram);
40          IPlotter plotter = plotterFactory.create(histogram.title());
41          plotter.createRegion();
42          plotter.region(0).plot(histogram, style);
43          if (show)
44              plotter.show();
45          return plotter;
46      }

71      /*
72       *  gets default plotter style based on histogram type
73       *  copied from org.hps.monitoring.drivers.ecal.EcalMonitoringUtilities.java
74       */
75      static IPlotterStyle getPlotterStyle(IBaseHistogram histogram) {
76          StyleRegistry styleRegistry = StyleRegistry.getStyleRegistry();
77          IStyleStore store = styleRegistry.getStore("DefaultStyleStore");
78          IPlotterStyle style = null;
79          if ((histogram instanceof IHistogram1D) || (histogram instanceof ICloud1D) || (histogram instanceof IProfile1D))
80              style = store.getStyle("DefaultHistogram1DStyle");
81          else if ((histogram instanceof IHistogram2D) || (histogram instanceof IProfile2D)) {
82              style = store.getStyle("DefaultColorMapStyle");
83              style.statisticsBoxStyle().setVisible(false);
84              style.setParameter("hist2DStyle", "colorMap");
85              style.dataStyle().fillStyle().setParameter("colorMapScheme", "rainbow");
86          }
87          if (style == null)
88              throw new RuntimeException("A default style could not be found for " + histogram.title());
89  
90          //custom stuff...mg
91          style.dataStyle().errorBarStyle().setVisible(false);
92          style.legendBoxStyle().setVisible(false);
93          style.dataStyle().outlineStyle().setVisible(false);
94  
95          return style;
96      }

99       *  gets default plotter style for a function type
100      *  copied from org.hps.monitoring.drivers.ecal.EcalMonitoringUtilities.java
101      */
102     static IPlotterStyle getPlotterStyle(IFunction func) {
103         StyleRegistry styleRegistry = StyleRegistry.getStyleRegistry();
104         IStyleStore store = styleRegistry.getStore("DefaultStyleStore");
105         IPlotterStyle style = null;
106         style = store.getStyle("DefaultFunctionStyle");
107         if (style == null)
108             throw new RuntimeException("A default style could not be found for " + func.title());
109         return style;
110     }

112     static IFitResult performGaussianFit(IHistogram1D histogram) {
113         IFunctionFactory functionFactory = aida.analysisFactory().createFunctionFactory(null);
114         IFitFactory fitFactory = aida.analysisFactory().createFitFactory();
115         IFunction function = functionFactory.createFunctionByName("Example Fit", "G");
116         IFitter fitter = fitFactory.createFitter("chi2", "jminuit");
117         double[] parameters = new double[3];
118         parameters[0] = histogram.maxBinHeight();
119         parameters[1] = histogram.mean();
120         parameters[2] = histogram.rms();
121         function.setParameters(parameters);
122         IFitResult fitResult = null;
123         Logger minuitLogger = Logger.getLogger("org.freehep.math.minuit");
124         minuitLogger.setLevel(Level.OFF);
125         minuitLogger.info("minuit logger test");
126 
127         try {
128             fitResult = fitter.fit(histogram, function);
129         } catch (RuntimeException e) {
130             LOGGER.info(e.getMessage());
131         }
132         return fitResult;
133     }

135     static void fitAndPutParameters(IHistogram1D hist, IFunction function) {
136         IFitResult fr = performGaussianFit(hist);
137         if (fr != null) {
138             IFunction currentFitFunction = fr.fittedFunction();
139             function.setParameters(currentFitFunction.parameters());
140         }
141     }
