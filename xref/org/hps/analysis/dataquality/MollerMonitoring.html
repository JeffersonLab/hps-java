34          TRK_QUALITY("Trk Quality"), 
35          VTX_QUALITY("Vtx Quality"), 
36          VERTEX_CUTS("Vtx Cuts"), 
37          TIMING("Timing"), 
38          TRACK_CUTS("Trk Cuts"), 
39          EVENT_QUALITY("Evt Quality"), 
40          FRONT_HITS("Front Hits");

42          private final String name;
43          private final static int firstVertexingCut = 6;
44  
45          Cut(String name) {
46              this.name = name;
47          }
48  
49          static int bitmask(EnumSet<Cut> flags) {
50              int mask = 0;
51              for (Cut flag : flags) {
52                  mask |= 1 << flag.ordinal();
53              }
54              return mask;
55          }

59              return name;

349     @Override
350     public void process(EventHeader event) {
351         /* make sure everything is there */
352         if (!event.hasCollection(ReconstructedParticle.class, finalStateParticlesColName)) {
353             return;
354         }
355         if (!event.hasCollection(ReconstructedParticle.class, unconstrainedV0CandidatesColName)) {
356             return;
357         }
358         // if (!event.hasCollection(ReconstructedParticle.class, beamConV0CandidatesColName)) {
359         // return;
360         // }
361         // if (!event.hasCollection(ReconstructedParticle.class, targetV0ConCandidatesColName)) {
362         // return;
363         // }
364 
365         // check to see if this event is from the correct trigger (or "all");
366         if (!matchTrigger(event)) {
367             return;
368         }
369         
370         int runNumber = event.getRunNumber();
371 
372         List<ReconstructedParticle> unConstrainedV0List = event.get(ReconstructedParticle.class,
373                 unconstrainedV0CandidatesColName);
374 
375         RelationalTable hitToStrips = TrackUtils.getHitToStripsTable(event);
376         RelationalTable hitToRotated = TrackUtils.getHitToRotatedTable(event);
377 
378         List<ReconstructedParticle> fspList = event.get(ReconstructedParticle.class, finalStateParticlesColName);
379         int npos = 0;
380         int ntrk = 0;
381         for (ReconstructedParticle fsp : fspList) {
382             if (isGBL != TrackType.isGBL(fsp.getType())) {
383                 continue;
384             }
385             if (fsp.getCharge() != 0) {
386                 ntrk++;
387             }
388             if (fsp.getCharge() > 0) {
389                 npos++;
390             }
391         }
392 
393         List<ReconstructedParticle> candidateList = new ArrayList<>();
394         List<ReconstructedParticle> vertCandidateList = new ArrayList<>();
395         for (ReconstructedParticle uncV0 : unConstrainedV0List) {
396             if (isGBL != TrackType.isGBL(uncV0.getType())) {
397                 continue;
398             }
399             Vertex uncVert = uncV0.getStartVertex();
400             // v0 & vertex-quality cuts
401             Hep3Vector v0MomRot = VecOp.mult(beamAxisRotation, uncV0.getMomentum());
402             Hep3Vector v0Vtx = VecOp.mult(beamAxisRotation, uncVert.getPosition());
403 
404             List<Track> tracks = new ArrayList<Track>();
405             ReconstructedParticle top = uncV0.getParticles().get(ReconParticleDriver.MOLLER_TOP);
406             ReconstructedParticle bot = uncV0.getParticles().get(ReconParticleDriver.MOLLER_BOT);
407             if (top.getCharge() != -1 || bot.getCharge() != -1) {
408                 throw new RuntimeException("incorrect charge on v0 daughters");
409             }
410             tracks.add(top.getTracks().get(0));
411             tracks.add(bot.getTracks().get(0));
412             if (tracks.size() != 2) {
413                 throw new RuntimeException("expected two tracks in vertex, got " + tracks.size());
414             }
415 
416             Double[] topIso = TrackUtils.getIsolations(top.getTracks().get(0), hitToStrips, hitToRotated);
417             Double[] botIso = TrackUtils.getIsolations(bot.getTracks().get(0), hitToStrips, hitToRotated);
418             double minL1Iso = -9999;
419             if (topIso[0] != null && botIso[0] != null) {
420                 double topL1Iso = Math.min(Math.abs(topIso[0]), Math.abs(topIso[1]));
421                 double botL1Iso = Math.min(Math.abs(botIso[0]), Math.abs(botIso[1]));
422                 minL1Iso = Math.min(topL1Iso, botL1Iso);
423             }
424 
425             double tTop = TrackUtils.getTrackTime(top.getTracks().get(0), hitToStrips, hitToRotated);
426             double tBot = TrackUtils.getTrackTime(bot.getTracks().get(0), hitToStrips, hitToRotated);
427             Hep3Vector pTopRot = VecOp.mult(beamAxisRotation, top.getMomentum());
428             Hep3Vector pBotRot = VecOp.mult(beamAxisRotation, bot.getMomentum());
429 
430             Hep3Vector topAtEcal = TrackUtils.getTrackPositionAtEcal(top.getTracks().get(0), runNumber);
431             Hep3Vector botAtEcal = TrackUtils.getTrackPositionAtEcal(bot.getTracks().get(0), runNumber);
432 
433             BilliorVertexer vtxFitter = new BilliorVertexer(TrackUtils.getBField(event.getDetector()).y());
434             vtxFitter.setBeamSize(beamSize);
435             vtxFitter.setBeamPosition(beamPos);
436             List<BilliorTrack> billiorTracks = new ArrayList<BilliorTrack>();
437             billiorTracks.add(new BilliorTrack(top.getTracks().get(0)));
438             billiorTracks.add(new BilliorTrack(bot.getTracks().get(0)));
439 
440             vtxFitter.doBeamSpotConstraint(true);
441             BilliorVertex bsconVertex = vtxFitter.fitVertex(billiorTracks);
442             ReconstructedParticle bscV0 = HpsReconParticleDriver.makeReconstructedParticle(top, bot, bsconVertex);
443             Hep3Vector bscMomRot = VecOp.mult(beamAxisRotation, bscV0.getMomentum());
444             Hep3Vector bscVtx = VecOp.mult(beamAxisRotation, bscV0.getStartVertex().getPosition());
445 
446             vtxFitter.doTargetConstraint(true);
447             BilliorVertex tarVertex = vtxFitter.fitVertex(billiorTracks);
448             ReconstructedParticle tarV0 = HpsReconParticleDriver.makeReconstructedParticle(top, bot, tarVertex);
449             Hep3Vector tarMomRot = VecOp.mult(beamAxisRotation, tarV0.getMomentum());
450             Hep3Vector tarVtx = VecOp.mult(beamAxisRotation, tarV0.getStartVertex().getPosition());
451 
452             vtxFitter.setBeamSize(vzcBeamSize);
453             vtxFitter.doTargetConstraint(true);
454             BilliorVertex vzcVertex = vtxFitter.fitVertex(billiorTracks);
455             ReconstructedParticle vzcV0 = HpsReconParticleDriver.makeReconstructedParticle(top, bot, vzcVertex);
456             Hep3Vector vzcMomRot = VecOp.mult(beamAxisRotation, vzcV0.getMomentum());
457             Hep3Vector vzcVtx = VecOp.mult(beamAxisRotation, vzcV0.getStartVertex().getPosition());
458 
459             // start applying cuts
460             EnumSet<Cut> bits = EnumSet.noneOf(Cut.class);
461 
462             boolean trackQualityCut = Math.max(top.getTracks().get(0).getChi2(), bot.getTracks().get(0).getChi2()) < (isGBL ? maxChi2GBLTrack
463                     : maxChi2SeedTrack);
464             if (trackQualityCut) {
465                 bits.add(Cut.TRK_QUALITY);
466             }
467 
468             boolean v0QualityCut = uncVert.getChi2() < maxUnconVertChi2 && bsconVertex.getChi2() < maxBsconVertChi2;
469             if (v0QualityCut) {
470                 bits.add(Cut.VTX_QUALITY);
471             }
472 
473             boolean vertexMomentumCut = v0MomRot.z() < v0PzMaxCut * beamEnergy
474                     && v0MomRot.z() > v0PzMinCut * beamEnergy && Math.abs(v0MomRot.x()) < v0PxCut * beamEnergy
475                     && Math.abs(v0MomRot.y()) < v0PyCut * beamEnergy;
476             boolean vertexPositionCut = Math.abs(v0Vtx.x()) < v0UnconVxCut && Math.abs(v0Vtx.y()) < v0UnconVyCut
477                     && Math.abs(v0Vtx.z()) < v0UnconVzCut && Math.abs(bscVtx.x()) < v0BsconVxCut
478                     && Math.abs(bscVtx.y()) < v0BsconVyCut;
479             if (vertexMomentumCut && vertexPositionCut) {
480                 bits.add(Cut.VERTEX_CUTS);
481             }
482 
483             boolean trackTimeDiffCut = Math.abs(tTop - tBot) < trkTimeDiff;
484             if (trackTimeDiffCut) {
485                 bits.add(Cut.TIMING);
486             }
487 
488             // boolean topBottomCut = top.getMomentum().y() * bot.getMomentum().y() < 0;
489             // boolean pMinCut = top.getMomentum().magnitude() > minTrkPCut * beamEnergy &&
490             // bot.getMomentum().magnitude() > minTrkPCut * beamEnergy;
491             boolean pMaxCut = top.getMomentum().magnitude() < maxTrkPCut * beamEnergy
492                     && bot.getMomentum().magnitude() < maxTrkPCut * beamEnergy;
493             // boolean pSumCut = top.getMomentum().magnitude() + bot.getMomentum().magnitude() < maxPSumCut * beamEnergy
494             // && top.getMomentum().magnitude() + bot.getMomentum().magnitude() < minPSumCut * beamEnergy;
495             // boolean pMaxCut = top.getMomentum().magnitude() < maxTrkPCut * beamEnergy &&
496             // bot.getMomentum().magnitude() < maxTrkPCut * beamEnergy;
497             if (pMaxCut) {
498                 bits.add(Cut.TRACK_CUTS);
499             }
500 
501             boolean eventTrkCountCut = ntrk >= 2 && ntrk <= nTrkMax;
502             boolean eventPosCountCut = npos >= 1 && npos <= nPosMax;
503             if (eventTrkCountCut && eventPosCountCut) {
504                 bits.add(Cut.EVENT_QUALITY);
505             }
506 
507             boolean frontHitsCut = topIso[0] != null && botIso[0] != null && topIso[2] != null && botIso[2] != null;
508             if (frontHitsCut) {
509                 bits.add(Cut.FRONT_HITS);
510             }
511 
512             for (Cut cut : Cut.values()) {
513                 if (bits.contains(cut)) {
514                     if (cut.ordinal() == Cut.firstVertexingCut) {// if we get here, we've passed all non-vertexing cuts
515                         candidateList.add(uncV0);
516                     }
517                 } else {
518                     break;
519                 }
520             }
521 
522             if (bits.containsAll(EnumSet.range(Cut.values()[0], Cut.values()[Cut.firstVertexingCut - 1]))) {
523                 candidateList.add(uncV0);
524             }
525             if (bits.equals(EnumSet.allOf(Cut.class))) {
526                 vertCandidateList.add(uncV0);
527             }
528         }
529 
530         nTriCand.fill(candidateList.size());
531         nVtxCand.fill(vertCandidateList.size());
532 
533         if (!candidateList.isEmpty()) {
534             // pick the best candidate...for now just pick a random one.
535             ReconstructedParticle bestCandidate = candidateList.get((int) (Math.random() * candidateList.size()));
536 
537             // fill some stuff:
538             ReconstructedParticle top = bestCandidate.getParticles().get(ReconParticleDriver.MOLLER_TOP);
539             ReconstructedParticle bot = bestCandidate.getParticles().get(ReconParticleDriver.MOLLER_BOT);
540             if (top.getCharge() != -1 || bot.getCharge() != -1) {
541                 throw new RuntimeException("vertex needs two e- but is missing one or both");
542             }
543 
544             Hep3Vector pEleRot = VecOp.mult(beamAxisRotation, top.getMomentum());
545             Hep3Vector pPosRot = VecOp.mult(beamAxisRotation, bot.getMomentum());
546             Hep3Vector v0Vtx = VecOp.mult(beamAxisRotation, bestCandidate.getStartVertex().getPosition());
547             Hep3Vector pBestV0Rot = VecOp.mult(beamAxisRotation, bestCandidate.getMomentum());
548 
549             // triTrackTime2D.fill(tEle, tPos);
550             // triTrackTimeDiff.fill(tEle - tPos);
551             triZVsMomentum.fill(bestCandidate.getMomentum().magnitude(), v0Vtx.z());
552             triMassMomentum.fill(bestCandidate.getMomentum().magnitude(), bestCandidate.getMass());
553             triTrackMomentum2D.fill(top.getMomentum().magnitude(), bot.getMomentum().magnitude());
554             triPyEleVsPyPos.fill(pEleRot.y(), pPosRot.y());
555             triPxEleVsPxPos.fill(pEleRot.x(), pPosRot.x());
556             triSumP.fill(bestCandidate.getMomentum().magnitude());
557             triDeltaP.fill(bot.getMomentum().magnitude() - top.getMomentum().magnitude());
558 
559             triPxPy.fill(pBestV0Rot.x(), pBestV0Rot.y());
560             triMass.fill(bestCandidate.getMass());
561             triZVsMass.fill(bestCandidate.getMass(), v0Vtx.z());
562             // triX.fill(v0Vtx.x());
563             // triY.fill(v0Vtx.y());
564             // triZ.fill(v0Vtx.z());
565             triPx.fill(pBestV0Rot.x());
566             triPy.fill(pBestV0Rot.y());
567             triPz.fill(pBestV0Rot.z());
568             triU.fill(pBestV0Rot.x() / pBestV0Rot.magnitude());
569             triV.fill(pBestV0Rot.y() / pBestV0Rot.magnitude());
570             // triXY.fill(v0Vtx.x(), v0Vtx.y());
571             // triZY.fill(v0Vtx.y(), v0Vtx.z());
572         }
573 
574         if (!vertCandidateList.isEmpty()) {
575             // pick the best candidate...for now just pick a random one.
576             ReconstructedParticle bestCandidate = vertCandidateList
577                     .get((int) (Math.random() * vertCandidateList.size()));
578             Vertex unconVertex = bestCandidate.getStartVertex();
579 
580             // fill some stuff:
581             ReconstructedParticle top = bestCandidate.getParticles().get(ReconParticleDriver.MOLLER_TOP);
582             ReconstructedParticle bot = bestCandidate.getParticles().get(ReconParticleDriver.MOLLER_BOT);
583             if (top.getCharge() != -1 || bot.getCharge() != -1) {
584                 throw new RuntimeException("vertex needs two e- but is missing one or both");
585             }
586 
587             Hep3Vector pEleRot = VecOp.mult(beamAxisRotation, top.getMomentum());
588             Hep3Vector pPosRot = VecOp.mult(beamAxisRotation, bot.getMomentum());
589             Hep3Vector v0Vtx = VecOp.mult(beamAxisRotation, unconVertex.getPosition());
590             Hep3Vector pBestV0Rot = VecOp.mult(beamAxisRotation, bestCandidate.getMomentum());
591 
592             // vertTrackTime2D.fill(tEle, tPos);
593             // vertTrackTimeDiff.fill(tEle - tPos);
594             vertZVsMomentum.fill(bestCandidate.getMomentum().magnitude(), v0Vtx.z());
595             vertMassMomentum.fill(bestCandidate.getMomentum().magnitude(), bestCandidate.getMass());
596             vertTrackMomentum2D.fill(top.getMomentum().magnitude(), bot.getMomentum().magnitude());
597             vertPyEleVsPyPos.fill(pEleRot.y(), pPosRot.y());
598             vertPxEleVsPxPos.fill(pEleRot.x(), pPosRot.x());
599             vertSumP.fill(bestCandidate.getMomentum().magnitude());
600             vertDeltaP.fill(bot.getMomentum().magnitude() - top.getMomentum().magnitude());
601 
602             vertPxPy.fill(pBestV0Rot.x(), pBestV0Rot.y());
603             vertMass.fill(bestCandidate.getMass());
604             vertZVsMass.fill(bestCandidate.getMass(), v0Vtx.z());
605             vertX.fill(v0Vtx.x());
606             vertY.fill(v0Vtx.y());
607             // vertZ.fill(v0Vtx.z());
608             vertPx.fill(pBestV0Rot.x());
609             vertPy.fill(pBestV0Rot.y());
610             vertPz.fill(pBestV0Rot.z());
611             vertU.fill(pBestV0Rot.x() / pBestV0Rot.magnitude());
612             vertV.fill(pBestV0Rot.y() / pBestV0Rot.magnitude());
613             vertXY.fill(v0Vtx.x(), v0Vtx.y());
614             vertZY.fill(v0Vtx.y(), v0Vtx.z());
615         }
616     }
