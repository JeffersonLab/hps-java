90      public void detectorChanged(Detector detector){
91  
92          aida.tree().cd("/");
93          tree = aida.tree();
94          histogramFactory = IAnalysisFactory.create().createHistogramFactory(tree);
95      
96          //Set Beam Energy
97          BeamEnergyCollection beamEnergyCollection = 
98                  this.getConditionsManager().getCachedConditions(BeamEnergyCollection.class, "beam_energies").getCachedData();        
99          ebeam = beamEnergyCollection.get(0).getBeamEnergy();      
100         
101         // Get the HpsSiSensor objects from the tracker detector element
102         sensors = detector.getSubdetector(SUBDETECTOR_NAME)
103                           .getDetectorElement().findDescendants(HpsSiSensor.class);
104    
105         // If the detector element had no sensors associated with it, throw
106         // an exception
107         if (sensors.size() == 0) {
108             throw new RuntimeException("No sensors were found in this detector.");
109         }
110 
111         trackP = aida.histogram1D("Track Momentum", 100, 0, 5.0);
112         trackTopP = aida.histogram1D("Track Top Momentum", 100, 0, 5.0);
113         trackBotP = aida.histogram1D("Track Bot Momentum", 100, 0, 5.0);
114         trackEleP = aida.histogram1D("Track Ele Momentum", 100, 0, 5.0);
115         trackEleTopP = aida.histogram1D("Track Ele Top Momentum", 100, 0, 5.0);
116         trackEleBotP = aida.histogram1D("Track Ele Bot Momentum", 100, 0, 5.0);
117         trackPosP = aida.histogram1D("Track Pos Momentum", 100, 0, 5.0);
118         trackPosTopP = aida.histogram1D("Track Pos Top Momentum", 100, 0, 5.0);
119         trackPosBotP = aida.histogram1D("Track Pos Bot Momentum", 100, 0, 5.0);
120         
121         for(int i = 0; i < nLay; i++){
122             int layer = i + 1;
123             trackerHitPositionTopZ.put(layer, histogramFactory.createHistogram1D("Tracker Hit Z Top Layer " + layer, 10000, 0, 1000));
124             trackerHitPositionTop.put(layer, histogramFactory.createHistogram2D("Tracker Hit X-Y Top Layer " + layer, 100, -20, 20, 100,0,20));
125             trackerHitPositionBotZ.put(layer, histogramFactory.createHistogram1D("Tracker Hit Z Bot Layer " + layer, 10000, 0, 1000));
126             trackerHitPositionBot.put(layer, histogramFactory.createHistogram2D("Tracker Hit X-Y Bot Layer " + layer, 100, -20, 20, 100,-20,0));
127         }
128         
129         for(HpsSiSensor sensor:sensors){
130             fittedRawHitTime.put(sensor.getName(), histogramFactory.createHistogram1D("Fitted Raw Hit Time " + sensor.getName(), 100, -100, 100));
131             clusterTime.put(sensor.getName(), histogramFactory.createHistogram1D("1D Cluster Hit Time " + sensor.getName(), 100, -100, 100));
132             helicalTrackHitTime.put(sensor.getName(), histogramFactory.createHistogram1D("Helical Track Hit Time " + sensor.getName(), 100, -100, 100));
133             fittedRawHitPositionZ.put(sensor.getName(), histogramFactory.createHistogram1D("Fitted Raw Hit Z " + sensor.getName(), 10000, 0, 1000));
134             clusterPositionZ.put(sensor.getName(), histogramFactory.createHistogram1D("1D Cluster Hit Z " + sensor.getName(), 10000, 0, 1000));
135             clusterDiff.put(sensor.getName(), histogramFactory.createHistogram1D("1D Cluster Hit Distance " + sensor.getName(), 100, 0, 100));
136             clusterDiff2D.put(sensor.getName(), histogramFactory.createHistogram2D("1D Cluster Hit Diff Y vs X " + sensor.getName(), 100, -20, 20, 100, -10, 10));
137             helicalTrackHitPositionZ.put(sensor.getName(), histogramFactory.createHistogram1D("Helical Track Hit Z " + sensor.getName(), 10000, 0, 1000));
138             fittedRawHitPosition.put(sensor.getName(), histogramFactory.createHistogram2D("Fitted Raw Hit X-Y " + sensor.getName(), 100, -100, 100, 100,-100,100));
139             clusterPositionSensorCoord.put(sensor.getName(), histogramFactory.createHistogram2D("1D Cluster Hit U-V " + sensor.getName(), 100, -15, 15, 100,-10,10));
140             clusterPosition.put(sensor.getName(), histogramFactory.createHistogram2D("1D Cluster Hit X-Y " + sensor.getName(), 100, -10, 10, 100,-10,10));
141             helicalTrackHitPosition.put(sensor.getName(), histogramFactory.createHistogram2D("Helical Track Hit X-Y " + sensor.getName(), 100, -20, 20, 100,-5,5));
142             xPosvsChan.put(sensor.getName(), histogramFactory.createHistogram2D("X vs Chan " + sensor.getName(), 640, 0, 640, 100,-20,20));
143             yPosvsChan.put(sensor.getName(), histogramFactory.createHistogram2D("Y vs Chan " + sensor.getName(), 640, 0, 640, 100,-20,20));
144         }

148     public void process(EventHeader event){
149         aida.tree().cd("/");
150         
151         // Get the list of fitted hits from the event
152         List<LCRelation> fittedHits = event.get(LCRelation.class, fittedHitsCollectionName);
153          
154         // Map the fitted hits to their corresponding raw hits
155         Map<RawTrackerHit, LCRelation> fittedRawTrackerHitMap = new HashMap<RawTrackerHit, LCRelation>();
156 
157         List<RawTrackerHit> rawHits = event.get(RawTrackerHit.class, rawTrackerHitCollectionName);
158         
159         List<TrackerHit> clusterHits = event.get(TrackerHit.class, stripClusterHitCollectionName);
160         
161         List<TrackerHit> helicalTrackHits = event.get(TrackerHit.class, helicalTrackerHitCollectionName);
162         
163         if(event.hasCollection(SimTrackerHit.class, trackerHitCollectionName)){
164             System.out.println("Event has tracker hits!");
165             List<SimTrackerHit> trackerHits = event.get(SimTrackerHit.class,trackerHitCollectionName);
166             for(SimTrackerHit hit : trackerHits){
167                 //HpsSiSensor sensor = ((HpsSiSensor) ((RawTrackerHit) hit.getRawHits().get(0)).getDetectorElement());
168                 int layer = hit.getLayer();
169                 if(hit.getPosition()[1] > 0){
170                     trackerHitPositionTopZ.get(layer).fill(hit.getPosition()[2]);
171                     trackerHitPositionTop.get(layer).fill(hit.getPosition()[0],hit.getPosition()[1]);
172                 }
173                 else{
174                     trackerHitPositionBotZ.get(layer).fill(hit.getPosition()[2]);
175                     trackerHitPositionBot.get(layer).fill(hit.getPosition()[0],hit.getPosition()[1]);
176                 }
177             }
178         }
179                  
180         for (LCRelation fittedHit : fittedHits) {
181             fittedRawTrackerHitMap.put(FittedRawTrackerHit.getRawTrackerHit(fittedHit), fittedHit);
182         }
183         
184         for (LCRelation hit : fittedHits) {
185             RawTrackerHit rth = (RawTrackerHit) hit.getFrom();
186             GenericObject pars = (GenericObject) hit.getTo();
187 
188             HpsSiSensor sensor = ((HpsSiSensor) rth.getDetectorElement());
189             //this is a clever way to get the parameters we want from the generic object
190             double t0 = ShapeFitParameters.getT0(pars);
191             double amp = ShapeFitParameters.getAmp(pars);
192             double chiProb = ShapeFitParameters.getChiProb(pars);
193             int channel = rth.getIdentifierFieldValue("strip");
194             fittedRawHitTime.get(sensor.getName()).fill(t0);
195             fittedRawHitPositionZ.get(sensor.getName()).fill(rth.getPosition()[2]);
196             fittedRawHitPosition.get(sensor.getName()).fill(rth.getPosition()[0],rth.getPosition()[1]);
197         }
198         
199         for(TrackerHit hit : clusterHits){
200             SiTrackerHitStrip1D local = ((SiTrackerHitStrip1D) hit).getTransformedHit(TrackerHitType.CoordinateSystem.SENSOR);
201             SiTrackerHitStrip1D global = ((SiTrackerHitStrip1D) hit).getTransformedHit(TrackerHitType.CoordinateSystem.GLOBAL);
202             HpsSiSensor sensor = ((HpsSiSensor) ((RawTrackerHit) hit.getRawHits().get(0)).getDetectorElement());
203             clusterPositionSensorCoord.get(sensor.getName()).fill(local.getPosition()[1],local.getPosition()[0]);
204             clusterTime.get(sensor.getName()).fill(hit.getTime());
205             clusterPositionZ.get(sensor.getName()).fill(hit.getPosition()[2]);
206             clusterPosition.get(sensor.getName()).fill(hit.getPosition()[0],hit.getPosition()[1]);
207             int channel = ((RawTrackerHit) hit.getRawHits().get(0)).getIdentifierFieldValue("strip");
208             xPosvsChan.get(sensor.getName()).fill(channel, hit.getPosition()[0]);
209             yPosvsChan.get(sensor.getName()).fill(channel, hit.getPosition()[1]);
210             double dist = 9999;
211             double [] pos1 = hit.getPosition();
212             double [] minpos2 = {9999,9999,9999};
213             for(TrackerHit hit2 : clusterHits){
214                 HpsSiSensor sensor2 = ((HpsSiSensor) ((RawTrackerHit) hit2.getRawHits().get(0)).getDetectorElement());
215                 if(!(sensor.getLayerNumber() == sensor2.getLayerNumber())){
216                     continue;
217                 }
218                 if(sensor.isTopLayer() == sensor2.isTopLayer()){
219                     continue;
220                 }
221                 if(sensor.equals(sensor2)){
222                     continue;
223                 }
224                 double [] pos2 = hit2.getPosition();
225                 if(Math.sqrt((pos1[0]-pos2[0])*(pos1[0]-pos2[0])+(pos1[1]-pos2[1])*(pos1[1]-pos2[1])) < dist){
226                     dist = Math.sqrt((pos1[0]-pos2[0])*(pos1[0]-pos2[0])+(pos1[1]-pos2[1])*(pos1[1]-pos2[1]));
227                     minpos2 = pos2;
228                 }
229                 //System.out.println(sensor.getName() + " " + sensor2.getName() + " " + dist);
230             }
231             //System.out.println("Min Distance " + dist);
232             clusterDiff.get(sensor.getName()).fill(dist);
233             clusterDiff2D.get(sensor.getName()).fill(pos1[0]-minpos2[0],pos1[1]-minpos2[1]);
234         }
235         
236         for(TrackerHit hit : helicalTrackHits){
237             int i = 1;
238             HpsSiSensor sensor =  ((HpsSiSensor) ((RawTrackerHit) hit.getRawHits().get(0)).getDetectorElement());
239             helicalTrackHitTime.get(sensor.getName()).fill(hit.getTime());
240             helicalTrackHitPositionZ.get(sensor.getName()).fill(hit.getPosition()[2]);
241             helicalTrackHitPosition.get(sensor.getName()).fill(hit.getPosition()[0],hit.getPosition()[1]);
242         }
243         
244         if(event.hasCollection(Track.class, trackCollectionName)){
245             List<Track> tracks = event.get(Track.class, trackCollectionName);
246             for(Track track : tracks){
247                 TrackState ts = track.getTrackStates().get(0);
248                 double[] p = ts.getMomentum();
249                 double pTot = Math.sqrt(p[0]*p[0] + p[1]*p[1] + p[2]*p[2]);
250                 trackP.fill(pTot);
251                 if(ts.getOmega() > 0){
252                     trackEleP.fill(pTot);
253                 }
254                 else{
255                     trackPosP.fill(pTot);
256                 }
257                 if(p[1] > 0){
258                     trackTopP.fill(pTot);
259                     if(ts.getOmega() > 0){
260                         trackEleTopP.fill(pTot);
261                     }
262                     else{
263                         trackPosTopP.fill(pTot);
264                     }
265                 }
266                 else{
267                     trackBotP.fill(pTot);
268                     if(ts.getOmega() > 0){
269                         trackEleBotP.fill(pTot);
270                     }
271                     else{
272                         trackPosBotP.fill(pTot);
273                     }
274                 }
275             }
276         }
277     }
