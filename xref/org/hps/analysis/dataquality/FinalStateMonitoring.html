170     @Override
171     public void process(EventHeader event) {
172         /* make sure everything is there */
173 
174         if (!event.hasCollection(ReconstructedParticle.class, finalStateParticlesColName)) {
175             if (debug)
176                 LOGGER.info(finalStateParticlesColName + " collection not found???");
177             return;
178         }
179 
180         RelationalTable hitToStrips = TrackUtils.getHitToStripsTable(event);
181         RelationalTable hitToRotated = TrackUtils.getHitToRotatedTable(event);
182 
183         // check to see if this event is from the correct trigger (or "all");
184         if (!matchTrigger(event))
185             return;
186 
187         nRecoEvents++;
188         int nPhotons = 0;  // number of photons
189         int nUnAssTracks = 0; // number of tracks w/o clusters
190         List<ReconstructedParticle> finalStateParticles = event.get(ReconstructedParticle.class,
191                 finalStateParticlesColName);
192         if (debug)
193             LOGGER.info("This events has " + finalStateParticles.size() + " final state particles");
194         for (ReconstructedParticle fsPart : finalStateParticles) {
195             if (debug)
196                 LOGGER.info("PDGID = " + fsPart.getParticleIDUsed() + "; charge = " + fsPart.getCharge() + "; pz = "
197                         + fsPart.getMomentum().x());
198 //            if (isGBL != TrackType.isGBL(fsPart.getType()))
199 //                continue;
200             // Extrapolate the track to the Ecal cluster position
201             boolean isPhoton = false;
202             boolean hasCluster = true;
203             Track fsTrack = null;
204             Cluster fsCluster = null;
205             // TODO: mg-May 14, 2014 use PID to do this instead...not sure if that's implemented yet
206             if (fsPart.getTracks().size() == 1)// should always be 1 or zero for final state particles
207 
208                 fsTrack = fsPart.getTracks().get(0);
209             // get the cluster
210             if (fsPart.getClusters().size() == 1)
211                 fsCluster = fsPart.getClusters().get(0);
212             else
213                 hasCluster = false;
214             if (fsPart.getCharge() == 0)
215                 isPhoton = true;
216 
217             // deal with electrons & positrons first
218             if (!isPhoton) {
219                 double charge = fsPart.getCharge();
220                 Hep3Vector mom = fsPart.getMomentum();
221                 if (charge < 0) {
222                     nTotEle++;
223                     elePx.fill(mom.x());
224                     elePy.fill(mom.y());
225                     elePz.fill(mom.z());
226                     elePzBeam.fill(mom.magnitude());
227                     if (mom.y() > 0) {
228                         elePzBeamTop.fill(mom.magnitude());
229                         elePTop.fill(mom.magnitude());
230                     } else {
231                         elePzBeamBottom.fill(mom.magnitude());
232                         elePBottom.fill(mom.magnitude());
233                     }
234                 } else {
235                     nTotPos++;
236                     posPx.fill(mom.x());
237                     posPy.fill(mom.y());
238                     posPz.fill(mom.z());
239                     if (mom.y() > 0)
240                         posPTop.fill(mom.magnitude());
241                     else
242                         posPBottom.fill(mom.magnitude());
243                 }
244             }
245             // now, the photons
246             if (isPhoton) {
247                 if (fsCluster == null)
248                     throw new RuntimeException("isPhoton==true but no cluster found: should never happen");
249                 double ene = fsPart.getEnergy();
250                 // TODO: mg-May 14, 2014....I would like to do this!!!!
251                 // double xpos = fsCluster.getPositionAtShowerMax(false)[0];// false-->assume a photon instead of
252                 // electron from calculating shower depth
253                 // double ypos = fsCluster.getPositionAtShowerMax(false)[1];
254                 // but I can't because ReconParticles don't know about HPSEcalClusters, and casting it as one doesn't
255                 // seem to work
256                 Hep3Vector clusterPosition = new BasicHep3Vector(fsCluster.getPosition());
257                 double xpos = clusterPosition.x();
258                 double ypos = clusterPosition.y();
259                 nPhotons++;
260                 nTotPhotons++;
261                 enePhoton.fill(ene);
262                 xPhoton.fill(xpos);
263                 yPhoton.fill(ypos);
264             }
265 
266             if (hasCluster && !isPhoton) {
267                 if (fsCluster == null)
268                     throw new RuntimeException("hasCluster==true but no cluster found: should never happen");
269                 nTotAss++;
270                 Hep3Vector mom = fsPart.getMomentum();
271                 double ene = fsPart.getEnergy();
272                 double eOverP = ene / mom.magnitude();
273                 Hep3Vector clusterPosition = new BasicHep3Vector(fsCluster.getPosition());// this gets position at
274                 // shower max assuming it's an
275                 // electron/positron
276                 Hep3Vector trackPosAtEcal = TrackUtils.extrapolateTrack(fsTrack, clusterPosition.z());
277                 double dx = trackPosAtEcal.x() - clusterPosition.x();// remember track vs detector coords
278                 double dy = trackPosAtEcal.y() - clusterPosition.y();// remember track vs detector coords
279 
280                 sumdelX += dx;
281                 sumdelY += dy;
282                 sumEoverP += eOverP;
283 
284                 eneOverp.fill(eOverP);
285                 deltaXAtCal.fill(dx);
286                 deltaYAtCal.fill(dy);
287                 /* here are some plots for debugging track-cluster matching */
288                 // trackXvsECalX.fill(trackPosAtEcal.x(), clusterPosition.x());
289                 // trackYvsECalY.fill(trackPosAtEcal.y(), clusterPosition.y());
290                 trackPvsECalE.fill(fsPart.getMomentum().magnitude(), fsPart.getEnergy());
291                 trackTvsECalT.fill(ClusterUtilities.getSeedHitTime(fsCluster),
292                         TrackUtils.getTrackTime(fsTrack, hitToStrips, hitToRotated));
293                 timeMatchDeltaT.fill(ClusterUtilities.getSeedHitTime(fsCluster)
294                         - TrackUtils.getTrackTime(fsTrack, hitToStrips, hitToRotated));
295                 // if(dy<-20)
296                 // LOGGER.info("Big deltaY...")
297 
298             }
299             if (!hasCluster) {// if there is no cluster, can't be a track or else it wouldn't be in list
300                 nUnAssTracks++; // count per event
301                 nTotUnAss++; // and keep a running total for averaging
302             }
303         }
304         nUnAssTracksHisto.fill(nUnAssTracks);
305         nPhotonsHisto.fill(nPhotons);
306     }

319     @Override
320     public void calculateEndOfRunQuantities() {
321         IAnalysisFactory analysisFactory = IAnalysisFactory.create();
322         IFitFactory fitFactory = analysisFactory.createFitFactory();
323         IFitter fitter = fitFactory.createFitter("chi2");
324         IFitResult result = fitBeamEnergyPeak(elePzBeam, fitter, "range=\"(-10.0,10.0)\"");
325         if (result != null) {
326             double[] pars = result.fittedParameters();
327             for (int i = 0; i < 5; i++)
328                 LOGGER.info("Beam Energy Peak:  " + result.fittedParameterNames()[i] + " = " + pars[i]);
329             monitoredQuantityMap.put(finalStateParticlesColName + " " + triggerType + " " + fpQuantNames[7],
330                     (double) pars[1]);
331             monitoredQuantityMap.put(finalStateParticlesColName + " " + triggerType + " " + fpQuantNames[8],
332                     (double) pars[2]);
333         }
334         monitoredQuantityMap.put(finalStateParticlesColName + " " + triggerType + " " + fpQuantNames[0],
335                 (double) nTotEle / nRecoEvents);
336         monitoredQuantityMap.put(finalStateParticlesColName + " " + triggerType + " " + fpQuantNames[1],
337                 (double) nTotPos / nRecoEvents);
338         monitoredQuantityMap.put(finalStateParticlesColName + " " + triggerType + " " + fpQuantNames[2],
339                 (double) nTotPhotons / nRecoEvents);
340         monitoredQuantityMap.put(finalStateParticlesColName + " " + triggerType + " " + fpQuantNames[3],
341                 (double) nTotUnAss / nRecoEvents);
342         monitoredQuantityMap.put(finalStateParticlesColName + " " + triggerType + " " + fpQuantNames[4],
343                 (double) sumdelX / nTotAss);
344         monitoredQuantityMap.put(finalStateParticlesColName + " " + triggerType + " " + fpQuantNames[5],
345                 (double) sumdelY / nTotAss);
346         monitoredQuantityMap.put(finalStateParticlesColName + " " + triggerType + " " + fpQuantNames[6],
347                 (double) sumEoverP / nTotAss);
348 
349         IPlotter plotter = analysisFactory.createPlotterFactory().create("Beam Energy Electrons");
350 
351         IPlotterStyle pstyle = plotter.style();
352         pstyle.legendBoxStyle().setVisible(false);
353         pstyle.dataStyle().fillStyle().setColor("green");
354         pstyle.dataStyle().lineStyle().setColor("black");
355         plotter.region(0).plot(elePzBeam);
356         // plotter.region(0).plot(result.fittedFunction());
357         if (outputPlots)
358             try {
359                 plotter.writeToFile(outputPlotDir + "beamEnergyElectrons.png");
360             } catch (IOException ex) {
361                 Logger.getLogger(FinalStateMonitoring.class.getName()).log(Level.SEVERE, null, ex);
362             }
363     }

374 
375         // return fitter.fit(h1d, "g+p1", init, range);

380             ifr = fitter.fit(h1d, "g+p1", init);
381         } catch (RuntimeException ex) {

383         }
384
