142     public void process(EventHeader event) {
143 
144         //check to see if this event is from the correct trigger (or "all");
145         if (!matchTrigger(event))
146             return;
147 
148         /*  increment the strip occupancy arrays */
149         Map<String, Integer> hitsPerSensor = new HashMap<String, Integer>();
150 
151         if (event.hasCollection(RawTrackerHit.class, rawTrackerHitCollectionName)) {
152 //            LOGGER.info("Found a raw hit collection");
153             List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, rawTrackerHitCollectionName);
154             for (RawTrackerHit hit : rawTrackerHits) {
155                 short[] adcValues = hit.getADCValues();
156 
157                 // Find the sample that has the largest amplitude. This should
158                 // correspond to the peak of the shaper signal if the SVT is timed
159                 // in correctly. Otherwise, the maximum sample value will default
160                 // to 0.
161                 int maxAmplitude = 0;
162                 int maxSamplePositionFound = -1;
163                 for (int sampleN = 0; sampleN < 6; sampleN++)
164                     if (adcValues[sampleN] > maxAmplitude) {
165                         maxAmplitude = adcValues[sampleN];
166                         maxSamplePositionFound = sampleN;
167                     }
168                 if (maxSamplePosition == -1 || maxSamplePosition == maxSamplePositionFound)
169                     occupancyMap.get(hit.getDetectorElement().getName())[hit.getIdentifierFieldValue("strip")]++; //                System.out.println("Filling occupancy");
170 
171                 Integer nHits = hitsPerSensor.get(hit.getDetectorElement().getName());
172                 if (nHits == null)
173                     nHits = 0;
174                 nHits++;
175                 hitsPerSensor.put(hit.getDetectorElement().getName(), nHits);
176             }
177             ++eventCountRaw;
178         }
179         for (HpsSiSensor sensor : sensors) {
180             IHistogram1D sensorHist = PlotAndFitUtilities.getSensorPlot(plotDir + triggerType + "/" + "nHitsPerEvent_", sensor, true);
181             Integer nHits = hitsPerSensor.get(sensor.getName());
182             if (nHits == null)
183                 sensorHist.fill(0);
184             else
185                 sensorHist.fill(nHits);
186         }
187         /*  fill the FittedTrackerHit related histograms */
188         if (event.hasCollection(LCRelation.class, fittedTrackerHitCollectionName)) {
189             List<LCRelation> fittedTrackerHits = event.get(LCRelation.class, fittedTrackerHitCollectionName);
190 
191             RelationalTable rthtofit = new BaseRelationalTable(RelationalTable.Mode.ONE_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
192             for (LCRelation hit : fittedTrackerHits)
193                 rthtofit.add(FittedRawTrackerHit.getRawTrackerHit(hit), FittedRawTrackerHit.getShapeFitParameters(hit));
194 
195             for (LCRelation hit : fittedTrackerHits) {
196                 RawTrackerHit rth = (RawTrackerHit) hit.getFrom();
197                 GenericObject pars = (GenericObject) hit.getTo();
198 
199                 HpsSiSensor sensor = ((HpsSiSensor) rth.getDetectorElement());
200                 //this is a clever way to get the parameters we want from the generic object
201                 double t0 = ShapeFitParameters.getT0(pars);
202                 double amp = ShapeFitParameters.getAmp(pars) ;
203                 double chiProb = ShapeFitParameters.getChiProb(pars);
204                 int channel = rth.getIdentifierFieldValue("strip");
205                 PlotAndFitUtilities.getSensorPlot(plotDir + triggerType + "/" + "nFitsPerHit_", sensor, true).fill(rthtofit.allFrom(rth).size());
206                 PlotAndFitUtilities.getSensorPlot(plotDir + triggerType + "/" + "t0Hit_", sensor, true).fill(t0);
207                 PlotAndFitUtilities.getSensorPlot(plotDir + triggerType + "/" + "amplitude_", sensor, true).fill(amp);
208                 if (makeExtraPlots)
209                     PlotAndFitUtilities.getSensorPlot2D(plotDir + triggerType + "/" + "t0AmpHit_", sensor, true).fill(t0, amp);
210                 PlotAndFitUtilities.getSensorPlot(plotDir + triggerType + "/" + "chiProb_", sensor, true).fill(chiProb);
211                 if (makeExtraPlots)
212                     PlotAndFitUtilities.getSensorPlot2D(plotDir + triggerType + "/" + "ampChanHit_", sensor, true).fill(channel, amp);
213                 if (amp > minChargeCut && makeExtraPlots) {
214                     PlotAndFitUtilities.getSensorPlot2D(plotDir + triggerType + "/" + "t0ChanBigHit_", sensor, true).fill(channel, t0);
215                     PlotAndFitUtilities.getSensorPlot2D(plotDir + triggerType + "/" + "chiprobChanBigHit_", sensor, true).fill(channel, chiProb);
216                 }
217                 if (amp > minChargeCut)
218                     PlotAndFitUtilities.getSensorPlot2D(plotDir + triggerType + "/" + "t0BigHitTrigTime_", sensor, true).fill(t0, event.getTimeStamp() % 24);
219             }
220             ++eventCountFit;
221         }
222 
223         if (event.hasItem(trackerHitCollectionName)) {
224 //            LOGGER.info("Found a Si cluster collection");
225             List<TrackerHit> siClusters = (List<TrackerHit>) event.get(trackerHitCollectionName);
226             for (TrackerHit cluster : siClusters) {
227                 HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) cluster.getRawHits().get(0)).getDetectorElement();
228                 double t0 = cluster.getTime();
229                 double dedx = cluster.getdEdx() / DopedSilicon.ENERGY_EHPAIR;
230 //                LOGGER.info("dedx = "+dedx);
231                 PlotAndFitUtilities.getSensorPlot2D(plotDir + triggerType + "/" + "clusterPositionPlot_", sensor, true).fill(cluster.getPosition()[0], cluster.getPosition()[1]);
232                 PlotAndFitUtilities.getSensorPlot(plotDir + triggerType + "/" + "t0Cluster_", sensor, true).fill(t0);
233                 PlotAndFitUtilities.getSensorPlot2D(plotDir + triggerType + "/" + "t0ClusterTrigTime_", sensor, true).fill(t0, event.getTimeStamp() % 24);
234                 PlotAndFitUtilities.getSensorPlot(plotDir + triggerType + "/" + "charge_", sensor, true).fill(dedx);
235                 if (dedx > minChargeCut && Math.abs(t0)<clTime)
236                     PlotAndFitUtilities.getSensorPlot2D(plotDir + triggerType + "/" + "chargeVsY_", sensor, true).fill(Math.abs(cluster.getPosition()[1]), dedx);
237             }
238         }
239     }

241     private void resetOccupancyMap() {
242         occupancyMap = new HashMap<String, int[]>();
243         avgOccupancyMap = new HashMap<String, Double>();
244         avgOccupancyNames = new HashMap<String, String>();
245         avgt0Names = new HashMap<String, String>();
246         sigt0Names = new HashMap<String, String>();
247         avgt0Map = new HashMap<String, Double>();
248         sigt0Map = new HashMap<String, Double>();
249         for (HpsSiSensor sensor : sensors) {
250             occupancyMap.put(sensor.getName(), new int[640]);
251             avgOccupancyMap.put(sensor.getName(), -999.);
252             String occName = "avgOcc_" + getNiceSensorName(sensor);
253             avgOccupancyNames.put(sensor.getName(), occName);
254             String avgt0Name = "avgt0_" + getNiceSensorName(sensor);
255             String sigt0Name = "sigmat0_" + getNiceSensorName(sensor);
256 
257             avgt0Names.put(sensor.getName(), avgt0Name);
258             sigt0Names.put(sensor.getName(), sigt0Name);
259         }
260     }

262     private String getNiceSensorName(HpsSiSensor sensor) {
263         return sensor.getName().replaceAll(nameStrip, "")
264                 .replace("module", "mod")
265                 .replace("layer", "lyr")
266                 .replace("sensor", "sens");

276     @Override
277     public void fillEndOfRunPlots() {
278         // Plot strip occupancies.
279         LOGGER.info("SvtMonitoring::endOfData  filling occupancy plots");
280         for (HpsSiSensor sensor : sensors) {
281             Double avg = 0.0;
282             //IHistogram1D sensorHist = aida.histogram1D(sensor.getName());
283             IHistogram1D sensorHist = PlotAndFitUtilities.getSensorPlot(plotDir + triggerType + "/" + "occupancy_", sensor, true);
284             sensorHist.reset();
285             int[] strips = occupancyMap.get(sensor.getName());
286             for (int i = 0; i < strips.length; i++) {
287                 double stripOccupancy = (double) strips[i] / (double) (eventCountRaw);
288                 if (stripOccupancy != 0)
289                     sensorHist.fill(i, stripOccupancy / timeWindowWeight);
290                 avg += stripOccupancy;
291             }
292             //do the end-of-run quantities here too since we've already done the loop.  
293             avg /= strips.length;
294             avgOccupancyMap.put(sensor.getName(), avg);
295         }

299     @Override
300     public void calculateEndOfRunQuantities() {
301         IAnalysisFactory analysisFactory = IAnalysisFactory.create();
302         IFitFactory fitFactory = analysisFactory.createFitFactory();
303         IFitter fitter = fitFactory.createFitter("chi2");
304         IPlotter plotterTop = analysisFactory.createPlotterFactory().create("t0 Top");
305         IPlotter plotterBottom = analysisFactory.createPlotterFactory().create("t0 Bottom");
306 
307         plotterTop.createRegions(4, 5);
308         IPlotterStyle pstyle = plotterTop.style();
309         pstyle.legendBoxStyle().setVisible(false);
310         plotterBottom.createRegions(4, 5);
311         IPlotterStyle pstyle2 = plotterBottom.style();
312         pstyle2.legendBoxStyle().setVisible(false);
313 
314         int irTop = 0;
315         int irBot = 0;
316         for (HpsSiSensor sensor : sensors) {
317             IHistogram1D sensPlot = PlotAndFitUtilities.getSensorPlot(plotDir + triggerType + "/" + "t0Hit_", sensor, true);
318             IFitResult result = fitGaussian(sensPlot, fitter, "range=\"(-8.0,8.0)\"");
319 
320             boolean isTop = sensor.isTopLayer();
321             if (isTop) {
322                 plotterTop.region(irTop).plot(sensPlot);
323                 plotterTop.region(irTop).plot(result.fittedFunction());
324                 irTop++;
325             } else {
326                 plotterBottom.region(irBot).plot(sensPlot);
327                 plotterBottom.region(irBot).plot(result.fittedFunction());
328                 irBot++;
329             }
330             avgt0Map.put(sensor.getName(), result.fittedParameters()[1]);
331             sigt0Map.put(sensor.getName(), result.fittedParameters()[2]);
332         }
333         if (outputPlots) {
334             try {
335                 plotterTop.writeToFile(outputPlotDir + "t0TopPlots.png");
336             } catch (IOException ex) {
337                 Logger.getLogger(SvtMonitoring.class.getName()).log(Level.SEVERE, null, ex);
338             }
339             try {
340                 plotterBottom.writeToFile(outputPlotDir + "t0BottomPlots.png");
341             } catch (IOException ex) {
342                 Logger.getLogger(SvtMonitoring.class.getName()).log(Level.SEVERE, null, ex);
343             }
344         }
345     }

347     @Override
348     public void dumpDQMData() {
349         for (HpsSiSensor sensor : sensors) {
350             String name = avgOccupancyNames.get(sensor.getName());
351             double occ = avgOccupancyMap.get(sensor.getName());
352             checkAndUpdate(name, occ);
353             String avgt0Name = avgt0Names.get(sensor.getName());
354             double avgt0 = avgt0Map.get(sensor.getName());
355             checkAndUpdate(avgt0Name, avgt0);
356             String sigt0Name = sigt0Names.get(sensor.getName());
357             double sigt0 = sigt0Map.get(sensor.getName());
358             checkAndUpdate(sigt0Name, sigt0);
359         }
360     }

380     IFitResult fitGaussian(IHistogram1D h1d, IFitter fitter, String range) {
381 //        return fitter.fit(h1d, "g", range);
382         double[] init = {20.0, 0.0, 4.0, 20, -1};
383         return fitter.fit(h1d, "g+p1", init, range);

387     void checkAndUpdate(String name, double val) {
388         boolean isnull = false;
389         try {
390             isnull = checkSelectionIsNULL(name);
391         } catch (SQLException ex) {
392             Logger.getLogger(SvtMonitoring.class.getName()).log(Level.SEVERE, null, ex);
393         }
394         if (!overwriteDB && !isnull) {
395             LOGGER.info("Not writing because " + name + " is already filled for this entry");
396             return; //entry exists and I don't want to overwrite                
397         }
398         String put = "update dqm SET " + name + " = " + val + " WHERE " + getRunRecoString();
399         LOGGER.info(put);
400         manager.updateQuery(put);
401     }
