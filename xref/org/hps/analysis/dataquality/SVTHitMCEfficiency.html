94      @Override
95      public void process(EventHeader event) {
96  
97          aida.tree().cd("/");
98  
99          //make sure the required collections exist
100         if (!event.hasCollection(RawTrackerHit.class, rawTrackerHitCollectionName)) {
101             return;
102         }
103         if (!event.hasCollection(LCRelation.class, fittedTrackerHitCollectionName)) {
104             return;
105         }
106 
107         if (!event.hasCollection(TrackerHit.class, siClusterCollectionName)) {
108             return;
109         }
110 
111         if (!event.hasCollection(SimTrackerHit.class, trackerHitCollectionName)) {
112             return;
113         }
114 
115         if (!event.hasCollection(LCRelation.class, svtTrueHitRelationName)) {
116             return;
117         }
118 
119         RelationalTable mcHittomcP = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_ONE, RelationalTable.Weighting.UNWEIGHTED);
120         //  Get the collections of SimTrackerHits
121         List<List<SimTrackerHit>> simcols = event.get(SimTrackerHit.class);
122         //  Loop over the SimTrackerHits and fill in the relational table
123         for (List<SimTrackerHit> simlist : simcols) {
124             for (SimTrackerHit simhit : simlist) {
125                 if (simhit.getMCParticle() != null) {
126                     mcHittomcP.add(simhit, simhit.getMCParticle());
127                 }
128             }
129         }
130 
131         RelationalTable rawtomc = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
132         List<LCRelation> trueHitRelations = event.get(LCRelation.class, svtTrueHitRelationName);
133         for (LCRelation relation : trueHitRelations) {
134             if (relation.getFrom() != null && relation.getTo() != null) {
135                 rawtomc.add(relation.getFrom(), relation.getTo());
136             }
137         }
138 
139         List<SimTrackerHit> simHits = event.get(SimTrackerHit.class, trackerHitCollectionName);
140         // make relational table for strip clusters to mc particle
141         List<TrackerHit> siClusters = event.get(TrackerHit.class, siClusterCollectionName);
142         RelationalTable clustertosimhit = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
143         for (TrackerHit cluster : siClusters) {
144             for (RawTrackerHit rth : (List<RawTrackerHit>) cluster.getRawHits()) {
145                 Set<SimTrackerHit> simTrackerHits = rawtomc.allFrom(rth);
146                 if (simTrackerHits != null) {
147                     for (SimTrackerHit simhit : simTrackerHits) {
148                         if (simhit != null) {
149                             clustertosimhit.add(cluster, simhit);
150                         }
151                     }
152                 }
153             }
154         }
155 
156         //relational tables from raw and fitted tracker hits to sim hit
157         RelationalTable rthtofit = new BaseRelationalTable(RelationalTable.Mode.ONE_TO_ONE, RelationalTable.Weighting.UNWEIGHTED);
158         RelationalTable fittomc = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
159         List<LCRelation> fittedTrackerHits = event.get(LCRelation.class, fittedTrackerHitCollectionName);
160         for (LCRelation hit : fittedTrackerHits) {
161             GenericObject oldfit = (GenericObject) rthtofit.to(FittedRawTrackerHit.getRawTrackerHit(hit));
162             if (oldfit == null || Math.abs(ShapeFitParameters.getT0(oldfit)) > Math.abs(FittedRawTrackerHit.getT0(hit))) {
163                 rthtofit.add(FittedRawTrackerHit.getRawTrackerHit(hit), FittedRawTrackerHit.getShapeFitParameters(hit));
164             }
165             Set<SimTrackerHit> simTrackerHits = rawtomc.allFrom(FittedRawTrackerHit.getRawTrackerHit(hit));
166             for (SimTrackerHit simhit : simTrackerHits) {
167                 fittomc.add(hit, simhit);
168             }
169         }
170 
171         for (SimTrackerHit simhit : simHits) {
172             Set<LCRelation> fittedRTH = fittomc.allTo(simhit);
173             LCRelation signalHit = null;
174             for (LCRelation frth : fittedRTH) {
175                 if (signalHit == null || Math.abs(FittedRawTrackerHit.getT0(frth)) < Math.abs(FittedRawTrackerHit.getT0(signalHit))) {
176                     signalHit = frth;
177                 }
178             }
179             if (signalHit != null) {
180 //                System.out.format("chiprob %f, t0 %f, A %f\n", signalHit.getShapeFitParameters().getChiProb(), signalHit.getT0(), signalHit.getAmp());
181                 getLayerPlot2D(plotDir + "allFits", simhit.getLayer()).fill(FittedRawTrackerHit.getT0(signalHit), FittedRawTrackerHit.getAmp(signalHit));
182                 getLayerPlot2D(plotDir + "fitT0ChiProb", simhit.getLayer()).fill(FittedRawTrackerHit.getT0(signalHit), ShapeFitParameters.getChiProb(FittedRawTrackerHit.getShapeFitParameters(signalHit)));
183                 getLayerPlot2D(plotDir + "fitAmpChiProb", simhit.getLayer()).fill(FittedRawTrackerHit.getAmp(signalHit), ShapeFitParameters.getChiProb(FittedRawTrackerHit.getShapeFitParameters(signalHit)));
184 //                if (signalHit.getShapeFitParameters().getChiProb() > 0.95) {
185 //                    getLayerPlot2D(plotDir + "toogoodFits", simhit.getLayer()).fill(signalHit.getT0(), signalHit.getAmp());
186 //                } else if (signalHit.getShapeFitParameters().getChiProb() < 0.05) {
187 //                    getLayerPlot2D(plotDir + "badFits", simhit.getLayer()).fill(signalHit.getT0(), signalHit.getAmp());
188 //                } else {
189 //                    getLayerPlot2D(plotDir + "goodFits", simhit.getLayer()).fill(signalHit.getT0(), signalHit.getAmp());
190 //                }
191             }
192 
193             int gotCluster = 0;
194             int[] gotClusterAtTime = new int[16];
195             Set<TrackerHit> clusters = clustertosimhit.allTo(simhit);
196             if (clusters != null) {
197                 for (TrackerHit clust : clusters) {
198                     getLayerPlot1D(plotDir + "signalClusterT0", simhit.getLayer()).fill(clust.getTime());
199                     getLayerPlot2D(plotDir + "signalClusterSizeT0", simhit.getLayer()).fill(clust.getTime(), clust.getRawHits().size());
200 
201                     for (int i = 0; i < 16; i++) {
202                         if (Math.abs(clust.getTime()) < i + 1) {
203                             gotClusterAtTime[i] = 1;
204                         }
205                     }
206                     if (Math.abs(clust.getTime()) < t0Cut) {
207                         gotCluster = 1;
208                         for (RawTrackerHit rth : (List<RawTrackerHit>) clust.getRawHits()) {
209                             GenericObject fit = (GenericObject) rthtofit.to(rth);
210                             getLayerPlot2D(plotDir + "goodClusterFits", simhit.getLayer()).fill(ShapeFitParameters.getT0(fit), ShapeFitParameters.getAmp(fit));
211                         }
212                     } else {
213 //                        LOGGER.info(clust.getRawHits().size());
214                         for (RawTrackerHit rth : (List<RawTrackerHit>) clust.getRawHits()) {
215                             GenericObject fit = (GenericObject) rthtofit.to(rth);
216                             getLayerPlot2D(plotDir + "badClusterFits", simhit.getLayer()).fill(ShapeFitParameters.getT0(fit), ShapeFitParameters.getAmp(fit));
217                         }
218                     }
219 
220                 }
221             }
222             Set<RawTrackerHit> rawhits = rawtomc.allTo(simhit);
223             int gotRawHit = !rawhits.isEmpty() ? 1 : 0;
224 
225             double y = simhit.getDetectorElement().getGeometry().getGlobalToLocal().transformed(simhit.getPositionVec()).x() + 20.0 * Math.signum(simhit.getPoint()[1]) * (simhit.getLayer() % 2 == 0 ? -1 : 1);
226             getLayerPlot(plotDir + "clusterEfficiency", simhit.getLayer()).fill(y, gotCluster);
227             getLayerPlot(plotDir + "readoutEfficiency", simhit.getLayer()).fill(y, gotRawHit);
228             if (gotRawHit == 1) {
229                 getLayerPlot(plotDir + "rthToClusterEfficiency", simhit.getLayer()).fill(y, gotCluster);
230             }
231             for (int i = 0; i < 16; i++) {
232                 getLayerPlot2D(plotDir + "clusterEfficiency2D", simhit.getLayer()).fill(y, i + 1, gotClusterAtTime[i]);
233                 if (gotRawHit == 1) {
234                     getLayerPlot2D(plotDir + "rthToClusterEfficiency2D", simhit.getLayer()).fill(y, i + 1, gotClusterAtTime[i]);
235                 }
236             }
237         }
238     }

252     private void resetEfficiencyMap() {
253         avgClusterEffMap = new HashMap<String, Double>();
254         avgClusterEffNames = new HashMap<String, String>();
255         for (SiSensor sensor : sensors) {
256             String effName = "avgClusterEff_" + getNiceSensorName(sensor);
257             avgClusterEffNames.put(sensor.getName(), effName);
258         }
259     }

276     private void getMean2D(IHistogram2D hist2D) {
277         int nx = hist2D.xAxis().bins();
278         int ny = hist2D.yAxis().bins();
279         double[][] means = new double[nx][ny];
280         for (int ix = 0; ix < nx; ix++) {
281             for (int iy = 0; iy < ny; iy++) {
282                 means[ix][iy] = hist2D.binHeight(ix, iy) / hist2D.binEntries(ix, iy);
283             }
284         }
285         hist2D.reset();
286         for (int ix = 0; ix < nx; ix++) {
287             for (int iy = 0; iy < ny; iy++) {
288                 double x = hist2D.xAxis().binCenter(ix);
289                 double y = hist2D.yAxis().binCenter(iy);
290                 hist2D.fill(x, y, means[ix][iy]);
291             }
292         }
293     }
