170     @Override
171     public void process(EventHeader event) {
172         aida.tree().cd("/");
173         if (!event.hasCollection(HelicalTrackHit.class, helicalTrackHitCollectionName))
174             return;
175 
176         if (!event.hasCollection(Track.class, l1to3CollectionName))
177             return;
178 
179         if (!event.hasCollection(Track.class, l4to6CollectionName))
180             return;
181 
182         List<Track> l1to3tracks = event.get(Track.class, l1to3CollectionName);
183         List<Track> l4to6tracks = event.get(Track.class, l4to6CollectionName);
184 
185         List<Track> l1to3tracksTop = splitTrackList(l1to3tracks, true);
186         List<Track> l1to3tracksBot = splitTrackList(l1to3tracks, false);
187         List<Track> l4to6tracksTop = splitTrackList(l4to6tracks, true);
188         List<Track> l4to6tracksBot = splitTrackList(l4to6tracks, false);
189 
190         List<Track> l1to3tracksEle = splitByCharge(l1to3tracks, -1);
191         List<Track> l1to3tracksPos = splitByCharge(l1to3tracks, 1);
192         List<Track> l4to6tracksEle = splitByCharge(l4to6tracks, -1);
193         List<Track> l4to6tracksPos = splitByCharge(l4to6tracks, 1);
194 
195         nTracks13Top.fill(l1to3tracksTop.size());
196         nTracks13Bot.fill(l1to3tracksBot.size());
197         nTracks46Top.fill(l4to6tracksTop.size());
198         nTracks46Bot.fill(l4to6tracksBot.size());
199 
200         nTracks13Ele.fill(l1to3tracksEle.size());
201         nTracks13Pos.fill(l1to3tracksPos.size());
202         nTracks46Ele.fill(l4to6tracksEle.size());
203         nTracks46Pos.fill(l4to6tracksPos.size());
204 
205         int ncombotop = 0;
206         int ncombobot = 0;
207 
208         for (Track trk46 : l4to6tracksTop) {
209             TrackState ts46 = trk46.getTrackStates().get(0);
210             for (Track trk13 : l1to3tracksTop)
211                 if (!onlyFEEStubs || (trk46.getMomentum()[0] > pcut && trk13.getMomentum()[0] > pcut)) {
212                     TrackState ts13 = trk13.getTrackStates().get(0);
213                     deld0Top.fill(ts46.getD0() - ts13.getD0());
214                     delphiTop.fill(Math.sin(ts46.getPhi()) - Math.sin(ts13.getPhi()));
215                     delwTop.fill(ts46.getOmega() - ts13.getOmega());
216                     delz0Top.fill(ts46.getZ0() - ts13.getZ0());
217                     dellambdaTop.fill(ts46.getTanLambda() - ts13.getTanLambda());
218                     ncombotop++;
219                 }
220         }
221 
222         for (Track trk46 : l4to6tracksBot) {
223             TrackState ts46 = trk46.getTrackStates().get(0);
224             for (Track trk13 : l1to3tracksBot)
225                 if (!onlyFEEStubs || (trk46.getMomentum()[0] > pcut && trk13.getMomentum()[0] > pcut)) {
226                     TrackState ts13 = trk13.getTrackStates().get(0);
227                     deld0Bot.fill(ts46.getD0() - ts13.getD0());
228                     delphiBot.fill(Math.sin(ts46.getPhi()) - Math.sin(ts13.getPhi()));
229                     delwBot.fill(ts46.getOmega() - ts13.getOmega());
230                     delz0Bot.fill(ts46.getZ0() - ts13.getZ0());
231                     dellambdaBot.fill(ts46.getTanLambda() - ts13.getTanLambda());
232                     ncombobot++;
233                 }
234         }
235 
236         nCombosTop.fill(ncombotop);
237         nCombosBot.fill(ncombobot);
238 
239         for (Track trk46 : l4to6tracksEle) {
240             TrackState ts46 = trk46.getTrackStates().get(0);
241             for (Track trk13 : l1to3tracksEle) {
242                 TrackState ts13 = trk13.getTrackStates().get(0);
243                 deld0Ele.fill(ts46.getD0() - ts13.getD0());
244                 delphiEle.fill(Math.sin(ts46.getPhi()) - Math.sin(ts13.getPhi()));
245                 delwEle.fill(ts46.getOmega() - ts13.getOmega());
246                 delz0Ele.fill(ts46.getZ0() - ts13.getZ0());
247                 dellambdaEle.fill(ts46.getTanLambda() - ts13.getTanLambda());
248                 d0Ele.fill(ts46.getD0(), ts13.getD0());
249                 phiEle.fill(Math.sin(ts46.getPhi()), Math.sin(ts13.getPhi()));
250                 wEle.fill(ts46.getOmega(), ts13.getOmega());
251                 lambdaEle.fill(ts46.getTanLambda(), ts13.getTanLambda());
252                 z0Ele.fill(ts46.getZ0(), ts13.getZ0());
253 
254             }
255         }
256 
257         for (Track trk46 : l4to6tracksPos) {
258             TrackState ts46 = trk46.getTrackStates().get(0);
259             for (Track trk13 : l1to3tracksPos) {
260                 TrackState ts13 = trk13.getTrackStates().get(0);
261                 deld0Pos.fill(ts46.getD0() - ts13.getD0());
262                 delphiPos.fill(Math.sin(ts46.getPhi()) - Math.sin(ts13.getPhi()));
263                 delwPos.fill(ts46.getOmega() - ts13.getOmega());
264                 delz0Pos.fill(ts46.getZ0() - ts13.getZ0());
265                 dellambdaPos.fill(ts46.getTanLambda() - ts13.getTanLambda());
266                 d0Pos.fill(ts46.getD0(), ts13.getD0());
267                 phiPos.fill(Math.sin(ts46.getPhi()), Math.sin(ts13.getPhi()));
268                 wPos.fill(ts46.getOmega(), ts13.getOmega());
269                 lambdaPos.fill(ts46.getTanLambda(), ts13.getTanLambda());
270                 z0Pos.fill(ts46.getZ0(), ts13.getZ0());
271             }
272         }
273 
274         /*
275          * l1to3tracksPos = null;
276          * l1to3tracksEle = null;
277          * l1to3tracksTop = null;
278          * l1to3tracksBot = null;
279          * 
280          * l4to6tracksPos = null;
281          * l4to6tracksEle = null;
282          * l4to6tracksTop = null;
283          * l4to6tracksBot = null;
284          */
285     }

292     private List<Track> splitTrackList(List<Track> trks, boolean doTop) {
293         List<Track> tracksHalf = new ArrayList<Track>();
294         boolean isTop = false;
295         boolean isBot = false;
296         for (Track trk : trks) {
297             isTop = false;
298             isBot = false;
299             for (TrackerHit hit : trk.getTrackerHits())
300                 if (hit.getPosition()[2] > 0)// remember, non-bend in tracking frame is z-direction
301                     isTop = true;
302                 else
303                     isBot = true;
304             if (isTop == true && isBot != true && doTop == true)  // if we want top tracks and all hits are in top
305                 tracksHalf.add(trk);
306             if (isBot == true && isTop != true && doTop == false) // if we want bottom tracks and all hits are in bottom
307                 tracksHalf.add(trk);
308         }
309         return tracksHalf;
310     }

312     private List<Track> splitByCharge(List<Track> trks, int charge) {
313         List<Track> tracksHalf = new ArrayList<>();
314         boolean isTop = false;
315         boolean isBot = false;
316         for (Track trk : trks) {
317             isTop = false;
318             isBot = false;
319             if (!(trk.getCharge() == charge)) { // XNOR but remember that the track charge is opposite because of
320                                                 // B-field definition...
321                 for (TrackerHit hit : trk.getTrackerHits())
322                     if (hit.getPosition()[2] > 0)// remember, non-bend in tracking frame is z-direction
323                         isTop = true;
324                     else
325                         isBot = true;
326                 if (isTop == true && isBot != true)  // if all hits are in top
327                     tracksHalf.add(trk);
328                 if (isBot == true && isTop != true) // if all hits are in bottom
329                     tracksHalf.add(trk);
330             }
331         }
332         return tracksHalf;
333     }
