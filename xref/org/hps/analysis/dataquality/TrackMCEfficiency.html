116     @Override
117     public void process(EventHeader event) {
118 
119         aida.tree().cd("/");
120 
121         //make sure the required collections exist
122         if (!event.hasCollection(RawTrackerHit.class, rawTrackerHitCollectionName)) {
123             if (debug)
124                 LOGGER.info(this.getClass().getSimpleName() + ": no collection found " + rawTrackerHitCollectionName);
125             return;
126         }
127         if (!event.hasCollection(LCRelation.class, fittedSVTHitCollectionName))
128             if (debug)
129                 LOGGER.info(this.getClass().getSimpleName() + ": no collection found " + fittedSVTHitCollectionName); //mg...2/1/2015...don't return if the fitted collection isn't there...
130         //allow us to run if we simulated in "simple" mode (i.e. no time evolution)
131         //            return;
132         if (!event.hasCollection(Track.class, trackCollectionName)) {
133             if (debug)
134                 LOGGER.info(this.getClass().getSimpleName() + ": no collection found " + trackCollectionName);
135             return;
136         }
137         if (!event.hasCollection(LCRelation.class, trackHitMCRelationsCollectionName)) {
138             if (debug)
139                 LOGGER.info(this.getClass().getSimpleName() + ": no collection found " + trackHitMCRelationsCollectionName);
140             return;
141         }
142         if (!event.hasCollection(TrackerHit.class, siClusterCollectionName)) {
143             if (debug)
144                 LOGGER.info(this.getClass().getSimpleName() + ": no collection found " + siClusterCollectionName);
145             return;
146         }
147 
148         if (!event.hasCollection(SimTrackerHit.class, trackerHitCollectionName)) {
149             if (debug)
150                 LOGGER.info(this.getClass().getSimpleName() + ": no collection found " + trackerHitCollectionName);
151             return;
152         }
153         //
154         //get the b-field
155         Hep3Vector IP = new BasicHep3Vector(0., 0., 1.);
156         double bfield = event.getDetector().getFieldMap().getField(IP).y();
157         //make some maps and relation tables        
158         Map<Track, TrackAnalysis> tkanalMap = new HashMap<Track, TrackAnalysis>();
159         RelationalTable hittomc = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
160         List<LCRelation> mcrelations = event.get(LCRelation.class, trackHitMCRelationsCollectionName);
161         if (debugTrackEfficiency)
162             LOGGER.info(this.getClass().getSimpleName() + ": number of MC relations = " + mcrelations.size());
163         for (LCRelation relation : mcrelations)
164             if (relation != null && relation.getFrom() != null && relation.getTo() != null)
165                 hittomc.add(relation.getFrom(), relation.getTo());
166         if (debugTrackEfficiency)
167             LOGGER.info(this.getClass().getSimpleName() + ": number of hittomc relations = " + hittomc.size());
168         RelationalTable mcHittomcP = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
169         //  Get the collections of SimTrackerHits
170         List<List<SimTrackerHit>> simcols = event.get(SimTrackerHit.class);
171         //  Loop over the SimTrackerHits and fill in the relational table
172         for (List<SimTrackerHit> simlist : simcols)
173             for (SimTrackerHit simhit : simlist)
174                 if (simhit.getMCParticle() != null)
175                     mcHittomcP.add(simhit, simhit.getMCParticle());
176         RelationalTable trktomc = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
177         RelationalTable rawtomc = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
178         if (event.hasCollection(LCRelation.class, "SVTTrueHitRelations")) {
179             List<LCRelation> trueHitRelations = event.get(LCRelation.class, "SVTTrueHitRelations");
180             for (LCRelation relation : trueHitRelations)
181                 if (relation != null && relation.getFrom() != null && relation.getTo() != null)
182                     rawtomc.add(relation.getFrom(), relation.getTo());
183         }
184         // make relational table for strip clusters to mc particle
185         List<TrackerHit> siClusters = event.get(TrackerHit.class, siClusterCollectionName);
186         RelationalTable clustertosimhit = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
187         for (TrackerHit cluster : siClusters) {
188             List<RawTrackerHit> rawHits = cluster.getRawHits();
189             for (RawTrackerHit rth : rawHits) {
190                 Set<SimTrackerHit> simTrackerHits = rawtomc.allFrom(rth);
191                 if (simTrackerHits != null)
192                     for (SimTrackerHit simhit : simTrackerHits)
193                         clustertosimhit.add(cluster, simhit);
194             }
195         }
196         //relational tables from mc particle to raw and fitted tracker hits
197         RelationalTable fittomc = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
198         if (event.hasCollection(LCRelation.class, fittedSVTHitCollectionName)) {
199             List<LCRelation> fittedTrackerHits = event.get(LCRelation.class, fittedSVTHitCollectionName);
200             for (LCRelation hit : fittedTrackerHits) {
201                 RawTrackerHit rth = FittedRawTrackerHit.getRawTrackerHit(hit);
202                 Set<SimTrackerHit> simTrackerHits = rawtomc.allFrom(rth);
203                 if (simTrackerHits != null)
204                     for (SimTrackerHit simhit : simTrackerHits)
205                         if (simhit.getMCParticle() != null)
206                             fittomc.add(hit, simhit.getMCParticle());
207             }
208         }
209         RelationalTable hittostrip = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
210         List<LCRelation> hitrelations = event.get(LCRelation.class, detectorFrameHitRelationsCollectionName);
211         for (LCRelation relation : hitrelations)
212             if (relation != null && relation.getFrom() != null && relation.getTo() != null)
213                 hittostrip.add(relation.getFrom(), relation.getTo());
214 
215         RelationalTable hittorotated = new BaseRelationalTable(RelationalTable.Mode.ONE_TO_ONE, RelationalTable.Weighting.UNWEIGHTED);
216         List<LCRelation> rotaterelations = event.get(LCRelation.class, trackHitRelationsCollectionName);
217         for (LCRelation relation : rotaterelations)
218             if (relation != null && relation.getFrom() != null && relation.getTo() != null)
219                 hittorotated.add(relation.getFrom(), relation.getTo());
220 
221         //  Instantiate the class that determines if a track is "findable"
222         FindableTrack findable = new FindableTrack(event);
223 
224         List<Track> tracks = event.get(Track.class, trackCollectionName);
225         if (debugTrackEfficiency)
226             LOGGER.info(this.getClass().getSimpleName() + ": nTracks = " + tracks.size());
227         for (Track trk : tracks) {
228             TrackAnalysis tkanal = new TrackAnalysis(trk, hittomc, rawtomc, hittostrip, hittorotated);
229             tkanalMap.put(trk, tkanal);
230             MCParticle mcp = tkanal.getMCParticleNew();
231             if (mcp != null) {//  Create a map between the tracks found and the assigned MC particle            
232                 if (debugTrackEfficiency)
233                     LOGGER.info(this.getClass().getSimpleName() + ": found MCP match");
234                 trktomc.add(trk, tkanal.getMCParticleNew());
235             }
236         }
237 
238         //  Now loop over all MC Particles
239         List<MCParticle> mclist = event.getMCParticles();
240         int _nchMCP = 0;
241         int _nchMCPBar = 0;
242         for (MCParticle mcp : mclist) {
243 
244             //  Calculate the pT and polar angle of the MC particle
245             double px = mcp.getPX();
246             double py = mcp.getPY();
247             double pz = mcp.getPZ();
248             double pt = Math.sqrt(px * px + py * py);
249             double p = Math.sqrt(pt * pt + pz * pz);
250             double cth = py / p;
251             double theta = 180. * Math.acos(cth) / Math.PI;
252             double eta = -Math.log(Math.tan(Math.atan2(pt, pz) / 2));
253             double phi = Math.atan2(px, pz);
254             //  Find the number of layers hit by this mc particle
255             if (debugTrackEfficiency)
256                 LOGGER.info("MC pt=" + pt);
257             int nhits = findable.LayersHit(mcp);
258             boolean isFindable = findable.InnerTrackerIsFindable(mcp, nlayers - 2);
259             if (debugTrackEfficiency)
260                 LOGGER.info("nhits Findable =" + nhits + "; is findable? " + isFindable);
261             //  Calculate the helix parameters for this MC particle
262             HelixParamCalculator helix = new HelixParamCalculator(mcp, bfield);
263             double d0 = helix.getDCA();
264             double z0 = helix.getZ0();
265 
266             //  Check cases where we have multiple tracks associated with this MC particle
267             Set<Track> trklist = trktomc.allTo(mcp);
268             int ntrk = trklist.size();
269 
270 //            Set<Track> trklistAxial = trktomcAxial.allTo(mcp);
271 //            int ntrkAxial = trklistAxial.size();
272             if (mcp.getPDGID() == 622) {
273                 boolean bothreco = true;
274                 boolean bothfindable = true;
275                 //it's the A'...let's see if we found both tracks.
276                 List<MCParticle> daughters = mcp.getDaughters();
277                 for (MCParticle d : daughters) {
278                     if (trktomc.allTo(d).isEmpty())
279                         bothreco = false;
280                     if (!findable.InnerTrackerIsFindable(d, nlayers - 2))
281                         bothfindable = false;
282                 }
283                 double vtxWgt = 0;
284                 if (bothreco)
285                     vtxWgt = 1.0;
286 //                VxEff.fill(mcp.getOriginX(), vtxWgt);
287 //                VyEff.fill(mcp.getOriginY(), vtxWgt);
288 //                VzEff.fill(mcp.getOriginZ(), vtxWgt);
289                 if (bothfindable) {
290 //                    VxEffFindable.fill(mcp.getOriginX(), vtxWgt);
291 //                    VyEffFindable.fill(mcp.getOriginY(), vtxWgt);
292 //                    VzEffFindable.fill(mcp.getOriginZ(), vtxWgt);
293                 }
294             }
295 
296 //            if (nhits == nlayers[0]) {
297             if (isFindable) {
298                 _nchMCP++;
299                 findableTracks++;
300                 double wgt = 0.;
301                 if (ntrk > 0)
302                     wgt = 1.;
303                 foundTracks += wgt;
304                 if (debugTrackEfficiency)
305                     LOGGER.info("...is findable; filling plots with weight " + wgt);
306                 peffFindable.fill(p, wgt);
307                 phieffFindable.fill(phi, wgt);
308                 ctheffFindable.fill(cth, wgt);
309 
310                 if (wgt == 0) {
311                     Set<SimTrackerHit> mchitlist = mcHittomcP.allTo(mcp);
312                     Set<HelicalTrackCross> hitlist = hittomc.allTo(mcp);
313                     if (debugTrackEfficiency)
314                         if (fittomc != null) {
315                             Set<FittedRawTrackerHit> fitlist = fittomc.allTo(mcp);
316                             LOGGER.info(this.getClass().getSimpleName() + ":  Missed a findable track with MC p = " + p);
317                             if (!hasHTHInEachLayer(hitlist, fitlist))
318                                 LOGGER.info("\t\tThis track failed becasue it's missing a helical track hit");
319                         }
320                 }
321 
322             }
323             if (debugTrackEfficiency){
324                 LOGGER.info("# of mc parents " + mcp.getParents().size());
325                 if(mcp.getParents().size() > 0 )
326                     LOGGER.info("PDG ID of parent 0 is " + mcp.getParents().get(0).getPDGID());
327             }
328             if (mcp.getParents().size() == 1 && mcp.getParents().get(0).getPDGID() == 622) {
329                 totelectrons++;
330 //                    findableelectrons++;
331                 double wgt = 0.;
332                 if (ntrk > 0)
333                     wgt = 1.;
334                 if (debugTrackEfficiency)
335                     LOGGER.info("...is from A'; filling plots with weight " + wgt);
336                 foundelectrons += wgt;
337                 peffElectrons.fill(p, wgt);
338                 phieffElectrons.fill(phi, wgt);
339                 ctheffElectrons.fill(cth, wgt);
340                 if (ntrk == 1) {
341                     Track trk = (Track) trklist.toArray()[0];
342                     TrackState ts = trk.getTrackStates().get(0);
343                     double deld0 = d0 - ts.getD0();
344                     double delz0 = z0 - ts.getZ0();
345                     double delp = p - calcMagnitude(ts.getMomentum());
346                     double delphi0 = phi - ts.getPhi();
347                     double deltanlam = cth - ts.getTanLambda();
348                     aida.histogram1D(resDir + "Momentum Resolution").fill(delp);
349                     aida.histogram1D(resDir + "z0 Resolution").fill(delz0);
350                     aida.histogram1D(resDir + "d0 Resolution").fill(deld0);
351                     aida.histogram1D(resDir + "phi0 Resolution").fill(delphi0);
352                     aida.histogram1D(resDir + "tanLambda Resolution").fill(deltanlam);
353                 }
354 
355                 //               }
356             }
357         }
358     }

377     private boolean hasHTHInEachLayer(Set<HelicalTrackCross> list, Set<FittedRawTrackerHit> fitlist) {
378         if (list.isEmpty())
379             return false;
380         if (!(list.toArray()[0] instanceof HelicalTrackCross))
381             return false;
382         for (int layer = 1; layer < nlayers - 2; layer += 2) {
383             boolean hasThisLayer = false;
384             for (HelicalTrackCross hit : list)
385                 if (hit.Layer() == layer)
386                     hasThisLayer = true;
387             if (!hasThisLayer) {
388 //                LOGGER.info("Missing reconstructed hit in layer = " + layer);
389                 boolean hasFitHitSL1 = false;
390                 boolean hasFitHitSL2 = false;
391                 FittedRawTrackerHit fitSL1 = null;
392                 FittedRawTrackerHit fitSL2 = null;
393 //                LOGGER.info("fitted hit list size = " + fitlist.size());
394                 for (FittedRawTrackerHit fit : fitlist) {
395 //                    LOGGER.info("fitted hit layer number = " + fit.getRawTrackerHit().getLayerNumber());
396                     if (fit.getRawTrackerHit().getLayerNumber() == layer) {
397                         hasFitHitSL1 = true;
398                         fitSL1 = fit;
399 //                        LOGGER.info("Found a hit in SL1 with t0 = " + fitSL1.getT0() + "; amp = " + fitSL1.getAmp() + "; chi^2 = " + fitSL1.getShapeFitParameters().getChiProb() + "; strip = " + fitSL1.getRawTrackerHit().getCellID());
400                     }
401                     if (fit.getRawTrackerHit().getLayerNumber() == layer + 1) {
402                         hasFitHitSL2 = true;
403                         fitSL2 = fit;
404 //                        LOGGER.info("Found a hit in SL2 with t0 = " + fitSL2.getT0() + "; amp = " + fitSL2.getAmp() + "; chi^2 = " + fitSL2.getShapeFitParameters().getChiProb() + "; strip = " + fitSL2.getRawTrackerHit().getCellID());
405 
406                     }
407                 }
408 //                if (!hasFitHitSL1)
409 //                    LOGGER.info("MISSING a hit in SL1!!!");
410 //                if (!hasFitHitSL2)
411 //                    LOGGER.info("MISSING a hit in SL2!!!");
412 
413                 return false;
414             }
415         }
416         return true;
417     }

419     private double calcMagnitude(double[] vec) {
420         return Math.sqrt(vec[0] * vec[0] + vec[1] * vec[1] + vec[2] * vec[2]);
421     }
