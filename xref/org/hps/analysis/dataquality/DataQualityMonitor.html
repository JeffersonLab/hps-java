52      @Override
53      protected void detectorChanged(Detector detector) {
54          BeamEnergyCollection beamEnergyCollection = this.getConditionsManager()
55                  .getCachedConditions(BeamEnergyCollection.class, "beam_energies").getCachedData();
56          if (beamEnergy == null && beamEnergyCollection != null && beamEnergyCollection.size() != 0)
57              beamEnergy = beamEnergyCollection.get(0).getBeamEnergy();
58          else {
59              LOGGER.log(Level.WARNING, "warning:  beam energy not found.  Using a 6.6 GeV as the default energy");
60              beamEnergy = 6.6;
61          }
62          this.runNumber = this.getConditionsManager().getRun();
63          is2019Run = DatabaseConditionsManager.isPhys2019Run(this.getConditionsManager().getRun());
64      }

105     @Override
106     public void endOfData() {
107         calculateEndOfRunQuantities();
108         fillEndOfRunPlots();
109         printDQMData();
110         if (printDQMStrings)
111             printDQMStrings();
112         LOGGER.info("Write to database =  " + connectToDB);
113         if (connectToDB) {
114             LOGGER.info("Connecting To Database...getting DQMDBManager");
115             manager = DQMDatabaseManager.getInstance();
116             // check to see if I need to make a new db entry
117             boolean entryExists = false;
118             try {
119                 entryExists = checkRowExists();
120                 if (entryExists)
121                     LOGGER.info("Found an existing run/reco entry in the dqm database; overwrite = " + overwriteDB);
122             } catch (SQLException ex) {
123                 Logger.getLogger(DataQualityMonitor.class.getName()).log(Level.SEVERE, null, ex);
124             }
125 
126             if (!entryExists)
127                 makeNewRow();
128             dumpDQMData();
129         }
130     }

132     private void makeNewRow() {
133         LOGGER.info("is the data base connected?  " + manager.isConnected);
134         if (manager.isConnected) {
135             String ins = "insert into dqm SET runNumber=" + runNumber;
136             // LOGGER.info(ins);
137             manager.updateQuery(ins);
138             ins = "update  dqm SET recoVersion='" + recoVersion + "' where runNumber=" + runNumber;
139             manager.updateQuery(ins);
140         }
141         LOGGER.info("Made a new row for runNumber=" + runNumber + "; recoVersion=" + recoVersion);
142     }

144     private boolean checkRowExists() throws SQLException {
145         String ins = "select * from dqm where " + getRunRecoString();
146         ResultSet res = manager.selectQuery(ins);
147         return res.next(); // this is a funny way of determining if the ResultSet has any entries
148     }

150     public boolean checkSelectionIsNULL(String var) throws SQLException {
151         String ins = "select " + var + " from dqm where " + getRunRecoString();
152         ResultSet res = manager.selectQuery(ins);
153         res.next();
154         double result = res.getDouble(var);
155         if (res.wasNull())
156             return true;
157         LOGGER.info("checkSelectionIsNULL::" + var + " = " + result);
158         return false;
159     }

175     public void dumpDQMData() {
176         for (Map.Entry<String, Double> entry : monitoredQuantityMap.entrySet()) {
177             String name = entry.getKey();
178             double val = entry.getValue();
179             boolean isnull = false;
180             try {
181                 isnull = checkSelectionIsNULL(name);
182             } catch (SQLException ex) {
183                 Logger.getLogger(SvtMonitoring.class.getName()).log(Level.SEVERE, null, ex);
184             }
185             if (!overwriteDB && !isnull) {
186                 LOGGER.info("Not writing because " + name + " is already filled for this entry");
187                 continue; // entry exists and I don't want to overwrite
188             }
189             String put = "update dqm SET " + name + " = " + val + " WHERE " + getRunRecoString();
190             LOGGER.info(put);
191             manager.updateQuery(put);
192 
193         }
194     }

200     public boolean matchTriggerType(TIData triggerData) {
201 //        System.out.println("matchTriggerType:: "+triggerType+" run number = "+runNumber);
202         if (triggerType.contentEquals("") || triggerType.contentEquals("all"))
203             return true;
204         if (triggerData.isSingle0Trigger() && triggerType.contentEquals("singles0"))
205             return true;
206         if (triggerData.isSingle1Trigger() && triggerType.contentEquals("singles1"))
207             return true;
208         if (triggerData.isPair0Trigger() && triggerType.contentEquals("pairs0"))
209             return true;
210         if (triggerData.isPair1Trigger() && triggerType.contentEquals("pairs1"))
211             return true;
212         return false;
213 
214     }

219             return true;
220         if (triggerData.isSingle0TopTrigger() && triggerType.contentEquals("singles0Top"))

279     public boolean matchTrigger(EventHeader event) {
280         boolean match = true;
281         if (!is2019Run) {
282             if (event.hasCollection(GenericObject.class, "TriggerBank")) {
283                 List<GenericObject> triggerList = event.get(GenericObject.class, "TriggerBank");
284                 for (GenericObject data : triggerList)
285                     if (AbstractIntData.getTag(data) == TIData.BANK_TAG) {
286                         TIData triggerData = new TIData(data);
287                         if (!matchTriggerType(triggerData))                  
288                             match = false;
289                     }
290             }
291         } else if (is2019Run) {
292             if (event.hasCollection(GenericObject.class, "TSBank")) {
293                 List<GenericObject> triggerList = event.get(GenericObject.class, "TSBank");
294                 for (GenericObject data : triggerList)
295                     if (AbstractIntData.getTag(data) == TSData2019.BANK_TAG) {
296                         TSData2019 triggerData = new TSData2019(data);
297                         if (!matchTriggerType2019(triggerData))                    
298                             match = false;                       
299                     }
300             }
301         } else if (debug)
302             LOGGER.info(this.getClass().getSimpleName() + ":  No trigger bank found...running over all trigger types");
303         return match;
304     }
