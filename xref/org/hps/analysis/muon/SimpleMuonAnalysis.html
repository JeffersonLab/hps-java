9   import org.lcsim.event.CalorimeterHit;
10  import org.lcsim.event.EventHeader;
11  import org.lcsim.geometry.Detector;
12  import org.lcsim.units.clhep.SystemOfUnits;
13  import org.lcsim.util.Driver;
14  import org.lcsim.util.aida.AIDA;
15  
16  public class SimpleMuonAnalysis extends Driver 
17  {
18      AIDA aida = AIDA.defaultInstance();
19      ICloud1D hite = aida.cloud1D("Hit Energy [MeV]");
20      ICloud1D nhits = aida.cloud1D("Number Of Hits Per Event");    
21      ICloud2D xy = aida.cloud2D("Hit Position XY");
22      IHistogram1D layer = aida.histogram1D("Hit Layer Number", 4, 0, 4);
23      
24      public void startOfData() 
25      {
26      }
27      
28      public void endOfData() 
29      {        
30      }
31      
32      public void detectorChanged(Detector det) 
33      {        
34      }
35      
36      public void process(EventHeader event) 
37      {
38          List<CalorimeterHit> hits = event.get(CalorimeterHit.class, "MUON_HITS");
39          nhits.fill(hits.size());
40          for (CalorimeterHit hit : hits) {
41              xy.fill(hit.getPosition()[0], hit.getPosition()[1]);
42              hite.fill(hit.getRawEnergy() * SystemOfUnits.MeV);
43              layer.fill(hit.getLayerNumber());
44          }
45      }
46  }
