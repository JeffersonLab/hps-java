439     public boolean isFiducial(CalorimeterHit hit) {
440         int ix = hit.getIdentifierFieldValue("ix");
441         int iy = hit.getIdentifierFieldValue("iy");
442         // Get the x and y indices for the cluster.
443         int absx = Math.abs(ix);
444         int absy = Math.abs(iy);
445 
446         // Check if the cluster is on the top or the bottom of the
447         // calorimeter, as defined by |y| == 5. This is an edge cluster
448         // and is not in the fiducial region.
449         if (absy == 5) {
450             return false;
451         }
452 
453         // Check if the cluster is on the extreme left or right side
454         // of the calorimeter, as defined by |x| == 23. This is also
455         // an edge cluster and is not in the fiducial region.
456         if (absx == 23) {
457             return false;
458         }
459 
460         // Check if the cluster is along the beam gap, as defined by
461         // |y| == 1. This is an internal edge cluster and is not in the
462         // fiducial region.
463         if (absy == 1) {
464             return false;
465         }
466 
467         // Lastly, check if the cluster falls along the beam hole, as
468         // defined by clusters with -11 <= x <= -1 and |y| == 2. This
469         // is not the fiducial region.
470         if (absy == 2 && ix <= -1 && ix >= -11) {
471             return false;
472         }
473 
474         // If all checks fail, the cluster is in the fiducial region.
475         return true;
476     }

495     private boolean isTopTrack(Track t) {
496         List<TrackerHit> hits = t.getTrackerHits();
497         int n[] = {0, 0};
498         int nHits = hits.size();
499         for (TrackerHit h : hits) {
500             HpsSiSensor sensor = ((HpsSiSensor) ((RawTrackerHit) h.getRawHits().get(0)).getDetectorElement());
501             if (sensor.isTopLayer()) {
502                 n[0] += 1;
503             } else {
504                 n[1] += 1;
505             }
506         }
507         if (n[0] == nHits && n[1] == 0) {
508             return true;
509         }
510         if (n[1] == nHits && n[0] == 0) {
511             return false;
512         }
513         throw new RuntimeException("mixed top and bottom hits on same track");
514 
515     }

517     private void setupSensors(EventHeader event) {
518         List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class,
519                 "SVTRawTrackerHits");
520         EventHeader.LCMetaData meta = event.getMetaData(rawTrackerHits);
521         // Get the ID dictionary and field information.
522         IIdentifierDictionary dict = meta.getIDDecoder().getSubdetector().getDetectorElement().getIdentifierHelper().getIdentifierDictionary();
523         int fieldIdx = dict.getFieldIndex("side");
524         int sideIdx = dict.getFieldIndex("strip");
525         for (RawTrackerHit hit : rawTrackerHits) {
526             // The "side" and "strip" fields needs to be stripped from the ID for sensor lookup.
527             IExpandedIdentifier expId = dict.unpack(hit.getIdentifier());
528             expId.setValue(fieldIdx, 0);
529             expId.setValue(sideIdx, 0);
530             IIdentifier strippedId = dict.pack(expId);
531             // Find the sensor DetectorElement.
532             List<IDetectorElement> des = DetectorElementStore.getInstance().find(strippedId);
533             if (des == null || des.size() == 0) {
534                 throw new RuntimeException("Failed to find any DetectorElements with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
535             } else if (des.size() == 1) {
536                 hit.setDetectorElement((SiSensor) des.get(0));
537             } else {
538                 // Use first sensor found, which should work unless there are sensors with duplicate IDs.
539                 for (IDetectorElement de : des) {
540                     if (de instanceof SiSensor) {
541                         hit.setDetectorElement((SiSensor) de);
542                         break;
543                     }
544                 }
545             }
546             // No sensor was found.
547             if (hit.getDetectorElement() == null) {
548                 throw new RuntimeException("No sensor was found for hit with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
549             }
550         }
551     }

597     void analyzeWabTrackingEfficiency(EventHeader event) {
598         setupSensors(event);
599         boolean isGG; // two photons  : presumably missing the electron track
600         boolean isEG; // electron + photon : good WAB
601         boolean isPG; // positron + photon : missing electron track
602         boolean isEP; // V0 : good trident or BH
603         String type = "unknown";
604 
605         // get the ReconstructedParticles in this event
606         List<ReconstructedParticle> rps = event.get(ReconstructedParticle.class, "FinalStateParticles");
607         // now add in the FEE candidates
608 //        rps.addAll(event.get(ReconstructedParticle.class, "OtherElectrons"));
609         if (rps.size() > 2) {
610             return; //currently allow both matched and gbl tracks
611         }
612         int[] pdgIds = new int[2];
613         int i = 0;
614         List<ReconstructedParticle> twoRPs = new ArrayList<>();
615         for (ReconstructedParticle rp : rps) {
616             boolean isGBL = TrackType.isGBL(rp.getType());
617             String trackType = isGBL ? "gbl " : "other ";
618             String particleType = "unknown";
619             int pdgId = rp.getParticleIDUsed().getPDG();
620             if (pdgId == 11) {
621                 particleType = "electron ";
622                 if (trackType.equals("gbl ") && !rp.getClusters().isEmpty()) {
623                     pdgIds[i++] = pdgId;
624                     twoRPs.add(rp);
625                     aida.histogram1D(particleType + "energy", 100, 0., 6.).fill(rp.getEnergy());
626                 }
627                 aida.histogram1D(particleType + trackType + "momentum", 100, 0., 6.).fill(rp.getMomentum().magnitude());
628             }
629             if (pdgId == -11) {
630                 particleType = "positron ";
631                 if (trackType.equals("gbl ") && !rp.getClusters().isEmpty()) {
632                     pdgIds[i++] = pdgId;
633                     twoRPs.add(rp);
634                     aida.histogram1D(particleType + "energy", 100, 0., 6.).fill(rp.getEnergy());
635                 }
636                 aida.histogram1D(particleType + trackType + "momentum", 100, 0., 6.).fill(rp.getMomentum().magnitude());
637             }
638             if (pdgId == 22) {
639                 particleType = "photon ";
640                 pdgIds[i++] = pdgId;
641                 twoRPs.add(rp);
642                 aida.histogram1D(particleType + "energy", 100, 0., 6.).fill(rp.getEnergy());
643             }
644         }
645         if (pdgIds[0] == 22 && pdgIds[1] == 22) {
646             isGG = true;
647             type = "gg";
648         }
649         if (pdgIds[0] == 22 && pdgIds[1] == 11) {
650             isEG = true;
651             type = "eg";
652         }
653         if (pdgIds[0] == 11 && pdgIds[1] == 22) {
654             isEG = true;
655             type = "eg";
656         }
657         if (pdgIds[0] == -11 && pdgIds[1] == 22) {
658             isPG = true;
659             type = "pg";
660         }
661         if (pdgIds[0] == 22 && pdgIds[1] == -11) {
662             isPG = true;
663             type = "pg";
664         }
665         if (pdgIds[0] == 11 && pdgIds[1] == -11) {
666             isEP = true;
667             type = "ep";
668         }
669         if (pdgIds[0] == -11 && pdgIds[1] == 11) {
670             isEP = true;
671             type = "ep";
672         }
673         if (twoRPs.size() == 2) {
674             aida.histogram1D("all esum", 100, 0., 5.).fill(twoRPs.get(0).getEnergy() + twoRPs.get(1).getEnergy());
675             aida.histogram1D(type + " esum", 100, 0., 5.).fill(twoRPs.get(0).getEnergy() + twoRPs.get(1).getEnergy());
676         }
677 
678 //        System.out.println("type " + type);
679         aida.tree()
680                 .mkdirs(type);
681         aida.tree()
682                 .cd(type);
683         //start by comparing egamma to gammagamma and assume that the electron track is missing in the latter sample.
684         for (ReconstructedParticle rp : twoRPs) {
685             analyzeReconstructedParticle(rp);
686         }
687 
688         aida.tree()
689                 .cd("..");
690     }

777         if (rp.getType() == 1) {
778             trackDir = "kf";
779         }
780         aida.tree().mkdirs(trackDir);
781         aida.tree().cd(trackDir);
782 
783         if (rp.getTracks().isEmpty()) {
784             System.out.println("RP with no Track with pdgId " + rp.getParticleIDUsed().getPDG());
785             return;
786         }
787         Track t = rp.getTracks().get(0);
788         if (t == null) {
789             System.out.println("null track in RP with pdgId " + rp.getParticleIDUsed().getPDG());
790         }
791 
792 //        aida.cloud1D("ReconstructedParticle Type").fill(rp.getType());
793 //        aida.cloud1D("Track Type").fill(t.getType());
794         //rotate into physiscs frame of reference
795         Hep3Vector rprot = VecOp.mult(beamAxisRotation, rp.getMomentum());
796         double theta = Math.acos(rprot.z() / rprot.magnitude());

798         int ndf = t.getNDF();
799         double chi2Ndf = t.getChi2() / t.getNDF();
