336     public boolean isFiducial(CalorimeterHit hit) {
337         int ix = hit.getIdentifierFieldValue("ix");
338         int iy = hit.getIdentifierFieldValue("iy");
339         // Get the x and y indices for the cluster.
340         int absx = Math.abs(ix);
341         int absy = Math.abs(iy);
342 
343         // Check if the cluster is on the top or the bottom of the
344         // calorimeter, as defined by |y| == 5. This is an edge cluster
345         // and is not in the fiducial region.
346         if (absy == 5) {
347             return false;
348         }
349 
350         // Check if the cluster is on the extreme left or right side
351         // of the calorimeter, as defined by |x| == 23. This is also
352         // an edge cluster and is not in the fiducial region.
353         if (absx == 23) {
354             return false;
355         }
356 
357         // Check if the cluster is along the beam gap, as defined by
358         // |y| == 1. This is an internal edge cluster and is not in the
359         // fiducial region.
360         if (absy == 1) {
361             return false;
362         }
363 
364         // Lastly, check if the cluster falls along the beam hole, as
365         // defined by clusters with -11 <= x <= -1 and |y| == 2. This
366         // is not the fiducial region.
367         if (absy == 2 && ix <= -1 && ix >= -11) {
368             return false;
369         }
370 
371         // If all checks fail, the cluster is in the fiducial region.
372         return true;
373     }

392     private boolean isTopTrack(Track t) {
393         List<TrackerHit> hits = t.getTrackerHits();
394         int n[] = {0, 0};
395         int nHits = hits.size();
396         for (TrackerHit h : hits) {
397             HpsSiSensor sensor = ((HpsSiSensor) ((RawTrackerHit) h.getRawHits().get(0)).getDetectorElement());
398             if (sensor.isTopLayer()) {
399                 n[0] += 1;
400             } else {
401                 n[1] += 1;
402             }
403         }
404         if (n[0] == nHits && n[1] == 0) {
405             return true;
406         }
407         if (n[1] == nHits && n[0] == 0) {
408             return false;
409         }
410         throw new RuntimeException("mixed top and bottom hits on same track");
411 
412     }

414     private void setupSensors(EventHeader event) {
415         List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
416         EventHeader.LCMetaData meta = event.getMetaData(rawTrackerHits);
417         // Get the ID dictionary and field information.
418         IIdentifierDictionary dict = meta.getIDDecoder().getSubdetector().getDetectorElement().getIdentifierHelper().getIdentifierDictionary();
419         int fieldIdx = dict.getFieldIndex("side");
420         int sideIdx = dict.getFieldIndex("strip");
421         for (RawTrackerHit hit : rawTrackerHits) {
422             // The "side" and "strip" fields needs to be stripped from the ID for sensor lookup.
423             IExpandedIdentifier expId = dict.unpack(hit.getIdentifier());
424             expId.setValue(fieldIdx, 0);
425             expId.setValue(sideIdx, 0);
426             IIdentifier strippedId = dict.pack(expId);
427             // Find the sensor DetectorElement.
428             List<IDetectorElement> des = DetectorElementStore.getInstance().find(strippedId);
429             if (des == null || des.size() == 0) {
430                 throw new RuntimeException("Failed to find any DetectorElements with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
431             } else if (des.size() == 1) {
432                 hit.setDetectorElement((SiSensor) des.get(0));
433             } else {
434                 // Use first sensor found, which should work unless there are sensors with duplicate IDs.
435                 for (IDetectorElement de : des) {
436                     if (de instanceof SiSensor) {
437                         hit.setDetectorElement((SiSensor) de);
438                         break;
439                     }
440                 }
441             }
442             // No sensor was found.
443             if (hit.getDetectorElement() == null) {
444                 throw new RuntimeException("No sensor was found for hit with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
445             }
446         }

484     void analyzeWabTrackingEfficiency(EventHeader event) {
485         setupSensors(event);
486         boolean isGG; // two photons  : presumably missing the electron track
487         boolean isEG; // electron + photon : good WAB
488         boolean isPG; // positron + photon : missing electron track
489         boolean isEP; // V0 : good trident or BH
490         String type = "unknown";
491 
492         // get the ReconstructedParticles in this event
493         List<ReconstructedParticle> rps = event.get(ReconstructedParticle.class, "FinalStateParticles");
494         // now add in the FEE candidates
495         rps.addAll(event.get(ReconstructedParticle.class, "OtherElectrons"));
496         if (rps.size() > 3) {
497             return; //currently allow both matched and gbl tracks
498         }
499         int[] pdgIds = new int[2];
500         int i = 0;
501         List<ReconstructedParticle> twoRPs = new ArrayList<>();
502         for (ReconstructedParticle rp : rps) {
503             boolean isGBL = TrackType.isGBL(rp.getType());
504             String trackType = isGBL ? "gbl " : "other ";
505             String particleType = "unknown";
506             int pdgId = rp.getParticleIDUsed().getPDG();
507             if (pdgId == 11) {
508                 particleType = "electron ";
509                 if (trackType.equals("gbl ") && !rp.getClusters().isEmpty()) {
510                     pdgIds[i++] = pdgId;
511                     twoRPs.add(rp);
512                     aida.histogram1D(particleType + "energy", 100, 0., 6.).fill(rp.getEnergy());
513                 }
514                 aida.histogram1D(particleType + trackType + "momentum", 100, 0., 6.).fill(rp.getMomentum().magnitude());
515             }
516             if (pdgId == -11) {
517                 particleType = "positron ";
518                 if (trackType.equals("gbl ") && !rp.getClusters().isEmpty()) {
519                     pdgIds[i++] = pdgId;
520                     twoRPs.add(rp);
521                     aida.histogram1D(particleType + "energy", 100, 0., 6.).fill(rp.getEnergy());
522                 }
523                 aida.histogram1D(particleType + trackType + "momentum", 100, 0., 6.).fill(rp.getMomentum().magnitude());
524             }
525             if (pdgId == 22) {
526                 particleType = "photon ";
527                 pdgIds[i++] = pdgId;
528                 twoRPs.add(rp);
529                 aida.histogram1D(particleType + "energy", 100, 0., 6.).fill(rp.getEnergy());
530             }
531         }
532         if (pdgIds[0] == 22 && pdgIds[1] == 22) {
533             isGG = true;
534             type = "gg";
535         }
536         if (pdgIds[0] == 22 && pdgIds[1] == 11) {
537             isEG = true;
538             type = "eg";
539         }
540         if (pdgIds[0] == 11 && pdgIds[1] == 22) {
541             isEG = true;
542             type = "eg";
543         }
544         if (pdgIds[0] == -11 && pdgIds[1] == 22) {
545             isPG = true;
546             type = "pg";
547         }
548         if (pdgIds[0] == 22 && pdgIds[1] == -11) {
549             isPG = true;
550             type = "pg";
551         }
552         if (pdgIds[0] == 11 && pdgIds[1] == -11) {
553             isEP = true;
554             type = "ep";
555         }
556         if (pdgIds[0] == -11 && pdgIds[1] == 11) {
557             isEP = true;
558             type = "ep";
559         }
560 //        System.out.println("type " + type);
561         aida.tree().mkdirs(type);
562         aida.tree().cd(type);
563         //start by comparing egamma to gammagamma and assume that the electron track is missing in the latter sample.
564         for (ReconstructedParticle rp : twoRPs) {
565             analyzeReconstructedParticle(rp);
566         }
567         aida.tree().cd("..");
568     }

570     void analyzeReconstructedParticle(ReconstructedParticle rp) {
571         boolean isElectron = rp.getParticleIDUsed().getPDG() == 11;
572         boolean isPositron = rp.getParticleIDUsed().getPDG() == -11;
573         boolean isPhoton = rp.getParticleIDUsed().getPDG() == 22;
574         String type = "";
575         if (isElectron) {
576             type = "electron";
577         }
578         if (isPositron) {
579             type = "positron";
580         }
581         if (isPhoton) {
582             type = "photon";
583         }
584 
585         aida.tree().mkdirs(type);
586         aida.tree().cd(type);
587 
588         if (isElectron || isPositron) {
589             analyzeTrack(rp);
590         }
591 
592         if (isPhoton) {
593             analyzeCluster(rp);
594         }
595 
596         aida.tree().cd("..");
597 
598     }

600     void analyzeCluster(ReconstructedParticle rp) {
601         Cluster c = rp.getClusters().get(0);
602         double p = rp.getMomentum().magnitude();
603         double e = rp.getEnergy();
604 
605         CalorimeterHit seedHit = ClusterUtilities.findSeedHit(c);
606         double seedHitEnergy = ClusterUtilities.findSeedHit(c).getCorrectedEnergy();
607         boolean isFiducial = isFiducial(seedHit);
608         String fid = isFiducial ? "fiducial" : "";
609         // debug diagnostics to set cuts
610 
611         if (c.getPosition()[1] > 0.) {
612             aida.histogram1D("Top cluster energy", 100, 0., 5.5).fill(c.getEnergy());
613         } else {
614             aida.histogram1D("Bottom cluster energy", 100, 0., 5.5).fill(c.getEnergy());
615         }
616         aida.histogram2D("Cluster x vs y", 320, -270.0, 370.0, 90, -90.0, 90.0).fill(c.getPosition()[0], c.getPosition()[1]);
617         aida.histogram1D("clusterSeedHit energy", 50, 0.5, 4.5).fill(seedHitEnergy);
618         aida.histogram1D("cluster nHits", 20, 0., 20.).fill(c.getCalorimeterHits().size());
619         aida.histogram2D("clusterSeedHit energy vs energy", 100, 0., 5.5, 50, 0.5, 4.5).fill(e, seedHitEnergy);
620         aida.histogram2D("cluster nHits vs energy", 100, 0., 5.5, 20, 0., 20.).fill(e, c.getCalorimeterHits().size());
621         aida.histogram2D("cluster time vs e", 100, 0., 5.5, 30, 30., 60.).fill(p, ClusterUtilities.getSeedHitTime(c));
622         if (isFiducial) {
623             if (c.getPosition()[1] > 0.) {
624                 aida.histogram1D("Top cluster energy " + fid, 100, 0.0, 5.5).fill(c.getEnergy());
625             } else {
626                 aida.histogram1D("Bottom cluster energy " + fid, 100, 0.0, 5.5).fill(c.getEnergy());
627             }
628             aida.histogram2D("Cluster x vs y " + fid, 320, -270.0, 370.0, 90, -90.0, 90.0).fill(c.getPosition()[0], c.getPosition()[1]);
629             aida.histogram1D("clusterSeedHit energy " + fid, 50, 0.5, 4.5).fill(seedHitEnergy);
630             aida.histogram1D("cluster nHits " + fid, 20, 0., 20.).fill(c.getCalorimeterHits().size());
631             aida.histogram2D("clusterSeedHit energy vs energy " + fid, 100, 0.0, 5.5, 50, 0.5, 4.5).fill(e, seedHitEnergy);
632             aida.histogram2D("cluster nHits vs energy " + fid, 100, 0.0, 5.5, 20, 0., 20.).fill(e, c.getCalorimeterHits().size());
633             aida.histogram2D("cluster time vs e " + fid, 100, 0.0, 5.5, 30, 30., 60.).fill(p, ClusterUtilities.getSeedHitTime(c));
634 
635             if (seedHitEnergy > 2.8) {
636                 if (c.getPosition()[1] > 0.) {
637                     aida.histogram1D("Top cluster energy seed hit > 2.8", 100, 0.0, 5.5).fill(c.getEnergy());
638                 } else {
639                     aida.histogram1D("Bottom cluster energy seed hit > 2.8", 100, 0.0, 5.5).fill(c.getEnergy());
640                 }
641                 if (seedHitEnergy > 3.0) {
642                     if (c.getPosition()[1] > 0.) {
643                         aida.histogram1D("Top cluster energy seed hit > 3.0", 100, 0.0, 5.5).fill(c.getEnergy());
644                     } else {
645                         aida.histogram1D("Bottom cluster energy seed hit > 3.0", 100, 0.0, 5.5).fill(c.getEnergy());
646                     }
647                 }
648             }
649         }

652     void analyzeTrack(ReconstructedParticle rp) {
653         boolean isGBL = TrackType.isGBL(rp.getType());
654         String trackDir = isGBL ? "gbl" : "htf";
655         if (rp.getType() == 1) {
656             trackDir = "kf";
657         }
658         aida.tree().mkdirs(trackDir);
659         aida.tree().cd(trackDir);
660 
661         if (rp.getTracks().isEmpty()) {
662             System.out.println("RP with no Track with pdgId " + rp.getParticleIDUsed().getPDG());
663             return;
664         }
665         Track t = rp.getTracks().get(0);
666         if (t == null) {
667             System.out.println("null track in RP with pdgId " + rp.getParticleIDUsed().getPDG());
668         }
669 
670 //        aida.cloud1D("ReconstructedParticle Type").fill(rp.getType());
671 //        aida.cloud1D("Track Type").fill(t.getType());
672         //rotate into physiscs frame of reference
673         Hep3Vector rprot = VecOp.mult(beamAxisRotation, rp.getMomentum());
674         double theta = Math.acos(rprot.z() / rprot.magnitude());
675         double chiSquared = t.getChi2();
676         int ndf = t.getNDF();
677         double chi2Ndf = t.getChi2() / t.getNDF();
678         double chisqProb = 1.;
679         if (ndf != 0) {
680             chisqProb = ChisqProb.gammp(ndf, chiSquared);
681         }
682         int nHits = t.getTrackerHits().size();
683         double dEdx = t.getdEdx();
684         double e = rp.getEnergy();
685         double p = rp.getMomentum().magnitude();
686 
687         String topOrBottom = isTopTrack(t) ? " top " : " bottom ";
688         aida.histogram1D("Track chisq per df" + topOrBottom, 100, 0., 50.).fill(chiSquared / ndf);
689         aida.histogram1D("Track chisq prob" + topOrBottom, 100, 0., 1.).fill(chisqProb);
690         aida.histogram1D("Track nHits" + topOrBottom, 7, 0.5, 7.5).fill(t.getTrackerHits().size());
691         aida.histogram1D("Track momentum" + topOrBottom, 100, 0., 5.0).fill(p);
692         aida.histogram1D("Track deDx" + topOrBottom, 100, 0.00004, 0.00013).fill(t.getdEdx());
693         aida.histogram1D("Track theta" + topOrBottom, 100, 0.010, 0.160).fill(theta);
694         aida.histogram2D("Track theta vs p" + topOrBottom, 100, 0.010, 0.160, 100, 0., 10.0).fill(theta, p);
695         aida.histogram1D("rp x0" + topOrBottom, 100, -0.50, 0.50).fill(TrackUtils.getX0(t));
696         aida.histogram1D("rp y0" + topOrBottom, 100, -5.0, 5.0).fill(TrackUtils.getY0(t));
697         aida.histogram1D("rp z0" + topOrBottom, 100, -1.0, 1.0).fill(TrackUtils.getZ0(t));
698 
699         //
700         aida.histogram1D("Track chisq per df" + topOrBottom + " " + nHits + " hits", 100, 0., 50.).fill(chiSquared / ndf);
701         aida.histogram1D("Track chisq prob" + topOrBottom + " " + nHits + " hits", 100, 0., 1.).fill(chisqProb);
702         aida.histogram1D("Track nHits" + topOrBottom + " " + nHits + " hits", 7, 0.5, 7.5).fill(t.getTrackerHits().size());
703         aida.histogram1D("Track momentum" + topOrBottom + " " + nHits + " hits", 100, 0., 5.0).fill(p);
704         aida.histogram1D("Track deDx" + topOrBottom + " " + nHits + " hits", 100, 0.00004, 0.00013).fill(t.getdEdx());
705         aida.histogram1D("Track theta" + topOrBottom + " " + nHits + " hits", 100, 0.010, 0.160).fill(theta);
706         aida.histogram2D("Track theta vs p" + topOrBottom + " " + nHits + " hits", 100, 0.010, 0.160, 100, 0., 10.0).fill(theta, p);
707         aida.histogram1D("rp x0" + topOrBottom + " " + nHits + " hits", 100, -0.50, 0.50).fill(TrackUtils.getX0(t));
708         aida.histogram1D("rp y0" + topOrBottom + " " + nHits + " hits", 100, -5.0, 5.0).fill(TrackUtils.getY0(t));
709         aida.histogram1D("rp z0" + topOrBottom + " " + nHits + " hits", 100, -1.0, 1.0).fill(TrackUtils.getZ0(t));
710 
711         boolean hasCluster = rp.getClusters().size() == 1;
712         if (hasCluster) {
713             analyzeCluster(rp);
714         }
715         //
716         aida.tree().cd("..");
717     }
