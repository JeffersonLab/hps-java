1   package org.hps.analysis.wab;
2   
3   import java.util.List;
4   import org.lcsim.event.EventHeader;
5   import org.lcsim.event.MCParticle;
6   import org.lcsim.event.MCParticle.SimulatorStatus;
7   import org.lcsim.util.Driver;
8   
9   /**
10   * Driver to select events in which the bremsstrahlung photon has converted in
11   * the tracker. Input files should be WAB events, i.e. events with just the
12   * incoming electron as particle 0 and the photon as particle 1
13   */
14  public class StripMcWabConversionDriver extends Driver
15  {
16  
17      private int _numberOfEventsWritten = 0;
18      private int _numberOfEventsRead = 0;
19  
20      protected void process(EventHeader event)
21      {
22          _numberOfEventsRead++;
23          boolean skipEvent = true;
24          List<MCParticle> mclist = event.getMCParticles();
25          //
26          // HPS Wide angle bremsstrahlung events list the electron first, followed by the bremsstrahlung photon.
27          // If that changes, so will the next line.
28          // TODO add some selection code here to recognize MC WAB events in mixed files.
29          // assert that particle 0 is electron, particle 1 is photon
30          // assert that sum of particle 0 and 1 has the beam energy and is directed along the beam direction
31          //
32          MCParticle brem = mclist.get(1);
33          SimulatorStatus stat = brem.getSimulatorStatus();
34          // select events in which the photon converted in the tracker
35          if (stat.isDecayedInTracker()) {
36              skipEvent = false;
37          }
38          if (skipEvent) {
39              throw new Driver.NextEventException();
40          } else {
41              _numberOfEventsWritten++;
42          }
43      }
44  
45      protected void endOfData()
46      {
47          System.out.println("Wrote " + _numberOfEventsWritten + " events of " + _numberOfEventsRead + " read.");
48      }
49  }
