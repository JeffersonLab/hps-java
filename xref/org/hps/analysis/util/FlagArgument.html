1   package org.hps.analysis.util;
2   
3   /**
4    * <code>FlagArgument</code> is a simple implementation of {@link
5    * org.hps.analysis.util.Argument Argument} that does not accept a
6    * value. Instead, it simply tracks whether or not an instance of the
7    * argument has been seen or not.
8    * 
9    * @see org.hps.analysis.util.Argument
10   */
11  public class FlagArgument extends Argument {
12      private boolean defined = false;
13      
14      /**
15       * Instantiates a new instance of a flag argument.
16       * @param shortOption - Defines the short option text used to
17       * declare this argument on the command line.
18       * @param fullOption - Defines the full option text used to
19       * declare this argument on the command line.
20       * @param description - Defines the description text used for
21       * displaying usage information.
22       * @param required - Specifies whether or not the argument is
23       * required for the associated command to run.
24       */
25      public FlagArgument(String shortOption, String fullOption, String description, boolean required) {
26          super(shortOption, fullOption, description, required);
27      }
28      
29      @Override
30      boolean acceptsValues() {
31          return false;
32      }
33      
34      @Override
35      boolean allowsMultipleInstances() {
36          return false;
37      }
38      
39      @Override
40      boolean isDefined() {
41          return defined;
42      }
43      
44      /**
45       * Sets whether or not an instance of this argument was seen.
46       * @param state - <code>true</code> indicates that an instance of
47       * this argument was seen and <code>false</code> that it was not.
48       */
49      void setDefined(boolean state) {
50          defined = state;
51      }
52  }
