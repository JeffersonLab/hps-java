1173     private static double PathCalc(double xc, double yc, double RC, double x1, double y1, double x2, double y2) {
1174         //  Find the angle between these points measured wrt the circle center
1175         double phi1 = Math.atan2(y1 - yc, x1 - xc);
1176         double phi2 = Math.atan2(y2 - yc, x2 - xc);
1177         double dphi = phi2 - phi1;
1178         //  Make sure dphi is in the valid range (-pi, pi)
1179         if (dphi > Math.PI)
1180             dphi -= 2. * Math.PI;
1181         if (dphi < -Math.PI)
1182             dphi += 2. * Math.PI;
1183         //  Return the arc length
1184         return -RC * dphi;
