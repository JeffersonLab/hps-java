16  /**
17   * <code>HodoscopeBoundsPrinterUtility</code> is a utility method
18   * that prints the positions, bounds, and miscellaneous parameters
19   * for a given hodoscope detector to the terminal. It is intended to
20   * allow for easy checking of the positioning of hodoscope elements
21   * after detector updates to ensure that the programmatic detector
22   * matches what is expected<br/><br/>
23   * It may be run using the command: <code>java -cp $HPS_JAVA
24   * org.hps.analysis.util.HodoscopeBoundsPrinterUtility -R
25   * $RUN_NUMBER -d $HODOSCOPE_DETECTOR_NAME</code>
26   */
27  public class HodoscopeBoundsPrinterUtility {
28      public static final void main(String args[]) throws ConditionsNotFoundException {
29          // Define the command line arguments.
30          UtilityArgumentParser argsParser = new UtilityArgumentParser("java -cp $HPS_JAVA org.hps.analysis.util.HodoscopeBoundsPrinterUtility");
31          argsParser.addSingleValueArgument("-d", "--detector", "Detector name", true);
32          argsParser.addSingleValueArgument("-R", "--run", "Run number", true);
33          argsParser.addFlagArgument("-h", "--help", "Display usage information", false);
34          
35          // Parse the command line arguments.
36          argsParser.parseArguments(args);
37          
38          // If the "display usage information" argument is present,
39          // then display the usage information and exit. Also do this
40          // if a required argument is not included.
41          if(argsParser.isDefined("-h") || !argsParser.verifyRequirements()) {
42              System.out.println(argsParser.getHelpText());
43              System.exit(0);
44          }
45          
46          // Get the required arguments.
47          String detectorName = ((SingleValueArgument) argsParser.getArgument("-d")).getValue();
48          int runNumber = Integer.parseInt(((SingleValueArgument) argsParser.getArgument("-R")).getValue());
49          
50          // Load the conditions database.
51          DatabaseConditionsManager conditions = DatabaseConditionsManager.getInstance();
52          conditions.setRun(runNumber);
53          conditions.setDetector(detectorName, runNumber);
54          
55          // Get the detector.
56          Detector detector = conditions.getDetectorObject();
57          HodoscopeDetectorElement hodoscopeDetectorElement = (HodoscopeDetectorElement) detector.getSubdetector("Hodoscope").getDetectorElement();
58          
59          // Get the list of all hodoscope scintillator identifiers
60          // from the detector.
61          List<IIdentifier> scintillators = hodoscopeDetectorElement.getScintillatorIdentifiers();
62          
63          // Iterate over the scintillators and store their bounds.
64          Map<Integer, String> hodoscopeBoundsMap = new HashMap<Integer, String>();
65          for(IIdentifier id : scintillators) {
66              Integer posvar = Integer.valueOf(hodoscopeDetectorElement.getScintillatorUniqueKey(id));
67              hodoscopeBoundsMap.put(posvar, getHodoscopeBounds(id, hodoscopeDetectorElement));
68          }
69          
70          // Sort the bounds data and print it to the terminal.
71          System.out.println(lblue("Hodoscope Geometry Bounds for Detector ") + yellow(detectorName)
72                  + lblue(" and Run Number ") + yellow(Integer.toString(runNumber)) + lblue(":"));
73          List<String> boundsData = new ArrayList<String>(hodoscopeBoundsMap.size());
74          boundsData.addAll(hodoscopeBoundsMap.values());
75          Collections.sort(boundsData);
76          for(String entry : boundsData) {
77              System.out.println(entry);
78          }
79      }

81      /**
82       * Creates a concise {@link java.lang.String String} object that
83       * contains the geometric details for the scintillator on which
84       * the argument hit occurred.
85       * @param hit - The hit.
86       * @return Returns a <code>String</code> describing the geometric
87       * details of the scintillator on which the hit occurred.
88       */
89      private static String getHodoscopeBounds(IIdentifier id, HodoscopeDetectorElement hodoscopeDetectorElement) {
90          int[] indices = hodoscopeDetectorElement.getHodoscopeIndices(id);
91          int holes = hodoscopeDetectorElement.getScintillatorChannelCount(id);
92          double[] dimensions = hodoscopeDetectorElement.getScintillatorHalfDimensions(id);
93          double[] position = hodoscopeDetectorElement.getScintillatorPosition(id);
94          double[] holeX = hodoscopeDetectorElement.getScintillatorHolePositions(id);
95          
96          String scintillatorBoundString = String.format(
97                  "Scintillator " + yellow("%1d") + ", " + yellow("%-6s") + ", Layer " + yellow("%1d") + "%n"
98                  + "\tx-Bounds = [ " + yellow("%7.2f") + " mm, " + yellow("%7.2f") + " mm ]%n"
99                  + "\ty-Bounds = [ " + yellow("%7.2f") + " mm, " + yellow("%7.2f") + " mm ]%n"
100                 + "\tz-Bounds = [ " + yellow("%7.2f") + " mm, " + yellow("%7.2f") + " mm ]%n"
101                 + "\tFADC Channels: " + yellow("%d") + "%n",
102                 (indices[0] + 1), indices[1] == 1 ? "Top" : "Bottom", (indices[2] + 1),
103                 position[0] - dimensions[0], position[0] + dimensions[0],
104                 position[1] - dimensions[1], position[1] + dimensions[1],
105                 position[2] - dimensions[2], position[2] + dimensions[2],
106                 holes);
107         
108         String holePositions = null;
109         if(holes == 1) {
110             holePositions = String.format("\t\tHole Position = " + yellow("%6.2f") + "%n", holeX[0]);
111         } else {
112             holePositions = String.format(
113                     "\t\tHole 1 Position = " + yellow("%6.2f") + " mm%n"
114                     + "\t\tHole 2 Position = " + yellow("%6.2f") + " mm%n",
115                     holeX[0], holeX[1]);
116         }
117         
118         return scintillatorBoundString + holePositions;
119     }
