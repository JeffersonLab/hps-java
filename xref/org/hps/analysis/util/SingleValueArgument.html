1   package org.hps.analysis.util;
2   
3   /**
4    * <code>SingleValueArgument</code> represents a command line
5    * argument that only allows for a single instance of itself. It is
6    * an implementation of {@link org.hps.analysis.util.Argument
7    * Argument}.
8    * 
9    * @see org.hps.analysis.util.Argument
10   */
11  public class SingleValueArgument extends Argument {
12      private String value = null;
13      
14      /**
15       * Instantiates a new instance of a single-value argument.
16       * @param shortOption - Defines the short option text used to
17       * declare this argument on the command line.
18       * @param fullOption - Defines the full option text used to
19       * declare this argument on the command line.
20       * @param description - Defines the description text used for
21       * displaying usage information.
22       * @param required - Specifies whether or not the argument is
23       * required for the associated command to run.
24       */
25      public SingleValueArgument(String shortOption, String fullOption, String description, boolean required) {
26          super(shortOption, fullOption, description, required);
27      }
28      
29      @Override
30      boolean acceptsValues() {
31          return true;
32      }
33      
34      @Override
35      boolean allowsMultipleInstances() {
36          return false;
37      }
38      
39      /**
40       * Gets the value that was given with this argument.
41       * @return Gets the value as a {@link java.util.String String}.
42       */
43      String getValue() {
44          return value;
45      }
46      
47      @Override
48      boolean isDefined() {
49          return value != null;
50      }
51      
52      /**
53       * Sets the value associated with this argument.
54       * @param value - The value.
55       */
56      void setValue(String value) {
57          this.value = value;
58      }
59  }
