10  /**
11   * <code>UtilityArgumentParser</code> is designed to automate the
12   * handling of command line argument parsing. It enables the user to
13   * define what arguments are valid and the easily access whether or
14   * not those arguments have been defined and, where applicable, what
15   * their values are.
16   * <br/><br/>
17   * The utility allows for different types of arguments to be defined.
18   * These include flag arguments which represent arguments that do
19   * not accept values (i.e. <code>-h</code>), single-value arguments
20   * which represent arguments that require a value but can only occur
21   * once (i.e. <code>-R $RUN_NUMBER</code>), and multiple-value
22   * arguments which represents arguments that take a value and can
23   * occur more than once (i.e. <code>-i $INPUT_FILE_1 -i
24   * $INPUT_FILE_2</code>).
25   */
26  class UtilityArgumentParser {
27      private final String command;
28      private List<Argument> arguments = new ArrayList<Argument>();
29      private Map<String, Argument> optionMap = new HashMap<String, Argument>();
30      
31      /**
32       * Instantiates a command line argument parser for the specified
33       * command.
34       * @param command - The command.
35       */
36      UtilityArgumentParser(String command) {
37          this.command = command;
38      }
39      
40      /**
41       * Adds a new {@link org.hps.analysis.util.FlagArgument
42       * FlagArgument} with the specified parameters to the set of
43       * allowed command line arguments.
44       * @param shortOption - Defines the short option text used to
45       * declare this argument on the command line.
46       * @param fullOption - Defines the full option text used to
47       * declare this argument on the command line.
48       * @param description - Defines the description text used for
49       * displaying usage information.
50       * @param required - Specifies whether or not the argument is
51       * required for the associated command to run.
52       */
53      void addFlagArgument(String shortOption, String fullOption, String description, boolean required) {
54          addArgument(new FlagArgument(shortOption, fullOption, description, required));
55      }
56      
57      /**
58       * Adds a new {@link org.hps.analysis.util.SingleValueArgument
59       * SingleValueArgument} with the specified parameters to the set
60       * of allowed command line arguments.
61       * @param shortOption - Defines the short option text used to
62       * declare this argument on the command line.
63       * @param fullOption - Defines the full option text used to
64       * declare this argument on the command line.
65       * @param description - Defines the description text used for
66       * displaying usage information.
67       * @param required - Specifies whether or not the argument is
68       * required for the associated command to run.
69       */
70      void addSingleValueArgument(String shortOption, String fullOption, String description, boolean required) {
71          addArgument(new SingleValueArgument(shortOption, fullOption, description, required));
72      }
73      
74      /**
75       * Adds a new {@link org.hps.analysis.util.MultipleValueArgument
76       * MultipleValueArgument} with the specified parameters to the
77       * set of allowed command line arguments.
78       * @param shortOption - Defines the short option text used to
79       * declare this argument on the command line.
80       * @param fullOption - Defines the full option text used to
81       * declare this argument on the command line.
82       * @param description - Defines the description text used for
83       * displaying usage information.
84       * @param required - Specifies whether or not the argument is
85       * required for the associated command to run.
86       */
87      void addMultipleValueArgument(String shortOption, String fullOption, String description, boolean required) {
88          addArgument(new MultipleValueArgument(shortOption, fullOption, description, required));
89      }
90      
91      /**
92       * Finds the argument, if it exists, which uses the specified
93       * command line option text.
94       * @param optionText - The command line text that corresponds to
95       * the argument.
96       * @return Returns the {@link org.hps.analysis.util.Argument
97       * Argument} object that corresponds to the indicated command
98       * line option text.
99       * @throws RuntimeException Occurs if there is no argument that
100      * corresponds to the specified command line argument text.
101      */
102     Argument getArgument(String optionText) {
103         if(optionMap.containsKey(optionText)) {
104             return optionMap.get(optionText);
105         } else {
106             throw new RuntimeException("Unrecognized option \"" + optionText + "\".");
107         }
108     }
109     
110     /**
111      * Returns a BASH-formatted description of the valid command line
112      * arguments.
113      * @return Returns the description as a {@link java.util.String
114      * String} object.
115      */
116     String getHelpText() {
117         // Get the longest short option.
118         int longestShortOption = 0;
119         for(Argument arg : arguments) {
120             if(arg.getShortOptionText() != null) { longestShortOption = Math.max(longestShortOption, arg.getShortOptionText().length()); };
121         }
122         
123         // Get the longest full option.
124         int longestFullOption = 0;
125         for(Argument arg : arguments) {
126             if(arg.getFullOptionText() != null) { longestFullOption = Math.max(longestFullOption, arg.getFullOptionText().length()); }
127         }
128         
129         // Get the longest description.
130         int longestDescription = 0;
131         for(Argument arg : arguments) {
132             if(arg.getDescription() != null) { longestDescription = Math.max(longestDescription, arg.getDescription().length()); }
133         }
134         
135         // Output the command prompt command.
136         StringBuffer textBuffer = new StringBuffer();
137         textBuffer.append(BashParameter.format(command, BashParameter.TEXT_LIGHT_BLUE));
138         textBuffer.append('\n');
139         
140         // Output each argument.
141         for(Argument arg : arguments) {
142             textBuffer.append(getFormattedArgumentText(arg, longestShortOption, longestFullOption, longestDescription));
143             textBuffer.append('\n');
144         }
145         
146         // Return the result.
147         return textBuffer.toString();
148     }

150     /**
151      * Specifies whether or not the argument corresponding to the
152      * specified command line option text is defined or not.
153      * @param optionText - The command line option text.
154      * @return Returns <code>true</code> if the argument is defined
155      * and <code>false</code> otherwise.
156      */
157     boolean isDefined(String optionText) {
158         if(optionMap.containsKey(optionText)) {
159             return optionMap.get(optionText).isDefined();
160         } else { return false; }
161     }
162     
163     /**
164      * Parses the command line arguments.
165      * @param args - The command line arguments.
166      */
167     void parseArguments(String[] args) {
168         // Iterate over the options and store them in the appropriate
169         // argument.
170         for(int i = 0; i < args.length; i++) {
171             // If the option is a valid option, parse it.
172             if(optionMap.containsKey(args[i])) {
173                 // Get the argument.
174                 Argument arg = optionMap.get(args[i]);
175                 
176                 // How the argument is handled depends on its type.
177                 // Flag arguments are set to defined if the option is
178                 // seen anywhere at all.
179                 if(arg instanceof FlagArgument) {
180                     ((FlagArgument) arg).setDefined(true);
181                 }
182                 
183                 // Single-value arguments can only set once. If there
184                 // are multiple instances, it causes an exception.
185                 if(arg instanceof SingleValueArgument) {
186                     if(arg.isDefined()) {
187                         throw new RuntimeException("Option \"" + arg.getShortOptionText() + "\" appears more than once, but does not allow for mutliple instances.");
188                     } else {
189                         ((SingleValueArgument) arg).setValue(args[++i]);
190                     }
191                 }
192                 
193                 // Multiple-value arguments can appear as many times
194                 // as needed.
195                 if(arg instanceof MultipleValueArgument) {
196                     ((MultipleValueArgument) arg).addValue(args[++i]);
197                 }
198             }
199             
200             // Otherwise, through an exception.
201             else {
202                 throw new RuntimeException("Unrecognized option \"" + args[i] + "\".");
203             }
204         }
205     }

207     /**
208      * Creates an exception indicating that a required argument is
209      * not defined.
210      */
211     void throwRequiredArgumentMissingError() {
212         throw new RuntimeException("One or more required arguments are missing. See usage information for more details.");
213     }

215     /**
216      * Checks that all of the required options are defined.
217      * @throws RuntimeException Occurs if a required argument is not
218      * defined.
219      */
220     boolean verifyRequirements() throws RuntimeException {
221         // Iterate over the arguments. If there is an instance of a
222         // required argument where no values have been defined, then
223         // the requirements are not satisfied.
224         for(Argument arg : arguments) {
225             if(arg.isRequired() && !arg.isDefined()) {
226                 return false;
227             }
228         }
229         
230         // If all required arguments are defined, then the
231         // requirements are satisfied.
232         return true;
233     }

235     /**
236      * Adds an argument to the set of valid arguments.
237      * @param arg - The argument to add.
238      */
239     private void addArgument(Argument arg) {
240         // Map the options to the argument object. If the option
241         // already exists, throw an error.
242         mapOption(arg.getFullOptionText(), arg, optionMap);
243         mapOption(arg.getShortOptionText(), arg, optionMap);
244         
245         // Create the argument object.
246         arguments.add(arg);
247     }

249     /**
250      * Gets the BASH-formatted text that describes an argument.
251      * @param arg - The argument.
252      * @param shortLength - The length of the longest short version
253      * of the command line option text that corresponds to the
254      * argument.
255      * @param fullLength - The length of the longest long version
256      * of the command line option text that corresponds to the
257      * argument.
258      * @param descriptionLength - The length of the longest
259      * description for an argument.
260      * @return Returns a {java.util.String String} object that
261      * describes the argument. It is formatted to display color in a
262      * BASH prompt.
263      */
264     private static final String getFormattedArgumentText(Argument arg, int shortLength, int fullLength, int descriptionLength) {
265         // Create a buffer to contain the argument text.
266         StringBuffer outputBuffer = new StringBuffer("\t\t");
267         
268         // Process the short option.
269         String shortText = String.format("%-" + shortLength + "s", arg.getShortOptionText() == null ? "" : arg.getShortOptionText());
270         outputBuffer.append(BashParameter.format(shortText, BashParameter.TEXT_YELLOW, BashParameter.PROPERTY_BOLD));
271         outputBuffer.append('\t');
272         
273         // Process the full option.
274         String fullText = String.format("%-" + fullLength + "s", arg.getFullOptionText() == null ? "" : arg.getFullOptionText());
275         outputBuffer.append(BashParameter.format(fullText, BashParameter.TEXT_YELLOW, BashParameter.PROPERTY_BOLD));
276         outputBuffer.append('\t');
277         
278         // Process the description.
279         String descriptionText = String.format("%-" + descriptionLength + "s", arg.getDescription() == null ? "" : arg.getDescription());
280         outputBuffer.append(descriptionText);
281         outputBuffer.append(' ');
282         
283         // Indicate whether the argument is mandatory or not.
284         if(arg.isRequired()) {
285             outputBuffer.append(BashParameter.format("[ REQUIRED ]", BashParameter.TEXT_RED, BashParameter.PROPERTY_BOLD));
286         } else {
287             outputBuffer.append(BashParameter.format("[ OPTIONAL ]", BashParameter.TEXT_LIGHT_GREY, BashParameter.PROPERTY_DIM));
288         }
289         
290         // Return the resultant string.
291         return outputBuffer.toString();
292     }

294     /**
295      * Maps a command line option text to an argument.
296      * @param text - The command line option text.
297      * @param arg - The argument.
298      * @param map - The map.
299      */
300     private static final void mapOption(String text, Argument arg, Map<String, Argument> map) {
301         if(text == null) { return; }
302         if(map.containsKey(text)) {
303             throw new RuntimeException("Argument " + text + " is already defined.");
304         } else {
305             map.put(text, arg);
306         }
307     }
