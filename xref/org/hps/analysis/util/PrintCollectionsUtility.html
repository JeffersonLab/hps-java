28  /**
29   * <code>PrintCollectionsUtility</code> is a utility designed to
30   * print all of the collections in an argument LCIO file from the
31   * command line. Users can specify how many events to search for new
32   * unique collection names. The utility will print both the names of
33   * all collections observed and the object types.
34   * <br/><br/>
35   * Additionally, users may specify that the utility should print the
36   * full collection data for specific collections. This will only
37   * occur if there is a print method for that object type implemented.
38   */
39  public class PrintCollectionsUtility {
40      /**
41       * Iterates over the events in a file and collates all the
42       * collection names which occur in any event. Method can also
43       * print the data in the collections if a <code>toString(T,
44       * int)</code> method (where <code>T</code> is the type of the
45       * object) is implemented.
46       * @param args - The command line arguments.
47       * @throws IOException - Should never occur. This is used
48       * internally to identify the end of the LCIO input file.
49       * @throws NumberFormatException - Occurs if a command line
50       * argument that is intended to be numerical is not.
51       * @throws ConditionsNotFoundException Occurs if the specified
52       * conditions set is not found.
53       */
54      public static void main(String[] args) throws IOException, NumberFormatException, ConditionsNotFoundException {
55          // Define the command line arguments.
56          UtilityArgumentParser argsParser = new UtilityArgumentParser("java -cp $HPS_JAVA org.hps.analysis.util.PrintCollectionsUtility");
57          argsParser.addSingleValueArgument("-i", "--input", "Input file name", true);
58          argsParser.addSingleValueArgument("-d", "--detector", "Detector name", true);
59          argsParser.addSingleValueArgument("-R", "--run", "Run number", true);
60          argsParser.addSingleValueArgument("-n", "--number", "Maximum events to process", false);
61          argsParser.addMultipleValueArgument("-p", "--print", "Print collection data", false);
62          argsParser.addFlagArgument("-h", "--help", "Display usage information", false);
63          
64          // Parse the command line arguments.
65          argsParser.parseArguments(args);
66          
67          // If the "display usage information" argument is present,
68          // then display the usage information and exit. Also do this
69          // if a required argument is not included.
70          if(argsParser.isDefined("-h") || !argsParser.verifyRequirements()) {
71              System.out.println(argsParser.getHelpText());
72              System.exit(0);
73          }
74          
75          // Get the required arguments.
76          String detectorName = ((SingleValueArgument) argsParser.getArgument("-d")).getValue();
77          int runNumber = Integer.parseInt(((SingleValueArgument) argsParser.getArgument("-R")).getValue());
78          
79          // Load the conditions database.
80          DatabaseConditionsManager conditions = DatabaseConditionsManager.getInstance();
81          conditions.setRun(runNumber);
82          conditions.setDetector(detectorName, runNumber);
83          
84          // Get the maximum number of events to run.
85          int maxEvents = Integer.MAX_VALUE;
86          if(argsParser.isDefined("-n")) {
87              maxEvents = Integer.parseInt(((SingleValueArgument) argsParser.getArgument("-n")).getValue());
88          }
89          
90          // Get the collections which should be printed.
91          List<String> printCollections = new ArrayList<String>();
92          if(argsParser.isDefined("-p")) {
93              printCollections = ((MultipleValueArgument) argsParser.getArgument("-p")).getValues();
94          }
95          
96          // Validate the input file.
97          String fileName = ((SingleValueArgument) argsParser.getArgument("-i")).getValue();
98          File inputFile = new File(fileName);
99          if(!inputFile.exists()) { throw new RuntimeException("Error: Input file \"" + args[0] + "\" does not exist."); }
100         
101         // Initialize the file reader.
102         LCIOReader reader = new LCIOReader(inputFile);
103         
104         // Get all collections names.
105         int processedEvents = 0;
106         Set<Pair<String, String>> collectionSet = new HashSet<Pair<String, String>>();
107         Map<String, Class<?>> collectionMap = new HashMap<String, Class<?>>();
108         eventLoop:
109         while(true) {
110             EventHeader event = null;
111             try { event = reader.read(); }
112             catch(EOFException e) { break eventLoop; }
113             
114             @SuppressWarnings("rawtypes")
115             Set<List> eventCollections = event.getLists();
116             for(List<?> eventCollection : eventCollections) {
117                 LCMetaData collectionMetaData = event.getMetaData(eventCollection);
118                 collectionSet.add(new Pair<String, String>(collectionMetaData.getName(), collectionMetaData.getType().getSimpleName()));
119                 collectionMap.put(collectionMetaData.getName(), collectionMetaData.getType());
120             }
121             
122             // If no collections were specified for printing, then
123             // continue to the next iteration.
124             if(printCollections.isEmpty()) {
125                 processedEvents++;
126                 if(processedEvents >= maxEvents) { break eventLoop; }
127                 continue eventLoop;
128             }
129             
130             // Otherwise, print the data.
131             System.out.println(BashParameter.format("Requested Print Collections for Event ", BashParameter.TEXT_LIGHT_BLUE)
132                     + BashParameter.format(Integer.toString(event.getEventNumber()), BashParameter.TEXT_GREEN, BashParameter.PROPERTY_BOLD));
133             collectionPrintLoop:
134             for(int i = 0; i < printCollections.size(); i++) {
135                 // Get the class type of the collection.
136                 Class<?> collectionType = collectionMap.get(printCollections.get(i));
137                 
138                 // Do nothing unless the event contains the collection.
139                 if(!event.hasCollection(collectionType, printCollections.get(i))) { continue; }
140                 
141                 // Indicate the current collection.
142                 System.out.println("\tPrinting data for collection \""
143                         + BashParameter.format(printCollections.get(i), BashParameter.TEXT_YELLOW, BashParameter.PROPERTY_BOLD) + "\"...");
144                 
145                 // Check to see if the collection exists in this event.
146                 if(!collectionMap.containsKey(printCollections.get(i))) {
147                     System.out.println("\t\tCollection does not exist.");
148                     continue collectionPrintLoop;
149                 }
150                 
151                 // Get the collection data.
152                 List<?> collectionData = event.get(collectionType, printCollections.get(i));
153                 
154                 // Print the data.
155                 Method toStringMethod = null;
156                 try {
157                     toStringMethod = PrintCollectionsUtility.class.getDeclaredMethod("toString", collectionType, int.class);
158                 } catch(NoSuchMethodException e) {
159                     System.out.println("\t\tNo print method exists for object type \"" + collectionType.getSimpleName() + "\".");
160                     continue collectionPrintLoop;
161                 }
162                 for(int j = 0; j < collectionData.size(); j++) {
163                     try {
164                         System.out.print(((String) toStringMethod.invoke(null, collectionData.get(j), 2)));
165                     } catch(IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
166                         System.out.println("\t\tError calling print method for object type \"" + collectionType.getSimpleName() + "\".");
167                         continue collectionPrintLoop;
168                     }
169                     System.out.println();
170                 }
171             }
172             
173             // Increment the number of processed events.
174             processedEvents++;
175             if(processedEvents >= maxEvents) { break eventLoop; }
176         }
177         
178         // Move the collection names into a sorted list.
179         List<Pair<String, String>> collectionNames = new ArrayList<Pair<String, String>>(collectionSet.size());
180         collectionNames.addAll(collectionSet);
181         collectionNames.sort(new Comparator<Pair<String, String>>() {
182             @Override
183             public int compare(Pair<String, String> arg0, Pair<String, String> arg1) {
184                 return arg0.getFirstElement().compareTo(arg1.getFirstElement());
185             }
186         });
187         
188         // Get the length of the longest collection name.
189         int longestNameLength = 0;
190         for(Pair<String, String> collectionName : collectionNames) {
191             longestNameLength = Math.max(longestNameLength, collectionName.getFirstElement().length());
192         }
193         
194         // Print the collections.
195         System.out.println(BashParameter.format("All File Collections:", BashParameter.TEXT_LIGHT_BLUE));
196         for(Pair<String, String> collectionName : collectionNames) {
197             System.out.printf("\tName = " + BashParameter.format("%-" + longestNameLength + "s", BashParameter.TEXT_YELLOW)
198                     + BashParameter.format(" | ", BashParameter.TEXT_LIGHT_GREY, BashParameter.PROPERTY_DIM)
199                     + "Type = " + BashParameter.format("%s", BashParameter.TEXT_YELLOW) + "%n",
200                     collectionName.getFirstElement(), collectionName.getSecondElement());
201         }
202     }

204     /**
205      * Creates an {@link java.util.String String} consisting of the
206      * specified number of indents.
207      * @param level - The number of indents.
208      * @return Returns a <code>String</code> with the specified
209      * number of indents.
210      */
211     private static final String getIndent(int level) {
212         StringBuffer outputBuffer = new StringBuffer();
213         for(int i = 0; i < level; i++) {
214             outputBuffer.append('\t');
215         }
216         return outputBuffer.toString();
217     }

219     /**
220      * Creates s textual representation of the data stored in a
221      * {@link org.lcsim.event.CalorimeterHit CalorimeterHit}
222      * object.
223      * @param truthHit - The object.
224      * @param indentLevel - The level of indention to prepend.
225      * @return Returns the textual representation.
226      */
227     @SuppressWarnings("unused")
228     private static final String toString(CalorimeterHit truthHit, int indentLevel) {
229         // Create a buffer to store the data and generate the indent
230         // level string.
231         StringBuffer outputBuffer = new StringBuffer();
232         String indent = getIndent(indentLevel);
233         
234         outputBuffer.append(String.format(indent + "Time      :: " + BashParameter.format("%f", BashParameter.TEXT_YELLOW) + " GeV%n", truthHit.getTime()));
235         outputBuffer.append(String.format(indent + "Energy    :: " + BashParameter.format("%f", BashParameter.TEXT_YELLOW) + " GeV%n", truthHit.getCorrectedEnergy()));
236         outputBuffer.append(String.format(indent + "Position  :: <" + BashParameter.format("%f, %f", BashParameter.TEXT_YELLOW) + ">%n", truthHit.getPosition()[0], truthHit.getPosition()[1]));
237         outputBuffer.append(String.format(indent + "Indices   :: <" + BashParameter.format("%d, %d", BashParameter.TEXT_YELLOW) + ">%n",
238                 truthHit.getIdentifierFieldValue("ix"), truthHit.getIdentifierFieldValue("iy")));
239         
240         // Return the result.
241         return outputBuffer.toString();
242     }

244     /**
245      * Creates s textual representation of the data stored in a
246      * {@link org.lcsim.event.SimCalorimeterHit SimCalorimeterHit}
247      * object.
248      * @param truthHit - The object.
249      * @param indentLevel - The level of indention to prepend.
250      * @return Returns the textual representation.
251      */
252     @SuppressWarnings("unused")
253     private static final String toString(SimCalorimeterHit truthHit, int indentLevel) {
254         // Create a buffer to store the data and generate the indent
255         // level string.
256         StringBuffer outputBuffer = new StringBuffer();
257         String indent = getIndent(indentLevel);
258         
259         outputBuffer.append(String.format(indent + "Time      :: " + yellow("%f") + " GeV%n", truthHit.getTime()));
260         outputBuffer.append(String.format(indent + "Energy    :: " + yellow("%f") + " GeV%n", truthHit.getCorrectedEnergy()));
261         outputBuffer.append(String.format(indent + "Position  :: <" + yellow("%f") + ", " + yellow("%f") + ">%n", truthHit.getPosition()[0], truthHit.getPosition()[1]));
262         outputBuffer.append(String.format(indent + "Indices   :: <" + yellow("%d") + ", " + yellow("%d") + ">%n",
263                 truthHit.getIdentifierFieldValue("ix"), truthHit.getIdentifierFieldValue("iy")));
264         outputBuffer.append(String.format(indent + "Particles :: " + yellow("%d") + "%n", truthHit.getMCParticleCount()));
265         for(int i = 0; i < truthHit.getMCParticleCount(); i++) {
266             outputBuffer.append(String.format(indent + "\tPDGID = " + BashParameter.format("%3d", BashParameter.TEXT_YELLOW) + ";  t = " + yellow("%13.2f")
267                     + ";  p = <" + yellow("%5.3f") + ", " + yellow("%5.3f") + ", " + yellow("%5.3f") + ">;   r = <" + yellow("%5.1f") + ", " + yellow("%5.1f")
268                     + ", " + yellow("%5.1f") + ">%n",
269                     truthHit.getMCParticle(i).getPDGID(), truthHit.getMCParticle(i).getProductionTime(), truthHit.getMCParticle(i).getMomentum().x(),
270                     truthHit.getMCParticle(i).getMomentum().y(), truthHit.getMCParticle(i).getMomentum().z(), truthHit.getMCParticle(i).getOriginX(),
271                     truthHit.getMCParticle(i).getOriginY(), truthHit.getMCParticle(i).getOriginZ()));
272         }
273         
274         // Return the result.
275         return outputBuffer.toString();
276     }

302     /**
303      * Creates s textual representation of the data stored in a
304      * {@link org.lcsim.event.GenericObject GenericObject} object.
305      * @param truthHit - The object.
306      * @param indentLevel - The level of indention to prepend.
307      * @return Returns the textual representation.
308      */
309     @SuppressWarnings("unused")
310     private static final String toString(GenericObject obj, int indentLevel) {
311         // Create a buffer to store the data and generate the indent
312         // level string.
313         StringBuffer outputBuffer = new StringBuffer();
314         String indent = getIndent(indentLevel);
315         
316         // Print the integer banks.
317         outputBuffer.append(String.format(indent + "Number of integers :: " + yellow("%d") + "%n", obj.getNInt()));
318         if(obj.getNInt() != 0) {
319             outputBuffer.append(indent + "\t { ");
320             for(int i = 0; i < obj.getNInt(); i++) {
321                 if(i != 0) { outputBuffer.append(", "); }
322                 outputBuffer.append(obj.getIntVal(i));
323             }
324             outputBuffer.append(String.format(" }%n"));
325         }
326         
327         // Print the float banks.
328         outputBuffer.append(String.format(indent + "Number of floats ::   " + yellow("%d") + "%n", obj.getNFloat()));
329         if(obj.getNFloat() != 0) {
330             outputBuffer.append(indent + "\t { ");
331             for(int i = 0; i < obj.getNFloat(); i++) {
332                 if(i != 0) { outputBuffer.append(", "); }
333                 outputBuffer.append(obj.getFloatVal(i));
334             }
335             outputBuffer.append(String.format(" }%n"));
336         }
337         
338         // Print the double banks.
339         outputBuffer.append(String.format(indent + "Number of doubles ::  " + yellow("%d") + "%n", obj.getNDouble()));
340         if(obj.getNDouble() != 0) {
341             outputBuffer.append(indent + "\t { ");
342             for(int i = 0; i < obj.getNDouble(); i++) {
343                 if(i != 0) { outputBuffer.append(", "); }
344                 outputBuffer.append(obj.getDoubleVal(i));
345             }
346             outputBuffer.append(String.format(" }%n"));
347         }
348         
349         // Return the result.
350         return outputBuffer.toString();
351     }
