244                 boolean foundRecoil = false;
245                 boolean foundElectron = false;
246

248                 StdhepRecord record = sr.nextRecord();
249                 // Only process StdhepEvent records.
250 
251                 if (record instanceof StdhepEvent) {
252                     // Convert to an LCCollection of MCParticle objects.
253                     List<MCParticle> mcpcoll = convert((StdhepEvent) record);
254                     for (MCParticle mcp : mcpcoll) {
255                         int pdgid = mcp.getPDGID();
256                         int status = mcp.getGeneratorStatus();
257                         System.out.println("PDG ID = " + pdgid);
258                         if ((abs(pdgid) == 611 || abs(pdgid) == 11) && status == 1) {
259                             double ene = mcp.getEnergy();
260                             Hep3Vector p = mcp.getMomentum();
261                             Hep3Vector o = mcp.getOrigin();
262                             if (debug)
263                                 System.out.println("P(x/y/z) = (" + p.x() + "," + p.y() + ","
264                                         + "" + p.z() + ")");
265                             if (debug)
266                                 System.out.println("o(x/y/z) = (" + o.x() + "," + o.y() + ","
267                                         + "" + o.z() + ")");
268                             int charge = 1;
269                             if (pdgid == 611)
270                                 charge = -1;
271                             HepLorentzVector pl = new BasicHepLorentzVector(ene, p);
272                             Hep3Vector pRotated = CoordinateTransformations.transformVectorToTracking(p);
273                             Hep3Vector oRotated = CoordinateTransformations.transformVectorToTracking(o);
274                             HelixParamCalculator hpc = new HelixParamCalculator(pRotated, oRotated, -charge, -B_FIELD);
275                             double[] trackParameters = new double[5];
276                             Track trkFromMCP = new BaseTrack();
277                             trackParameters[BaseTrack.D0] = hpc.getDCA();
278                             trackParameters[BaseTrack.OMEGA] = hpc.getMCOmega();
279                             trackParameters[BaseTrack.PHI] = hpc.getPhi0();
280                             trackParameters[BaseTrack.TANLAMBDA] = hpc.getSlopeSZPlane();
281                             trackParameters[BaseTrack.Z0] = hpc.getZ0();
282 
283                             if (debug)
284                                 System.out.println("d0=" + trackParameters[BaseTrack.D0]
285                                         + "; omega=" + trackParameters[BaseTrack.OMEGA]
286                                         + "; phi=" + trackParameters[BaseTrack.PHI]
287                                         + "; TanLambda=" + trackParameters[BaseTrack.TANLAMBDA]
288                                         + "; z0=" + trackParameters[BaseTrack.Z0]);
289                             ((BaseTrack) trkFromMCP).setTrackParameters(trackParameters, B_FIELD);
290                             Boolean passEcal = false;
291                             double[] atEcal = {-999, -999, -999};
292                             if (p.z() > _pCut) {
293                                 TrackState state = TrackUtils.extrapolateTrackUsingFieldMap(trkFromMCP, extStartPos, ecalPosition, stepSize, bFieldMap);
294                                 atEcal = state.getReferencePoint();
295                                 trkFromMCP.getTrackStates().add(state);
296 
297                                 passEcal = inFiducialRegion(atEcal[1], atEcal[2], 0);
298                                 if (debug)
299                                     System.out.println("X at ECal = " + atEcal[1] + "; Y at ECal = " + atEcal[2] + "; PassEcal = " + passEcal);
300                                 if (Double.isNaN(atEcal[1]))
301                                     System.out.println("Projection to ECal is NaN...momentum = " + p.z());
302                             }
303 
304                             List<Boolean> passLayerFull = new ArrayList<Boolean>();
305                             for (int ii = 0; ii < nLayers; ii++) {
306                                 boolean inAcceptance = isWithinAcceptance(trkFromMCP, ii + 1);
307                                 passLayerFull.add(inAcceptance);
308                             }
309 
310                             if (pdgid == 611) {  //electron from A'
311                                 passEleFull = passLayerFull;
312                                 passEleEcal = passEcal;
313                                 pEle = pl;
314                                 atEcalEle = atEcal;
315                                 foundElectron = true;
316                             } else if (pdgid == -611 || pdgid == -11) {//positron
317                                 pPos = pl;
318                                 passPosEcal = passEcal;
319                                 passPosFull = passLayerFull;
320                                 atEcalPos = atEcal;
321                             } else if (pdgid == 11)
322                                 if (!foundElectron && trident) {
323 //                            System.out.println("This is a trident????");
324                                     foundElectron = true;
325                                     passEleFull = passLayerFull;
326                                     passEleEcal = passEcal;
327                                     pEle = pl;
328                                     atEcalEle = atEcal;
329                                 }
330                             if (!foundRecoil) {
331                                 foundRecoil = true;
332                                 pRecoil = pl;
333                                 passRecFull = passLayerFull;
334                                 passRecEcal = passEcal;
335                                 atEcalRec = atEcal;
336                             }
337                         }
338                     }
339                     boolean passEleMinLayers = false;
340                     boolean passPosMinLayers = false;
341                     boolean passRecMinLayers = false;
342                     int nlayPass = 0;
343                     for (boolean pass : passEleFull)
344                         if (pass)
345                             nlayPass++;
346                     if (nlayPass >= minLayerHits)
347                         passEleMinLayers = true;
348 
349                     nlayPass = 0;
350                     for (boolean pass : passPosFull)
351                         if (pass)
352                             nlayPass++;
353                     if (nlayPass >= minLayerHits)
354                         passPosMinLayers = true;
355                     System.out.println("Number of layers hit: positron = " + nlayPass);
356                     nlayPass = 0;
357                     for (boolean pass : passRecFull)
358                         if (pass)
359                             nlayPass++;
360                     if (nlayPass >= minLayerHits)
361                         passRecMinLayers = true;
362 
363                     pPosPassGen.fill(pPos.v3().magnitude());
364                     if (passPosEcal)
365                         pPosPassECal.fill(pPos.v3().magnitude());
366                     if (passPosEcal && passPosMinLayers)
367                         pPosPassECalAndMinLayers.fill(pPos.v3().magnitude());
368                     if((passEleMinLayers||passRecMinLayers)&&passPosEcal && passPosMinLayers)
369                          pPosPassECalAndPosMinLayersAndEleMinLayers.fill(pPos.v3().magnitude());
370                     if (passPosEcal && !passPosMinLayers)
371                         System.out.println("Passed ECal but not passPosMinLayers");
372                     if (!passPosEcal && passPosMinLayers)
373                         System.out.println("Passed passPosMinLayers but not ECal");
374 
375                     nread++;
376                 }
377 
378             }
379 
380         } catch (EOFException e) {
381             // End of Stdhep file.
382         }
383 
384         return nread;
385     }

601     private static boolean inFiducialRegion(double x, double y, double offsetY) {
602         boolean in_fid = false;
603 
604         y = Math.abs(y);
605 
606         if (x > x_edge_low + shiftECalinX && x < x_edge_high + shiftECalinX && y > y_edge_low + offsetY && y < y_edge_high + offsetY)
607             if ((x > x_gap_low + shiftECalinX && x < x_gap_high + shiftECalinX && y > y_edge_low + offsetY && y < y_gap_high + offsetY) != true)
608                 in_fid = true;
609         return in_fid;
610     }
