584     private double getInvMass(Track track1, StraightLineTrack slt1, Track track2, StraightLineTrack slt2) {
585         double esum = 0.;
586         double pxsum = 0.;
587         double pysum = 0.;
588         double pzsum = 0.;
589         double chargesum = 0.;
590         double me = 0.000511;
591         // Loop over jets
592 
593         double p1x = track1.getTrackStates().get(0).getMomentum()[0];
594         double p1y = track1.getTrackStates().get(0).getMomentum()[1];
595         double p1z = track1.getTrackStates().get(0).getMomentum()[2];
596         double p1mag2 = p1x * p1x + p1y * p1y + p1z * p1z;
597         double e1 = Math.sqrt(p1mag2 + me * me);
598         double dydx1 = slt1.dydx();
599         double dzdx1 = slt1.dzdx();
600         double s1sq = 1 + 1 / (dydx1 * dydx1) + (dzdx1 * dzdx1) / (dydx1 * dydx1);
601         double truep1y = Math.sqrt(p1mag2 / s1sq);
602         if (dydx1 < 0)
603             truep1y = -truep1y;
604         double truep1x = truep1y / dydx1;
605         double truep1z = dzdx1 * truep1x;
606 
607         double p2x = track2.getTrackStates().get(0).getMomentum()[0];
608         double p2y = track2.getTrackStates().get(0).getMomentum()[1];
609         double p2z = track2.getTrackStates().get(0).getMomentum()[2];
610         double p2mag2 = p2x * p2x + p2y * p2y + p2z * p2z;
611         double e2 = Math.sqrt(p2mag2 + me * me);
612 
613         double dydx2 = slt2.dydx();
614         double dzdx2 = slt2.dzdx();
615         double s2sq = 1 + 1 / (dydx2 * dydx2) + (dzdx2 * dzdx2) / (dydx2 * dydx2);
616         double truep2y = Math.sqrt(p2mag2 / s2sq);
617         if (dydx2 < 0)
618             truep2y = -truep2y;
619         double truep2x = truep2y / dydx2;
620         double truep2z = dzdx2 * truep2x;
621 
622         pxsum =
623                 truep1x + truep2x;
624         pysum =
625                 truep1y + truep2y;
626         pzsum =
627                 truep1z + truep2z;
628 
629         esum =
630                 e1 + e2;
631 //        double p1dotp2 = p1x * p2x + p1y * p2y + p1z * p2z;
632         double p1dotp2 = truep1x * truep2x + truep1y * truep2y + truep1z * truep2z;
633         double e1e2 = e1 * e2;
634         double invmass = Math.sqrt(2 * me * me + 2 * (e1e2 - p1dotp2));
635         // Compute total momentum and hence event mass
636         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
637         double evtmass = Math.sqrt(esum * esum - psum * psum);
638 //        System.out.println("invmass= " + invmass + "; evtmass=" + evtmass);
639         return invmass;
640     }

643     private double findDoca(double y, double z, double px, double py, double pz) {
644         double xoca = 0;
645         double sy = py / px;
646         double sz = pz / px;
647         xoca =
648                 -(y * sy + z * sz) / (sy * sy + sz + sz);
649         double doca = Math.sqrt(Math.pow(y + sy * xoca, 2) + Math.pow(z + sz * xoca, 2));
650         return doca;
651     }

654     private double findXoca(double y, double z, double px, double py, double pz) {
655         double xoca = 0;
656         double sy = py / px;
657         double sz = pz / px;
658         xoca =
659                 -(y * sy + z * sz) / (sy * sy + sz + sz);
660         return xoca;
661     }

663     private double[] findPoca(double y, double z, double px, double py, double pz) {
664         double poca[] = {0, 0, 0};
665         double sy = py / px;
666         double sz = pz / px;
667         poca[0] = -(y * sy + z * sz) / (sy * sy + sz * sz);
668         poca[1] = y + sy * poca[0];
669         poca[2] = z + sz * poca[0];
670         return poca;
671     }

693     private Hep3Vector getTrueMomentum(Track track1, StraightLineTrack slt1) {
694         double[] truep = {0, 0, 0};
695         double me = 0.000511;
696         double p1x = track1.getTrackStates().get(0).getMomentum()[0];
697         double p1y = track1.getTrackStates().get(0).getMomentum()[1];
698         double p1z = track1.getTrackStates().get(0).getMomentum()[2];
699         double p1mag2 = p1x * p1x + p1y * p1y + p1z * p1z;
700         double e1 = Math.sqrt(p1mag2 + me * me);
701         double dydx1 = slt1.dydx();
702         double dzdx1 = slt1.dzdx();
703         double s1sq = 1 + 1 / (dydx1 * dydx1) + (dzdx1 * dzdx1) / (dydx1 * dydx1);
704         truep[1] = Math.sqrt(p1mag2 / s1sq);
705         if (dydx1 < 0)
706             truep[1] = -truep[1];
707         truep[0] = truep[1] / dydx1;
708         truep[2] = dzdx1 * truep[0];
709         return new BasicHep3Vector(truep[0], truep[1], truep[2]);
710     }

720     private HelicalTrackStrip getNearestHit(HelicalTrackStrip cl, List<HelicalTrackHit> toththits) {
721         Hep3Vector corigin = cl.origin();
722         Hep3Vector u = cl.u();
723         double umeas = cl.umeas();
724         Hep3Vector uvec = VecOp.mult(umeas, u);
725         Hep3Vector clvec = VecOp.add(corigin, uvec);
726         int layer = cl.layer();
727         HelicalTrackStrip nearest = null;
728         double mindist = 99999999;
729         for (HelicalTrackHit hth : toththits) {
730             HelicalTrackCross cross = (HelicalTrackCross) hth;
731             for (HelicalTrackStrip str : cross.getStrips())
732                 if (str != cl) {
733                     Hep3Vector strorigin = str.origin();
734                     Hep3Vector stru = str.u();
735                     double strumeas = str.umeas();
736                     Hep3Vector struvec = VecOp.mult(strumeas, stru);
737                     Hep3Vector strvec = VecOp.add(strorigin, struvec);
738                     int strlayer = cl.layer();
739                     if (layer == strlayer && VecOp.sub(clvec, strvec).magnitude() < mindist) {
740                         mindist = VecOp.sub(clvec, strvec).magnitude();
741                         nearest =
742                                 str;
743                     }
744 
745                 }
746         }
747 
748         return nearest;
749     }

751     private Double getNearestDistance(HelicalTrackStrip cl, List<HelicalTrackHit> toththits) {
752         Hep3Vector corigin = cl.origin();
753         Hep3Vector u = cl.u();
754         double umeas = cl.umeas();
755         Hep3Vector uvec = VecOp.mult(umeas, u);
756         Hep3Vector clvec = VecOp.add(corigin, uvec);
757         int layer = cl.layer();
758         HelicalTrackStrip nearest = null;
759         Double mindist = 99999999.0;
760         for (HelicalTrackHit hth : toththits) {
761             HelicalTrackCross cross = (HelicalTrackCross) hth;
762             for (HelicalTrackStrip str : cross.getStrips())
763                 if (str != cl) {
764                     Hep3Vector strorigin = str.origin();
765                     Hep3Vector stru = str.u();
766                     double strumeas = str.umeas();
767                     Hep3Vector struvec = VecOp.mult(strumeas, stru);
768                     Hep3Vector strvec = VecOp.add(strorigin, struvec);
769                     int strlayer = str.layer();
770                     if (layer == strlayer && VecOp.sub(clvec, strvec).magnitude() < Math.abs(mindist)) {
771                         mindist = VecOp.sub(clvec, strvec).magnitude();
772                         if (Math.abs(clvec.z()) > Math.abs(strvec.z()))
773                             mindist = -mindist;
774                         nearest =
775                                 str;
776                     }
777 
778                 }
779         }
780         return mindist;
781     }
