77      private boolean isTopTrack(Track t) {
78          List<TrackerHit> hits = t.getTrackerHits();
79          int n[] = {0, 0};
80          int nHits = hits.size();
81          for (TrackerHit h : hits) {
82              HpsSiSensor sensor = ((HpsSiSensor) ((RawTrackerHit) h.getRawHits().get(0)).getDetectorElement());
83              if (sensor.isTopLayer()) {
84                  n[0] += 1;
85              } else {
86                  n[1] += 1;
87              }
88          }
89          if (n[0] == nHits && n[1] == 0) {
90              return true;
91          }
92          if (n[1] == nHits && n[0] == 0) {
93              return false;
94          }
95          throw new RuntimeException("mixed top and bottom hits on same track");
96  
97      }

99      private void setupSensors(EventHeader event) {
100         List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
101         EventHeader.LCMetaData meta = event.getMetaData(rawTrackerHits);
102         // Get the ID dictionary and field information.
103         IIdentifierDictionary dict = meta.getIDDecoder().getSubdetector().getDetectorElement().getIdentifierHelper().getIdentifierDictionary();
104         int fieldIdx = dict.getFieldIndex("side");
105         int sideIdx = dict.getFieldIndex("strip");
106         for (RawTrackerHit hit : rawTrackerHits) {
107             // The "side" and "strip" fields needs to be stripped from the ID for sensor lookup.
108             IExpandedIdentifier expId = dict.unpack(hit.getIdentifier());
109             expId.setValue(fieldIdx, 0);
110             expId.setValue(sideIdx, 0);
111             IIdentifier strippedId = dict.pack(expId);
112             // Find the sensor DetectorElement.
113             List<IDetectorElement> des = DetectorElementStore.getInstance().find(strippedId);
114             if (des == null || des.size() == 0) {
115                 throw new RuntimeException("Failed to find any DetectorElements with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
116             } else if (des.size() == 1) {
117                 hit.setDetectorElement((SiSensor) des.get(0));
118             } else {
119                 // Use first sensor found, which should work unless there are sensors with duplicate IDs.
120                 for (IDetectorElement de : des) {
121                     if (de instanceof SiSensor) {
122                         hit.setDetectorElement((SiSensor) de);
123                         break;
124                     }
125                 }
126             }
127             // No sensor was found.
128             if (hit.getDetectorElement() == null) {
129                 throw new RuntimeException("No sensor was found for hit with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
130             }
131         }
132     }
