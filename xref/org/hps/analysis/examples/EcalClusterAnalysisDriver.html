1   package org.hps.analysis.examples;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   import org.hps.recon.ecal.cluster.ClusterUtilities;
6   import org.hps.record.triggerbank.TriggerModule;
7   import org.lcsim.event.Cluster;
8   import org.lcsim.event.EventHeader;
9   import org.lcsim.util.Driver;
10  import org.lcsim.util.aida.AIDA;
11  
12  public class EcalClusterAnalysisDriver extends Driver {
13  
14      private AIDA aida = AIDA.defaultInstance();
15  
16      public void process(EventHeader event) {
17          if (event.hasCollection(Cluster.class, "EcalClustersCorr")) {
18              List<Cluster> clusters = event.get(Cluster.class, "EcalClustersCorr");
19              aida.histogram1D("Number of Clusters in Event", 10, 0., 10.).fill(clusters.size());
20              double maxEnergy = 0;
21              List<Double> times = new ArrayList<Double>();
22              List<Double> energies = new ArrayList<Double>();
23              List<Boolean> isFiducial = new ArrayList<Boolean>();
24              List<Boolean> isTop = new ArrayList<Boolean>();
25              for (Cluster cluster : clusters) {
26                  if (cluster.getEnergy() > maxEnergy) {
27                      maxEnergy = cluster.getEnergy();
28                  }
29                  double[] cPos = cluster.getPosition();
30                  times.add(ClusterUtilities.findSeedHit(cluster).getTime());
31                  energies.add(cluster.getEnergy());
32                  isFiducial.add(TriggerModule.inFiducialRegion(cluster));
33                  isTop.add(cluster.getPosition()[1] > 0.);
34                  aida.histogram2D("cluster x vs y", 320, -270.0, 370.0, 90, -90.0, 90.0).fill(cPos[0], cPos[1]);
35                  String half = cluster.getPosition()[1] > 0. ? "Top ": "Bottom ";
36                  aida.histogram1D(half+"Cluster Energy", 100, 0., 6.).fill(cluster.getEnergy());
37              }
38              if (clusters.size() == 2) {
39                  double e0 = energies.get(0);
40                  double e1 = energies.get(1);
41                  double esum = e0 + e1;
42                  aida.histogram1D("Two-Cluster Energy sum", 100, 0., 6.).fill(energies.get(0) + energies.get(1));
43                  aida.histogram1D("Two-Cluster time difference", 100, -5, 5.).fill(times.get(0) - times.get(1));
44                  boolean isTopBottom = false;
45                  if (isTop.get(0) && !isTop.get(1)) {
46                      isTopBottom = true;
47                      aida.histogram1D("Two-Cluster Energy sum top-bottom", 100, 0., 6.).fill(energies.get(0) + energies.get(1));
48                      aida.histogram1D("Two-Cluster time difference top-bottom", 100, -5, 5.).fill(times.get(0) - times.get(1));
49                  }
50                  if (!isTop.get(0) && isTop.get(1)) {
51                      isTopBottom = true;
52                      aida.histogram1D("Two-Cluster time difference top-bottom", 100, -5, 5.).fill(times.get(1) - times.get(0));
53                      aida.histogram1D("Two-Cluster Energy sum top-bottom", 100, 0., 6.).fill(energies.get(0) + energies.get(1));
54  
55                  }
56                  if (isTop.get(0) && isTop.get(1)) {
57                      aida.histogram1D("Two-Cluster time difference top-top", 100, -5, 5.).fill(times.get(0) - times.get(1));
58                  }
59                  if (!isTop.get(0) && !isTop.get(1)) {
60                      aida.histogram1D("Two-Cluster time difference bottom-bottom", 100, -5, 5.).fill(times.get(0) - times.get(1));
61                  }
62                  if (isFiducial.get(0) && isFiducial.get(1)) {
63                      double dt = -9999.;
64                      aida.histogram1D("Two-Cluster Energy sum both fiducial", 100, 0., 6.).fill(energies.get(0) + energies.get(1));
65                      aida.histogram1D("Two-Cluster time difference both fiducial", 100, -5, 5.).fill(times.get(0) - times.get(1));
66                      if (isTop.get(0) && !isTop.get(1)) {
67                          aida.histogram1D("Two-Cluster Energy sum both fiducial top-bottom", 100, 0., 6.).fill(energies.get(0) + energies.get(1));
68                          aida.histogram1D("Two-Cluster time difference both fiducial top-bottom", 100, -5, 5.).fill(times.get(0) - times.get(1));
69                          dt = times.get(0) - times.get(1);
70                      }
71                      if (!isTop.get(0) && isTop.get(1)) {
72                          aida.histogram1D("Two-Cluster time difference both fiducial top-bottom", 100, -5, 5.).fill(times.get(1) - times.get(0));
73                          aida.histogram1D("Two-Cluster Energy sum both fiducial top-bottom", 100, 0., 6.).fill(energies.get(0) + energies.get(1));
74                          dt = times.get(1) - times.get(0);
75                      }
76                      if (isTopBottom) {
77                          aida.histogram2D("Two-Cluster both fiducial top-bottom E0 vs E1", 100, 0., 5., 100, 0., 5.).fill(energies.get(0), energies.get(1));
78                          aida.histogram2D("Two-Cluster both fiducial top-bottom Esum vs E0", 100, 0., 5., 100, 0., 5.).fill(energies.get(0), energies.get(0) + energies.get(1));
79                          if (e0 > 1.8 && e0 < 2.8) {
80                              aida.histogram1D("Two-Cluster both fiducial Energy sum top-bottom 1.8<e0<2.8", 100, 0., 6.).fill((esum));
81                              aida.histogram1D("Two-Cluster both fiducial time difference top-bottom 1.8<e0<2.8", 100, -5., 5.).fill((dt));
82                              if (dt > 0. && dt < 1.) {
83                                  aida.histogram1D("Two-Cluster both fiducial Energy sum top-bottom 1.8<e0<2.8 0.<dt<1.", 100, 0., 6.).fill((esum));
84                              }
85                          }
86                      }
87                  }
88              }
89              aida.histogram1D("Max Cluster Energy", 100, 0., 6.).fill(maxEnergy);
90          }
91      }
92  }
