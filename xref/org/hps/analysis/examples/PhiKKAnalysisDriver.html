22  /**
23   * Simple analysis to search for φ→K⁺K⁻ Loop over existing V0s, recalculate mass
24   * using different particle hypotheses for the constituent particles.
25   */
26  public class PhiKKAnalysisDriver extends Driver {
27  
28      private AIDA aida = AIDA.defaultInstance();
29      String vertexCollectionName = "UnconstrainedV0Vertices";
30      String[] particleType = {"electron", "positron"};
31      String[] names = {"X", "Y", "Z"};
32      double[] p1 = new double[4];
33      double[] p2 = new double[4];
34      double[] pV = new double[3];
35      double emass = 0.000511;
36      double kmass = 0.493667;
37      double[] masses = {0.000511, 0.10566, 0.13957, 0.493667};
38      String[] particleNames = {"electron", "muon", "pion", "kaon"};
39      double[] minVals = {0.0, 0.15, 0.25, 0.95};
40      double[] maxVals = {0.85, 0.85, 0.85, 1.45};
41      double phimass = 1.019;
42      double emass2 = emass * emass;
43      double kmass2 = kmass * kmass;
44      boolean debug = false;
45      double clusterEcut = 0.3;
46  
47      // skimming
48      boolean _skimEvents = true;
49      int _numberOfEventsSelected;
50  
51      protected void process(EventHeader event) {
52          boolean skipEvent = true;
53          // just a few sanity checks
54          List<Cluster> eventClusters = event.get(Cluster.class, "EcalClustersCorr");
55          aida.histogram1D("Number of Clusters in Event", 10, 0., 10.).fill(eventClusters.size());
56          boolean isMC = event.hasCollection(MCParticle.class, "MCParticle");
57          MCParticle mcp = null;
58          if (isMC) {
59              if (!event.get(MCParticle.class, "MCParticle").isEmpty()) {
60                  mcp = event.get(MCParticle.class, "MCParticle").get(0);
61              }
62          }
63          for (Cluster c : eventClusters) {
64              aida.histogram1D("event cluster nHits", 20, 0., 20.).fill(c.getCalorimeterHits().size());
65              aida.histogram1D("event cluster energy", 100, 0., 1.5).fill(c.getEnergy());
66              aida.histogram2D("event cluster x vs y", 320, -270.0, 370.0, 90, -90.0, 90.0).fill(c.getPosition()[0], c.getPosition()[1]);
67              if (mcp != null) {
68                  aida.tree().mkdirs("all clusters");
69                  aida.tree().cd("all clusters");
70                  analyzeCluster(c, "MC " + mcp.getType().getName());
71                  aida.tree().cd("..");
72              }
73          }
74          List<ReconstructedParticle> V0List = event.get(ReconstructedParticle.class, "UnconstrainedV0Candidates");
75  //        if (V0List.size() != 2) {
76  //            return;
77  //        }
78          List<Vertex> vertices = event.get(Vertex.class, vertexCollectionName);
79          for (Vertex v : vertices) {
80              boolean goodMuPair = false;
81              ReconstructedParticle muPlus = null;
82              ReconstructedParticle muMinus = null;
83  
84              ReconstructedParticle V0 = v.getAssociatedParticle();
85  
86              Cluster[] clusters = new Cluster[2];
87              List<ReconstructedParticle> particles = V0.getParticles();
88              for (int i = 0; i < 2; ++i) {
89                  ReconstructedParticle rp = particles.get(i);
90                  Cluster clus = null;
91                  if (rp.getClusters().size() == 1) {
92                      clus = rp.getClusters().get(0);
93                      clusters[i] = clus;
94                  }
95                  boolean isGBL = TrackType.isGBL(rp.getTracks().get(0).getType());
96                  String trackType = isGBL ? "gbl" : "matchedTracks";
97                  aida.tree().mkdirs(trackType);
98                  aida.tree().cd(trackType);
99                  Track t = rp.getTracks().get(0);
100                 int nHits = t.getTrackerHits().size();
101                 double dEdx = t.getdEdx();
102                 aida.histogram1D(particleType[i] + " track nHits", 20, 0., 20.).fill(nHits);
103                 aida.histogram1D(particleType[i] + " track dEdx", 200, 0., 0.003).fill(dEdx);
104                 // cluster plots
105                 if (clus != null) {
106                     aida.histogram1D(particleType[i] + " cluster nHits", 20, 0., 20.).fill(clus.getCalorimeterHits().size());
107                     aida.histogram1D(particleType[i] + " cluster energy", 100, 0., 1.5).fill(clus.getEnergy());
108                     aida.histogram2D(particleType[i] + " cluster x vs y", 320, -270.0, 370.0, 90, -90.0, 90.0).fill(clus.getPosition()[0], clus.getPosition()[1]);
109                     if (clus.getEnergy() < clusterEcut) {
110                         aida.histogram1D(particleType[i] + " MIP track nHits", 20, 0., 20.).fill(nHits);
111                         aida.histogram1D(particleType[i] + " MIP track dEdx", 200, 0., 0.003).fill(dEdx);
112                         aida.histogram1D(particleType[i] + " MIP cluster nHits", 20, 0., 20.).fill(clus.getCalorimeterHits().size());
113                     }
114                 }
115                 aida.tree().cd("..");
116             }
117             ReconstructedParticle ele = particles.get(0);
118             ReconstructedParticle pos = particles.get(1);
119 
120             boolean isGBL = TrackType.isGBL(ele.getTracks().get(0).getType());
121             String trackType = isGBL ? "gbl" : "matchedTracks";
122 
123             boolean noClusters = (ele.getClusters().isEmpty() && pos.getClusters().isEmpty());
124             boolean oneCluster = ((ele.getClusters().size() == 1 && pos.getClusters().isEmpty()) || (ele.getClusters().size() == 0 && pos.getClusters().size() == 1));
125             boolean twoClusters = (ele.getClusters().size() == 1 && pos.getClusters().size() == 1);
126 
127             aida.tree().mkdirs(trackType);
128             aida.tree().cd(trackType);
129 
130             Hep3Vector vertexPosition = v.getPosition();
131             Map<String, Double> vals = v.getParameters();
132             // System.out.println(vals);
133             p1[0] = vals.get("p1X");
134             p1[1] = vals.get("p1Y");
135             p1[2] = vals.get("p1Z");
136             p2[0] = vals.get("p2X");
137             p2[1] = vals.get("p2Y");
138             p2[2] = vals.get("p2Z");
139             double v0p = vals.get("V0P");
140             pV[0] = vals.get("V0Px");
141             pV[1] = vals.get("V0Py");
142             pV[2] = vals.get("V0Pz");
143             double k1 = 0;
144             double k2 = 0.;
145             for (int i = 0; i < 3; ++i) {
146                 k1 += p1[i] * p1[i];
147                 k2 += p2[i] * p2[i];
148             }
149             k1 = sqrt(k1 + kmass2);
150             k2 = sqrt(k2 + kmass2);
151             Momentum4Vector kvec1 = new Momentum4Vector(p1[0], p1[1], p1[2], k1);
152             Momentum4Vector kvec2 = new Momentum4Vector(p2[0], p2[1], p2[2], k2);
153             Lorentz4Vector kksum = kvec1.plus(kvec2);
154             double kkmass = kksum.mass();
155             double invMass = vals.get("invMass");
156 
157             aida.histogram1D("vertex invariant mass", 100, 0., 0.4).fill(invMass);
158             aida.histogram1D("vertex x", 100, -3., 3.).fill(vertexPosition.x());
159             aida.histogram1D("vertex y", 100, -1., 1.).fill(vertexPosition.y());
160             aida.histogram1D("vertex z", 200, -25., 0.).fill(vertexPosition.z());
161             aida.histogram2D("vertex x vs y", 100, -3., 3., 100, -1., 1.).fill(vertexPosition.x(), vertexPosition.y());
162             aida.histogram1D("vertex invariant mass K+K-", 100, 0.95, 1.95).fill(kkmass);
163             aida.histogram1D("vertex invariant mass phi search", 100, 0.98, 1.06).fill(kkmass);
164             aida.histogram1D("vertex momentum", 100, 0., 5.0).fill(v0p);
165             if (noClusters) {
166                 aida.histogram1D("vertex invariant mass no clusters", 100, 0., 0.4).fill(invMass);
167                 aida.histogram1D("vertex invariant mass phi search no clusters", 100, 0.98, 1.06).fill(kkmass);
168             }
169             if (oneCluster) {
170                 aida.histogram1D("vertex invariant mass one cluster", 100, 0., 0.4).fill(invMass);
171                 aida.histogram1D("vertex invariant mass phi search one cluster", 100, 0.98, 1.06).fill(kkmass);
172             }
173             if (twoClusters) {
174                 aida.histogram1D("vertex invariant mass two clusters", 100, 0., 0.4).fill(invMass);
175                 aida.histogram1D("vertex invariant mass phi search two clusters", 100, 0.98, 1.06).fill(kkmass);
176                 aida.histogram2D("electron vs positron cluster energy", 100, 0., 1.5, 100, 0., 1.5).fill(clusters[0].getEnergy(), clusters[1].getEnergy());
177                 if (clusters[0].getEnergy() < clusterEcut && clusters[1].getEnergy() < clusterEcut) {
178                     double deltaT = ClusterUtilities.findSeedHit(clusters[0]).getTime() - ClusterUtilities.findSeedHit(clusters[1]).getTime();
179                     if (abs(deltaT) < 5.) {
180                         //two in-time MIP-like clusters: keep
181                         skipEvent = false;
182                         aida.histogram1D(particleType[0] + " track momentum", 200, 0., 4.5).fill(ele.getMomentum().magnitude());
183                         aida.histogram1D(particleType[1] + " track momentum", 200, 0., 4.5).fill(pos.getMomentum().magnitude());
184                         aida.histogram2D("electron vs positron track momentum", 100, 0., 4.5, 100, 0., 4.5).fill(ele.getMomentum().magnitude(), pos.getMomentum().magnitude());
185                         aida.histogram1D(particleType[0] + " two MIP clusters nHits", 20, 0., 20.).fill(clusters[0].getCalorimeterHits().size());
186                         aida.histogram1D(particleType[1] + " two MIP clusters nHits", 20, 0., 20.).fill(clusters[1].getCalorimeterHits().size());
187                         aida.histogram1D(particleType[0] + " two MIP clusters energy", 100, 0., 0.3).fill(clusters[0].getEnergy());
188                         aida.histogram1D(particleType[1] + " two MIP clusters energy", 100, 0., 0.3).fill(clusters[1].getEnergy());
189 
190                         aida.histogram1D(particleType[0] + " two MIP clusters track dEdx", 200, 0., 0.003).fill(ele.getTracks().get(0).getdEdx());
191                         aida.histogram1D(particleType[1] + " two MIP clusters track dEdx", 200, 0., 0.003).fill(pos.getTracks().get(0).getdEdx());
192 
193                         aida.histogram1D("vertex invariant mass phi search two clusters below " + clusterEcut, 100, 0.98, 1.06).fill(kkmass);
194                         aida.histogram1D("vertex invariant mass two clusters below " + clusterEcut, 100, 0., 0.4).fill(invMass);
195                         aida.histogram1D("two cluster deltaT", 100, -5., 5.).fill(deltaT);
196                         aida.histogram2D(particleType[0] + " cluster x vs y two clusters below " + clusterEcut, 320, -270.0, 370.0, 90, -90.0, 90.0).fill(clusters[0].getPosition()[0], clusters[0].getPosition()[1]);
197                         aida.histogram2D(particleType[1] + " cluster x vs y two clusters below " + clusterEcut, 320, -270.0, 370.0, 90, -90.0, 90.0).fill(clusters[1].getPosition()[0], clusters[1].getPosition()[1]);
198                         aida.histogram1D("vertex x MIP clusters", 100, -3., 3.).fill(vertexPosition.x());
199                         aida.histogram1D("vertex y MIP clusters", 100, -1., 1.).fill(vertexPosition.y());
200                         aida.histogram1D("vertex z MIP clusters", 200, -25., 0.).fill(vertexPosition.z());
201                         aida.histogram2D("vertex x vs y MIP clusters", 100, -3., 3., 100, -1., 1.).fill(vertexPosition.x(), vertexPosition.y());
202                         for (int j = 0; j < masses.length; ++j) {
203                             double mass2 = masses[j] * masses[j];
204                             k1 = 0;
205                             k2 = 0.;
206                             for (int i = 0; i < 3; ++i) {
207                                 k1 += p1[i] * p1[i];
208                                 k2 += p2[i] * p2[i];
209                             }
210                             k1 = sqrt(k1 + mass2);
211                             k2 = sqrt(k2 + mass2);
212                             kvec1 = new Momentum4Vector(p1[0], p1[1], p1[2], k1);
213                             kvec2 = new Momentum4Vector(p2[0], p2[1], p2[2], k2);
214                             kksum = kvec1.plus(kvec2);
215                             kkmass = kksum.mass();
216 //                        aida.histogram1D("vertex invariant mass phi search two clusters below " + clusterEcut + " " + particleNames[j] + " hypothesis", 200, 0., 2.).fill(kkmass);
217                             aida.histogram1D("vertex invariant mass phi search two clusters below " + clusterEcut + " " + particleNames[j] + " hypothesis", 200, minVals[j], maxVals[j]).fill(kkmass);
218                         } //end of loop over particle hypotheses
219                         goodMuPair = true;
220                         muPlus = pos;
221                         muMinus = ele;
222                         analyzeCluster(muPlus.getClusters().get(0), "mu+");
223                         analyzeCluster(muMinus.getClusters().get(0), "mu-");
224                     } // end of deltaT cut
225                 }
226             }
227             aida.tree().cd("..");
228         }
229         if (skipEvent && _skimEvents) {
230             throw new Driver.NextEventException();
231         } else {
232             _numberOfEventsSelected++;
233         }
234     }

238         System.out.println("Selected " + _numberOfEventsSelected + " events");
239     }

242         _skimEvents = b;

245     void analyzeCluster(Cluster cluster, String type) {
246         if (cluster.getCalorimeterHits().size() == 1) {
247             aida.tree().mkdirs("clusterAnalysis");
248             aida.tree().cd("clusterAnalysis");
249             CalorimeterHit seed = cluster.getCalorimeterHits().get(0);
250             int ix = seed.getIdentifierFieldValue("ix");
251             int iy = seed.getIdentifierFieldValue("iy");
252             if (ix > 5 && type.equals("MC mu+")) {
253                 aida.histogram1D(type + " crystal energy", 100, 0.1, 0.3).fill(cluster.getEnergy());
254             }
255             if (ix < -5 && type.equals("MC mu-")) {
256                 aida.histogram1D(type + " crystal energy", 100, 0.1, 0.3).fill(cluster.getEnergy());
257             }
258             aida.histogram2D("cluster ix vs iy",47, -23.5, 23.5, 11, -5.5, 5.5).fill(ix, iy);
259             aida.histogram1D(ix + " " + iy + " " + type + " crystal energy", 50, 0.1, 0.3).fill(cluster.getEnergy());
260             aida.tree().cd("..");
261         }
