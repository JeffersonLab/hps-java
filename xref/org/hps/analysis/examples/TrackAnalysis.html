79      private void doAnalysis(Track trk, RelationalTable hittomc, RelationalTable rthtosimhit, RelationalTable hittostrip, RelationalTable hittorotated) {
80  
81          //  Get the number of hits on the track
82          _nhits = trk.getTrackerHits().size();
83  
84          _hasLayerOne = false;
85          //  Loop over the hits on the track and make sure we have HelicalTrackHits (which contain the MC particle)
86          for (TrackerHit hit : trk.getTrackerHits()) {
87              //  get the set of MCParticles associated with this hit and update the hit count for each MCParticle
88              Set<MCParticle> mclist = hittomc.allFrom(hit);
89              for (MCParticle mcp : mclist) {
90                  Integer mchits = 0;
91                  if (mcmap.containsKey(mcp))
92                      mchits = mcmap.get(mcp);
93                  mchits++;
94                  mcmap.put(mcp, mchits);
95              }
96              if (hit instanceof HelicalTrackCross)
97                  countHit((HelicalTrackCross) hit);
98              else if (hit instanceof HelicalTrack2DHit)
99                  countHit((HelicalTrack2DHit) hit);
100             else if (!(hit  instanceof HelicalTrack2DHit )) //probably SOITrackerHit
101                 countHit(hit, rthtosimhit, hittostrip, hittorotated);
102         }
103 
104         //  Find the MCParticle that has the most hits on the track
105         int nbest = 0;
106         MCParticle mcbest = null;
107         for (MCParticle mcp : mcmap.keySet()) {
108             int count = mcmap.get(mcp);
109             if (count > nbest) {
110                 nbest = count;
111                 mcbest = mcp;
112             }
113         }
114 
115         if (nbest > 0)
116             _mcp = mcbest;
117         _purity = (double) nbest / (double) _nhits;
118         _nbadhits = _nhits - nbest;
119 
120 //single strip layer accounting.
121         int nbestAll = 0;
122         MCParticle mcbestAll = null;
123         for (MCParticle mcp : mcmapAll.keySet()) {
124             int count = mcmapAll.get(mcp);
125             if (count > nbestAll) {
126                 nbestAll = count;
127                 mcbestAll = mcp;
128             }
129         }
130 
131         if (nbestAll > 0)
132             _mcpNew = mcbestAll;
133         _purityNew = (double) nbestAll / (double) _nhitsNew;
134         _nbadhitsNew = _nhitsNew - nbestAll;
135 
136         for (TrackerHit hit : trk.getTrackerHits())
137             if (hit instanceof HelicalTrackCross)
138                 checkForBadHit((HelicalTrackCross) hit);
139 
140         if (_nAxialhits > 0)
141             if (mcmapAxial.containsKey(_mcpNew))
142                 _nbadAxialhits = _nAxialhits - mcmapAxial.get(_mcpNew);
143             else
144                 _nbadAxialhits = _nAxialhits;
145         if (_nZhits > 0)
146             if (mcmapZ.containsKey(_mcpNew))
147                 _nbadZhits = _nZhits - mcmapZ.get(_mcpNew);
148             else
149                 _nbadZhits = _nZhits;
150     }

152     private void countHit(HelicalTrackCross cross) {
153         List<HelicalTrackStrip> clusterlist = cross.getStrips();
154 
155         for (HelicalTrackStrip cl : clusterlist) {
156             int layer = cl.layer();
157             if (layer == 1)
158                 _hasLayerOne = true;
159 
160             _nStripHitsPerLayer[layer - 1] = cl.rawhits().size();
161             _hitLocationPerLayer.put(layer, clusterPosition(cl));
162             _nhitsNew++;
163             double axdotu = VecOp.dot(cl.u(), axial);
164 //            System.out.println(new BasicHep3Vector(cross.getPosition()).toString() + cl.u());
165             boolean isAxial = false;
166             if (axdotu > 0.5) {
167                 isAxial = true;
168                 _nAxialhits++;
169             } else
170                 _nZhits++;
171             List<MCParticle> mcPartList = cl.MCParticles();
172             _nMCHitsPerLayer[layer - 1] = mcPartList.size();
173             for (MCParticle mcp : mcPartList) {
174                 Integer mchits = 0;
175                 if (mcmapAll.containsKey(mcp))
176                     mchits = mcmapAll.get(mcp);
177                 mchits++;
178                 mcmapAll.put(mcp, mchits);
179                 if (isAxial) {
180                     Integer mchitsAxial = 0;
181                     if (mcmapAxial.containsKey(mcp))
182                         mchitsAxial = mcmapAxial.get(mcp);
183                     mchitsAxial++;
184                     mcmapAxial.put(mcp, mchitsAxial);
185                 } else {
186                     Integer mchitsZ = 0;
187                     if (mcmapZ.containsKey(mcp))
188                         mchitsZ = mcmapZ.get(mcp);
189                     mchitsZ++;
190                     mcmapZ.put(mcp, mchitsZ);
191                 }
192             }
193         }
194     }

196     private void countHit(TrackerHit hit, RelationalTable rthtosimhit, RelationalTable hittostrip, RelationalTable hittorotated) {
197         TrackerHit unrotatedHit = (TrackerHit) hittorotated.from(hit);
198 //        System.out.println("ID: " + unrotatedHit.getCellID());
199         Set<TrackerHit> hitlist = hittostrip.allFrom(unrotatedHit);
200 //        System.out.println("size: " + hitlist.size());
201         for (TrackerHit cl : hitlist) {
202             int layer = -1;
203             int module = -1;
204             List<RawTrackerHit> rawHits = cl.getRawHits();
205 //                System.out.println("RawHits: " + rawHits.size());
206             for (RawTrackerHit rawHit : rawHits) {
207 //                    System.out.println(rawHit.getCellID());
208                 IIdentifier id = new Identifier(rawHit.getCellID());
209                 //===> int newLayer = SvtUtils.getInstance().getHelper().getValue(id, "layer");
210                 int newLayer = ((HpsSiSensor) rawHit.getDetectorElement()).getLayerNumber();
211                 if (layer != -1 && layer != newLayer)
212                     System.out.format("TrackerHit has hits from multiple layers: %d and %d\n", layer, newLayer);
213                 layer = newLayer;
214                 //===> int newModule = SvtUtils.getInstance().getHelper().getValue(id, "module");
215                 int newModule = ((HpsSiSensor) rawHit.getDetectorElement()).getModuleNumber();
216                 if (module != -1 && module != newModule)
217                     System.out.format("TrackerHit has hits from multiple modules: %d and %d\n", module, newModule);
218                 module = newModule;
219 //                    System.out.println(SvtUtils.getInstance().getHelper().getValue(id, "strip"));
220             }
221 
222             if (layer == 1)
223                 _hasLayerOne = true;
224             DiagonalizedCovarianceMatrix covariance = new DiagonalizedCovarianceMatrix(cl);
225             _nStripHitsPerLayer[layer - 1] = cl.getRawHits().size();
226             _hitLocationPerLayer.put(layer, new BasicHep3Vector(hit.getPosition()));
227             _nhitsNew++;
228 
229             double axdotu = VecOp.dot(transformVectorToTracking(covariance.getMeasuredVector()), axial);
230 //            System.out.println(transformVectorToTracking(new BasicHep3Vector(cl.getPosition())).toString() + transformVectorToTracking(covariance.getMeasuredVector()));
231             boolean isAxial = false;
232             if (axdotu > 0.5) {
233                 isAxial = true;
234                 _nAxialhits++;
235             } else
236                 _nZhits++;
237             //  get the set of MCParticles associated with this hit and update the hit count for each MCParticle
238 
239             Set<MCParticle> mcPartList = new HashSet<MCParticle>();
240             for (RawTrackerHit rawHit : rawHits) {
241                 Set<SimTrackerHit> simhits = (Set<SimTrackerHit>) rthtosimhit.allFrom(rawHit);
242                 for (SimTrackerHit simhit : simhits)
243                     if (simhit != null && simhit.getMCParticle() != null)
244                         mcPartList.add(simhit.getMCParticle());
245             }
246 //            System.out.println("MCParticle count: " + mcPartList.size());
247             _nMCHitsPerLayer[layer - 1] = mcPartList.size();
248             for (MCParticle mcp : mcPartList) {
249                 Integer mchits = 0;
250                 if (mcmapAll.containsKey(mcp))
251                     mchits = mcmapAll.get(mcp);
252                 mchits++;
253                 mcmapAll.put(mcp, mchits);
254                 if (isAxial) {
255                     Integer mchitsAxial = 0;
256                     if (mcmapAxial.containsKey(mcp))
257                         mchitsAxial = mcmapAxial.get(mcp);
258                     mchitsAxial++;
259                     mcmapAxial.put(mcp, mchitsAxial);
260                 } else {
261                     Integer mchitsZ = 0;
262                     if (mcmapZ.containsKey(mcp))
263                         mchitsZ = mcmapZ.get(mcp);
264                     mchitsZ++;
265                     mcmapZ.put(mcp, mchitsZ);
266                 }
267             }
268         }
269     }

271     private void countHit(HelicalTrack2DHit hit2d) {
272         _nhitsNew++;
273         _nAxialhits++;
274         List<MCParticle> mcPartList = hit2d.getMCParticles();
275         //assume that lone hits are all axial
276         boolean isAxial = true;
277         for (MCParticle mcp : mcPartList) {
278             Integer mchits = 0;
279             if (mcmapAll.containsKey(mcp))
280                 mchits = mcmapAll.get(mcp);
281             mchits++;
282             mcmapAll.put(mcp, mchits);
283             Integer mchitsAxial = 0;
284             if (mcmapAxial.containsKey(mcp))
285                 mchitsAxial = mcmapAxial.get(mcp);
286             mchitsAxial++;
287             mcmapAxial.put(mcp, mchitsAxial);
288         }

291     private void checkForBadHit(HelicalTrackCross cross) {
292         List<HelicalTrackStrip> clusterlist = cross.getStrips();
293         for (HelicalTrackStrip cl : clusterlist) {
294             trackLayerList.add(cl.layer());
295             if (!(cl.MCParticles().contains(_mcpNew))) {
296                 badHitList.add(cl.layer());
297                 badhits.put(_mcpNew, cross);
298             }
299             if (cl.MCParticles().size() > 1)
300                 sharedHitList.add(cl.layer());
301         }

304     public static Hep3Vector clusterPosition(HelicalTrackStrip cl) {
305         Hep3Vector corigin = cl.origin();
306         Hep3Vector u = cl.u();
307         double umeas = cl.umeas();
308         Hep3Vector uvec = VecOp.mult(umeas, u);
309         return VecOp.add(corigin, uvec);

393     public static class DiagonalizedCovarianceMatrix {
394 
395         double[] measurement_errors = new double[3];
396         Hep3Vector[] measurement_vectors = new Hep3Vector[3];
397 
398         public DiagonalizedCovarianceMatrix(TrackerHit hit) {
399             SymmetricMatrix cov = new SymmetricMatrix(3, hit.getCovMatrix(), true);
400             RealMatrix covMatrix = new Array2DRowRealMatrix(3, 3);
401             for (int i = 0; i < 3; i++)
402                 for (int j = 0; j < 3; j++)
403                     covMatrix.setEntry(i, j, cov.e(i, j));
404             EigenDecomposition decomposed = new EigenDecomposition(covMatrix);
405             BasicHep3Matrix localToGlobal = new BasicHep3Matrix();
406             for (int i = 0; i < 3; i++)
407                 for (int j = 0; j < 3; j++)
408                     localToGlobal.setElement(i, j, decomposed.getV().getEntry(i, j));
409 //            SymmetricMatrix localToGlobal = decomposed.getV().operate(new ArrayRealVector(3))
410             {
411                 double eigenvalue = decomposed.getRealEigenvalue(0);
412 //                Hep3Vector eigenvector = VecOp.mult(localToGlobal, new BasicHep3Vector());
413                 Hep3Vector eigenvector = VecOp.mult(Math.signum(eigenvalue), new BasicHep3Vector(decomposed.getVT().getRow(0)));
414                 measurement_errors[0] = eigenvalue;
415                 measurement_vectors[0] = eigenvector;
416                 measurement_errors[2] = eigenvalue;
417                 measurement_vectors[2] = eigenvector;
418             }
419             {
420                 double eigenvalue = decomposed.getRealEigenvalue(1);
421                 Hep3Vector eigenvector = VecOp.mult(Math.signum(eigenvalue), new BasicHep3Vector(decomposed.getVT().getRow(1)));
422                 if (eigenvalue > measurement_errors[0]) {
423                     measurement_errors[0] = eigenvalue;
424                     measurement_vectors[0] = eigenvector;
425                 }
426                 if (eigenvalue < measurement_errors[2]) {
427                     measurement_errors[2] = eigenvalue;
428                     measurement_vectors[2] = eigenvector;
429                 }
430             }
431             {
432                 double eigenvalue = decomposed.getRealEigenvalue(2);
433                 Hep3Vector eigenvector = VecOp.mult(Math.signum(eigenvalue), new BasicHep3Vector(decomposed.getVT().getRow(2)));
434                 if (eigenvalue > measurement_errors[0]) {
435                     measurement_errors[1] = measurement_errors[0];
436                     measurement_vectors[1] = measurement_vectors[0];
437                     measurement_errors[0] = eigenvalue;
438                     measurement_vectors[0] = eigenvector;
439                 }
440                 if (eigenvalue < measurement_errors[2]) {
441                     measurement_errors[1] = measurement_errors[2];
442                     measurement_vectors[1] = measurement_vectors[2];
443                     measurement_errors[2] = eigenvalue;
444                     measurement_vectors[2] = eigenvector;
445                 }
446                 if (measurement_vectors[1] == null) {
447                     measurement_errors[1] = eigenvalue;
448                     measurement_vectors[1] = eigenvector;
449                 }
450             }
451 //            for (int i = 0; i < 3; i++) {
452 //                System.out.format("%d: resolution %f, vector %s\n", i, measurement_errors[i], measurement_vectors[i].toString());
453 //            }
454         }
