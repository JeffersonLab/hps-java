55      public void process(EventHeader event) {
56          if (event.hasCollection(ReconstructedParticle.class, "FinalStateParticles")) {
57              //skip "monster" events
58              List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
59              int nRawTrackerHits = rawTrackerHits.size();
60              aida.histogram1D("SVT number of RawTrackerHits", 100, 0., 1000.).fill(nRawTrackerHits);
61              if (nRawTrackerHits > 250) {
62                  return;
63              }
64              setupSensors(event);
65              analyzeRP(event);
66              analyzeV0(event);
67          }
68          if (event.hasCollection(Cluster.class, clusterCollection)) {
69              analyzeClusters(event);
70          }
71      }

73      private void analyzeRP(EventHeader event) {
74          for (String s : finalStateParticleCollectionNames) {
75              if (event.hasCollection(ReconstructedParticle.class, s)) {
76                  String dir = s + " ReconstructedParticle Analysis";
77                  aida.tree().mkdirs(dir);
78                  aida.tree().cd(dir);
79                  List<ReconstructedParticle> rpList = event.get(ReconstructedParticle.class, s);
80                  for (ReconstructedParticle rp : rpList) {
81                      int pdgId = rp.getParticleIDUsed().getPDG();
82                      if (pdgId == 11) {
83                          String trackType = "SeedTrack ";
84                          if (TrackType.isGBL(rp.getType())) {
85                              trackType = "GBL ";
86                          }
87                          if (rp.getType() == 1) {
88                              trackType = "Kalman ";
89                          }
90  //            if (!TrackType.isGBL(rp.getType())) {
91  //                continue;
92  //            }
93                          Track t = rp.getTracks().get(0);
94                          int trackFinderType = t.getType();
95                          int nHits = t.getTrackerHits().size();
96                          String id = pdgId == 11 ? "electron" : "positron";
97                          Hep3Vector pmom = rp.getMomentum();
98                          double thetaY = atan2(pmom.y(), pmom.z());//asin(pmom.y() / pmom.magnitude());
99                          double z0 = t.getTrackStates().get(0).getZ0();
100                         String torb = isTopTrack(t) ? " top " : " bottom ";
101                         aida.histogram1D(trackType + torb + id + " track momentum", 100, 0., 10.).fill(rp.getMomentum().magnitude());
102 
103                         aida.cloud1D(trackType + torb + id + " |thetaY|").fill(abs(thetaY));
104                         aida.histogram1D(trackType + torb + id + " z0", 100, -2., 2.).fill(z0);
105                         aida.cloud2D(trackType + torb + id + " |thetaY| vs z0").fill(abs(thetaY), z0);
106                         aida.profile1D(trackType + torb + id + " |thetaY| vs z0 profile", 10, 0.01, 0.1).fill(abs(thetaY), z0);
107                         if (trackType.equals("MatchedTrack ") && torb.equals(" bottom ") && id.equals("electron")) {
108                             zProfileBottomMatched.fill(abs(thetaY), z0);
109                         }
110                         if (trackType.equals("GBL ") && torb.equals(" bottom ") && id.equals("electron")) {
111                             zProfileBottomGBL.fill(abs(thetaY), z0);
112                         }
113 
114                         List<Cluster> clusters = rp.getClusters();
115                         if (!clusters.isEmpty()) {
116                             Cluster c = clusters.get(0);
117                             analyzeCluster(c, id);
118                             double[] cPos = c.getPosition();
119                             aida.histogram2D(id + " cluster x vs y", 320, -270.0, 370.0, 90, -90.0, 90.0).fill(cPos[0], cPos[1]);
120                             aida.histogram1D(trackType + torb + id + " cluster energy", 100, 0., 10.).fill(c.getEnergy());
121                             aida.histogram1D(trackType + torb + id + " cluster energy over track momentum EoverP", 100, 0., 2.).fill(c.getEnergy() / rp.getMomentum().magnitude());
122                             aida.histogram1D(trackType + torb + id + " track momentum with cluster", 100, 0., 10.).fill(rp.getMomentum().magnitude());
123                             aida.histogram1D(trackType + torb + id + " track momentum with cluster " + nHits + " hits", 100, 0., 10.).fill(rp.getMomentum().magnitude());
124                             aida.histogram1D(trackType + " " + trackFinderType + torb + id + " track momentum with cluster " + nHits + " hits", 100, 0., 10.).fill(rp.getMomentum().magnitude());
125                         }
126                     }
127                     if (pdgId == 22) {
128                         String id = "photon";
129                         Cluster c = rp.getClusters().get(0);
130                         analyzeCluster(c, id);
131                         double[] cPos = c.getPosition();
132                         aida.histogram2D(id + " cluster x vs y", 320, -270.0, 370.0, 90, -90.0, 90.0).fill(cPos[0], cPos[1]);
133                         aida.histogram1D(id + " momentum", 100, 0., 10.).fill(rp.getMomentum().magnitude());
134                     }
135                 }
136                 aida.tree().cd("..");
137             }
138         }

149     private void analyzeV0(EventHeader event) {
150         String[] v0Dirs = {"UnconstrainedV0Candidates", "UnconstrainedV0Candidates_KF"};
151         for (String dir : v0Dirs) {
152             if (event.hasCollection(ReconstructedParticle.class, dir)) {
153                 aida.tree().mkdirs(dir);
154                 aida.tree().cd(dir);
155                 List<ReconstructedParticle> v0List = event.get(ReconstructedParticle.class, dir);
156                 for (ReconstructedParticle v0 : v0List) {
157                     String trackType = "SeedTrack ";
158                     int minNhits = 5;
159                     if (TrackType.isGBL(v0.getType())) {
160                         trackType = "GBL ";
161                     }
162                     if (v0.getType() == 1) {
163                         trackType = "Kalman ";
164                         minNhits = 10;
165                     }
166                     Vertex uncVert = v0.getStartVertex();
167                     Hep3Vector pVtxRot = VecOp.mult(beamAxisRotation, v0.getMomentum());
168                     Hep3Vector vtxPosRot = VecOp.mult(beamAxisRotation, uncVert.getPosition());
169                     double theta = Math.acos(pVtxRot.z() / pVtxRot.magnitude());
170                     double phi = Math.atan2(pVtxRot.y(), pVtxRot.x());
171 
172                     // this always has 2 tracks.
173                     List<ReconstructedParticle> trks = v0.getParticles();
174                     ReconstructedParticle ele = trks.get(0);
175                     ReconstructedParticle pos = trks.get(1);
176                     int eNhits = ele.getTracks().get(0).getTrackerHits().size();
177                     int pNhits = pos.getTracks().get(0).getTrackerHits().size();
178                     double eMom = ele.getMomentum().magnitude();
179                     double pMom = pos.getMomentum().magnitude();
180                     aida.histogram1D("electron track nHits", 20, 0., 20.).fill(eNhits);
181                     aida.histogram1D("positron track nHits", 20, 0., 20.).fill(pNhits);
182                     aida.histogram1D("electron momentum", 100, 0., 6.0).fill(eMom);
183                     aida.histogram1D("positron momentum", 100, 0., 6.0).fill(pMom);
184 
185                     if (eNhits >= minNhits && pNhits >= minNhits) {
186                         aida.histogram1D("v0 x", 50, -5., 5.).fill(vtxPosRot.x());
187                         aida.histogram1D("v0 y", 50, -2., 2.).fill(vtxPosRot.y());
188                         aida.histogram1D("v0 z", 50, -25., 0.).fill(vtxPosRot.z());
189                         aida.histogram1D("v0 x ele " + eNhits + " pos " + pNhits + " hits on track", 50, -5., 5.).fill(vtxPosRot.x());
190                         aida.histogram1D("v0 y ele " + eNhits + " pos " + pNhits + " hits on track", 50, -2., 2.).fill(vtxPosRot.y());
191                         aida.histogram1D("v0 z ele " + eNhits + " pos " + pNhits + " hits on track", 50, -25., 0.).fill(vtxPosRot.z());
192                         aida.histogram1D("v0 energy", 100, 0., 10.).fill(v0.getEnergy());
193                         aida.histogram1D("v0 mass", 50, 0., 0.5).fill(v0.getMass());
194                         aida.histogram2D("v0 mass vs Z vertex", 50, 0., 0.5, 100, -20., 20.).fill(v0.getMass(), vtxPosRot.z());
195                         aida.histogram2D("v0 mass vs Z vertex ele " + eNhits + " pos " + pNhits + " hits on track", 50, 0., 0.5, 100, -20., 20.).fill(v0.getMass(), vtxPosRot.z());
196                         aida.profile1D("v0 mass vs Z vertex profile", 50, 0.05, 0.25).fill(v0.getMass(), vtxPosRot.z());
197                         if (ele.getClusters().isEmpty()) {
198                             aida.histogram1D("psum no electron ECal Cluster", 100, 0., 6.0).fill(eMom + pMom);
199                             aida.histogram1D("psum no electron ECal Cluster ele " + eNhits + " pos " + pNhits + " hits on track", 100, 0., 6.0).fill(eMom + pMom);
200                         }
201                         aida.histogram1D("psum", 100, 0., 6.0).fill(eMom + pMom);
202                         aida.histogram1D("psum ele " + eNhits + " pos " + pNhits + " hits on track", 100, 0., 6.0).fill(eMom + pMom);
203                         aida.histogram2D("electron vs positron momentum", 100, 0., 6.0, 100, 0., 6.).fill(eMom, pMom);
204                         if (ele.getClusters().size() > 0 && pos.getClusters().size() > 0) {
205                             aida.histogram1D("psum both ECal Clusters", 100, 0., 6.0).fill(eMom + pMom);
206                             aida.histogram1D("esum both ECal Clusters", 100, 0., 6.0).fill(ele.getClusters().get(0).getEnergy() + pos.getClusters().get(0).getEnergy());
207                         }
208                     }
209                 }
210                 aida.tree().cd("..");
211             }
212         }

246                 aida.histogram1D(topOrBottom + " cluster energy, seed > " + minSeedEnergy, 100, 0., 6.0).fill(c.getEnergy());
247                 aida.histogram1D(topOrBottom + " cluster energy, seed > " + minSeedEnergy + " " + nClusters + " event clusters", 100, 0., 6.0).fill(c.getEnergy());
248                 aida.histogram1D(fiducial + topOrBottom + " cluster energy, seed > " + minSeedEnergy, 100, 0., 6.0).fill(c.getEnergy());
249                 aida.histogram1D(fiducial + topOrBottom + " cluster energy, seed > " + minSeedEnergy + " " + nClusters + " event clusters", 100, 0., 6.0).fill(c.getEnergy());
250             }
251             List<CalorimeterHit> clusterHits = c.getCalorimeterHits();
252             CalorimeterHit seedHit = ClusterUtilities.findSeedHit(c);
253             aida.histogram1D("cluster seed time", 100, 0., 200.).fill(seedHit.getTime());
254             if (seed != seedHit) {
255                 System.out.println("Panic!");
256             }
257             for (CalorimeterHit hit : clusterHits) {
258                 aida.histogram1D("cluster hit corrected energy", 200, 0., 5.).fill(hit.getCorrectedEnergy());
259                 aida.histogram1D("cluster hit corrected energy low end", 200, 0., 0.1).fill(hit.getCorrectedEnergy());
260 //                aida.histogram1D("cluster hit raw energy", 200, 0., 5.).fill(hit.getRawEnergy());  //!? always zero?
261                 if (seedHit != hit) {
262                     aida.histogram1D("cluster seed hit time - hit time", 100, -20., 20.).fill(seedHit.getTime() - hit.getTime());
263                 }
264             }
265         }
266         aida.tree().cd("..");

283     private boolean isTopTrack(Track t) {
284         List<TrackerHit> hits = t.getTrackerHits();
285         int n[] = {0, 0};
286         int nHits = hits.size();
287         for (TrackerHit h : hits) {
288             HpsSiSensor sensor = ((HpsSiSensor) ((RawTrackerHit) h.getRawHits().get(0)).getDetectorElement());
289             if (sensor.isTopLayer()) {
290                 n[0] += 1;
291             } else {
292                 n[1] += 1;
293             }
294         }
295         if (n[0] == nHits && n[1] == 0) {
296             return true;
297         }
298         if (n[1] == nHits && n[0] == 0) {
299             return false;
300         }
301         throw new RuntimeException("mixed top and bottom hits on same track");
302 
303     }

305     private void setupSensors(EventHeader event) {
306         List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
307         EventHeader.LCMetaData meta = event.getMetaData(rawTrackerHits);
308         // Get the ID dictionary and field information.
309         IIdentifierDictionary dict = meta.getIDDecoder().getSubdetector().getDetectorElement().getIdentifierHelper().getIdentifierDictionary();
310         int fieldIdx = dict.getFieldIndex("side");
311         int sideIdx = dict.getFieldIndex("strip");
312         for (RawTrackerHit hit : rawTrackerHits) {
313             // The "side" and "strip" fields needs to be stripped from the ID for sensor lookup.
314             IExpandedIdentifier expId = dict.unpack(hit.getIdentifier());
315             expId.setValue(fieldIdx, 0);
316             expId.setValue(sideIdx, 0);
317             IIdentifier strippedId = dict.pack(expId);
318             // Find the sensor DetectorElement.
319             List<IDetectorElement> des = DetectorElementStore.getInstance().find(strippedId);
320             if (des == null || des.size() == 0) {
321                 throw new RuntimeException("Failed to find any DetectorElements with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
322             } else if (des.size() == 1) {
323                 hit.setDetectorElement((SiSensor) des.get(0));
324             } else {
325                 // Use first sensor found, which should work unless there are sensors with duplicate IDs.
326                 for (IDetectorElement de : des) {
327                     if (de instanceof SiSensor) {
328                         hit.setDetectorElement((SiSensor) de);
329                         break;
330                     }
331                 }
332             }
333             // No sensor was found.
334             if (hit.getDetectorElement() == null) {
335                 throw new RuntimeException("No sensor was found for hit with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
336             }
337         }
