82      private Hep3Vector getUnitVector(SiSensor sensor, String type) {
83  
84          Hep3Vector unit_vec = new BasicHep3Vector(-99, -99, -99);
85  
86          for (ChargeCarrier carrier : ChargeCarrier.values()) {
87              if (sensor.hasElectrodesOnSide(carrier)) {
88                  int channel = 1;
89                  long cell_id = sensor.makeStripId(channel, carrier.charge()).getValue();
90                  IIdentifier id = new Identifier(cell_id);
91                  SiTrackerIdentifierHelper _sid_helper = (SiTrackerIdentifierHelper) sensor.getIdentifierHelper();
92                  SiSensorElectrodes electrodes = sensor.getReadoutElectrodes(carrier);
93                  ITransform3D local_to_global = new Transform3D();// sensor.getGeometry().getLocalToGlobal();
94                  ITransform3D electrodes_to_global = electrodes.getLocalToGlobal();
95                  ITransform3D global_to_hit = local_to_global.inverse();
96                  ITransform3D electrodes_to_hit = Transform3D.multiply(global_to_hit, electrodes_to_global);
97                  if (type == "measured") {
98                      unit_vec = electrodes_to_hit.rotated(electrodes.getMeasuredCoordinate(0));
99                  } else if (type == "unmeasured") {
100                     unit_vec = electrodes_to_hit.rotated(electrodes.getUnmeasuredCoordinate(0));
101                 } else {
102                     throw new UnsupportedOperationException(String.format("type=\"%s\" not supported", type));
103                 }
104             }
105         }
106         return unit_vec;
107     }

109     private Hep3Vector[] getUnitVectors(SiSensor sensor) {
110 
111         Hep3Vector unit_vecU = new BasicHep3Vector(-99, -99, -99);
112         Hep3Vector unit_vecV = new BasicHep3Vector(-99, -99, -99);
113         Hep3Vector unit_vecW = new BasicHep3Vector(-99, -99, -99);
114 
115         for (ChargeCarrier carrier : ChargeCarrier.values()) {
116             if (sensor.hasElectrodesOnSide(carrier)) {
117                 int channel = 1;
118                 long cell_id = sensor.makeStripId(channel, carrier.charge()).getValue();
119                 IIdentifier id = new Identifier(cell_id);
120                 SiTrackerIdentifierHelper _sid_helper = (SiTrackerIdentifierHelper) sensor.getIdentifierHelper();
121                 SiSensorElectrodes electrodes = sensor.getReadoutElectrodes(carrier);
122                 ITransform3D local_to_global = new Transform3D();// sensor.getGeometry().getLocalToGlobal();
123                 ITransform3D electrodes_to_global = electrodes.getLocalToGlobal();
124                 ITransform3D global_to_hit = local_to_global.inverse();
125                 ITransform3D electrodes_to_hit = Transform3D.multiply(global_to_hit, electrodes_to_global);
126 
127                 unit_vecU = electrodes_to_hit.rotated(electrodes.getMeasuredCoordinate(0));
128                 unit_vecV = electrodes_to_hit.rotated(electrodes.getUnmeasuredCoordinate(0));
129                 unit_vecW = VecOp.cross(unit_vecU, unit_vecV);
130             }
131         }
132         return new Hep3Vector[]{unit_vecU, unit_vecV, unit_vecW};
133     }
