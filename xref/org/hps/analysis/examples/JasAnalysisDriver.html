753         return Math.sqrt(x * x + y * y);

756     protected double drcalc(Hep3Vector pos, SymmetricMatrix cov) {
757         double x = pos.x();
758         double y = pos.y();
759         double r2 = x * x + y * y;
760         return Math.sqrt((x * x * cov.e(0, 0) + y * y * cov.e(1, 1) + 2. * x * y * cov.e(0, 1)) / r2);
761     }
762 
763     protected double drphicalc(Hep3Vector pos, SymmetricMatrix cov) {
764         double x = pos.x();
765         double y = pos.y();
766         double r2 = x * x + y * y;
767         return Math.sqrt((y * y * cov.e(0, 0) + x * x * cov.e(1, 1) - 2. * x * y * cov.e(0, 1)) / r2);
768     }

770     private double getphi(double x, double y) {
771         double phi = Math.atan2(y, x);
772         if (phi < 0.) {
773             phi += 2. * Math.PI;
774         }
775         return phi;
776     }

778     private double getdxdy(Hep3Vector hitpos, Hep3Vector posonhelix) {
779         return Math.sqrt(Math.pow(hitpos.x() - posonhelix.x(), 2) + Math.pow(hitpos.y() - posonhelix.y(), 2));
780     }

782     private double getdxdyErr(Hep3Vector hitpos, Hep3Vector posonhelix, SymmetricMatrix cov) {
783         double dxdySq = Math.pow(hitpos.x() - posonhelix.x(), 2) + Math.pow(hitpos.y() - posonhelix.y(), 2);
784         double ErrSqDxDySq = 4 * (cov.e(0, 0) * Math.pow(hitpos.x() - posonhelix.x(), 2) + cov.e(1, 1)
785                 * Math.pow(hitpos.y() - posonhelix.y(), 2));
786         double error = Math.sqrt(ErrSqDxDySq / dxdySq) / 2;
787         return error;
788     }

879     private double getInvMass(Track track1, StraightLineTrack slt1, Track track2, StraightLineTrack slt2) {
880         double esum = 0.;
881         double pxsum = 0.;
882         double pysum = 0.;
883         double pzsum = 0.;
884         double chargesum = 0.;
885         double me = 0.000511;
886         // Loop over jets
887 
888         double p1x = track1.getTrackStates().get(0).getMomentum()[0];
889         double p1y = track1.getTrackStates().get(0).getMomentum()[1];
890         double p1z = track1.getTrackStates().get(0).getMomentum()[2];
891         double p1mag2 = p1x * p1x + p1y * p1y + p1z * p1z;
892         double e1 = Math.sqrt(p1mag2 + me * me);
893         double dydx1 = slt1.dydx();
894         double dzdx1 = slt1.dzdx();
895         double s1sq = 1 + 1 / (dydx1 * dydx1) + (dzdx1 * dzdx1) / (dydx1 * dydx1);
896         double truep1y = Math.sqrt(p1mag2 / s1sq);
897         if (dydx1 < 0) {
898             truep1y = -truep1y;
899         }
900         double truep1x = truep1y / dydx1;
901         double truep1z = dzdx1 * truep1x;
902 
903         double p2x = track2.getTrackStates().get(0).getMomentum()[0];
904         double p2y = track2.getTrackStates().get(0).getMomentum()[1];
905         double p2z = track2.getTrackStates().get(0).getMomentum()[2];
906         double p2mag2 = p2x * p2x + p2y * p2y + p2z * p2z;
907         double e2 = Math.sqrt(p2mag2 + me * me);
908 
909         double dydx2 = slt2.dydx();
910         double dzdx2 = slt2.dzdx();
911         double s2sq = 1 + 1 / (dydx2 * dydx2) + (dzdx2 * dzdx2) / (dydx2 * dydx2);
912         double truep2y = Math.sqrt(p2mag2 / s2sq);
913         if (dydx2 < 0) {
914             truep2y = -truep2y;
915         }
916         double truep2x = truep2y / dydx2;
917         double truep2z = dzdx2 * truep2x;
918 
919         pxsum = truep1x + truep2x;
920         pysum = truep1y + truep2y;
921         pzsum = truep1z + truep2z;
922 
923         esum = e1 + e2;
924         // double p1dotp2 = p1x * p2x + p1y * p2y + p1z * p2z;
925         double p1dotp2 = truep1x * truep2x + truep1y * truep2y + truep1z * truep2z;
926         double e1e2 = e1 * e2;
927         double invmass = Math.sqrt(2 * me * me + 2 * (e1e2 - p1dotp2));
928         // Compute total momentum and hence event mass
929         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
930         double evtmass = Math.sqrt(esum * esum - psum * psum);
931         // System.out.println("invmass= " + invmass + "; evtmass=" + evtmass);
932         return invmass;
933     }

935     // find the DOCA to the beamline extrpolating linearly from the reference point
936     private double findDoca(double y, double z, double px, double py, double pz) {
937         double xoca = 0;
938         double sy = py / px;
939         double sz = pz / px;
940         xoca = -(y * sy + z * sz) / (sy * sy + sz + sz);
941         double doca = Math.sqrt(Math.pow(y + sy * xoca, 2) + Math.pow(z + sz * xoca, 2));
942         return doca;
943     }

945     // find the XOCA to the beamline extrpolating linearly from the reference point
946     private double findXoca(double y, double z, double px, double py, double pz) {
947         double xoca = 0;
948         double sy = py / px;
949         double sz = pz / px;
950         xoca = -(y * sy + z * sz) / (sy * sy + sz + sz);
951         return xoca;
952     }

954     private double[] findPoca(double y, double z, double px, double py, double pz) {
955         double poca[] = {0, 0, 0};
956         double sy = py / px;
957         double sz = pz / px;
958         poca[0] = -(y * sy + z * sz) / (sy * sy + sz * sz);
959         poca[1] = y + sy * poca[0];
960         poca[2] = z + sz * poca[0];
961         return poca;
962     }

983     private Hep3Vector getTrueMomentum(Track track1, StraightLineTrack slt1) {
984         double[] truep = {0, 0, 0};
985         double me = 0.000511;
986         double p1x = track1.getTrackStates().get(0).getMomentum()[0];
987         double p1y = track1.getTrackStates().get(0).getMomentum()[1];
988         double p1z = track1.getTrackStates().get(0).getMomentum()[2];
989         double p1mag2 = p1x * p1x + p1y * p1y + p1z * p1z;
990         double e1 = Math.sqrt(p1mag2 + me * me);
991         double dydx1 = slt1.dydx();
992         double dzdx1 = slt1.dzdx();
993         double s1sq = 1 + 1 / (dydx1 * dydx1) + (dzdx1 * dzdx1) / (dydx1 * dydx1);
994         truep[1] = Math.sqrt(p1mag2 / s1sq);
995         if (dydx1 < 0) {
996             truep[1] = -truep[1];
997         }
998         truep[0] = truep[1] / dydx1;
999         truep[2] = dzdx1 * truep[0];
1000 
1001         return new BasicHep3Vector(truep[0], truep[1], truep[2]);
1002     }
