1   package org.hps.analysis.examples;
2   
3   import java.util.List;
4   import org.lcsim.event.Cluster;
5   import org.lcsim.event.EventHeader;
6   import org.lcsim.event.RawTrackerHit;
7   import org.lcsim.util.Driver;
8   import org.lcsim.util.aida.AIDA;
9   
10  public class StripMuMuDriver extends Driver {
11  
12      private AIDA aida = AIDA.defaultInstance();
13      double clusterMaxEcut = 0.3;
14      double clusterMinEcut = 0.1;
15      int nMaxSvtRawTrackerHits = 500;
16      int _numberOfEventsSelected;
17  
18      public void process(EventHeader event) {
19          boolean skipEvent = true;
20          List<Cluster> clusters = event.get(Cluster.class, "EcalClustersCorr");
21          int nMuClusters = 0;
22          for (Cluster c : clusters) {
23              aida.histogram1D("event cluster nHits", 20, 0., 20.).fill(c.getCalorimeterHits().size());
24              aida.histogram1D("event cluster energy", 100, 0., 1.5).fill(c.getEnergy());
25              aida.histogram2D("event cluster x vs y", 320, -270.0, 370.0, 90, -90.0, 90.0).fill(c.getPosition()[0], c.getPosition()[1]);
26  
27              if (c.getEnergy() < clusterMaxEcut && c.getEnergy() > clusterMinEcut) {
28                  nMuClusters++;
29                  aida.histogram2D("muon cluster x vs y", 320, -270.0, 370.0, 90, -90.0, 90.0).fill(c.getPosition()[0], c.getPosition()[1]);
30  
31              }
32          }
33          aida.histogram1D("Number of Clusters in Event", 10, 0., 10.).fill(clusters.size());
34          aida.histogram1D("Number of muon Clusters in Event", 10, 0., 10.).fill(nMuClusters);
35  
36          if (nMuClusters > 1) {
37              skipEvent = false;
38          }
39  
40          //skip "monster" events
41          List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
42          int nRawTrackerHits = rawTrackerHits.size();
43          aida.histogram1D("SVT number of RawTrackerHits", 100, 0., 1000.).fill(nRawTrackerHits);
44          if (nRawTrackerHits > nMaxSvtRawTrackerHits) {
45              skipEvent = true;
46          }
47  
48          if (skipEvent) {
49              throw new Driver.NextEventException();
50          } else {
51              _numberOfEventsSelected++;
52          }
53      }
54  
55      @Override
56      protected void endOfData() {
57          System.out.println("Selected " + _numberOfEventsSelected + " events");
58      }
59  
60  }
