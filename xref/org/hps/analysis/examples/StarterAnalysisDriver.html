25  public class StarterAnalysisDriver extends Driver {
26  
27      int nevents = 0;
28      int naccepted = 0;
29      private boolean debug = false;
30      private double bfield = 0.5;
31      private AIDA aida = AIDA.defaultInstance();
32      IHistogram1D trkPz;
33      IHistogram2D trkMC;
34  
35      @Override
36      protected void detectorChanged(Detector detector) {
37          for (String str : detector.getFields().keySet()) {
38              System.out.println(str);
39              ((Field) detector.getFields().get(str)).getName();
40          }
41          trkPz = aida.histogram1D("Track momentum (Pz)", 200, 0, 10.0);
42          trkMC = aida.histogram2D("Track momentum vs. MCParticle momentum (Pz)", 200, 0, 10.0, 50, 0, 10.0);
43      }
44  
45      @Override
46      public void process(EventHeader event) {
47          nevents++;
48  
49          List<LCIOTrackAnalysis> tkanalList = processTracks(event);
50  
51          //only look at events with 2 or more tracks
52  //        if (tkanalList.size() < 2) {
53  //            return;
54  //        }
55  
56  //        System.out.println("Event with " + tkanalList.size() + " tracks");
57  
58          int ok = 0;
59          for (LCIOTrackAnalysis tkanal : tkanalList) {   //remember, these tracks are in the lcsim tracking frame!     
60              Track track = tkanal.getTrack();
61              BaseTrackState ts = (BaseTrackState) track.getTrackStates().get(0);
62  //            ts.computeMomentum(bfield);
63              trkPz.fill(ts.getMomentum()[0]);
64  //            BaseTrackState.computeMomentum(track.getTrackStates().get(0), 0.5);
65              MCParticle mcp = tkanal.getMCParticle();
66              if (mcp != null) {
67                  System.out.println("chisq: " + track.getChi2() + ", track pz: " + ts.getMomentum()[0] + ", MC momentum: " + mcp.getMomentum());
68                  trkMC.fill(ts.getMomentum()[0], mcp.getPZ());
69                  if (Math.abs(tkanal.getMCParticle().getPDGID()) == 611) {
70                      ok++;
71                  }
72                  //do some stuff to makes sure tracks are great
73                  //is there an e+e- from the muonium
74              }
75          }
76  
77          if (ok == 2) {
78              naccepted++;
79          }
80      }
81  
82      @Override
83      public void endOfData() {
84  
85          System.out.println("# of muonium events= " + naccepted + "; # of total = " + nevents);
86      }
87  
88      private List<LCIOTrackAnalysis> processTracks(EventHeader event) {
89          List<LCIOTrackAnalysis> tkanalList = new ArrayList<LCIOTrackAnalysis>();
90  
91          List<Track> tracklist = event.get(Track.class, "MatchedTracks");
92  
93          if (debug) {
94              List<List<TrackerHit>> hitlists = event.get(TrackerHit.class);
95              for (List<TrackerHit> hitlist : hitlists) {
96                  System.out.println(event.getMetaData(hitlist).getName());
97                  for (TrackerHit hit : hitlist) {
98                      System.out.println(hit);
99                  }
100             }
101         }
102 
103         RelationalTable hittomc = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
104         List<LCRelation> mcrelations = event.get(LCRelation.class, "HelicalTrackMCRelations");
105         if (debug) {
106             System.out.println("HelicalTrackMCRelations:" + mcrelations.size());
107         }
108         for (LCRelation relation : mcrelations) {
109             if (relation != null && relation.getFrom() != null && relation.getTo() != null) {
110                 hittomc.add(relation.getFrom(), relation.getTo());
111                 if (debug) {
112                     System.out.println("to:" + relation.getTo());
113                     System.out.println("from:" + relation.getFrom());
114                 }
115             }
116         }
117 
118         RelationalTable hittostrip = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
119         List<LCRelation> hitrelations = event.get(LCRelation.class, "HelicalTrackHitRelations");
120         if (debug) {
121             System.out.println("HelicalTrackHitRelations:" + hitrelations.size());
122         }
123         for (LCRelation relation : hitrelations) {
124             if (relation != null && relation.getFrom() != null && relation.getTo() != null) {
125                 hittostrip.add(relation.getFrom(), relation.getTo());
126                 if (debug) {
127                     System.out.println("to:" + relation.getTo());
128                     System.out.println("from:" + relation.getFrom());
129                 }
130             }
131         }
132 
133         RelationalTable hittorotated = new BaseRelationalTable(RelationalTable.Mode.ONE_TO_ONE, RelationalTable.Weighting.UNWEIGHTED);
134         List<LCRelation> rotaterelations = event.get(LCRelation.class, "RotatedHelicalTrackHitRelations");
135         if (debug) {
136             System.out.println("RotatedHelicalTrackHitRelations:" + rotaterelations.size());
137         }
138         for (LCRelation relation : rotaterelations) {
139             if (relation != null && relation.getFrom() != null && relation.getTo() != null) {
140                 hittorotated.add(relation.getFrom(), relation.getTo());
141                 if (debug) {
142                     System.out.println("to:" + relation.getTo());
143                     System.out.println("from:" + relation.getFrom());
144                 }
145             }
146         }
147 
148         for (Track track : tracklist) {   //remember, these tracks are in the lcsim tracking frame!     
149             BaseTrackState ts = (BaseTrackState) track.getTrackStates().get(0);
150             ts.computeMomentum(bfield);
151             tkanalList.add(new LCIOTrackAnalysis(track, hittomc, hittostrip, hittorotated));
152         }
153         return tkanalList;
154     }
