1   package org.hps.analysis.examples;
2   
3   import static java.lang.Math.sqrt;
4   import java.util.List;
5   import java.util.Map;
6   import org.lcsim.event.EventHeader;
7   import org.lcsim.event.Vertex;
8   import org.lcsim.util.Driver;
9   import org.lcsim.util.fourvec.Lorentz4Vector;
10  import org.lcsim.util.fourvec.Momentum4Vector;
11  
12  public class VertexAnalysis extends Driver {
13  
14      String vertexCollectionName = "TargetConstrainedV0Vertices";
15      String[] names = {"X", "Y", "Z"};
16      double[] p1 = new double[4];
17      double[] p2 = new double[4];
18      double emass = 0.000511;
19      double mass2 = emass * emass;
20  
21      boolean debug = false;
22  
23      protected void process(EventHeader event) {
24          List<Vertex> vertices = event.get(Vertex.class, vertexCollectionName);
25          for (Vertex v : vertices) {
26              Map<String, Double> vals = v.getParameters();
27              // System.out.println(vals);
28              p1[0] = vals.get("p1X");
29              p1[1] = vals.get("p1Y");
30              p1[2] = vals.get("p1Z");
31              p2[0] = vals.get("p2X");
32              p2[1] = vals.get("p2Y");
33              p2[2] = vals.get("p2Z");
34              double e1 = 0;
35              double e2 = 0.;
36              for (int i = 0; i < 3; ++i) {
37                  e1 += p1[i] * p1[i];
38                  e2 += p2[i] * p2[i];
39              }
40              e1 = sqrt(e1 + mass2);
41              e2 = sqrt(e2 + mass2);
42              Momentum4Vector vec1 = new Momentum4Vector(p1[0], p1[1], p1[2], e1);
43              Momentum4Vector vec2 = new Momentum4Vector(p2[0], p2[1], p2[2], e2);
44              Lorentz4Vector sum = vec1.plus(vec2);
45              double mass = sum.mass();
46              double invMass = vals.get("invMass");
47              if (debug)
48                  System.out.println("mass: " + mass + " invMass: " + invMass + " delta: " + (invMass - mass));
49  
50          }
51      }
52  }
