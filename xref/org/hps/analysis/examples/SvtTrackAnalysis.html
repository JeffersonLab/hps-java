26          // Check if an event has a specific collection and if it doesn't, 
27          // skip it  The default track collection is still "MatchedTracks"

30          if (!event.hasCollection(Track.class, matchedTracksColName)) return;
31          
32          // Once the collection has been confirmed to exist in the event, 
33          // you can retrieve a reference to the collection from the event
34          // as follows.
35          List<Track> tracks = event.get(Track.class, matchedTracksColName);
36      
37          // The track collection can be iterated over and each track can 
38          // then be analyzed
39          for (Track track : tracks) { 
40             // Analysis Code goes here ... 
41          }
42          
43          // The other collections can be retrieved in a similar manner
44          
45          if (!event.hasCollection(Track.class, tracksS456ColName)) return;
46          tracks = event.get(Track.class, tracksS456ColName);
47         
48          // Analysis goes here ...
49          
50          if (!event.hasCollection(Track.class, tracksS123C4ColName)) return;
51          tracks = event.get(Track.class, tracksS123C4ColName);
52  
53          // Analysis goes here ...
54          
55          if (!event.hasCollection(Track.class, tracksS123C5ColName)) return;
56          tracks = event.get(Track.class, tracksS123C5ColName);
57      
58          // Analysis goes here ...
59          
60          //
61          // If looking at all track collections, a filter can be used to look
62          // at only specific tracks
63          //
64          
65          // Check if the event has a LCIO Track collection to begin
66          if (!event.hasCollection(Track.class));
67          
68          // Get all LCIO Track collections from an event
69          List<List<Track>> trackCollections = event.get(Track.class);
70      
71          // Iterate over all track collections
72          for (List<Track> trackCollection : trackCollections) { 
73              
74              // Check if the collection has tracks.  If it does not, skip it.
75              if (trackCollection.isEmpty()) continue;
76             
77              // Use the track type to check if the tracks are from the
78              // collections of interest
79              
80              if (TrackType.getType(StrategyType.MATCHED_TRACKS) != trackCollection.get(0).getType()) continue;
81

84              }
85          }
86      }
