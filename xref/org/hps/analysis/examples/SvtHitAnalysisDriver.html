178     private void setupSensors(EventHeader event) {
179         List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
180         EventHeader.LCMetaData meta = event.getMetaData(rawTrackerHits);
181         // Get the ID dictionary and field information.
182         IIdentifierDictionary dict = meta.getIDDecoder().getSubdetector().getDetectorElement().getIdentifierHelper().getIdentifierDictionary();
183         int fieldIdx = dict.getFieldIndex("side");
184         int sideIdx = dict.getFieldIndex("strip");
185         for (RawTrackerHit hit : rawTrackerHits) {
186             // The "side" and "strip" fields needs to be stripped from the ID for sensor lookup.
187             IExpandedIdentifier expId = dict.unpack(hit.getIdentifier());
188             expId.setValue(fieldIdx, 0);
189             expId.setValue(sideIdx, 0);
190             IIdentifier strippedId = dict.pack(expId);
191             // Find the sensor DetectorElement.
192             List<IDetectorElement> des = DetectorElementStore.getInstance().find(strippedId);
193             if (des == null || des.size() == 0) {
194                 throw new RuntimeException("Failed to find any DetectorElements with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
195             } else if (des.size() == 1) {
196                 hit.setDetectorElement((SiSensor) des.get(0));
197             } else {
198                 // Use first sensor found, which should work unless there are sensors with duplicate IDs.
199                 for (IDetectorElement de : des) {
200                     if (de instanceof SiSensor) {
201                         hit.setDetectorElement((SiSensor) de);
202                         break;
203                     }
204                 }
205             }
206             // No sensor was found.
207             if (hit.getDetectorElement() == null) {
208                 throw new RuntimeException("No sensor was found for hit with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
209             }
210         }
