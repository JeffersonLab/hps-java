1   package org.hps.analysis.examples;
2   
3   import java.util.List;
4   import org.lcsim.event.EventHeader;
5   import org.lcsim.event.ReconstructedParticle;
6   import org.lcsim.util.Driver;
7   
8   /**
9    * Simple Driver to select events containing a single positron This is defined
10   * as one ReconstructedParticle which has been identified as a positron with a
11   * track which has greater than or equal the numberOfHitsOnTrack (default is 6).
12   * By default, only one ReconstructedParticle is allowed per event.
13   *
14   * TODO Move to recon.filtering
15   */
16  public class StripSinglePositronDriver extends Driver {
17  
18      private boolean _writeRunAndEventNumbers = false;
19      private double _energyCut = 0.;
20      private int _nHitsOnTrack = 6;
21      private int _nReconstructedParticles = 1;
22  
23      private int _numberOfEventsWritten = 0;
24  
25      /**
26       *
27       * @param event
28       */
29      @Override
30      protected void process(EventHeader event) {
31          boolean skipEvent = true;
32          List<ReconstructedParticle> rps = event.get(ReconstructedParticle.class, "FinalStateParticles");
33          if (rps.size() <= _nReconstructedParticles) {
34              for (ReconstructedParticle rp : rps) {
35                  double energy = rp.getEnergy();
36                  int pdgId = rp.getParticleIDUsed().getPDG();
37                  if (pdgId == -11) {
38                      if (energy >= _energyCut && rp.getTracks().get(0).getTrackerHits().size() >= _nHitsOnTrack) // positron
39                      {
40                          skipEvent = false;
41                      }
42                  }
43              }
44          }
45          if (skipEvent) {
46              throw new Driver.NextEventException();
47          } else {
48              if (_writeRunAndEventNumbers) {
49                  System.out.println(event.getRunNumber() + " " + event.getEventNumber());
50              }
51              _numberOfEventsWritten++;
52          }
53      }
54  
55      /**
56       * Positrons having energy below the cut will be rejected.
57       *
58       * @param cut
59       */
60      public void setEnergyCut(double cut) {
61          _energyCut = cut;
62      }
63  
64      /**
65       * Tracks having fewer than the number of hits will be rejected.
66       *
67       * @param cut
68       */
69      public void setNumberOfHitsOnTrack(int cut) {
70          _nHitsOnTrack = cut;
71      }
72  
73      /**
74       * Events having more than the number of ReconstructedParticles will be
75       * rejected.
76       *
77       * @param cut
78       */
79      public void setNumberOfReconstructedParticles(int cut) {
80          _nReconstructedParticles = cut;
81      }
82  
83      /**
84       * Write out run and event numbers of events passing the cuts if desired
85       *
86       * @param b
87       */
88      public void setWriteRunAndEventNumbers(boolean b) {
89          _writeRunAndEventNumbers = b;
90      }
91  
92      @Override
93      protected void endOfData() {
94          System.out.println("Wrote " + _numberOfEventsWritten + " events");
95      }
96  }
