1   package org.hps.analysis.examples;

3   import java.io.BufferedReader;
4   import java.io.FileReader;
5   import java.io.IOException;
6   import java.util.LinkedHashSet;
7   import java.util.Set;
8   import java.util.StringTokenizer;
9   import org.apache.commons.math3.util.Pair;
10  import org.lcsim.event.EventHeader;
11  import org.lcsim.geometry.Detector;
12  import org.lcsim.util.Driver;
13  
14  /**
15   * @todo Move to recon.filtering
16   */
17  public class StripRunAndEventDriver extends Driver
18  {
19  
20      private String _eventListFileName;
21      private int _nEventsToStrip;
22      private int _nEventsStripped;
23      Set<Pair<Integer, Integer>> _eventsToStrip = new LinkedHashSet<Pair<Integer, Integer>>();
24  
25      @Override
26      protected void detectorChanged(Detector detector)
27      {
28          BufferedReader br = null;
29          String currentLine = null;
30          System.out.println("detector changed reading in "+_eventListFileName);
31          try {
32              br = new BufferedReader(new FileReader(_eventListFileName));
33              while ((currentLine = br.readLine()) != null) {
34                  String runEvent = currentLine.trim();
35                  StringTokenizer st = new StringTokenizer(runEvent, " ");
36                  Integer run = Integer.parseInt(st.nextToken());
37                  Integer event = Integer.parseInt(st.nextToken());
38                  _eventsToStrip.add(new Pair(run, event));
39              }
40          } catch (IOException e) {
41              e.printStackTrace();
42              throw new RuntimeException(e);
43          } catch (NumberFormatException e) {
44              System.out.println("bad number format in skim list: " + currentLine);
45              throw new RuntimeException(e);
46          } finally {
47              try {
48                  br.close();
49              } catch (IOException e) {
50                  e.printStackTrace();
51              }
52          }
53          _nEventsToStrip = _eventsToStrip.size();
54          System.out.println("preparing to strip "+_nEventsToStrip+" events");
55      }
56  
57      @Override
58      protected void process(EventHeader event)
59      {
60          boolean skipEvent = true;
61          int runNum = event.getRunNumber();
62          int eventNum = event.getEventNumber();
63          Pair<Integer,Integer> pair = new Pair<Integer,Integer>(runNum, eventNum);
64          if(_eventsToStrip.contains(pair)) skipEvent = false;
65          if (skipEvent) {
66              throw new Driver.NextEventException();
67          } else {
68              _nEventsStripped++;
69          }
70      }
71  
72      @Override
73      protected void endOfData()
74      {
75          System.out.println("Stripped " + _nEventsStripped + " of " + _nEventsToStrip + " requested");
76      }
77  
78      public void setEventListFileName(String s)
79      {
80          System.out.println("run and event list is "+s);
81          _eventListFileName = s;
82      }
83  
84  }
