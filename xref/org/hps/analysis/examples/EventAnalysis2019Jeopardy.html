149     private void analyzeV0(EventHeader event) {
150         String dir = "Unconstrained V0 Analysis";
151         aida.tree().mkdirs(dir);
152         aida.tree().cd(dir);
153         List<ReconstructedParticle> v0List = event.get(ReconstructedParticle.class, "UnconstrainedV0Candidates");
154         for (ReconstructedParticle v0 : v0List) {
155             if (!TrackType.isGBL(v0.getType())) {
156                 continue;
157             }
158             Vertex uncVert = v0.getStartVertex();
159             Hep3Vector pVtxRot = VecOp.mult(beamAxisRotation, v0.getMomentum());
160             Hep3Vector vtxPosRot = VecOp.mult(beamAxisRotation, uncVert.getPosition());
161             double theta = Math.acos(pVtxRot.z() / pVtxRot.magnitude());
162             double phi = Math.atan2(pVtxRot.y(), pVtxRot.x());
163             aida.histogram1D("v0 x", 50, -5., 5.).fill(vtxPosRot.x());
164             aida.histogram1D("v0 y", 50, -5., 5.).fill(vtxPosRot.y());
165             aida.histogram1D("v0 z", 50, -25., 0.).fill(vtxPosRot.z());
166             aida.histogram1D("v0 energy", 100, 0., 10.).fill(v0.getEnergy());
167             aida.histogram1D("v0 mass", 50, 0., 0.5).fill(v0.getMass());
168             // this always has 2 tracks.
169             List<ReconstructedParticle> trks = v0.getParticles();
170             ReconstructedParticle ele = trks.get(0);
171             ReconstructedParticle pos = trks.get(1);
172             double eMom = ele.getMomentum().magnitude();
173             double pMom = pos.getMomentum().magnitude();
174             aida.histogram1D("electron momentum", 100, 0., 6.0).fill(eMom);
175             aida.histogram1D("positron momentum", 100, 0., 6.0).fill(pMom);
176             aida.histogram2D("electron vs positron momentum", 100, 0., 6.0, 100, 0., 6.).fill(eMom, pMom);
177         }
178         aida.tree().cd("..");
179     }

235     private boolean isTopTrack(Track t) {
236         List<TrackerHit> hits = t.getTrackerHits();
237         int n[] = {0, 0};
238         int nHits = hits.size();
239         for (TrackerHit h : hits) {
240             HpsSiSensor sensor = ((HpsSiSensor) ((RawTrackerHit) h.getRawHits().get(0)).getDetectorElement());
241             if (sensor.isTopLayer()) {
242                 n[0] += 1;
243             } else {
244                 n[1] += 1;
245             }
246         }
247         if (n[0] == nHits && n[1] == 0) {
248             return true;
249         }
250         if (n[1] == nHits && n[0] == 0) {
251             return false;
252         }
253         throw new RuntimeException("mixed top and bottom hits on same track");
254 
255     }

257     private void setupSensors(EventHeader event) {
258         List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
259         EventHeader.LCMetaData meta = event.getMetaData(rawTrackerHits);
260         // Get the ID dictionary and field information.
261         IIdentifierDictionary dict = meta.getIDDecoder().getSubdetector().getDetectorElement().getIdentifierHelper().getIdentifierDictionary();
262         int fieldIdx = dict.getFieldIndex("side");
263         int sideIdx = dict.getFieldIndex("strip");
264         for (RawTrackerHit hit : rawTrackerHits) {
265             // The "side" and "strip" fields needs to be stripped from the ID for sensor lookup.
266             IExpandedIdentifier expId = dict.unpack(hit.getIdentifier());
267             expId.setValue(fieldIdx, 0);
268             expId.setValue(sideIdx, 0);
269             IIdentifier strippedId = dict.pack(expId);
270             // Find the sensor DetectorElement.
271             List<IDetectorElement> des = DetectorElementStore.getInstance().find(strippedId);
272             if (des == null || des.size() == 0) {
273                 throw new RuntimeException("Failed to find any DetectorElements with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
274             } else if (des.size() == 1) {
275                 hit.setDetectorElement((SiSensor) des.get(0));
276             } else {
277                 // Use first sensor found, which should work unless there are sensors with duplicate IDs.
278                 for (IDetectorElement de : des) {
279                     if (de instanceof SiSensor) {
280                         hit.setDetectorElement((SiSensor) de);
281                         break;
282                     }
283                 }
284             }
285             // No sensor was found.
286             if (hit.getDetectorElement() == null) {
287                 throw new RuntimeException("No sensor was found for hit with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
288             }
289         }
