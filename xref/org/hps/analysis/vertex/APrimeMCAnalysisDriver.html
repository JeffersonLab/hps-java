95      protected void process(EventHeader event) {
96          List<MCParticle> mcparts = event.getMCParticles();
97          MCParticle aprime = null;
98          MCParticle recoilMC = mcparts.get(0);
99          MCParticle positronMC = mcparts.get(1);
100         MCParticle electronMC = mcparts.get(2);
101         if (debug) {
102             System.out.println("recoil " + recoilMC);
103             System.out.println("electron " + electronMC);
104             System.out.println("positron " + positronMC);
105 
106         }
107 
108         Hep3Vector ip = recoilMC.getOrigin();
109         Hep3Vector aprimeVtx = electronMC.getOrigin();
110         Lorentz4Vector l4v = lorentzVectorSum(electronMC, positronMC);
111         double aprimeMass = l4v.mass();
112         if (debug) {
113             System.out.println("aprime 4vec: " + l4v);
114             System.out.println("aprime vertex: " + aprimeVtx);
115         }
116         
117         List<Vertex> vertices = event.get(Vertex.class, vertexCollectionName);
118         aida.cloud1D("number of Vertices)").fill(vertices.size());
119         if (vertices.size() == 2) {
120             for (Vertex v : vertices) {
121                 aida.tree().cd("/");
122 
123                 ReconstructedParticle rp = v.getAssociatedParticle();
124                 int type = rp.getType();
125                 if (TrackType.isGBL(type)) {
126                     SymmetricMatrix cov = v.getCovMatrix();
127                     Hep3Vector pos = v.getPosition();
128 
129                     // try some transforms here...
130 //            rotMat.rotate(cov);
131 //            rotMat.rotate(pos);
132 //            xformMat.transform(aprimeVtx);
133                     if (debug) {
134                         System.out.println("cov matrix: \n" + cov);
135                         System.out.println("vertex pos: \n" + pos);
136                     }
137                     Map<String, Double> vals = v.getParameters();
138                     //System.out.println(vals);
139                     p1[0] = vals.get("p1X");
140                     p1[1] = vals.get("p1Y");
141                     p1[2] = vals.get("p1Z");
142                     p2[0] = vals.get("p2X");
143                     p2[1] = vals.get("p2Y");
144                     p2[2] = vals.get("p2Z");
145                     double e1 = 0;
146                     double e2 = 0.;
147                     for (int i = 0; i < 3; ++i) {
148                         e1 += p1[i] * p1[i];
149                         e2 += p2[i] * p2[i];
150                     }
151                     e1 = sqrt(e1 + mass2);
152                     e2 = sqrt(e2 + mass2);
153                     Momentum4Vector vec1 = new Momentum4Vector(p1[0], p1[1], p1[2], e1);
154                     Momentum4Vector vec2 = new Momentum4Vector(p2[0], p2[1], p2[2], e2);
155                     Lorentz4Vector sum = vec1.plus(vec2);
156                     double mass = sum.mass();
157                     double invMass = vals.get("invMass");
158                     if (debug) {
159                         System.out.println("mass: " + mass + " invMass: " + invMass + " delta: " + (invMass - mass));
160                     }
161 
162                     analyzeVertex(v, electronMC, positronMC);
163 
164                     aida.cloud1D("type").fill(type);
165                     vtx_x.fill(pos.x());
166                     aida.cloud2D("vertex x vs z").fill(pos.z(), pos.x());
167 
168                     vtx_xMC.fill(aprimeVtx.x());
169                     aida.cloud2D("vertex MC x vs MC z").fill(aprimeVtx.z(), aprimeVtx.x());
170                     aida.cloud2D("vertex MC x vs MC y").fill(aprimeVtx.x(), aprimeVtx.y());
171                     vtx_xRes.fill(pos.x() - aprimeVtx.x());
172                     double xPull = (pos.x() - aprimeVtx.x()) / sqrt(cov.diagonal(0));
173                     vtx_xPull.fill(xPull);
174 
175                     vtx_y.fill(pos.y());
176                     aida.cloud2D("vertex y vs z").fill(pos.z(), pos.y());
177                     vtx_yMC.fill(aprimeVtx.y());
178                     aida.cloud2D("vertex MC y vs MC z").fill(aprimeVtx.z(), aprimeVtx.y());
179                     vtx_yRes.fill(pos.y() - aprimeVtx.y());
180                     double yPull = (pos.y() - aprimeVtx.y()) / sqrt(cov.diagonal(1));
181 
182                     vtx_yPull.fill(yPull);
183 
184                     vtx_z.fill(pos.z());
185                     vtx_zMC.fill(aprimeVtx.z());
186                     vtx_zRes.fill(pos.z() - aprimeVtx.z());
187                     double zPull = (pos.z() - aprimeVtx.z()) / sqrt(cov.diagonal(2));
188                     if (abs(zPull) < 10.) {
189                         vtx_zPull.fill(zPull);
190                         aida.cloud2D("vertex z pull vs MC z").fill(zPull, aprimeVtx.z());
191                         aida.cloud1D("vertex z significance pull lt 10").fill(pos.z() / zPull);
192                     }
193 
194                     vtx_zvsMCz.fill(aprimeVtx.z(), pos.z());
195 
196                     aida.cloud1D("vertex chisq").fill(v.getChi2());
197                     aida.cloud1D("vertex chi2 prob").fill(v.getProbability());
198                     vtxMass.fill(rp.getMass());
199                     vtxMassRes.fill(rp.getMass() - aprimeMass);
200                     vtxMassResvsMCz.fill(aprimeVtx.z(), rp.getMass() - aprimeMass);
201                     List<ReconstructedParticle> parts = rp.getParticles();
202                     ReconstructedParticle electron = null;
203                     ReconstructedParticle positron = null;
204                     for (ReconstructedParticle part : parts) {
205                         if (part.getCharge() < 0) {
206                             electron = part;
207                         }
208                         if (part.getCharge() > 0) {
209                             positron = part;
210                         }
211                     }
212                     analyzeParticle("electron", electron);
213                     analyzeParticle("positron", positron);
214 
215                     BilliorVertex bvRefit = reVertex(v);
216                     if (debug) {
217                         System.out.println("aprime vertex: " + aprimeVtx);
218                         System.out.println("vertex pos: " + pos);
219                         System.out.println("bvRefit: " + bvRefit);
220                     }
221                     
222                     vtxrf_zRes.fill(bvRefit.getPosition().z() - aprimeVtx.z());
223                     // check mass correction here...
224                     // corrM = events["uncM"]-0.15e-3*(events["elePX"]/events["eleP"]-events["posPX"]/events["posP"])*events["uncVZ"]/events["uncM"]
225                     double uncM = bvRefit.getParameters().get("invMass");
226                     double uncVZ = bvRefit.getPosition().z();
227                     double elePX = bvRefit.getParameters().get("p1Y");
228                     double elePY = bvRefit.getParameters().get("p1Z");
229                     double elePZ = bvRefit.getParameters().get("p1X");
230                     double posPX = bvRefit.getParameters().get("p2Y");
231                     double posPY = bvRefit.getParameters().get("p2Z");
232                     double posPZ = bvRefit.getParameters().get("p2X");
233                     double eleP = sqrt(elePX * elePX + elePY * elePY + elePZ * elePZ);
234                     double posP = sqrt(posPX * posPX + posPY * posPY + posPZ * posPZ);
235                     double corrM = uncM - 0.15e-3 * (elePX / eleP - posPX / posP) * uncVZ / uncM;
236                     if (debug) {
237                         System.out.println("uncM " + uncM + " corrM " + corrM);
238                     }
239                     vtxrf_mRes.fill(uncM - aprimeMass);
240                     vtxrf_corrmRes.fill(corrM - aprimeMass);
241                     vtxcorrMassResvsMCz.fill(aprimeVtx.z(), corrM - aprimeMass);
242                     aida.tree().cd("/");
243                 }// only analyze GBL tracks
244             }
245         }//only analyze events with one vertex (2 in collection because of multiple track collections
246     }

248     private void analyzeParticle(String dir, ReconstructedParticle p) {
249         aida.tree().mkdirs(dir);
250         aida.tree().cd(dir);
251         //
252         Track t = p.getTracks().get(0);
253         double chiSquared = t.getChi2();
254         int ndf = t.getNDF();
255         double chisqProb = ChisqProb.gammp(ndf, chiSquared);
256         aida.cloud1D("Track chisq per df").fill(chiSquared / ndf);
257         aida.cloud1D("Track chisq prob").fill(chisqProb);
258         aida.cloud1D("Track nHits").fill(t.getTrackerHits().size());
259         aida.tree().cd("..");
260     }

262     private BilliorVertex reVertex(Vertex v0) {
263         double zVtx = v0.getPosition().z();
264         ReconstructedParticle rp = v0.getAssociatedParticle();
265         List<ReconstructedParticle> parts = rp.getParticles();
266         ReconstructedParticle electron = null;
267         ReconstructedParticle positron = null;
268         for (ReconstructedParticle part : parts) {
269             if (part.getCharge() < 0) {
270                 electron = part;
271             }
272             if (part.getCharge() > 0) {
273                 positron = part;
274             }
275         }
276         Track eletrk = electron.getTracks().get(0);
277         Track postrk = positron.getTracks().get(0);
278 //        //propagate track to the vertex position...
279 //        Hep3Vector elexcpt = TrackUtils.extrapolateTrack(eletrk,zVtx);
280 //        Hep3Vector posxcpt = TrackUtils.extrapolateTrack(postrk,zVtx);
281         // first let's see if we can recreate this vertex...
282         SeedTrack stEle1 = TrackUtils.makeSeedTrackFromBaseTrack(eletrk);
283         SeedTrack stEle2 = TrackUtils.makeSeedTrackFromBaseTrack(postrk);
284         BilliorTrack eleBilliorTrack = new BilliorTrack(stEle1.getSeedCandidate().getHelix());
285         BilliorTrack posBilliorTrack = new BilliorTrack(stEle2.getSeedCandidate().getHelix());
286         BilliorVertex bv = fitVertex(eleBilliorTrack, posBilliorTrack, _bfield);
287         double invMass = bv.getParameters().get("invMass");
288         Hep3Vector vtxpos = bv.getPosition();
289         aida.cloud1D("vtx x diff").fill(vtxpos.x() - v0.getPosition().x());
290         aida.cloud1D("vtx y diff").fill(vtxpos.y() - v0.getPosition().y());
291         aida.cloud1D("vtx z diff").fill(vtxpos.z() - v0.getPosition().z());
292         if(debug) {
293             System.out.println(bv);
294         }
295         // this looks good.
296         // quick check of track states...
297         List<TrackState> eleTrackStates = eletrk.getTrackStates();
298         if(debug) {
299             System.out.println("electron track states:");
300             for (TrackState ts : eleTrackStates) {
301                 System.out.println(ts);
302             }
303             System.out.println("positron track states:");
304             List<TrackState> posTrackStates = postrk.getTrackStates();
305             for (TrackState ts : posTrackStates) {
306                 System.out.println(ts);
307             }
308         }
309         // now, transport track state to vtx position, revertex...
310         // TODO transport covariance matrix as well...
311         // TODO determine whether to iterate
312         TrackState eleTrackState = eletrk.getTrackStates().get(0);

314         if (debug) {
315             System.out.println("eleTrackState: " + eleTrackState);
316             System.out.println("posTrackState: " + posTrackState);
317         }
318         double[] eleParams = eleTrackState.getParameters();
319         double[] eleCovMat = eleTrackState.getCovMatrix();
320         double[] eleParamsXtrap = TrackUtils.getParametersAtNewRefPoint(vtxpos.v(), eleTrackState.getReferencePoint(), eleParams);
321         double[] posParams = posTrackState.getParameters();
322         double[] posCovMat = posTrackState.getCovMatrix();
323         double[] posParamsXtrap = TrackUtils.getParametersAtNewRefPoint(vtxpos.v(), posTrackState.getReferencePoint(), posParams);
324         if (debug) {
325             System.out.println("ele before: " + Arrays.toString(eleParams));
326             System.out.println("ele after: " + Arrays.toString(eleParamsXtrap));
327             System.out.println("pos before: " + Arrays.toString(posParams));
328             System.out.println("pos after: " + Arrays.toString(posParamsXtrap));

331         TrackState eleTsAtVtx = new BaseTrackState(eleParamsXtrap, eleCovMat, vtxpos.v(), 5);
332         TrackState posTsAtVtx = new BaseTrackState(posParamsXtrap, posCovMat, vtxpos.v(), 5);
333 
334         // create BilliorTracks from trackstate constructor
335         BilliorTrack eleBtAtVtx = new BilliorTrack(eleTsAtVtx, eletrk.getChi2(), eletrk.getNDF());
336         BilliorTrack posBtAtVtx = new BilliorTrack(posTsAtVtx, postrk.getChi2(), postrk.getNDF());
337         // revertex
338         BilliorVertex bvRefit = fitVertex(eleBtAtVtx, posBtAtVtx, _bfield);
339 
340         return bvRefit;
341     }

343     private BilliorVertex fitVertex(BilliorTrack electron, BilliorTrack positron, double bField) {
344         // Create a vertex fitter from the magnetic field.
345         double[] beamSize = {0.001, 0.2, 0.02};
346         BilliorVertexer vtxFitter = new BilliorVertexer(bField);
347         // TODO: The beam size should come from the conditions database.
348         vtxFitter.setBeamSize(beamSize);
349 
350         // Perform the vertexing based on the specified constraint.
351         vtxFitter.doBeamSpotConstraint(false);
352 
353         // Add the electron and positron tracks to a track list for
354         // the vertex fitter.
355         List<BilliorTrack> billiorTracks = new ArrayList<BilliorTrack>();
356 
357         billiorTracks.add(electron);
358 
359         billiorTracks.add(positron);
360 
361         // Find and return a vertex based on the tracks.
362         return vtxFitter.fitVertex(billiorTracks);
363     }

374     static private String myDate() {
375         Calendar cal = new GregorianCalendar();
376         Date date = new Date();
377         cal.setTime(date);
378         DecimalFormat formatter = new DecimalFormat("00");
379         String day = formatter.format(cal.get(Calendar.DAY_OF_MONTH));
380         String month = formatter.format(cal.get(Calendar.MONTH) + 1);
381         return cal.get(Calendar.YEAR) + month + day;
382     }

384     private Lorentz4Vector lorentzVectorSum(MCParticle p1, MCParticle p2) {
385         Momentum4Vector vec1 = new Momentum4Vector(p1.getPX(), p1.getPY(), p1.getPZ(), p1.getEnergy());
386         Momentum4Vector vec2 = new Momentum4Vector(p2.getPX(), p2.getPY(), p2.getPZ(), p2.getEnergy());
387         return vec1.plus(vec2);
388     }
