1266     private double invMass(Hep3Vector p1, Hep3Vector p2) {
1267         double me2 = 0.000511 * 0.000511;

1269         double pxsum = p1.x() + p2.x();
1270         double pysum = p1.y() + p2.y();
1271         double pzsum = p1.z() + p2.z();
1272 
1273         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
1274         double evtmass = esum * esum - psum * psum;
1275 
1276         if (evtmass > 0)
1277             return Math.sqrt(evtmass);
1278         else
1279             return -99;
