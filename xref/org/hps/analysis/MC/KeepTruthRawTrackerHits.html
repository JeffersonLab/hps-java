1   package org.hps.analysis.MC;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.lcsim.event.EventHeader;
7   import org.lcsim.event.LCRelation;
8   import org.lcsim.event.RawTrackerHit;
9   import org.lcsim.event.RelationalTable;
10  import org.lcsim.event.SimTrackerHit;
11  import org.lcsim.event.TrackerHit;
12  import org.lcsim.event.base.BaseRelationalTable;
13  import org.lcsim.recon.tracking.digitization.sisim.SiTrackerHitStrip1D;
14  import org.lcsim.util.Driver;
15  
16  /**
17   * This driver keeps all truth hits that correspond to a track's 1D strip hits
18   */
19  public class KeepTruthRawTrackerHits extends Driver {
20  
21      //Collection Names
22      private String simHitCollectionName = "TrackerHits_truth";
23      private String rawHitRelationsCollectionName = "SVTTrueHitRelations";
24      
25      private String stripHitInputCollectionName = "StripClusterer_SiTrackerHitStrip1D";
26      private String stripHitOutputCollectionName = "StripClusterer_SiTrackerHitStrip1D_truth";
27  
28      public KeepTruthRawTrackerHits() {
29      
30      }
31      
32      public void setSimHitCollectionName(String simHitCollectionName) {
33          this.simHitCollectionName = simHitCollectionName;
34      }
35  
36      public void setRawHitRelationsCollectionName(String rawHitRelationsCollectionName) {
37          this.rawHitRelationsCollectionName = rawHitRelationsCollectionName;
38      }
39      
40      
41      //Build relation table for truth track hits
42      public static RelationalTable getRawToTruthTable(EventHeader event,String RawHitRelationsCollectionName) {
43          RelationalTable hitToTruth = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
44          List<LCRelation> hitrelations = event.get(LCRelation.class, RawHitRelationsCollectionName);
45          for (LCRelation relation : hitrelations)
46              if (relation != null && relation.getFrom() != null && relation.getTo() != null){
47                  hitToTruth.add(relation.getFrom(), relation.getTo());
48              }
49          return hitToTruth;
50      }
51      
52      protected void process(EventHeader event) {
53          List<TrackerHit> striphits = event.get(TrackerHit.class, stripHitInputCollectionName);
54          List<SimTrackerHit> simHits = event.get(SimTrackerHit.class, simHitCollectionName);
55          List<SiTrackerHitStrip1D> truthStripHits = new ArrayList<SiTrackerHitStrip1D>();
56          RelationalTable rawToTruth = getRawToTruthTable(event,rawHitRelationsCollectionName);
57          //Loop over all 1D strip hits
58          for(TrackerHit hit:striphits){
59              boolean keepHit = false;
60              List<RawTrackerHit> rawhits = hit.getRawHits();
61              //Loop over raw hits on strip hits
62              for(RawTrackerHit rawhit:rawhits){
63                  //Loop over all sim tracker hits on raw hit
64                  for (SimTrackerHit simhit : simHits) {
65                      //Check to see if the raw hit contains a sim hit from the list of truth hits
66                      //If so, keep the hit and break the loop
67                      if(rawToTruth.allFrom(rawhit).contains(simhit)){
68                          keepHit = true;
69                          break;
70                      }
71                  }
72                  if(keepHit)
73                      break;
74              }
75              if(keepHit){
76                  truthStripHits.add(new SiTrackerHitStrip1D(hit));
77              }
78          }
79          event.put(stripHitOutputCollectionName, truthStripHits, SiTrackerHitStrip1D.class, 0);
80      }
81  }
