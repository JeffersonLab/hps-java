66      private void doTruth(EventHeader event, Track trk, FieldMap bFieldMap, List<HpsSiSensor> sensors, Subdetector trackerSubdet){
67      
68          List<SimTrackerHit> trackerHits = event.get(SimTrackerHit.class, trackerHitsCollectionName);
69          List<SimTrackerHit> trackerHits_Inactive = null;
70          List<Track> tracks = event.get(Track.class, trackCollectionName);
71          
72          if(event.hasCollection(SimTrackerHit.class , inactiveTrackerHitsCollectionName)){
73              trackerHits_Inactive = event.get(SimTrackerHit.class, inactiveTrackerHitsCollectionName);
74          }
75          Map<MCParticle, List<SimTrackerHit>> trackerHitMap = BuildTrackerHitMap(trackerHits);
76          Map<MCParticle, List<SimTrackerHit>> trackerInHitMap = BuildTrackerHitMap(trackerHits_Inactive);
77          
78          RelationalTable rawtomc = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
79          if (event.hasCollection(LCRelation.class, "SVTTrueHitRelations")) {
80              List<LCRelation> trueHitRelations = event.get(LCRelation.class, "SVTTrueHitRelations");
81              for (LCRelation relation : trueHitRelations)
82                  if (relation != null && relation.getFrom() != null && relation.getTo() != null)
83                      rawtomc.add(relation.getFrom(), relation.getTo());
84          }
85          _pTruth = new TrackTruthMatching(trk, rawtomc, trackerHits);
86          
87          MCParticle truthp = _pTruth.getMCParticle();
88          
89          if(truthp == null)
90              return;
91          
92          List<TrackerHit> trk_hits = trk.getTrackerHits();
93          for(Track track : tracks){
94              if(track.equals(trk))
95                  continue;
96              boolean sharedHit = false;
97              List<TrackerHit> hits = track.getTrackerHits();
98              int sharedLay = 9999;
99              for(TrackerHit hit : hits){
100                 for(TrackerHit trk_hit : trk_hits){
101                     int lay = ((RawTrackerHit) trk_hit.getRawHits().get(0)).getLayerNumber();
102                     if(!_sharedTrk.containsKey(lay)){
103                         _sharedTrk.put(lay,new HashSet<Track>());
104                     }
105                     if(hit.equals(trk_hit)){
106                         sharedHit = true;
107                         sharedLay = lay;
108                         break;
109                     }
110                 }
111                 if(sharedHit)
112                     break;
113             }
114             if(sharedHit){
115                 if(_sharedTrk.containsKey(sharedLay)){
116                     _sharedTrk.get(sharedLay).add(track);
117                 }
118                 else{
119                     _sharedTrk.put(sharedLay,new HashSet<Track>());
120                 }
121             }
122         }
123         
124         _truthActHits = trackerHitMap.get(truthp);
125         List<SimTrackerHit> truthInActHits = trackerInHitMap.get(truthp);
126         
127         ComputeSVTVars(truthp, _truthActHits, truthInActHits, bFieldMap, sensors, trackerSubdet);
128         
129         List<SimCalorimeterHit> calHits = event.get(SimCalorimeterHit.class, ecalHitsCollectionName);
130         Map<MCParticle, List<SimCalorimeterHit>> calHitMap = BuildCalHitMap(calHits);
131         
132         List<SimCalorimeterHit> truthEcalHits = calHitMap.get(truthp);
133         if(truthEcalHits == null)
134             return;
135         
136         IDDecoder calDecoder = event.getMetaData(calHits).getIDDecoder();
137         
138         ComputeEcalVars(event,truthEcalHits,calDecoder);
139     }

141     private void ComputeSVTVars(MCParticle p, List<SimTrackerHit> hits_act, List<SimTrackerHit> hits_in, FieldMap bFieldMap, List<HpsSiSensor> sensors, Subdetector trackerSubdet){
142         // loop over particle's hits
143         boolean inactiveprev = false;
144         int layerprev = 0;
145         List<Pair<SimTrackerHit,String>> allhits = orderHits(hits_act, hits_in, trackerSubdet, sensors);
146         
147         Hep3Vector startPosition = allhits.get(0).getFirst().getPositionVec();
148         double[] startP = allhits.get(0).getFirst().getMomentum();
149         Hep3Vector startMomentum = new BasicHep3Vector(startP[0], startP[1], startP[2]);
150         
151         for(Pair<SimTrackerHit,String> allhit : allhits){
152             boolean inactive = allhit.getSecond().equals("inactive");
153             SimTrackerHit hit = allhit.getFirst();
154             IDetectorElement de = trackerSubdet.getDetectorElement().findDetectorElement(hit.getPositionVec());
155             int layer = trackHitLayerNum(de, sensors, inactive);
156             Hep3Vector endPosition = hit.getPositionVec();
157             Hep3Vector endMomentum = new BasicHep3Vector(hit.getMomentum()[0], hit.getMomentum()[1], hit.getMomentum()[2]);
158             if(layer == layerprev){
159                 if(inactive != inactiveprev)
160                     inactiveprev = false;
161                 continue;
162             }
163             
164             Pair<Hep3Vector,Hep3Vector> extrapPair = extrapolateTrack(endPosition,endMomentum,startPosition,5,bFieldMap,p.getCharge());
165             if(extrapPair == null) continue;
166             Hep3Vector extrapPos = extrapPair.getFirst();
167             Hep3Vector extrapP = extrapPair.getSecond();
168 
169             double thetaX = MCFullDetectorTruth.deltaThetaX(VecOp.neg(extrapP),startMomentum);
170             double thetaY = MCFullDetectorTruth.deltaThetaY(VecOp.neg(extrapP),startMomentum);
171             
172             double[] theta = {thetaX,thetaY};
173             double[] residual = {startPosition.x() - extrapPos.x(),startPosition.y() - extrapPos.y()};
174             
175             if(!inactive)
176                 _actHitPos.put(layer,endPosition);
177             else
178                 _inactHitPos.put(layer,endPosition);
179             if(!inactiveprev){
180                 _actHitP.put(layerprev,startMomentum);
181                 _actHitTheta.put(layerprev,theta);
182                 _actHitResidual.put(layerprev,residual);
183             }
184             else{
185                 _inactHitP.put(layerprev,startMomentum);
186                 _inactHitTheta.put(layerprev,theta);
187                 _inactHitResidual.put(layerprev,residual);
188             }
189             startPosition = endPosition;
190             startMomentum = endMomentum;
191             inactiveprev = inactive;
192             layerprev = layer;
193         }
194     }

196     private List<Pair<SimTrackerHit,String>> orderHits(List<SimTrackerHit> hits_act, List<SimTrackerHit> hits_in, Subdetector trackerSubdet, List<HpsSiSensor> sensors){
197         if(hits_act == null) 
198             return null;
199         List<Pair<SimTrackerHit,String>> hitlist = new ArrayList<Pair<SimTrackerHit,String>>();
200         if(hits_in == null){
201             for(SimTrackerHit hit : hits_act){
202                 hitlist.add(new Pair<SimTrackerHit,String>(hit,"active"));
203             }
204             return hitlist;
205         }
206         
207         int n_inact_used = 0;
208 
209         for (SimTrackerHit hit : hits_act) {
210             boolean inactive = false;
211             SimTrackerHit hit_act = hit;
212             do{
213                 inactive = false;
214                 int n_inact = 0;
215                 for(SimTrackerHit hit_in:hits_in){
216                     n_inact++;
217                     if(n_inact_used >= n_inact) continue;
218                     IDetectorElement de = trackerSubdet.getDetectorElement().findDetectorElement(hit_in.getPositionVec());
219                     int layer_in = trackHitLayerNum(de, sensors, true);
220                     if(layer_in >= hit_act.getLayer()) continue;
221                     hit = hit_in;
222                     inactive = true;
223                     n_inact_used = n_inact;
224                     break;
225                 }
226                 
227                 if(!inactive){
228                     hit = hit_act;
229                     hitlist.add(new Pair<SimTrackerHit,String>(hit,"active"));
230                 }
231                 else
232                     hitlist.add(new Pair<SimTrackerHit,String>(hit,"inactive"));
233                 
234             } while(inactive);
235         }
236         if(n_inact_used > 0){
237             int n_inact = 0;
238             for(SimTrackerHit hit:hits_in){
239                 n_inact++;
240                 if(n_inact_used >= n_inact) continue;
241                 hitlist.add(new Pair<SimTrackerHit,String>(hit,"inactive"));
242 
243             }
244         }
245         return hitlist;
246     }

248     private void ComputeEcalVars(EventHeader event, List<SimCalorimeterHit> truthEcalHits,IDDecoder calDecoder){
249         _ecalNHits = truthEcalHits.size();
250         int i = 0;
251         for(SimCalorimeterHit hit : truthEcalHits){
252             calDecoder.setID(hit.getCellID());
253             int[] index = {calDecoder.getValue("ix"),calDecoder.getValue("iy")};
254             _ecalHitIndex.put(i,index);
255             _ecalHitPos.put(i,hit.getPositionVec());
256             _ecalHitE.put(i, hit.getCorrectedEnergy());
257             i++;
258         }
259     }

261     public static Map<MCParticle, List<SimTrackerHit>> BuildTrackerHitMap(List<SimTrackerHit> trackerHits){
262         Map<MCParticle, List<SimTrackerHit>> trackerHitMap = new HashMap<MCParticle, List<SimTrackerHit>>();
263         if(trackerHits == null) { return trackerHitMap; }
264         for (SimTrackerHit hit : trackerHits) {
265             MCParticle p = hit.getMCParticle();
266             if (p == null) {
267                 throw new RuntimeException("Tracker hit points to null MCParticle!");
268             }
269             if (trackerHitMap.get(p) == null) {
270                 trackerHitMap.put(p, new ArrayList<SimTrackerHit>());
271             }
272             trackerHitMap.get(p).add(hit);
273         }        
274         return trackerHitMap;
275     }

277     public static Map<MCParticle, List<SimCalorimeterHit>> BuildCalHitMap(List<SimCalorimeterHit> calHits){
278         // map particle to a list of its sim cal hits
279         Map<MCParticle, List<SimCalorimeterHit>> calHitMap = new HashMap<MCParticle, List<SimCalorimeterHit>>();
280         for (SimCalorimeterHit hit : calHits) {
281             int nmc = hit.getMCParticleCount();
282             for (int i = 0; i < nmc; i++) {
283                 MCParticle p = hit.getMCParticle(i);
284                 if (p == null) {
285                     throw new RuntimeException("Cal hit points to null MCParticle!");
286                 }
287                 if (calHitMap.get(p) == null) {
288                     calHitMap.put(p, new ArrayList<SimCalorimeterHit>());
289                 }
290                 if (!calHitMap.get(p).contains(hit)) {
291                     calHitMap.get(p).add(hit);
292                 }
293             }
294         }
295         return calHitMap;
296     }

298     public static int trackHitLayerNum(IDetectorElement de, List<HpsSiSensor> sensors, boolean inactive) {
299         if(!inactive){
300             try{
301                 HpsSiSensor sensor = ((HpsSiSensor) de);
302                 return sensor.getLayerNumber();
303             }
304             catch(ClassCastException e){
305                 System.out.println("Sim tracker hit does not correspond to a sensor! " + e.getMessage());
306                 return -1;
307             }
308         }
309         else{
310             HpsSiSensor sensor = null;
311             for (HpsSiSensor s : sensors){
312                 try{
313                     String name = de.getName() + "_sensor0";
314                     if(s.getName().equals(name)){
315                         sensor = s;
316                         break;
317                     }
318                 }
319                 catch(ClassCastException e){
320                     System.out.println("Sim tracker hit does not correspond to a sensor! " + e.getMessage());
321                     return -1;
322                 }
323             }
324             if(sensor != null){
325                 return sensor.getLayerNumber();
326             }
327             else{
328                 return -1;
329             }
330         }
331     }

333     public static double deltaThetaX(Hep3Vector p1, Hep3Vector p2){
334         double theta1 = p1.x()/p1.z();
335         double theta2 = p2.x()/p2.z();
336         return theta2 - theta1;
337     }

339     public static double deltaThetaY(Hep3Vector p1, Hep3Vector p2){
340         double theta1 = p1.y()/p1.z();
341         double theta2 = p2.y()/p2.z();
342         return theta2 - theta1;
343     }

345     public static Pair<Hep3Vector,Hep3Vector> extrapolateTrack(Hep3Vector endPosition, Hep3Vector endMomentum, Hep3Vector startPosition, double stepSize, FieldMap fieldMap, double q) {
346         if(endPosition == null || endMomentum == null || startPosition == null) return null;
347         // Start by transforming detector vectors into tracking frame
348         Hep3Vector currentPosition = CoordinateTransformations.transformVectorToTracking(endPosition);
349         Hep3Vector currentMomentum = VecOp.neg(CoordinateTransformations.transformVectorToTracking(endMomentum));
350         double startPositionZ = startPosition.z();
351 
352         // Retrieve the y component of the bfield in the middle of detector
353         double bFieldY = fieldMap.getField(new BasicHep3Vector(0, 0, 500.0)).y();       
354     
355         // HACK: LCSim doesn't deal well with negative fields so they are
356         // turned to positive for tracking purposes. As a result,
357         // the charge calculated using the B-field, will be wrong
358         // when the field is negative and needs to be flipped.
359         if (bFieldY < 0)
360             q = q * (-1);
361 
362         // Swim the track through the B-field until the end point is reached.
363         // The position of the track will be incremented according to the step
364         // size up to ~90% of the final position. At this point, a finer
365         // track size will be used.
366         boolean stepSizeChange = false;
367         while (currentPosition.x() > startPositionZ) {      
368             // The field map coordinates are in the detector frame so the
369             // extrapolated track position needs to be transformed from the
370             // track frame to detector.
371             Hep3Vector currentPositionDet = CoordinateTransformations.transformVectorToDetector(currentPosition);
372 
373             // Get the field at the current position along the track.
374             bFieldY = fieldMap.getField(currentPositionDet).y();
375 
376             // Get a trajectory (Helix or Line objects) created with the
377             // track parameters at the current position.
378             Trajectory trajectory = TrackUtils.getTrajectory(currentMomentum, new org.lcsim.spacegeom.SpacePoint(currentPosition), q, bFieldY);
379 
380             // Using the new trajectory, extrapolated the track by a step and
381             // update the extrapolated position.
382             //if(CoordinateTransformations.transformVectorToTracking(startPosition).x() > currentPosition.x()){
383             if(currentMomentum.x() > 0){
384                 //System.out.println("Track is going Forwards!");
385                 return null;
386             }
387             
388             //If trajectory step crosses starting position, adjust step size to fall on starting position
389             if(trajectory.getPointAtDistance(stepSize).x() < startPositionZ){
390                 stepSize = VecOp.sub(CoordinateTransformations.transformVectorToTracking(startPosition),currentPosition).magnitude();
391             }
392             
393             currentPosition = trajectory.getPointAtDistance(stepSize);
394 
395             // Calculate the momentum vector at the new position. This will
396             // be used when creating the trajectory that will be used to
397             // extrapolate the track in the next iteration.
398             currentMomentum = VecOp.mult(currentMomentum.magnitude(), trajectory.getUnitTangentAtLength(stepSize));
399             
400             //System.out.println("Position " + CoordinateTransformations.transformVectorToDetector(currentPosition) + "   Momentum " + CoordinateTransformations.transformVectorToDetector(currentMomentum) + "   startPositionZ " + startPositionZ);
401         
402             //System.out.println(currentPosition.x() + "  " + currentPosition.y() + "  " + currentPosition.z() + "  ");
403 
404             // If the position of the track along X (or z in the detector frame)
405             // is at 90% of the total distance, reduce the step size.
406             if (currentPosition.x() / startPositionZ > .80 && !stepSizeChange) {
407                 stepSize /= 10;
408                 //System.out.println("Changing step size: " + stepSize);
409                 stepSizeChange = true;
410             }
411         }
412 
413         // Transform vector back to detector frame
414         Hep3Vector returnPos = CoordinateTransformations.transformVectorToDetector(currentPosition);
415         Hep3Vector returnP = CoordinateTransformations.transformVectorToDetector(currentMomentum);
416         return new Pair<>(returnPos,returnP);
417     }
