58      private void doAnalysis(Track trk, RelationalTable rawtomc, List<SimTrackerHit> allsimhits, boolean isKalman){
59          //  Get the number of hits on the track
60          _nhits = trk.getTrackerHits().size();
61          
62          if (!isKalman)
63              _nhits = _nhits*2;
64          
65          _isTop = trk.getTrackStates().get(0).getTanLambda() > 0;
66  
67          for (TrackerHit hit : trk.getTrackerHits()) {
68              //  get the set of MCParticles associated with this hit and update the hit count for each MCParticle
69              List<RawTrackerHit> rawhits = hit.getRawHits();
70              int trackhitlayer = ((RawTrackerHit) hit.getRawHits().get(0)).getLayerNumber();
71              Set<MCParticle> mcPartList1 = new HashSet<MCParticle>();
72              Set<MCParticle> mcPartList2 = new HashSet<MCParticle>();
73              for(RawTrackerHit rawhit : rawhits){
74                  //Get all the SimTrackerHits associated with the raw hits
75                  Set<SimTrackerHit> simhits = rawtomc.allFrom(rawhit);
76                  for (SimTrackerHit simhit : simhits){
77                      if (simhit != null && simhit.getMCParticle() != null){
78                          simhitsontrack.add(simhit);
79                          MCParticle simhitpart = simhit.getMCParticle();
80                          int simlay = simhit.getLayer();
81                          if(simlay % 2 == 1)
82                              mcPartList1.add(simhitpart);
83                          if(simlay % 2 == 0)
84                              mcPartList2.add(simhitpart);
85                          if(mapGoodHitList.get(simhitpart) == null){
86                              Map<Integer,Boolean> dummy = new HashMap<Integer,Boolean>();
87                              dummy.put(simlay, true);
88                              mapGoodHitList.put(simhitpart, dummy);
89                          }
90                          if(mapGoodHitList.get(simhitpart).get(simlay) == null)
91                              mapGoodHitList.get(simhitpart).put(simlay,false);
92                          if(mcmap.get(simhitpart) == null)
93                              mcmap.put(simhitpart, 1);
94                          if(mapGoodHitList.get(simhitpart).get(simlay) == false)
95                              mcmap.put(simhitpart,mcmap.get(simhitpart) + 1);
96                          mapGoodHitList.get(simhitpart).put(simlay,true);
97                      }
98                      else{
99                          System.out.println("SIM HIT IS NULL!!!!!!!!");
100                     }
101                 }
102             }
103             _nMCHitsPerLayer.put(trackhitlayer,mcPartList1.size());
104             _nMCHitsPerLayer.put(trackhitlayer + 1,mcPartList2.size());
105             _hitMCPartList.put(trackhitlayer,mcPartList1);
106             _hitMCPartList.put(trackhitlayer + 1,mcPartList2);
107             trackerlayerhitlist.add(trackhitlayer);
108             trackerlayerhitlist.add(trackhitlayer + 1);
109         }
110         
111         //Create good hit map list
112         for (TrackerHit hit : trk.getTrackerHits()){
113             List<RawTrackerHit> rawhits = hit.getRawHits();
114             for(RawTrackerHit rawhit : rawhits){
115                 int layer = rawhit.getLayerNumber();
116                 for(MCParticle p : mapGoodHitList.keySet()){
117                     if(mapGoodHitList.get(p).get(layer) == null)
118                         mapGoodHitList.get(p).put(layer,false);
119                 }
120             }
121         }
122         
123 
124         _ngoodhits = GetMax(mcmap.values());
125         MCParticle mcbest = null;
126         
127         boolean has2 = false;
128         for(int i = 0; i < 12; i++){
129             if(_nMCHitsPerLayer.get(i+1) == null) continue;
130             if(_nMCHitsPerLayer.get(i+1) > 1){
131                 has2 = true;
132                 break;
133             }
134         }
135         
136         has2 = false;
137         
138         //Choose a list of the best MC Particles
139         //If more than one, choose the one with the innermost hits
140         for (MCParticle mcp : mcmap.keySet()) {
141             if(has2){
142                 if(mapGoodHitList.get(mcp) != null){
143                     for(int i = 0; i < mapGoodHitList.get(mcp).size(); i++){
144                         System.out.println(mapGoodHitList.get(mcp).get(i+1));
145                     }
146                 }
147             }
148             if (mcmap.get(mcp) == _ngoodhits) {
149                 mcbestlist.put(mcp, mapGoodHitList.get(mcp));
150             }
151         }
152         
153         //If there is more than 1 best MC, chose the one with the higher energy
154         if(mcbestlist.size() > 1)
155             mcbest = ChooseBest(mcbestlist,trackerlayerhitlist);
156         else if(mcbestlist.size() == 1){
157             List<MCParticle> plist = new ArrayList<MCParticle>();
158             plist.addAll(mcbestlist.keySet());
159             mcbest = plist.get(0);
160         }
161 
162         if (_ngoodhits > 0)
163             _mcp = mcbest;
164         _purity = (double) _ngoodhits / (double) _nhits;
165         _nbadhits = _nhits - _ngoodhits;
166         _goodHitList = mapGoodHitList.get(mcbest);
167         
168         
169         
170         for(SimTrackerHit hit : allsimhits){
171             if(hit.getMCParticle().equals(mcbest))
172                 mcbesthits.add(hit);
173         }
174         
175         for(SimTrackerHit besthit : mcbesthits){
176             boolean hastrackerhit = false;
177             for(SimTrackerHit hit : simhitsontrack){
178                 if(hit.equals(besthit)){
179                     hastrackerhit = true;
180                     break;
181                 }
182             }
183             if(!hastrackerhit)
184                 _hitListNotMatched.add(besthit);
185         }
186     }

188     //Get the maximum integer in the collection
189     private int GetMax(Collection<Integer> ilist){
190         int imax = 0;
191         for(Integer i : ilist){
192             if(i.intValue() > imax){
193                 imax = i.intValue();
194             }
195         }
196         return imax;
197     }

199     //This function chooses the particle with the highest energy
200     //This function needs to be improved
201     private MCParticle ChooseBest(Map<MCParticle,Map<Integer,Boolean>> pmap,List<Integer> trackerlayerhitlist){
202         List<MCParticle> plist = new ArrayList<MCParticle>();
203         plist.addAll(pmap.keySet());
204         MCParticle Pbest = plist.get(0);
205         double maxenergy = 0;
206         for(Map.Entry<MCParticle,Map<Integer,Boolean>> map : pmap.entrySet()){
207             MCParticle p = map.getKey();
208             double energy = p.getEnergy();
209             if(energy > maxenergy){
210                 energy = maxenergy;
211                 Pbest = p;
212             }
213             
214         }
215         return Pbest;
216     }
