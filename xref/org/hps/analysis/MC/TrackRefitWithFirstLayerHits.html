40   * This driver selects all 3D hits in the first layer of the track (L1 or L2),
41   * and refits on all 3D hits in that layer. This seed track can be fed to "FirstHitGBLRefitterDriver"
42   * to do a GBL Refit.
43   */

102     protected void process(EventHeader event) {
103         List<Track> tracks = event.get(Track.class, trackCollectionName);
104         List<TrackerHit> hits = event.get(TrackerHit.class,helicalTrackHitCollectionName);
105         List<LCRelation> helicalTrackHitToStripHitRelations = event.get(LCRelation.class,helicalTrackHitToStripHitRelationsCollectionName);
106         List<LCRelation> trackToFirstLayHitsRelationList = new ArrayList<LCRelation>();
107         List<Track> refitTracks = new ArrayList<Track>();
108         List<LCRelation> trackToTrackRefitList = new ArrayList<LCRelation>();
109         List<LCRelation> hthToStripHitRefitList = new ArrayList<LCRelation>();
110         List<HelicalTrackHit> helicalTrackHits_refit = new ArrayList<HelicalTrackHit>();
111         
112         //Loop over all the GBL Tracks in the event
113         for(Track track:tracks){
114             //Get Layer of the first hit on track (either L1 or L2)
115             int firstLayHit = (((HpsSiSensor) ((RawTrackerHit) track.getTrackerHits().get(0).getRawHits().get(0)).getDetectorElement()).getLayerNumber() + 1) / 2;
116             List<TrackerHit> newHits = new ArrayList<TrackerHit>();
117             //Loop over all helical track hits in the event
118             for(TrackerHit hit:hits){
119                 int layer = (((HpsSiSensor) ((RawTrackerHit) hit.getRawHits().get(0)).getDetectorElement()).getLayerNumber() + 1) / 2;
120                 //Check to see if the layer is the same as the first hit on track
121                 if(layer != firstLayHit)
122                     continue;
123                 //Check to see if hits are both top or bottom
124                 if(hit.getPosition()[1]*track.getTrackStates().get(0).getTanLambda() < 0)
125                     continue;
126                 boolean sameHit = checkSameHit(hit,track.getTrackerHits().get(0));
127                 //Check to see if the 3D hit is already on track
128                 if(sameHit)
129                     continue;
130                 //If the hit passes these requirements, add it to the list of hits to be refit
131                 newHits.add(hit);
132                 trackToFirstLayHitsRelationList.add(new BaseLCRelation(track, hit));
133             }
134             //Loop over the list of new hits for each track
135             //For each new hit, output a seed track to be refit
136             for(TrackerHit hit:newHits){
137                 Pair<Track,Pair<List<LCRelation>,List<HelicalTrackHit>>> newtrackPair = fitNewTrack(track,hit,helicalTrackHitToStripHitRelations); //track with new hits
138                 Track newtrack = newtrackPair.getFirstElement();
139                 refitTracks.add(newtrack);
140                 hthToStripHitRefitList.addAll(newtrackPair.getSecondElement().getFirstElement());
141                 helicalTrackHits_refit.addAll(newtrackPair.getSecondElement().getSecondElement());
142                 trackToTrackRefitList.add(new BaseLCRelation(track,newtrack));
143             }
144             
145         }
146         event.put(trackToFirstLayHitsRelationCollectionName, trackToFirstLayHitsRelationList, LCRelation.class, 0);
147         event.put(refitTracksCollectionName, refitTracks, Track.class, 0);
148         event.put(trackToTrackRefitListRelationCollectionName, trackToTrackRefitList, LCRelation.class, 0);
149         event.put(helicalTrackHitRefitCollectionName, helicalTrackHits_refit, HelicalTrackHit.class, 0);
150         event.put(helicalTrackHitRefitToStripHitRelationsCollectionName, hthToStripHitRefitList, LCRelation.class, 0);
151         addRotatedHitsToEvent(event, helicalTrackHits_refit); 
152     }

154     //Checks to see if two 3D hits are the same
155     private boolean checkSameHit(TrackerHit hit, TrackerHit trackerHit) {
156         List<RawTrackerHit> rawhits = hit.getRawHits();
157         List<RawTrackerHit> rawtrackerhits = trackerHit.getRawHits();
158         
159         //If the rawhits are different array sizes, they can't be the same hit
160         if(rawhits.size() != rawtrackerhits.size())
161             return false;
162         
163         boolean[] isSameRawHit = new boolean[rawhits.size()];
164         boolean[] isSameRawTrackerHit = new boolean[rawtrackerhits.size()];
165         
166         //Initialize boolean arrays
167         for(int i = 0; i < rawhits.size(); i++){
168             isSameRawHit[i] = false;
169         }
170         for(int i = 0; i < rawtrackerhits.size(); i++){
171             isSameRawTrackerHit[i] = false;
172         }
173         
174         int i = 0;
175         for(RawTrackerHit rawhit:rawhits){
176             int j = 0;
177             for(RawTrackerHit rawtrackerhit:rawtrackerhits){
178                 if(rawhit.equals(rawtrackerhit)){
179                     isSameRawHit[i] = true;
180                     isSameRawTrackerHit[j] = true;
181                     break;
182                 }
183                 j++;
184             }
185             i++;
186         }
187         
188         //If any of the raw hits are not the same, this is not the same 3D hit
189         for(int j = 0; j < isSameRawHit.length; j++){
190             if(!isSameRawHit[j])
191                 return false;
192         }
193         for(int j = 0; j < isSameRawTrackerHit.length; j++){
194             if(!isSameRawTrackerHit[j])
195                 return false;
196         }
197         return true;
198     }

200     private Pair<Track,Pair<List<LCRelation>,List<HelicalTrackHit>>> fitNewTrack(Track track, TrackerHit hitL1, List<LCRelation> helicalTrackHitToStripHitRelations){
201         //List all the standard seed tracking strategies, only use one for now. That's probably sufficient
202         String[] strategyResources = new String[4];
203         strategyResources[0] = "/org/hps/recon/tracking/strategies/HPS_s345_c2_e16.xml";
204         strategyResources[1] = "/org/hps/recon/tracking/strategies/HPS_s456_c3_e21.xml";
205         strategyResources[2] = "/org/hps/recon/tracking/strategies/HPS_s123_c4_e56.xml";
206         strategyResources[3] = "/org/hps/recon/tracking/strategies/HPS_s123_c5_e46.xml";
207         
208         List<SeedStrategy> sFinallist = StrategyXMLUtils.getStrategyListFromInputStream(this.getClass().getResourceAsStream(strategyResources[0]));
209         List<LCRelation> hthToStripHitRefitList = new ArrayList<LCRelation>();
210         List<HelicalTrackHit> helicalHits = new ArrayList<HelicalTrackHit>();
211 
212         HelicalTrackStrip strip1 = null;
213         HelicalTrackStrip strip2 = null;
214         TrackerHit hit1 = null;
215         TrackerHit hit2 = null;
216         HelicalTrackHit hth = null;
217         
218         //Build the HelicalTrackStrip for the first layer refit
219         for(LCRelation rel:helicalTrackHitToStripHitRelations){
220             if(((TrackerHit) rel.getFrom()).equals(hitL1)){
221                 HelicalTrackStrip strip = makeDigiStrip(new SiTrackerHitStrip1D((TrackerHit) rel.getTo()));
222                 if(strip1 == null){
223                     strip1 = strip;
224                     hit1 = (TrackerHit) rel.getTo();
225                 }
226                 else{
227                     strip2 = strip;
228                     hit2 = (TrackerHit) rel.getTo();
229                     break;
230                 }
231             }
232         }
233         
234         //Build the HelicalTrackHit for the first layer refit
235         if(strip1 != null && strip2 != null){
236             HelicalTrackCross trackCross = new HelicalTrackCross(strip1,strip2);
237             hth = (HelicalTrackHit) trackCross;
238             helicalHits.add(hth);
239         }
240         
241         List<TrackerHit> hits = track.getTrackerHits();
242         
243         //Build the HelicalTrackStrips for the remaining layers
244         for(TrackerHit hit:hits){
245             if(hit.equals(hits.get(0))){
246                 hthToStripHitRefitList.add(new BaseLCRelation(hth,hit1));
247                 hthToStripHitRefitList.add(new BaseLCRelation(hth,hit2));
248                 continue;
249             }
250             strip1 = null;
251             strip2 = null;
252             for(LCRelation rel:helicalTrackHitToStripHitRelations){
253                 if(checkSameHit((TrackerHit) rel.getFrom(),hit)){
254                     HelicalTrackStrip strip = makeDigiStrip(new SiTrackerHitStrip1D((TrackerHit) rel.getTo()));
255                     if(strip1 == null){
256                         strip1 = strip;
257                         hit1 = (TrackerHit) rel.getTo();
258                     }
259                     else{
260                         strip2 = strip;
261                         hit2 = (TrackerHit) rel.getTo();
262                         break;
263                     }
264                 }
265             }
266             //Build the HelicalTrackHit for the remaining layers
267             if(strip1 != null && strip2 != null){
268                 HelicalTrackCross trackCross = new HelicalTrackCross(strip1,strip2);
269                 hth = (HelicalTrackHit) trackCross;
270                 helicalHits.add(hth);
271                 hthToStripHitRefitList.add(new BaseLCRelation(hth,hit1));
272                 hthToStripHitRefitList.add(new BaseLCRelation(hth,hit2));
273             }
274         }
275         
276         SeedCandidate trackseed = new SeedCandidate(helicalHits, sFinallist.get(0), TrackUtils.getHTF(track.getTrackStates().get(0)), bfield);
277         
278         //  Initialize the reference point to the origin
279         double[] ref = new double[] {0., 0., 0.};
280         
281             //  Create a new SeedTrack (SeedTrack extends BaseTrack)
282         SeedTrack trk = new SeedTrack();
283             
284             //  Add the hits to the track
285         for (HelicalTrackHit hit : trackseed.getHits()) {
286             trk.addHit((TrackerHit) hit);
287         }
288             
289             //  Retrieve the helix and save the relevant bits of helix info
290         HelicalTrackFit helix = trackseed.getHelix();
291         trk.setTrackParameters(helix.parameters(), bfield); // Sets first TrackState.
292         trk.setCovarianceMatrix(helix.covariance()); // Modifies first TrackState.
293         trk.setChisq(helix.chisqtot());
294         trk.setNDF(helix.ndf()[0]+helix.ndf()[1]);
295             
296             //  Flag that the fit was successful and set the reference point
297         trk.setFitSuccess(true);
298         trk.setReferencePoint(ref); // Modifies first TrackState.
299         trk.setRefPointIsDCA(true);
300             
301             //  Set the strategy used to find this track
302         trk.setStratetgy(trackseed.getSeedStrategy());
303             
304             //  Set the SeedCandidate this track is based on
305         trk.setSeedCandidate(trackseed);
306         
307         return new Pair<>((Track) trk,new Pair<>(hthToStripHitRefitList,helicalHits));
308     }

311     private HelicalTrackStrip makeDigiStrip(SiTrackerHitStrip1D h) {
312 
313         SiTrackerHitStrip1D local = h.getTransformedHit(TrackerHitType.CoordinateSystem.SENSOR);
314         SiTrackerHitStrip1D global = h.getTransformedHit(TrackerHitType.CoordinateSystem.GLOBAL);
315 
316         ITransform3D trans = local.getLocalToGlobal();
317         Hep3Vector org = trans.transformed(_orgloc);
318         Hep3Vector u = global.getMeasuredCoordinate();
319         Hep3Vector v = global.getUnmeasuredCoordinate();
320 
321         double umeas = local.getPosition()[0];
322         double vmin = VecOp.dot(local.getUnmeasuredCoordinate(), local.getHitSegment().getStartPoint());
323         double vmax = VecOp.dot(local.getUnmeasuredCoordinate(), local.getHitSegment().getEndPoint());
324         double du = Math.sqrt(local.getCovarianceAsMatrix().diagonal(0));
325 
326         IDetectorElement de = h.getSensor();
327         String det = _ID.getName(de);
328         int lyr = _ID.getLayer(de);
329         BarrelEndcapFlag be = _ID.getBarrelEndcapFlag(de);
330 
331         double dEdx = h.getdEdx();
332         double time = h.getTime();
333         List<RawTrackerHit> rawhits = h.getRawHits();
334         HelicalTrackStrip strip = new HelicalTrackStrip(org, u, v, umeas, du, vmin, vmax, dEdx, time, rawhits, det, lyr, be);
335 
336         return strip;
337     }

340     private void addRotatedHitsToEvent(EventHeader event, List<HelicalTrackHit> helicalHits) {
341 
342         List<HelicalTrackHit> rotatedhits = new ArrayList<HelicalTrackHit>();
343         List<LCRelation> hthrelations = new ArrayList<LCRelation>();
344         List<HelicalTrackCross> stereohits = new ArrayList<HelicalTrackCross>();
345         for(HelicalTrackHit hit:helicalHits){
346             stereohits.add((HelicalTrackCross) hit);
347         }
348         
349         for (HelicalTrackCross cross : stereohits) {
350             List<HelicalTrackStrip> rotatedstriphits = new ArrayList<HelicalTrackStrip>();
351             for (HelicalTrackStrip strip : cross.getStrips()) {
352 
353                 Hep3Vector origin = strip.origin();
354                 Hep3Vector u = strip.u();
355                 Hep3Vector v = strip.v();
356                 double umeas = strip.umeas();
357                 double du = strip.du();
358                 double vmin = strip.vmin();
359                 double vmax = strip.vmax();
360                 double dedx = strip.dEdx();
361                 double time = strip.time();
362                 List<RawTrackerHit> rthList = strip.rawhits();
363                 String detname = strip.detector();
364                 int layer = strip.layer();
365                 BarrelEndcapFlag bec = strip.BarrelEndcapFlag();
366                 Hep3Vector neworigin = CoordinateTransformations.transformVectorToTracking(origin);
367                 Hep3Vector newu = CoordinateTransformations.transformVectorToTracking(u);
368                 Hep3Vector newv = CoordinateTransformations.transformVectorToTracking(v);
369                 HelicalTrackStrip newstrip = new HelicalTrackStrip(neworigin, newu, newv, umeas, du, vmin, vmax, dedx, time, rthList, detname, layer, bec);
370                 rotatedstriphits.add(newstrip);
371             }
372             List<HelicalTrackStrip> strip1 = new ArrayList<HelicalTrackStrip>();
373             List<HelicalTrackStrip> strip2 = new ArrayList<HelicalTrackStrip>();
374             strip1.add(rotatedstriphits.get(0));
375             strip2.add(rotatedstriphits.get(1));
376             HelicalTrackCross newhit = new HelicalTrackCross(strip1.get(0),strip2.get(0));
377 
378             rotatedhits.add(newhit);
379             hthrelations.add(new MyLCRelation(cross, newhit));
380         }
381         event.put("Rotated" + _outname, rotatedhits, HelicalTrackHit.class, 0);
382         event.put("Rotated" + _hitrelname, hthrelations, LCRelation.class, 0);
383     }
