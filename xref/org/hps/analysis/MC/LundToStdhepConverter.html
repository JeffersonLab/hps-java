9   import java.io.File;

12  import java.io.InputStreamReader;

15  /**
16   * Reads in text file in LUND format and creates events in stdhep file format.
17   */
18  /* Header Format:
19  /* 1 Number of particles in event, blah, blah blah (user defined)
20   /*
21   /* Particle Format: 
22   /* 1 index 
23   /* 2 lifetime (ns) 
24   /* 3 type (1 is active) 
25   /* 4 pdgID 
26   /* 5 index of parent 
27   /* 6 index of child
28   /* 7 x momentum (GeV) 
29   /* 8 y momentum (GeV) 
30   /* 9 z momentum (GeV) 
31   /* 10 energy (GeV) 
32   /* 11 mass (GeV) 
33   /* 12 x vertex (cm) 
34   /* 13 y vertex (cm) 
35   /* 14 z vertex (cm)
36   /*
37   */
38  public class LundToStdhepConverter {
39  
40      public static void main(String args[]) throws IOException {
41  
42          // remind user of correct usage
43          if (args.length < 1) {
44              usage();
45          }
46          if (args.length == 1 && args[0].equals("-h")) {
47              usage();
48          }
49          String fileName = args[0];
50  
51          FileInputStream fin1 = new FileInputStream(fileName);
52  
53          double zOffset = 0.;
54          if (args.length > 1) {
55              zOffset = Double.parseDouble(args[1]);
56              System.out.println("Offsetting z vertex by "+zOffset+" mm.");
57          }
58          File outputDir = new File(".");
59          if (args.length > 2) {
60              outputDir = new File(args[2]);
61          }
62          // check if output directory exists
63  
64          if (!outputDir.exists()) {
65              System.out.println("\n\n  Directory " + outputDir + " does not exist!");
66              System.exit(1);
67          }
68  
69          int dot = fileName.lastIndexOf(".txt");
70          String stdhepFileName = fileName.substring(0, dot);
71  
72          stdhepFileName += ".stdhep";
73  
74          String outputFile = outputDir + "/" + stdhepFileName;
75          System.out.println(outputFile);
76          StdhepWriter w = null;
77          try {
78              w = new StdhepWriter(outputFile, "Imported Stdhep Events v1.0", "From file " + fileName, 10);
79              w.setCompatibilityMode(false);
80          } catch (java.io.IOException ex) {
81              System.err.println("Error opening file: " + outputFile);
82              ex.printStackTrace();
83              System.exit(1);
84          }
85  
86          /*--------------------------------------------------------*/
87   /* NEVHEP          - event number (or some special meaning*/
88   /*                    (see documentation for details)     */
89   /* NHEP            - actual number of entries in current  */
90   /*                    event.                              */
91   /* ISTHEP[IHEP]    - status code for IHEP'th entry - see  */
92   /*                    documentation for details           */
93   /* IDHEP [IHEP]    - IHEP'th particle identifier according*/
94   /*                    to PDG.                             */
95   /* JMOHEP[IHEP][0] - pointer to position of 1st mother    */
96   /* JMOHEP[IHEP][1] - pointer to position of 2nd mother    */
97   /* JDAHEP[IHEP][0] - pointer to position of 1st daughter  */
98   /* JDAHEP[IHEP][1] - pointer to position of 2nd daughter  */
99   /* PHEP  [IHEP][0] - X momentum [Gev/c]                   */
100  /* PHEP  [IHEP][1] - Y momentum [Gev/c]                   */
101  /* PHEP  [IHEP][2] - Z momentum [Gev/c]                   */
102  /* PHEP  [IHEP][3] - Energy [Gev]                         */
103  /* PHEP  [IHEP][4] - Mass[Gev/c^2]                        */
104  /* VHEP  [IHEP][0] - X vertex [mm]                        */
105  /* VHEP  [IHEP][1] - Y vertex [mm]                        */
106  /* VHEP  [IHEP][2] - Z vertex [mm]                        */
107  /* VHEP  [IHEP][3] - production time [mm/c]               */
108  /*========================================================*/
109  /* IstHep convention: */
110  /* 0      - final state particle if JdaHEP=0 */
111  /*          intermediate particle if JdaHEP>0 */
112  /*          (NEUGEN extension; was "null") */
113  /* 1      - final state particle */
114  /* 2      - intermediate state */
115  /* 3      - documentation line */
116  /* 4-10   - reserved for future */
117  /* 11-200 - reserved for specific model use */
118  /* 201+   - reserved for users */
119         int _nmax = 500000;
120         int _eventnum = 0;
121         int _nhep = 0;
122         int[] _isthep = new int[_nmax];
123         int[] _idhep = new int[_nmax];
124         int[] _jmohep = new int[2 * _nmax];
125         int[] _jdahep = new int[2 * _nmax];
126         double[] _phep = new double[5 * _nmax];
127         double[] _vhep = new double[4 * _nmax];
128 
129         //Dummy values...
130         int nevtreq = 1;
131         int nevtgen = 1;
132         int nevtwrt = 1;
133         float stdecom = 2.F;
134         float stdxsec = 2.F;
135         double stdseed1 = 137.;
136         double stdseed2 = 138.;
137 
138         double[] mom = new double[3];
139         double[] pos = new double[3];
140 
141         String thisLine;
142 
143         // write a begin run record
144         w.writeRecord(new StdhepBeginRun(nevtreq, nevtgen, nevtwrt, stdecom, stdxsec, stdseed1, stdseed2));
145         // now loop over contents of this file...
146         FileInputStream fin2 = new FileInputStream(fileName);
147         try {
148             BufferedReader myInput = new BufferedReader(new InputStreamReader(fin2));
149             double[] values = new double[14];
150             int nEvent = 0;
151             // read header to get number of particles in event.
152             while ((thisLine = myInput.readLine()) != null) {
153                 // tokenize the string and convert to double values
154                 StringTokenizer st = new java.util.StringTokenizer(thisLine, " ");
155                 int nParticles = Integer.valueOf(st.nextToken()).intValue();
156                 if (nParticles >= _nmax) {
157                     throw new RuntimeException("\nAre you sure you want to create an event with more than " + _nmax + " particles?\nIf so, please recompile.");
158                 }
159                 _nhep = 0;
160                 // read file line by line, each line represents one particle
161                 for (int i = 0; i < nParticles; ++i) {
162                     thisLine = myInput.readLine();
163                     int j = 0;
164                     // tokenize the string and convert to double values
165                     st = new java.util.StringTokenizer(thisLine, " ");
166                     int numTokens = st.countTokens();
167 
168                     while (st.hasMoreElements()) {
169                         values[j++] = Double.parseDouble(st.nextToken());
170                     }
171                     pos[0] = values[11] * 10.; //convert cm to mm
172                     mom[0] = values[6];
173                     pos[1] = values[12] * 10.;//convert cm to mm
174                     mom[1] = values[7];
175                     pos[2] = values[13] * 10. + zOffset;//convert cm to mm and translate z by zOffset
176                     mom[2] = values[8];
177 
178                     double energy = values[9];
179                     // now populate the HEPEVT "common block"
180                     _isthep[_nhep] = 1; // final state particle
181                     _idhep[_nhep] = (int) values[3];
182                     _phep[0 + 5 * _nhep] = mom[0]; //px
183                     _phep[1 + 5 * _nhep] = mom[1]; //py
184                     _phep[2 + 5 * _nhep] = mom[2]; //pz
185                     _phep[3 + 5 * _nhep] = energy; //E
186                     _phep[4 + 5 * _nhep] = values[10]; // mass
187                     _vhep[0 + 4 * _nhep] = pos[0]; //x
188                     _vhep[1 + 4 * _nhep] = pos[1]; //y
189                     _vhep[2 + 4 * _nhep] = pos[2]; //z
190                     // increment the number of particles in this event
191                     _nhep++;
192                 }
193                 //
194                 // Create an StdhepEvent and write it out...
195                 //
196                 StdhepEvent event = new StdhepEvent(_eventnum++, _nhep, _isthep, _idhep, _jmohep, _jdahep, _phep, _vhep);
197                 w.writeRecord(event);
198                 nEvent++;
199             }
200             // write an end run record
201             w.writeRecord(new StdhepEndRun(nevtreq, nevtgen, nevtwrt, stdecom, stdxsec, stdseed1, stdseed2));
202             // close the file
203             try {
204                 System.out.println(" Closing file: " + outputFile + " with " + nEvent + " events");
205                 w.close();
206             } catch (java.io.IOException ex) {
207                 System.err.println("Error closing file: " + outputFile);
208                 ex.printStackTrace();
209                 System.exit(1);
210             }
211             fin2.close();
212         } catch (EOFException ex) {
213             ex.printStackTrace();
214         }
215     }
216 
217     public static void usage() {
218         System.out.println("LundToStdhepConvertor: \n  an application to read in text files in LUND format and convert to stdhep format.\n");
219         System.out.println("Usage: \n\n >> java LundToStdhepConvertor InputFile.txt <zOffSet[mm]> <output directory> \n");
220         System.out.println(" Where: \n InputFile.txt    is an input text file in Lund format to process ");
221         System.out.println(" Writes to the current working directory unless output directory is specified");
222         System.out.println("\n e.g. >> java LundToStdhepConvertor LundEvents.txt -5 \n");
223         System.out.println("  will convert the events in LundEvents.txt to LundEvents.stdhep offset by -5mm in z in the same directory");
224         System.exit(0);
225     }
