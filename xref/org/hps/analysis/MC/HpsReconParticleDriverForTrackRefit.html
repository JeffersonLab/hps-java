83      public void findV0s(List<ReconstructedParticle> electrons, List<ReconstructedParticle> positrons,  List<ReconstructedParticle> uncV0Candidates, List<Vertex> uncV0Vertices, List<ReconstructedParticle> bscV0Candidates, List<Vertex> bscV0Vertices, List<ReconstructedParticle> targV0Candidates, List<Vertex> targV0Vertices) {
84          for (ReconstructedParticle positron : positrons) {
85              for (ReconstructedParticle electron : electrons) {
86                  // Don't vertex a GBL track with a SeedTrack.
87                  if (TrackType.isGBL(positron.getType()) != TrackType.isGBL(electron.getType()))
88                      continue;
89                  // Make V0 candidates
90                  this.makeV0Candidates(electron, positron, uncV0Candidates, uncV0Vertices, bscV0Candidates, bscV0Vertices, targV0Candidates, targV0Vertices);
91              }
92          }
93      }

122     /**
123      * Fits a vertex from an electron/positron track pair using the indicated
124      * constraint.
125      *
126      * @param constraint - The constraint type to use.
127      * @param electron - The electron track.
128      * @param positron - The positron track.
129      * @return Returns the reconstructed vertex as a <code>BilliorVertex
130      * </code> object.
131      * mg--8/14/17--add the displaced vertex refit for the UNCONSTRAINED and
132      * BS_CONSTRAINED fits
133      */
134     private BilliorVertex fitVertex(Constraint constraint, ReconstructedParticle electron, ReconstructedParticle positron) {
135 
136         // Covert the tracks to BilliorTracks.
137         BilliorTrack electronBTrack = toBilliorTrack(electron.getTracks().get(0));
138         BilliorTrack positronBTrack = toBilliorTrack(positron.getTracks().get(0));
139 
140         // Create a vertex fitter from the magnetic field.
141         BilliorVertexer vtxFitter = new BilliorVertexer(bField);
142         // TODO: The beam size should come from the conditions database.
143         vtxFitter.setBeamSize(beamSize);
144         vtxFitter.setBeamPosition(beamPosition);
145         vtxFitter.setStoreCovTrkMomList(_storeCovTrkMomList);
146         vtxFitter.setDebug(debug);
147 
148         // Perform the vertexing based on the specified constraint.
149         switch (constraint) {
150             case UNCONSTRAINED:
151                 vtxFitter.doBeamSpotConstraint(false);
152                 break;
153             case BS_CONSTRAINED:
154                 vtxFitter.doBeamSpotConstraint(true);
155                 break;
156             case TARGET_CONSTRAINED:
157                 vtxFitter.doTargetConstraint(true);
158                 break;
159         }
160 
161         // Add the electron and positron tracks to a track list for
162         // the vertex fitter.
163         List<BilliorTrack> billiorTracks = new ArrayList<BilliorTrack>();
164         billiorTracks.add(electronBTrack);
165         billiorTracks.add(positronBTrack);
166 
167         // Find a vertex based on the tracks.
168         BilliorVertex vtx = vtxFitter.fitVertex(billiorTracks);
169         
170         int minLayEle = 6;
171         int minLayPos = 6;
172         List<TrackerHit> allTrackHits = electron.getTracks().get(0).getTrackerHits();
173         for (TrackerHit temp : allTrackHits) {
174             // Retrieve the sensor associated with one of the hits. This will
175             // be used to retrieve the layer number
176             HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) temp.getRawHits().get(0)).getDetectorElement();
177 
178             // Retrieve the layer number by using the sensor
179             int layer = (sensor.getLayerNumber() + 1) / 2;
180             if (layer < minLayEle)
181                 minLayEle = layer;
182         }
183         allTrackHits = positron.getTracks().get(0).getTrackerHits();
184         for (TrackerHit temp : allTrackHits) {
185             // Retrieve the sensor associated with one of the hits. This will
186             // be used to retrieve the layer number
187             HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) temp.getRawHits().get(0)).getDetectorElement();
188 
189             // Retrieve the layer number by using the sensor
190             int layer = (sensor.getLayerNumber() + 1) / 2;
191             if (layer < minLayPos)
192                 minLayPos = layer;
193         }
194         vtx.setLayerCode(minLayPos + minLayEle);
195         vtx.setProbability(DOF[constraint.ordinal()]);
196 
197         // mg 8/14/17 
198         // if this is an unconstrained or BS constrained vertex, propogate the 
199         // tracks to the vertex found in previous fit and do fit again
200         //  ...  this is required because the vertex fit assumes trajectories 
201         // change linearly about the reference point (which we initially guess to be 
202         // (0,0,0) while for long-lived decays there is significant curvature
203         if (constraint == Constraint.BS_CONSTRAINED || constraint == Constraint.UNCONSTRAINED) {
204             List<ReconstructedParticle> recoList = new ArrayList<ReconstructedParticle>();
205             recoList.add(electron);
206             recoList.add(positron);
207             List<BilliorTrack> shiftedTracks = shiftTracksToVertex(recoList, vtx.getPosition());
208 //            if (constraint == Constraint.BS_CONSTRAINED) {
209 //                Hep3Vector beamRelToNewRef = new BasicHep3Vector(-vtx.getPosition().z() + beamPosition[0], -vtx.getPosition().x() + beamPosition[1], 0);
210 ////                vtxFitter.setBeamPosition(beamRelToNewRef.v());
211 //                //mg 5/11/2018:  use referencePostion, separate from beam position  
212 //                vtxFitter.setReferencePosition(beamRelToNewRef.v());
213 //            }
214  //mg 5/11/2018:  use referencePosition, separate from beam position  
215             Hep3Vector newRefPoint = new BasicHep3Vector(vtx.getPosition().z() , vtx.getPosition().x() , 0);          
216             vtxFitter.setReferencePosition(newRefPoint.v());
217 
218             BilliorVertex vtxNew = vtxFitter.fitVertex(shiftedTracks);
219             Hep3Vector vtxPosNew = VecOp.add(vtx.getPosition(), vtxNew.getPosition());//the refit vertex is measured wrt the original vertex position
220             vtxNew.setPosition(vtxPosNew);//just change the position...the errors and momenta are correct in re-fit
221             vtxNew.setLayerCode(vtx.getLayerCode());
222             vtxNew.setProbability(DOF[constraint.ordinal()]);
223             //  mg 5/11/2018:  correct for the beamspot position in the fitter instead of HPSReconParticle...
224 //            Hep3Vector vtxPosNew = VecOp.add(vtx.getPosition(), vtxNew.getPosition());//the refit vertex is measured wrt the original vertex position
225 //            vtxNew.setPosition(vtxPosNew);//just change the position...the errors and momenta are correct in re-fit
226 
227             return vtxNew;
228         } else
229             return vtx;
230     }

235     private void makeV0Candidates(ReconstructedParticle electron, ReconstructedParticle positron, List<ReconstructedParticle> uncV0Candidates, List<Vertex> uncV0Vertices, List<ReconstructedParticle> bscV0Candidates, List<Vertex> bscV0Vertices, List<ReconstructedParticle> targV0Candidates, List<Vertex> targV0Vertices) {
236         boolean eleIsTop = (electron.getTracks().get(0).getTrackerHits().get(0).getPosition()[2] > 0);
237         boolean posIsTop = (positron.getTracks().get(0).getTrackerHits().get(0).getPosition()[2] > 0);
238         
239         if ((eleIsTop == posIsTop))
240             return;
241         if (electron.getClusters() == null || positron.getClusters() == null)
242             return;
243         
244         // Create candidate particles for each constraint.
245         for (Constraint constraint : Constraint.values()) {
246 
247             // Generate a candidate vertex and particle.
248 
249             BilliorVertex vtxFit = fitVertex(constraint, electron, positron);
250             ReconstructedParticle candidate = makeReconstructedParticle(electron, positron, vtxFit);
251             // Add the candidate vertex and particle to the
252             // appropriate LCIO collection.
253             switch (constraint) {
254 
255                 case UNCONSTRAINED:
256                     // patch the track parameters at the found vertex
257                     if (_patchVertexTrackParameters)
258                         patchVertex(vtxFit);
259                     if (eleIsTop != posIsTop) {
260                         uncV0Vertices.add(vtxFit);
261                         uncV0Candidates.add(candidate);
262                     }
263                     break;
264 
265                 case BS_CONSTRAINED:
266                     if (eleIsTop != posIsTop) {
267                         bscV0Vertices.add(vtxFit);
268                         bscV0Candidates.add(candidate);
269                     }
270                     break;
271 
272                 case TARGET_CONSTRAINED:
273                     if (eleIsTop != posIsTop) {
274                         targV0Vertices.add(vtxFit);
275                         targV0Candidates.add(candidate);
276                     }
277                     break;
278 
279             }
280         }
281     }

283     /**
284      * Creates a reconstructed V0 candidate particle from an electron, positron,
285      * and billior vertex.
286      *
287      * @param electron - The electron.
288      * @param positron - The positron.
289      * @param vtxFit - The billior vertex.
290      * @return Returns a reconstructed particle with properties generated from
291      * the child particles and vertex given as an argument.
292      */
293     public static ReconstructedParticle makeReconstructedParticle(ReconstructedParticle electron, ReconstructedParticle positron, BilliorVertex vtxFit) {
294 
295         // Create a new reconstructed particle to represent the V0
296         // candidate and populate it with the electron and positron.
297         ReconstructedParticle candidate = new BaseReconstructedParticle();
298         ((BaseReconstructedParticle) candidate).setStartVertex(vtxFit);
299         candidate.addParticle(electron);
300         candidate.addParticle(positron);
301 
302         // Set the type of the V0 particle.  This will only be needed for pass 2.
303         ((BaseReconstructedParticle) candidate).setType(electron.getType());
304 
305         // TODO: The calculation of the total fitted momentum should be
306         //       done within BilloirVertex.
307         // Calculate the candidate particle momentum and associate it
308         // with the reconstructed candidate particle.
309         ((BaseReconstructedParticle) candidate).setMass(vtxFit.getParameters().get("invMass"));
310         Hep3Vector fittedMomentum = new BasicHep3Vector(vtxFit.getParameters().get("p1X"),
311                 vtxFit.getParameters().get("p1Y"),
312                 vtxFit.getParameters().get("p1Z"));
313         fittedMomentum = VecOp.add(fittedMomentum, new BasicHep3Vector(vtxFit.getParameters().get("p2X"),
314                 vtxFit.getParameters().get("p2Y"),
315                 vtxFit.getParameters().get("p2Z")));
316         //mg 10/24/2017...billiorvertex now returns momentum in JLAB frame
317         //         fittedMomentum = CoordinateTransformations.transformVectorToDetector(fittedMomentum);
318 
319         // If both the electron and positron have an associated Ecal cluster,
320         // calculate the total energy and assign it to the V0 particle
321         double v0Energy = 0;
322         if (!electron.getClusters().isEmpty() && !positron.getClusters().isEmpty()) {
323             v0Energy += electron.getClusters().get(0).getEnergy();
324             v0Energy += positron.getClusters().get(0).getEnergy();
325         }
326 
327         HepLorentzVector fourVector = new BasicHepLorentzVector(v0Energy, fittedMomentum);
328         //((BasicHepLorentzVector) fourVector).setV3(fourVector.t(), fittedMomentum);
329         ((BaseReconstructedParticle) candidate).set4Vector(fourVector);
330 
331         // Set the charge of the particle
332         double particleCharge = electron.getCharge() + positron.getCharge();
333         ((BaseReconstructedParticle) candidate).setCharge(particleCharge);
334 
335         // VERBOSE :: Output the fitted momentum data.
336 //        printDebug("Fitted momentum in tracking frame: " + fittedMomentum.toString());
337 //        printDebug("Fitted momentum in detector frame: " + fittedMomentum.toString());
338         // Add the ReconstructedParticle to the vertex.
339         vtxFit.setAssociatedParticle(candidate);
340 
341         // Set the vertex position as the reference point of the V0 particle
342         ((BaseReconstructedParticle) candidate).setReferencePoint(vtxFit.getPosition());
343 
344         // Return the V0 candidate.
345         return candidate;
346     }

378     private void patchVertex(BilliorVertex v) {
379         ReconstructedParticle rp = v.getAssociatedParticle();
380         List<ReconstructedParticle> parts = rp.getParticles();
381         ReconstructedParticle rp1 = parts.get(0);
382         ReconstructedParticle rp2 = parts.get(1);
383         //p1
384         Track et = rp1.getTracks().get(0);
385         double etrackMom = rp1.getMomentum().magnitude();
386         HelicalTrackFit ehtf = TrackUtils.getHTF(et);
387         // propagate this to the vertex z position...
388         // Note that HPS y is lcsim z
389         double es = HelixUtils.PathToZPlane(ehtf, v.getPosition().y());
390         //Hep3Vector epointOnTrackAtVtx = HelixUtils.PointOnHelix(ehtf, es);
391         Hep3Vector edirOfTrackAtVtx = HelixUtils.Direction(ehtf, es);
392         Hep3Vector emomAtVtx = VecOp.mult(etrackMom, VecOp.unit(edirOfTrackAtVtx));
393         //p2
394         Track pt = rp2.getTracks().get(0);
395         double ptrackMom = rp2.getMomentum().magnitude();
396         HelicalTrackFit phtf = TrackUtils.getHTF(pt);
397         // propagate this to the vertex z position...
398         // Note that HPS y is lcsim z
399         double ps = HelixUtils.PathToZPlane(phtf, v.getPosition().y());
400         //Hep3Vector ppointOnTrackAtVtx = HelixUtils.PointOnHelix(phtf, ps);
401         Hep3Vector pdirOfTrackAtVtx = HelixUtils.Direction(phtf, ps);
402         Hep3Vector pmomAtVtx = VecOp.mult(ptrackMom, VecOp.unit(pdirOfTrackAtVtx));
403 
404         double mass = invMass(emomAtVtx, pmomAtVtx);
405         v.setVertexTrackParameters(emomAtVtx, pmomAtVtx, mass);
406     }

408     private double invMass(Hep3Vector p1, Hep3Vector p2) {
409         double me2 = 0.000511 * 0.000511;
410         double esum = sqrt(p1.magnitudeSquared() + me2) + sqrt(p2.magnitudeSquared() + me2);
411         double pxsum = p1.x() + p2.x();
412         double pysum = p1.y() + p2.y();
413         double pzsum = p1.z() + p2.z();
414 
415         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
416         double evtmass = esum * esum - psum * psum;
417 
418         if (evtmass > 0)
419             return Math.sqrt(evtmass);
420         else
421             return -99;
422     }

424     private List<BilliorTrack> shiftTracksToVertex(List<ReconstructedParticle> particles, Hep3Vector vtxPos) {
425         ///     Ok...shift the reference point....        
426         double[] newRef = {vtxPos.z(), vtxPos.x(), 0.0};//the  TrackUtils.getParametersAtNewRefPoint method only shifts in xy tracking frame
427         List<BilliorTrack> newTrks = new ArrayList<BilliorTrack>();
428         for (ReconstructedParticle part : particles) {
429             BaseTrackState oldTS = (BaseTrackState) part.getTracks().get(0).getTrackStates().get(0);
430             double[] newParams = TrackUtils.getParametersAtNewRefPoint(newRef, oldTS);
431             SymmetricMatrix newCov = TrackUtils.getCovarianceAtNewRefPoint(newRef, oldTS.getReferencePoint(), oldTS.getParameters(), new SymmetricMatrix(5, oldTS.getCovMatrix(), true));
432             //mg...I don't like this re-casting, but toBilliorTrack only takes Track as input
433             BaseTrackState newTS = new BaseTrackState(newParams, newRef, newCov.asPackedArray(true), TrackState.AtIP, bField);
434             BilliorTrack electronBTrackShift = this.toBilliorTrack(newTS);
435             newTrks.add(electronBTrackShift);
436         }
437         return newTrks;
