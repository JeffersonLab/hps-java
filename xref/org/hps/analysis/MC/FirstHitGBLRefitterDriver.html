107     @Override
108     protected void detectorChanged(Detector detector) {
109         bfield = Math.abs(TrackUtils.getBField(detector).magnitude());
110         _scattering.getMaterialManager().buildModel(detector);
111         _scattering.setBField(bfield); // only absolute of B is needed as it's used for momentum calculation only
112         
113         bFieldMap = detector.getFieldMap();
114         
115         // Get the HpsSiSensor objects from the tracker detector element
116         sensors = detector.getSubdetector(SUBDETECTOR_NAME)
117                           .getDetectorElement().findDescendants(HpsSiSensor.class);
118         
119         trackerSubdet = detector.getSubdetector(SUBDETECTOR_NAME);
120     }

122     @Override
123     protected void process(EventHeader event) {
124         if (!event.hasCollection(Track.class, inputCollectionName)){
125             System.out.println("System has no input collection");
126             return;
127         }
128 
129         setupSensors(event);
130         List<Track> tracks = event.get(Track.class, inputCollectionName);
131         List<LCRelation> refitTrackToTrackRelations = event.get(LCRelation.class, refitTrackToTrackRelationsName);
132         RelationalTable hitToStrips = getHitToStripsTable(event,helicalTrackHitRelationsCollectionName);
133 
134         List<Track> refittedTracks = new ArrayList<Track>();
135         List<LCRelation> trackRelations = new ArrayList<LCRelation>();
136 
137         List<GBLKinkData> kinkDataCollection = new ArrayList<GBLKinkData>();
138         List<LCRelation> kinkDataRelations = new ArrayList<LCRelation>();
139         
140         List<LCRelation> trackToMCParticleRelations = new ArrayList<LCRelation>();
141         List<LCRelation> trackToTrackRefitRelations = new ArrayList<LCRelation>();
142 
143         Map<Track, Track> inputToRefitted = new HashMap<Track, Track>();
144         for (Track track : tracks) {
145             if (getStripHits(track, hitToStrips).size() == 0)
146                 continue;
147 
148             Pair<Track, GBLKinkData> newTrack = MakeGblTracks.refitTrack(TrackUtils.getHTF(track), getStripHits(track, hitToStrips), track.getTrackerHits(), 5, track.getType(), _scattering, bfield, storeTrackStates);
149             if (newTrack == null)
150                 continue;
151 
152             Track gblTrk = newTrack.getFirst();
153 
154             refittedTracks.add(gblTrk);
155             trackRelations.add(new BaseLCRelation(track, gblTrk));
156             inputToRefitted.put(track, gblTrk);
157             MCFullDetectorTruth truthMatch = new MCFullDetectorTruth(event, track, bFieldMap, sensors, trackerSubdet);
158             if(truthMatch != null){
159                 MCParticle p = truthMatch.getMCParticle();
160                 trackToMCParticleRelations.add(new BaseLCRelation(gblTrk,p));
161             }
162             kinkDataCollection.add(newTrack.getSecond());
163             kinkDataRelations.add(new BaseLCRelation(newTrack.getSecond(), gblTrk));
164             Track oldgblTrk = null;
165             for(LCRelation rel:refitTrackToTrackRelations){
166                 if(rel.getTo().equals(track)){
167                     oldgblTrk = (Track) rel.getFrom();
168                     break;
169                 }
170             }
171             trackToTrackRefitRelations.add(new BaseLCRelation(oldgblTrk,gblTrk));
172         }
173 
174         // Put the tracks back into the event and exit
175         int flag = 1 << LCIOConstants.TRBIT_HITS;
176         event.put(outputCollectionName, refittedTracks, Track.class, flag);
177         event.put(trackRelationCollectionName, trackRelations, LCRelation.class, 0);
178         event.put(kinkDataCollectionName, kinkDataCollection, GBLKinkData.class, 0);
179         event.put(kinkDataRelationsName, kinkDataRelations, LCRelation.class, 0);
180         event.put(trackToMCParticleRelationsName, trackToMCParticleRelations, LCRelation.class, 0);
181         event.put(trackToTrackRefitRelationsName, trackToTrackRefitRelations, LCRelation.class, 0);
182     }

184     private void setupSensors(EventHeader event) {
185         List<RawTrackerHit> rawTrackerHits = null;
186         if (event.hasCollection(RawTrackerHit.class, rawHitCollectionName))
187             rawTrackerHits = event.get(RawTrackerHit.class, rawHitCollectionName);
188         if (event.hasCollection(RawTrackerHit.class, "RawTrackerHitMaker_RawTrackerHits"))
189             rawTrackerHits = event.get(RawTrackerHit.class, "RawTrackerHitMaker_RawTrackerHits");
190 
191         EventHeader.LCMetaData meta = event.getMetaData(rawTrackerHits);
192         // Get the ID dictionary and field information.
193         IIdentifierDictionary dict = meta.getIDDecoder().getSubdetector().getDetectorElement().getIdentifierHelper().getIdentifierDictionary();
194         int fieldIdx = dict.getFieldIndex("side");
195         int sideIdx = dict.getFieldIndex("strip");
196         for (RawTrackerHit hit : rawTrackerHits) {
197             // if sensor already has a DetectorElement, skip it
198             if (hit.getDetectorElement() != null)
199                 continue;
200 
201             // The "side" and "strip" fields needs to be stripped from the ID for sensor lookup.
202             IExpandedIdentifier expId = dict.unpack(hit.getIdentifier());
203             expId.setValue(fieldIdx, 0);
204             expId.setValue(sideIdx, 0);
205             IIdentifier strippedId = dict.pack(expId);
206             // Find the sensor DetectorElement.
207             List<IDetectorElement> des = DetectorElementStore.getInstance().find(strippedId);
208             if (des == null || des.size() == 0)
209                 throw new RuntimeException("Failed to find any DetectorElements with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
210             else if (des.size() == 1)
211                 hit.setDetectorElement((SiSensor) des.get(0));
212             else
213                 // Use first sensor found, which should work unless there are sensors with duplicate IDs.
214                 for (IDetectorElement de : des)
215                     if (de instanceof SiSensor) {
216                         hit.setDetectorElement((SiSensor) de);
217                         break;
218                     }
219             // No sensor was found.
220             if (hit.getDetectorElement() == null)
221                 throw new RuntimeException("No sensor was found for hit with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
222         }
223     }

225     //These functions are adapted from the same ones in TrackUtils, but without the "cache"
226     static RelationalTable getHitToStripsTable(EventHeader event,String HelicalTrackHitRelationsCollectionName) {
227         RelationalTable hitToStrips = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
228         List<LCRelation> hitrelations = event.get(LCRelation.class, HelicalTrackHitRelationsCollectionName);
229         for (LCRelation relation : hitrelations)
230             if (relation != null && relation.getFrom() != null && relation.getTo() != null){
231                 hitToStrips.add(relation.getFrom(), relation.getTo());
232             }
233         return hitToStrips;
234     }
235 
236     static RelationalTable getHitToRotatedTable(EventHeader event, String RotatedHelicalTrackHitRelationsCollectionName) {
237         RelationalTable hitToRotated = new BaseRelationalTable(RelationalTable.Mode.ONE_TO_ONE, RelationalTable.Weighting.UNWEIGHTED);
238         List<LCRelation> rotaterelations = event.get(LCRelation.class, RotatedHelicalTrackHitRelationsCollectionName);
239         for (LCRelation relation : rotaterelations)
240             if (relation != null && relation.getFrom() != null && relation.getTo() != null) {
241                 hitToRotated.add(relation.getFrom(), relation.getTo());
242             }
243         return hitToRotated;
244     }

246     public static List<TrackerHit> getStripHits(Track track, RelationalTable hitToStrips) {
247         List<TrackerHit> hits = new ArrayList<TrackerHit>();
248         for (TrackerHit hit : track.getTrackerHits()) {
249             hits.addAll(hitToStrips.allFrom(hit));
250         }
251         return hits;
252     }
