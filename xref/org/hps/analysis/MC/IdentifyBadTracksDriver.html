31  /**
32   * Driver to skim selected events from LCIO files based on 
33   * track picking up the wrong hit.
34   *
35   * It outputs the bad track, the MCParticle associated with the bad hit,
36   * the truth hits associated with the truth match, and the track of the 
37   * other particle associated with the bad hit (if any)
38   * This driver can only be run on MC readout with full truth.
39   */
40  public class IdentifyBadTracksDriver extends Driver{

71      public void detectorChanged(Detector detector){
72          
73          bFieldMap = detector.getFieldMap();
74          
75          // Get the HpsSiSensor objects from the tracker detector element
76          sensors = detector.getSubdetector(SUBDETECTOR_NAME)
77                            .getDetectorElement().findDescendants(HpsSiSensor.class);
78          
79          trackerSubdet = detector.getSubdetector(SUBDETECTOR_NAME);
80          
81          matcher.setBFieldMap(detector.getFieldMap());
82          BeamEnergyCollection beamEnergyCollection = 
83                  this.getConditionsManager().getCachedConditions(BeamEnergyCollection.class, "beam_energies").getCachedData();        
84          beamEnergy = beamEnergyCollection.get(0).getBeamEnergy();
85          matcher.setBeamEnergy(beamEnergy);

88      @Override
89      protected void process(EventHeader event){
90          List<Track> allTracks = event.get(Track.class, trackColName);
91          List<Track> badTracks = new ArrayList<Track>();
92          List<SimTrackerHit> truthHits = new ArrayList<SimTrackerHit>();
93          List<MCFullDetectorTruth> truthMatchWithBadTrack = new ArrayList<MCFullDetectorTruth>();
94          List<LCRelation> trackBadToTruthMatchRelations = new ArrayList<LCRelation>();
95          List<LCRelation> trackToTruthMatchRelations = new ArrayList<LCRelation>();
96          List<LCRelation> trackBadToBadMCParticleRelations = new ArrayList<LCRelation>();
97          List<LCRelation> trackBadToBadReconParticleRelations = new ArrayList<LCRelation>();
98          List<ReconstructedParticle> otherReconParticles = new ArrayList<ReconstructedParticle>();
99          List<Cluster> clusters = event.get(Cluster.class, ecalClustersCollectionName);
100         //Loop over all tracks
101         for(Track track:allTracks){
102             //Match the track to a MC truth particle
103             MCFullDetectorTruth truthMatch = new MCFullDetectorTruth(event, track, bFieldMap, sensors, trackerSubdet);
104             if(truthMatch.getMCParticle() == null){
105                 continue;
106             }
107             //Add the truth match to track LC relation
108             trackToTruthMatchRelations.add(new BaseLCRelation(track, truthMatch.getMCParticle()));
109             
110             //Check to see if the truth match contains a hit not associated with the truth particle
111             //(i.e. the purity is less than 1.0)
112             if((truthMatch.getPurity() == 1.0)){
113                 continue;
114             }
115             //Identify MCParticle responsible for bad hit
116             MCParticle badPart = SelectBadMCParticle(truthMatch,track);
117             Track badTrk = SelectBadTrack(truthMatch,track);
118             ReconstructedParticle badReconPart = null;
119             if(badTrk != null){
120                 badReconPart = makeReconstructedParticles(clusters,badTrk);
121                 otherReconParticles.add(badReconPart);
122             }
123 
124             truthHits = truthMatch.getActiveHitListMCParticle();
125             badTracks.add(track);
126             truthMatchWithBadTrack.add(truthMatch);
127             trackBadToTruthMatchRelations.add(new BaseLCRelation(track, truthMatch.getMCParticle()));
128             trackBadToBadMCParticleRelations.add(new BaseLCRelation(track, badPart));
129             trackBadToBadReconParticleRelations.add(new BaseLCRelation(track, badReconPart));
130         }
131 
132         //Fill the collections
133         event.put(otherParticleColName, otherReconParticles, ReconstructedParticle.class, 0);
134         event.put(badMCParticleRelationsColName, trackBadToBadMCParticleRelations, LCRelation.class, 0);
135         event.put(otherParticleRelationsColName, trackBadToBadReconParticleRelations, LCRelation.class, 0);
136         event.put(simhitOutputColName, truthHits, SimTrackerHit.class, 0);
137         event.put(badTrackColName, badTracks, Track.class, 0);
138         event.put(trackBadToTruthMatchRelationsOutputColName, trackBadToTruthMatchRelations, LCRelation.class, 0);
139         event.put(trackToTruthMatchRelationsOutputColName, trackToTruthMatchRelations, LCRelation.class, 0);
140     }

142     //This function identifies MCParticle responsible for bad hit
143     //It selects the particle that at the innermost bad hit
144     //If this has multiple particles, select the higher momentum particle
145     //TODO Use better selection criteria or incorporate all particles that contribute to bad hits on track
146 
147     MCParticle SelectBadMCParticle(MCFullDetectorTruth truthMatch, Track trk){
148         MCParticle badP = null;
149         List<TrackerHit> hits = trk.getTrackerHits();
150         //for(TrackerHit hit : hits){
151         for(int layer = 1; layer < 13; layer++){
152             if (truthMatch.getHitList(layer) == null || truthMatch.getHitMCParticleList(layer) == null)
153                 continue;
154             if(truthMatch.getHitList(layer))
155                 continue;
156             Set<MCParticle> badPList = truthMatch.getHitMCParticleList(layer);
157             if(badPList ==  null)
158                 continue;
159             double maxP = 0.0;
160             for(MCParticle part : badPList){
161                 double p = part.getMomentum().magnitude();
162                 if(p > maxP){
163                     badP = part;
164                     break;
165                 }
166             }
167             if(badP != null)
168                 return badP;
169         }
170         return badP;
171     }
172     
173     //This function identifies a Track responsible for bad hit
174     //It selects the track that at the innermost bad hit
175     //If this has multiple tracks, select the higher momentum track
176     //TODO Use better selection criteria or incorporate all tracks that contribute to bad hits on track
177     
178     Track SelectBadTrack(MCFullDetectorTruth truthMatch, Track trk){
179         Track badTrk = null;
180         List<TrackerHit> hits = trk.getTrackerHits();
181         for(TrackerHit hit : hits){
182             int layer = ((RawTrackerHit) hit.getRawHits().get(0)).getLayerNumber();
183             if(truthMatch.getHitList(layer))
184                 continue;
185             Set<Track> badTrkList = truthMatch.getHitTrackList(layer);
186             if(badTrkList ==  null)
187                 continue;
188             double maxP = 0.0;
189             for(Track track : badTrkList){
190                 double p = Math.abs(TrackUtils.getHTF(track.getTrackStates().get(0)).p(bFieldMap.getField(new BasicHep3Vector(0, 0, 500)).y()));
191                 if(p > maxP){
192                     badTrk = track;
193                     break;
194                 }
195             }
196             if(badTrk != null){
197                 return badTrk;
198             }
199         }
200         return badTrk;
201     }

203     protected ReconstructedParticle makeReconstructedParticles(List<Cluster> clusters, Track track) {
204 
205         // Loop through all of the track collections and try to match every
206         // track to a cluster. Allow a cluster to be matched to multiple
207         // tracks and use a probability (to be coded later) to determine what
208         // the best match is.
209 
210         // Create a reconstructed particle to represent the track.
211         ReconstructedParticle particle = new BaseReconstructedParticle();
212 
213         // Store the track in the particle.
214         particle.addTrack(track);
215 
216         // Set the type of the particle. This is used to identify
217         // the tracking strategy used in finding the track associated with
218         // this particle.
219         ((BaseReconstructedParticle) particle).setType(track.getType());
220 
221         // Derive the charge of the particle from the track.
222         int flipSign = 1;
223         int charge = (int) Math.signum(track.getTrackStates().get(0).getOmega());
224         ((BaseReconstructedParticle) particle).setCharge(charge * flipSign);
225 
226         // initialize PID quality to a junk value:
227         ((BaseReconstructedParticle) particle).setGoodnessOfPid(9999);
228 
229         // Extrapolate the particle ID from the track. Positively
230         // charged particles are assumed to be positrons and those
231         // with negative charges are assumed to be electrons.
232         if (particle.getCharge() > 0) {
233             ((BaseReconstructedParticle) particle).setParticleIdUsed(new SimpleParticleID(-11, 0, 0, 0));
234         } else if (particle.getCharge() < 0) {
235             ((BaseReconstructedParticle) particle).setParticleIdUsed(new SimpleParticleID(11, 0, 0, 0));
236         }
237 
238         // normalized distance of the closest match:
239         double smallestNSigma = Double.MAX_VALUE;
240 
241         // try to find a matching cluster:
242         Cluster matchedCluster = null;
243         for (Cluster cluster : clusters) {
244             //double clusTime = ClusterUtilities.getSeedHitTime(cluster);
245             //double trkT = TrackUtils.getTrackTime(track, hitToStrips, hitToRotated);
246 
247                     
248             //if the option to use corrected cluster positions is selected, then
249             //create a copy of the current cluster, and apply corrections to it
250             //before calculating nsigma.  Default is don't use corrections.  
251             Cluster originalCluster = cluster;
252                     
253             // normalized distance between this cluster and track:
254             final double thisNSigma = matcher.getNSigmaPosition(cluster, particle);
255 
256             // ignore if matching quality doesn't make the cut:
257             if (thisNSigma > MAXNSIGMAPOSITIONMATCH)
258                 continue;
259 
260             // ignore if we already found a cluster that's a better match:
261             if (thisNSigma > smallestNSigma)
262                 continue;
263 
264             // we found a new best cluster candidate for this track:
265             smallestNSigma = thisNSigma;
266             matchedCluster = originalCluster;
267         }
268 
269         // If a cluster was found that matches the track...
270         if (matchedCluster != null) {
271 
272             // add cluster to the particle:
273             particle.addCluster(matchedCluster);
274 
275             // use pid quality to store track-cluster matching quality:
276             ((BaseReconstructedParticle) particle).setGoodnessOfPid(smallestNSigma);
277 
278             // propogate pid to the cluster:
279             final int pid = particle.getParticleIDUsed().getPDG();
280             if (Math.abs(pid) == 11) {
281                 if (!disablePID)
282                     ((BaseCluster) matchedCluster).setParticleId(pid);
283             }
284         }
285         return particle;
286     }
