29  /**
30   * This driver creates an MCParticle relation to be persisted for each track collection
31   * It also saves a TruthTrack
32   */
33  public class TrackToMCParticleRelationsDriver extends Driver {
34      
35      //Collection Names
36      private String trackCollectionName = "GBLTracks";
37      
38      //If the tracks are kalman tracks
39      private boolean kalmanTracks     = true;
40  
41      private double bfield;
42      private double bfield_y;
43  
44      private boolean debug = false;
45      private boolean saveTruthTracks = true;
46      
47      
48      public void setTrackCollectionName(String trackCollectionName) {
49          this.trackCollectionName = trackCollectionName;
50      }
51  
52      public void setKalmanTracks(boolean val) {
53          kalmanTracks = val;
54      }
55  
56      public void setDebug(boolean val) {
57          debug = val;
58      }
59      
60      public void setSaveTruthTracks(boolean val) {
61          saveTruthTracks = val;
62      }
63      
64      /*
65        public void setTrackCollectionNames(String [] trackCollectionNames) {
66        this.trackCollectionNames = trackCollectionNames;
67        }
68      */
69      
70      @Override
71      protected void detectorChanged(Detector detector) {
72          Hep3Vector fieldInTracker = TrackUtils.getBField(detector);
73          bfield = Math.abs(fieldInTracker.y());
74          bfield_y = fieldInTracker.y();
75      }
76      
77      @Override 
78      protected void process(EventHeader event) {
79          
80          //Retrieve track collection
81          List<Track> trackCollection = new ArrayList<Track>();
82          if (trackCollectionName != null) {
83              if (event.hasCollection(Track.class, trackCollectionName)) {
84                  trackCollection = event.get(Track.class, trackCollectionName);
85              }
86          }
87          
88          //Retrieve rawhits to mc
89          
90          RelationalTable rawtomc = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
91          if (event.hasCollection(LCRelation.class, "SVTTrueHitRelations")) {
92              List<LCRelation> trueHitRelations = event.get(LCRelation.class, "SVTTrueHitRelations");
93              for (LCRelation relation : trueHitRelations)
94                  if (relation != null && relation.getFrom() != null && relation.getTo() != null)
95                      rawtomc.add(relation.getFrom(), relation.getTo());
96          }
97          else
98              return;
99  
100         //Retrieve all simulated hits
101         String MCHitInputCollectionName = "TrackerHits";
102         List<SimTrackerHit> allsimhits = event.get(SimTrackerHit.class, MCHitInputCollectionName);
103 
104 
105         //MCParticleRelations
106         
107         List<LCRelation> trackToMCParticleRelations    =  new ArrayList<LCRelation>();
108         
109         //Truth Tracks and Relations
110         List<LCRelation> trackToTruthTrackRelations    =  new ArrayList<LCRelation>();
111         List<Track>      truthTrackCollection          =  new ArrayList<Track>();
112         
113         for (Track track : trackCollection) {
114             
115             //Truth Matching tool
116             TrackTruthMatching ttm = new TrackTruthMatching(track, rawtomc, allsimhits, kalmanTracks);
117             if (ttm != null) {
118                 MCParticle mcp = ttm.getMCParticle();
119                 
120                 if (mcp != null) {
121                     trackToMCParticleRelations.add(new BaseLCRelation(track,mcp));
122                     
123                     //Hep3Vector origin = new BasicHep3Vector(0.,0.,0.);
124                     HelicalTrackFit mcp_htf  = TrackUtils.getHTF(mcp,bfield);
125                     HelicalTrackFit trk_htf  = TrackUtils.getHTF(track);
126                 
127                     if (debug) {
128                         System.out.println("--------------------");
129                         System.out.println(trackCollectionName+" Track:");
130                         System.out.printf("d0 %f z0 %f R %f phi %f lambda %s\n", trk_htf.dca(), trk_htf.z0(), trk_htf.R(), trk_htf.phi0(), trk_htf.slope());
131                         System.out.printf("Nhits = %d \n",ttm.getNHits());
132                         System.out.printf("NGoodHits = %d  purity = %f\n",ttm.getNGoodHits(),ttm.getPurity());
133                         
134                     }
135                     BaseTrack truth_trk  = new BaseTrack();
136                     truth_trk.setTrackParameters(mcp_htf.parameters(),bfield);
137                     truth_trk.getTrackStates().clear();
138                     double[] ref = new double[] { 0., 0., 0. };
139                     SymmetricMatrix cov = new SymmetricMatrix(5);
140                     TrackState stateIP = new BaseTrackState(mcp_htf.parameters(),ref,cov.asPackedArray(true),TrackState.AtIP,bfield);
141                     truth_trk.getTrackStates().add(stateIP);
142                     truth_trk.setChisq(-1);
143                     truth_trk.setNDF(-1);
144                     truth_trk.setFitSuccess(false);
145                     truth_trk.setRefPointIsDCA(true);
146                     truth_trk.setTrackType(-1);
147                     truthTrackCollection.add(truth_trk);
148                     trackToTruthTrackRelations.add(new BaseLCRelation(track,truth_trk));
149                     
150                     
151                     if (debug) {
152                         double d0    = truth_trk.getTrackStates().get(0).getD0();
153                         double z0    = truth_trk.getTrackStates().get(0).getZ0();
154                         double C     = truth_trk.getTrackStates().get(0).getOmega();
155                         double phi   = truth_trk.getTrackStates().get(0).getPhi();
156                         double slope = truth_trk.getTrackStates().get(0).getTanLambda();
157                         System.out.printf("TruthTrack \n");
158                         System.out.printf("d0 %f z0 %f R %f phi %f lambda %s\n", d0, z0, 1./C, phi, slope);
159                         System.out.printf("MCParticle  \n");
160                         Hep3Vector pVec = mcp.getMomentum();
161                         System.out.printf("ptTrue = %f \n", Math.sqrt(pVec.x()*pVec.x() + pVec.z()*pVec.z()));
162                         double mom_param = 2.99792458e-04;
163                         double trkMom = trk_htf.R() * bfield * mom_param;
164                         System.out.printf("pt = %f \n", trkMom);
165                         
166                         System.out.println("--------------------");
167                     }
168                 }//mcp not null
169                 else {
170                     System.out.printf("PF::FakeTrack");
171                 }
172             } //ttm not null
173             else {
174                 System.out.printf("Error::TrackTruthMatching returns null \n");
175             }
176         }
177         
178         int flag = 1 << LCIOConstants.TRBIT_HITS;
179         event.put(trackCollectionName+"Truth", truthTrackCollection, Track.class, flag);
180         event.put(trackCollectionName+"ToTruthTrackRelations", trackToTruthTrackRelations, LCRelation.class, 0);
181         event.put(trackCollectionName+"ToMCParticleRelations", trackToMCParticleRelations, LCRelation.class, 0);
182     }//closes process
183 }
