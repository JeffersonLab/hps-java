31  /**
32   * Driver used to persist additional {@link org.lcsim.event.Track} information via a 
33   * {@link org.lcsim.event.GenericObject} collection. This driver is only useful for 
34   * refitting the truth hits on a track.
35   */
36  public final class TrackDataDriverWithTruth extends Driver {
37  
38      /** logger **/
39      private static final Logger LOGGER = Logger.getLogger(TrackDataDriver.class.getPackage().getName());
40  
41      /** The B field map */
42      FieldMap bFieldMap = null;
43      double bField = 0;
44  
45      /** Collection Names */
46      private String TRK_COLLECTION_NAME = "GBLTracks";
47      private String helicalTrackHitRelationsCollectionName = "HelicalTrackHitRelations";
48      private String rotatedHelicalTrackHitRelationsCollectionName = "RotatedHelicalTrackHitRelations";
49  
50      /** Collection name of TrackResidualData objects */
51      private static final String TRK_RESIDUALS_COL_NAME = "TrackResiduals";
52  
53      /** 
54       * Collection name of LCRelations between a Track and  TrackResidualsData
55       * objects.
56       */
57      private static final String TRK_RESIDUALS_REL_COL_NAME = "TrackResidualsRelations";
58      List<HpsSiSensor> sensors = null;
59  
60      /** Default constructor */
61      public TrackDataDriverWithTruth() {
62      }
63  
64      public void setTrackCollectionName(String input) {
65          TRK_COLLECTION_NAME = input;
66      }
67  
68      public void setHelicalTrackHitRelationsCollectionName(String helicalTrackHitRelationsCollectionName) {
69          this.helicalTrackHitRelationsCollectionName = helicalTrackHitRelationsCollectionName;
70      }
71  
72      public void setRotatedHelicalTrackHitRelationsCollectionName(String rotatedHelicalTrackHitRelationsCollectionName) {
73          this.rotatedHelicalTrackHitRelationsCollectionName = rotatedHelicalTrackHitRelationsCollectionName;
74      }
75      /**
76       * Method called by the framework when a new {@link Detector} geometry is
77       * loaded. This method is called at the beginning of every run and 
78       * provides access to the {@link Detector} object itself.
79       * 
80       * @param detector LCSim {@link Detector} geometry 
81       */
82      @Override
83      protected void detectorChanged(Detector detector) {
84  
85          // Get the field map from the detector object
86          bFieldMap = detector.getFieldMap();
87          bField = TrackUtils.getBField(detector).magnitude();
88          sensors = detector.getSubdetector("Tracker").getDetectorElement().findDescendants(HpsSiSensor.class);
89  
90      }
91  
92      /**
93       * Method called by the framework to process the event.
94       * 
95       * @param event : LCSim event
96       */
97      @Override
98      protected void process(EventHeader event) {
99  
100         // Check if the event contains a collection of the type Track. If it
101         // doesn't skip the event.
102         if (!event.hasCollection(Track.class))
103             return;
104         
105         int runNumber = event.getRunNumber();
106 
107         // Get all collections of the type Track from the event. This is
108         // required since the event contains a track collection for each of the
109         // different tracking strategies.
110         List<List<Track>> trackCollections = null;
111         if (TRK_COLLECTION_NAME == null) {
112             trackCollections = event.get(Track.class);
113         } else {
114             trackCollections = new ArrayList<List<Track>>();
115             trackCollections.add(event.get(Track.class, TRK_COLLECTION_NAME));
116         }
117 
118         // Get the collection of LCRelations relating RotatedHelicalTrackHits to
119         // HelicalTrackHits
120 
121         RelationalTable hitToStrips = TruthGBLRefitterDriver.getHitToStripsTable(event,helicalTrackHitRelationsCollectionName);
122         RelationalTable hitToRotated = TruthGBLRefitterDriver.getHitToRotatedTable(event,rotatedHelicalTrackHitRelationsCollectionName);
123 
124         //        List<HelicalTrackHit> rotatedHths = event.get(HelicalTrackHit.class, ROTATED_HTH_COL_NAME);
125 
126         // Create a container that will be used to store all TrackData objects.
127         List<TrackData> trackDataCollection = new ArrayList<TrackData>();
128 
129         // Create a container that will be used to store all LCRelations between
130         // a TrackData object and the corresponding Track
131         List<LCRelation> trackDataRelations = new ArrayList<LCRelation>();
132 
133         // Create a collection to hold the track residuals
134         List<TrackResidualsData> trackResidualsCollection = new ArrayList<TrackResidualsData>();
135 
136         // Create a collection of LCRelations between a track and the track
137         // residuals
138         List<LCRelation> trackToTrackResidualsRelations = new ArrayList<LCRelation>();
139 
140         double xResidual;
141         double yResidual;
142 
143         float totalT0;
144         float totalHits;
145         float trackTime;
146 
147         int trackerVolume = -1;
148 
149         boolean isFirstHit;
150 
151         Hep3Vector stereoHitPosition;
152         Hep3Vector trackPosition;
153         HelicalTrackHit helicalTrackHit;
154 
155         List<Double> t0Residuals = new ArrayList<Double>();
156         List<Double> trackResidualsX = new ArrayList<Double>();
157         List<Float> trackResidualsY = new ArrayList<Float>();
158         List<Integer> stereoLayers = new ArrayList<Integer>();
159 
160         // Loop over each of the track collections retrieved from the event
161         for (List<Track> tracks : trackCollections) {
162 
163             // Loop over all the tracks in the event
164             for (Track track : tracks) {
165                 totalT0 = 0;
166                 totalHits = 0;
167                 t0Residuals.clear();
168                 trackResidualsX.clear();
169                 trackResidualsY.clear();
170                 stereoLayers.clear();
171 
172                 // Change the position of a HelicalTrackHit to be the corrected one.
173                 // Loop over all stereo hits comprising a track
174                 for (TrackerHit rotatedStereoHit : track.getTrackerHits()) {
175                     HpsSiSensor sensor = null;
176                     isFirstHit = true;
177                     // Loop over the clusters comprising the stereo hit
178                     for (HelicalTrackStrip cluster : ((HelicalTrackCross) rotatedStereoHit).getStrips()) {
179 
180                         totalT0 += cluster.time();
181                         totalHits++;
182 
183                         if (isFirstHit) {
184                             sensor = (HpsSiSensor) ((RawTrackerHit) cluster.rawhits().get(0)).getDetectorElement();
185                             if (sensor.isTopLayer()) {
186                                 trackerVolume = 0;
187                             } else if (sensor.isBottomLayer()) {
188                                 trackerVolume = 1;
189                             }
190                             isFirstHit = false;
191                         }
192                     }
193                     stereoHitPosition = ((HelicalTrackHit) rotatedStereoHit).getCorrectedPosition();
194 
195                     trackPosition = TrackUtils.extrapolateTrackPositionToSensor(track, sensor, sensors, bField);
196                     if (trackPosition != null) {
197                         // Add the stereo layer number associated with the track residual
198                         stereoLayers.add(((HelicalTrackHit) rotatedStereoHit).Layer());
199                         // Extrapolate the track to the stereo hit position and calculate track residuals
200                         Hep3Vector stereoHitPositionDetector = CoordinateTransformations.transformVectorToDetector(stereoHitPosition);
201                         xResidual = trackPosition.x() - stereoHitPositionDetector.x();
202                         yResidual = trackPosition.y() - stereoHitPositionDetector.y();
203                         trackResidualsX.add(xResidual);
204                         trackResidualsY.add((float) yResidual);
205                     }
206 
207                     // Change the persisted position of both 
208                     // RotatedHelicalTrackHits and HelicalTrackHits to the
209                     // corrected position.
210                     helicalTrackHit = (HelicalTrackHit) hitToRotated.from(rotatedStereoHit);
211                     ((HelicalTrackHit) rotatedStereoHit).setPosition(stereoHitPosition.v());
212                     stereoHitPosition = CoordinateTransformations.transformVectorToDetector(stereoHitPosition);
213                     helicalTrackHit.setPosition(stereoHitPosition.v());
214 
215                 }
216 
217                 //
218                 // Add a track state that contains the extrapolated track position and 
219                 // parameters at the face of the Ecal.
220                 //
221                 LOGGER.fine("Extrapolating track with type " + Integer.toString(track.getType()));
222 
223                 // Extrapolate the track to the face of the Ecal and get the TrackState
224                 if (TrackType.isGBL(track.getType())) {
225                     TrackState stateEcal = TrackUtils.getTrackExtrapAtEcalRK(track, bFieldMap,runNumber);
226                     if (stateEcal != null)
227                         track.getTrackStates().add(stateEcal);
228                 }
229 
230                 LOGGER.fine(Integer.toString(track.getTrackStates().size()) + " track states for this track at this point:");
231                 for (TrackState state : track.getTrackStates()) {
232                     String s = "type " + Integer.toString(track.getType()) + " location " + Integer.toString(state.getLocation()) + " refPoint (" + state.getReferencePoint()[0] + " " + state.getReferencePoint()[1] + " " + state.getReferencePoint()[2] + ") " + " params: ";
233                     for (int i = 0; i < 5; ++i)
234                         s += String.format(" %f", state.getParameter(i));
235                     LOGGER.fine(s);
236                 }
237 
238                 // The track time is the mean t0 of hits on a track
239                 trackTime = totalT0 / totalHits;
240 
241                 // Calculate the track isolation constants for each of the 
242                 // layers
243                 Double[] isolations = TrackUtils.getIsolations(track, hitToStrips, hitToRotated, 6);
244                 double qualityArray[] = new double[isolations.length];
245                 for (int i = 0; i < isolations.length; i++) {
246                     qualityArray[i] = isolations[i] == null ? -99999999.0 : isolations[i];
247                 }
248 
249                 // Create a new TrackData object and add it to the event
250                 TrackData trackData = new TrackData(trackerVolume, trackTime, qualityArray);
251                 trackDataCollection.add(trackData);
252                 trackDataRelations.add(new BaseLCRelation(trackData, track));
253 
254                 // Create a new TrackResidualsData object and add it to the event
255                 TrackResidualsData trackResiduals = new TrackResidualsData((int) trackerVolume, stereoLayers, trackResidualsX, trackResidualsY);
256                 trackResidualsCollection.add(trackResiduals);
257                 trackToTrackResidualsRelations.add(new BaseLCRelation(trackResiduals, track));
258 
259             }
260         }
261 
262         // Add all collections to the event
263         event.put(TrackData.TRACK_DATA_COLLECTION, trackDataCollection, TrackData.class, 0);
264         event.put(TrackData.TRACK_DATA_RELATION_COLLECTION, trackDataRelations, LCRelation.class, 0);
265         event.put(TRK_RESIDUALS_COL_NAME, trackResidualsCollection, TrackResidualsData.class, 0);
266         event.put(TRK_RESIDUALS_REL_COL_NAME, trackToTrackResidualsRelations, LCRelation.class, 0);
267     }
