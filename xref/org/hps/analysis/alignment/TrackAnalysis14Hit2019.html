18  public class TrackAnalysis14Hit2019 extends Driver {
19  
20      private AIDA aida = AIDA.defaultInstance();
21  
22      public void process(EventHeader event) {
23          if (event.hasCollection(ReconstructedParticle.class, "FinalStateParticles_KF")) {
24              List<ReconstructedParticle> rps = event.get(ReconstructedParticle.class, "FinalStateParticles_KF");
25              for (ReconstructedParticle rp : rps) {
26                  int pdgId = rp.getParticleIDUsed().getPDG();
27                  // only consider charge particles
28                  if (abs(pdgId) == 11) {
29                      // only consider tracks with all sensors hit
30                      if (rp.getTracks().get(0).getTrackerHits().size() == 14) {
31                          //only consider particles with associated cluster
32                          if (!rp.getClusters().isEmpty()) {
33                              String type = pdgId == 11 ? " electron" : " positron";
34                              Cluster c = rp.getClusters().get(0);
35                              Track t = rp.getTracks().get(0);
36                              double p = rp.getMomentum().magnitude();
37                              double e = rp.getEnergy();
38                              if (e > 0.5) {
39                                  aida.tree().mkdirs(type + " Analysis");
40                                  aida.tree().cd(type + " Analysis");
41                                  CalorimeterHit seed = c.getCalorimeterHits().get(0);
42                                  int ix = seed.getIdentifierFieldValue("ix");
43                                  int iy = seed.getIdentifierFieldValue("iy");
44                                  String fid = TriggerModule.inFiducialRegion(c) ? " fiducial " : " non fid ";
45                                  String topOrBottom = iy > 0 ? " top " : " bottom ";
46                                  aida.histogram2D("cluster ix vs iy" + type, 47, -23.5, 23.5, 11, -5.5, 5.5).fill(ix, iy);
47                                  aida.histogram1D("EoverP ix: " + ix + " iy: " + iy + type, 100, 0., 2.0).fill(e / p);
48                                  aida.histogram1D("Track momentum" + topOrBottom + type, 200, 0., 7.0).fill(p);
49                                  aida.histogram2D("E vs p" + topOrBottom + type, 100, 0., 7., 100, 0., 7.).fill(e, p);
50                                  aida.histogram2D("E vs p" + topOrBottom + fid + type, 100, 0., 7., 100, 0., 7.).fill(e, p);
51                                  aida.histogram1D("EoverP" + topOrBottom + fid + type, 100, 0., 2.0).fill(e / p);
52                                  aida.tree().cd("..");
53                              }
54                          }
55                      }
56                  }
57              }
58          }
59      }
60  }
