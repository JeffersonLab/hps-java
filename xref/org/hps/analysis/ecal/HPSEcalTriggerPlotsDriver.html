4   import static org.hps.recon.ecal.EcalUtils.nBit;
5   import hep.aida.IHistogram2D;

7   import java.util.ArrayList;
8   import java.util.List;
9   
10  import org.hps.readout.ecal.TriggerDriver;
11  import org.hps.recon.ecal.EcalUtils;
12  import org.lcsim.event.CalorimeterHit;
13  import org.lcsim.event.Cluster;
14  import org.lcsim.event.EventHeader;
15  import org.lcsim.util.Driver;
16  import org.lcsim.util.aida.AIDA;

18  /**
19   * Diagnostic plots for HPS ECal.
20   */
21  public class HPSEcalTriggerPlotsDriver extends Driver {
22      // LCSim collection names.
23      String ecalCollectionName = "EcalHits";
24      String clusterCollectionName = "EcalClusters";
25      
26      // Histogram factory.
27      AIDA aida = AIDA.defaultInstance();
28      
29      // Energy cuts for hit histograms (in MeV).
30      double energyCut[] = { 0, 1, 2, 5, 10, 20, 50, 100, 200, 500, 1000 };
31      
32      // Energy cut histograms.
33      IHistogram2D hitXYPlot[] = new IHistogram2D[energyCut.length];
34      
35      // Additional histograms.
36      IHistogram2D crystalDeadTime;
37      IHistogram2D clusterHitXYPlot;
38      IHistogram2D seedHitXYPlot;
39      IHistogram2D triggerClusterHitXYPlot;
40      IHistogram2D triggerSeedHitXYPlot;
41      
42      private double tp = 6.95;
43      private double threshold = 10; //ADC counts
44  
45      public void setEcalCollectionName(String ecalCollectionName) {
46          this.ecalCollectionName = ecalCollectionName;
47      }
48  
49      public void setClusterCollectionName(String clusterCollectionName) {
50          this.clusterCollectionName = clusterCollectionName;
51      }
52      
53      public void startOfData() {
54          // Initialize a hit histogram for each declared energy.
55          for(int e = 0; e < energyCut.length; e++) {
56              hitXYPlot[e] = aida.histogram2D("Trigger Plots: " + ecalCollectionName +
57                      " : Hits above " + energyCut[e] + " MeV", 46, -23, 23, 11, -5.5, 5.5);
58          }
59          // Initialize the remaining plots.
60          crystalDeadTime = aida.histogram2D("Trigger Plots: " + ecalCollectionName +
61                  " : Crystal dead time", 46, -23, 23, 11, -5.5, 5.5);
62          clusterHitXYPlot = aida.histogram2D("Trigger Plots: " + clusterCollectionName +
63                  " : Crystals in clusters", 47, -23.5, 23.5, 11, -5.5, 5.5);
64          seedHitXYPlot = aida.histogram2D("Trigger Plots: " + clusterCollectionName +
65                  " : Seed hits", 47, -23.5, 23.5, 11, -5.5, 5.5);
66          triggerClusterHitXYPlot = aida.histogram2D("Trigger Plots: " + clusterCollectionName +
67                  " : Crystals in clusters, with trigger", 47, -23.5, 23.5, 11, -5.5, 5.5);
68          triggerSeedHitXYPlot = aida.histogram2D("Trigger Plots: " + clusterCollectionName +
69                  " : Seed hits, with trigger", 47, -23.5, 23.5, 11, -5.5, 5.5);
70      }
71      
72      public void process(EventHeader event) {
73          // If the current event has the indicated hit collection,
74          // use it as the hit list.
75          List<CalorimeterHit> hits;
76          if(event.hasCollection(CalorimeterHit.class, ecalCollectionName)) {
77              hits = event.get(CalorimeterHit.class, ecalCollectionName);
78          }
79          // If it does not, then use an empty list to avoid crashing.
80          else { hits = new ArrayList<CalorimeterHit>(0); }
81          
82          // If the current event has the indicated cluster collection,
83          // use it as the cluster list.
84          List<Cluster> clusters;
85          if(event.hasCollection(Cluster.class, clusterCollectionName)) {
86              clusters = event.get(Cluster.class, clusterCollectionName);
87          }
88          // If it does not, then use an empty list to avoid crashing.
89          else { clusters = new ArrayList<Cluster>(0); }
90          
91          // Populate hit plots.
92          for (CalorimeterHit hit : hits) {
93              // Get the hit crystal position.
94              int ix = hit.getIdentifierFieldValue("ix");
95              int iy = hit.getIdentifierFieldValue("iy");
96              double energy = hit.getRawEnergy();
97              
98              // Loop through the energy plots and fill them if the hit
99              // is over the current energy threshold/
100             for(int e = 0; e < energyCut.length; e++) {
101                 if(energy > energyCut[e] * EcalUtils.MeV) {
102                     hitXYPlot[e].fill(ix - 0.5 * Math.signum(ix), iy);
103                 }
104             }
105             
106             // Generate the dead time plot.
107             double deadTime = 0;
108             for (int time = 0; time < 500; time++) {
109                 if (hit.getRawEnergy() * pulseAmplitude(time) > threshold) { deadTime += 1e-6; } // units of milliseconds
110                 else if (time > 2 * tp || deadTime != 0) { break; }
111             }
112             crystalDeadTime.fill(ix - 0.5 * Math.signum(ix), iy, deadTime);
113         }
114         
115         // Check the trigger bit.
116         boolean trigger = TriggerDriver.triggerBit();
117         
118         // Populate cluster based plots.
119         for (Cluster cluster : clusters) {
120             // Get the cluster's seed hit position.
121             CalorimeterHit seed = cluster.getCalorimeterHits().get(0);
122             int ix = seed.getIdentifierFieldValue("ix");
123             int iy = seed.getIdentifierFieldValue("iy");
124             
125             // Fill the seed hit plot.
126             seedHitXYPlot.fill(ix, iy);
127             
128             // If the trigger bit is set, add the seed to the trigger
129             // plot histogram as well.
130             if(trigger) { triggerSeedHitXYPlot.fill(ix, iy); }
131             
132             // Populate the component hit histogram.
133             for (CalorimeterHit hit : cluster.getCalorimeterHits()) {
134                 // Get the component hit location.
135                 ix = hit.getIdentifierFieldValue("ix");
136                 iy = hit.getIdentifierFieldValue("iy");
137                 
138                 // Add it to the plot.
139                 clusterHitXYPlot.fill(ix, iy);
140                 
141                 // If the trigger bit is set, add the component to
142                 // the trigger cluster histogram too.
143                 if(trigger) { triggerClusterHitXYPlot.fill(ix, iy); }
144             }
145         }
146     }
147     
148     private double pulseAmplitude(double time) {
149         if (time <= 0.0) { return 0.0; }
150         return EcalUtils.readoutGain * ((time * time / (2 * tp * tp * tp)) * Math.exp(-time / tp)) * ((Math.pow(2, nBit) - 1) / maxVolt);
151     }
152 }
