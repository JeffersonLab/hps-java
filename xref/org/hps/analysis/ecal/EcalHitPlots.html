67          hitCountPlot = aida.histogram1D(detector.getDetectorName() + " : " + inputCollection + " : Hit Count In Event", 10, -0.5, 9.5);

69  
70          // Create the plotter regions.
71          plotter.createRegions(1, 2);
72          plotter.region(0).plot(hitCountPlot);
73          plotter.region(1).plot(hitTimePlot);
74  
75          // Setup the plotter.
76          plotter2 = plotterFactory.create("Hit Energies");
77          plotter2.setTitle("Hit Energies");
78          //plotterFrame.addPlotter(plotter2);
79          plotter2.style().dataStyle().errorBarStyle().setVisible(false);
80  
81          hitEnergyPlot = aida.histogram1D(detector.getDetectorName() + " : " + inputCollection + " : Hit Energy", 1000, -0.1, maxE);
82          hitMaxEnergyPlot = aida.histogram1D(detector.getDetectorName() + " : " + inputCollection + " : Maximum Hit Energy In Event", 1000, -0.1, maxE);
83  
84          if (logScale) {
85              plotter2.style().yAxisStyle().setParameter("scale", "log");
86          }

149     @Override
150     public void process(EventHeader event) {
151         int orTrigTime = -1;
152         int topTrigTime = -1;
153         int botTrigTime = -1;
154 
155         if (event.hasCollection(GenericObject.class, "TriggerBank")) {
156             List<GenericObject> triggerList = event.get(GenericObject.class, "TriggerBank");
157             for (GenericObject data : triggerList) {
158                 if (AbstractIntData.getTag(data) == BaseTriggerData.BANK_TAG) {
159                     BaseTriggerData triggerData = new BaseTriggerData(data);
160                     
161                     int orTrig = triggerData.getOrTrig();
162                     if (orTrig != 0) {
163                         for (int i = 0; i < 32; i++) {
164                             if ((1 << (31 - i) & orTrig) != 0) {
165                                 orTrigTime = i;
166                                 orTrigTimePlot.fill(i);
167                                 break;
168                             }
169                         }
170                     }
171                     int topTrig = triggerData.getTopTrig();
172                     if (topTrig != 0) {
173                         for (int i = 0; i < 32; i++) {
174                             if ((1 << (31 - i) & topTrig) != 0) {
175                                 topTrigTime = i;
176                                 topTrigTimePlot.fill(i);
177                                 break;
178                             }
179                         }
180                     }
181                     int botTrig = triggerData.getBotTrig();
182                     if (botTrig != 0) {
183                         for (int i = 0; i < 32; i++) {
184                             if ((1 << (31 - i) & botTrig) != 0) {
185                                 botTrigTime = i;
186                                 botTrigTimePlot.fill(i);
187                                 break;
188                             }
189                         }
190                     }
191                     break;
192                 } else if (AbstractIntData.getTag(data) == SSPData.BANK_TAG) {
193                     //SSPData triggerData = new SSPData(data);
194                     // TODO: TOP, BOTTOM, AND, and OR trigger are test
195                     // run-specific parameters and are not supported by
196                     // SSPData.
197                     int orTrig = 0; //triggerData.getOrTrig();
198                     if(orTrig != 0) {
199                         for (int i = 0; i < 32; i++) {
200                             if ((1 << (31 - i) & orTrig) != 0) {
201                                 orTrigTime = i;
202                                 orTrigTimePlot.fill(i);
203                                 break;
204                             }
205                         }
206                     }
207                     int topTrig = 1; //triggerData.getTopTrig();
208                     if (topTrig != 0) {
209                         for (int i = 0; i < 32; i++) {
210                             if ((1 << (31 - i) & topTrig) != 0) {
211                                 topTrigTime = i;
212                                 topTrigTimePlot.fill(i);
213                                 break;
214                             }
215                         }
216                     }
217                     int botTrig = 0; //triggerData.getBotTrig();
218                     if (botTrig != 0) {
219                         for (int i = 0; i < 32; i++) {
220                             if ((1 << (31 - i) & botTrig) != 0) {
221                                 botTrigTime = i;
222                                 botTrigTimePlot.fill(i);
223                                 break;
224                             }
225                         }
226                     }
227                     break;
228                 }
229             }
230         }
231 
232         if (event.hasCollection(CalorimeterHit.class, inputCollection)) {
233             List<CalorimeterHit> hits = event.get(CalorimeterHit.class, inputCollection);
234             hitCountPlot.fill(hits.size());
235             double maxEnergy = 0;
236             double topTime = Double.POSITIVE_INFINITY;
237             double botTime = Double.POSITIVE_INFINITY;
238             double orTime = Double.POSITIVE_INFINITY;
239             for (CalorimeterHit hit : hits) {
240                 /*
241                 if (hit.getIdentifierFieldValue("iy") > 0) {
242                     topX.fill(hit.getIdentifierFieldValue("ix"),hit.getPosition()[0]);
243                     topY.fill(hit.getIdentifierFieldValue("iy"),hit.getPosition()[1]);
244                 } else {
245                     botX.fill(hit.getIdentifierFieldValue("ix"),hit.getPosition()[0]);
246                     botY.fill(hit.getIdentifierFieldValue("iy"),hit.getPosition()[1]);                    
247                 }
248                 */
249                 hitEnergyPlot.fill(hit.getRawEnergy());
250                 hitTimePlot.fill(hit.getTime());
251                 if (hit.getTime() < orTime) {
252                     orTime = hit.getTime();
253                 }
254                 if (hit.getIdentifierFieldValue("iy") > 0 && hit.getTime() < topTime) {
255                     topTime = hit.getTime();
256                 }
257                 if (hit.getIdentifierFieldValue("iy") < 0 && hit.getTime() < botTime) {
258                     botTime = hit.getTime();
259                 }
260                 if (hit.getRawEnergy() > maxEnergy) {
261                     maxEnergy = hit.getRawEnergy();
262                 }
263             }
264             if (orTime != Double.POSITIVE_INFINITY) {
265                 orTimePlot.fill(orTime);
266                 orTimePlot2D.fill(orTime, orTrigTime);
267             }
268             if (topTime != Double.POSITIVE_INFINITY) {
269                 topTimePlot.fill(topTime);
270                 topTimePlot2D.fill(topTime, topTrigTime);
271             }
272             if (botTime != Double.POSITIVE_INFINITY) {
273                 botTimePlot.fill(botTime);
274                 botTimePlot2D.fill(botTime, botTrigTime);
275             }
276             hitMaxEnergyPlot.fill(maxEnergy);
277             for (int i = 0; i < hits.size(); i++) {
278                 CalorimeterHit hit1 = hits.get(i);
279                 int x1 = hit1.getIdentifierFieldValue("ix");
280                 int y1 = hit1.getIdentifierFieldValue("iy");
281                 for (int j = i + 1; j < hits.size(); j++) {
282                     CalorimeterHit hit2 = hits.get(j);
283                     int x2 = hit2.getIdentifierFieldValue("ix");
284                     int y2 = hit2.getIdentifierFieldValue("iy");
285                     if ((Math.abs(x1 - x2) <= 1 || x1 * x2 == -1) && (Math.abs(y1 - y2) <= 1)) {
286                         if (x1 != x2 || y1 != y2) {
287                             edgePlot.fill((x1 + x2) / 2.0, (y1 + y2) / 2.0);
288                         }
289                     }
290                 }
291             }
292         } else {
293             hitCountPlot.fill(0);
294         }
295     }
