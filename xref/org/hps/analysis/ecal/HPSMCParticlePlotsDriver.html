58          public int compare(MCParticle p1, MCParticle p2) {
59              double e1 = p1.getEnergy();
60              double e2 = p2.getEnergy();
61              if (e1 < e2) {
62                  return -1;
63              } else if (e1 == e2) {
64                  return 0;
65              } else {
66                  return 1;
67              }
68          }
69      }

201     @Override
202     public void process(EventHeader event) {
203 
204         // MCParticles
205         List<MCParticle> mcparticles = event.get(MCParticle.class).get(0);
206 
207         // Final State particles.
208         List<MCParticle> fsParticles = makeGenFSParticleList(mcparticles);
209 
210         // System.out.println("fsParticles="+fsParticles.size());
211         fsCountPlot.fill(fsParticles.size());
212 
213         for (int i = 0; i < fsParticles.size(); ++i)
214             fsCountVsEventPlot.fill(event.getEventNumber());
215 
216         double fsGammaEmax = this.getHighestPhotonE(fsParticles);
217         // if (fsGammaEmax>0.5) System.out.println("Emax large " + fsGammaEmax);
218         int[] nelectrons = {0, 0};
219         int[] npositrons = {0, 0};
220         int[] ngammas = {0, 0};
221         int count = 0;
222         double trigThr = 0.2;
223         for (MCParticle fs : fsParticles) {
224             // System.out.println("Index " + count);
225 
226             double fsE = fs.getEnergy();
227             double theta = Math.atan2(Math.sqrt(fs.getPX() * fs.getPX() + fs.getPY() * fs.getPY()), fs.getPZ());
228             double thetay = Math.atan2(fs.getPY(), fs.getPZ());
229             int fsPdg = fs.getPDGID();
230             fsEPlot.fill(fsE);
231             this.fsThetayPlot.fill(Math.abs(thetay));
232             fsCountTypePlot.fill(fsPdg);
233             if (fsE > 0.5)
234                 fsCountTypePlot500.fill(fsPdg);
235             if (ParticleTypeClassifier.isElectron(fsPdg)) {
236                 fsElectronEPlot.fill(fsE);
237                 fsElectronThetaPlot.fill(theta);
238                 fsElectronThetayPlot.fill(Math.abs(thetay));
239                 if (fsE > trigThr)
240                     fsElectronThetayTrigPlot.fill(Math.abs(thetay));
241                 fsElectronThetayEPlot.fill(Math.abs(thetay), fsE);
242                 nelectrons[0]++;
243                 if (fsGammaEmax > 0.5)
244                     nelectrons[1]++;
245             } else if (ParticleTypeClassifier.isPositron(fsPdg)) {
246                 fsPositronEPlot.fill(fsE);
247                 fsPositronThetaPlot.fill(theta);
248                 fsPositronThetayPlot.fill(Math.abs(thetay));
249                 if (fsE > trigThr)
250                     fsPositronThetayTrigPlot.fill(Math.abs(thetay));
251                 fsPositronThetayEPlot.fill(Math.abs(thetay), fsE);
252                 npositrons[0]++;
253                 if (fsGammaEmax > 0.5)
254                     npositrons[1]++;
255             } else if (ParticleTypeClassifier.isPhoton(fsPdg)) {
256                 fsGammaEPlot.fill(fsE);
257                 fsGammaThetaPlot.fill(theta);
258                 fsGammaThetayPlot.fill(Math.abs(thetay));
259                 if (fsE > trigThr)
260                     fsGammaThetayTrigPlot.fill(Math.abs(thetay));
261                 fsGammaThetayEPlot.fill(Math.abs(thetay), fsE);
262                 ngammas[0]++;
263                 if (fsGammaEmax > 0.5) {
264                     ngammas[1]++;
265                     // System.out.println("Counting high E gamma at count "+ count);
266                 }
267             }
268         }
269 
270         fsCountEventTypePlot.fill(getEventTypeId(nelectrons[0], npositrons[0], ngammas[0]));
271         fsCountEventTypePlot2.fill(getEventTypeId(nelectrons[1], npositrons[1], ngammas[1]));
272 
273         // Sort MCParticles on energy.
274         // Collections.sort(fsParticles, new MCParticleEComparator());
275 
276         // Energy of top two FS particles.
277         // double e2 = fsParticles.get(0).getEnergy() + fsParticles.get(1).getEnergy();
278 
279         // Energy of top three FS particles.
280         // double e3 = e2 + fsParticles.get(2).getEnergy();
281 
282         if (!fsParticles.isEmpty()) {
283             // primary particle with most E
284             double primaryE = getPrimary(fsParticles).getEnergy();
285             primaryEPlot.fill(primaryE);
286         }
287 
288         // event electron energy
289         double eventE = getPrimaryElectronE(fsParticles);
290         eventEPlot.fill(eventE);
291     }

293     public int getEventTypeId(int ne, int np, int ng) {
294         // get a unique ID depending on final state particle count
295         if (ne == 0 && np == 0 & ng == 0)
296             return -1;
297         if (ne == 1 && np == 0 & ng == 0)
298             return 1;
299         if (ne == 0 && np == 1 & ng == 0)
300             return 2;
301         if (ne == 0 && np == 0 & ng == 1)
302             return 3;
303         if (ne == 1 && np == 1 & ng == 0)
304             return 4;
305         if (ne == 1 && np == 0 & ng == 1)
306             return 5;
307         if (ne == 0 && np == 1 & ng == 1)
308             return 6;
309         if (ne == 1 && np == 1 & ng == 1)
310             return 7;
311         if (ne == 2 && np == 1)
312             return 8;
313         if (ne == 2 && np == 1 & ng == 0)
314             return 9;
315         if (ne == 2 && np == 1 & ng == 1)
316             return 10;
317         if (ne > 2)
318             return 11;
319         if (np > 1)
320             return 12;
321         if (ng > 1)
322             return 13;
323 
324         return 0;
325     }

327     public double getHighestPhotonE(List<MCParticle> particles) {
328         double Emax = -1;
329         double E = 0;
330         int count = 0;
331         for (MCParticle particle : particles) {
332             if (ParticleTypeClassifier.isPhoton(particle.getPDGID())) {
333                 E = particle.getEnergy();
334                 if (E > Emax) {
335                     Emax = E;
336                     // System.out.println("Emax from photon with index " + count);
337                 }
338                 count++;
339             }
340         }
341         return Emax;
342     }

344     private double getPrimaryElectronE(List<MCParticle> particles) {
345         double totalE = 0;
346         for (MCParticle particle : particles) {
347             if (Math.abs(particle.getPDGID()) == 11) {
348                 totalE += particle.getEnergy();
349             }
350         }
351         return totalE;
352     }
353 
354     private MCParticle getPrimary(List<MCParticle> particles) {
355         double maxE = 0;
356         MCParticle primary = null;
357         for (MCParticle particle : particles) {
358             if (particle.getEnergy() > maxE) {
359                 maxE = particle.getEnergy();
360                 primary = particle;
361             }
362         }
363         return primary;
364     }

366     public static List<MCParticle> makeGenFSParticleList(List<MCParticle> mcparticles) {
367         List<MCParticle> fsParticles = new ArrayList<MCParticle>();
368         for (MCParticle mcparticle : mcparticles) {
369             if (mcparticle.getGeneratorStatus() == MCParticle.FINAL_STATE) {
370                 double theta = Math.atan2(
371                         Math.sqrt(mcparticle.getPX() * mcparticle.getPX() + mcparticle.getPY() * mcparticle.getPY()),
372                         mcparticle.getPZ());
373                 if (theta > 1e-3) {
374                     fsParticles.add(mcparticle);
375                 }
376             }
377         }
378         return fsParticles;
379     }
