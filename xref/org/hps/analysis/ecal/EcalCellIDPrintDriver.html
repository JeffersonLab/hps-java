1   package org.hps.analysis.ecal;
2   
3   import java.io.IOException;
4   import java.io.PrintWriter;
5   import java.util.List;
6   
7   import org.lcsim.event.EventHeader;
8   import org.lcsim.event.RawCalorimeterHit;
9   import org.lcsim.event.RawTrackerHit;
10  import org.lcsim.geometry.Detector;
11  import org.lcsim.geometry.IDDecoder;
12  import org.lcsim.geometry.Subdetector;
13  import org.lcsim.util.Driver;
14  
15  public class EcalCellIDPrintDriver extends Driver {
16  
17      Subdetector ecal;
18      IDDecoder dec;
19      String ecalName = "Ecal";
20      String ecalCollectionName = "EcalReadoutHits";
21      String outputFileName;
22      PrintWriter outputStream = null;
23  
24      public EcalCellIDPrintDriver() {
25      }
26  
27      public void setEcalCollectionName(String ecalCollectionName) {
28          this.ecalCollectionName = ecalCollectionName;
29      }
30  
31      public void setEcalName(String ecalName) {
32          this.ecalName = ecalName;
33      }
34  
35      public void setOutputFileName(String outputFileName) {
36          this.outputFileName = outputFileName;
37      }
38  
39      public void startOfData() {
40          if (outputFileName != null) {
41              try {
42                  outputStream = new PrintWriter(outputFileName);
43              } catch (IOException ex) {
44                  throw new RuntimeException("Invalid outputFilePath!");
45              }
46          } else {
47              outputStream = new PrintWriter(System.out, true);
48          }
49      }
50  
51      public void detectorChanged(Detector detector) {
52          // Get the Subdetector.
53          ecal = (Subdetector) detector.getSubdetector(ecalName);
54          dec = ecal.getIDDecoder();
55      }
56  
57      public void process(EventHeader event) {
58          // Get the list of ECal hits.
59          if (event.hasCollection(RawCalorimeterHit.class, ecalCollectionName)) {
60              List<RawCalorimeterHit> hits = event.get(RawCalorimeterHit.class, ecalCollectionName);
61              //outputStream.println("Reading RawCalorimeterHit from event " + event.getEventNumber());
62              for (RawCalorimeterHit hit : hits) {
63                  dec.setID(hit.getCellID());
64                  outputStream.printf("x=%d\ty=%d\n", dec.getValue("ix"), dec.getValue("iy"));
65              }
66          }
67          if (event.hasCollection(RawTrackerHit.class, ecalCollectionName)) {
68              List<RawTrackerHit> hits = event.get(RawTrackerHit.class, ecalCollectionName);
69              //outputStream.println("Reading RawCalorimeterHit from event " + event.getEventNumber());
70              for (RawTrackerHit hit : hits) {
71                  dec.setID(hit.getCellID());
72                  outputStream.printf("x=%d\ty=%d\n", dec.getValue("ix"), dec.getValue("iy"));
73              }
74          }
75      }
76  }
