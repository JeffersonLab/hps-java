121     @Override
122     public void process(EventHeader event) {
123         if (event.hasCollection(Cluster.class, inputCollection)) {
124             List<Cluster> clusters = event.get(Cluster.class, inputCollection);
125             clusterCountPlot.fill(clusters.size());
126             double maxEnergy = 0;
127             for (Cluster cluster : clusters) {
128                 // if ((botTrig == 0 && cluster.getEnergy() > 130 && cluster.getPosition()[1] < 0) || (topTrig == 0 &&
129                 // cluster.getEnergy() > 130 && cluster.getPosition()[1] > 0)) {
130                 // if ((botTrig == 0 && cluster.getPosition()[1] < 0) || (topTrig == 0 && cluster.getPosition()[1] > 0))
131                 // {
132                 clusterEnergyPlot.fill(cluster.getEnergy());
133                 if (cluster.getEnergy() > maxEnergy) {
134                     maxEnergy = cluster.getEnergy();
135                 }
136                 int size = 0;
137                 double[] times = new double[cluster.getCalorimeterHits().size()];
138                 // System.out.format("cluster:\n");
139                 for (CalorimeterHit hit : cluster.getCalorimeterHits()) {
140                     if (hit.getRawEnergy() != 0) {
141                         times[size] = hit.getTime();
142                         clusterTimes.fill(hit.getTime());
143                         size++;
144                         // System.out.format("x=%d, y=%d, time=%f, energy=%f\n", hit.getIdentifierFieldValue("ix"),
145                         // hit.getIdentifierFieldValue("iy"), hit.getTime(), hit.getRawEnergy());
146                     }
147                 }
148                 clusterSizePlot.fill(size);
149                 clusterTimeSigma.fill(Math.sqrt(StatUtils.variance(times, 0, size)));
150 
151                 List<CalorimeterHit> hits = cluster.getCalorimeterHits();
152                 for (int i = 0; i < hits.size(); i++) {
153                     CalorimeterHit hit1 = hits.get(i);
154                     if (hit1.getRawEnergy() == 0) {
155                         continue;
156                     }
157                     int x1 = hit1.getIdentifierFieldValue("ix");
158                     int y1 = hit1.getIdentifierFieldValue("iy");
159                     for (int j = i + 1; j < hits.size(); j++) {
160                         CalorimeterHit hit2 = hits.get(j);
161                         if (hit2.getRawEnergy() == 0) {
162                             continue;
163                         }
164                         int x2 = hit2.getIdentifierFieldValue("ix");
165                         int y2 = hit2.getIdentifierFieldValue("iy");
166                         if ((Math.abs(x1 - x2) <= 1 || x1 * x2 == -1) && (Math.abs(y1 - y2) <= 1)) {
167                             if (x1 != x2 || y1 != y2) {
168                                 edgePlot.fill((x1 + x2) / 2.0, (y1 + y2) / 2.0);
169                             }
170                         }
171                     }
172                 }
173                 // }
174             }
175             clusterMaxEnergyPlot.fill(maxEnergy);
176         } else {
177             clusterCountPlot.fill(0);
178         }
179     }
