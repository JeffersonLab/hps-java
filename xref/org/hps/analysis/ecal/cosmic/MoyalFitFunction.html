5   public class MoyalFitFunction extends AbstractIFunction {
6   
7       /**
8        * No argument constructor.
9        */
10      public MoyalFitFunction() {
11          this("");
12      }
13      
14      /**
15       * Constructor with function title.
16       * The no arg constructor uses this one.
17       * @param title The title of the function.
18       */
19      public MoyalFitFunction(String title) {
20          super();                
21          this.variableNames = new String[] { "x0" };
22          this.parameterNames = new String[] { "pedestal", "norm", "mpv", "width" };
23          init(title);
24      }
25      
26      @Override
27      public double value(double[] x) {
28          double a = -(x[0] - this.parameter("mpv")) / this.parameter("width");
29          return this.parameter("pedestal") + this.parameter("norm") * (1 / (Math.sqrt(2 * Math.PI) * this.parameter("width"))) * Math.exp(-0.5 * (Math.exp(a) - a));
30      }                  
31  }
