24  /**
25   * This Driver will create a histogram for every crystal in the ECAL and plot its corrected energy
26   * for the event.
27   */
28  public class CrystalEnergyPlotsDriver extends Driver {
29  
30      EcalConditions conditions = null;
31      EcalChannelCollection channels = null;
32  
33      List<List<IPlotter>> plotterLists = new ArrayList<List<IPlotter>>();
34  
35      AIDA aida = AIDA.defaultInstance();
36      IAnalysisFactory analysisFactory = aida.analysisFactory();
37  
38      public void detectorChanged(Detector detector) {
39          
40          conditions = DatabaseConditionsManager.getInstance().getEcalConditions();
41          
42          channels = conditions.getChannelCollection();
43  
44          Set<Integer> crates = new HashSet<Integer>();
45          Set<Integer> slots = new HashSet<Integer>();
46          Set<Integer> channels = new HashSet<Integer>();
47  
48          for (EcalChannel channel : conditions.getChannelCollection()) {
49              crates.add(channel.getCrate());
50              slots.add(channel.getSlot());
51              channels.add(channel.getChannel());
52          }
53                                          
54          for (Integer crate : crates) {
55              IPlotterFactory plotterFactory = analysisFactory.createPlotterFactory("ECAL Energy - Crate " + crate);
56              IPlotterStyle style = plotterFactory.createPlotterStyle();
57              style.dataStyle().lineStyle().setVisible(false);
58              style.legendBoxStyle().setVisible(false);
59              int plottersIndex = crate - 1;
60              plotterLists.add(new ArrayList<IPlotter>());
61              List<IPlotter> plotters = plotterLists.get(plottersIndex);
62              for (Integer slot : slots) {
63                  IPlotter plotter = plotterFactory.create("Slot " + slot);
64                  plotters.add(plotter);
65                  plotter.createRegions(4, 4);
66                  for (Integer channel : channels) {
67                                          
68                      String energyPlotName = "Crystal Energy : " + crate + " : " + slot + " : " + channel;
69                      IHistogram1D energyHistogram = aida.histogram1D(energyPlotName, 500, -50., 200.);
70                      
71                      plotter.region(channel).plot(energyHistogram, style);
72                  }
73                  plotter.show();
74              }
75          }
76      }
77  
78      public void process(EventHeader event) {
79          if (event.hasCollection(CalorimeterHit.class, "EcalCalHits")) {
80              List<CalorimeterHit> hits = event.get(CalorimeterHit.class, "EcalCalHits");
81              for (CalorimeterHit hit : hits) {
82                  EcalChannel channel = channels.findGeometric(hit.getCellID());
83                  if (channel != null) {
84                      aida.histogram1D("Crystal Energy : " + channel.getCrate() + " : " + channel.getSlot() + " : " + channel.getChannel())
85                          .fill(hit.getCorrectedEnergy() * 1000);
86                  }                   
87              }
88          }
89      }
90  }
