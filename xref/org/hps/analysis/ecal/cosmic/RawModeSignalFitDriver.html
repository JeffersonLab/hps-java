30  /**
31   * <p>
32   * This Driver will perform a function fit on ECAL window mode data
33   * to determine the likelihood of a signal being present, e.g. from a cosmic
34   * ray MIP signal.  By default, the mean and sigma are fixed in the fit, and the
35   * pedestal and normalization are allowed to float.  The pedestal can also be 
36   * configured as fixed using the {@link #setFixPedestal(boolean)} method.
37   * <p>
38   * Those hits with a signal significance greater than a settable
39   * threshold (by default set to 4 sigma) will be written into an output collection
40   * of selected hits that can be used by other Drivers.   
41   */
42  public class RawModeSignalFitDriver extends Driver {
43  
44      // ECAL conditions data.
45      EcalConditions conditions = null;
46      EcalChannelCollection channels = null;
47          
48      // AIDA setup.    
49      AIDA aida = AIDA.defaultInstance();
50      IAnalysisFactory analysisFactory = aida.analysisFactory();
51      IFunctionFactory functionFactory = analysisFactory.createFunctionFactory(null);
52      IFitFactory fitFactory = analysisFactory.createFitFactory();
53      IFitter fitter = fitFactory.createFitter();
54      
55      // DPS used to fit a hit's ADC samples.
56      IDataPointSet adcDataPointSet;
57      
58      // Per channel histograms filled when doing the fit.
59      Map<EcalChannel, IHistogram1D> signalNormHistograms = new HashMap<EcalChannel, IHistogram1D>();
60      Map<EcalChannel, IHistogram1D> pedestalNormHistograms = new HashMap<EcalChannel, IHistogram1D>();
61      Map<EcalChannel, IHistogram1D> signalSignificanceHistograms = new HashMap<EcalChannel, IHistogram1D>();
62      
63      // The function that will be used for the signal fit.
64      IFunction fitFunction;
65      
66      // The output hits collection with the selected hits.
67      String outputHitsCollectionName = "EcalCosmicReadoutHits";
68      
69      // The input hits collection with all the raw data hits.
70      String inputHitsCollectionName = "EcalReadoutHits";
71          
72      HPSEcal3 ecal = null;
73      static String ecalName = "Ecal";
74      
75      // The minimum number of required hits for event processing to continue.  
76      int minimumHits = 3;
77      
78      // This determines whether the pedestal is fixed in the fit parameters.
79      boolean fixPedestal = false;
80      
81      boolean printFits = false;
82      
83      // This is the required significance for signal hits (4 sigma default).
84      static double signalSignificanceThreshold = 4.0;
85      
86      // Global fit parameters for Moyal distribution.
87      static double signalMean = 45.698; 
88      static double signalSigma = 2.2216;
89      
90      // The initial value of the function normalization, which is not fixed in the fit.
91      static double norm = 60.0;
92

130     /**
131      * Initialize conditions dependent class variables.
132      * @param detector The current Detector object.
133      */
134     public void detectorChanged(Detector detector) {
135         ecal = (HPSEcal3)detector.getSubdetector(ecalName);
136         conditions = DatabaseConditionsManager.getInstance().getEcalConditions();
137         channels = conditions.getChannelCollection();                
138         for (EcalChannel channel : conditions.getChannelCollection()) {            
139             signalNormHistograms.put(channel, aida.histogram1D(inputHitsCollectionName + "/Signal Norm : Channel " + String.format("%03d", channel.getChannelId()), 500, 0, 500.));
140             pedestalNormHistograms.put(channel, aida.histogram1D(inputHitsCollectionName + "/Pedestal Norm : Channel " + String.format("%03d", channel.getChannelId()), 500, 0, 500.));
141             signalSignificanceHistograms.put(channel, aida.histogram1D(inputHitsCollectionName + "/Signal Significance : Channel " + String.format("%03d", channel.getChannelId()), 200, -5., 35.));
142         }        
143     }

145     /**
146      * Perform start of job initialize.
147      * The DataPointSet for the ADC values is initialized and global fit parameters are set here.
148      */
149     public void startOfData() {
150         adcDataPointSet = aida.analysisFactory().createDataPointSetFactory(null).create("ADC DataPointSet", 2);
151         
152         fitFunction = new MoyalFitFunction();        
153         fitFunction.setParameter("mpv", signalMean);
154         fitFunction.setParameter("width", signalSigma);
155         fitFunction.setParameter("norm", norm);                
156         
157         fitter.fitParameterSettings("mpv").setFixed(true);
158         fitter.fitParameterSettings("width").setFixed(true);
159         if (fixPedestal) {
160             fitter.fitParameterSettings("pedestal").setFixed(true);
161         }
162     }
163 
164     /**
165      * Process the event, performing a signal fit for every raw data hit in the input collection.
166      * The hits that pass the sigma selection cut are added to a new hits collection, which can be
167      * converted to a CalorimeterHit collection and then clustered.
168      * @throws NextEventException if there are not enough hits that pass the selection cut.
169      */
170     public void process(EventHeader event) {
171         if (event.hasCollection(RawTrackerHit.class, inputHitsCollectionName)) {
172             List<RawTrackerHit> hits = event.get(RawTrackerHit.class, inputHitsCollectionName);
173             List<RawTrackerHit> selectedHitsList = new ArrayList<RawTrackerHit>();
174             for (RawTrackerHit hit : hits) {
175                 EcalChannel channel = channels.findGeometric(hit.getCellID());
176                 if (channel != null) {
177                                         
178                     EcalChannelConstants channelConstants = conditions.getChannelConstants(channel);
179                     double noise = channelConstants.getCalibration().getNoise();                    
180                     
181                     // Clear the DPS from previous fit.
182                     adcDataPointSet.clear();
183                     
184                     // Loop over all ADC values of the hit.
185                     for (int adcSample = 0; adcSample < hit.getADCValues().length; adcSample++) {
186                         // Insert a DP into the DPS for each sample. 
187                         adcDataPointSet.addPoint();
188                         
189                         // Coordinate 1 is the ADC sample number.
190                         adcDataPointSet.point(adcSample).coordinate(0).setValue(adcSample);
191                         
192                         // Coordinate 2 is the ADC sample value and its errors, which is set to the 
193                         // noise from the EcalCalibration condition object for plus and minus.
194                         adcDataPointSet.point(adcSample).coordinate(1).setValue(hit.getADCValues()[adcSample]);
195                         adcDataPointSet.point(adcSample).coordinate(1).setErrorMinus(noise);
196                         adcDataPointSet.point(adcSample).coordinate(1).setErrorPlus(noise);                                                
197                     }
198                     
199                     // Fit the ADC signal.
200                     IFitResult fitResult = fitAdcSamples(channel, adcDataPointSet);
201                                      
202                     if (printFits) {
203                         ((FitResult)fitResult).printResult();
204                     }
205                      
206                     // Calculate the signal significance which is norm over error.
207                     double signalSignificance = fitResult.fittedParameter("norm") / fitResult.errors()[1];
208                     
209                     // Fill signal significance histogram.
210                     this.signalSignificanceHistograms.get(channel).fill(signalSignificance);                    
211                                         
212                     // Is the significance over the threshold?
213                     if (signalSignificance >= signalSignificanceThreshold) {
214                         //System.out.println(fitResult.fittedParameter("norm") + " " + fitResult.errors()[1] + " " + signalSignificance);
215                         // Add the hit to the output list.
216                         selectedHitsList.add(hit);
217                     }                   
218                 } else {                    
219                     throw new RuntimeException("EcalChannel not found for cell ID 0x" + String.format("%08d", Long.toHexString(hit.getCellID())));
220                 }
221             }            
222             
223             // Is the hit list greater than the minimum hits?
224             if (selectedHitsList.size() >= minimumHits) {
225                 // Write the hits to a new collection of selected hits.
226                 event.put(outputHitsCollectionName, selectedHitsList, RawTrackerHit.class, event.getMetaData(hits).getFlags(), ecal.getReadout().getName());
227             } else {
228                 // Discontinue processing this event because there aren't enough hits for clustering.
229                 throw new NextEventException();
230             }
231         }
232     }
233     
234     /**
235      * Fit all of the ADC samples in a hit using a DataPointSet and then return the fit result.
236      * @param channel The ECAL channel conditions information.
237      * @param adcDataPointSet The DataPointSet to use for the fit containing all 100 ADC samples.
238      * @return The significance which is the normalization divided by its error.
239      */
240     IFitResult fitAdcSamples(EcalChannel channel, IDataPointSet adcDataPointSet) {
241         EcalChannelConstants channelConstants = conditions.getChannelConstants(channel);
242         fitFunction.setParameter("pedestal", channelConstants.getCalibration().getPedestal());
243         IFitResult fitResult = fitter.fit(adcDataPointSet, fitFunction);
244         this.signalNormHistograms.get(channel).fill(fitResult.fittedParameter("norm"));
245         this.pedestalNormHistograms.get(channel).fill(fitResult.fittedParameter("pedestal"));
246         return fitResult;
247     }
