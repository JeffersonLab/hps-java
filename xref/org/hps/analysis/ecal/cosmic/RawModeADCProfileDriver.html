20  /**
21   * This Driver makes profile plots of the raw mode ADC data.
22   */
23  public class RawModeADCProfileDriver extends Driver {
24  
25      EcalConditions conditions = null;
26      EcalChannelCollection channels = null;
27      Map<EcalChannel, IProfile1D> adcProfiles = new HashMap<EcalChannel, IProfile1D>();       
28      AIDA aida = AIDA.defaultInstance();
29      IAnalysisFactory analysisFactory = aida.analysisFactory();    
30      String inputHitsCollectionName = "EcalReadoutHits";
31             
32      public void setInputHitsCollectionName(String inputHitsCollectionName) {
33          this.inputHitsCollectionName = inputHitsCollectionName;
34      }
35          
36      public void detectorChanged(Detector detector) {
37          conditions = DatabaseConditionsManager.getInstance().getEcalConditions();
38          channels = conditions.getChannelCollection();
39          for (EcalChannel channel : conditions.getChannelCollection()) {            
40              // Create ADC profile histogram, assuming ADC sample values of 0 to 99, with profile range -0.5 to 99.5, so bins are centered.
41              adcProfiles.put(channel, aida.profile1D(inputHitsCollectionName + "/ADC Values : " + String.format("%03d", channel.getChannelId()), 100, -0.5, 99.5));
42          }
43      }
44  
45      public void process(EventHeader event) {                       
46          if (event.hasCollection(RawTrackerHit.class, inputHitsCollectionName)) {
47              List<RawTrackerHit> hits = event.get(RawTrackerHit.class, inputHitsCollectionName);
48              for (RawTrackerHit hit : hits) {
49                  EcalChannel channel = channels.findGeometric(hit.getCellID());                
50                  if (channel != null) {                    
51                      IProfile1D profile = adcProfiles.get(channel);                                        
52                      for (int adcIndex = 0; adcIndex < hit.getADCValues().length; adcIndex++) {
53                          // Fill the Profile1D with ADC value.
54                          profile.fill(adcIndex, hit.getADCValues()[adcIndex]);
55                      }                    
56                  } else {
57                      System.err.println("EcalChannel not found for cell ID 0x" + String.format("%08d", Long.toHexString(hit.getCellID())));
58                  }
59              }
60          }
61      } 
62  }
