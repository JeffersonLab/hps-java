1   package org.hps.analysis.ecal.cosmic;
2   
3   import hep.aida.ref.function.AbstractIFunction;
4   
5   /**
6    * <p>
7    * This class implements a function for fitting a signal plus pedestal
8    * in non-pedestal subtracted raw mode ADC data.
9    * <p>
10   * It has four function parameters:
11   * <ul>
12   * <li><b>mean</b> - the mean of the Landau function
13   * <li><b>sigma</b> - the width of the Landau
14   * <li><b>norm</b> - the normalization parameter
15   * <li><b>pedestal</b> - the pedestal constant
16   * </ul> 
17   * <p>
18   * The class is designed to be used with the AIDA fitting API. 
19   */
20  public class LandauFitFunction extends AbstractIFunction {
21  
22      // This is the backing function used to get the Landau PDF values.
23      LandauPdf landauPdf = new LandauPdf();    
24        
25      /**
26       * No argument constructor.
27       */
28      public LandauFitFunction() {
29          this("");
30      }
31      
32      /**
33       * Constructor with function title.
34       * The no arg constructor uses this one.
35       * @param title The title of the function.
36       */
37      public LandauFitFunction(String title) {
38          super();                
39          this.variableNames = new String[] { "x0" };
40          this.parameterNames = new String[] { "mean", "sigma", "norm", "pedestal" };        
41          init(title);
42      }
43  
44      /**
45       * Get the Y value of the Landau function at X.
46       * @value v The input X value (array of length 1).
47       */
48      @Override
49      public double value(double[] v) {
50          return this.parameter("pedestal") + this.parameter("norm") * landauPdf.getValue(v[0]);
51      }
52  
53      /**
54       * Set a parameter value on the function.
55       * If these are mean or sigma values, they are pushed to the {@link #landauPdf} object.
56       */
57      @Override
58      public void setParameter(String key, double value) throws IllegalArgumentException {
59          super.setParameter(key, value);
60          if (key.equals("mean")) {
61              landauPdf.setMean(value);
62          } else if (key.equals("sigma")) {
63              landauPdf.setSigma(value);
64          }         
65      }
66  
67      /**
68       * Set all parameters at once.
69       * The mean and sigma are pushed to the {@link #landauPdf} object.
70       */
71      @Override
72      public void setParameters(double[] parameters) throws IllegalArgumentException {        
73          super.setParameters(parameters);
74          landauPdf.setMean(parameters[0]);
75          landauPdf.setSigma(parameters[1]);
76      }
77  }
