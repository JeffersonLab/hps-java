18  /**
19   * This code looks at raw FADC spectra and applies geometric cuts to select
20   * vertical cosmic events. Signals are then fitted with a "3-pole" function to
21   * extract the integral. In order to cross-check the results, the integral is
22   * evaluated also summing the FADC samples. The output is a .root file
23   * containing the histogram spectra for each crystal, both for the fit and the
24   * sum, the 2-D correlation between the two and the histograms of pedestal
25   * values for each crystal.
26   */
27  public class CosmicCalibrationFit extends Driver {

34      private EcalCosmicPulseFitter pulseFitter = new EcalCosmicPulseFitter();
35      private EcalConditions ecalConditions = null;

37      // Tune-able parameters:
38      // optimum signal window, 35-55 (in units of 4ns) (35-55 in 2016)
39      private int MINS = 25;
40      private int MAXS = 45;
41      // pedestal window for calculating an average pedestal (10-30 in 2016)
42      private int MINP = 0;
43      private int MAXP = 20;
44      // number of bins used to calculate the pedestal
45      private int NWIN = MAXP - MINP;
46      // threshold in mV (2.5mV in 2015, 3.5mV in 2016)
47      private double THR = 3.0;// for 2016
48      // 0 is strict requires 2 hits vertically, 1 is loose requiring 1 hit vertically
49      // (fit available only with strict geo
50      // cut)
51      private int cutType = 0;
52  
53      // ///////////////////
54      // Don't change these:
55      // ///////////////////
56      // puts the crystals in an array:
57      private int NX = 46;
58      private int NY = 10;
59      // number of 4ns time samples in the measured spectrum
60      private int NSAMP = 100;
61      // conversion from FADC to mV
62      private double ADC2V = 0.25;

90          ecalConditions = DatabaseConditionsManager.getInstance().getEcalConditions();
91          pulseFitter.setDetector(detector);

93          // Instantiate the tree and histogram factory
94          aida.tree().cd("/");
95          // Create histograms

97          String titleID_CorrTot = "CorrTot";

101         // aida.histogram1D( titleID_PedDiffTot,150,0,150);
102

114 
115                     String titleID_Ped = String.format("Cry_%d_%d_Ped", jx, jy);
116                     aida.histogram1D(titleID_Ped, 400, 0, 60);
117 
118                     String titleID_PedF = String.format("Cry_%d_%d_PedF", jx, jy);

120 
121                 } // end !ishole
122             } // end jy iteration
123         } // end jx iteration
124     }

126     // read in channels, store in arrays
127     public void process(EventHeader event) {
128         if (!event.hasCollection(RawTrackerHit.class, "EcalReadoutHits"))
129             throw new Driver.NextEventException();
130         List<RawTrackerHit> hitList = event.get(RawTrackerHit.class, "EcalReadoutHits");
131 
132         // Put all raw hits into a hash map with ix,iy array position
133         Map<Point, RawTrackerHit> hitMap = new LinkedHashMap<Point, RawTrackerHit>();
134 
135         // Loop over the raw hits in the event
136         for (RawTrackerHit hit : hitList) {
137             int ix = hit.getIdentifierFieldValue("ix");
138             int iy = hit.getIdentifierFieldValue("iy");
139             // crystals are stored in array of 0-45,0-9 with numbering starting at bottom
140             // left of Ecal (-23,-5)
141             int xx = ix + 23;
142             if (xx > 23)
143                 xx -= 1;
144             int yy = iy + 5;
145             if (yy > 5)
146                 yy -= 1;
147 
148             Point hitIndex = new Point(xx, yy);
149             hitMap.put(hitIndex, hit);
150         }
151 
152         int pedestal[][] = new int[NX][NY];
153         int pulse[][] = new int[NX][NY];
154         float signal[][] = new float[NX][NY];
155         int time0 = 0;
156 
157         // loop over crystal y
158         for (int iy = 0; iy < NY; iy++) {
159             // loop over crystal x
160             for (int ix = 0; ix < NX; ix++) {
161                 int trigger = 0;
162 
163                 if (!ishole(ix, iy)) {
164 
165                     // loop over time samples, integrate adc values, use for pedestal
166                     pedestal[ix][iy] = 0;
167 
168                     Point cid = new Point(ix, iy);
169                     RawTrackerHit ihit = hitMap.get(cid);
170 
171                     // This may happen if - for any reason - a FADC channel is missing data.
172                     // In principle, cosmics data should be taken with no 0-suppression.
173                     if (ihit == null) {
174                         pulse[ix][iy] = 0;
175                         signal[ix][iy] = 0;
176                         continue;
177                     }
178 
179                     final short samples[] = ihit.getADCValues();
180 
181                     for (int nTime = MINP; nTime < MAXP; nTime++) {
182                         int adc = samples[nTime];
183                         pedestal[ix][iy] += adc;
184                     } // end loop over time samples
185 
186                     // loop over time samples, integrate adc values, use for signal and trigger
187                     pulse[ix][iy] = 0;
188                     for (int nTime = MINS; nTime < MAXS; nTime++) {
189                         int adc = samples[nTime];
190                         double peak = (adc - pedestal[ix][iy] / NWIN) * ADC2V;
191 
192                         if (peak > THR) {
193                             trigger = trigger + 1;
194                             time0 = nTime;
195                             // System.out.println("\t greater than thresh:\t"+peak+"\t"+THR);
196                         }
197                         pulse[ix][iy] += adc;
198                     } // end loop over time samples
199 
200                     // subtract pedestal from pulse and plot
201                     if (trigger >= 1) {
202                         signal[ix][iy] = (float) ((pulse[ix][iy] - pedestal[ix][iy]) * ADC2V);
203 
204                         // Crystal has passed threshold trigger cut, now must pass geometry cuts
205                         int geomCut0 = 0;// 0 passes, ix+1
206                         int geomCut1 = 0;// 0 passes, ix-1
207                         int geomCut2 = 1;// 0 passes, iy+1
208                         int geomCut3 = 1;// 0 passes, iy-1
209                         int geomCut4 = 1;// 0 passes, if iy is 9,4, iy-2
210                         int geomCut5 = 1;// 0 passes, if iy is 0,5, iy+2
211 
212                         // define geometry cuts-no other hit on left and right passing raw thresh
213                         // loop over time samples, integrate adc values
214                         if (!ishole(ix + 1, iy) && (ix + 1) < 46) {
215                             Point cidxp1 = new Point(ix + 1, iy);
216                             RawTrackerHit ihitxp1 = hitMap.get(cidxp1);
217 
218                             // This may happen if - for any reason - a FADC channel is missing data.
219                             // In principle, cosmics data should be taken with no 0-suppression.
220                             if (ihitxp1 == null) {
221                                 pulse[ix + 1][iy] = 0;
222                                 signal[ix + 1][iy] = 0;
223                                 continue;
224                             }
225 
226                             final short samplesxp1[] = ihitxp1.getADCValues();
227                             pedestal[ix + 1][iy] = 0;
228                             // calculate the pedestal for the crystal ix+1
229                             for (int nTime = MINP; nTime < MAXP; nTime++) {
230                                 int ped = samplesxp1[nTime];
231                                 pedestal[ix + 1][iy] += ped;
232                             } // end loop over time samples
233                               // check if hit in adj crystal passes threshold
234                             pulse[ix + 1][iy] = 0;
235                             for (int nTime = MINS; nTime < MAXS; nTime++) {
236                                 int adc = samplesxp1[nTime];
237                                 double peak = (adc - pedestal[ix + 1][iy] / NWIN) * ADC2V;
238                                 if (peak > THR) {
239                                     geomCut0 = 1;
240                                     break;
241                                 }
242                             } // end loop over time samples
243                         }
244 
245                         if (!ishole(ix - 1, iy) && (ix - 1) > -1) {
246                             Point cidxm1 = new Point(ix - 1, iy);
247                             RawTrackerHit ihitxm1 = hitMap.get(cidxm1);
248                             // This may happen if - for any reason - a FADC channel is missing data.
249                             // In principle, cosmics data should be taken with no 0-suppression.
250                             if (ihitxm1 == null) {
251                                 pulse[ix - 1][iy] = 0;
252                                 signal[ix - 1][iy] = 0;
253                                 continue;
254                             }
255                             final short samplesxm1[] = ihitxm1.getADCValues();
256 
257                             pedestal[ix - 1][iy] = 0;
258                             for (int nTime = MINP; nTime < MAXP; nTime++) {
259                                 int ped = samplesxm1[nTime];
260                                 pedestal[ix - 1][iy] += ped;
261                             } // end loop over time samples
262                               // check if hit in adj crystal passes threshold
263                             pulse[ix - 1][iy] = 0;
264                             for (int nTime = MINS; nTime < MAXS; nTime++) {
265                                 int adc = samplesxm1[nTime];
266                                 double peak = (adc - pedestal[ix - 1][iy] / NWIN) * ADC2V;
267                                 if (peak > THR) {
268                                     geomCut1 = 1;
269                                     break;
270                                 }
271                             } // end loop over time samples
272                         }
273 
274                         if (!ishole(ix, iy + 1) && (iy + 1) < 10 && iy != 4) {
275                             geomCut2 = 0;
276                             Point cidyp1 = new Point(ix, iy + 1);
277                             RawTrackerHit ihityp1 = hitMap.get(cidyp1);
278                             // This may happen if - for any reason - a FADC channel is missing data.
279                             // In principle, cosmics data should be taken with no 0-suppression.
280                             if (ihityp1 == null) {
281                                 pulse[ix][iy + 1] = 0;
282                                 signal[ix][iy + 1] = 0;
283                                 continue;
284                             }
285                             final short samplesyp1[] = ihityp1.getADCValues();
286 
287                             pedestal[ix][iy + 1] = 0;
288                             for (int nTime = MINP; nTime < MAXP; nTime++) {
289                                 int ped = samplesyp1[nTime];
290                                 pedestal[ix][iy + 1] += ped;
291                             } // end loop over time samples
292                               // check if hit in adj crystal passes threshold
293                             pulse[ix][iy + 1] = 0;
294                             for (int nTime = MINS; nTime < MAXS; nTime++) {
295                                 int adc = samplesyp1[nTime];
296                                 double peak = (adc - pedestal[ix][iy + 1] / NWIN) * ADC2V;
297                                 if (peak > THR) {
298                                     geomCut2 = geomCut2 + 1;
299                                     // break;
300                                 }
301                             } // end loop over time samples
302                         }
303 
304                         if (!ishole(ix, iy - 1) && (iy - 1) > -1 && iy != 5) {
305                             geomCut3 = 0;
306                             Point cidym1 = new Point(ix, iy - 1);
307                             RawTrackerHit ihitym1 = hitMap.get(cidym1);
308                             if (ihitym1 == null) {
309                                 pulse[ix][iy - 1] = 0;
310                                 signal[ix][iy - 1] = 0;
311                                 continue;
312                             }
313                             final short samplesym1[] = ihitym1.getADCValues();
314 
315                             pedestal[ix][iy - 1] = 0;
316                             for (int nTime = MINP; nTime < MAXP; nTime++) {
317                                 int ped = samplesym1[nTime];
318                                 pedestal[ix][iy - 1] += ped;
319                             } // end loop over time samples
320                               // check if hit in adj crystal passes threshold
321                             pulse[ix][iy - 1] = 0;
322                             for (int nTime = MINS; nTime < MAXS; nTime++) {
323                                 int adc = samplesym1[nTime];
324                                 double peak = (adc - pedestal[ix][iy - 1] / NWIN) * ADC2V;
325                                 if (peak > THR) {
326                                     geomCut3 = geomCut3 + 1;
327                                     // break;
328                                 }
329 
330                             } // end loop over time samples
331                         }
332 
333                         // ///////////////////Add in additional vert geom constraint for edges/////////
334                         // if the crystal is along an edge, it must have a hit two above or two below
335                         // since it does not have 1 above and 1 below
336                         if (iy == 9 || iy == 4 || ishole(ix, iy + 1)) // look at iy-2
337                         {
338                             geomCut4 = 0;
339                             Point cidym2 = new Point(ix, iy - 2);
340                             RawTrackerHit ihitym2 = hitMap.get(cidym2);
341                             if (ihitym2 == null) {
342                                 pulse[ix][iy - 2] = 0;
343                                 signal[ix][iy - 2] = 0;
344                                 // continue;
345                             }
346                             final short samplesym2[] = ihitym2.getADCValues();
347 
348                             pedestal[ix][iy - 2] = 0;
349                             for (int nTime = MINP; nTime < MAXP; nTime++) {
350                                 int ped = samplesym2[nTime];
351                                 pedestal[ix][iy - 2] += ped;
352                             } // end loop over time samples
353                             pulse[ix][iy - 2] = 0;
354                             for (int nTime = MINS; nTime < MAXS; nTime++) {
355                                 int adc = samplesym2[nTime];
356                                 double peak = (adc - pedestal[ix][iy - 2] / NWIN) * ADC2V;
357                                 if (peak > THR) {
358                                     geomCut4 = geomCut4 + 1;
359                                     // break;
360                                 }
361 
362                             } // end loop over time samples
363                         } // end for iy-2
364                         if (iy == 0 || iy == 5 || ishole(ix, iy - 1)) // look at iy+2
365                         {
366                             geomCut5 = 0;
367                             pedestal[ix][iy + 2] = 0;
368                             Point cidyp2 = new Point(ix, iy + 2);
369                             RawTrackerHit ihityp2 = hitMap.get(cidyp2);
370                             if (ihityp2 == null) {
371                                 pulse[ix][iy + 2] = 0;
372                                 signal[ix][iy + 2] = 0;
373                                 // continue;
374                             }
375                             final short samplesyp2[] = ihityp2.getADCValues();
376                             for (int nTime = MINP; nTime < MAXP; nTime++) {
377                                 int ped = samplesyp2[nTime];
378                                 pedestal[ix][iy + 2] += ped;
379                             } // end loop over time samples
380                             pulse[ix][iy + 2] = 0;
381                             for (int nTime = MINS; nTime < MAXS; nTime++) {
382                                 int adc = samplesyp2[nTime];
383                                 double peak = (adc - pedestal[ix][iy + 2] / NWIN) * ADC2V;
384                                 if (peak > THR) {
385                                     geomCut5 = geomCut5 + 1;
386                                     // break;
387                                 }
388                             } // end loop over time samples
389                         } // end for iy+2
390 
391                         // ///////////////////////////////////////////////////////////////////////////
392                         // System.out.println("print of
393                         // geoCuts:\t"+geomCut0+","+geomCut1+","+geomCut2+","+geomCut3+","+geomCut4+","+geomCut5);
394                         // System.out.println("cut type:\t"+cutType);
395 
396                         if (cutType == 0) // strict geometry cut
397                         {
398                             if (geomCut0 == 0 && geomCut1 == 0 && geomCut2 >= 1 && geomCut3 >= 1 && geomCut4 >= 1
399                                     && geomCut5 >= 1) {
400                                 // double rangeMin = MINS-5;
401                                 // double rangeMax = MAXS+5;
402                                 double sumADC = 0;
403                                 double minADC = 0;
404                                 double maxADC = 10000;
405                                 double fitQuality = -1;
406 
407                                 // Fill histograms
408                                 String titleID = String.format("Cry_%d_%d", ix, iy);
409                                 aida.histogram1D(titleID).fill(signal[ix][iy]);
410 
411                                 String titleID_Ped = String.format("Cry_%d_%d_Ped", ix, iy);
412                                 aida.histogram1D(titleID_Ped).fill(pedestal[ix][iy] / NWIN * ADC2V);
413 
414                                 // System.out.println("Fitting:\t"+ix+"\t"+iy);
415                                 // System.out.println(findChannel(ihit.getCellID()).getCalibration().getNoise()
416                                 // );
417 
418                                 // Fit the pulse
419                                 IFitResult fitResult = pulseFitter.fitCosmicPulse(ihit, time0, maxADC);
420 
421                                 // System.out.println("Fitting:\t"+ix+"\t"+iy);
422 
423                                 if (fitResult != null) {
424                                     fitQuality = fitResult.quality();
425 
426                                     if (fitQuality > 0) {
427                                         // pulseTime = fitResult.fittedParameter("time0")*nsPerSample;
428                                         // Getting fit results
429                                         sumADC = fitResult.fittedParameter("integral");
430                                         minADC = fitResult.fittedParameter("pedestal");
431 
432                                         String titleID_CorrTot = "CorrTot";
433                                         aida.histogram2D(titleID_CorrTot).fill(signal[ix][iy], sumADC * ADC2V);
434 
435                                         // String titleID_PedDiffTot = "PedDiffTot";
436                                         // aida.histogram1D(titleID_PedDiffTot).fill(minADC*ADC2V-pedestal[ix][iy]/NWIN*ADC2V);
437 
438                                         String titleID_Fit = String.format("Cry_%d_%d_Fit", ix, iy);
439                                         aida.histogram1D(titleID_Fit).fill(sumADC * ADC2V);
440 
441                                         String titleID_Corr = String.format("Cry_%d_%d_Corr", ix, iy);
442                                         aida.histogram2D(titleID_Corr).fill(signal[ix][iy], sumADC * ADC2V);
443 
444                                         String titleID_PedF = String.format("Cry_%d_%d_PedF", ix, iy);
445                                         aida.histogram1D(titleID_PedF).fill(minADC * ADC2V);
446 
447                                     }
448                                 }
449 
450                             }
451                         } else if (cutType == 1) // loose geometry cut
452                         {
453                             if (geomCut0 == 0 && geomCut1 == 0) {
454                                 if (geomCut2 == 0 || geomCut3 == 0) {
455                                     String titleID = String.format("Cry_%d_%d", ix, iy);
456                                     aida.histogram1D(titleID).fill(signal[ix][iy]);
457                                 }
458                             }
459                         }
460                     } // end of trigger==1
461 
462                 } // end !ishole
463 
464             } // end loop over x
465 
466         } // end loop over y
467 
468     }// end process event

470     boolean ishole(int x, int y) {
471         return (x > 12 && x < 22 && y > 3 && y < 6);
472     }
