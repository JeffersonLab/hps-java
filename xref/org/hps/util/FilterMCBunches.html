34  /**
35   * Selects LCIO events passing a cut; spaces out these events with blank events.
36   * Intended use is to clean up a photon-run MC file before running trigger and
37   * readout sim. Can also be used to chain multiple LCIO files together.
38   */
39  public class FilterMCBunches {
40

43          DefaultLoggingConfig.initialize();

46      /**
47       * Defines command line options for this program.
48       *
49       * @return The command line options.
50       */
51      private static Options createCommandLineOptions() {
52          Options options = new Options();
53          Option opt_e = new Option("e", true, "Interval between non-empty events");
54          opt_e.setRequired(true);
55          options.addOption(opt_e);
56          //options.addOption(new Option("e", true, "Interval between non-empty events"));
57          options.addOption(new Option("n", true, "Number of events to read"));
58          options.addOption(new Option("w", true, "Number of events to write"));
59          options.addOption(new Option("a", false, "All events - no cuts"));
60          options.addOption(new Option("d", false, "Filter requiring enough ECal hit energy in both top and bottom"));
61          options.addOption(new Option("t", false, "Filter requiring enough SimTrackerHits to make tracks in top and bottom"));
62          options.addOption(new Option("r", false, "Filter requiring enough RawTrackerHits to make tracks in top and bottom"));
63          options.addOption(new Option("p", false, "Filter requiring at least 2 particles with enough hits to make tracks"));
64          options.addOption(new Option("E", true, "Energy cut for EcalHit cut"));
65          options.addOption(new Option("L", true, "Layer count required for tracker cuts"));
66          return options;
67      }

69      public static void main(String[] args) {
70          // Set up command line parsing.
71          Options options = createCommandLineOptions();
72          CommandLineParser parser = new PosixParser();
73  
74          // Parse command line arguments.
75          CommandLine cl = null;
76          try {
77              cl = parser.parse(options, args);
78          } catch (ParseException e) {
79              throw new RuntimeException("Problem parsing command line options.", e);
80          }
81  
82          String[] parsedArgs = cl.getArgs();
83  
84          if (parsedArgs.length < 2) {
85              System.out.println("FilterMCBunches <input files> <output file>");
86              HelpFormatter help = new HelpFormatter();
87              help.printHelp(" ", options);
88              System.exit(1);
89          }
90  
91          String outFileName = parsedArgs[parsedArgs.length - 1];
92  
93          LCIOWriter writer = null;
94          try {
95              writer = new LCIOWriter(new File(outFileName));
96          } catch (IOException e) {
97              throw new RuntimeException(e);
98          }
99  
100         EventTester tester = null;
101         if (!cl.hasOption("a")) {
102             if (cl.hasOption("E")) {
103                 tester = new EcalEventTester(Double.valueOf(cl.getOptionValue("E")));
104             } else {
105                 tester = new EcalEventTester(0.05);
106             }
107         }
108         if (cl.hasOption("d")) {
109             if (cl.hasOption("E")) {
110                 tester = new EcalPairEventTester(Double.valueOf(cl.getOptionValue("E")));
111             } else {
112                 tester = new EcalPairEventTester(0.05);
113             }
114         }
115         if (cl.hasOption("t")) {
116             if (cl.hasOption("L")) {
117                 tester = new TrackerEventTester(Integer.valueOf(cl.getOptionValue("L")));
118             } else {
119                 tester = new TrackerEventTester(4);
120             }
121         }
122         if (cl.hasOption("r")) {
123             if (cl.hasOption("L")) {
124                 tester = new RTHEventTester(Integer.valueOf(cl.getOptionValue("L")));
125             } else {
126                 tester = new RTHEventTester(4);
127             }
128         }
129         if (cl.hasOption("p")) {
130             if (cl.hasOption("L")) {
131                 tester = new PairEventTester(Integer.valueOf(cl.getOptionValue("L")));
132             } else {
133                 tester = new PairEventTester(3);
134             }
135         }
136 
137         int nEmpty = 0;
138         if (cl.hasOption("e")) {
139             nEmpty = Integer.valueOf(cl.getOptionValue("e"));
140         } else {
141             System.out.println("You need to specify the number of empty bunches!");
142             System.exit(1);
143         }
144 
145         int nEvents = -1;
146         int nEventsToWrite = -1;
147         if (cl.hasOption("n")) {
148             nEvents = Integer.valueOf(cl.getOptionValue("n"));
149         }
150         if (cl.hasOption("w")) {
151             nEventsToWrite = Integer.valueOf(cl.getOptionValue("w"));
152         }
153         int readEvents = 0;
154         int writtenEvents = 0;
155 
156         String detectorName = null;
157 
158         fileLoop:
159         for (int fileNumber = 0; fileNumber < parsedArgs.length - 1; fileNumber++) {
160             LCIOReader reader = null;
161             String inFileName = parsedArgs[fileNumber];
162             try {
163                 reader = new LCIOReader(new File(inFileName));
164                 System.out.println("Opened input file " + inFileName);
165             } catch (IOException e) {
166                 throw new RuntimeException(e);
167             }
168 
169             eventLoop:
170             while (true) {
171                 if (nEvents != -1 && readEvents == nEvents) {
172                     break fileLoop;
173                 }
174                 EventHeader event;
175                 try {
176                     event = reader.read();
177                 } catch (IOException e) {
178                     break eventLoop;
179                 }
180                 readEvents++;
181 
182                 if (detectorName == null) {
183                     detectorName = event.getDetectorName();
184                 }
185 
186                 if (tester == null || tester.goodEvent(event)) {
187                     writtenEvents++;
188                     try {
189                         writer.write(event);
190                     } catch (IOException e) {
191                         throw new RuntimeException(e);
192                     }
193 
194                     for (int i = 1; i < nEmpty; i++) {
195                         try {
196                             writer.write(new BaseLCSimEvent(event.getRunNumber(), event.getEventNumber(), detectorName));
197                         } catch (IOException e) {
198                             throw new RuntimeException(e);
199                         }
200                     }
201                     if (nEventsToWrite != -1 && writtenEvents == nEventsToWrite) {
202                         break fileLoop;
203                     }
204                 }
205             }
206             try {
207                 reader.close();
208             } catch (IOException e) {
209                 throw new RuntimeException(e);
210             }
211         }
212 
213         System.out.format("Read %d events, wrote %d of them\n", readEvents, writtenEvents);
214         if (tester != null) {
215             tester.endOfRun();
216         }
217 
218         try {
219             writer.close();
220         } catch (IOException e) {
221             throw new RuntimeException(e);
222         }
223     }

225     static int countPairs(Set<Integer> ints) {
226         int pairs = 0;
227         while (!ints.isEmpty()) {
228             Integer next = ints.iterator().next();
229             ints.remove(next);
230             if (ints.remove(next ^ 1)) {
231                 pairs++;
232             }
233         }
234         return pairs;
235     }

246     private static class RTHEventTester extends EventTester {
247 
248         int nEvents = 0;
249         int nEcal = 0;
250         int nRTH = 0;
251         private int hitsNeeded;
252 
253         public RTHEventTester(int hitsNeeded) {
254             this.hitsNeeded = hitsNeeded;
255             LCSimConditionsManagerImplementation.register();
256         }
257 
258         @Override
259         public boolean goodEvent(EventHeader event) {
260             nEvents++;
261             List<RawCalorimeterHit> ecalHits = event.get(RawCalorimeterHit.class, "EcalReadoutHits");
262             List<RawTrackerHit> trackerHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
263             nEcal += ecalHits.size();
264             nRTH += trackerHits.size();
265 
266             Set<Integer> topLayers = new HashSet<Integer>();
267             Set<Integer> botLayers = new HashSet<Integer>();
268             for (RawTrackerHit hit : trackerHits) {
269                 IDDecoder dec = hit.getIDDecoder();
270                 dec.setID(hit.getCellID());
271                 if (dec.getValue("module") % 2 == 0) {
272                     topLayers.add(dec.getValue("layer"));
273                 } else {
274                     botLayers.add(dec.getValue("layer"));
275                 }
276             }
277 
278 //            System.out.format("%d SimCalorimeterHits, %d SimTrackerHits, %d top layers, %d bottom layers\n", ecalHits.size(), trackerHits.size(), topLayers.size(), botLayers.size());
279             return (countPairs(topLayers) >= hitsNeeded && countPairs(botLayers) >= hitsNeeded);
280         }
281 
282         @Override
283         void endOfRun() {
284             System.out.format("%d events, %f RawCalorimeterHits and %f RawTrackerHits on average\n", nEvents, ((double) nEcal) / nEvents, ((double) nRTH) / nEvents);
285         }
286     }

378     private static class PairEventTester extends EventTester {
379 
380         private int hitsNeeded;
381 
382         public PairEventTester(int hitsNeeded) {
383             this.hitsNeeded = hitsNeeded;
384             LCSimConditionsManagerImplementation.register();
385         }
386 
387         @Override
388         public boolean goodEvent(EventHeader event) {
389 //            List<SimCalorimeterHit> ecalHits = event.getSimCalorimeterHits("EcalHits");
390             List<SimTrackerHit> trackerHits = event.get(SimTrackerHit.class, "TrackerHits");
391             List<MCParticle> mcParticles = event.getMCParticles();
392 
393             Map<MCParticle, Set<Integer>> particleMap = new HashMap<MCParticle, Set<Integer>>();
394             List<MCParticle> particleList = new LinkedList<MCParticle>(mcParticles);
395             List<MCParticle> priParticles = new ArrayList<MCParticle>();
396 
397             while (!particleList.isEmpty()) {
398                 Iterator<MCParticle> iter = particleList.iterator();
399                 while (iter.hasNext()) {
400                     MCParticle particle = iter.next();
401 
402                     if (!particle.getSimulatorStatus().isCreatedInSimulation()) {
403                         particleMap.put(particle, new HashSet<Integer>());
404                         priParticles.add(particle);
405                         iter.remove();
406                         continue;
407                     }
408 
409                     for (MCParticle parent : particle.getParents()) {
410                         if (particleMap.containsKey(parent)) {
411                             particleMap.put(particle, particleMap.get(parent));
412                             iter.remove();
413                             break;
414                         }
415                     }
416                 }
417             }
418 
419 
420             for (SimTrackerHit hit : trackerHits) {
421                 IDDecoder dec = hit.getIDDecoder();
422                 dec.setID(hit.getCellID());
423                 particleMap.get(hit.getMCParticle()).add(dec.getValue("layer"));
424             }
425 
426             int particlesWithTracks = 0;
427             for (MCParticle priParticle : priParticles) {
428                 int nLayers = countPairs(particleMap.get(priParticle));
429 //                System.out.format("layers hit: %d minimum number needed: %d\n", nLayers, nHits);
430                 if (nLayers >= hitsNeeded) {
431                     particlesWithTracks++;
432                 }
433             }
434 
435 //            System.out.format("%d of %d primary particles have tracks\n", particlesWithTracks, priParticles.size());
436             return (particlesWithTracks >= 2);
437         }
