1   package org.hps.util;
2   
3   import hep.io.stdhep.StdhepEvent;
4   import hep.io.stdhep.StdhepWriter;
5   import hep.physics.particle.properties.ParticleType;
6   import hep.physics.particle.properties.ParticlePropertyManager;
7   import java.io.IOException;
8   import java.util.Random;
9   
10  /**
11   * Simple java program to generate single particles of fixed momentum uniformly
12   * distributed in pseudorapidity.
13   */
14  public class GenerateSingleParticles {
15  
16      //  Edit the following declarations to set desired paramters
17      static String dir = "/nfs/sulky21/g.ec.u12/users/mgraham/DarkPhoton/SingleParticleSamples/";        // Output directory
18  //    static String file = "electron_6GeV_10spills";  // File name
19      static String file = "electron_6GeV_10000_XDir";  // File name
20      static String ext = "stdhep";      // File extension
21  //    static int nevt = 15000*10;            // Number of events to generate
22      static int nevt = 10000;            // Number of events to generate
23      static int pdgid = 11;             // PDG code for particle to generate (11=electrons, 13 = muons, 211 = pions)
24      static boolean flipsign = false;    // Set to false if you only want a specific charge
25      static double pmin = 6.;         // Minimum particle momentum at 90 degrees (GeV)
26      static double pmax = 1.;         // Maximum particle momentum at 90 degrees (GeV)
27      static boolean pfixed = true;      // Set to false for fixed pt
28      static double sigx = 0.001;        // Luminous region size in x (mm)
29      static double sigy = 0.001;        // Luminous region size in y (mm)
30      static double sigz = 0.001;         // Luminous region size in z (mm)
31  
32      /*
33      static double sigpx = 0.001;        // momentum spread
34      static double sigpy = 0.001;        // 
35      static double sigpz = 0.001;         //
36  */
37      static double sigpx = 0.6;        // momentum spread
38      static double sigpy = 0.6;        //
39      static double sigpz = 0.001;         //
40      public static void main(String[] args) throws IOException {
41  
42          //  Instantiate the random number generators
43          Random generator = new Random();
44  
45          //  Decode the pdgid
46          ParticleType pid = ParticlePropertyManager.getParticlePropertyProvider().get(pdgid);
47          String pname = pid.getName();
48  
49          //  Open the output file
50          String fname = dir + file + "." + ext;
51          StdhepWriter sw = new StdhepWriter(fname, "Single particles",
52                  "Single particles", nevt);
53          sw.setCompatibilityMode(false);
54  
55          //  Loop over the events
56          for (int icross = 0; icross < nevt; icross++) {
57  
58              //  Generate the momentum at 90 degrees
59              double ptot = pmin + (pmax - pmin) * generator.nextDouble();
60              double px = sigpx * generator.nextGaussian();
61              double py = sigpy * generator.nextGaussian();
62              //  Generate the pseudorapidity and calculate the polar angle
63  
64              //  Figure out the transverse momentum and mass
65  
66              double pt = Math.sqrt(px * px + py * py);
67              double theta = Math.asin(pt / ptot);
68              double m = pid.getMass();
69  
70              //  Generate the aximutha angle
71              double phi = 2. * Math.PI * generator.nextDouble();
72  
73              //  Generate the IP
74              double x0 = sigx * generator.nextGaussian();
75              double y0 = sigy * generator.nextGaussian();
76              double z0 = sigz * generator.nextGaussian();
77              double t0 = 0.;
78  
79              //  Fill the event record variables
80              //  Set the event number
81              int nevhep = icross;
82  
83              //  Set the number of particles in the event
84              int nhep = 1;
85  
86              //  Set the status code of the particle
87              int isthep[] = new int[2];
88              isthep[0] = 1;
89  
90              //  Set the particle ID of the particle
91              int idhep[] = new int[2];
92              idhep[0] = pdgid;
93              if (flipsign && generator.nextDouble() > 0.5) {
94                  idhep[0] = -pdgid;
95              }
96  
97              //  Set the mother and daughter pointers
98              int jmohep[] = new int[2];
99              int jdahep[] = new int[2];
100             for (int i = 0; i < 2; i++) {
101                 jmohep[i] = 0;
102                 jdahep[i] = 0;
103             }
104 
105             //  Set the momentum/energy/mass
106             double phep[] = new double[5];
107 //            phep[0] = pt * Math.cos(phi);
108 //            phep[1] = pt * Math.sin(phi);
109 //            phep[2] = pt / Math.tan(theta);
110 // swap x and z
111             phep[2] = pt * Math.cos(phi);
112             phep[1] = pt * Math.sin(phi);
113             phep[0] = pt / Math.tan(theta);
114             double p2 = pt * pt + phep[2] * phep[2];
115             phep[3] = Math.sqrt(p2 + m * m);
116             phep[4] = m;
117 
118             //  Set the particle origin
119             double vhep[] = new double[4];
120 //            vhep[0] = x0;
121 //            vhep[1] = y0;
122 //           vhep[2] = z0;
123             vhep[2] = x0;
124             vhep[1] = y0;
125             vhep[0] = z0;
126             vhep[3] = t0;
127 
128             //  Create an event record
129             StdhepEvent ev = new StdhepEvent(nevhep, nhep, isthep, idhep, jmohep, jdahep, phep, vhep);
130 
131             //  Write out the event record
132             sw.writeRecord(ev);
133         }
134 
135         //  Done with generating particles - close the file
136         sw.close();
137     }
138 }
