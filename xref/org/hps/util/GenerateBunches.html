1   package org.hps.util;
2   
3   import hep.io.stdhep.StdhepEvent;
4   import hep.io.stdhep.StdhepWriter;
5   import hep.physics.particle.properties.ParticleType;
6   import hep.physics.particle.properties.ParticlePropertyManager;
7   import java.io.IOException;
8   import java.util.Random;
9   
10  /**
11   * Simple java program to generate single particles of fixed momentum uniformly
12   * distributed in pseudorapidity.
13   */
14  public class GenerateBunches {
15  
16      //  Edit the following declarations to set desired paramters
17      static String dir = "/a/surrey10/vol/vol0/g.hps/mgraham/DarkPhoton/SingleParticleSamples/5.5GeV-100u/";        // Output directory
18  // static String dir = "/a/surrey10/vol/vol0/g.hps/mgraham/DarkPhoton/SingleParticleSamples/";
19      static String file = "electron_5.5GeV_7.5ns_100na_400bunches_";  // File name
20  //    static String file = "electron_6GeV_25ns_10bunches_";  // File name
21  
22      static int nevt = 400;            // Number of events/file to generate
23      static int nfiles = 400;             //number of files to generate
24      static int start=400;
25  
26      static double current = 100;  //nanoamps
27      static double bunchtime=7.5;  //nanoseconds
28  
29      static double convert=6.25;   //electrons/ns in a nanoamp
30      static String ext = "stdhep";      // File extension
31  
32  //    static int nele = 15000;            // Number of elecrons in a bunch (25ns)
33      static int nele = (int) Math.round(current*bunchtime*convert);            // Number of elecrons in a bunch (7.5ns)
34      
35      
36      static int pdgid = 11;             // PDG code for particle to generate (11=electrons, 13 = muons, 211 = pions)
37      static boolean flipsign = false;    // Set to false if you only want a specific charge
38      static double pmin = 5.5;         // Minimum particle momentum at 90 degrees (GeV)
39      static double pmax = 5.5;         // Maximum particle momentum at 90 degrees (GeV)
40      static boolean pfixed = true;      // Set to false for fixed pt
41      static double sigx = 0.1;        // Luminous region size in x (mm)
42      static double sigy = 0.1;        // Luminous region size in y (mm)
43      static double sigz = 0.001;         // Luminous region size in z (mm)
44      static double sigpx = 0.001;        // momentum spread
45      static double sigpy = 0.001;        // 
46      static double sigpz = 0.001;         //
47      static double xoff = -1.0;         // offset of the beam in the xdir (so that it goes through the target)
48  
49      public static void main(String[] args) throws IOException {
50  
51          //  Instantiate the random number generators
52          Random generator = new Random();
53          generator.setSeed(start);
54          //  Decode the pdgid
55          ParticleType pid = ParticlePropertyManager.getParticlePropertyProvider().get(pdgid);
56          String pname = pid.getName();
57  
58  
59          for (int n = start; n < start+nfiles; n++) {
60              //  Open the output file
61              String fname = dir + file + n + "." + ext;
62              StdhepWriter sw = new StdhepWriter(fname, "Bunch",
63                      "Bunch", nevt);
64              sw.setCompatibilityMode(false);
65  
66              //  Loop over the events
67              for (int icross = 0; icross < nevt; icross++) {
68  
69                  //  Generate the momentum at 90 degrees
70  
71  
72                  //  Fill the event record variables
73                  //  Set the event number
74                  int nevhep = icross;
75  
76                  //  Set the number of particles in the event
77                  int nhep = nele;
78                  int isthep[] = new int[nhep];
79                  int idhep[] = new int[nhep];
80                  int jmohep[] = new int[2 * nhep];
81                  int jdahep[] = new int[2 * nhep];
82                  double phep[] = new double[5 * nhep];
83                  double vhep[] = new double[4 * nhep];
84                  for (int jj = 0; jj < nhep; jj++) {
85  
86                      double ptot = pmin + (pmax - pmin) * generator.nextDouble();
87                      double px = sigpx * generator.nextGaussian();
88                      double py = sigpy * generator.nextGaussian();
89                      //  Generate the pseudorapidity and calculate the polar angle
90  
91                      //  Figure out the transverse momentum and mass
92  
93                      double pt = Math.sqrt(px * px + py * py);
94                      double theta = Math.asin(pt / ptot);
95                      double m = pid.getMass();
96  
97                      //  Generate the aximutha angle
98                      double phi = 2. * Math.PI * generator.nextDouble();
99  
100                     //  Generate the IP
101                     double x0 = sigx * generator.nextGaussian();
102                     double y0 = sigy * generator.nextGaussian();
103                     double z0 = sigz * generator.nextGaussian();
104                     double t0 = 0.;
105                     //  Set the status code of the particle
106 
107                     isthep[jj] = 1;
108 
109                     //  Set the particle ID of the particle
110 
111                     idhep[jj] = pdgid;
112                     if (flipsign && generator.nextDouble() > 0.5)
113                         idhep[jj] = -pdgid;
114 
115                     //  Set the mother and daughter pointers
116 
117                     for (int i = 0; i < 2; i++) {
118                         jmohep[2 * jj + i] = 0;
119                         jdahep[2 * jj + i] = 0;
120                     }
121 
122                     //  Set the momentum/energy/mass
123 
124 //            phep[0] = pt * Math.cos(phi);
125 //            phep[1] = pt * Math.sin(phi);
126 //            phep[2] = pt / Math.tan(theta);
127 // swap x and z
128                     phep[5 * jj + 2] = pt * Math.cos(phi);
129                     phep[5 * jj + 1] = pt * Math.sin(phi);
130                     phep[5 * jj + 0] = pt / Math.tan(theta);
131                     double p2 = pt * pt + phep[2] * phep[2];
132                     phep[5 * jj + 3] = Math.sqrt(p2 + m * m);
133                     phep[5 * jj + 4] = m;
134 
135                     //  Set the particle origin
136 
137 //            vhep[0] = x0;
138 //            vhep[1] = y0;
139 //           vhep[2] = z0;
140                     vhep[4 * jj + 2] = x0;
141                     vhep[4 * jj + 1] = y0;
142                     vhep[4 * jj + 0] = z0 + xoff;
143                     vhep[4 * jj + 3] = t0;
144                 }
145                 //  Create an event record
146                 StdhepEvent ev = new StdhepEvent(nevhep, nhep, isthep, idhep, jmohep, jdahep, phep, vhep);
147 
148                 //  Write out the event record
149                 sw.writeRecord(ev);
150             }
151 
152             //  Done with generating particles - close the file
153             sw.close();
154         }
155     }
