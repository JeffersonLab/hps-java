413         boolean fndNumber = false;

415         boolean fndEvents = false;
416         int evts = -1;

429                 evts = (int) tok.nval;

453     static private double findMaxWeight() {
454         Random generator = new Random();
455         int ntrials = 100000;
456         double maxlength = maxLen;
457         double maxWeight = 0;
458         for (int i = 0; i < ntrials; i++) {
459             double x = generator.nextDouble() * maxlength;
460             double wght = expWeight(x);
461             if (wght > maxWeight)
462                 maxWeight = wght;
463         }
464 
465         return maxWeight;
466     }

468     static private double getDecayLength(double MaxWeight, double gamma) {
469         Random generator = new Random();
470         double maxlength = maxLen;
471         double dl = 0;
472         double draw = generator.nextDouble();
473         double tmpwght = 0;
474         while (tmpwght < draw) {
475             dl = generator.nextDouble() * maxlength;
476             tmpwght = expWeight(dl, gamma) / MaxWeight;
477         }
478         return dl;
479     }

571                     for (int ii = 0; ii < nLayers; ii++) {

573                         // System.out.println("path length "+pathL);

584                         passLayerTarget50.add(inAcceptance(posLTg50, ySize[ii], zSize[ii], zGap[ii]));
585                     }
586 
587                     if (pdgid == 611) { // electron from A'
588                         passEleTarget50 = passLayerTarget50;
589                         passEleNoMax = passLayerNoMax;
590                         passEleFull = passLayerFull;
591                         pEle = pl;
592                     } else if (pdgid == -611 || pdgid == -11) {// positron
593                         pPos = pl;
594                         passPosTarget50 = passLayerTarget50;
595                         passPosNoMax = passLayerNoMax;
596                         passPosFull = passLayerFull;
597                     } else if (pdgid == 11)

631     public static boolean inAcceptance(Hep3Vector position, double yExt, double zExt, double zGap) {
632         double ypos = position.y();
633         if (Math.abs(ypos) > yExt)
634             return false;
635         double zpos = position.z();
636         if (Math.abs(zpos) < zGap)
637             return false;
638         if (Math.abs(zpos) > zGap + zExt)
639             return false;
640 
641         return true;
642     }

651         if (p1.magnitude() < _pCut)

664     public static Double PathToXPlane(double x0, double y0, double xc, double yc, double RC, double x) {
665         // Create a list to hold the path lengths
666         Double path;
667 
668         double y = yc + Math.signum(RC) * Math.sqrt(RC * RC - Math.pow(x - xc, 2));
669         // System.out.println("x = "+x+"; y = "+y);
670         double s = PathCalc(xc, yc, RC, x0, y0, x, y);
671 
672         // System.out.println("PathToXPlane :  s = "+s+"; sFromClass = "+sFromClass);
673 
674         path = s;
675 
676         return path;
677     }

679     private static double PathCalc(double xc, double yc, double RC, double x1, double y1, double x2, double y2) {
680         // Find the angle between these points measured wrt the circle center
681         double phi1 = Math.atan2(y1 - yc, x1 - xc);
682         double phi2 = Math.atan2(y2 - yc, x2 - xc);
683         double dphi = phi2 - phi1;
684         // Make sure dphi is in the valid range (-pi, pi)
685         if (dphi > Math.PI)
686             dphi -= 2. * Math.PI;
687         if (dphi < -Math.PI)
688             dphi += 2. * Math.PI;
689         // Return the arc length
690         return -RC * dphi;
691     }

693     public static Hep3Vector PointOnHelix(double xc, double yc, double RC, double phi0, double z0, double slope,
694             double s) {
695         // Find the azimuthal direction at this path length
696         double phi = phi0 - s / RC;
697         // Calculate the position on the helix at this path length
698         double x = xc - RC * Math.sin(phi);
699         double y = yc + RC * Math.cos(phi);
700         double z = z0 + s * slope;
701         // Return the position as a Hep3Vector
702         return new BasicHep3Vector(x, y, z);
703     }

713     public static double getInvMass(Hep3Vector p1, Hep3Vector p2) {
714         double esum = 0.;
715         double pxsum = 0.;
716         double pysum = 0.;
717         double pzsum = 0.;
718         double me = 0.000511;
719         if (_isMuon)
720             me = 0.1057;
721         // Loop over tracks
722 
723         double p1x = p1.x();
724         double p1y = p1.y();
725         double p1z = p1.z();
726         double p1mag2 = p1x * p1x + p1y * p1y + p1z * p1z;
727         double e1 = Math.sqrt(p1mag2 + me * me);
728         pxsum += p1x;
729         pysum += p1y;
730         pzsum += p1z;
731         esum += e1;
732 
733         double p2x = p2.x();
734         double p2y = p2.y();
735         double p2z = p2.z();
736         double p2mag2 = p2x * p2x + p2y * p2y + p2z * p2z;
737         double e2 = Math.sqrt(p2mag2 + me * me);
738         pxsum += p2x;
739         pysum += p2y;
740         pzsum += p2z;
741         esum += e2;
742         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
743         double evtmass = esum * esum - psum * psum;
744 
745         if (evtmass > 0)
746             return Math.sqrt(evtmass);
747         else
748             return -99;
749     }
