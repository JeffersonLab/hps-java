1   package org.hps.util;

3   import org.lcsim.event.EventHeader;
4   import org.lcsim.util.Driver;
5   
6   /**
7    * Lightweight timer driver. Add one copy of this driver to measure time per event; add multiple copies to measure time between two points in the execute list.
8    */
9   public class TimerDriver extends Driver {
10  
11      private static long time = Long.MIN_VALUE;
12      private long timeElapsed;
13      private int count = 0;
14      private String name = "unnamed";
15      private boolean verbose = false;
16  
17      public void setName(String name) {
18          this.name = name;
19      }
20  
21      @Override
22      public void process(EventHeader event) {
23          long oldTime = time;
24          time = System.nanoTime();
25          if (oldTime != Long.MIN_VALUE) {
26              long dt = time - oldTime;
27              timeElapsed += dt;
28              count++;
29              if (verbose) {
30                  System.out.format("Timer %s: dt %f, timeElapsed %d, count %d, mean time %f ms\n", name, dt / 1.0e6, timeElapsed, count, (timeElapsed / 1.0e6) / count);
31              } else {
32                  System.out.format("Timer %s: dt %f, mean time %f ms\n", name, dt / 1.0e6, (timeElapsed / 1.0e6) / count);
33              }
34          }
35      }
36  }
