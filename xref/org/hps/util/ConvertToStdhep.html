305     private static int process(String infile) throws IOException {
306         Random generator = new Random();
307 
308         FileReader lc = new FileReader(infile);
309         StreamTokenizer lctok = new StreamTokenizer(lc);
310         lctok.resetSyntax();
311         lctok.wordChars(33, 255);
312         lctok.parseNumbers();
313 
314         lctok.whitespaceChars(0, ' ');
315         lctok.eolIsSignificant(true);
316         int nevts = getnevts(lctok);
317         lc.close();
318         System.out.println("Found " + nevts + " events");
319 
320         FileReader fr = new FileReader(infile);
321 
322         StreamTokenizer tok = new StreamTokenizer(fr);
323 
324         tok.resetSyntax();
325         tok.wordChars(33, 255);
326         tok.parseNumbers();
327 
328         tok.whitespaceChars(0, ' ');
329         tok.eolIsSignificant(true);
330 
331         // System.out.println("Found " + nevts + "  events");
332         int nreq = (int) nevts;
333         int ngen = (int) nevts;
334         int nwrit = (int) nevts;
335         float ecm = (float) _ecm;
336         float xsec = (float) 99999997952.;
337         double rn1 = 12345321;
338         double rn2 = 66666666;
339         // StdhepBeginRun sb = new StdhepBeginRun(nreq, ngen, nwrit, ecm, xsec, rn1, rn2);
340         // sw.writeRecord(sb);
341 
342         tok.resetSyntax();
343         tok.wordChars(33, 255);
344         tok.wordChars('0', '9'); // java.io.StreamTokenizer fails to parse
345         tok.wordChars('e', 'e'); // scientific notation like "1.09E-008".
346         tok.wordChars('E', 'E'); // The solution is to read and parse
347         tok.wordChars('.', '.'); // coordinates as "words".
348         tok.wordChars('+', '+'); // You run into trouble if the input file
349         tok.wordChars('-', '-'); // contains text with "e" or "E" which is
350         tok.whitespaceChars(0, ' ');
351         tok.eolIsSignificant(true);
352 
353         double[] beam = {0, 0, 0, 0};
354         int nevhep = 0;
355         for (int icross = 0; icross < nwrit; icross++) {
356             Hep3Vector beamVecOrig = new BasicHep3Vector(sigx * generator.nextGaussian() + _xoff, sigy
357                     * generator.nextGaussian() + _yoff, sigz * generator.nextGaussian() + _zoff);
358             Hep3Vector beamVec = rot.rotated(beamVecOrig);
359             beam[0] = beamVec.x();
360             beam[1] = beamVec.y();
361             beam[2] = beamVec.z();
362 
363             double tmpDecLen = 0;
364 
365             if (fileType.contains("lhe")) {
366                 writeLHEEvent(tok, beam, icross);
367             } else if (fileType.contains("dat")) {
368                 writeDATEvent(tok, beam, tmpDecLen, icross);
369             }
370 
371         }
372         fr.close();
373 
374         return nwrit;
375 
376     }

394     @SuppressWarnings("static-access")
395     private static List<Double> getNumbersInLine(StreamTokenizer tok) throws IOException {
396         List<Double> nums = new ArrayList<Double>();
397         while (tok.nextToken() != tok.TT_EOF) {
398             if (tok.ttype == tok.TT_EOL) {
399                 break;
400             }
401             String tokVal = tok.sval;
402             // System.out.println(tokVal);
403             nums.add(Double.valueOf(tokVal).doubleValue());
404         }
405 
406         return nums;
407     }

409     @SuppressWarnings("static-access")
410     private static int getNumberOfEvents(StreamTokenizer tok) throws IOException {
411         boolean fndNumber = false;
412         boolean fndOf = false;
413         boolean fndEvents = false;
414         int evts = -1;
415         while (tok.nextToken() != tok.TT_EOF) {
416             // System.out.println(tok.toString());
417             if (tok.ttype == tok.TT_EOL) {
418                 break;
419             }
420             if (tok.ttype == tok.TT_WORD && tok.sval.contentEquals("Number")) {
421                 // System.out.println(tok.toString());
422                 fndNumber = true;
423             }
424             if (tok.ttype == tok.TT_WORD && tok.sval.contentEquals("of")) {
425                 fndOf = true;
426             }
427             if (tok.ttype == tok.TT_WORD && tok.sval.contentEquals("Events")) {
428                 fndEvents = true;
429             }
430             if (tok.ttype == tok.TT_NUMBER && fndEvents && fndOf && fndNumber) {
431                 evts = (int) tok.nval;
432             }
433         }
434         return evts;
435     }

587     static private void writeLHEEvent(StreamTokenizer tok, double[] beam, int nevhep) throws IOException {
588         Random generator = new Random();
589         getToNextEvent(tok);
590         List<Double> nums = getNumbersInLine(tok);
591 
592         if (nums.size() != 6) {
593             throw new RuntimeException("Unexpected entry for number of particles");
594         }
595         int nhep = nums.get(0).intValue();
596         // System.out.println("Number of particles for event " + nevhep + ": " + nhep);
597 
598         double decLen = 0;
599         double maxWght = 0;
600 
601         /*
602          * No longer needed if (expDecay) { maxWght = findMaxWeight(); }
603          */
604         int isthep[] = new int[nhep];
605         int idhep[] = new int[nhep];
606         int jmohep[] = new int[2 * nhep];
607         int jdahep[] = new int[2 * nhep];
608         double phep[] = new double[5 * nhep];
609         double vhep[] = new double[4 * nhep];
610         int i = 0;
611         int idhepTmp = 0;
612         double[] ApMom = {0, 0, 0};
613         double ApMass = 0;
614         double ApEnergy = 0;
615         boolean acceptEvent = false;
616         int nElePass = 0;
617         for (int npart = 0; npart < nhep; npart++) {
618             List<Double> vals = getNumbersInLine(tok);
619             if (vals.size() != 13) {
620                 throw new RuntimeException("Unexpected entry for a particle");
621             }
622             if (vals.get(1).intValue() != 9) {// ignore the vertex for now
623             // int ip = vals.get(0).intValue();
624             // if (ip != i + 1) {
625             // throw new RuntimeException("Particle numbering mismatch");
626             // }
627                 isthep[i] = vals.get(1).intValue();
628 
629                 if (vals.get(1).intValue() == -1) {
630                     isthep[i] = 3;
631                 }
632                 idhepTmp = vals.get(0).intValue();
633                 // System.out.println(idhepTmp);
634                 idhep[i] = idhepTmp;
635                 if (idhep[i] == 611) {
636                     idhep[i] = 11;
637                 }
638                 if (idhep[i] == -611) {
639                     idhep[i] = -11;
640                 }
641                 jmohep[2 * i] = vals.get(2).intValue();
642                 jmohep[2 * i + 1] = vals.get(3).intValue();
643                 jdahep[2 * i] = vals.get(4).intValue();
644                 jdahep[2 * i + 1] = vals.get(5).intValue();
645                 for (int j = 0; j < 5; j++) {
646                     phep[5 * i + j] = vals.get(j + 6);
647                 }
648                 Hep3Vector pRot = rotateToDetector(phep[5 * i + 0], phep[5 * i + 1], phep[5 * i + 2]);
649                 phep[5 * i + 0] = pRot.x();
650                 phep[5 * i + 1] = pRot.y();
651                 phep[5 * i + 2] = pRot.z();
652                 if (Math.abs(idhepTmp) == 622) {
653                     /*
654                      * ApMom[0] = phep[5 * i + 2]; ApMom[1] = phep[5 * i + 1]; ApMom[2] = phep[5 * i + 0];
655                      */
656                     ApMom[0] = pRot.x();
657                     ApMom[1] = pRot.y();
658                     ApMom[2] = pRot.z();
659                     ApMass = vals.get(10);
660                     ApEnergy = vals.get(9);
661                 }
662                 for (int j = 0; j < 4; j++) {
663                     vhep[4 * i + j] = beam[j];
664                 }
665 
666                 if (!trident && (Math.abs(idhepTmp) == 611 || Math.abs(idhepTmp) == 622)) {
667                     double gamma = ApEnergy / ApMass;
668                     if (expDecay) {
669                         decLen = getDecayLength(gamma);
670                         // decLen = getDecayLength(maxWght, gamma);
671                     }
672                     if (flatDecay) {
673                         decLen = generator.nextDouble() * maxLen;
674                     }
675                     if (!expDecay && !flatDecay) {
676                         vhep[4 * i + 0] = beam[0] + decLen;
677                     } else {
678                         double totApMom = Math.sqrt(ApMom[0] * ApMom[0] + ApMom[1] * ApMom[1] + ApMom[2] * ApMom[2]);
679                         System.out.println("Decay at : " + decLen);
680                         vhep[4 * i + 0] = beam[0] + decLen * ApMom[0] / totApMom;
681                         vhep[4 * i + 1] = beam[1] + decLen * ApMom[1] / totApMom;
682                         vhep[4 * i + 2] = beam[2] + decLen * ApMom[2] / totApMom;
683                     }
684                 }
685 
686                 if (Math.abs(idhepTmp) == 611 || Math.abs(idhepTmp) == 11) {
687                     if (inAcceptance(pRot)) {
688                         // acceptEvent = true;
689                         nElePass++;
690                     }
691                 }
692 
693                 // swap x and z axes...
694                 /*
695                  * Don't do this anymore! We do stuff in JLAB frame now... double px = phep[5 * i + 0]; double pz =
696                  * phep[5 * i + 2]; phep[5 * i + 0] = pz; phep[5 * i + 2] = px;
697                  */
698                 // if (i == 0 || i == nhep - 1) {
699                 // System.out.println(i + " st: " + isthep[i] + " id: " + idhep[i] +
700                 // " jmo: " + jmohep[2 * i] + " " + jmohep[2 * i + 1] +
701                 // " jda: " + jdahep[2 * i] + " " + jdahep[2 * i + 1]);
702                 // System.out.println("p: " + phep[5 * i] + " " + phep[5 * i + 1] + " " +
703                 // phep[5 * i + 2] + " " + phep[5 * i + 3] + " " + phep[5 * i + 4]);
704                 // System.out.println("v: " + vhep[4 * i] + " " + vhep[4 * i + 1] + " " +
705                 // vhep[4 * i + 2] + " " + vhep[4 * i + 3]);
706                 // }
707                 i++;
708             }
709         }
710         StdhepEvent ev = new StdhepEvent(nevhep, nhep, isthep, idhep, jmohep, jdahep, phep, vhep);
711         if (nElePass >= _nEleRequired) {
712             acceptEvent = true;
713         }
714         if (acceptEvent) {
715             sw.writeRecord(ev);
716         } else {
717             // System.out.println("LHE Event failed acceptance cuts");
718         }
719     }

739     public static boolean inAcceptance(Hep3Vector ph) {
740         boolean ok = false;
741         double[] p = {ph.x(), ph.y(), ph.z()};
742         double ptot = Math.sqrt(p[0] * p[0] + p[1] * p[1] + p[2] * p[2]);
743 
744         double sinThx = p[0] / Math.sqrt(p[0] * p[0] + p[2] * p[2]);
745         double sinThy = p[1] / Math.sqrt(p[1] * p[1] + p[2] * p[2]);
746         // for now, just use thetay (non-bend direction)
747         // System.out.println("px = "+p[0]+"; py = "+p[1]+"; pz = "+p[2]);
748         // System.out.println(sinThy+" "+sinThx+" "+ptot);
749 
750         if (Math.abs(sinThy) > 0.012 && ptot > 0.1) {
751             ok = true;
752         }
753         return ok;
