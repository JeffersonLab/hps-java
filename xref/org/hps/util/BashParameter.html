52      public static final String format(String text, BashParameter... params) {
53          // Only one text and background color, at most, are
54          // allowed. Track whether more than one is seen.
55          boolean sawTextColor = false;
56          boolean sawBackgroundColor = false;
57          
58          // Buffer the formatted text.
59          StringBuffer buffer = new StringBuffer();
60          
61          for(BashParameter param : params) {
62              // If the default parameter is declared, it must be
63              // the only parameters defined, since it overrides
64              // all other parameters.
65              if(param == BashParameter.DEFAULT && params.length != 1) {
66                  throw new IllegalArgumentException("Error: The default style parameter can not be employed in conjunction with other style parameters.");
67              }
68              
69              // Only one text color may be defined at a time, as
70              // subsequent colors will override the earlier ones.
71              if(param.isTextColor()) {
72                  if(sawTextColor) {
73                      throw new IllegalArgumentException("Error: Only one text color may be defined.");
74                  } else {
75                      sawTextColor = true;
76                  }
77              }
78              
79              // Only one background color may be defined at a
80              // time, as subsequent colors will override the
81              // earlier ones.
82              if(param.isBackgroundColor()) {
83                  if(sawBackgroundColor) {
84                      throw new IllegalArgumentException("Error: Only one background color may be defined.");
85                  } else {
86                      sawBackgroundColor = true;
87                  }
88              }
89              
90              // Append the full bash text style code to the
91              // buffer.
92              buffer.append(param.getTextCode());
93          }
94          
95          // Add the text to the buffer, and then reset the style
96          // to the default.
97          buffer.append(text);
98          buffer.append(BashParameter.DEFAULT.getTextCode());
99          
100         // Return the text.
101         return buffer.toString();
102     }
