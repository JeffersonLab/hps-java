1   package org.hps.util.test;

3   import java.io.File;
4   
5   /**
6    * Convenience class for test file output
7    */
8   @SuppressWarnings("serial")
9   public class TestOutputFile extends File {
10  
11      private static final String TEST_OUTPUT_DIR = "target/test-output";
12  
13      /**
14       * Create an output file in a test-specific directory
15       *
16       * @param testClass the TestCase class
17       * @param filename The name of the file
18       */
19      public TestOutputFile(Class<?> testClass, String filename) {
20          super(TEST_OUTPUT_DIR + File.separator + testClass.getSimpleName() + File.separator + filename);
21          File dir = getParentFile();
22          if (!dir.exists()) {
23              dir.mkdirs();
24          }
25      }
26  
27      /**
28       * Create an output file in the generic output directory
29       * @param filename The name of the file
30       */
31      public TestOutputFile(String filename) {
32          super(TEST_OUTPUT_DIR + File.separator + filename);
33          File dir = getParentFile();
34          if (!dir.exists()) {
35              dir.mkdirs();
36          }
37      }
38  }
