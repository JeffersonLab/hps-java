1   package org.hps.util.test;
2   
3   import java.io.File;
4   import java.net.URL;
5   
6   import org.lcsim.util.cache.FileCache;
7   
8   /**
9    * Static utilities for accessing test files at JLAB, such as EVIO and LCIO input data
10   * and reference AIDA plots
11   *
12   * Additional test utility methods may be added here, as needed.
13   */
14  public final class TestUtil {
15  
16      /**
17       * Base URL for all test files
18       */
19      private static final String JLAB_BASE = "https://hpsweb.jlab.org/test/hps-java";
20  
21      /**
22       * Base URL for test data files such as EVIO or LCIO input data
23       */
24      private static final String JLAB_DATA = JLAB_BASE + "/data";
25  
26      /**
27       * Base URL for reference plots
28       */
29      private static final String JLAB_REF = JLAB_BASE + "/referencePlots";
30  
31      /**
32       * Download a test data file by name
33       * @param fileName The name of the file, relative to the test data directory
34       * @return The test data file
35       * @throws RuntimeException If there is an error downloading the file or it does not exist
36       */
37      public static File downloadTestFile(String fileName) {
38          try {
39              URL url = new URL(JLAB_DATA + "/" + fileName);
40              FileCache cache = new FileCache();
41              cache.setPrintStream(null);
42              File file = cache.getCachedFile(url);
43              return file;
44          } catch (Exception e) {
45              throw new RuntimeException(e);
46          }
47      }
48  
49      /**
50       * Download reference AIDA plots by name
51       * @param name The name of the IT
52       * @return The AIDA reference file
53       * @throws RuntimeException If there is an error downloading the plots or the file does not exist
54       */
55      public static File downloadRefPlots(String name) {
56          try {
57              URL url = new URL(JLAB_REF + "/" + name + "/" + name + "-ref.aida");
58              FileCache cache = new FileCache();
59              cache.setPrintStream(null);
60              File file = cache.getCachedFile(url);
61              return file;
62          } catch (Exception e) {
63              throw new RuntimeException(e);
64          }
65      }
66  
67      /**
68       * Private constructor to disallow class instantiation
69       */
70      private TestUtil() {
71      }
72  }
