24  /**
25   * Extracts LCIO events with positron hits at Ecal. For an event, loop all hits
26   * at Ecal. If any hit has contribution from positron, then the event is
27   * extracted to be saved in output. Hits could be cut with lower energy limit.
28   */
29  public class ExtractEventsWithHitAtHodoEcal {
30      static {
31          LCSimConditionsManagerImplementation.register();
32          DefaultLoggingConfig.initialize();
33      }
34  
35      /**
36       * Defines command line options for this program.
37       *
38       * @return The command line options.
39       */
40      private static Options createCommandLineOptions() {
41          Options options = new Options();
42  
43          options.addOption(new Option("n", true, "Number of events to read"));
44          options.addOption(new Option("w", true, "Number of events to write"));
45          options.addOption(new Option("M", true, "Number of required hits at hodoscope"));
46          options.addOption(new Option("N", true, "Number of required positron hits at Ecal"));
47          options.addOption(new Option("E", true, "Lower energy threshold for hits at Ecal"));
48          options.addOption(new Option("t", true, "Lower energy threshold for sum of energy of all hits at top or bot of Ecal"));
49          options.addOption(new Option("e", true, "Interval between non-empty events"));
50          
51          return options;
52      }
53  
54      public static void main(String[] args) {
55          // Set up command line parsing.
56          Options options = createCommandLineOptions();
57          CommandLineParser parser = new PosixParser();
58  
59          // Parse command line arguments.
60          CommandLine cl = null;
61          try {
62              cl = parser.parse(options, args);
63          } catch (ParseException e) {
64              throw new RuntimeException("Problem parsing command line options.", e);
65          }
66  
67          String[] parsedArgs = cl.getArgs();
68  
69          if (parsedArgs.length < 2) {
70              System.out.println("FilterMCBunches <input files> <output file>");
71              HelpFormatter help = new HelpFormatter();
72              help.printHelp(" ", options);
73              System.exit(1);
74          }
75  
76          String outFileName = parsedArgs[parsedArgs.length - 1];
77  
78          LCIOWriter writer = null;
79          try {
80              writer = new LCIOWriter(new File(outFileName));
81          } catch (IOException e) {
82              throw new RuntimeException(e);
83          }
84          
85          int numHitHodo = 0;
86          if (cl.hasOption("M")) numHitHodo = Integer.valueOf(cl.getOptionValue("M"));
87          
88          int numPositronHitEcal = 0;
89          if (cl.hasOption("N")) numPositronHitEcal = Integer.valueOf(cl.getOptionValue("N"));
90          
91          double eCutHit = 0;
92          if (cl.hasOption("E")) eCutHit = Double.valueOf(cl.getOptionValue("E"));
93          
94          double eCutTotal = 0;
95          if (cl.hasOption("t")) eCutTotal = Double.valueOf(cl.getOptionValue("t"));
96  
97          int nEvents = -1;
98          if (cl.hasOption("n")) nEvents = Integer.valueOf(cl.getOptionValue("n"));
99          
100         int nEventsToWrite = -1;
101         if (cl.hasOption("w")) nEventsToWrite = Integer.valueOf(cl.getOptionValue("w"));
102         
103         int nEmpty = 0;
104         if (cl.hasOption("e")) nEmpty = Integer.valueOf(cl.getOptionValue("e"));
105         
106         int readEvents = 0;
107         int writtenEvents = 0;
108 
109         String detectorName = null;
110         
111         ExtractEventsWithHitAtHodoEcal extractor = new ExtractEventsWithHitAtHodoEcal();
112 
113         fileLoop: for (int fileNumber = 0; fileNumber < parsedArgs.length - 1; fileNumber++) {
114             LCIOReader reader = null;
115             String inFileName = parsedArgs[fileNumber];
116             try {
117                 reader = new LCIOReader(new File(inFileName));
118                 System.out.println("Opened input file " + inFileName);
119             } catch (IOException e) {
120                 throw new RuntimeException(e);
121             }
122 
123             eventLoop: while (true) {
124                 if (nEvents != -1 && readEvents == nEvents) {
125                     break fileLoop;
126                 }
127                 EventHeader event;
128                 try {
129                     event = reader.read();
130                 } catch (IOException e) {
131                     break eventLoop;
132                 }
133                 readEvents++;
134 
135                 if (detectorName == null) {
136                     detectorName = event.getDetectorName();
137                 }
138 
139                 if (extractor.goodEvent(event, numHitHodo, numPositronHitEcal, eCutHit, eCutTotal)) {
140                     writtenEvents++;
141                     try {
142                         writer.write(event);
143                     } catch (IOException e) {
144                         throw new RuntimeException(e);
145                     }
146                     
147                     for (int i = 1; i < nEmpty; i++) {
148                         try {
149                             writer.write(new BaseLCSimEvent(event.getRunNumber(), event.getEventNumber(), detectorName));
150                         } catch (IOException e) {
151                             throw new RuntimeException(e);
152                         }
153                     }
154                     
155                     if (nEventsToWrite != -1 && writtenEvents == nEventsToWrite) {
156                         break fileLoop;
157                     }
158                 }
159             }
160             try {
161                 reader.close();
162             } catch (IOException e) {
163                 throw new RuntimeException(e);
164             }
165         }
166 
167         System.out.format("Read %d events, wrote %d of them\n", readEvents, writtenEvents);
168 
169         try {
170             writer.close();
171         } catch (IOException e) {
172             throw new RuntimeException(e);
173         }
174     }

176     /**
177      * An event is regarded as good if there is positron hits at Ecal
178      * @param event: LCIO event
179      * @param numHitHodo: # of required hits at hodo
180      * @param numPositronHitEcal: # of required positron hits at Ecal
181      * @param eCutHit: Lower energy threshold for hits at Ecal
182      * @param eCutTotal: Lower energy threshold for sum of energy of all hits at top or bot of Ecal
183      * @return Return true if an event is good
184      */
185     public boolean goodEvent(EventHeader event, int numHitHodo, int numPositronHitEcal, double eCutHit, double eCutTotal) {        
186         if(numHitHodo > 0) {
187             List<SimTrackerHit> hodoHits = event.get(SimTrackerHit.class, "HodoscopeHits");
188             if(hodoHits.size() < numHitHodo) return false;
189         }
190         
191         List<SimCalorimeterHit> ecalHits = event.get(SimCalorimeterHit.class, "EcalHits");
192         
193         if(eCutTotal > 0) {
194             double topE = 0, botE = 0;
195             for (SimCalorimeterHit hit : ecalHits) {
196                 if (hit.getIdentifierFieldValue("iy") > 0) {
197                     topE += hit.getRawEnergy();
198                 } else {
199                     botE += hit.getRawEnergy();
200                 }
201             }
202             if(topE < eCutTotal || botE < eCutTotal) return false;
203         }
204         
205         int numPosHit = 0;
206         if(numPositronHitEcal > 0) {
207             for (SimCalorimeterHit simHit : ecalHits) {            
208                 if(simHit.getRawEnergy() >= eCutHit) {
209                     for(int i = 0; i < simHit.getMCParticleCount(); i++) {
210                         MCParticle particle = simHit.getMCParticle(i);
211                         if(particle.getPDGID() == -11) numPosHit++;
212                     }            
213                 }
214             }
215             if (numPosHit < numPositronHitEcal) return false;
216         }
217         
218         return true;
219     }
