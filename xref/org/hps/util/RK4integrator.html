1   package org.hps.util;
2   
3   import hep.physics.vec.BasicHep3Vector;
4   import hep.physics.vec.Hep3Vector;
5   import hep.physics.vec.VecOp;
6   
7   import org.lcsim.geometry.FieldMap;
8   import static org.lcsim.constants.Constants.fieldConversion;
9   
10  public class RK4integrator {
11      private double h;
12      private double h2;
13      private double alpha;
14      private FieldMap fM;
15      
16      public RK4integrator(double Q, double dx, FieldMap fM) {
17          alpha = Q * fieldConversion; // Q is the charge in units of the proton charge
18          h = dx; // Step size in mm
19          h2 = h / 2.0;
20          this.fM = fM; // Magnetic field map
21      }
22      
23      public void setQ(int input) {
24          setAlpha(input * fieldConversion);
25      }
26      
27      public void setAlpha(double input) {
28          alpha = input;
29      }
30      
31      public void setFieldmap(FieldMap input) {
32          fM = input;
33      }
34      
35      public void setStepSize(double input) {
36          h = input;
37          h2 = h / 2.0;
38      }
39      
40      public Hep3Vector integrationPosition(Hep3Vector r0, Hep3Vector p0, double s) {
41          return integrate(r0, p0, s).getFirstElement();
42      }
43      
44      public Hep3Vector integrationMomentum(Hep3Vector r0, Hep3Vector p0, double s) {
45          return integrate(r0, p0, s).getFirstElement();
46      }
47      
48      public Pair<Hep3Vector, Hep3Vector> integrate(Hep3Vector r0, Hep3Vector p0, double s) {
49          // r0 is the initial point in mm
50          // p0 is the initial momentum in GeV/c
51          // s is the distance to propagate (approximate to distance dx)
52          Hep3Vector r = new BasicHep3Vector(r0.v());
53          Hep3Vector p = new BasicHep3Vector(p0.v());
54  
55          int nStep = (int) (s / h) + 1;
56          for (int step = 0; step < nStep; step++) {
57  
58              Pair<Hep3Vector, Hep3Vector> k1 = f(r, p);
59              Hep3Vector r1 = VecOp.add(r, VecOp.mult(h2, k1.getFirstElement()));
60              Hep3Vector p1 = VecOp.add(p, VecOp.mult(h2, k1.getSecondElement()));
61              Pair<Hep3Vector, Hep3Vector> k2 = f(r1, p1);
62              Hep3Vector r2 = VecOp.add(r, VecOp.mult(h2, k2.getFirstElement()));;
63              Hep3Vector p2 = VecOp.add(p, VecOp.mult(h2, k2.getSecondElement()));
64              Pair<Hep3Vector, Hep3Vector> k3 = f(r2, p2);
65              Hep3Vector r3 = VecOp.add(r, VecOp.mult(h, k3.getFirstElement()));;
66              Hep3Vector p3 = VecOp.add(p, VecOp.mult(h, k3.getSecondElement()));
67              Pair<Hep3Vector, Hep3Vector> k4 = f(r3, p3);
68              
69              Hep3Vector rtemp = VecOp.add(VecOp.mult(1.0/6.0, k1.getFirstElement()), VecOp.mult(1.0/3.0, k2.getFirstElement()));
70              rtemp = VecOp.add(rtemp, VecOp.mult(1.0/3.0, k3.getFirstElement()));
71              rtemp = VecOp.add(rtemp, VecOp.mult(1.0/6.0, k4.getFirstElement()));
72              rtemp = VecOp.mult(h, rtemp);
73              r = VecOp.add(r, rtemp);
74              
75              Hep3Vector ptemp = VecOp.add(VecOp.mult(1.0/6.0, k1.getSecondElement()), VecOp.mult(1.0/3.0, k2.getSecondElement()));
76              ptemp = VecOp.add(ptemp, VecOp.mult(1.0/3.0, k3.getSecondElement()));
77              ptemp = VecOp.add(ptemp, VecOp.mult(1.0/6.0, k4.getSecondElement()));
78              ptemp = VecOp.mult(h, ptemp);
79              p = VecOp.add(p, ptemp);
80  
81          }
82  
83          return new Pair<Hep3Vector, Hep3Vector> (r,p);
84      }
85      
86      private Pair<Hep3Vector, Hep3Vector> f(Hep3Vector x, Hep3Vector p) { // Return all the derivatives
87  
88          Hep3Vector pNorm = VecOp.mult(1.0 / p.magnitude(), p);
89          Hep3Vector d = VecOp.mult(alpha, VecOp.cross(pNorm, fM.getField(x)));
90  
91          return new Pair<Hep3Vector, Hep3Vector> (pNorm, d);
92      }
93  
94  }
