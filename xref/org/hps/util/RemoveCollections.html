1   package org.hps.util;
2   
3   import java.util.Arrays;
4   import java.util.HashSet;
5   import java.util.Set;
6   import org.lcsim.event.EventHeader;
7   import org.lcsim.util.Driver;
8   
9   /**
10   * This driver removes collections from an event
11   */
12  public class RemoveCollections extends Driver {
13  
14      protected Set<String> collections = new HashSet<String>();
15  
16      public RemoveCollections() {
17      }
18  
19      public void RemoveCollectionsFromEvent(String[] collectionNames) {
20          this.collections = new HashSet<String>(Arrays.asList(collectionNames));
21      }
22      
23      public void setCollectionNames(String[] collectionNames) {
24          this.collections = new HashSet<String>(Arrays.asList(collectionNames));
25      }
26  
27      protected void process(EventHeader event) {
28          for (String col : collections)
29              if (event.hasItem(col))
30                  event.remove(col);
31      }
32  }
