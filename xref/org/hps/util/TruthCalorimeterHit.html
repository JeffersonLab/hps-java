1   package org.hps.util;
2   
3   import java.util.ArrayList;
4   
5   import org.lcsim.event.CalorimeterHit;
6   import org.lcsim.event.EventHeader.LCMetaData;
7   import org.lcsim.event.base.BaseSimCalorimeterHit;
8   
9   import hep.physics.vec.Hep3Vector;
10  
11  public class TruthCalorimeterHit extends BaseSimCalorimeterHit {
12      public TruthCalorimeterHit(double rawEnergy, double correctedEnergy, double energyError, double time, long id,
13              Hep3Vector positionVec, int type, Object[] mcparts, float[] energies, float[] times, int[] pdgs, LCMetaData metaData) {
14          // Instantiate the regular calorimeter hit fields.
15          this.rawEnergy = rawEnergy;
16          this.correctedEnergy = correctedEnergy;
17          this.energyError = energyError;
18          this.time = time;
19          this.id = id;
20          this.positionVec = positionVec;
21          this.type = type;
22          
23          // Set the metadata.
24          setMetaData(metaData);
25          
26          // Store the truth information.
27          this.nContributions = mcparts.length;
28          this.particle = mcparts;
29          this.energyContrib = energies;
30          this.times = times;
31          this.pdg = pdgs;
32          this.steps = new ArrayList<float[]>(nContributions);
33      }
34      
35      public TruthCalorimeterHit(CalorimeterHit baseHit, Object[] mcparts, float[] energies, float[] times, int[] pdgs) {
36          this(baseHit.getRawEnergy(), baseHit.getCorrectedEnergy(), baseHit.getEnergyError(), baseHit.getTime(),
37                  baseHit.getCellID(), baseHit.getPositionVec(), baseHit.getType(), mcparts, energies, times,
38                  pdgs, baseHit.getMetaData());
39      }
40      
41      public void setTime(float newTime) {
42          time = newTime;
43      }
44      
45      @Override
46      public double getTime() {
47          // SimCalorimeterHit usually calculates time based on the
48          // input particles. We would prefer that it simply return the
49          // given time, as the time calculation is performed in a
50          // different manner here.
51          return time;
52      }
53  }
