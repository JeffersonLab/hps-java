1   package org.hps.util;
2   
3   import org.lcsim.event.EventHeader;
4   import org.lcsim.event.Track;
5   
6   public class TwoTrackFilterDriver extends LCIOFilterDriver {
7   
8       private String trackCollectionName = "MatchedTracks";
9   
10      public TwoTrackFilterDriver() {
11      }
12  
13      public void setTrackCollectionNamePath(String trackCollection) {
14          this.trackCollectionName = trackCollection;
15      }
16  
17      @Override
18      boolean eventFilter(EventHeader event) {
19          boolean pass = false;
20  
21          if (!event.hasCollection(Track.class, trackCollectionName)) {
22              throw new RuntimeException("Error, event doesn't have the track collection");
23          }
24  
25          if (this.debug) {
26              System.out.printf("%s: %d tracks in event %d\n", this.getClass().getSimpleName(), event.get(Track.class, trackCollectionName).size(), event.getEventNumber());
27          }
28  
29          if (event.get(Track.class, trackCollectionName).size() > 1) {
30              if (this.debug) {
31                  System.out.printf("%s: write event %d \n", this.getClass().getSimpleName(), event.getEventNumber());
32              }
33              pass = true;
34          }
35  
36  
37          return pass;
38      }
39  }
