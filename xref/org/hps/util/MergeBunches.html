75      public void process(EventHeader event) {
76  
77          if (bunchCounter == 0)
78              newEvent = new BaseLCSimEvent(event.getRunNumber(), eventCounter, event.getDetectorName());
79          mergeEvents(newEvent, event, offsetT);
80          bunchCounter++;
81          offsetT += deltaT;
82          if (bunchCounter == nBunches) {
83              System.out.println("Writing events #" + eventCounter);
84              eventCounter++;
85              offsetT = startT;
86              bunchCounter = 0;
87              try {
88                  writer.write(newEvent);
89              } catch (IOException ex) {
90                  Logger.getLogger(MergeBunches.class.getName()).log(Level.SEVERE, null, ex);
91              }
92          }
93      }

95      /**
96       * Merges all collections from the given events and applies a time offset to all entries in all collections of the
97       * overlay event.
98       * 
99       * @param event the event where everything is merged into
100      * @param overlayEvent the event overlaid
101      * @param overlayTime the time offset for the overlay event
102      */
103     protected void mergeEvents(EventHeader event, EventHeader overlayEvent, double overlayTime) {
104 
105         // need to copy list of collections to avoid concurrent modification
106         List<LCMetaData> overlayCollections = new ArrayList<LCMetaData>(overlayEvent.getMetaData());
107         for (LCMetaData overlayCollection : overlayCollections) {
108             String overlayCollectionName = overlayCollection.getName();
109             if (event.hasItem(overlayCollectionName)) {
110                 this.mergeCollections(event.getMetaData((List) event.get(overlayCollectionName)), overlayCollection,
111                         overlayTime);
112             } else {
113                 // event does not contain corresponding collection from overlayEvent, just put it there
114                 // First move hits and apply timing cuts
115                 List collection = this.moveCollectionToTime(overlayCollection, overlayTime);
116                 this.putCollection(overlayCollection, (List) overlayEvent.get(overlayCollectionName), event);
117             }
118         }
119     }

121     /**
122      * Shifts an event in time. Moves all entries in all collections in the event by the given offset in time.
123      * 
124      * @param event the event to move in time
125      * @param time the time shift applied to all entries in all collections
126      */
127     protected void moveEventToTime(EventHeader event, double time) {
128         // need to copy list of collections to avoid concurrent modification
129         List<LCMetaData> collections = new ArrayList<LCMetaData>(event.getMetaData());
130         for (LCMetaData collection : collections) {
131             List movedCollection = this.moveCollectionToTime(collection, time);
132             if (movedCollection != null) {
133                 // replace the original collection
134                 event.remove(collection.getName());
135                 this.putCollection(collection, movedCollection, event);
136             }
137         }
138     }

140     /**
141      * Shifts a collection in time. Moves all entries in the collection by the given offset in time. If a readout time
142      * is set for the given collection, all entries outside of that window will be removed.
143      * 
144      * @param collection the collection to move in time
145      * @param time the time shift applied to all entries in the collection
146      * @return returns the list of moved entries
147      */
148     protected List moveCollectionToTime(LCMetaData collection, double time) {
149         EventHeader event = collection.getEvent();
150         String collectionName = collection.getName();
151         Class collectionType = collection.getType();
152         int flags = collection.getFlags();
153         if (this.getHistogramLevel() > HLEVEL_NORMAL)
154             System.out.println("Moving collection: " + collectionName + " of type " + collectionType + " to " + time
155                     + "ns");
156 
157         double timeWindow = 0;
158 
159         List movedCollection;
160         if (collectionType.isAssignableFrom(MCParticle.class)) {
161             // MCParticles
162             // don't create new list, just move existing particles
163             movedCollection = event.get(MCParticle.class, collectionName);
164             for (MCParticle mcP : (List<MCParticle>) movedCollection) {
165                 ((SIOMCParticle) mcP).setTime(mcP.getProductionTime() + time);
166             }
167         } else if (collectionType.isAssignableFrom(SimTrackerHit.class)) {
168             // SimTrackerHits
169             movedCollection = new ArrayList<SimTrackerHit>();
170             for (SimTrackerHit hit : (List<SimTrackerHit>) event.get(SimTrackerHit.class, collectionName)) {
171                 ((BaseSimTrackerHit) hit).setTime(hit.getTime() + time);
172                 movedCollection.add(hit);
173             }
174         } else if (collectionType.isAssignableFrom(SimCalorimeterHit.class)) {
175             // SimCalorimeterHits
176             movedCollection = new ArrayList<SimCalorimeterHit>();
177             // check if hit contains PDGIDs
178             boolean hasPDG = LCIOUtil.bitTest(flags, LCIOConstants.CHBIT_PDG);
179             List<SimCalorimeterHit> hits = event.get(SimCalorimeterHit.class, collectionName);
180             int nSimCaloHits = hits.size();
181             int nHitsMoved = 0;
182             for (SimCalorimeterHit hit : event.get(SimCalorimeterHit.class, collectionName)) {
183                 // check if earliest energy deposit is later than relevant time window
184 
185                 BaseSimCalorimeterHit movedHit = null;
186                 nHitsMoved++;
187                 if (this.getHistogramLevel() > HLEVEL_HIGH && nHitsMoved % 100 == 0)
188                     System.out.print("Moved " + nHitsMoved + " / " + nSimCaloHits + " hits\n");
189                 movedHit = (BaseSimCalorimeterHit) hit;
190                 movedHit.shiftTime(time);// adds this amount of time to all the times of SimCalorimeterHits
191                 movedCollection.add(movedHit);
192             }
193         } else if (collectionType.isAssignableFrom(GenericObject.class)) {
194             // nothing to do for GenericObjects
195             return event.get(GenericObject.class, collectionName);
196         } else {
197             System.err.println("Unable to move collection: " + collectionName + " of type " + collectionType);
198             return null;
199         }
200         return movedCollection;
201     }

212         String[] readout = collection.getStringParameters().get("READOUT_NAME");
213         if (readout != null) {

215         } else {
216             event.put(collection.getName(), entries, collection.getType(), collection.getFlags());

218         if (this.getHistogramLevel() > HLEVEL_NORMAL)
219             System.out.println("Putting collection " + collection.getName() + " into event.");

222     /**
223      * Merges two collections and applies a time offset to all entries in the overlay collection.
224      * 
225      * @param collection the collection where the overlay collection is merged into
226      * @param overlayCollection the collection overlaid
227      * @param overlayTime the time offset for the overlay collection
228      * @return returns <c>false</c> if unable to merge collections, otherwise <c>true</c>
229      */
230     protected boolean mergeCollections(LCMetaData collection, LCMetaData overlayCollection, double overlayTime) {
231         String collectionName = collection.getName();
232         Class collectionType = collection.getType();
233         Class overlayCollectionType = overlayCollection.getType();
234         if (this.getHistogramLevel() > HLEVEL_NORMAL)
235             System.out.println("Merging collection: " + collectionName + " of type " + collectionType + ".");
236         if (!collectionType.equals(overlayCollectionType)) {
237             System.err.println("Can not merge collections: " + collectionName + " of type " + collectionType + " and "
238                     + overlayCollectionType);
239             return false;
240         }
241 
242         // move the overlay hits in time, signal should have been moved already
243         List overlayEntries = this.moveCollectionToTime(overlayCollection, overlayTime);
244         // List overlayEntries = overlayCollection.getEvent().get(overlayCollectionType, overlayCollection.getName());
245         // Check if there are actually entries to overlay
246         if (overlayEntries.isEmpty())
247             return true;
248         EventHeader event = collection.getEvent();
249 
250         if (collectionType.isAssignableFrom(MCParticle.class)) {
251             // Nothing to do. Only add mc particles that are connected to something kept in the event.
252             // This is done in the other steps below.
253             if (!collectionName.equals(event.MC_PARTICLES)) {
254                 event.get(MCParticle.class, collectionName).addAll(overlayEntries);
255             }
256 
257         } else if (collectionType.isAssignableFrom(SimTrackerHit.class)) {
258             // SimTrackerHits: just append all hits from overlayEvents
259             List<SimTrackerHit> signalTrackerHits = event.get(SimTrackerHit.class, collectionName);
260 
261             // add contributing mc particles to lists
262             for (SimTrackerHit hit : (List<SimTrackerHit>) overlayEntries) {
263                 SimTrackerHit overlayHit = copySimTrackerHit(hit, collection);
264                 signalTrackerHits.add(overlayHit);
265             }
266             // System.out.println("Overlaid TrackerHits!");
267         } else if (collectionType.isAssignableFrom(SimCalorimeterHit.class)) {
268             // SimCalorimeterHits: need to merge hits in cells which are hit in both events
269             // check if map has already been filled
270             Map<Long, SimCalorimeterHit> hitMap;
271             List<SimCalorimeterHit> signalCaloHits = event.get(SimCalorimeterHit.class, collectionName);
272             if (!caloHitMap.containsKey(collectionName)) {
273                 // build map of cells which are hit in signalEvent
274                 hitMap = new HashMap<Long, SimCalorimeterHit>();
275                 for (SimCalorimeterHit hit : signalCaloHits) {
276                     hitMap.put(hit.getCellID(), hit);
277                 }
278                 caloHitMap.put(collectionName, hitMap);
279             } else {
280                 hitMap = caloHitMap.get(collectionName);
281             }
282 
283             boolean hasPDG = LCIOUtil.bitTest(collection.getFlags(), LCIOConstants.CHBIT_PDG);
284             // loop over the hits from the overlay event
285             int nHitsMerged = 0;
286             int nSimCaloHits = overlayEntries.size();
287             for (SimCalorimeterHit hit : (List<SimCalorimeterHit>) overlayEntries) {
288                 long cellID = hit.getCellID();
289 
290                 nHitsMerged++;
291                 if (this.getHistogramLevel() > HLEVEL_HIGH && nHitsMerged % 100 == 0)
292                     System.out.print("Merged " + nHitsMerged + " / " + nSimCaloHits + " hits\n");
293                 if (hitMap.containsKey(cellID)) {
294                     SimCalorimeterHit oldHit = hitMap.get(hit.getCellID());
295                     int nHitMcP = oldHit.getMCParticleCount();
296                     int nOverlayMcP = hit.getMCParticleCount();
297                     int nMcP = nHitMcP + nOverlayMcP;
298                     // arrays of mc particle contributions to the hit
299                     Object[] mcpList = new Object[nMcP];
300                     float[] eneList = new float[nMcP];
301                     float[] timeList = new float[nMcP];
302                     int[] pdgList = null;
303                     if (hasPDG)
304                         pdgList = new int[nMcP];
305                     double rawEnergy = 0.;
306                     // fill arrays with values from hit
307                     for (int i = 0; i != nHitMcP; i++) {
308                         mcpList[i] = oldHit.getMCParticle(i);
309                         eneList[i] = (float) oldHit.getContributedEnergy(i);
310                         timeList[i] = (float) oldHit.getContributedTime(i);
311                         if (hasPDG)
312                             pdgList[i] = oldHit.getPDG(i);
313                         rawEnergy += eneList[i];
314                     }
315                     // add values of overlay hit
316                     for (int i = 0; i != nOverlayMcP; i++) {
317                         int j = nHitMcP + i;
318                         MCParticle hitMC = hit.getMCParticle(i);
319                         if (hitMC != null) {
320                             if (!mcParticleReferences.containsKey(hitMC)) {
321                                 this.addOverlayMcParticle(hitMC);
322                             }
323                             mcpList[j] = mcParticleReferences.get(hitMC);
324                         }
325                         eneList[j] = (float) hit.getContributedEnergy(i);
326                         timeList[j] = (float) hit.getContributedTime(i);
327                         if (hasPDG)
328                             pdgList[j] = hit.getPDG(i);
329                         rawEnergy += eneList[j];
330                     }
331                     // need to set time to 0 so it is recalculated from the timeList
332                     SimCalorimeterHit mergedHit = new BaseSimCalorimeterHit(oldHit.getCellID(), rawEnergy, 0., mcpList,
333                             eneList, timeList, pdgList, collection);
334                     // replace old hit with merged hit
335                     signalCaloHits.remove(oldHit);
336                     signalCaloHits.add(mergedHit);
337                     hitMap.put(cellID, mergedHit);
338                 } else {
339                     SimCalorimeterHit overlayHit = copySimCalorimeterHit(hit, collection, hasPDG);
340                     signalCaloHits.add(overlayHit);
341                     hitMap.put(cellID, overlayHit);
342                 }
343 
344             }
345         } else if (collectionType.isAssignableFrom(GenericObject.class)) {
346             // need to implement all kinds of possible GenericObjects separately
347             if (collectionName.equals("MCParticleEndPointEnergy")) {
348                 // TODO decide what to do with this collection in the overlay events
349                 // TODO would need to resolve the position of kept mc particles and keep the same position here
350                 // event.get(GenericObject.class, collectionName).addAll(overlayEntries);
351                 // event.remove("MCParticleEndPointEnergy");
352             } else {
353                 System.err.println("Can not merge collection " + collectionName + " of type " + collectionType
354                         + ". Unhandled type.");
355                 return false;
356             }
357         }
358         return true;
359     }

361     /**
362      * Deep copy of an SimCalorimeterHit. Necessary in order to be able to close an overlay event.
363      * 
364      * @param hit The hit to be copied
365      * @param meta The meta data that will be attached to the hit
366      * @param hasPDG Flag if the pdg code of the mc contriutions should be saved
367      * @return The copied SimCalorimeterHit
368      */
369     protected SimCalorimeterHit copySimCalorimeterHit(SimCalorimeterHit hit, LCMetaData meta, boolean hasPDG) {
370         long id = hit.getCellID();
371         double rawEnergy = hit.getRawEnergy();
372         double time = hit.getTime();
373         int nMCP = hit.getMCParticleCount();
374         Object[] mcparts = new Object[nMCP];
375         float[] energies = new float[nMCP];
376         float[] times = new float[nMCP];
377         int[] pdgs = null;
378         if (hasPDG)
379             pdgs = new int[nMCP];
380         // fill arrays with values from hit
381         for (int i = 0; i != nMCP; i++) {
382             MCParticle hitMC = hit.getMCParticle(i);
383             if (hitMC != null) {
384                 this.addOverlayMcParticle(hitMC);
385                 mcparts[i] = mcParticleReferences.get(hitMC);
386             }
387             energies[i] = (float) hit.getContributedEnergy(i);
388             times[i] = (float) hit.getContributedTime(i);
389             if (hasPDG)
390                 pdgs[i] = hit.getPDG(i);
391         }
392 
393         BaseSimCalorimeterHit copyHit = new BaseSimCalorimeterHit(id, rawEnergy, time, mcparts, energies, times, pdgs,
394                 meta);
395 
396         return copyHit;
397     }

399     /**
400      * Deep copy of an SimTrackerHit. Necessary in order to be able to close an overlay event.
401      * 
402      * @param hit The hit to be copied
403      * @param meta The meta data that will be attached to the hit
404      * @return The copied SimTrackerHit
405      */
406     protected SimTrackerHit copySimTrackerHit(SimTrackerHit hit, LCMetaData meta) {
407 
408         double[] position = new double[3];
409         double[] momentum = new double[3];
410         double[] hitp = hit.getPosition();
411         double[] hitm = hit.getMomentum();
412         for (int i = 0; i != 3; i++) {
413             position[i] = hitp[i];
414             momentum[i] = hitm[i];
415         }
416         double dEdx = hit.getdEdx();
417         double pathLength = hit.getPathLength();
418         double time = hit.getTime();
419         int cellID = hit.getCellID();
420         MCParticle hitMC = hit.getMCParticle();
421         MCParticle mcParticle = null;
422         if (hitMC != null) {
423             this.addOverlayMcParticle(hitMC);
424             mcParticle = mcParticleReferences.get(hitMC);
425         }
426         IDetectorElement de = null;
427 
428         return new BaseSimTrackerHit(position, dEdx, momentum, pathLength, time, cellID, mcParticle, meta, de);
429     }

431     /**
432      * Deep copy of an mc particle. Necessary in order to be able to close an overlay event. The parent and daught
433      * relations are <b>not</b> set for the copied mc particle. Because those should most likely also point to copies
434      * this should be handled somewhere else.
435      * 
436      * @param mcParticle The mc particle to be copied
437      * @return the copied mc particle
438      */
439     static public MCParticle copyMcParticle(MCParticle mcParticle) {
440         Hep3Vector origin = new BasicHep3Vector(mcParticle.getOriginX(), mcParticle.getOriginY(),
441                 mcParticle.getOriginZ());
442         HepLorentzVector p = new BasicHepLorentzVector(mcParticle.getEnergy(), new double[] {mcParticle.getPX(),
443                 mcParticle.getPY(), mcParticle.getPZ()});
444         ParticleType ptype = mcParticle.getType().getParticlePropertyProvider().get(mcParticle.getPDGID());
445         int status = mcParticle.getGeneratorStatus();
446         double time = mcParticle.getProductionTime();
447         BaseMCParticle copyMcP = new BaseMCParticle(origin, p, ptype, status, time);
448         // override the mass and charge from the particle type to prevent unknown particle exceptions
449         copyMcP.setMass(mcParticle.getMass());
450         copyMcP.setCharge(mcParticle.getCharge());
451         copyMcP.setSimulatorStatus(mcParticle.getSimulatorStatus().getValue());
452         return copyMcP;
453     }

455     /**
456      * Copies an mc particle and stores it together with the copy in a map. Adds it to the list of mc particles as well
457      * as the overlay mc particles. Also copies and keeps all ancestors.
458      * 
459      * @param particle
460      */
461     protected void addOverlayMcParticle(MCParticle particle) {
462         if (!mcParticleReferences.containsKey(particle)) {
463             // keep a copy of the mc particle instead of the original in order to close the background event
464             MCParticle mcp = copyMcParticle(particle);
465             mcParticleReferences.put(particle, mcp);
466             overlayMcParticles.add(mcp);
467             allMcParticles.add(mcp);
468             List<MCParticle> parents = particle.getParents();
469             // keep the parents as well and set the parent daughter relations
470             for (MCParticle parent : parents) {
471                 this.addOverlayMcParticle(parent);
472                 ((BaseMCParticle) mcParticleReferences.get(parent)).addDaughter(mcp);
473             }
474         }
