14  /**
15   * Get information about available HPS detector resources.
16   */
17  public class DetectorDataResources {
18      
19      private DetectorDataResources() {        
20      }
21      
22      /**
23       * Find a list of available detector names.
24       * <p>
25       * Only those detectors that have names starting with "HPS" in their <code>detector.properties</code> files will be
26       * returned.
27       *
28       * @return the list of available HPS detector names
29       */
30      public static Set<String> getDetectorNames() {
31          final ClassLoader classLoader = DetectorDataResources.class.getClassLoader();
32          final Set<String> detectorNames = new TreeSet<String>();
33          final URL url = DetectorDataResources.class.getResource("DetectorDataResources.class");
34          final String protocol = url.getProtocol();
35          if (!"jar".equals(protocol)) {
36              throw new RuntimeException("Unsupported URL protocol: " + url.getProtocol());
37          }
38          try {
39              final JarURLConnection con = (JarURLConnection) url.openConnection();
40              final JarFile archive = con.getJarFile();
41              final Enumeration<JarEntry> entries = archive.entries();
42              while (entries.hasMoreElements()) {
43                  final JarEntry entry = entries.nextElement();
44                  if (entry.getName().endsWith("detector.properties")) {
45                      final InputStream inputStream = classLoader.getResourceAsStream(entry.getName());
46                      if (inputStream == null) {
47                          throw new RuntimeException("Failed to load jar entry: " + entry.getName());
48                      }
49                      final Properties properties = new Properties();
50                      properties.load(inputStream);
51                      final String detectorName = properties.getProperty("name");
52                      detectorNames.add(detectorName);
53                  }
54              }
55              archive.close();
56          } catch (final IOException e) {
57              throw new RuntimeException(e);
58          }
59          return detectorNames;
60      }   
61  }
