26  /**
27   * A rewrite of the LCSim detector converter that supports setting of the run number for conditions system
28   * initialization so that database conditions are available.
29   */
30  public class DetectorConverter {
31  
32      private static Options OPTIONS = new Options();
33      static {
34          OPTIONS.addOption(new Option("i", "input-file", true, "input compact.xml file"));
35          OPTIONS.getOption("i").setRequired(true);
36          OPTIONS.addOption(new Option("o", "ouput-file", true, "output file"));
37          OPTIONS.getOption("o").setRequired(true);
38          OPTIONS.addOption(new Option("f", "format", true, "output format (lcdd, etc.)"));
39          OPTIONS.addOption(new Option("r", "run-number", true, "run number for conditions initialization"));
40      }
41  
42      private DefaultParser parser = new DefaultParser();
43      private File inputFile = null;
44      private File outputFile = null;
45      private String outputFormat = null;
46      private Integer runNumber = null;
47  
48      public static void main(String args[]) throws Exception {
49          DetectorConverter cnv = new DetectorConverter();
50          cnv.run(args);
51      }
52  
53      private static List<Converter> getConverters() {
54          Iterator<Converter> iter = getServices(Converter.class);
55          List<Converter> result = new ArrayList<Converter>();
56          while (iter.hasNext())
57              result.add(iter.next());
58          return result;
59      }
60  
61      private static <T> Iterator<T> getServices(Class<T> providerClass) {
62          return ServiceLoader.load(providerClass, DetectorConverter.class.getClassLoader()).iterator();
63      }
64  
65      private void run(String args[]) throws Exception {
66          CommandLine cl = parser.parse(OPTIONS, args);
67          if (!cl.hasOption("i")) {
68              throw new RuntimeException("Missing required -i arg.");
69          }
70          if (!cl.hasOption("o")) {
71              throw new RuntimeException("Missing required -o arg.");
72          }
73          inputFile = new File(cl.getOptionValue("i"));
74          outputFile = new File(cl.getOptionValue("o"));
75          if (cl.hasOption("f")) {
76              outputFormat = cl.getOptionValue("f");
77          }
78          if (cl.hasOption("r")) {
79              runNumber = Integer.parseInt(cl.getOptionValue("r"));
80          } 
81  
82          Converter cnv = null;
83  
84          if (outputFormat != null) {
85              for (Converter c : getConverters()) {
86                  if (c.getOutputFormat().equalsIgnoreCase(outputFormat)) {
87                      cnv = c;
88                      break;
89                  }
90              }
91          } else {
92              for (Converter c : getConverters()) {
93                  if (c.getFileFilter().accept(outputFile)) {
94                      cnv = c;
95                      break;
96                  }
97              }
98          }
99          if (cnv == null)
100             throw new IllegalArgumentException("No converter found for format: " + outputFormat);
101 
102         if (runNumber != null) {
103             DatabaseConditionsManager mgr = DatabaseConditionsManager.getInstance();
104             String name = "DUMMY";
105             ConditionsReader dummyReader = ConditionsReader.createDummy();
106             ((ConditionsManagerImplementation) mgr).setConditionsReader(dummyReader, name);
107             DummyDetector detector = new DummyDetector(name);
108             mgr.registerConditionsConverter(new DummyConditionsConverter(detector));
109             mgr.setDetector(name, runNumber);
110         }
111 
112         InputStream in = new BufferedInputStream(new FileInputStream(inputFile));
113         OutputStream out = new BufferedOutputStream(new FileOutputStream(outputFile));
114         cnv.convert(inputFile.getCanonicalPath(), in, out);
115         in.close();
116         out.close();
117     }
118 }
