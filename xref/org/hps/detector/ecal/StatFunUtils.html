8       /**
9        * Function for double to round up the arg. value to 'places' digits after the semicolon 
10       * @param value double number to round
11       * @param places round up to places
12       * @return rounded value 
13       */
14      public static double round(double value, int places) {
15          if (places < 0 || places > 16) {
16              throw new IllegalArgumentException();
17          }
18          long factor = (long) Math.pow(10, places);        
19          value = value * factor;
20          long tmp = Math.round(value); 
21          return (double) tmp / factor;
22      }
