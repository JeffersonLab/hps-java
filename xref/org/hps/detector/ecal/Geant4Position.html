3   /**
4    * Calculate G4 position for a crystal, by it's front and back face coordinates calculates the position for the center
5    * of the crystal calculates the rotation of the crystal in convention of Tait-Bryan angles, {phi, theta, psi}
6    * phi-rotation around X theta-rotation around Y psi-rotation around Z
7    */
8   public final class Geant4Position {
9   
10      /* center coordinates */
11      private GVector centerPoint;
12      /* crystal vector formed by front face center point and back face center point */
13      private GVector crysvec;
14      /* { phi,theta,psi=0}; Tait-Bryan angles to define crystal orientation in the space */
15      private double[] taitbriangles;
16  
17      /* crystal center coordinates as a double[3] array */
18      private double[] centerArr;
19  
20      /**
21       * constructor
22       * 
23       * @param point1 (x,y,z) of crystal front face center point as GVector
24       * @param point2 (x,y,z) of crystal back face center point as GVector
25       */
26      public Geant4Position(GVector point1, GVector point2) {
27          centerPoint = point2.Add(point1).Multiply(0.5);
28          centerArr = new double[3];
29          centerArr[0] = centerPoint.x();
30          centerArr[1] = centerPoint.y();
31          centerArr[2] = centerPoint.z();
32          crysvec = (point2.Substract(centerPoint)).getUnitVector();
33          taitbriangles = new double[3];
34          this.calculateTaitBryanAngles();
35      }
36  
37      /**
38       * @return center coordinates as a GVector
39       */
40      public GVector getCenter() {
41          return this.centerPoint;
42      }
43  
44      /**
45       * @return center coordinates as an array double[x_center, y_center, z_center]
46       */
47      public double[] getCenterArr() {
48          return centerArr;
49      }
50  
51      /**
52       * @return Tait-Bryan angles as an array doubles [phi, theta, psi]
53       */
54      public double[] getTaitBryanAngles() {
55          return this.taitbriangles;
56      }
57  
58      /*
59       * calculates Tait-Bryan angles for the object orientation defined by initial and final state vectors throws
60       * arithmetical exception if the phi angle is calculated wrong or 0
61       */
62      private void calculateTaitBryanAngles() throws ArithmeticException {
63          CrystalTaitBryanAngleCalculator crysTBang = new CrystalTaitBryanAngleCalculator(crysvec);
64          if (crysTBang.getPhi() == Double.NaN) {
65              throw new ArithmeticException("Calculated Tait-Bryan phi angle is NaN.");
66          }
67          this.taitbriangles[0] = -crysTBang.getPhi();
68          this.taitbriangles[1] = -crysTBang.getTheta();
69          this.taitbriangles[2] = -crysTBang.getPsi();
70      }
71  
72      /**
73       * Print crystal center coordinates(X,Y,Z) and Tait-Bryan angles
74       */
75      public void Print() {
76          this.centerPoint.Print();
77      }
78  
79      /**
80       * @return String of Geant4Position object attributes
81       */
82      public String toString() {
83          return ("\"Crys Center coordinates:\\t\"" + this.centerPoint.toString() + "Tait Brian angles phi = "
84                  + Math.toDegrees(this.taitbriangles[0]) + "\t theta = " + Math.toDegrees(this.taitbriangles[1])
85                  + "\t psi = " + Math.toDegrees(this.taitbriangles[2]) + "\n");
86      }
87  
88  }
