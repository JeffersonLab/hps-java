3   import hep.physics.vec.BasicHep3Vector;
4   import hep.physics.vec.Hep3Vector;
5   import hep.physics.vec.VecOp;
6   
7   import org.lcsim.detector.DetectorElement;
8   import org.lcsim.detector.IDetectorElement;
9   import org.lcsim.detector.IGeometryInfo;
10  import org.lcsim.detector.identifier.IIdentifier;
11  import org.lcsim.detector.solids.Trd;
12  
13  /**
14   * This class implements the geometry API for ECal crystals in the HPS experiment. 
15   */
16  public final class EcalCrystal extends DetectorElement {
17      
18      private Hep3Vector positionFront;
19  
20      /**
21       * Class constructor.
22       * @param name The name of the DetectorElement.
23       * @param parent The parent component.
24       * @param path The physical path.
25       * @param id The component's ID.
26       */
27      public EcalCrystal(String name, IDetectorElement parent, String path, IIdentifier id) {
28          super(name, parent, path, id);
29      }
30      
31      /**
32       * Get the X index of this crystal.
33       * @return The X index of this crystal.
34       */
35      public int getX() {
36          return getIdentifierHelper().getValue(getIdentifier(), "ix");
37      }
38      
39      /**
40       * Get the Y index of this crystal.
41       * @return The Y index of this crystal.
42       */
43      public int getY() {
44          return getIdentifierHelper().getValue(getIdentifier(), "iy");
45      }
46                 
47      /**
48       * Get the global center position of the XY plane in the front of the crystal.
49       * This is used in the reconstruction clustering algorithm for determining the 
50       * "corrected" hit position, so it is best to cache it once used, for performance 
51       * purposes.
52       * @return The center position of the XY plane in the front of the crystal.
53       */
54      public Hep3Vector getPositionFront() {
55          if (positionFront == null) {
56              IGeometryInfo geom = getGeometry();
57              double[] p = geom.transformLocalToGlobal(VecOp.add(geom.transformGlobalToLocal(geom.getPosition()), 
58                              (Hep3Vector) new BasicHep3Vector(0, 0, -1 * ((Trd) geom.getLogicalVolume().getSolid()).getZHalfLength()))).v();
59              positionFront = new BasicHep3Vector(p[0], p[1], p[2]);
60          }
61          return positionFront;
62      }
63      
64  }
