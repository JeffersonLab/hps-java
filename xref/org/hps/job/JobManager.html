1   package org.hps.job;
2   
3   import java.util.HashSet;
4   import java.util.Set;
5   
6   import org.apache.commons.cli.CommandLine;
7   import org.apache.commons.cli.Options;
8   import org.hps.logging.config.DefaultLoggingConfig;
9   import org.lcsim.job.JobControlManager;
10  
11  /**
12   * Extension of standard LCSim job manager.
13   * <p>
14   * Provides setup of database conditions system and adds option to provide conditions system tags.
15   */
16  public final class JobManager extends JobControlManager {
17      
18      /**
19       * Run the job manager from the command line.
20       *
21       * @param args the command line arguments
22       */
23      public static void main(final String args[]) {    
24          
25          // Initialize default logging config.
26          DefaultLoggingConfig.initialize();
27         
28          // Run the job.
29          final JobManager job = new JobManager();
30          job.parse(args);
31          job.run();
32      }
33  
34      /**
35       * Class constructor.
36       */
37      public JobManager() {
38          conditionsSetup = new DatabaseConditionsManagerSetup();
39      }
40      
41      /**
42       * Get the conditions setup.
43       * @return the conditions setup
44       */
45      public DatabaseConditionsManagerSetup getDatabaseConditionsManagerSetup() {
46          return (DatabaseConditionsManagerSetup) this.conditionsSetup;
47      }
48      
49      /**
50       * Override creation of command line options.
51       * @return the overridden command line options
52       */
53      @Override
54      protected Options createCommandLineOptions() {
55          Options options = super.createCommandLineOptions();
56          options.addOption("t", "tag", true, "conditions system tag (can be used multiple times)");
57          return options;
58      }
59      
60      /**
61       * Override command line parsing.
62       * @return the overridden, parsed command line
63       */
64      @Override
65      public CommandLine parse(final String args[]) {
66          CommandLine commandLine = super.parse(args);
67          if (commandLine.hasOption("t")) {
68              Set<String> tags = new HashSet<String>();
69              for (String tag : commandLine.getOptionValues("t")) {
70                  tags.add(tag);
71              }
72              getDatabaseConditionsManagerSetup().setTags(tags);
73          }
74          return commandLine;
75      }
76  }
