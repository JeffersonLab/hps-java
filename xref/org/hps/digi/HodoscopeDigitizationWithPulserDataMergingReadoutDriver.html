30  /**
31   * Class <code>HodoscopeDigitizationWithPulserDataMergingReadoutDriver</code> is an
32   * implementation of the {@link
33   * org.hps.digi.DigitizationWithPulserDataMergingReadoutDriver} for a subdetector of type {@link
34   * org.lcsim.geometry.subdetector.Hodoscope_v1 Hodoscope_v1}. It
35   * handles all of the hodoscope-specific functions needed by the
36   * superclass.
37   * 
38   * @author Tongtong Cao <caot@jlab.org>
39   */
40  public class HodoscopeDigitizationWithPulserDataMergingReadoutDriver extends DigitizationWithPulserDataMergingReadoutDriver<Hodoscope_v1> {
41      // The DAQ configuration manager for FADC parameters.
42      private FADCConfigHodo2019 config = new FADCConfigHodo2019();
43      private boolean configStat = false; // Indicates if DAQ configuration is loaded
44  
45      // The number of nanoseconds in a clock-cycle (sample).
46      private static final int nsPerSample = 4;   
47      
48      /** Stores the set of all channel IDs for  the hodoscope. */
49      private Set<Long> channelIDSet = new HashSet<Long>();
50      /** Maps hodoscope channels to the gain for that channel. */
51      private Map<Long, HodoscopeGain> channelToGainsMap = new HashMap<Long, HodoscopeGain>();
52      /** Maps hodoscope channels to the time shifts for that channel. */
53      private Map<Long, HodoscopeTimeShift> channelToTimeShiftsMap = new HashMap<Long, HodoscopeTimeShift>();
54      /** Maps hodoscope channels to the noise sigma and pedestals for that channel. */
55      private Map<Long, HodoscopeCalibration> channelToCalibrationsMap = new HashMap<Long, HodoscopeCalibration>();
56      /** Factor for gain conversion from self-define-unit/ADC to MeV/ADC. */
57      private double factorGainConversion = 0.000833333;
58      /** Gain scaling factor for raw energy (self-defined unit) of FADC hits.
59       * In DAQ configuration, gains are scaled by the gain scaling factor for two-hole tiles. 
60       * Such gains from DAQ configuration should be divided by the factor.
61       */
62      
63      private HodoscopeConditions hodoConditions = null;
64      
65      public HodoscopeDigitizationWithPulserDataMergingReadoutDriver() {
66          // Set the default values for each subdetector-dependent
67          // parameter.
68          setGeometryName("Hodoscope");
69          
70          setInputHitCollectionName("HodoscopeHits");
71          setOutputHitCollectionName("HodoscopeRawHits");
72          setTruthRelationsCollectionName("HodoscopeTruthRelations");
73          setTriggerPathTruthRelationsCollectionName("HodoscopeTriggerPathTruthRelations");
74          setReadoutHitCollectionName("HodoscopeReadoutHits");
75          
76          setNumberSamplesAfter(10);
77          setNumberSamplesBefore(6);
78          setPulseTimeParameter(4.0);
79          setPhotoelectronsPerMeV(10.0);
80      }
81      
82      /**
83       * Sets whether or not the DAQ configuration is applied into the driver
84       * the EvIO data stream or whether to read the configuration from data files.
85       * 
86       * @param state - <code>true</code> indicates that the DAQ configuration is
87       * applied into the readout system, and <code>false</code> that it
88       * is not applied into the readout system.
89       */
90      public void setDaqConfigurationAppliedintoReadout(boolean state) {
91          // If the DAQ configuration should be read, attach a listener
92          // to track when it updates.               
93          if (state) {
94              ConfigurationManager2019.addActionListener(new ActionListener() {
95                  @Override
96                  public void actionPerformed(ActionEvent e) {
97                      // Get the DAQ configuration.
98                      DAQConfig2019 daq = ConfigurationManager2019.getInstance();
99  
100                     // Load the DAQ settings from the configuration manager.
101                     numSamplesAfter = daq.getHodoFADCConfig().getNSA() / nsPerSample;
102                     numSamplesBefore = daq.getHodoFADCConfig().getNSB() / nsPerSample;
103                     readoutWindow = daq.getHodoFADCConfig().getWindowWidth() / nsPerSample;
104                     pulserDataWindow = readoutWindow;                    
105 
106                     // Get the FADC configuration.
107                     config = daq.getHodoFADCConfig();
108                     configStat = true;
109                     integrationThreshold = config.getThreshold((int)10);
110                 }
111             });
112         }
113 
114     }

117     @Override
118     public void detectorChanged(Detector detector) {
119         // Get a copy of the calorimeter conditions for the detector.
120         hodoConditions = DatabaseConditionsManager.getInstance().getHodoConditions();        
121         
122         // Populate the channel ID collections.
123         populateChannelCollections();
124         
125         // Run the superclass method.
126         super.detectorChanged(detector);
127     }

129     @Override
130     protected Set<Long> getChannelIDs() {
131         return channelIDSet;
132     }
133     
134     @Override
135     protected Long getID(RawTrackerHit hit) {
136         return Long.valueOf(hodoConditions.getChannels().findGeometric(hit.getCellID()).getChannelId().intValue());
137     }
138     
139     @Override
140     protected double getGainConditions(long channelID) {
141         if(channelToGainsMap.containsKey(Long.valueOf(channelID))) {
142             return channelToGainsMap.get(Long.valueOf(channelID)).getGain() * factorGainConversion;
143         } else {
144             throw new IllegalArgumentException("No gain conditions exist for hodoscope channel ID \"" + channelID + "\".");
145         }
146     }
147     
148     @Override
149     protected double getNoiseConditions(long channelID) {
150         if(channelToCalibrationsMap.containsKey(Long.valueOf(channelID))) {
151             return channelToCalibrationsMap.get(Long.valueOf(channelID)).getNoise();
152         } else {
153             throw new IllegalArgumentException("No noise conditions exist for hodoscope channel ID \"" + channelID + "\".");
154         }
155     }
156     
157     @Override
158     protected double getPedestalConditions(long channelID) { 
159         if (channelToCalibrationsMap.containsKey(Long.valueOf(channelID))) {
160             return channelToCalibrationsMap.get(Long.valueOf(channelID)).getPedestal();
161         } else {
162             throw new IllegalArgumentException(
163                     "No pedestal conditions exist for hodoscope channel ID \"" + channelID + "\".");
164         }
165     }
166     
167     @Override
168     protected double getTimeShiftConditions(long channelID) {
169         if(channelToTimeShiftsMap.containsKey(Long.valueOf(channelID))) {
170             return channelToTimeShiftsMap.get(Long.valueOf(channelID)).getTimeShift();
171         } else {
172             throw new IllegalArgumentException("No time shift conditions exist for hodoscope channel ID \"" + channelID + "\".");
173         }
174     }
175     
176     @Override
177     protected int getTimestampFlag() {
178         return ReadoutTimestamp.SYSTEM_HODOSCOPE;
179     }

181     /**
182      * Populates the channel ID set and maps all existing channels to
183      * their respective conditions.
184      */
185     private void populateChannelCollections() {
186         // Load the conditions database and get the hodoscope channel
187         // collection data.
188         final DatabaseConditionsManager conditions = DatabaseConditionsManager.getInstance();
189         final HodoscopeGainCollection gains = conditions.getCachedConditions(HodoscopeGainCollection.class, "hodo_gains").getCachedData();
190         final HodoscopeChannelCollection channels = conditions.getCachedConditions(HodoscopeChannelCollection.class, "hodo_channels").getCachedData();
191         final HodoscopeTimeShiftCollection timeShifts = conditions.getCachedConditions(HodoscopeTimeShiftCollection.class, "hodo_time_shifts").getCachedData();
192         final HodoscopeCalibrationCollection calibrations = conditions.getCachedConditions(HodoscopeCalibrationCollection.class, "hodo_calibrations").getCachedData();
193         
194         // Map the gains to channel IDs.
195         for(HodoscopeGain gain : gains) {
196             channelToGainsMap.put(Long.valueOf(gain.getChannelId().intValue()), gain);
197         }
198         
199         // Map the pedestals and noise to channel IDs.
200         for(HodoscopeCalibration calibration : calibrations) {
201             channelToCalibrationsMap.put(Long.valueOf(calibration.getChannelId().intValue()), calibration);
202         }
203         
204         // Map time shifts to channel IDs.
205         for(HodoscopeTimeShift timeShift :  timeShifts) {
206             channelToTimeShiftsMap.put(Long.valueOf(timeShift.getChannelId().intValue()), timeShift);
207         }
208         
209         // Store the set of all channel IDs.
210         for(HodoscopeChannel channel : channels) {
211             channelIDSet.add(Long.valueOf(channel.getChannelId().intValue()));
212         }
213     }
