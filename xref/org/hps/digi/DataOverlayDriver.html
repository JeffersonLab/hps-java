23  /**
24   * <p>
25   * Driver to overlay from another event stream onto an EventHeader,
26   * adding ADC traces from RawTrackerHit objects with the same IDs,
27   * and then replacing the collection that exists.
28   * </p>
29   *
30   * <p>
31   * New collections will be created in the target output event if
32   * they do not exist.
33   * </p>
34   *
35   * <p>
36   * Example collections from 2019 data...
37   * </p>
38   *
39   * <pre>
40   * Collection Name                Collection Type
41   * --------------------------------------------------
42   * HodoReadoutHits                RawTrackerHit
43   * EcalReadoutHits                RawTrackerHit
44   * FADCGenericHits                LCGenericObject
45   * VTPBank                        LCGenericObject
46   * SVTRawTrackerHits              RawTrackerHit
47   * TSBank                         LCGenericObject
48   * TriggerBank                    LCGenericObject
49   * --------------------------------------------------
50   * </pre>
51   *
52   * <p>
53   * Hit times for RawTrackerHit output collections are hard-coded to zero for now
54   * (they are always zero in the data as well).
55   * </p>
56   *
57   * @author Jeremy McCormick, SLAC
58   */
59  public class DataOverlayDriver extends Driver {
60  
61      private static Logger LOGGER = Logger.getLogger(DataOverlayDriver.class.getPackage().getName());
62  
63      private final Queue<String> inputFilePaths = new LinkedList<String>();
64      LCIOReader reader = null;
65  
66      /**
67       * Collections names to read.
68       */
69      static List<String> COLLECTION_NAMES =
70              Arrays.asList("EcalReadoutHits", "HodoReadoutHits", "SVTRawTrackerHits");
71      
72      static List<String> READOUT_NAMES = 
73              Arrays.asList("EcalHits","HodoscopeHits","TrackerHits");

100     public void setInputFileList(String inputFileListPath) {
101         try {
102             BufferedReader reader = new BufferedReader(new FileReader(inputFileListPath));
103             String line = reader.readLine();
104             while (line != null) {
105                 line = reader.readLine().trim();
106                 if (line.length() > 0) {
107                     inputFilePaths.add(line);
108                 }
109             }
110             reader.close();
111         } catch (IOException e) {
112             throw new RuntimeException(e);
113         }
114     }

116     private EventHeader readNextEvent() throws EndOfDataException {
117         EventHeader event = null;
118         try {
119             LOGGER.info("Reading next event...");
120             event = this.reader.read();
121         } catch (EOFException eof) {
122             openNextFile();
123             try {
124                 LOGGER.info("Reading first event in new file...");
125                 event = this.reader.read();
126             } catch (IOException e) {
127                 e.printStackTrace();
128             }
129         } catch (IOException ioe) {
130             ioe.printStackTrace();
131         } finally {
132             if (event == null) {
133                 throw new RuntimeException("Failed to read next event.");
134             }
135         }
136         return event;
137     }
138 
139     @Override
140     protected void process(EventHeader event) {
141         try {
142             LOGGER.fine("Processing event: " + event.getEventNumber());
143             EventHeader overlayEvent = this.readNextEvent();
144             LOGGER.fine("Read overlay event: " + overlayEvent.getEventNumber());
145             this.overlayEvent(overlayEvent, event);
146         } catch (EndOfDataException e) {
147             throw new RuntimeException("Ran out of overlay events!");
148         }
149     }
150 
151     /**
152      * Overlay source event onto target.
153      *
154      * Collections in source not present in the target will be copied as new collections.
155      *
156      * If raw data collections are present in the target, source collections will be merged
157      * and ADC values added together for matching hits using the cell ID.
158      *
159      * @param overlayEvent the source event, typically pulser data
160      * @param targetEvent the target event, typically a SLIC signal event
161      */
162     private void overlayEvent(EventHeader overlayEvent, EventHeader targetEvent) {
163         LOGGER.fine("Overlaying event " + overlayEvent.getEventNumber() + " onto primary event " + targetEvent.getEventNumber());
164         int cnt=0;
165         for (String collName : COLLECTION_NAMES) {
166             if (!targetEvent.hasCollection(RawTrackerHit.class, collName)) {
167                 // Copy raw data collection into output event.
168                 LOGGER.finer("Adding new raw data collection to output event: " + collName);
169                 List<RawTrackerHit> overlayColl = overlayEvent.get(RawTrackerHit.class, collName);
170                 //                System.out.println( overlayEvent.get(RawTrackerHit.class, collName).getClass().getSimpleName());
171                 //                System.out.println(overlayEvent.getMetaData(overlayColl).getFlags().getClass().getSimpleName());
172                 // System.out.println(READOUT_NAMES.get(cnt).getClass().getSimpleName());
173                 targetEvent.put(collName,
174                         overlayEvent.get(RawTrackerHit.class, collName),
175                         RawTrackerHit.class,
176                                 overlayEvent.getMetaData(overlayColl).getFlags(),READOUT_NAMES.get(cnt));
177             } else {
178                 // Add ADC values for matching collections and add the new collection to the event.
179                 List<RawTrackerHit> overlayHits = overlayEvent.get(RawTrackerHit.class, collName);
180                 LOGGER.finer("Overlay " + collName + ": " + overlayHits.size());
181                 List<RawTrackerHit> hits = targetEvent.get(RawTrackerHit.class, collName);
182                 LOGGER.finer("Primary " + collName + ": " + hits.size());
183                 List<RawTrackerHit> newHits = this.mergeRawTrackerHitCollections(overlayHits, hits);
184                 LOGGER.finer("Merged " + collName + ": " + newHits.size() + '\n');
185                 targetEvent.put(collName, newHits, RawTrackerHit.class, targetEvent.getMetaData(hits).getFlags(),READOUT_NAMES.get(cnt));
186             }
187             cnt++;
188         }
189     }

191     static private void addHits(List<RawTrackerHit> hits, Map<Long, List<RawTrackerHit>> map) {
192         for (RawTrackerHit hit : hits) {
193             Long id = hit.getCellID();
194             if (!map.containsKey(id)) {
195                 map.put(id, new ArrayList<RawTrackerHit>());
196             }
197             map.get(id).add(hit);
198         }
199     }
200 
201     private List<RawTrackerHit> mergeRawTrackerHitCollections(List<RawTrackerHit> coll1, List<RawTrackerHit> coll2) {
202         List<RawTrackerHit> newColl = new ArrayList<RawTrackerHit>();
203         Map<Long, List<RawTrackerHit>> hitMap = new HashMap<Long, List<RawTrackerHit>>();
204         addHits(coll1, hitMap);
205         addHits(coll2, hitMap);
206         for (Long id : hitMap.keySet()) {
207             List<RawTrackerHit> hits = hitMap.get(id);
208             if (hits.size() == 1) {
209                 newColl.add(hits.get(0));
210                 LOGGER.info("Added single hit: " + String.format("0x%08X", hits.get(0).getCellID()));
211             } else {
212                 RawTrackerHit combinedHit = combineHits(hits);
213                 newColl.add(combinedHit);
214                 LOGGER.info("Added " + hits.size() + " hits to: " + String.format("0x%08X", combinedHit.getCellID()));
215             }
216         }
217         return newColl;
218     }
219 
220     private RawTrackerHit combineHits(List<RawTrackerHit> hits) {
221         short[] adcValues = new short[hits.get(0).getADCValues().length];
222         long id = hits.get(0).getCellID();
223         for (RawTrackerHit hit : hits) {
224             for (int i = 0; i < adcValues.length; i++) {
225                 adcValues[0] += hit.getADCValues()[i];
226             }
227         }
228         return new BaseRawTrackerHit(id, 0, adcValues);
229     }

231     @Override
232     protected void startOfData() {
233         if (this.inputFilePaths.size() == 0) {
234             throw new RuntimeException("No input file paths were provided!");
235         }
236         String filePath = this.inputFilePaths.remove();
237         try {
238             LOGGER.info("Opening overlay file: " + filePath);
239             reader = new LCIOReader(new File(filePath));
240         } catch (IOException e) {
241             LOGGER.severe("Failed to read from: " + filePath);
242             throw new RuntimeException(e);
243         }
244     }
245 
246     private void openNextFile() throws EndOfDataException {
247         try {
248             LOGGER.info("Closing previous reader...");
249             reader.close();
250         } catch (IOException e) {
251             throw new RuntimeException(e);
252         }
253         if (inputFilePaths.size() == 0) {
254             throw new EndOfDataException();
255         }
256         String path = inputFilePaths.remove();
257         LOGGER.info("Opening next file: " + path);
258         try {
259             reader = new LCIOReader(new File(path));
260         } catch (IOException e) {
261             throw new RuntimeException(e);
262         }
263     }
