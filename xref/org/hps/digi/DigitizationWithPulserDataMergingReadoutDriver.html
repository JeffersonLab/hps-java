43  /**
44   * Class <code>DigitizationWithPulserDataMergingReadoutDriver</code> performs digitization
45   * of truth hits from SLIC by converting them into emulated pulses and merges pulser data,
46   * and then performing pulse integration. The results are output in
47   * the form of {@link org.lcsim.event.RawCalorimeterHit
48   * RawCalorimeterHit} objects.
49   * <br/><br/>
50   * The truth hit information is retained by also producing an output
51   * collection of {@link org.lcsim.event.LCRelation LCRelation}
52   * objects linking the raw hits to the original {@link
53   * org.lcsim.event.SimCalorimeterHit SimCalorimeterHit} objects from
54   * which they were generated.
55   * <br/><br/>
56   * <code>DigitizationReadoutDriver</code> is itself abstract. It is
57   * designed with the intent to function for both the hodoscope and
58   * the calorimeter. As such, it requires its implementing classes to
59   * handle certain subdetector-specific tasks.
60   * 
61   * @author Tongtong Cao <caot@jlab.org>
62   */
63  public abstract class DigitizationWithPulserDataMergingReadoutDriver<D extends Subdetector> extends ReadoutDriver {

307     @Override
308     public void startOfData() {
309         // Validate that all the collection names are defined.
310         if(truthHitCollectionName == null || PulserDataCollectionName == null || outputHitCollectionName == null || truthRelationsCollectionName == null
311                 || triggerTruthRelationsCollectionName == null || readoutCollectionName == null) {
312             throw new RuntimeException("One or more collection names is not defined!");
313         }
314         
315         // Calculate the correct time offset. This is a function of
316         // the integration samples and the output delay.
317         // Threshold-crossing sample is part of NSA.
318         localTimeOffset = 4 * numSamplesAfter;
319         
320         // Validate that a real mode was selected.
321         if(mode != 1 && mode != 3 && mode != 7) {
322             throw new IllegalArgumentException("Error: Mode " + mode + " is not a supported output mode.");
323         }
324         
325         // Add the driver dependencies.
326         addDependency(truthHitCollectionName);
327         addDependency(PulserDataCollectionName);
328         
329         // Define the LCSim collection parameters for this driver's
330         // output. Note: Since these are not persisted, the flags and
331         // readout name are probably not necessary.
332         LCIOCollectionFactory.setCollectionName(outputHitCollectionName);
333         LCIOCollectionFactory.setProductionDriver(this);
334         LCIOCollectionFactory.setFlags((0 + (1 << LCIOConstants.CHBIT_LONG) + (1 << LCIOConstants.RCHBIT_ID1)));
335         LCIOCollectionFactory.setReadoutName(truthHitCollectionName);
336         LCIOCollection<RawCalorimeterHit> hitCollectionParams = LCIOCollectionFactory.produceLCIOCollection(RawCalorimeterHit.class);
337         ReadoutDataManager.registerCollection(hitCollectionParams, false);
338         
339         LCIOCollectionFactory.setCollectionName(triggerTruthRelationsCollectionName);
340         LCIOCollectionFactory.setProductionDriver(this);
341         LCIOCollection<LCRelation> triggerTruthCollectionParams = LCIOCollectionFactory.produceLCIOCollection(LCRelation.class);
342         ReadoutDataManager.registerCollection(triggerTruthCollectionParams, false);
343         
344         // Define the LCSim collection data for the on-trigger output.
345         LCIOCollectionFactory.setCollectionName(readoutCollectionName);
346         LCIOCollectionFactory.setProductionDriver(this);
347         mode13HitCollectionParams = LCIOCollectionFactory.produceLCIOCollection(RawTrackerHit.class);
348         
349         LCIOCollectionFactory.setCollectionName(readoutCollectionName);
350         LCIOCollectionFactory.setProductionDriver(this);
351         LCIOCollectionFactory.setFlags(1 << LCIOConstants.RCHBIT_TIME);
352         mode7HitCollectionParams = LCIOCollectionFactory.produceLCIOCollection(RawCalorimeterHit.class);
353         
354         LCIOCollectionFactory.setCollectionName(truthRelationsCollectionName);
355         LCIOCollectionFactory.setProductionDriver(this);
356         truthRelationsCollectionParams = LCIOCollectionFactory.produceLCIOCollection(LCRelation.class);
357         
358         // Run the superclass method.
359         super.startOfData();
360     }

383     @Override
384     public void process(EventHeader event) {
385         
386         /*
387          * Get current SLIC hits and current raw hits in pulser data.
388          */
389         
390         // Get current SLIC hits.
391         Collection<SimCalorimeterHit> hits = ReadoutDataManager.getData(ReadoutDataManager.getCurrentTime(), ReadoutDataManager.getCurrentTime() + 2.0,
392                 truthHitCollectionName, SimCalorimeterHit.class);
393         
394         // Get current raw hits in pulser data.
395         Collection<RawTrackerHit> rawHits = ReadoutDataManager.getData(ReadoutDataManager.getCurrentTime(), ReadoutDataManager.getCurrentTime() + 2.0,
396                 PulserDataCollectionName, RawTrackerHit.class);        
397         
398         // Once an overlaid event is input, reset adcBufferMap to ensure that other overlaid events do not affect the current event. 
399         if(hits.size()!=0 || rawHits.size()!=0) {
400             // Get the set of all possible channel IDs.
401             Set<Long> cells = getChannelIDs();
402             
403             // Reset adcBufferMap.
404             for(Long cellID : cells)
405                 adcBufferMap.get(cellID).setAll((int) Math.round(getPedestalConditions(cellID)));
406         }
407         
408         /* To merge MC data with pulser data, three different cases are handled separately.
409          * Case 1: If pulser data does not have a channel in MC data, directly buffer samples
410          * 
411          * Case 2: If MC data does not have a channel in pulser data,
412          * 1) add noise into MC hits
413          * 2) convert MC hits into a window of ADC samples
414          * 3) add pedestal
415          * 4) buffer samples  
416          * 
417          * Case 3: If MC data has a channel that is also in pulser data, 
418          * 1) convert MC hits into a window of ADC samples
419          * 2) merge with samples of pulser data
420          * 3) buffer merged samples
421          * 
422          * MC hits are digitized into ADC samples with the same time window of pulser data.
423          * Before the time window, the window is extended with NSB ADC samples, and values of the ADC samples are set as pedestal.
424          * After the time window, the window is extended with NSA ADC samples, and values of the ADC samples are set as pedestal.
425          * The extension is allowed since enough empty events are inserted into neighbored overlaid events.
426          */
427         
428         // Add the truth hits to the truth hit buffer. The buffer is
429         // only incremented when the ADC buffer is incremented, which
430         // is handled below.
431         // Save cell IDs of hits as keys in the MC hit Cell ID hash map, and set values as 1.
432         // The hash map is used to check if MC data has a channel that is also in pulser data.
433         Map<Long,Integer> hitCellIDMap = new HashMap<Long,Integer>(hits.size());
434         for(SimCalorimeterHit hit : hits) {
435             // Store the truth data.
436             Long hitCellID = hit.getCellID(); // For Ecal, cell ID is geometry ID; For hodo, cell ID is channel ID after hodoscope preprocessing
437             
438             ObjectRingBuffer<SimCalorimeterHit> hitBuffer = truthBufferMap.get(hitCellID);
439             hitBuffer.addToCell(0, hit);
440             
441             // Save cell IDs of hits as keys in the hit Cell ID hash map, and set values as 1.
442             if(hitCellIDMap.get(hitCellID) == null)
443                 hitCellIDMap.put(hitCellID,1);
444         }
445         
446         // handle pulser data: case 1.
447         // If cellID of a raw hit is not included by keys in the MC hit Cell ID hash map for MC hits, directly buffer ADC samples.
448         // If included, set value as 2 for the corresponding key in the MC hit Cell ID hash map.
449         // Save raw hits in the raw hit hash map, where keys are raw hit cell IDs and values are raw hits.
450         // The hash map is used for case 3
451         Map<Long, RawTrackerHit> rawHitsMap = new HashMap<Long,RawTrackerHit>(rawHits.size());
452         for(RawTrackerHit rawHit : rawHits) {
453             Long rawHitID = getID(rawHit); // For Ecal, ID is geometry ID; For hodo, ID is channel ID, which is converted from geometry ID. 
454             if(hitCellIDMap.get(rawHitID) == null) {
455                 // Get the ADC buffer for the channel.
456                 IntegerRingBuffer adcBuffer = adcBufferMap.get(rawHitID);
457                 
458                 // Get ADC samples for the channel.
459                 short[] adcSamples = rawHit.getADCValues();                 
460 
461                 // Length of ADC sample array should be equal to setup for time window of ADC samples
462                 if(adcSamples.length != pulserDataWindow) 
463                     throw new RuntimeException("Error: time window of pulser data is not correctly set.");                
464                 
465                 // Buffer ADC samples in pulser data
466                 for(int i = 0; i < pulserDataWindow; i++) 
467                     adcBuffer.setValue(i - pulserSamplesShift, (int)adcSamples[i]);  
468             }
469             else {
470                 hitCellIDMap.put(rawHitID, 2);
471                 rawHitsMap.put(rawHitID, rawHit);
472             }
473         }
474         
475         // handle MC hits: case 2 and case 3
476         // In the MC hit Cell ID hash map, if value for cell ID of a MC hit is 1, handle the hit as case 2.
477         // If value for cell ID of a MC hit is 2, handle the hit as case 3.
478         for(SimCalorimeterHit hit : hits) {
479             Long hitCellID = hit.getCellID();            
480 
481             // Check to see if the hit time seems valid. This is done
482             // by calculating the time of the next readout cycle in
483             // ns and subtracting the time of the current hit (with
484             // adjustment for simulation time passed) from it. If the
485             // hit would fall in a previous readout cycle, something
486             // is probably wrong.
487             if(READOUT_PERIOD + readoutTime() - (ReadoutDataManager.getCurrentTime() + hit.getTime()) >= READOUT_PERIOD) {
488                 throw new RuntimeException("Error: Trying to add a hit to the analog pipeline, but the time seems incorrect.");
489             }
490             
491             // Get the ADC buffer for the channel.
492             IntegerRingBuffer adcBuffer = adcBufferMap.get(hitCellID);
493 
494             // Get the pedestal for the channel.
495             int pedestal = (int) Math.round(getPedestalConditions(hitCellID));           
496             
497             // Get the buffer for the current truth hit's channel.
498             DoubleRingBuffer voltageBuffer = voltageBufferMap.get(hitCellID);
499             
500             // Get the truth hit energy deposition.
501             double energyAmplitude = hit.getRawEnergy();
502             
503             if(hitCellIDMap.get(hitCellID) == 1) {                
504                 // If noise should be added, calculate a random value for
505                 // the noise and add it to the truth energy deposition.
506                 if(addNoise) {
507                     // Calculate a randomized noise value.
508                     double noiseSigma = Math.sqrt(Math.pow(getNoiseConditions(hitCellID) * getGainConditions(hitCellID) * EcalUtils.MeV, 2)
509                             + hit.getRawEnergy() * EcalUtils.MeV / pePerMeV);
510                     double noise = RandomGaussian.getGaussian(0, noiseSigma);
511                     
512                     // Increment the truth energy deposition by this amount.
513                     energyAmplitude += noise;
514                 }
515                 
516                 // Simulate the pulse for each position in the preamp
517                 // pulse buffer for the subdetector channel on which the
518                 // hit occurred.
519                 for(int i = 0; i < pulserDataWindow; i++) {
520                     // Calculate the voltage deposition for the current
521                     // buffer time.
522                     double voltageDeposition = energyAmplitude * pulseAmplitude((i + 1) * READOUT_PERIOD + readoutTime()
523                             - (ReadoutDataManager.getCurrentTime() + hit.getTime()) - getTimeShiftConditions(hitCellID), hitCellID);
524                     
525                     // Increase the current buffer time's voltage value
526                     // by the calculated amount.
527                     voltageBuffer.addToCell(i, voltageDeposition);
528                     
529                     // Scale the current value of the preamplifier buffer
530                     // to a 12-bit ADC value where the maximum represents
531                     // a value of maxVolt.
532                     double currentValue = voltageBuffer.getValue(i) * ((Math.pow(2, nBit) - 1) / maxVolt);
533                     
534                     // An ADC value is not allowed to exceed 4095. If a
535                     // larger value is observed, 4096 (overflow) is given
536                     // instead. (This corresponds to >2 Volts.)
537                     int digitizedValue = Math.min((int) Math.round(pedestal + currentValue), (int) Math.pow(2, nBit));
538 
539                     // Write this value to the ADC buffer.
540                     adcBuffer.setValue(i, digitizedValue);
541                 } 
542             }
543             
544             else {
545                 // Get ADC samples for the channel.
546                 short[] ADCSamples = rawHitsMap.get(hitCellID).getADCValues(); 
547                 
548                 // Get digitized samples for MC hits
549                 int[] digitizedValue = new int[pulserDataWindow];
550                 
551                 // Simulate the pulse for each position in the preamp
552                 // pulse buffer for the subdetector channel on which the
553                 // hit occurred.
554 
555                 for(int i = 0; i < pulserDataWindow; i++) {
556                     // Calculate the voltage deposition for the current
557                     // buffer time.
558                     double voltageDeposition = energyAmplitude * pulseAmplitude((i + 1) * READOUT_PERIOD + readoutTime()
559                             - (ReadoutDataManager.getCurrentTime() + hit.getTime()) - getTimeShiftConditions(hitCellID), hitCellID);
560                     
561                     // Increase the current buffer time's voltage value
562                     // by the calculated amount.
563                     voltageBuffer.addToCell(i, voltageDeposition);
564                     
565                     // Scale the current value of the preamplifier buffer
566                     // to a 12-bit ADC value where the maximum represents
567                     // a value of maxVolt.
568                     double currentValue = voltageBuffer.getValue(i) * ((Math.pow(2, nBit) - 1) / maxVolt);
569                     
570                     // An ADC value is not allowed to exceed 4095. If a
571                     // larger value is observed, 4096 (overflow) is given
572                     // instead. (This corresponds to >2 Volts.)
573                     digitizedValue[i] = Math.min((int) Math.round(currentValue), (int) Math.pow(2, nBit));                    
574                 }
575                 
576                 // Write this value to the ADC buffer.  
577                 // If pulserSamplesShift is larger than 0, merged sample window is [-pulserSamplesShift, pulserDataWindow]
578                 if(pulserSamplesShift >= 0) {
579                     for(int i = -pulserSamplesShift; i < 0; i++) adcBuffer.setValue(i , (int)ADCSamples[i + pulserSamplesShift]);
580                     for(int i = 0; i < pulserDataWindow - pulserSamplesShift; i++) adcBuffer.setValue(i, digitizedValue[i] + ADCSamples[i + pulserSamplesShift]);
581                     for(int i = pulserDataWindow - pulserSamplesShift; i < pulserDataWindow; i++) adcBuffer.setValue(i, digitizedValue[i]);
582                 }
583                 // If pulserSamplesShift is less than 0, merged sample window is [0, -pulserSamplesShift + pulserDataWindow]
584                 else {
585                     for(int i = 0; i < -pulserSamplesShift; i++) adcBuffer.setValue(i, digitizedValue[i]);
586                     for(int i = -pulserSamplesShift; i < pulserDataWindow; i++) adcBuffer.setValue(i, digitizedValue[i] + ADCSamples[i + pulserSamplesShift]);
587                     for(int i = pulserDataWindow; i < pulserDataWindow - pulserSamplesShift; i++) adcBuffer.setValue(i, (int)ADCSamples[i + pulserSamplesShift]);
588                 }
589             }            
590         }       
591           
592         /*
593          * Next step is to integrate hits from the pulses. Hit
594          * integration is only performed once per readout period. The
595          * readout period, defined by the hardware, is by default 4
596          * nanoseconds.
597          */
598         
599         // Check whether the appropriate amount of time has passed to
600         // perform another integration step. If so, create a list to
601         // contain any newly integrated hits and perform integration.
602         List<RawCalorimeterHit> newHits = null;
603         List<LCRelation> newTruthRelations = null;
604         while(ReadoutDataManager.getCurrentTime() - readoutTime() + ReadoutDataManager.getBeamBunchSize() >= READOUT_PERIOD) {
605             if(newHits == null) { newHits = new ArrayList<RawCalorimeterHit>(); }
606             if(newTruthRelations == null) { newTruthRelations = new ArrayList<LCRelation>(); }
607             readHits(newHits, newTruthRelations);
608             readoutCounter++;
609         }
610     }

612     // TODO: Document this.
613     private void readHits(List<RawCalorimeterHit> newHits, List<LCRelation> newTruthRelations) {
614         // Perform hit integration as needed for each subdetector
615         // channel in the buffer map.
616         for(Long cellID : adcBufferMap.keySet()) {
617             // Get the ADC buffer for the channel.
618             IntegerRingBuffer adcBuffer = adcBufferMap.get(cellID);   
619             
620             // Get the pedestal for the channel.
621             int pedestal = (int) Math.round(getPedestalConditions(cellID));
622             
623             // Store the pedestal subtracted value so that it may
624             // be checked against the integration threshold.
625             int pedestalSubtractedValue = adcBuffer.getValue() - pedestal;
626             
627             // Get the total ADC value that has been integrated
628             // on this channel.
629             Integer sum = channelIntegrationSumMap.get(cellID);
630             
631             // If any readout hits exist on this channel, add the
632             // current ADC values to them.
633             
634             // If the ADC sum is undefined, then there is not an
635             // ongoing integration. If the pedestal subtracted
636             // value is also over the integration threshold, then
637             // integration should be initiated.
638             if(sum == null && pedestalSubtractedValue > integrationThreshold) {
639                 // Store the current local time in units of
640                 // events (4 ns). This will indicate when the
641                 // integration started and, in turn, should end.
642                 channelIntegrationTimeMap.put(cellID, readoutCounter);
643                 
644                 // Integrate the ADC values for a number of
645                 // samples defined by NSB and threshold
646                 // crossing sample. 
647                 int sumBefore = 0;
648                 for(int i = 0; i <= numSamplesBefore; i++) {
649                     sumBefore += adcBuffer.getValue(-(numSamplesBefore - i));
650                 }
651                 
652                 // This will represent the total integral sum at
653                 // the current point in time. Store it in the sum
654                 // buffer so that it may be incremented later as
655                 // additional samples are read.
656                 channelIntegrationSumMap.put(cellID, sumBefore);
657                 
658                 // Collect and store truth information for trigger
659                 // path hits.
660                 channelIntegrationADCMap.put(cellID, new ArrayList<Integer>());
661                 
662                 // Get the truth information in the
663                 // integration samples for this channel.
664                 Set<SimCalorimeterHit> truthHits = new HashSet<SimCalorimeterHit>();
665                 for(int i = 0; i < numSamplesBefore + 4; i++) {
666                     channelIntegrationADCMap.get(cellID).add(adcBuffer.getValue(-(numSamplesBefore - i)));
667                     truthHits.addAll(truthBufferMap.get(cellID).getValue(-(numSamplesBefore - i)));
668                 }
669                 
670                 // Store all the truth hits that occurred in
671                 // the truth buffer in the integration period
672                 // for this channel as well. These will be
673                 // passed through the chain to allow for the
674                 // accessing of truth information during the
675                 // trigger simulation.
676                 channelIntegrationTruthMap.put(cellID, truthHits);
677             }
678             
679             // If the integration sum is defined, then pulse
680             // integration is ongoing.
681             if(sum != null) {
682                 // Three cases are treated separataly
683                 // Case 1: CHANNEL_INTEGRATION_DEADTIME > numSamplesAfter
684                 // Case 2: CHANNEL_INTEGRATION_DEADTIME == numSamplesAfter 
685                 // Case 3: CHANNEL_INTEGRATION_DEADTIME < numSamplesAfter
686                 if(CHANNEL_INTEGRATION_DEADTIME > numSamplesAfter) { // Case 1
687                     //Continue integration until NSA, the threshold-crossing sample has been added before.
688                     if (channelIntegrationTimeMap.get(cellID) + numSamplesAfter - 1 >= readoutCounter) { 
689                         channelIntegrationADCMap.get(cellID).add(adcBuffer.getValue(0));
690 
691                         // Add the new ADC sample.
692                         channelIntegrationSumMap.put(cellID, sum + adcBuffer.getValue(0));
693 
694                         // Add the new truth information, if trigger
695                         // path truth output is enabled.
696                         if (writeTriggerTruth) {
697                             channelIntegrationTruthMap.get(cellID).addAll(truthBufferMap.get(cellID).getValue(0));
698                         }
699                     }
700 
701                     // If integration is complete, a hit may be added
702                     // to data manager.
703                     else if (channelIntegrationTimeMap.get(cellID) + numSamplesAfter - 1 == readoutCounter - 1) {//At NSA + 1, hit is added into data manager
704                         // Add a new calorimeter hit.
705                         RawCalorimeterHit newHit = new BaseRawCalorimeterHit(cellID, sum,
706                                 64 * channelIntegrationTimeMap.get(cellID));
707                         newHits.add(newHit);
708                         // Cycle-clock for events is 2 ns, while cycle-clock for samples is 4 ns                        
709                         integrationTime = channelIntegrationTimeMap.get(cellID) * 4 + 2;
710                         // Add the truth relations for this hit, if
711                         // trigger path truth is enabled.
712                         if (writeTriggerTruth) {
713                             Set<SimCalorimeterHit> truthHits = channelIntegrationTruthMap.get(cellID);
714                             for (SimCalorimeterHit truthHit : truthHits) {
715                                 newTruthRelations.add(new BaseLCRelation(newHit, truthHit));
716                             }
717                         }
718                     }
719 
720                     // Do not clear the channel for integration until  deadtime has passed.
721                     // The threshold-crossing sample counts as the first sample in the deadtime.
722                     else if (channelIntegrationTimeMap.get(cellID) + CHANNEL_INTEGRATION_DEADTIME - 1 <= readoutCounter
723                             - 1) { // No new integration until over deadtime
724                         channelIntegrationSumMap.remove(cellID);
725                     }
726                 } // Case 1 ends
727                 else if(CHANNEL_INTEGRATION_DEADTIME == numSamplesAfter){ // Case 2
728                     // Continue integration until NSA, the threshold-crossing sample has been added before.
729                     if (channelIntegrationTimeMap.get(cellID) + numSamplesAfter - 1 >= readoutCounter) {
730                         channelIntegrationADCMap.get(cellID).add(adcBuffer.getValue(0));
731 
732                         // Add the new ADC sample.
733                         channelIntegrationSumMap.put(cellID, sum + adcBuffer.getValue(0));
734 
735                         // Add the new truth information, if trigger
736                         // path truth output is enabled.
737                         if (writeTriggerTruth) {
738                             channelIntegrationTruthMap.get(cellID).addAll(truthBufferMap.get(cellID).getValue(0));
739                         }
740                     }  
741                     // If integration is complete, a hit may be added
742                     // to data manager.
743                     else if (channelIntegrationTimeMap.get(cellID) + numSamplesAfter - 1 == readoutCounter - 1) {//At NSA + 1, hit is added into data manager
744                         // Add a new calorimeter hit.
745                         RawCalorimeterHit newHit = new BaseRawCalorimeterHit(cellID, sum,
746                                 64 * channelIntegrationTimeMap.get(cellID));
747                         newHits.add(newHit);
748                         // Cycle-clock for events is 2 ns, while cycle-clock for samples is 4 ns 
749                         integrationTime = channelIntegrationTimeMap.get(cellID) * 4 + 2;
750 
751                         // Add the truth relations for this hit, if
752                         // trigger path truth is enabled.
753                         if (writeTriggerTruth) {
754                             Set<SimCalorimeterHit> truthHits = channelIntegrationTruthMap.get(cellID);
755                             for (SimCalorimeterHit truthHit : truthHits) {
756                                 newTruthRelations.add(new BaseLCRelation(newHit, truthHit));
757                             }
758                         }
759                         channelIntegrationSumMap.remove(cellID);
760                     }
761                 } // Case 2 ends
762                 else { // Case 3
763                     if (channelIntegrationTimeMap.get(cellID) + CHANNEL_INTEGRATION_DEADTIME - 1 >= readoutCounter) {
764                         // Continue integration until CHANNEL_INTEGRATION_DEADTIME
765                         channelIntegrationADCMap.get(cellID).add(adcBuffer.getValue(0));
766 
767                         // Add the new ADC sample.
768                         channelIntegrationSumMap.put(cellID, sum + adcBuffer.getValue(0));
769 
770                         // Add the new truth information, if trigger
771                         // path truth output is enabled.
772                         if (writeTriggerTruth) {
773                             channelIntegrationTruthMap.get(cellID).addAll(truthBufferMap.get(cellID).getValue(0));
774                         }
775                         
776                         // If sample at the end of deadtime is less than threshold, new integration could be started from next sample
777                         if(channelIntegrationTimeMap.get(cellID) + CHANNEL_INTEGRATION_DEADTIME == readoutCounter && pedestalSubtractedValue <= integrationThreshold)                            
778                             flagStartNewIntegration.put(cellID, true);                           
779                     }  
780                     else if (channelIntegrationTimeMap.get(cellID) + numSamplesAfter - 1 >= readoutCounter) {
781                         if(flagStartNewIntegration.get(cellID) == true) { // Flag for previous sample is true
782                             if(pedestalSubtractedValue <= integrationThreshold) { // If sample is less than threshold, then do not start new integration
783                                 channelIntegrationADCMap.get(cellID).add(adcBuffer.getValue(0));
784 
785                                 // Add the new ADC sample.
786                                 channelIntegrationSumMap.put(cellID, sum + adcBuffer.getValue(0));
787 
788                                 // Add the new truth information, if trigger
789                                 // path truth output is enabled.
790                                 if (writeTriggerTruth) {
791                                     channelIntegrationTruthMap.get(cellID).addAll(truthBufferMap.get(cellID).getValue(0));
792                                 }                                
793                             }
794                             else { // if sample is larger than threshold, a hit is added into data manager and start new integration
795                                 // Add a new calorimeter hit.
796                                 RawCalorimeterHit newHit = new BaseRawCalorimeterHit(cellID, sum,
797                                         64 * channelIntegrationTimeMap.get(cellID));
798                                 newHits.add(newHit);
799                                 integrationTime = channelIntegrationTimeMap.get(cellID) * 4 + 2;
800 
801                                 // Add the truth relations for this hit, if
802                                 // trigger path truth is enabled.
803                                 if (writeTriggerTruth) {
804                                     Set<SimCalorimeterHit> truthHits = channelIntegrationTruthMap.get(cellID);
805                                     for (SimCalorimeterHit truthHit : truthHits) {
806                                         newTruthRelations.add(new BaseLCRelation(newHit, truthHit));
807                                     }
808                                 }                                                                
809             
810                                 //Start new integration
811                                 channelIntegrationTimeMap.put(cellID, readoutCounter);
812                                 flagStartNewIntegration.put(cellID, false);
813                                 
814                                 // Integrate the ADC values for a number of
815                                 // samples defined by NSB from before threshold
816                                 // crossing. Note that this stops one sample
817                                 // before the current sample. This current sample
818                                 // is handled in the subsequent code block.
819                                 int sumBefore = 0;
820                                 for(int i = 0; i <= numSamplesBefore; i++) {
821                                     sumBefore += adcBuffer.getValue(-(numSamplesBefore - i));
822                                 }
823                                 
824                                 // This will represent the total integral sum at
825                                 // the current point in time. Store it in the sum
826                                 // buffer so that it may be incremented later as
827                                 // additional samples are read.
828                                 channelIntegrationSumMap.put(cellID, sumBefore);
829                                 
830                                 // Collect and store truth information for trigger
831                                 // path hits.
832                                 channelIntegrationADCMap.put(cellID, new ArrayList<Integer>());
833                                 
834                                 // Get the truth information in the
835                                 // integration samples for this channel.
836                                 Set<SimCalorimeterHit> truthHits = new HashSet<SimCalorimeterHit>();
837                                 for(int i = 0; i < numSamplesBefore + 4; i++) {
838                                     channelIntegrationADCMap.get(cellID).add(adcBuffer.getValue(-(numSamplesBefore - i)));
839                                     truthHits.addAll(truthBufferMap.get(cellID).getValue(-(numSamplesBefore - i)));
840                                 }
841                                 
842                                 // Store all the truth hits that occurred in
843                                 // the truth buffer in the integration period
844                                 // for this channel as well. These will be
845                                 // passed through the chain to allow for the
846                                 // accessing of truth information during the
847                                 // trigger simulation.
848                                 channelIntegrationTruthMap.put(cellID, truthHits);                                                                    
849                             }                                                          
850                         }
851                         else { // Flag for previous sample is false
852                             channelIntegrationADCMap.get(cellID).add(adcBuffer.getValue(0));
853 
854                             // Add the new ADC sample.
855                             channelIntegrationSumMap.put(cellID, sum + adcBuffer.getValue(0));
856 
857                             // Add the new truth information, if trigger
858                             // path truth output is enabled.
859                             if (writeTriggerTruth) {
860                                 channelIntegrationTruthMap.get(cellID).addAll(truthBufferMap.get(cellID).getValue(0));
861                             }
862                             if(pedestalSubtractedValue <= integrationThreshold)
863                                 flagStartNewIntegration.put(cellID, true);                                                       
864                         }  
865                     }
866                     else if (channelIntegrationTimeMap.get(cellID) + numSamplesAfter - 1 == readoutCounter - 1) {//If reach NSA + 1, hit is added into data manager, and flag is set as false
867                         // Add a new calorimeter hit.
868                         RawCalorimeterHit newHit = new BaseRawCalorimeterHit(cellID, sum,
869                                 64 * channelIntegrationTimeMap.get(cellID));
870                         newHits.add(newHit);
871                         integrationTime = channelIntegrationTimeMap.get(cellID) * 4 + 2;
872 
873                         // Add the truth relations for this hit, if
874                         // trigger path truth is enabled.
875                         if (writeTriggerTruth) {
876                             Set<SimCalorimeterHit> truthHits = channelIntegrationTruthMap.get(cellID);
877                             for (SimCalorimeterHit truthHit : truthHits) {
878                                 newTruthRelations.add(new BaseLCRelation(newHit, truthHit));
879                             }
880                         }
881                         channelIntegrationSumMap.remove(cellID);
882                         flagStartNewIntegration.put(cellID, false);
883                     }
884                 } // Case 3 ends                 
885             }
886             
887             // Step to the next entry in the adc buffer.
888             adcBuffer.stepForward();  
889             
890             // Step to the next entry in the voltage buffer.
891             if(voltageBufferMap.get(cellID) != null) { // A channel could be from pulser data, while MC data has no such channel.
892                 DoubleRingBuffer voltageBuffer = voltageBufferMap.get(cellID);
893                 voltageBuffer.clearValue();
894                 voltageBuffer.stepForward();
895             }
896                        
897             // Step the truth buffer for this channel forward.
898             // The new cell should be cleared of any old values.
899             if(truthBufferMap.get(cellID) != null) { // A channel could be from pulser data, while MC data has no such channel.
900                 truthBufferMap.get(cellID).stepForward();
901                 truthBufferMap.get(cellID).clearValue();
902             }
903         }
904         
905         // Write the trigger path output data to the readout data
906         // manager. Truth data is optional.
907         
908     
909         ReadoutDataManager.addData(outputHitCollectionName, integrationTime, newHits, RawCalorimeterHit.class);
910         if(writeTriggerTruth) {
911             ReadoutDataManager.addData(triggerTruthRelationsCollectionName, integrationTime, newTruthRelations, LCRelation.class);
912         }
913     }

915     /**
916      * Finds all root particles associated with the interactions that
917      * created the argument particle.
918      * @param particle - The particle.
919      * @return Returns a {@link java.util.List List} containing each
920      * particle object in the argument particle's particle tree which
921      * has no parent particle.
922      */
923     private static final List<MCParticle> getRootParticleList(MCParticle particle) {
924         // If the particle has no parents, it should be added to the
925         // list and the list returned.
926         if(particle.getParents().isEmpty()) {
927             List<MCParticle> list = new ArrayList<MCParticle>(1);
928             list.add(particle);
929             return list;
930         }
931         
932         // If there is only one parent, just return the results from
933         // that parent.
934         else if(particle.getParents().size() == 1) {
935             return getRootParticleList(particle.getParents().get(0));
936         }
937         
938         // Otherwise, run the method on each parent particle and
939         // return the results from that instead.
940         else {
941             // Store the parent particle roots.
942             List<MCParticle> list = new ArrayList<MCParticle>();
943             
944             // Get the root particles for each parent and add them to
945             // the list.
946             for(MCParticle parent : particle.getParents()) {
947                 List<MCParticle> parentParticles = getRootParticleList(parent);
948                 list.addAll(parentParticles);
949             }
950             
951             // Return the compiled particle list.
952             return list;
953         }
954     }

956     /**
957      * Flattens the particle tree to a set containing both the root
958      * particle and any particles that are descended from it.
959      * @param root - The root of the particle tree.
960      * @return Returns a set containing the argument particle and all
961      * of its descendants.
962      */
963     private static final Set<MCParticle> getParticleTreeAsSet(MCParticle root) {
964         // Create a set to store the particle tree.
965         Set<MCParticle> particleSet = new HashSet<MCParticle>();
966         
967         // Add the root particle to the tree, and then recursively
968         // add any daughter particles to the tree.
969         particleSet.add(root);
970         addDaughtersToSet(root, particleSet);
971         
972         // Return the particle set.
973         return particleSet;
974     }

977      * Adds all the daughter particles of the argument to the set.
978      * Daughters of each daughter particle are then recursively added
979      * to the set as well.
980      * @param particle - The particle to add.
981      * @param set - The set to which to add the particle.
982      */
983     private static final void addDaughtersToSet(MCParticle particle, Set<MCParticle> set) {
984         // Add each daughter particle to the set, and recursively add
985         // its daughters as well.
986         for(MCParticle daughter : particle.getDaughters()) {
987             set.add(daughter);
988             addDaughtersToSet(daughter, set);
989         }

1029     @Override
1030     protected Collection<TriggeredLCIOData<?>> getOnTriggerData(double triggerTime) {
1031         // Create a list to store the extra collections.
1032         List<TriggeredLCIOData<?>> collectionsList = null;
1033         if(writeTruth) {
1034             collectionsList = new ArrayList<TriggeredLCIOData<?>>(5);
1035         } else {
1036             collectionsList = new ArrayList<TriggeredLCIOData<?>>(2);
1037         }
1038         
1039         // Readout drivers need to produce readout timestamps to
1040         // specify when they occurred in terms of simulation time.
1041         // The readout timestamp for the subdetector data should be
1042         // defined as the start simulation time of the ADC buffer.
1043         ReadoutTimestamp timestamp = new ReadoutTimestamp(getTimestampFlag(), triggerTime - (readoutOffset * 4) + 4);
1044         
1045         // Make the readout timestamp collection parameters object.
1046         LCIOCollectionFactory.setCollectionName(ReadoutTimestamp.collectionName);
1047         LCIOCollection<ReadoutTimestamp> timestampCollection = LCIOCollectionFactory.produceLCIOCollection(ReadoutTimestamp.class);
1048         TriggeredLCIOData<ReadoutTimestamp> timestampData = new TriggeredLCIOData<ReadoutTimestamp>(timestampCollection);
1049         timestampData.getData().add(timestamp);
1050         collectionsList.add(timestampData);
1051         
1052         // Instantiate some lists to store truth data, if truth is to
1053         // be output.
1054         List<SimCalorimeterHit> triggerTruthHits = null;
1055         List<LCRelation> triggerTruthRelations = null;
1056         if(writeTruth) {
1057             triggerTruthHits = new ArrayList<SimCalorimeterHit>();
1058             triggerTruthRelations = new ArrayList<LCRelation>();
1059         }
1060         
1061         // Get the appropriate collection of readout hits and output
1062         // them to the readout data manager.
1063         if(mode == 7) {
1064             List<RawCalorimeterHit> readoutHits = getMode7Hits(triggerTime);
1065             TriggeredLCIOData<RawCalorimeterHit> readoutData = new TriggeredLCIOData<RawCalorimeterHit>(mode7HitCollectionParams);
1066             readoutData.getData().addAll(readoutHits);
1067             collectionsList.add(readoutData);
1068         } else {
1069             List<RawTrackerHit> readoutHits = null;
1070             if(mode == 1) { readoutHits = getMode1Hits(triggerTime); }
1071             else { readoutHits = getMode3Hits(triggerTime); }
1072             TriggeredLCIOData<RawTrackerHit> readoutData = new TriggeredLCIOData<RawTrackerHit>(mode13HitCollectionParams);
1073             readoutData.getData().addAll(readoutHits);
1074             collectionsList.add(readoutData);
1075             
1076             // FIXME: Truth information is currently only supported for Mode-1 operation.
1077             if(writeTruth && mode == 1) {
1078                 for(RawTrackerHit hit : readoutHits) {
1079                     Collection<SimCalorimeterHit> truthHits = getTriggerTruthValues(hit.getCellID(), triggerTime);
1080                     triggerTruthHits.addAll(truthHits);
1081                     for(CalorimeterHit truthHit : truthHits) {
1082                         triggerTruthRelations.add(new BaseLCRelation(hit, truthHit));
1083                     }
1084                 }
1085             }
1086         }
1087         
1088         // Add the truth collections if they exist.
1089         if(writeTruth) {
1090             // Add the truth hits to the output collection.
1091             LCIOCollection<SimCalorimeterHit> truthHitCollection = ReadoutDataManager.getCollectionParameters(truthHitCollectionName, SimCalorimeterHit.class);
1092             TriggeredLCIOData<SimCalorimeterHit> truthData = new TriggeredLCIOData<SimCalorimeterHit>(truthHitCollection);
1093             truthData.getData().addAll(triggerTruthHits);
1094             collectionsList.add(truthData);
1095             
1096             // MC particles need to be extracted from the truth hits
1097             // and included in the readout data to ensure that the
1098             // full truth chain is available.
1099             Set<MCParticle> truthParticles = new java.util.HashSet<MCParticle>();
1100             for(SimCalorimeterHit simHit : triggerTruthHits) {
1101                 for(int i = 0; i < simHit.getMCParticleCount(); i++) {
1102                     List<MCParticle> rootParticles = getRootParticleList(simHit.getMCParticle(i));
1103                     for(MCParticle rootParticle : rootParticles) {
1104                         truthParticles.addAll(getParticleTreeAsSet(rootParticle));
1105                     }
1106                 }
1107             }
1108             
1109             // Create the truth MC particle collection.
1110             LCIOCollection<MCParticle> truthParticleCollection = ReadoutDataManager.getCollectionParameters("MCParticle", MCParticle.class);
1111             TriggeredLCIOData<MCParticle> truthParticleData = new TriggeredLCIOData<MCParticle>(truthParticleCollection);
1112             truthParticleData.getData().addAll(truthParticles);
1113             collectionsList.add(truthParticleData);
1114             
1115             // Add the truth relations to the output data.

1179      * Clones an object of type {@link org.lcsim.event.CalorimeterHit
1180      * CalorimeterHit} and returns a copy that is shifted in time by
1181      * the specified amount.
1182      * @param hit - The hit to clone.
1183      * @param newTime - The new time for the hit.
1184      * @return Returns a time-shifted hit as an object of type {@link
1185      * org.lcsim.event.CalorimeterHit CalorimeterHit}, unless the
1186      * input hit was a {@link org.lcsim.event.SimCalorimeterHit
1187      * SimCalorimeterHit} object, in which case the truth information
1188      * will be retained.
1189      */
1190     private static final CalorimeterHit cloneHitToTime(CalorimeterHit hit, double newTime) {
1191         if(hit instanceof SimCalorimeterHit) {
1192             // Cast the hit to a simulated calorimeter hit.
1193             SimCalorimeterHit simHit = (SimCalorimeterHit) hit;
1194             
1195             // Create the necessary data objects to clone the
1196             // hit.
1197             int[] pdgs = new int[simHit.getMCParticleCount()];
1198             float[] times = new float[simHit.getMCParticleCount()];
1199             float[] energies = new float[simHit.getMCParticleCount()];
1200             Object[] particles = new Object[simHit.getMCParticleCount()];
1201             for(int i = 0; i < simHit.getMCParticleCount(); i++) {
1202                 particles[i] = simHit.getMCParticle(i);
1203                 pdgs[i] = simHit.getMCParticle(i).getPDGID();
1204                 
1205                 // Note -- Despite returning the value for these
1206                 // methods as a double, they are actually stored
1207                 // internally as floats, so this case is always safe.

1209                 // of each of the contributing truth particles. This
1210                 // means that we have to give a fake truth time to
1211                 // actually get the correct hit time.
1212                 times[i] = (float) newTime;
1213                 energies[i] = (float) simHit.getContributedEnergy(i);
1214             }
1215             
1216             // Create the new hit and shift its time position.
1217             BaseSimCalorimeterHit cloneHit = new BaseSimCalorimeterHit(simHit.getCellID(), simHit.getRawEnergy(), newTime,
1218                     particles, energies, times, pdgs, simHit.getMetaData());

1222         } else {
1223             return new BaseCalorimeterHit(hit.getRawEnergy(), hit.getCorrectedEnergy(), hit.getEnergyError(), newTime,
1224                     hit.getCellID(), hit.getPositionVec(), hit.getType(), hit.getMetaData());

1241      * Generates the hits which should be output for a given trigger
1242      * time in Mode-1 format.
1243      * @param triggerTime - The trigger time.
1244      * @return Returns the readout hits for the given trigger time as
1245      * Mode-1 hits.
1246      */
1247     private List<RawTrackerHit> getMode1Hits(double triggerTime) {
1248         // Create a list to store the Mode-1 hits.
1249         List<RawTrackerHit> hits = new ArrayList<RawTrackerHit>();
1250         
1251         // Iterate over each channel.
1252         for(Long cellID : adcBufferMap.keySet()) {
1253             // Get the ADC values at the time of the trigger.
1254             short[] adcValues = getTriggerADCValues(cellID, triggerTime);
1255             
1256             // Iterate across the ADC values. If the ADC value is
1257             // sufficiently high to produce a hit, then it should be
1258             // written out.
1259             boolean isAboveThreshold = false;
1260             for(int i = 0; i < adcValues.length; i++) {
1261                 // Check that there is a threshold-crossing at some
1262                 // point in the ADC buffer.
1263                 if(adcValues[i] > getPedestalConditions(cellID) + integrationThreshold) {
1264                     isAboveThreshold = true;
1265                     break;
1266                 }
1267             }
1268             
1269             // If so, create a new hit and add it to the list.
1270             if(isAboveThreshold) {
1271                 hits.add(new BaseRawTrackerHit(cellID, 0, adcValues));
1272             }
1273         }
1274         
1275         // Return the hits.
1276         return hits;
1277     }

1280      * Generates the hits which should be output for a given trigger
1281      * time in Mode-3 format.
1282      * @param triggerTime - The trigger time.
1283      * @return Returns the readout hits for the given trigger time as
1284      * Mode-3 hits.
1285      */
1286     private List<RawTrackerHit> getMode3Hits(double triggerTime) {
1287         // Create a list to store the Mode-3 hits.
1288         List<RawTrackerHit> hits = new ArrayList<RawTrackerHit>();
1289         
1290         // Iterate across the ADC values and extract Mode-3 hits.
1291         for(Long cellID : adcBufferMap.keySet()) {
1292             int pointerOffset = 0;
1293             int numSamplesToRead = 0;
1294             int thresholdCrossing = 0;
1295             short[] adcValues = null;
1296             short[] window = getTriggerADCValues(cellID, triggerTime);
1297             
1298             for(int i = 0; i < ReadoutDataManager.getReadoutWindow(); i++) {
1299                 if(numSamplesToRead != 0) {
1300                     adcValues[adcValues.length - numSamplesToRead] = window[i - pointerOffset];
1301                     numSamplesToRead--;
1302                     if (numSamplesToRead == 0) {
1303                         hits.add(new BaseRawTrackerHit(cellID, thresholdCrossing, adcValues));
1304                     }
1305                 } else if ((i == 0 || window[i - 1] <= getPedestalConditions(cellID) + integrationThreshold) && window[i]
1306                         > getPedestalConditions(cellID) + integrationThreshold) {
1307                     thresholdCrossing = i;
1308                     pointerOffset = Math.min(numSamplesBefore, i);
1309                     numSamplesToRead = pointerOffset + Math.min(numSamplesAfter, ReadoutDataManager.getReadoutWindow() - i - pointerOffset - 1);
1310                     adcValues = new short[numSamplesToRead];
1311                 }
1312             }
1313         }
1314         
1315         // Return the hits.
1316         return hits;
1317     }

1396      * Gets a list of all truth hits that occurred in the ADC output
1397      * window around a given trigger time from the truth buffer.
1398      * @param cellID - The channel ID.
1399      * @param triggerTime - The trigger time.
1400      * @return Returns all truth hits that occurred within the ADC
1401      * readout window around the trigger time for the specified
1402      * channel.
1403      */
1404     private Collection<SimCalorimeterHit> getTriggerTruthValues(long cellID, double triggerTime) {
1405         // Calculate the offset between the current position and the
1406         // trigger time.
1407         int readoutLatency = getReadoutLatency(triggerTime);
1408         
1409         // Get the truth pipeline.
1410         ObjectRingBuffer<SimCalorimeterHit> pipeline = truthBufferMap.get(cellID);
1411         
1412         // Extract the truth for the requested channel. Note that one
1413         // extra sample is included over the range of ADC samples as
1414         // sometimes, the truth hit occurs a little earlier than may
1415         // be expected due to a delay from pulse propagation.
1416         double baseHitTime = 0;
1417         List<SimCalorimeterHit> channelHits = new ArrayList<SimCalorimeterHit>();
1418         for(int i = 0; i < readoutWindow + 4; i++) {
1419             // Hit times should be specified with respect to the
1420             // start of the readout window.
1421             for(SimCalorimeterHit hit : pipeline.getValue(-(readoutLatency - i))) {
1422                 channelHits.add((SimCalorimeterHit) cloneHitToTime(hit, baseHitTime));
1423             }
1424             
1425             // Increment the base hit time.
1426             baseHitTime += 4.0;
1427         }
1428         
1429         // Return the result.
1430         return channelHits;

1433     /**
1434      * Returns pulse amplitude at the given time (relative to hit time). Gain is
1435      * applied.
1436      *
1437      * @param time Units of ns. Relative to hit time (negative=before the start
1438      * of the pulse).
1439      * @param cellID Crystal ID as returned by hit.getCellID().
1440      * @return Amplitude, units of volts/GeV.
1441      */
1442     private double pulseAmplitude(double time, long cellID) {
1443         //normalization constant from cal gain (MeV/integral bit) to amplitude gain (amplitude bit/GeV)
1444         // Determine the gain. Gain may either be fixed across all
1445         // channels, or be obtained from the conditions database
1446         // depending on the behavior defined in the steering file.
1447         // The gain should also be normalized.
1448         double gain;
1449         if(fixedGain > 0) {
1450             gain = READOUT_PERIOD / (fixedGain * EcalUtils.MeV * ((Math.pow(2, nBit) - 1) / maxVolt));
1451         } else {
1452             gain = READOUT_PERIOD / (getGainConditions(cellID) * EcalUtils.MeV * ((Math.pow(2, nBit) - 1) / maxVolt));
1453         }
1454         
1455         // Calculate the correct pulse amplitude and return it.
1456         return gain * pulseAmplitude(time, pulseShape, tp);
1457     }

1459     /**
1460      * Calculates the amplitude of a pulse at the given time, where
1461      * the time is relative to the hit time, and for a given pulse
1462      * shape.
1463      * @param time - The time in the pulse. This is in units of ns
1464      * and is relative to the hit time. A negative value represents
1465      * the pulse shape before the hit occurs.
1466      * @param shape - The type of pulse for which the calculation is
1467      * to be performed.
1468      * @param shapingTime - A fitting parameter that influences the
1469      * shape of the pulse.
1470      * @return Returns the pulse amplitude in units of inverse ns.
1471      * The amplitude is normalized so that the pulse integral is one.
1472      */
1473     private static final double pulseAmplitude(double time, PulseShape shape, double shapingTime) {
1474         // There can not be a pulse response from a hit that has not
1475         // occurred yet, so any time before zero must produce a pulse
1476         // amplitude of zero as well.
1477         if(time <= 0.0) {
1478             return 0.0;
1479         }
1480         
1481         // Perform the calculation appropriate to the specified pulse
1482         // shape.
1483         switch (shape) {
1484             case CRRC:
1485                 // Peak Location: tp
1486                 // Peak Value:    1/(tp * e)
1487                 return ((time / (shapingTime * shapingTime)) * Math.exp(-time / shapingTime));
1488             case DoubleGaussian:
1489                 // According to measurements, the output signal can
1490                 // be fitted by two Gaussians: one for the rise of
1491                 // the signal and one for the fall.
1492                 // Peak Location: 3 * riseTime
1493                 // Peak Value:    1/norm
1494                 double norm = ((riseTime + fallTime) / 2) * Math.sqrt(2 * Math.PI); //to ensure the total integral is equal to 1: = 33.8
1495                 return funcGaus(time - 3 * riseTime, (time < 3 * riseTime) ? riseTime : fallTime) / norm;
1496             case ThreePole:
1497                 // Peak Location: 2 * tp
1498                 // Peak Value:    2/(tp * e^2)
1499                 return ((time * time / (2 * shapingTime * shapingTime * shapingTime)) * Math.exp(-time / shapingTime));
1500             default:
1501                 return 0.0;
1502         }
1503     }
