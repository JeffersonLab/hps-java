41  /**
42   * <p>
43   * This class converts EVIO to LCIO, performing an LCSim job in the same session. The processed events are then
44   * (optionally) written to disk using an LCIOWriter.
45   * <p>
46   * To run this class from the command line:<br>
47   * java -cp hps-distribution-bin.jar EvioToLcio [options] [evioFiles]
48   * <p>
49   * The available arguments can be printed using:<br>
50   * java -cp hps-distribution-bin.jar EvioToLcio -h
51   * <p>
52   * Extra arguments are treated as paths to EVIO files.
53   * <p>
54   * This class attempts to automatically configure itself for Test Run or Engineering Run based on the run numbers in the
55   * EVIO file. It will use an appropriate default detector unless one is given on the command line, and it will also use
56   * the correct event builder. It will not handle jobs correctly with files from both the Test and Engineering Run, so
57   * don't do this!
58   * <p>
59   * The run number in the conditions system can be initialized in one of three ways.<br/>
60   * <ol>
61   * <li>user specified run number in which case the conditions system is frozen for the rest of the job</li>
62   * <li>run number from an EVIO PRESTART event</li>
63   * <li>run number from a header bank in a physics event</li>
64   * </ol>
65   * <p>
66   * In the case where a file has no PRESTART event and there are header banks present, the "-m" command line option can
67   * be used to buffer a number of EVIO events. If there is a head bank found while adding these events to queue, the
68   * conditions system will be initialized from it.
69   *
70   */
71  public final class EvioToLcio {

151     /**
152      * Buffer up to <code>maxBufferSize</code> events in the <code>eventQueue</code>. This method will also initialize
153      * the conditions system using a run number if a header bank is found.
154      *
155      * @param reader the EVIO reader
156      * @param eventQueue the event queue
157      * @param maxBufferSize the maximum number of records to buffer
158      */
159     private void bufferEvents(final EvioReader reader, final EvioEventQueue eventQueue, final int maxBufferSize) {
160         EvioEvent evioEvent = null;
161         while (eventQueue.size() < maxBufferSize) {
162             try {
163                 // Read the next event.
164                 evioEvent = reader.nextEvent();
165 
166                 if (evioEvent == null) { // This catches an end of file or bad event.
167                     break;
168                 }
169 
170                 // Add the event to the queue.
171                 eventQueue.addRecord(evioEvent);
172 
173             } catch (IOException | EvioException e) {
174                 LOGGER.log(Level.SEVERE, e.getMessage(), e);
175             }
176 
177             if (evioEvent != null) {
178 
179                 // Parse the event.
180                 try {
181                     reader.parseEvent(evioEvent);
182                 } catch (final EvioException e) {
183                     LOGGER.log(Level.SEVERE, e.getMessage(), e);
184                     continue;
185                 }
186 
187                 // Get head bank from event.
188                 final BaseStructure headBank = EvioEventUtilities.getHeadBank(evioEvent);
189 
190                 // Is head bank available in this event?
191                 if (headBank != null) {
192 
193                     // Get the run number from the head bank.
194                     runNumber = headBank.getIntData()[1];
195                     LOGGER.finer("got head bank with run number " + runNumber);
196 
197                     // Is conditions system not frozen?
198                     if (!DatabaseConditionsManager.getInstance().isFrozen()) {
199                         // Check if the conditions system needs to be updated from the head bank.
200                         this.checkConditions(runNumber, false);
201                     }
202                 } else {
203                     LOGGER.finer("event " + evioEvent.getEventNumber() + " does not have a head bank");
204                 }
205             }
206         }
207         LOGGER.finer("buffered " + eventQueue.size() + " events");
208     }

210     /**
211      * Check if the conditions system and event builder need to be initialized or updated given a run number.
212      *
213      * @param runNumber The run number.
214      * @param freeze True to freeze conditions system after it is setup.
215      */
216     private void checkConditions(final int runNumber, final boolean freeze) {
217 
218         // Is the event builder uninitialized?
219         if (eventBuilder == null) {
220             // Setup event builder.
221             this.setupEventBuilder(runNumber);
222         }
223 
224         // Update the conditions system with the new run number.
225         try {
226             // This call may be ignored by the conditions system if the run number is not new.
227             ConditionsManager.defaultInstance().setDetector(detectorName, runNumber);
228         } catch (final ConditionsNotFoundException e) {
229             throw new RuntimeException("Error initializing conditions system.", e);
230         }
231 
232         if (freeze) {
233             // Freeze the conditions system so subsequent run numbers are ignored.
234             DatabaseConditionsManager.getInstance().freeze();
235         }
236     }

248     public void parse(String[] args) {
249         
250         if (args.length == 0) {
251             this.printUsage();
252         }
253         
254         final CommandLineParser parser = new PosixParser();
255         CommandLine cl = null;
256         try {
257             cl = parser.parse(OPTIONS, args);
258         } catch (final ParseException e) {
259             throw new RuntimeException("Problem parsing command line options.", e);
260         }
261         
262         if (cl.hasOption("h")) {
263             this.printUsage();
264         }
265 
266         // Add all extra arguments to the EVIO file list.
267         evioFileList = new ArrayList<String>(Arrays.asList(cl.getArgs()));
268 
269         // Process text file containing list of EVIO file paths, one per line.
270         if (cl.hasOption("f")) {
271             // Add additional EVIO files to process from text file.
272             final File file = new File(cl.getOptionValue("f"));
273             if (!file.exists()) {
274                 throw new RuntimeException("The file " + file.getPath() + " does not exist.");
275             }
276             final Path filePath = file.toPath();
277             List<String> lines = null;
278             try {
279                 lines = Files.readAllLines(filePath, Charset.defaultCharset());
280             } catch (final IOException e) {
281                 throw new RuntimeException(e);
282             }
283             for (final String line : lines) {
284                 final File evioFile = new File(line);
285                 if (evioFile.exists()) {
286                     evioFileList.add(evioFile.getPath());
287                 } else {
288                     throw new RuntimeException("The EVIO file " + line + " does not exist.");
289                 }
290             }
291         }
292 
293         // Is the EVIO file list empty?
294         if (evioFileList.isEmpty()) {
295             // There weren't any EVIO files provided on the command line so exit.
296             LOGGER.severe("No EVIO files were provided with command line arguments or -f option.");
297             this.printUsage();
298         }
299 
300         // Get the LCIO output file.
301         if (cl.hasOption("l")) {
302             LOGGER.config("set LCIO file to " + lcioFileName);
303             lcioFileName = cl.getOptionValue("l");
304         }
305 
306         // Get the steering file or resource for the LCSim job.
307         if (cl.hasOption("x")) {
308             final String lcsimXmlName = cl.getOptionValue("x");
309             if (cl.hasOption("r")) {
310                 steeringStream = EvioToLcio.class.getResourceAsStream(lcsimXmlName);
311                 if (steeringStream == null) {
312                     final IllegalArgumentException e = new IllegalArgumentException(
313                             "XML steering resource was not found.");
314                     LOGGER.log(Level.SEVERE, "LCSim steering resource " + lcsimXmlName + " was not found.", e);
315                     throw e;
316                 }
317                 LOGGER.config("using steering resource " + lcsimXmlName);
318             } else {
319                 try {
320                     steeringStream = new FileInputStream(lcsimXmlName);
321                     LOGGER.config("using steering file " + lcsimXmlName);
322                 } catch (final FileNotFoundException e) {
323                     LOGGER.severe("The XML steering file " + lcsimXmlName + " does not exist.");
324                     throw new IllegalArgumentException("Steering file does not exist.", e);
325                 }
326             }
327         }
328 
329         // Get the max number of events to process.
330         if (cl.hasOption("n")) {
331             maxEvents = Integer.valueOf(cl.getOptionValue("n"));
332             if (maxEvents <= 0) {
333                 throw new IllegalArgumentException("Value of -n option is invalid: " + maxEvents);
334             }
335             LOGGER.config("set max events to " + maxEvents);
336         }
337 
338         // Setup the LCIO writer.
339         if (lcioFileName != null) {
340             try {
341                 writer = new LCIOWriter(lcioFileName);
342                 LOGGER.config("initialized LCIO writer with file " + lcioFileName);
343             } catch (final IOException e) {
344                 LOGGER.severe("Problem initializing the LCIO writer.");
345                 throw new RuntimeException(e);
346             }
347         }
348 
349         // Process the LCSim job variable definitions, if any.
350         jobManager = new JobManager();
351                                
352         // Enable dry run because events will be processed individually.
353         jobManager.setDryRun(true);
354         
355         // Event marker printing from command line arg.
356         if (cl.hasOption("e")) {
357             eventPrintInterval = Long.parseLong(cl.getOptionValue("e"));
358             jobManager.setEventPrintInterval(eventPrintInterval);
359             LOGGER.config("Set event print interval to " + this.eventPrintInterval);
360         }
361         
362         // Enable headless mode so no plots are shown.
363         if (cl.hasOption("b")) {
364             LOGGER.config("Headless mode is enabled.  No plots will be shown.");
365             jobManager.enableHeadlessMode();
366         }
367         
368         // Set a steering variable and value using "key=value" format.
369         if (cl.hasOption("D")) {
370             final String[] steeringOptions = cl.getOptionValues("D");
371             for (final String def : steeringOptions) {
372                 final String[] s = def.split("=");
373                 if (s.length != 2) {
374                     final IllegalArgumentException e = new IllegalArgumentException("Bad variable format: " + def);
375                     LOGGER.log(Level.SEVERE, "bad variable format: " + def, e);
376                     throw e;
377                 }
378                 final String key = s[0];
379                 final String value = s[1];
380                 jobManager.addVariableDefinition(key, value);
381                 LOGGER.config("set steering variable: " + key + "=" + value);
382             }
383         }
384 
385         // Configure the LCSim job manager.
386         if (steeringStream != null) {
387             // Running a steering file with the job; use full job manager setup.
388             jobManager.setup(steeringStream);            
389         } else {
390             // No steering file is being used; configure job manager manually.
391             jobManager.initializeLoop(); /* enables event printing */
392             jobManager.getConditionsSetup().configure(); /* activates SVT detector setup */
393         }       
394         jobManager.configure();
395         LOGGER.config("LCSim job manager was successfully configured.");
396 
397         // Get the user specified detector name.
398         if (cl.hasOption("d")) {
399             detectorName = cl.getOptionValue("d");
400             LOGGER.config("User set detector to " + detectorName + " with command option.");
401         } else {
402             throw new RuntimeException("Missing -d argument with name of detector.");
403         }
404 
405         // Get the user specified run number.
406         if (cl.hasOption("R")) {
407             runNumber = Integer.parseInt(cl.getOptionValue("R"));
408             LOGGER.config("User set run number to " + runNumber + " with command option.");
409         }
410 
411         // Add conditions system tag filters.
412         if (cl.hasOption("t")) {
413             final String[] tags = cl.getOptionValues("t");
414             for (final String tag : tags) {
415                 LOGGER.config("adding conditions tag " + tag);
416                 DatabaseConditionsManager.getInstance().addTag(tag);
417             }
418         }
419 
420         // Print out the EVIO file list before the job starts.
421         final StringBuffer buff = new StringBuffer();
422         buff.append("The job will include the following EVIO files ...");
423         buff.append('\n');
424         for (final String evioFileName : evioFileList) {
425             buff.append(evioFileName);
426             buff.append('\n');
427         }
428         LOGGER.config(buff.toString());
429 
430         // Get whether to debug print XML from the EVIO events.
431         if (cl.hasOption("v")) {
432             printXml = true;
433             LOGGER.config("print XML enabled");
434         }
435 
436         // Get the maximum number of EVIO events to buffer.
437         if (cl.hasOption("m")) {
438             maxBufferSize = Integer.parseInt(cl.getOptionValue("m"));
439             LOGGER.config("max event buffer size set to " + maxBufferSize);
440         }
441 
442         // memory mapping
443         if (cl.hasOption("M")) {
444             useMemoryMapping = true;
445             LOGGER.config("EVIO reader memory mapping is enabled.");
446         }
447         
448         // skip events
449         if (cl.hasOption("s")) {
450             skipEvents = Integer.parseInt(cl.getOptionValue("s"));
451             LOGGER.config("Skip events set to " + skipEvents);
452         }        
453     }

455     /**
456      * This method will execute the EVIO to LCIO conversion and optionally process the events with LCSim Drivers from a
457      * steering file. Then the resultant LCIO events will be written to disk if this option is enabled in the command
458      * line arguments.
459      */
460     public void run() {
461 
462         // Is there a run number from the command line options?
463         if (runNumber != null) {
464             // Initialize the conditions system before the job starts and freeze it.
465             this.checkConditions(runNumber, true);
466         }
467 
468         // Loop over the input EVIO files.
469         EvioReader reader = null;
470         fileLoop: for (final String evioFileName : evioFileList) {
471 
472             // Get the next EVIO input file.
473             final File evioFile = new File(evioFileName);
474             if (!evioFile.exists()) {
475                 throw new RuntimeException("EVIO file " + evioFile.getPath() + " does not exist.");
476             }
477             LOGGER.info("Opening EVIO file " + evioFileName);
478 
479             // Open the EVIO reader.
480             try {
481                 reader = new EvioReader(evioFile, false, !useMemoryMapping);
482             } catch (final Exception e) {
483                 throw new RuntimeException("Error opening the EVIO file reader.", e);
484             }
485 
486             boolean firstEvent = true;
487             long eventTime = 0; // in ms
488             
489             // Skip some events in each file first if requested.
490             if (skipEvents > 0) {
491                 LOGGER.info("Skipping " + skipEvents + " events ...");
492                 for (int i = 0; i < skipEvents; i++) {
493                     try {
494                         EvioEvent event = reader.nextEvent();
495                         if (event == null) {
496                             LOGGER.log(Level.WARNING, "Got null event object when skipping event num " + i);
497                             //throw new RuntimeException("Got null event object when skipping events.");
498                         }
499                     } catch (EvioException | IOException e) { /* error when skipping events */
500                         LOGGER.log(Level.SEVERE, e.getMessage(), e);
501                         throw new RuntimeException(e); // Stop the job.
502                     }
503                 }
504                 LOGGER.info("Done skipping events.");
505             }
506 
507             // Loop over events.
508             final EvioEventQueue eventQueue = new EvioEventQueue(-1L, maxBufferSize);
509             eventLoop: for (;;) {                
510 
511                 // Buffer the EVIO events into the queue.
512                 this.bufferEvents(reader, eventQueue, maxBufferSize);
513 
514                 LOGGER.fine("buffered " + eventQueue.size() + " events");
515                 
516                 // Is the event queue empty?
517                 if (eventQueue.size() == 0) {
518                     // Break from the event processing loop.
519                     break eventLoop;
520                 }
521 
522                 // Loop over the EVIO events in the buffer until it is empty.
523                 recordLoop: while (eventQueue.hasNext()) {
524 
525                     // Read and parse the next EVIO event.
526                     EvioEvent evioEvent = null;
527                     try {
528                         eventQueue.next();
529                         evioEvent = (EvioEvent) eventQueue.getCurrentRecord();
530                         // The parseEvent method does not need to be called here.
531                         // The events were already parsed when buffering.
532                         if (evioEvent == null) {
533                             new RuntimeException("Failed to read EVIO event.").printStackTrace();
534                             continue recordLoop;
535                         }
536                         LOGGER.finer("processing EVIO event " + evioEvent.getEventNumber());
537                     } catch (final IOException e) {
538                         // This means the EVIO event has bad data.
539                         LOGGER.log(Level.SEVERE, e.getMessage(), e);
540                         continue recordLoop;
541                     } catch (final NoSuchRecordException e) {
542                         // This means the queue does not have any more events.
543                         // We checked hasNext() already so it should not happen.
544                         LOGGER.log(Level.SEVERE, e.getMessage(), e);
545                         break recordLoop;
546                     }
547 
548                     // Print out event XML if enabled.
549                     if (printXml) {
550                         LOGGER.info(evioEvent.toXML());
551                     }
552 
553                     // Is this a PRESTART event?
554                     if (EvioEventUtilities.isPreStartEvent(evioEvent)) {
555 
556                         LOGGER.info("got PRESTART event");
557                         
558                         // Get the pre start event's data bank.
559                         final int[] data = EvioEventUtilities.getControlEventData(evioEvent);
560 
561                         if (data == null) {
562                             // This should never happen but just ignore it.
563                             LOGGER.severe("PRESTART event is missing a data bank.");
564                         } else {
565                             // Check if conditions system needs to be updated from the pre start data.
566                             this.checkConditions(data[1], false);
567                         }
568                     }
569 
570                     // Is this an END event?
571                     if (EvioEventUtilities.isEndEvent(evioEvent)) {
572                         
573                         LOGGER.info("got END event");
574                         
575                         final int[] data = EvioEventUtilities.getControlEventData(evioEvent);
576                         if (data == null) {
577                             // This should never happen but just ignore it.
578                             LOGGER.severe("The END event is missing a data bank.");
579                         } else {
580                             final int seconds = data[0];
581                             final int totalEvents = data[2];
582                             LOGGER.info("EVIO END event with " + totalEvents + " events and " + seconds + " seconds");
583                         }
584                     }
585 
586                     // Setup state in the LCSimEventBuilder based on the EVIO control event.
587                     if (eventBuilder != null) {
588                         LOGGER.finer("event builder reading in EVIO event " + evioEvent.getEventNumber());
589                         eventBuilder.readEvioEvent(evioEvent);
590                     }
591 
592                     // Is this a physics event?
593                     if (EvioEventUtilities.isPhysicsEvent(evioEvent)) {
594 
595                         // Print physics event number, which is actually a sequence number from
596                         // the reader, not the actual event number from the data.
597                         LOGGER.finer("got physics event number " + evioEvent.getEventNumber());
598 
599                         // Is the event builder initialized?
600                         if (eventBuilder == null) {
601                             // Die here, because the event builder should be setup by now.
602                             throw new RuntimeException("The LCSimEventBuilder was never initialized.");
603                         }
604 
605                         // Build the LCIO event.
606                         final EventHeader lcioEvent = eventBuilder.makeLCSimEvent(evioEvent);
607                         eventTime = lcioEvent.getTimeStamp() / 1000000;
608                         LOGGER.finer("created LCIO event " + lcioEvent.getEventNumber() + " with timestamp " + eventTime);
609                         if (firstEvent) {
610                             LOGGER.info("first physics event time: " + eventTime / 1000 + " - " + new Date(eventTime));
611                             firstEvent = false;
612                         }
613 
614                         // Activate Driver process methods.
615                         LOGGER.finer("Job manager processing event " + lcioEvent.getEventNumber());
616                         jobManager.processEvent(lcioEvent);
617 
618                         // Write out this LCIO event.
619                         if (writer != null) {
620                             try {
621                                 writer.write(lcioEvent);
622                                 writer.flush();
623                                 LOGGER.finer("wrote LCSim event " + lcioEvent.getEventNumber());
624                             } catch (final IOException e) {
625                                 throw new RuntimeException("Error writing LCIO file.", e);
626                             }
627                             LOGGER.finer("wrote event #" + lcioEvent.getEventNumber());
628                         }
629 
630                         // Increment number of events processed.
631                         nEvents++;
632 
633                         // Check if max events was reached and end job if this is true.
634                         if (maxEvents != -1 && nEvents >= maxEvents) {
635                             LOGGER.info("maxEvents " + maxEvents + " was reached");
636                             break fileLoop;
637                         }
638                     }
639                 }
640             } // eventLoop
641             LOGGER.info("Last physics event time: " + eventTime / 1000 + " - " + new Date(eventTime));
642 
643             // Close the EVIO reader.
644             try {
645                 reader.close();
646                 LOGGER.fine("EVIO reader closed.");
647             } catch (final IOException e) {
648                 LOGGER.warning(e.getMessage());
649                 e.printStackTrace();
650             }
651         } // fileLoop
652 
653         // If the processing stopped because of max events then need to cleanup the reader.
654         if (reader != null) {
655             if (!reader.isClosed()) {
656                 // Close the EVIO reader.
657                 try {
658                     reader.close();
659                     LOGGER.fine("EVIO reader closed.");
660                 } catch (final IOException e) {
661                     LOGGER.log(Level.WARNING, e.getMessage(), e);
662                 }
663             }
664         }
665 
666         // Trigger endOfData on LCSim Drivers.
667         jobManager.finish();
668 
669         // Close the LCIO writer.
670         if (writer != null) {
671             try {
672                 writer.close();
673                 LOGGER.info("LCIO output writer closed okay.");
674             } catch (final IOException e) {
675                 e.printStackTrace();
676                 LOGGER.warning(e.getMessage());
677             }
678         }
679 
680         LOGGER.info("Job finished successfully!");
681     }

683     /**
684      * Setup the LCSimEventBuilder based on the current detector name and run number.
685      *
686      * @param detectorName The detector name to be assigned to the event builder.
687      * @param runNumber The run number which determines which event builder to use.
688      * @return The LCSimEventBuilder for the Test Run or Engineering Run.
689      */
690     private void setupEventBuilder(final int runNumber) {
691         if (DatabaseConditionsManager.isPhys2019Run(runNumber)) { 
692             LOGGER.info("using 2019 Physics Run builder."); 
693             eventBuilder = new LCSimPhys2019EventBuilder(); 
694         } else {
695             // Configure conditions system for Eng Run or default.
696             LOGGER.info("using Eng Run event builder");
697             eventBuilder = new LCSimEngRunEventBuilder();
698         }
699         final ConditionsManager conditions = ConditionsManager.defaultInstance();
700         conditions.addConditionsListener(eventBuilder);
701     }
