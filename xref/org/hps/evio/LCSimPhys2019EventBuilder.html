1   package org.hps.evio;
2   
3   import java.util.logging.Level;
4   import java.util.logging.Logger;
5   
6   import org.hps.record.evio.EvioEventUtilities;
7   import org.jlab.coda.jevio.EvioEvent;
8   import org.lcsim.event.EventHeader;
9   
10  /**
11   * This is the {@link org.hps.record.LCSimEventBuilder} implementation for the
12   * 2019 Physics Run for converting EVIO to LCIO events. 
13   * <p>
14   * This implementation makes uses a new SVT EVIO reader that was updated to 
15   * parse RSSI frames. 
16   */
17  public class LCSimPhys2019EventBuilder extends LCSimEngRunEventBuilder { 
18  
19      /**
20       * Setup logger.
21       */
22      private static final Logger LOGGER = Logger.getLogger(LCSimPhys2019EventBuilder.class.getPackage().getName());
23      
24      protected TSEvioReader tsReader = null;
25      
26      /** Constructor */
27      public LCSimPhys2019EventBuilder() {
28          super(); 
29          svtReader = new Phys2019SvtEvioReader(); 
30          vtpReader = new VTPEvioReader();
31          tsReader = new TSEvioReader();
32  
33          // in 2019 the RF signal changed crates relative to previous runs:
34          ecalReader.setRfBankTag(0x27);
35  
36          svtEventFlagger = new SvtEventFlagger();  
37      }
38      
39      
40      /**
41       * Make an lcsim event from EVIO data.
42       *
43       * @param evioEvent the input EVIO event
44       */
45      @Override
46      public EventHeader makeLCSimEvent(final EvioEvent evioEvent) {
47  
48          LOGGER.finest("creating LCSim event from EVIO event " + evioEvent.getEventNumber());
49  
50          if (!EvioEventUtilities.isPhysicsEvent(evioEvent)) {
51              throw new RuntimeException("Not a physics event: event tag " + evioEvent.getHeader().getTag());
52          }
53  
54          // Create a new LCSimEvent.
55          final EventHeader lcsimEvent = super.makeLCSimEvent(evioEvent);
56          LOGGER.finest("created new LCSim event " + lcsimEvent.getEventNumber());
57  
58          // Make TS collection
59          // into one list.
60          try {
61              tsReader.makeHits(evioEvent, lcsimEvent);
62          } catch (final Exception e) {
63              LOGGER.log(Level.SEVERE, "Error reading TS bank", e);
64          }
65  
66          //if (svtEventFlagger != null) this.svtEventFlagger.writeFlags(lcsimEvent);
67  
68          return lcsimEvent;
69      }
70      
71  }
