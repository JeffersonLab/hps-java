143     /**
144      *  Make {@link RawTrackerHit}s out of all sample sets in an SVT EVIO bank
145      *  and put them into an LCSim event.
146      *
147      *  
148      *  @param event - EVIO event to process
149      *  @param lcsimEvent - LCSim event to put collections into 
150      *  @return true if the raw hits were created successfully, false otherwise 
151      * @throws SvtEvioReaderException 
152      */
153     @Override
154     public boolean makeHits(EvioEvent event, EventHeader lcsimEvent) throws SvtEvioReaderException {
155 
156         LOGGER.finest("Physics Event: " + event.toString());
157         
158         // Retrieve the data banks encapsulated by the physics bank.  The ROC
159         // bank range is set in the subclass.
160         List<BaseStructure> dataBanks 
161                 = SvtEvioUtils.getDataBanks(event, 
162                     this.getMinRocBankTag(), this.getMaxRocBankTag(),
163                     this.getMinDataBankTag(), this.getMaxDataBankTag());
164         
165         // Return false if data banks weren't found
166         if (dataBanks.isEmpty()) return false;  
167         //System.out.println("Total data banks found: " + dataBanks.size());
168     
169         // Setup the DAQ map if it's not setup
170         if (!this.isDaqMapSetup)
171             this.setupDaqMap(lcsimEvent.getDetector().getSubdetector(
172                     SUBDETECTOR_NAME));
173 
174         List<RawTrackerHit> rawHits = new ArrayList<RawTrackerHit>();
175 
176         // Loop over all of the data banks contained by the ROC banks and 
177         // processed them
178         for (BaseStructure dataBank : dataBanks) {
179 
180             //System.out.println("Processing data bank: " + dataBank.toString());
181             
182             // Get the set of 32 bit words encapsulated by the bank.  
183             int[] data = dataBank.getIntData();
184             //System.out.println("Total number of integers contained by the data bank: " + data.length);
185            
186             // Process the data
187             rawHits.addAll(this.makeHits(dataBank.getHeader().getNumber(), data)); 
188 
189         }
190         
191         //System.out.println("Total number of RawTrackerHits created: " + rawHits.size());
192 
193         // Turn on 64-bit cell ID.
194         int flag = LCIOUtil.bitSet(0, 31, true);
195         // Add the collection of raw hits to the LCSim event
196         lcsimEvent.put(SVT_HIT_COLLECTION_NAME, rawHits, RawTrackerHit.class, flag, READOUT_NAME);
197 
198         
199         return true;
200     }

212     /**
213      * Extract the header information and store it in a {@link SvtHeaderDataInfo} object.
214      * @param num - bank num (ROC id)
215      * @param data - SVT data block.
216      * @return the {@link SvtHeaderDataInfo}.
217      */
218     protected SvtHeaderDataInfo extractSvtHeader(int num, int[] data) {
219         // Extract the header information
220         int svtHeader = SvtEvioUtils.getSvtHeader(data);
221         // Extract the tail information
222         int svtTail = SvtEvioUtils.getSvtTail(data);
223         return new SvtHeaderDataInfo(num, svtHeader, svtTail);
224 
225     }

227     /**
228      * Copy the multisample header data for the samples into a long array.
229      * @param samples
230      * @param index
231      * @param multisampleHeaderData
232      * @return the length of the extracted samples or 0 if not a multisample header
233      */
234     protected int extractMultisampleHeaderData(int[] samples, int index, int[] multisampleHeaderData) {
235         //System.out.println("extractMultisampleHeaderData: index " + index);
236         if( SvtEvioUtils.isMultisampleHeader(samples) && !SvtEvioUtils.isMultisampleTail(samples) ) {
237             //System.out.println("extractMultisampleHeaderData: this is a multisample header so add the words to index " + index);
238             System.arraycopy(samples, 0, multisampleHeaderData, index, samples.length);
239             return samples.length;
240         } else {
241             //System.out.println("extractMultisampleHeaderData: this is a NOT multisample header ");
242             return 0;
243         }
244     }

246     /**
247      * Checks that the SVT header data count is consistent with the bank size.
248      * @param sampleCount - sample count from the size.
249      * @param headerData - header extracted from the bank.
250      * @throws SvtEvioHeaderException
251      */
252     protected void checkSvtSampleCount(int sampleCount, SvtHeaderDataInfo headerData) throws SvtEvioHeaderException {
253         if( sampleCount != SvtEvioUtils.getSvtTailMultisampleCount(headerData.getTail())*4)
254             throw new SvtEvioHeaderException("ROC " + headerData.getNum() + " multisample count " + sampleCount + " is not consistent with bank size " + SvtEvioUtils.getSvtTailMultisampleCount(headerData.getTail()));
255     }

257     /**
258      * Add the multisample headers to the {@link SvtHeaderDataInfo} object.
259      * @param headerData - object to add multisample headers to.
260      * @param n - number of multisample headers
261      * @param multisampleHeaders - multisample headers to copy
262      */
263     protected void setMultiSampleHeaders(SvtHeaderDataInfo headerData, int n, int[] multisampleHeaders) {
264         //copy out the headers that are non-zero
265         int[] vals = new int[n];
266         System.arraycopy(multisampleHeaders, 0, vals, 0, n);
267         //logger.info("setMultiSampleHeaders: adding " + vals.length + " multisample headers");
268         headerData.setMultisampleHeaders(vals);
269     }

285     
286     /**
287      *  Make a {@link RawTrackerHit} from a set of samples.
288      * 
289      *  @param data : Sample block of data
290      *  @param channel : Channel number associated with these samples
291      *  @return A raw hit
292      */
293     protected RawTrackerHit makeHit(int[] data, int channel) { 
294 
295         // Get the sensor associated with this sample
296         HpsSiSensor sensor = this.getSensor(data);
297         
298         if (sensor != null) {
299             // Use the channel number to create the cell ID
300             long cellID = sensor.makeChannelID(channel);
301             
302             // Set the hit time.  For now this will be zero
303             int hitTime = 0;
304         
305             // Create and return a RawTrackerHit
306             return new BaseRawTrackerHit(hitTime, cellID, SvtEvioUtils.getSamples(data), null, sensor);
307         }else {
308             LOGGER.warning("makeHit: Bad sensor codes in hit evio data. " );
309             return(null);
310         }
311     }
