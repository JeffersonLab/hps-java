24      static public void main(String[] args) {
25  
26          Options options = new Options();
27          options.addOption(new Option("q", false, "quiet - don't print event contents"));
28          options.addOption(new Option("c", false, "print control events"));
29          options.addOption(new Option("t", false, "print event timestamps"));
30          options.addOption(new Option("s", false, "sequential read (not mem-mapped)"));
31  
32          // Parse the command line options.
33          if (args.length == 0) {
34              printUsage(options);
35          }
36          final CommandLineParser parser = new PosixParser();
37          CommandLine cl = null;
38          try {
39              cl = parser.parse(options, args);
40          } catch (final ParseException e) {
41              throw new RuntimeException("Problem parsing command line options.", e);
42          }
43  
44          if (cl.getArgs().length < 1) {
45              throw new RuntimeException("Missing EVIO file name.");
46          }
47  
48          boolean quiet = cl.hasOption("q");
49          boolean printControlEvents = cl.hasOption("c");
50          boolean seqRead = cl.hasOption("s");
51          boolean printTimestamps = cl.hasOption("t");
52  
53          IntBankDefinition headBankDefinition = new IntBankDefinition(HeadBankData.class, new int[]{0x2e, 0xe10f});
54          IntBankDefinition tiBankDefinition = new IntBankDefinition(TIData.class, new int[]{0x2e, 0xe10a});
55  
56  //        String evioFileName = args[0];
57          for (String evioFileName : cl.getArgs()) {
58              File evioFile = new File(evioFileName);
59              if (!evioFile.exists()) {
60                  throw new RuntimeException("File " + evioFileName + " does not exist.");
61              }
62              System.out.println("Opened file " + evioFileName);
63              try {
64                  org.jlab.coda.jevio.EvioReader reader = new org.jlab.coda.jevio.EvioReader(evioFile, true, seqRead);
65                  int eventN = 1;
66                  int badEvents = 0;
67                  int[] lastData = new int[]{0, 0, 0, 0, 0};
68                  long minDelta = 0, maxDelta = 0;
69                  long lastTI = 0;
70                  fileLoop:
71                  while (true) {
72                      if (!quiet) {
73                          System.out.println("Reading event " + eventN);
74                      }
75                      try {
76                          EvioEvent event = reader.nextEvent();
77                          if (event == null) {
78                              break fileLoop;
79                          }
80                          reader.parseEvent(event);
81                          //printBytes(event.getRawBytes()); // DEBUG
82                          if (!quiet) {
83                              System.out.println("Successfully read event " + eventN);// + " which contains " + event.getTotalBytes() + " bytes.");
84                              printBank(event, "");
85                          }
86                          if (printControlEvents && EvioEventUtilities.isControlEvent(event) && !EvioEventUtilities.isEpicsEvent(event)) {
87                              int[] controlEventData = EvioEventUtilities.getControlEventData(event);
88                              if (controlEventData == null) {
89                                  printBank(event, "");
90                              }
91                              System.out.print(event.getHeader().getTag() + "\t");
92  
93                              for (int i = 0; i < controlEventData.length; i++) {
94                                  System.out.print(controlEventData[i] + "\t");
95                              }
96                              Date timestamp = new Date(controlEventData[0] * 1000L);
97                              System.out.println(timestamp);
98                          }
99  
100                         if (printTimestamps) {
101                             int thisTimestamp = 0;
102                             BaseStructure headBank = headBankDefinition.findBank(event);
103                             if (headBank != null) {
104                                 int[] data = headBank.getIntData();
105                                 thisTimestamp = data[3];
106                                 if (data[3] != 0) {
107                                     if (lastData[3] == 0) {
108                                         System.out.print("first_head\t");
109                                         printInts(data);
110                                     }
111                                     lastData = data;
112                                 }
113                             }
114                             BaseStructure tiBank = tiBankDefinition.findBank(event);
115                             if (tiBank != null) {
116                                 TIData tiData = new TIData(tiBank.getIntData());
117                                 if (lastTI == 0) {
118                                     System.out.format("first_TItime\t%d\n", tiData.getTime());
119                                 }
120                                 lastTI = tiData.getTime();
121                                 if (thisTimestamp != 0) {
122                                     long delta = thisTimestamp * 1000000000L - tiData.getTime();
123                                     if (minDelta == 0 || minDelta > delta) {
124                                         minDelta = delta;
125                                     }
126                                     if (maxDelta == 0 || maxDelta < delta) {
127                                         maxDelta = delta;
128                                     }
129 //                                    System.out.format("%d %d %d %d %d %d\n", thisTimestamp, tiData.getTime(), delta, minDelta, maxDelta, maxDelta-minDelta);
130                                 }
131                             }
132                         }
133                     } catch (Exception e) {
134                         System.out.println("Caught Exception processing event " + eventN + " which was...");
135                         e.printStackTrace();
136                         ++badEvents;
137                     }
138                     ++eventN;
139                     if (!quiet) {
140                         System.out.println("-------");
141                     }
142                 }
143                 System.out.println("There were " + badEvents + " bad events out of " + eventN + " total.");
144                 if (printTimestamps) {
145                     System.out.print("last_head\t");
146                     printInts(lastData);
147                     System.out.format("last_TItime\t%d\n", lastTI);
148                     System.out.format("ti_offset\t%d\t%d\t%d\n", minDelta, maxDelta, maxDelta - minDelta);
149                 }
150                 reader.close();
151             } catch (Exception e) {
152 //                throw new RuntimeException(e);
153                 e.printStackTrace();
154             }
155         }
156     }

158     private static void printInts(int[] data) {
159         for (int i = 0; i < data.length; i++) {
160             System.out.format("%d\t", data[i]);
161         }
162         System.out.println();
163     }

172     private static void printBank(BaseStructure bank, String indent) throws EvioException {
173         System.out.println(indent + "Bank contains " + bank.getTotalBytes() + " bytes.");
174         System.out.println(indent + "Bank has " + bank.getChildCount() + " sub-banks.");
175         System.out.format(indent + "Bank tag: 0x%x length: %d type: %s num: %d\n", bank.getHeader().getTag(), bank.getHeader().getLength(), bank.getHeader().getDataType(), bank.getHeader().getNumber());
176         if (bank.getChildCount() > 0) {
177             for (BaseStructure child : bank.getChildrenList()) {
178                 printBank(child, indent + "\t");
179             }
180         }
181         if (bank.getHeader().getDataType() == DataType.COMPOSITE) {
182 //            for (CompositeData cdata : bank.getCompositeData()) {
183             CompositeData cdatalist[] = bank.getCompositeData();
184             for (CompositeData cdata : cdatalist) {
185                 switch (bank.getHeader().getTag()) {
186                     case 0xe101:
187                         printWindow(cdata, indent + "\t");
188                         break;
189                     case 0xe102:
190                         printComposite(cdata, indent + "\t");
191                         break;
192                     case 0xe103:
193                         printComposite(cdata, indent + "\t");
194                         break;
195                 }
196             }
197 //            }
198         }
199         if (bank.getHeader().getDataType() == DataType.UINT32) {
200             int[] data = bank.getIntData();
201             if (data.length < 100) {
202                 for (int i = 0; i < data.length; i++) {
203                     System.out.format(indent + "0x%x\n", data[i]);
204                 }
205             }
206         }
207     }

214     private static void printWindow(CompositeData cdata, String indent) {
215         while (cdata.index() + 1 < cdata.getItems().size()) {
216             System.out.println(indent + "Byte count: " + cdata.getRawBytes().length);
217             System.out.println(indent + "Slot: " + cdata.getByte());
218             System.out.println(indent + "Trigger: " + cdata.getInt());
219             System.out.println(indent + "Timestamp: " + cdata.getLong());
220             int nchannels = cdata.getNValue();
221             System.out.println(indent + "NChannels: " + nchannels);
222             for (int j = 0; j < nchannels; j++) {
223                 System.out.println(indent + "Channel: " + cdata.getByte());
224                 int nSamples = cdata.getNValue();
225                 System.out.println(indent + "NSamples: " + nSamples);
226                 for (int i = 0; i < nSamples; i++) {
227                     cdata.getShort();
228                 }
229             }
230         }
