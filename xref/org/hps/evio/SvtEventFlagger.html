61      public void writeFlags(EventHeader event) {
62          Date eventDate = getEventTimeStamp(event);
63          Long evTime = event.getTimeStamp();
64          Long trigArrT0 = evTime + trigDel + (24 - (evTime + trigPhase0)%24 );
65          Long trigSyncTime0 = trigArrT0 + (840 - (trigArrT0 + syncPhase0)%840);
66          Long trigArrT1 = evTime + trigDel + (24 - (evTime + trigPhase1)%24 );
67          Long trigSyncTime1 = trigArrT1 + (840 - (trigArrT1 + syncPhase1)%840);
68  
69          // Remove reads that have already happened
70          for(int ri = readTimes0.size()-1; ri >= 0; ri--) {
71              if (readTimes0.size() == 0) break;
72              if (readTimes0.get(ri) < evTime - 840) {
73                  readTimes0.remove(ri);
74                  readEvTimes0.remove(ri);
75              }
76          }
77          for(int ri = readTimes1.size()-1; ri >= 0; ri--) {
78              if (readTimes1.size() == 0) break;
79              if (readTimes1.get(ri) < evTime - 840) {
80                  readTimes1.remove(ri);
81                  readEvTimes1.remove(ri);
82              }
83          }
84  
85          // Add new reads to buffer from this trigger
86          fillAPVbuffer(readTimes0, readEvTimes0, evTime, trigSyncTime0);
87          fillAPVbuffer(readTimes1, readEvTimes1, evTime, trigSyncTime1);
88  
89          // Check if this event overlaps a read in the buffer
90          readoutOverlapGood = true;
91          if (readTimes0.get(0) - evTime < cut0H && readTimes0.get(0) - evTime > cut0L) {
92              readoutOverlapGood = false;
93          }
94          if (readTimes1.get(0) - evTime < cut1H && readTimes1.get(0) - evTime > cut1L) {
95              readoutOverlapGood = false;
96          }
97  
98          //Check date against the HV position state for that time
99          if (eventDate != null) {
100             biasGood = false;
101             if (svtBiasConstants != null) {
102                 SvtBiasConstant biasConstant = svtBiasConstants.find(eventDate);
103                 if (biasConstant != null) {
104                     biasGood = true;
105                 }
106             }
107 
108             positionGood = false;
109             if (svtPositionConstants != null) {
110                 SvtMotorPosition positionConstant = svtPositionConstants.find(eventDate);
111                 if (positionConstant != null) {
112 //                    System.out.format("%f %f %f %f\n", positionConstant.getBottom(), nominalAngleBottom, positionConstant.getTop(), nominalAngleTop);
113                     if (Math.abs(positionConstant.getBottom() - nominalAngleBottom) < angleTolerance && Math.abs(positionConstant.getTop() - nominalAngleTop) < angleTolerance) {
114                         positionGood = true;
115                     }
116                 }
117             }
118         }
119 
120         latencyGood = false;
121         if (svtTimingConstants != null) {
122             if (svtTimingConstants.getOffsetTime() <= 27) {
123                 latencyGood = true;
124             } else {
125                 if (((event.getTimeStamp() - 4 * svtTimingConstants.getOffsetPhase()) % 24) < 16) {
126                     latencyGood = true;
127                 }
128             }
129         }
130 
131         burstmodeNoiseGood = isBurstmodeNoiseGood(event);
132 
133         event.getIntegerParameters().put("svt_bias_good", new int[]{biasGood ? 1 : 0});
134         event.getIntegerParameters().put("svt_position_good", new int[]{positionGood ? 1 : 0});
135         event.getIntegerParameters().put("svt_burstmode_noise_good", new int[]{burstmodeNoiseGood ? 1 : 0});
136         event.getIntegerParameters().put("svt_latency_good", new int[]{latencyGood ? 1 : 0});
137         event.getIntegerParameters().put("svt_readout_overlap_good", new int[]{readoutOverlapGood ? 1 : 0});
138     }

140     private Date getEventTimeStamp(EventHeader event) {
141         List<GenericObject> intDataCollection = event.get(GenericObject.class, "TriggerBank");
142         for (GenericObject data : intDataCollection) {
143             if (AbstractIntData.getTag(data) == HeadBankData.BANK_TAG) {
144                 Date date = HeadBankData.getDate(data);
145                 if (date != null) {
146                     return date;
147                 }
148             }
149         }
150         return null;
151     }

153     public void initialize() {
154         try {
155             svtBiasConstants = DatabaseConditionsManager.getInstance().getCachedConditions(SvtBiasConstant.SvtBiasConstantCollection.class, "svt_bias_constants").getCachedData();
156         } catch (Exception e) {
157             svtBiasConstants = null;
158         }
159         try {
160             svtPositionConstants = DatabaseConditionsManager.getInstance().getCachedConditions(SvtMotorPosition.SvtMotorPositionCollection.class, "svt_motor_positions").getCachedData();
161             SvtAlignmentConstant.SvtAlignmentConstantCollection alignmentConstants = DatabaseConditionsManager.getInstance().getCachedConditions(SvtAlignmentConstant.SvtAlignmentConstantCollection.class, "svt_alignments").getCachedData();
162 
163             for (final SvtAlignmentConstant constant : alignmentConstants) {
164                 switch (constant.getParameter()) {
165                     case 13100:
166 //                    System.out.format("nominal top angle: %f\n", constant.getValue());
167                         nominalAngleTop = constant.getValue();
168                         break;
169                     case 23100:
170 //                    System.out.format("nominal bottom angle: %f\n", constant.getValue());
171                         nominalAngleBottom = -constant.getValue();
172                         break;
173                 }
174             }
175         } catch (Exception e) {
176             svtPositionConstants = null;
177         }
178 
179         try {
180             svtTimingConstants = DatabaseConditionsManager.getInstance().getCachedConditions(SvtTimingConstants.SvtTimingConstantsCollection.class, "svt_timing_constants").getCachedData().get(0);
181         } catch (Exception e) {
182             svtTimingConstants = null;
183         }
184 
185         try {
186             svtSyncPhaseColl = DatabaseConditionsManager.getInstance().getCachedConditions(SvtReadoutSyncPhase.SvtReadoutSyncPhaseCollection.class, "svt_readout_sync_phases").getCachedData();
187             trigDel = Long.valueOf(svtSyncPhaseColl.get(0).getTrigDel().intValue());
188             syncPhase0 = Long.valueOf(svtSyncPhaseColl.get(0).getPhase0().intValue());
189             syncPhase1 = Long.valueOf(svtSyncPhaseColl.get(0).getPhase1().intValue());
190             cut0L = Long.valueOf(svtSyncPhaseColl.get(0).getCut0L().intValue());
191             cut0H = Long.valueOf(svtSyncPhaseColl.get(0).getCut0H().intValue());
192             cut1L = Long.valueOf(svtSyncPhaseColl.get(0).getCut1L().intValue());
193             cut1H = Long.valueOf(svtSyncPhaseColl.get(0).getCut1H().intValue());
194             trigPhase0 = syncPhase0%24;
195             trigPhase1 = syncPhase1%24;
196 
197             LOG.config("svt_readout_sync_phases found phase0: " + syncPhase0);
198         } catch (Exception e) {
199             svtSyncPhaseColl = null;
200             LOG.config("svt_readout_sync_phases not found.");
201         }
202

205     private static boolean isBurstmodeNoiseGood(EventHeader event) {
206         if (event.hasCollection(RawTrackerHit.class, "SVTRawTrackerHits")) {
207             // Get RawTrackerHit collection from event.
208             List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
209             return countSmallHits(rawTrackerHits) <= 3;
210         }
211         return false;
212     }

214     private static int countSmallHits(List<RawTrackerHit> rawHits) {
215         int smallHitCount = 0;
216         Map<HpsSiSensor, Set<Integer>> hitMap = new HashMap<HpsSiSensor, Set<Integer>>();
217 
218         for (RawTrackerHit hit : rawHits) {
219             HpsSiSensor sensor = (HpsSiSensor) hit.getDetectorElement();
220             Set<Integer> hitStrips = hitMap.get(sensor);
221             if (hitStrips == null) {
222                 hitStrips = new HashSet<Integer>();
223                 hitMap.put(sensor, hitStrips);
224             }
225             int strip = hit.getIdentifierFieldValue("strip");
226             hitStrips.add(strip);
227         }
228 
229         for (RawTrackerHit hit : rawHits) {
230             if (isSmallHit(hitMap, hit)) {
231                 smallHitCount++;
232             }
233         }
234         return smallHitCount;
235     }

237     private static boolean isSmallHit(Map<HpsSiSensor, Set<Integer>> hitMap, RawTrackerHit hit) {
238         HpsSiSensor sensor = (HpsSiSensor) hit.getDetectorElement();
239         int strip = hit.getIdentifierFieldValue("strip");
240         double pedestal = sensor.getPedestal(strip, 0);
241         double noise = sensor.getNoise(strip, 0);
242 
243         if (hitMap.get(sensor) != null && (hitMap.get(sensor).contains(strip - 1) || hitMap.get(sensor).contains(strip + 1))) {
244             return false;
245         }
246         for (int i = 0; i < 6; i++) {
247             if (hit.getADCValues()[i] > pedestal + 4.0 * noise) {
248                 return false;
249             }
250         }
251         return true;
252     }

263     public static void AddHeaderInfoToMetaData(List<SvtHeaderDataInfo> headers, EventHeader lcsimEvent) {
264         int[] svtHeaders = new int[headers.size()];
265         int[] svtTails = new int[headers.size()];
266         for (int iSvtHeader = 0; iSvtHeader < headers.size(); ++iSvtHeader) {
267             svtHeaders[iSvtHeader] = headers.get(iSvtHeader).getHeader();
268             svtTails[iSvtHeader] = headers.get(iSvtHeader).getTail();
269 
270             lcsimEvent.getIntegerParameters().put("svt_event_header_roc" + headers.get(iSvtHeader).getNum(), new int[]{headers.get(iSvtHeader).getHeader()});
271             lcsimEvent.getIntegerParameters().put("svt_event_tail_roc" + headers.get(iSvtHeader).getNum(), new int[]{headers.get(iSvtHeader).getTail()});
272 
273             int nMS = headers.get(iSvtHeader).getNumberOfMultisampleHeaders();
274             int[] multisampleHeadersArray = new int[4 * nMS];
275             for (int iMS = 0; iMS < nMS; ++iMS) {
276                 int[] multisampleHeader = headers.get(iSvtHeader).getMultisampleHeader(iMS);
277                 System.arraycopy(multisampleHeader, 0, multisampleHeadersArray, iMS * 4, multisampleHeader.length);
278             }
279             lcsimEvent.getIntegerParameters().put("svt_multisample_headers_roc" + headers.get(iSvtHeader).getNum(), multisampleHeadersArray);
280         }

284     private static final Pattern rocIdPattern = Pattern.compile("svt_.*_roc(\\d+)");
285 
286     public static int getRocFromSvtHeaderName(String seq) {
287         Matcher m = rocIdPattern.matcher(seq);
288         if (m == null) {
289             throw new RuntimeException("null matcher, don't think this should happen");
290         }
291         if (!m.matches()) {
292             return -1;
293         } else {
294             return Integer.parseInt(m.group(1));
295         }
296     }

298     public static List<SvtHeaderDataInfo> getHeaderInfoToMetaData(EventHeader lcsimEvent) {
299         Map<Integer, Integer> headers = new HashMap<Integer, Integer>();
300         Map<Integer, Integer> tails = new HashMap<Integer, Integer>();
301         Map<Integer, Integer[]> multisampleHeaders = new HashMap<Integer, Integer[]>();
302 
303         for (Map.Entry<String, int[]> entry : lcsimEvent.getIntegerParameters().entrySet()) {
304 
305             int roc = getRocFromSvtHeaderName(entry.getKey());
306 
307             if (roc == -1) {
308                 continue;
309             }
310             //LOGGER.logger.fine("processing entry \"" + entry.getKey()+ "\"" + " for roc "  + roc);
311             int[] value = entry.getValue();
312 
313             if (entry.getKey().contains("svt_event_header_roc")) {
314                 headers.put(roc, value[0]);
315             }
316 
317             if (entry.getKey().contains("svt_event_tail_roc")) {
318                 tails.put(roc, value[0]);
319             }
320 
321             // really need to copy?
322             if (entry.getKey().contains("svt_multisample_headers_roc")) {
323                 Integer[] tmp = ArrayUtils.toObject(value); //new Integer[value.length];
324                 multisampleHeaders.put(roc, tmp);
325             }
326 
327         }
328 
329         // create the new objects
330         List<SvtHeaderDataInfo> headerDataInfo = new ArrayList<SvtHeaderDataInfo>();
331         for (Integer roc : headers.keySet()) {
332             int header = headers.get(roc);
333             int tail = tails.get(roc);
334             Integer[] ms = multisampleHeaders.get(roc);
335             headerDataInfo.add(new SvtHeaderDataInfo(roc, header, tail, ms));
336         }
337 
338         return headerDataInfo;
339 
340     }

342     public static int fillAPVbuffer(Vector<Long> readTimes, Vector<Long> readEvTimes, Long evT, Long trigSyncT) {
343         for (int ss = 0; ss < 6; ss++) {
344             if (ss > 0) {
345                 readTimes.add(readTimes.get(readTimes.size()-1) + 3360);
346                 readEvTimes.add(evT);
347             } else {
348                 if (readTimes.size() == 0) {
349                     readTimes.add(trigSyncT);
350                     readEvTimes.add(evT);
351                 } else if (readTimes.get(readTimes.size()-1) + 3360 > trigSyncT) {
352                     readTimes.add(readTimes.get(readTimes.size()-1) + 3360);
353                     readEvTimes.add(evT);
354                 } else {
355                     readTimes.add(trigSyncT);
356                     readEvTimes.add(evT);
357                 }
358             }
359         }
360         return readTimes.size();
361     }
