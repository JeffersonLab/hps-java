1   package org.hps.evio;
2   
3   import java.util.List;
4   import java.util.ArrayList;
5   
6   import org.apache.commons.lang3.ArrayUtils;
7   import org.jlab.coda.jevio.BaseStructure;
8   import org.jlab.coda.jevio.BaseStructureHeader;
9   import org.hps.record.evio.EvioEventConstants;
10  import org.jlab.coda.jevio.EvioEvent;
11  
12  import org.lcsim.event.EventHeader;
13  
14  /**
15   * This code reads VTP banks from the EVIO, and writes it's values as a genericObject.
16   * In GenericObject there are two arrays. One represents Trigger words, and
17   * the other represents rocIDs. In particular we want to know the rocID of i-th  VTP word,
18   * the we should use the i-th value of the array "rocID_vals" (see description in the code).
19   */
20  public class VTPEvioReader extends EvioReader {
21  
22      public VTPEvioReader() {
23      }
24  
25      @Override
26      public boolean makeHits(EvioEvent event, EventHeader lcsimEvent) {
27  
28          boolean foundVTP = false;
29          int[] vtp_vals = new int[]{};           // This array contains VTP words (as integer) for the given event
30          int[] rocID_vals = new int[]{};         // This array contains rocIDs for the same index VTP word
31  
32          // ======== Defining GenericObjects for words, and rocIDs
33          VTPGenericObject vtp_generic = new VTPGenericObject();
34          VTPGenericObject vtp_generic_rocid = new VTPGenericObject();
35          
36          
37          // ====== In order to write generic object in the file, 1st we should add generic objects
38          // ====== in the List, so we will add above genericObjects into the list below
39          List<VTPGenericObject> vtp_list = new ArrayList();
40  
41          
42          // ===== Looping over all banks in the EVIO
43          for (BaseStructure bank : event.getChildrenList()) {
44              BaseStructureHeader header = bank.getHeader();
45              int rocID = header.getTag();  // getting the tag of the bnk
46              
47              // We want to select banks with tag=rocID_Top or rocID_Bot===== As of Jun-3-2019, these rocIDs are
48              // 60011 (Top) and 60012 (Bot)
49              if (rocID == EvioEventConstants.VTP_TOP_RocID || rocID == EvioEventConstants.VTP_BOT_RocID ) {
50  
51                  for (BaseStructure childBank : bank.getChildrenList()) {
52  
53                      // ==== Althought all the subbanks here should be VTP, it is worth to check, if
54                      // ==== the Bank tag is VTP Bank tag
55                      
56                      if (childBank.getHeader().getTag() == EvioEventConstants.VTP_BANK_TAG) {
57                          int[] vals = childBank.getIntData();
58  
59                          // Concatinating arrays "vtp_vals" to the array from this bank
60                          vtp_vals = ArrayUtils.addAll(vtp_vals, vals);
61                          
62                          int len_vals = vals.length;
63                          
64                          // ==== A temporary array which will contin rocIDs, which by the way has same values for all
65                          // ==== elements in banks under the same rocID
66                          int[] rocid_values = new int[len_vals];
67                          
68                          for( int i_val = 0 ; i_val < len_vals; i_val++ ){
69                              rocid_values[i_val] = rocID;
70                          }
71                          
72                          // Concatinating array rocID_vals with the rocid_values (which represents rocIDs only for the given bank)
73                          rocID_vals = ArrayUtils.addAll(rocID_vals, rocid_values);
74                      }
75                  }
76  
77                  foundVTP = true;
78              }
79          }
80  
81  
82          // Filling generic objects with corresponding arrays of Integers
83          vtp_generic.setValues(vtp_vals);
84          vtp_generic_rocid.setValues(rocID_vals);
85          
86          // Adding generic object to the list
87          vtp_list.add(vtp_generic);
88          vtp_list.add(vtp_generic_rocid);
89          
90          // Writing VTP data into the event
91          lcsimEvent.put("VTPBank", vtp_list, VTPGenericObject.class, 0);
92          
93          return foundVTP;
94      }
95  
96  }
