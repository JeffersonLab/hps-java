1   package org.hps.evio;
2   
3   import org.jlab.coda.jevio.EvioEvent;
4   
5   import org.lcsim.event.EventHeader;
6   
7   /**
8    *  Abstract class containing shared methods used by EVIO readers.
9    */
10  public abstract class EvioReader {
11  
12      // Debug flag
13      protected boolean debug = false;
14      
15      // Name of the hit collection that will be created
16      protected String hitCollectionName = null;
17  
18      /**
19       *  Make a LCIO hit collection (e.g. {@link org.lcsim.event.RawTrackerHit}, 
20       *  {@link org.lcsim.event.CalorimeterHit} from raw EVIO data.
21       * 
22       *  @param event : The EVIO event to read the raw data from
23       *  @param lcsimEvent : The LCSim event to write the collections to
24       *  @return True if the appropriate EVIO bank is found, false otherwise 
25       * @throws Exception 
26       * 
27       */
28      abstract boolean makeHits(EvioEvent event, EventHeader lcsimEvent) throws Exception;
29  
30      /**
31       *  Set the hit collection name.
32       * 
33       *  @param hitCollectionName : Name of the hit collection
34       */
35      public void setHitCollectionName(String hitCollectionName) {
36          this.hitCollectionName = hitCollectionName;
37      }
38  
39      /**
40       *  Enable/disable debug output.
41       * 
42       *  @param debug : Set to true to enable, false to disable.
43       */
44      public void setDebug(boolean debug) {
45          this.debug = debug;
46      }
47  }
