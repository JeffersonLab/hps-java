3   import org.hps.record.LCSimEventBuilder;
4   import org.jlab.coda.jevio.EvioEvent;
5   import org.lcsim.conditions.ConditionsEvent;
6   import org.lcsim.conditions.ConditionsManager;
7   import org.lcsim.conditions.ConditionsManagerImplementation;
8   import org.lcsim.conditions.ConditionsReader;
9   import org.lcsim.event.EventHeader;
10  import org.lcsim.event.base.BaseLCSimEvent;
11  import org.lcsim.util.loop.DummyConditionsConverter;
12  import org.lcsim.util.loop.DummyDetector;
13  
14  /**
15   * This is an event builder that does nothing except create empty events.
16   * It uses a "dummy" LCSim detector. 
17   */
18  public class DummyEventBuilder implements LCSimEventBuilder {
19  
20      String dummyName = "NONE";
21      
22      public DummyEventBuilder() {
23          setDummyDetector();
24      }
25  
26      private void setDummyDetector() {
27          ConditionsManager cond = ConditionsManager.defaultInstance();
28          ConditionsReader dummyReader = ConditionsReader.createDummy();
29          ((ConditionsManagerImplementation)cond).setConditionsReader(dummyReader, dummyName);
30          DummyDetector detector = new DummyDetector(dummyName);
31          cond.registerConditionsConverter(new DummyConditionsConverter(detector));
32      }
33      
34      @Override
35      public EventHeader makeLCSimEvent(EvioEvent evioEvent) {
36          return new BaseLCSimEvent(0, evioEvent.getHeader().getNumber(), dummyName);
37      }
38      
39      @Override
40      public void readEvioEvent(EvioEvent evioEvent) {
41      }
42  
43      @Override
44      public void conditionsChanged(ConditionsEvent conditionsEvent) {
45      }   
46  }
