54      //make some dummy FpgaData to use in case there isn't any real FpgaData
55      private Map<Integer, FpgaData> makeFpgaData(Subdetector subdetector) {
56          double[] temps = new double[HPSSVTConstants.TOTAL_HYBRIDS_PER_FPGA * HPSSVTConstants.TOTAL_TEMPS_PER_HYBRID];
57          for (int i = 0; i < HPSSVTConstants.TOTAL_HYBRIDS_PER_FPGA * HPSSVTConstants.TOTAL_TEMPS_PER_HYBRID; i++) {
58              temps[i] = 23.0;
59          }
60  
61          Map<Integer, FpgaData> fpgaData = new HashMap<Integer, FpgaData>();
62          List<HpsSiSensor> sensors = subdetector.getDetectorElement().findDescendants(HpsSiSensor.class);
63  
64          for (HpsSiSensor sensor : sensors) {
65              if (sensor instanceof HpsTestRunSiSensor && !fpgaNumbers.contains(((HpsTestRunSiSensor) sensor).getFpgaID())) {
66                  fpgaNumbers.add(((HpsTestRunSiSensor) sensor).getFpgaID());
67              }
68          }
69          //===> for (Integer fpgaNumber : SvtUtils.getInstance().getFpgaNumbers()) {
70          for (Integer fpgaNumber : fpgaNumbers) {
71              fpgaData.put(fpgaNumber, new FpgaData(fpgaNumber, temps, 0));
72          }
73  
74          return fpgaData;
75      }

77      @Override
78      public void writeData(EventHeader event, EventBuilder builder) {
79  
80          List<RawTrackerHit> hits = event.get(RawTrackerHit.class, hitCollectionName);
81          Map<Integer, FpgaData> fpgaData = makeFpgaData(event.getDetector().getSubdetector(subdetectorName));
82  
83          if (verbosity >= 1) {
84              System.out.println("Writing " + hits.size() + " SVT hits");
85              System.out.println("Writing " + fpgaData.size() + " FPGA data");
86          }
87  
88          Map<Integer, List<int[]>> fpgaHits = new HashMap<Integer, List<int[]>>();
89  
90          //===> for (Integer fpgaNumber : SvtUtils.getInstance().getFpgaNumbers()) {
91          for (Integer fpgaNumber : fpgaNumbers) {
92              fpgaHits.put(fpgaNumber, new ArrayList<int[]>());
93          }
94  
95          for (RawTrackerHit hit : hits) {
96              //===> int fpgaAddress = SvtUtils.getInstance().getFPGA((SiSensor) hit.getDetectorElement());
97              int fpgaAddress = ((HpsTestRunSiSensor) hit.getDetectorElement()).getFpgaID();
98              //int hybridNumber = SvtUtils.getInstance().getHybrid((SiSensor) hit.getDetectorElement());
99              int hybridNumber = ((HpsTestRunSiSensor) hit.getDetectorElement()).getFpgaID();
100             int sensorChannel = hit.getIdentifierFieldValue("strip");
101             int apvNumber = SVTData.getAPV(sensorChannel);
102             int channelNumber = SVTData.getAPVChannel(sensorChannel);
103 
104             int[] data = new int[4];
105             SVTData.createSVTDataPacket(hybridNumber, apvNumber, channelNumber, fpgaAddress, hit.getADCValues(), data);
106             fpgaHits.get(fpgaAddress).add(data);
107         }
108 
109         // SVT container bank.
110         EvioBank svtBank = new EvioBank(SVT_BANK_TAG, DataType.BANK, SVT_BANK_NUMBER);
111 
112         // Iterate over FPGA's 0 - 6
113 //        for (int fpgaNumber = 0; fpgaNumber < SVT_TOTAL_NUMBER_FPGAS; fpgaNumber++) {
114         //===> for (Integer fpgaNumber : SvtUtils.getInstance().getFpgaNumbers()) {
115         for (Integer fpgaNumber : fpgaNumbers) {
116             FpgaData fpgaDatum = fpgaData.get(fpgaNumber);
117             int[] header = fpgaDatum.extractData();
118 
119             // Get the raw int data buffer for this FPGA.
120             int[] dataBuffer = new int[header.length + 4 * fpgaHits.get(fpgaNumber).size() + 2];
121             int ptr = 0;
122 
123             dataBuffer[ptr++] = 0;
124 
125             System.arraycopy(header, 0, dataBuffer, ptr, header.length);
126             ptr += header.length;
127 
128             for (int[] data : fpgaHits.get(fpgaNumber)) {
129                 System.arraycopy(data, 0, dataBuffer, ptr, data.length);
130                 ptr += data.length;
131             }
132 
133             dataBuffer[ptr++] = fpgaDatum.getTail();
134 
135             if (ptr != dataBuffer.length) {
136                 throw new RuntimeException("tried to fill SVT buffer of length " + dataBuffer.length + " with " + ptr + " ints");
137             }
138 
139             if (verbosity >= 2) {
140                 System.out.println(this.getClass().getSimpleName() + ": FPGA " + fpgaNumber + " : Data size: " + dataBuffer.length);
141             }
142 
143             // Bank for this FPGA's frame data.
144             EvioBank frameBank = new EvioBank(fpgaNumber, DataType.UINT32, fpgaNumber);
145             try {
146                 // Add the FPGA bank to the SVT bank
147                 builder.addChild(svtBank, frameBank);
148                 // Add the SVT data to the FPGA bank
149                 frameBank.appendIntData(dataBuffer);
150             } catch (EvioException e) {
151                 throw new RuntimeException(e);
152             }
153         }
154         // Add the SVT bank to the Main EVIO bank
155         try {
156             builder.addChild(builder.getEvent(), svtBank);
157         } catch (EvioException e) {
158             throw new RuntimeException(e);
159         }
160     }

162     @Override
163     public void writeData(EventHeader event, EventHeader toEvent) {
164         List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, hitCollectionName);
165         if (verbosity >= 1) {
166             System.out.println("Writing " + rawTrackerHits.size() + " SVT hits");
167         }
168         int flags = 1 << LCIOConstants.TRAWBIT_ID1;
169         toEvent.put(hitCollectionName, rawTrackerHits, RawTrackerHit.class, flags, readoutName);
170 
171         List<LCRelation> trueHitRelations = event.get(LCRelation.class, relationCollectionName);
172         toEvent.put(relationCollectionName, trueHitRelations, LCRelation.class, 0);
173 
174         List<FpgaData> fpgaData = new ArrayList(makeFpgaData(event.getDetector().getSubdetector(subdetectorName)).values());
175         if (verbosity >= 1) {
176             System.out.println("Writing " + fpgaData.size() + " FPGA data");
177         }
178         toEvent.put(fpgaDataCollectionName, fpgaData, FpgaData.class, 0);
179     }
