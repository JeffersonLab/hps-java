1   package org.hps.logging.config;
2   
3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.util.logging.LogManager;
6   
7   /**
8    * Read the default logging configuration and load it into the global log manager.
9    */
10  public class DefaultLoggingConfig {
11  
12      private static final String LOG_PROP = "logging.properties";
13  
14      /**
15       * Class constructor which reads in a logging properties file from a classpath resource.
16       */
17      public DefaultLoggingConfig() {
18          InputStream inputStream = DefaultLoggingConfig.class.getResourceAsStream(LOG_PROP);
19          try {
20              System.out.println("Reading default logging config from " + LOG_PROP);
21              LogManager.getLogManager().readConfiguration(inputStream);
22          } catch (SecurityException | IOException e) {
23              throw new RuntimeException("Initialization of default logging configuration failed.", e);
24          }
25      }
26      
27      /**
28       * Initialize default logging if java system properties are not set.
29       */
30      public static void initialize() {
31          if (System.getProperty("java.util.logging.config.class") == null &&
32                  System.getProperty("java.util.logging.config.file") == null) {
33              // Config is only read in if there is not an externally set class or file already.
34              new DefaultLoggingConfig();
35          }
36      }
37  }
