176     @Override
177     public void process(EventHeader event) {
178         aida.tree().cd("/");
179 
180         if (!event.hasCollection(Track.class, trackCollectionName)) {
181             nTracks.fill(0);
182             return;
183         }
184    
185         List<Track> tracks = event.get(Track.class, trackCollectionName);
186         nTracks.fill(tracks.size());
187 
188         for (Track trk : tracks) {
189             Hep3Vector momentum = new BasicHep3Vector(trk.getTrackStates().get(0).getMomentum());
190             double pmag = momentum.magnitude();
191             double pt = Math.sqrt(momentum.z() * momentum.z() + momentum.y() * momentum.y());
192             double theta = Math.asin(pt / pmag);
193 
194             trkPx.fill(momentum.y());
195             trkPy.fill(momentum.z());
196             trkPz.fill(momentum.x());
197             trkChi2.fill(trk.getChi2() / trk.getNDF());
198 
199             nhits.fill(trk.getTrackerHits().size());
200             charge.fill(-trk.getCharge());
201             trkd0.fill(trk.getTrackStates().get(0).getParameter(ParameterName.d0.ordinal()));
202             trkphi.fill(Math.sin(trk.getTrackStates().get(0).getParameter(ParameterName.phi0.ordinal())));
203             trkomega.fill(trk.getTrackStates().get(0).getParameter(ParameterName.omega.ordinal()));
204             trklam.fill(trk.getTrackStates().get(0).getParameter(ParameterName.tanLambda.ordinal()));
205             trkz0.fill(trk.getTrackStates().get(0).getParameter(ParameterName.z0.ordinal()));
206 
207             if (pmag > feeMomentumCut && trk.getCharge() > 0) { // remember, hps-java track charge is opposite the real
208                 // charge
209                 hfeeMom.fill(momentum.magnitude());
210                 hfeeTheta.fill(theta);
211             }
212 
213             TrackState stateAtEcal = TrackUtils.getTrackStateAtECal(trk);
214             if (stateAtEcal == null) {
215                 System.out.println("Couldn't get track state at ECal");
216                 continue;
217             }
218 
219             Hep3Vector posAtEcal = new BasicHep3Vector(stateAtEcal.getReferencePoint());
220             // Hep3Vector posAtEcal = hpstrk.getPositionAtZMap(svt_l12, ecal_face, 5.0,
221             // event.getDetector().getFieldMap())[0];
222             List<Cluster> clusters = event.get(Cluster.class, ecalCollectionName);
223             if (clusters != null) {
224                 if (debug)
225                     System.out.println("Found " + clusters.size() + " clusters");
226                 Cluster clust = findClosestCluster(posAtEcal, clusters);
227                 if (clust != null) {
228                     if (debug)
229                         System.out.println("\t\t\t Found the best clusters");
230                     Hep3Vector clusterPos = new BasicHep3Vector(clust.getPosition());
231                     double eOverP = clust.getEnergy() / pmag;
232                     double dx = posAtEcal.y() - clusterPos.x();
233                     double dy = posAtEcal.z() - clusterPos.y();
234                     heOverP.fill(eOverP);
235                     hdelXECal.fill(dx);
236                     hdelYECal.fill(dy);
237                     heVsP.fill(pmag, clust.getEnergy());
238                     if (pmag > feeMomentumCut && trk.getCharge() > 0) { // remember, hps-java track charge is opposite
239                         // the real charge
240                         hfeePOverE.fill(pmag / clust.getEnergy());
241                         hfeeClustPos.fill(posAtEcal.x(), posAtEcal.y());
242                     }
243                 }
244             }
245 
246             //////   Do Track-Hodoscope Matching  //// 
247             // Get RawTrackerHit collection from event.
248             List<SimpleGenericObject> reconHits = event.get(SimpleGenericObject.class, hodoClustersCollectionName);
249 
250             //System.out.println("Size of reconHitsi is " + reconHits.size());
251             int n_hits = reconHits.get(0).getNInt();
252 
253             TrackState stateAtHodo1 = TrackUtils.getTrackStateAtHodoL1(trk);
254             TrackState stateAtHodo2 = TrackUtils.getTrackStateAtHodoL2(trk);
255             if (stateAtHodo1 != null && stateAtHodo2 != null) {
256                 Hep3Vector posAtH1 = new BasicHep3Vector(stateAtHodo1.getReferencePoint());
257                 Hep3Vector posAtH2 = new BasicHep3Vector(stateAtHodo2.getReferencePoint());
258                 boolean isMatchHL1 = false;
259                 boolean isMatchHL2 = false;
260                 for (int ihit = 0; ihit < n_hits; ihit++) {
261                     int detid = reconHits.get(5).getIntVal(ihit);
262                     int layer = reconHits.get(2).getIntVal(ihit);
263                     DetectorElement thisTile = hodoMap.get(new Identifier(detid));
264                     if (thisTile == null) {
265                         System.out.println("Could not find this tile~~~~");
266                         continue;
267                     }
268                     if (layer == 0)
269                         isMatchHL1 = isMatchHL1 || TrackUtils.detectorElementContainsPoint(CoordinateTransformations.transformVectorToDetector(posAtH1),
270                                 (DetectorElement) thisTile, 20.0);
271                     if (layer == 1)
272                         isMatchHL2 = isMatchHL2 || TrackUtils.detectorElementContainsPoint(CoordinateTransformations.transformVectorToDetector(posAtH2),
273                                 (DetectorElement) thisTile, 20.0);
274                 }
275                 if (posAtH1.z() > 0)
276                     if (isMatchHL1)
277                         htrkProjH1TopMatch.fill(posAtH1.y(), posAtH1.z());
278                     else
279                         htrkProjH1TopNoMatch.fill(posAtH1.y(), posAtH1.z());
280                 else if (isMatchHL1)
281                     htrkProjH1BotMatch.fill(posAtH1.y(), -1 * posAtH1.z());
282                 else
283                     htrkProjH1BotNoMatch.fill(posAtH1.y(), -1 * posAtH1.z());
284 
285                 if (posAtH2.z() > 0)
286                     if (isMatchHL2)
287                         htrkProjH2TopMatch.fill(posAtH2.y(), posAtH2.z());
288                     else
289                         htrkProjH2TopNoMatch.fill(posAtH2.y(), posAtH2.z());
290                 else if (isMatchHL2)
291                     htrkProjH2BotMatch.fill(posAtH2.y(), -1 * posAtH2.z());
292                 else
293                     htrkProjH2BotNoMatch.fill(posAtH2.y(), -1 * posAtH2.z());
294             }
295         }
296     }

314     private Cluster findClosestCluster(Hep3Vector posonhelix, List<Cluster> clusters) {
315         Cluster closest = null;
316         double minDist = 9999;
317         for (Cluster cluster : clusters) {
318             double[] clPos = cluster.getPosition();
319             double dist = Math.sqrt(Math.pow(clPos[1] - posonhelix.z(), 2)); // coordinates!!!
320             if (dist < minDist) {
321                 closest = cluster;
322                 minDist = dist;
323             }
324         }
325         return closest;
326     }
