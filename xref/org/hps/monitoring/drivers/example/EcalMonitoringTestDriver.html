1   package org.hps.monitoring.drivers.example;

3   import hep.aida.IAnalysisFactory;
4   import hep.aida.IHistogram1D;
5   import hep.aida.IHistogramFactory;
6   import hep.aida.IPlotter;
7   
8   import java.util.List;
9   
10  import org.lcsim.event.EventHeader;
11  import org.lcsim.event.RawTrackerHit;
12  import org.lcsim.util.Driver;
13  
14  /**
15   * This Driver just plots the number of RawTrackerHit objects in the 'EcalReadoutHits'
16   * collection and the FADC values as a basic test that ECAL raw data is accessible to
17   * the monitoring app.
18   */
19  public class EcalMonitoringTestDriver extends Driver {
20  
21      static IHistogramFactory histogramFactory = IAnalysisFactory.create().createHistogramFactory(null);
22      IHistogram1D nHitsH1D = histogramFactory.createHistogram1D("EcalReadoutHits : Number of Hits", 443, 0, 443.);
23      IHistogram1D adcValuesH1D = histogramFactory.createHistogram1D("EcalReadoutHits : ADC Values", 300, 0, 300.);
24  
25      public void startOfData() {
26          IPlotter plotter = IAnalysisFactory.create().createPlotterFactory("ECAL Test").create("ECAL Test");
27          plotter.createRegions(2);
28          plotter.region(0).plot(nHitsH1D);
29          plotter.region(1).plot(adcValuesH1D);
30          plotter.show();
31      }
32  
33      public void process(EventHeader event) {
34          if (event.hasCollection(RawTrackerHit.class, "EcalReadoutHits")) {
35              List<RawTrackerHit> hits = event.get(RawTrackerHit.class, "EcalReadoutHits");
36              // System.out.println("EcalReadoutHits has " + hits.size() + " hits");
37              nHitsH1D.fill(hits.size());
38              for (RawTrackerHit hit : hits) {
39                  for (short adcValue : hit.getADCValues()) {
40                      adcValuesH1D.fill(adcValue);
41                  }
42              }
43          } else {
44              // For testing purposes, consider it a fatal error if collection is missing.
45              throw new RuntimeException("No RawTrackerHit collection was found with name 'EcalReadoutHits'.");
46          }
47      }
48  }
