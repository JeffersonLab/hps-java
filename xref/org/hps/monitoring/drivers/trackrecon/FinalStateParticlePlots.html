180     @Override
181     public void process(EventHeader event) {
182         if (removeRandomEvents && event.hasCollection(GenericObject.class, "TSBank")) {
183             List<GenericObject> triggerList = event.get(GenericObject.class, "TSBank");
184             for (GenericObject data : triggerList) {
185                 if (AbstractIntData.getTag(data) == TSData2019.BANK_TAG) {
186                     TSData2019 triggerData = new TSData2019(data);
187                     if (triggerData.isPulserTrigger() || triggerData.isFaradayCupTrigger()) {
188                         return;
189                     }
190                 }
191             }
192         }
193         /* make sure everything is there */
194         if (!event.hasCollection(ReconstructedParticle.class, finalStateParticlesColName)) {
195             return;
196         }
197         nRecoEvents++;
198 
199         List<ReconstructedParticle> fspList = event.get(ReconstructedParticle.class,
200                 finalStateParticlesColName);
201 
202         int eleCnt = 0;
203         int posCnt = 0;
204         int photCnt = 0;
205         for (ReconstructedParticle fsp : fspList) {
206             double charge = fsp.getCharge();
207             if (charge < 0) {
208                 eleCnt++;
209                 Hep3Vector mom = fsp.getMomentum();
210                 elePx.fill(mom.x());
211                 elePy.fill(mom.y());
212                 elePz.fill(mom.z());
213                 TrackState stateAtEcal = TrackUtils.getTrackStateAtECal((fsp.getTracks().get(0)));
214                 Hep3Vector tPos = new BasicHep3Vector(stateAtEcal.getReferencePoint());
215                 if (fsp.getClusters().size() != 0) {
216                     eleProjXYEcalMatch.fill(tPos.y(), tPos.z());
217                 } else {
218                     eleProjXYEcalNoMatch.fill(tPos.y(), tPos.z());
219                 }
220             } else if (charge > 0) {
221                 posCnt++;
222                 Hep3Vector mom = fsp.getMomentum();
223                 posPx.fill(mom.x());
224                 posPy.fill(mom.y());
225                 posPz.fill(mom.z());
226                 TrackState stateAtEcal = TrackUtils.getTrackStateAtECal((fsp.getTracks().get(0)));
227                 Hep3Vector tPos = new BasicHep3Vector(stateAtEcal.getReferencePoint());
228                 if (fsp.getClusters().size() != 0) {
229                     posProjXYEcalMatch.fill(tPos.y(), tPos.z());// tracking frame!
230                 } else {
231                     posProjXYEcalNoMatch.fill(tPos.y(), tPos.z());
232                 }
233             } else if (fsp.getClusters().size() != 0) {
234                 photCnt++;
235                 Cluster clu = fsp.getClusters().get(0);
236                 photEne.fill(clu.getEnergy());
237                 photXYECal.fill(clu.getPosition()[0], clu.getPosition()[1]);
238             } else {
239                 System.out.println("This FSP had no tracks or clusters???");
240             }
241         }
242 
243         for (ReconstructedParticle fsp1 : fspList) {
244             if (fsp1.getCharge() != 0) {
245                 continue;
246             }
247             for (ReconstructedParticle fsp2 : fspList) {
248                 if (fsp1 == fsp2) {
249                     continue;
250                 }
251                 if (fsp2.getCharge() != 0) {
252                     continue;
253                 }
254 //                if (fsp1.getClusters().get(0) == null || fsp2.getClusters().get(0) == null)
255 //                    continue;//this should never happen
256                 Cluster clu1 = fsp1.getClusters().get(0);
257                 Cluster clu2 = fsp2.getClusters().get(0);
258                 double pi0ene = clu1.getEnergy() + clu2.getEnergy();
259                 double pi0diff = Math.abs(clu1.getEnergy() - clu2.getEnergy());
260                 double pi0mass = getClusterPairMass(clu1, clu2);
261                 if (pi0diff > pi0EdifCut) {
262                     continue;
263                 }
264                 if (pi0ene < pi0EsumCut) {
265                     continue;
266                 }
267                 if (clu1.getPosition()[1] * clu2.getPosition()[1] < 0) {//top bottom
268                     pi0Ene.fill(pi0ene);
269                     pi0Diff.fill(pi0diff);
270                     pi0Mass.fill(pi0mass);
271                 }
272             }
273         }
274         nEle.fill(eleCnt);
275         nPos.fill(posCnt);
276         nPhot.fill(photCnt);
277     }

279     public double getClusterPairMass(Cluster clu1, Cluster clu2) {
280         double x0 = clu1.getPosition()[0];
281         double y0 = clu1.getPosition()[1];
282         double z0 = clu1.getPosition()[2];
283         double x1 = clu2.getPosition()[0];
284         double y1 = clu2.getPosition()[1];
285         double z1 = clu2.getPosition()[2];
286         double e0 = clu1.getEnergy();
287         double e1 = clu2.getEnergy();
288         double xlen0 = sqrt(x0 * x0 + y0 * y0 + z0 * z0);
289         double xlen1 = sqrt(x1 * x1 + y1 * y1 + z1 * z1);
290         Hep3Vector p0 = new BasicHep3Vector(x0 / xlen0 * e0, y0 / xlen0 * e0, z0 / xlen0 * e0);
291         Hep3Vector p1 = new BasicHep3Vector(x1 / xlen1 * e1, y1 / xlen1 * e1, z1 / xlen1 * e1);
292 
293         double esum = sqrt(p1.magnitudeSquared()) + sqrt(p0.magnitudeSquared());
294         double pxsum = p1.x() + p0.x();
295         double pysum = p1.y() + p0.y();
296         double pzsum = p1.z() + p0.z();
297 
298         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
299         double evtmass = esum * esum - psum * psum;
300 
301         if (evtmass > 0) {
302             return Math.sqrt(evtmass);
303         } else {
304             return -99;
305         }
306     }
