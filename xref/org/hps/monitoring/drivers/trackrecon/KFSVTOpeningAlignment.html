328     @Override
329     public void process(EventHeader event) {
330         if (removeRandomEvents && event.hasCollection(GenericObject.class, "TSBank")) {
331             List<GenericObject> triggerList = event.get(GenericObject.class, "TSBank");
332             for (GenericObject data : triggerList) {
333                 if (AbstractIntData.getTag(data) == TSData2019.BANK_TAG) {
334                     TSData2019 triggerData = new TSData2019(data);
335                     if (triggerData.isPulserTrigger() || triggerData.isFaradayCupTrigger()) {
336                         return;
337                     }
338                 }
339             }
340         }
341         aida.tree().cd("/");
342         int nTrkTop = 0;
343         int nTrkBot = 0;
344         String stripDataRelationsInputCollectionName = "KFGBLStripClusterDataRelations";
345         if (!event.hasCollection(LCRelation.class, stripDataRelationsInputCollectionName)) {
346             System.out.format("\nKalmanKinkFitDriver: the data collection %s is missing.\n", stripDataRelationsInputCollectionName);
347         }
348         String trackCollectionName = "KalmanFullTracks";
349         if (event.hasCollection(Track.class, trackCollectionName)) {
350             List<Track> kalmanFullTracks = event.get(Track.class, trackCollectionName);
351             for (Track trk : kalmanFullTracks) {
352                 if (trk.getNDF() >= 7) {
353                     KalmanKinkFit knkFt = new KalmanKinkFit(event, KI, trk);
354                     if (knkFt.doFits()) {
355                         if (knkFt.innerMomentum()[2] < 0.) {
356                             nTrkBot++;
357                             hProjAngBot.fill(knkFt.projectedAngle());
358                             hAngBot.fill(knkFt.scatteringAngle());
359                             hChiInBot.fill(knkFt.innerChi2());
360                             hDofInBot.fill(knkFt.innerDOF());
361                             hChiOutBot.fill(knkFt.outerChi2());
362                             hDofOutBot.fill(knkFt.outerDOF());
363                             deld0Bot.fill(knkFt.outerHelix()[ParameterName.d0.ordinal()] - knkFt.innerHelix()[ParameterName.d0.ordinal()]);
364                             delphiBot.fill(knkFt.outerHelix()[ParameterName.phi0.ordinal()] - knkFt.innerHelix()[ParameterName.phi0.ordinal()]);
365                             delz0Bot.fill(knkFt.outerHelix()[ParameterName.z0.ordinal()] - knkFt.innerHelix()[ParameterName.z0.ordinal()]);
366                             delwBot.fill(knkFt.outerHelix()[ParameterName.omega.ordinal()] - knkFt.innerHelix()[ParameterName.omega.ordinal()]);
367                         } else {
368                             nTrkTop++;
369                             hProjAngTop.fill(knkFt.projectedAngle());
370                             hAngTop.fill(knkFt.scatteringAngle());
371                             hChiInTop.fill(knkFt.innerChi2());
372                             hDofInTop.fill(knkFt.innerDOF());
373                             hChiOutTop.fill(knkFt.outerChi2());
374                             hDofOutTop.fill(knkFt.outerDOF());
375                             deld0Top.fill(knkFt.outerHelix()[ParameterName.d0.ordinal()] - knkFt.innerHelix()[ParameterName.d0.ordinal()]);
376                             delphiTop.fill(knkFt.outerHelix()[ParameterName.phi0.ordinal()] - knkFt.innerHelix()[ParameterName.phi0.ordinal()]);
377                             delz0Top.fill(knkFt.outerHelix()[ParameterName.z0.ordinal()] - knkFt.innerHelix()[ParameterName.z0.ordinal()]);
378                             delwTop.fill(knkFt.outerHelix()[ParameterName.omega.ordinal()] - knkFt.innerHelix()[ParameterName.omega.ordinal()]);
379                         }
380 
381                         if (debug) {
382                             System.out.format("KinkFit: event number %d\n", event.getEventNumber());
383                             System.out.format("  KinkFit: chi^2 of inner helix = %8.4f with %d dof\n", knkFt.innerChi2(), knkFt.innerDOF());
384                             System.out.format("  KinkFit: chi^2 of outer helix = %8.4f with %d dof\n", knkFt.outerChi2(), knkFt.outerDOF());
385                             Vec inHx = new Vec(5, knkFt.innerHelix());
386                             Vec outHx = new Vec(5, knkFt.outerHelix());
387                             System.out.format("  KinkFit: inner helix = %s\n", inHx.toString());
388                             System.out.format("  KinkFit: outer helix = %s\n", outHx.toString());
389                             Vec inP = new Vec(3, knkFt.innerMomentum());
390                             Vec outP = new Vec(3, knkFt.outerMomentum());
391                             System.out.format("  KinkFit: inner momentum = %s\n", inP.toString());
392                             System.out.format("  KinkFit: outer momentum = %s\n", outP.toString());
393                             System.out.format("  KinkFit: scattering angle = %10.6f radians\n", knkFt.scatteringAngle());
394                             System.out.format("  KinkFit: projected scattering angle = %10.6f radians\n", knkFt.projectedAngle());
395                         }
396                     }
397                 }
398             }
399         }
400         KI.clearInterface();
401         nTracksTop.fill(nTrkTop);
402         nTracksBot.fill(nTrkBot);

425     private List<Track> splitTrackList(List<Track> trks, boolean doTop) {
426         List<Track> tracksHalf = new ArrayList<Track>();
427         boolean isTop = false;
428         boolean isBot = false;
429         for (Track trk : trks) {
430             isTop = false;
431             isBot = false;
432             for (TrackerHit hit : trk.getTrackerHits()) {
433                 if (hit.getPosition()[2] > 0)// remember, non-bend in tracking frame is z-direction
434                 {
435                     isTop = true;
436                 } else {
437                     isBot = true;
438                 }
439             }
440             if (isTop == true && isBot != true && doTop == true) // if we want top tracks and all hits are in top
441             {
442                 tracksHalf.add(trk);
443             }
444             if (isBot == true && isTop != true && doTop == false) // if we want bottom tracks and all hits are in bottom
445             {
446                 tracksHalf.add(trk);
447             }
448         }
449         return tracksHalf;
450     }

452     private Track checkFullTrack(List<Track> fullTracks, Track t03, Track t46) {
453         List<TrackerHit> trkHitsL03 = t03.getTrackerHits();
454         List<TrackerHit> trkHitsL46 = t46.getTrackerHits();
455 
456         for (Track fullTr : fullTracks) {
457             List<TrackerHit> trkHitsFull = fullTr.getTrackerHits();
458             if (trkHitsFull.containsAll(trkHitsL03) && trkHitsFull.containsAll(trkHitsL46)) {
459                 return fullTr;
460             }
461         }
462 
463         return null;
464     }
