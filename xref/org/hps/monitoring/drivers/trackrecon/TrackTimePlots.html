228     @Override
229     public void process(EventHeader event) {
230         int trigTime = (int) (event.getTimeStamp() % 24);
231 
232         // ===> IIdentifierHelper helper = SvtUtils.getInstance().getHelper();
233         List<SiTrackerHitStrip1D> hits = event.get(SiTrackerHitStrip1D.class, hitCollection);
234         for (SiTrackerHitStrip1D hit : hits) {
235             // ===> IIdentifier id = hit.getSensor().getIdentifier();
236             // ===> int layer = helper.getValue(id, "layer");
237             int layer = ((HpsSiSensor) hit.getSensor()).getLayerNumber();
238             int module = ((HpsSiSensor) hit.getSensor()).getModuleNumber();
239             // ===> int module = helper.getValue(id, "module");
240             // System.out.format("%d, %d, %d\n",hit.getCellID(),layer,module);
241             SiSensor sensor = hit.getSensor();
242             t0.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(hit.getTime());
243 
244         }
245         //
246 
247         List<Track> tracks = event.get(Track.class, trackCollectionName);
248         for (Track track : tracks) {
249             int trackModule;
250             String moduleName = "Top";
251             if (track.getTrackerHits().get(0).getPosition()[2] > 0)
252                 trackModule = 0;
253             else {
254                 moduleName = "Bottom";
255                 trackModule = 1;
256             }
257             double minTime = Double.POSITIVE_INFINITY;
258             double maxTime = Double.NEGATIVE_INFINITY;
259             int hitCount = 0;
260             double trackTime = 0;
261             for (TrackerHit hitCross : track.getTrackerHits())
262                 for (HelicalTrackStrip hit : ((HelicalTrackCross) hitCross).getStrips()) {
263                     SiSensor sensor = (SiSensor) ((RawTrackerHit) hit.rawhits().get(0)).getDetectorElement();
264                     trackHitT0.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(hit.time());
265                     trackTime += hit.time();
266                     hitCount++;
267                     if (hit.time() > maxTime)
268                         maxTime = hit.time();
269                     if (hit.time() < minTime)
270                         minTime = hit.time();
271                 }
272             trackTimeMinMax.get(moduleName).fill(minTime, maxTime);
273             trackTimeRange.get(moduleName).fill(maxTime - minTime);
274             trackTime /= hitCount;
275             trackT0.get(moduleName).fill(trackTime);
276             trackTrigTime.get(moduleName).fill(trackTime, trigTime);
277 
278             for (TrackerHit hitCross : track.getTrackerHits())
279                 for (HelicalTrackStrip hit : ((HelicalTrackCross) hitCross).getStrips()) {
280                     int layer = hit.layer();
281                     int module = ((RawTrackerHit) hit.rawhits().get(0)).getIdentifierFieldValue("module");
282                     SiSensor sensor = (SiSensor) ((RawTrackerHit) hit.rawhits().get(0)).getDetectorElement();
283                     trackHitDt.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(hit.time() - trackTime);
284                     trackHit2D.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(hit.time() - trackTime, event.getTimeStamp() % 24);
285                     trackHitDtChan.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(hit.umeas(), hit.time() - trackTime);
286                 }
287         }
288     }

306     // module 0-4...0,2 = top; 1,3=bottom
307     // this computePlotterRegion puts top&bottom modules on same region
308     // and assume plotter is split in 3 columns, 4 rows...L0-5 on top 2 rows; L6-11 on bottom 2
309     private int computePlotterRegion(int layer) {
310 
311         if (layer == 0)
312             return 0;
313         if (layer == 1)
314             return 1;
315         if (layer == 2)
316             return 4;
317         if (layer == 3)
318             return 5;
319         if (layer == 4)
320             return 8;
321         if (layer == 5)
322             return 9;
323 
324         if (layer == 6)
325             return 2;
326         if (layer == 7)
327             return 3;
328         if (layer == 8)
329             return 6;
330         if (layer == 9)
331             return 7;
332         if (layer == 10)
333             return 10;
334         if (layer == 11)
335             return 11;
336         return -1;
337     }

339     private String getColor(int module) {
340         String color = "Black";
341         if (module == 1)
342             color = "Green";
343         if (module == 2)
344             color = "Blue";
345         if (module == 3)
346             color = "Red";
347         return color;
348     }

350     IPlotterStyle createStyle(HpsSiSensor sensor, String xAxisTitle, String yAxisTitle) {
351         IPlotterStyle style = SvtPlotUtils.createStyle(plotterFactory, xAxisTitle, yAxisTitle);
352 
353         if (sensor.isTopLayer()) {
354             style.dataStyle().fillStyle().setColor("31, 137, 229, 1");
355             style.dataStyle().outlineStyle().setColor("31, 137, 229, 1");
356         } else {
357             style.dataStyle().fillStyle().setColor("93, 228, 47, 1");
358             style.dataStyle().outlineStyle().setColor("93, 228, 47, 1");
359         }
360 
361         return style;
