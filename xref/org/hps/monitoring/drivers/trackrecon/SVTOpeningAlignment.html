503     private List<Track> splitTrackList(List<Track> trks, boolean doTop) {
504         List<Track> tracksHalf = new ArrayList<Track>();
505         boolean isTop = false;
506         boolean isBot = false;
507         for (Track trk : trks) {
508             isTop = false;
509             isBot = false;
510             for (TrackerHit hit : trk.getTrackerHits()) {
511                 if (hit.getPosition()[2] > 0)// remember, non-bend in tracking frame is z-direction
512                 {
513                     isTop = true;
514                 } else {
515                     isBot = true;
516                 }
517             }
518             if (isTop == true && isBot != true && doTop == true) // if we want top tracks and all hits are in top
519             {
520                 tracksHalf.add(trk);
521             }
522             if (isBot == true && isTop != true && doTop == false) // if we want bottom tracks and all hits are in bottom
523             {
524                 tracksHalf.add(trk);
525             }
526         }
527         return tracksHalf;
528     }

530     private Track checkFullTrack(List<Track> fullTracks, Track t03, Track t46) {
531         List<TrackerHit> trkHitsL03 = t03.getTrackerHits();
532         List<TrackerHit> trkHitsL46 = t46.getTrackerHits();
533 
534         for (Track fullTr : fullTracks) {
535             List<TrackerHit> trkHitsFull = fullTr.getTrackerHits();
536             if (trkHitsFull.containsAll(trkHitsL03) && trkHitsFull.containsAll(trkHitsL46)) {
537                 return fullTr;
538             }
539         }
540 
541         return null;
542     }

544     private void analyzeMatchedTrack(String torb, Track t) {
545         aida.tree().mkdirs("track quality");
546         aida.tree().cd("track quality");
547         String pdg = t.getCharge() < 0 ? " electron " : " positron ";
548         aida.histogram1D("chisq per ndf " + pdg + torb, 100, 0., 15.).fill(t.getChi2() / t.getNDF());
549         aida.histogram1D("number of hits " + pdg + torb, 20, -0.5, 19.5).fill(t.getTrackerHits().size());
550         aida.histogram1D("charge " + pdg + torb, 3, -1.5, 1.5).fill(t.getCharge());
551         TrackState ts = t.getTrackStates().get(0);
552         aida.histogram1D("tanLambda " + pdg + torb, 50, 0, 0.06).fill(abs(ts.getTanLambda()));
553         aida.histogram1D("z0 " + pdg + torb, 50, -1.0, 1.0).fill(ts.getZ0());
554         aida.histogram1D("d0 " + pdg + torb, 50, -3.0, 3.0).fill(ts.getD0());
555 
556         aida.tree().cd("..");
557     }
