142     @Override
143     public void process(EventHeader event) {
144         if (removeRandomEvents && event.hasCollection(GenericObject.class, "TSBank")) {
145             List<GenericObject> triggerList = event.get(GenericObject.class, "TSBank");
146             for (GenericObject data : triggerList) {
147                 if (AbstractIntData.getTag(data) == TSData2019.BANK_TAG) {
148                     TSData2019 triggerData = new TSData2019(data);
149                     if (triggerData.isPulserTrigger() || triggerData.isFaradayCupTrigger()) {
150                         return;
151                     }
152                 }
153             }
154         }
155         /* make sure everything is there */
156         if (!event.hasCollection(ReconstructedParticle.class, finalStateParticlesColName)) {
157             return;
158         }
159         if (!event.hasCollection(ReconstructedParticle.class, unconstrainedV0CandidatesColName)) {
160             return;
161         }
162         if (!event.hasCollection(ReconstructedParticle.class, beamConV0CandidatesColName)) {
163             return;
164         }
165         if (!event.hasCollection(ReconstructedParticle.class, targetConV0CandidatesColName)) {
166             return;
167         }
168         nRecoEvents++;
169 
170         List<ReconstructedParticle> unConstrainedV0List = event.get(ReconstructedParticle.class,
171                 unconstrainedV0CandidatesColName);
172         nV0.fill(unConstrainedV0List.size());
173         for (ReconstructedParticle uncV0 : unConstrainedV0List) {
174             Vertex uncVert = uncV0.getStartVertex();
175             unconVx.fill(uncVert.getPosition().x());
176             unconVy.fill(uncVert.getPosition().y());
177             unconVz.fill(uncVert.getPosition().z());
178             unconMass.fill(uncV0.getMass());
179             unconChi2.fill(uncVert.getChi2());
180             massVsVtxZ.fill(uncV0.getMass(), uncVert.getPosition().z());
181             // this always has 2 tracks.
182             List<ReconstructedParticle> trks = uncV0.getParticles();
183             Track ele = trks.get(0).getTracks().get(0);
184             Track pos = trks.get(1).getTracks().get(0);
185             // if track #0 has charge>0 it's the electron! This seems mixed up, but remember the track
186             // charge is assigned assuming a positive B-field, while ours is negative
187             if (trks.get(0).getCharge() > 0) {
188                 pos = trks.get(0).getTracks().get(0);
189                 ele = trks.get(1).getTracks().get(0);
190             }
191             pEleVspPos.fill(getMomentum(ele), getMomentum(pos));
192             pxEleVspxPos.fill(ele.getTrackStates().get(0).getMomentum()[1],
193                     pos.getTrackStates().get(0).getMomentum()[1]);
194             pyEleVspyPos.fill(ele.getTrackStates().get(0).getMomentum()[2],
195                     pos.getTrackStates().get(0).getMomentum()[2]);
196         }
197     }

203     private double getMomentum(Track trk) {
204 
205         double px = trk.getTrackStates().get(0).getMomentum()[0];
206         double py = trk.getTrackStates().get(0).getMomentum()[1];
207         double pz = trk.getTrackStates().get(0).getMomentum()[2];
208         return Math.sqrt(px * px + py * py + pz * pz);
209     }
