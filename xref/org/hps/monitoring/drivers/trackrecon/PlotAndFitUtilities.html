27      /*
28       * creates a new plotter with one region and puts the histogram in it
29       * copied from org.hps.monitoring.drivers.ecal.EcalMonitoringUtilities.java
30       */
31      static IPlotter plot(IPlotterFactory plotterFactory, IBaseHistogram histogram, IPlotterStyle style, boolean show) {
32          if (style == null)
33              style = getPlotterStyle(histogram);
34          IPlotter plotter = plotterFactory.create(histogram.title());
35          plotter.createRegion();
36          plotter.region(0).plot(histogram, style);
37          if (show)
38              plotter.show();
39          return plotter;
40      }

66      /*
67       * gets default plotter style based on histogram type
68       * copied from org.hps.monitoring.drivers.ecal.EcalMonitoringUtilities.java
69       */
70      static IPlotterStyle getPlotterStyle(IBaseHistogram histogram) {
71          StyleRegistry styleRegistry = StyleRegistry.getStyleRegistry();
72          IStyleStore store = styleRegistry.getStore("DefaultStyleStore");
73          IPlotterStyle style = null;
74          if ((histogram instanceof IHistogram1D) || (histogram instanceof ICloud1D) || (histogram instanceof IProfile1D))
75              style = store.getStyle("DefaultHistogram1DStyle");
76          else if ((histogram instanceof IHistogram2D) || (histogram instanceof IProfile2D)) {
77              style = store.getStyle("DefaultColorMapStyle");
78              style.statisticsBoxStyle().setVisible(false);
79              style.setParameter("hist2DStyle", "colorMap");
80              style.dataStyle().fillStyle().setParameter("colorMapScheme", "rainbow");
81          }
82          if (style == null)
83              throw new RuntimeException("A default style could not be found for " + histogram.title());
84  
85          // custom stuff...mg
86          style.dataStyle().errorBarStyle().setVisible(false);
87          style.legendBoxStyle().setVisible(false);
88          style.dataStyle().outlineStyle().setVisible(false);
89  
90          return style;
91      }

93      /*
94       * gets default plotter style for a function type
95       * copied from org.hps.monitoring.drivers.ecal.EcalMonitoringUtilities.java
96       */
97      static IPlotterStyle getPlotterStyle(IFunction func) {
98          StyleRegistry styleRegistry = StyleRegistry.getStyleRegistry();
99          IStyleStore store = styleRegistry.getStore("DefaultStyleStore");
100         IPlotterStyle style = null;
101         style = store.getStyle("DefaultFunctionStyle");
102         if (style == null)
103             throw new RuntimeException("A default style could not be found for " + func.title());
104         return style;
105     }

107     static IFitResult performGaussianFit(IHistogram1D histogram) {
108         IFunctionFactory functionFactory = aida.analysisFactory().createFunctionFactory(null);
109         IFitFactory fitFactory = aida.analysisFactory().createFitFactory();
110         IFunction function = functionFactory.createFunctionByName("Example Fit", "G");
111         IFitter fitter = fitFactory.createFitter("chi2", "jminuit");
112         double[] parameters = new double[3];
113         parameters[0] = histogram.maxBinHeight();
114         parameters[1] = histogram.mean();
115         parameters[2] = histogram.rms();
116         function.setParameters(parameters);
117         IFitResult fitResult = null;
118         Logger minuitLogger = Logger.getLogger("org.freehep.math.minuit");
119         minuitLogger.setLevel(Level.OFF);
120         minuitLogger.info("minuit logger test");
121 
122         try {
123             fitResult = fitter.fit(histogram, function);
124         } catch (RuntimeException e) {
125             System.out.println(e.getMessage());
126         }
127         return fitResult;
128     }
129 
130     static void fitAndPutParameters(IHistogram1D hist, IFunction function) {
131         IFitResult fr = performGaussianFit(hist);
132         if (fr != null) {
133             IFunction currentFitFunction = fr.fittedFunction();
134             function.setParameters(currentFitFunction.parameters());
135         }
136     }
