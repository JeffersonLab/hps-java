544     private int getModuleNumber(TrackerHit hit) {
545         int module = -666;
546         int layer = ((RawTrackerHit) hit.getRawHits().get(0)).getLayerNumber();
547         if (layer < 2) {
548             module = 1;
549         } else if (layer < 4) {
550             module = 2;
551         } else if (layer < 6) {
552             module = 3;
553         } else if (layer < 8) {
554             module = 4;
555         } else if (layer < 10) {
556             module = 5;
557         } else if (layer < 12) {
558             module = 6;
559         } else {
560             module = 7;
561         }
562         return module;
563     }
