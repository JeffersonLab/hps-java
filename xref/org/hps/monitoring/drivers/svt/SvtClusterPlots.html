105     private int computePlotterRegion(HpsSiSensor sensor) {
106 
107         if (sensor.getLayerNumber() < 7) {
108             if (sensor.isTopLayer()) {
109                 return 6 * (sensor.getLayerNumber() - 1);
110             } else {
111                 return 6 * (sensor.getLayerNumber() - 1) + 1;
112             }
113         } else if (sensor.isTopLayer()) {
114             if (sensor.getSide() == HpsSiSensor.POSITRON_SIDE) {
115                 return 6 * (sensor.getLayerNumber() - 7) + 2;
116             } else {
117                 return 6 * (sensor.getLayerNumber() - 7) + 3;
118             }
119         } else if (sensor.isBottomLayer()) {
120             if (sensor.getSide() == HpsSiSensor.POSITRON_SIDE) {
121                 return 6 * (sensor.getLayerNumber() - 7) + 4;
122             } else {
123                 return 6 * (sensor.getLayerNumber() - 7) + 5;
124             }
125         }
126         return -1;
127     }

129     IPlotterStyle createStyle(HpsSiSensor sensor, String xAxisTitle, String yAxisTitle) {
130 
131         // Create a default style
132         IPlotterStyle style = this.plotterFactory.createPlotterStyle();
133 
134         // Set the style of the X axis
135         style.xAxisStyle().setLabel(xAxisTitle);
136         style.xAxisStyle().labelStyle().setFontSize(14);
137         style.xAxisStyle().setVisible(true);
138 
139         // Set the style of the Y axis
140         style.yAxisStyle().setLabel(yAxisTitle);
141         style.yAxisStyle().labelStyle().setFontSize(14);
142         style.yAxisStyle().setVisible(true);
143 
144         // Turn off the histogram grid
145         style.gridStyle().setVisible(false);
146 
147         // Set the style of the data
148         style.dataStyle().lineStyle().setVisible(false);
149         style.dataStyle().outlineStyle().setVisible(false);
150         style.dataStyle().outlineStyle().setThickness(4);
151         style.dataStyle().fillStyle().setVisible(true);
152         style.dataStyle().fillStyle().setOpacity(.30);
153 
154         if (sensor == null) {
155             style.dataStyle().fillStyle().setColor("255, 38, 38, 1");
156             style.dataStyle().outlineStyle().setColor("255, 38, 38, 1");
157             style.dataStyle().fillStyle().setOpacity(.70);
158         } else if (sensor.isTopLayer()) {
159             style.dataStyle().fillStyle().setColor("31, 137, 229, 1");
160             style.dataStyle().outlineStyle().setColor("31, 137, 229, 1");
161         } else if (sensor.isBottomLayer()) {
162             style.dataStyle().fillStyle().setColor("93, 228, 47, 1");
163             style.dataStyle().outlineStyle().setColor("93, 228, 47, 1");
164         }
165         style.dataStyle().errorBarStyle().setVisible(false);
166 
167         style.regionBoxStyle().backgroundStyle().setOpacity(.20);
168         if (sensor != null && sensor.isAxial()) {
169             style.regionBoxStyle().backgroundStyle().setColor("246, 246, 34, 1");
170         }
171 
172         // Turn off the legend
173         style.legendBoxStyle().setVisible(false);
174 
175         return style;
176     }

181     private void resetPlots() {
182 
183         // Clear the fitted raw hit map of old values
184         // fittedRawTrackerHitMap.clear();
185         // Since all plots are mapped to the name of a sensor, loop
186         // through the sensors, get the corresponding plots and clear them.
187         for (HpsSiSensor sensor : sensors) {
188             clusterChargePlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
189             singleHitClusterChargePlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
190             clusterTimePlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
191         }
192 
193         for (IHistogram2D histogram : hitTimeTrigTimePlots.values()) {
194             histogram.reset();
195         }
196 
197         for (int i = 0; i < 6; i++) {
198             for (int j = 0; j < 2; j++) {
199                 hitTimeTrigTimePlots1D[i][j].reset();
200                 hitTimeTrigTimePlots2D[i][j].reset();
201             }
202         }
203     }

230     protected void detectorChanged(Detector detector) {
231 
232         // Get the HpsSiSensor objects from the geometry
233         sensors = detector.getSubdetector(SUBDETECTOR_NAME).getDetectorElement().findDescendants(HpsSiSensor.class);
234         timingConstants = DatabaseConditionsManager.getInstance().getCachedConditions(SvtTimingConstants.SvtTimingConstantsCollection.class, "svt_timing_constants").getCachedData().get(0);
235         if (sensors.size() == 0) {
236             throw new RuntimeException("No sensors were found in this detector.");
237         }
238 
239         // // If the tree already exist, clear all existing plots of any old data
240         // // they might contain.
241         // if (tree != null) {
242         // this.resetPlots();
243         // return;
244         // }
245         tree = AIDA.defaultInstance().tree();
246         tree.cd("/");
247         boolean dirExists = false;
248         String dirName = "/Clusters";
249         for (String st : tree.listObjectNames()) {
250             System.out.println(st);
251             if (st.contains(dirName)) {
252                 dirExists = true;
253             }
254         }
255         tree.setOverwrite(true);
256         if (!dirExists) {
257             tree.mkdir(dirName);
258         }
259         tree.cd(dirName);
260 
261         histogramFactory = analysisFactory.createHistogramFactory(tree);
262 
263         plotters.put("Cluster Amplitude: L0-L3", plotterFactory.create("4a Cluster Amplitude: L0-L3"));
264         plotters.get("Cluster Amplitude: L0-L3").createRegions(4, 4);
265         plotters.put("Cluster Amplitude: L4-L6", plotterFactory.create("4b Cluster Amplitude: L4-L6"));
266         plotters.get("Cluster Amplitude: L4-L6").createRegions(6, 4);
267         plotters.put("Cluster Time: L0-L3", plotterFactory.create("4c Cluster Time: L0-L3"));
268         plotters.get("Cluster Time: L0-L3").createRegions(4, 4);
269         plotters.put("Cluster Time: L4-L6", plotterFactory.create("4d Cluster Time: L4-L6"));
270         plotters.get("Cluster Time: L4-L6").createRegions(6, 4);
271         if (this.doClusterYPlots) {

273             plotters.get("L0-L3 Cluster Y").createRegions(4, 4);
274             plotters.put("L4-L6 Cluster Y", plotterFactory.create("4i L4-L6 Cluster Y"));
275             plotters.get("L4-L6 Cluster Y").createRegions(6, 4);
276         }
277         for (HpsSiSensor sensor : sensors) {
278 
279             clusterChargePlots.put(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()),
280                     histogramFactory.createHistogram1D(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()) + " - Cluster Charge", 100, 0, 2000));

283                             + " - Single Hit Cluster Charge", 100, 0, 2000));

285                     histogramFactory.createHistogram1D(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()) + " - Cluster Time", 100, -75, 75));

289                         .region(SvtPlotUtils.computePlotterRegionSvtUpgrade(sensor))

297                         .plot(clusterTimePlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())), this.createStyle(null, "Cluster Time [ns]", ""));

342         plotters.get("Cluster-trigger timing by phase").createRegions(2, 6);
343

345                 plotterFactory.create("4g Cluster-trigger timing and amplitude by phase"));

369     public void process(EventHeader event) {
370         if (removeRandomEvents && event.hasCollection(GenericObject.class, "TSBank")) {
371             List<GenericObject> triggerList = event.get(GenericObject.class, "TSBank");
372             for (GenericObject data : triggerList) {
373                 if (AbstractIntData.getTag(data) == TSData2019.BANK_TAG) {
374                     TSData2019 triggerData = new TSData2019(data);
375                     if (triggerData.isPulserTrigger() || triggerData.isFaradayCupTrigger()) {
376                         return;
377                     }
378                 }
379             }
380         }
381         if (runNumber == -1) {
382             runNumber = event.getRunNumber();
383         }
384 
385         // If the event doesn't contain any clusters, skip it
386         if (!event.hasCollection(SiTrackerHitStrip1D.class, clusterCollectionName)) {
387             return;
388         }
389 
390         if (event.hasCollection(RawTrackerHit.class, "SVTRawTrackerHits")) {
391             // Get RawTrackerHit collection from event.
392             List<RawTrackerHit> rawHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
393 
394             if (dropSmallHitEvents && SvtPlotUtils.countSmallHits(rawHits) > 3) {
395                 return;
396             }
397         }
398 
399         // Get the list of clusters in the event
400         List<SiTrackerHitStrip1D> clusters = event.get(SiTrackerHitStrip1D.class, clusterCollectionName);
401 
402         for (SiTrackerHitStrip1D cluster : clusters) {
403 
404             // Get the sensor associated with this cluster
405             HpsSiSensor sensor = (HpsSiSensor) cluster.getSensor();
406             double absClY = Math.abs(cluster.getPosition()[1]);
407             // Fill all plots
408             clusterChargePlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(cluster.getdEdx() / DopedSilicon.ENERGY_EHPAIR);
409             if (this.doClusterYPlots) {
410                 clusterYPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(absClY);
411             }
412             if (cluster.getRawHits().size() == 1) {
413                 singleHitClusterChargePlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(cluster.getdEdx() / DopedSilicon.ENERGY_EHPAIR);
414             }
415             double trigPhase = (((event.getTimeStamp() - 4 * timingConstants.getOffsetPhase()) % 24) - 12);
416             if (cutOutLowChargeClusters) {
417                 if (cluster.getdEdx() / DopedSilicon.ENERGY_EHPAIR < clusterChargeCut) {
418                     continue;
419                 }
420             }
421             clusterTimePlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(cluster.getTime());
422             if (sensor.isTopLayer()) {
423                 hitTimeTrigTimePlots1D[(int) ((event.getTimeStamp() / 4) % 6)][TOP].fill(cluster.getTime());
424                 hitTimeTrigTimePlots2D[(int) ((event.getTimeStamp() / 4) % 6)][TOP].fill(cluster.getTime(),
425                         cluster.getdEdx() / DopedSilicon.ENERGY_EHPAIR);
426                 hitTimeTrigTimePlots.get("Top").fill(cluster.getTime(), trigPhase);
427             } else {
428                 hitTimeTrigTimePlots1D[(int) ((event.getTimeStamp() / 4) % 6)][BOTTOM].fill(cluster.getTime());
429                 hitTimeTrigTimePlots2D[(int) ((event.getTimeStamp() / 4) % 6)][BOTTOM].fill(cluster.getTime(),
430                         cluster.getdEdx() / DopedSilicon.ENERGY_EHPAIR);
431                 hitTimeTrigTimePlots.get("Bottom").fill(cluster.getTime(), trigPhase);
432             }
433         }
434     }

436     public void endOfData() {
437         if (saveRootFile) {
438             String rootFile = "run" + runNumber + "_cluster_analysis.root";
439             RootFileStore store = new RootFileStore(rootFile);
440             try {
441                 store.open();
442                 store.add(tree);

445                 e.printStackTrace();
446             }
447         }
