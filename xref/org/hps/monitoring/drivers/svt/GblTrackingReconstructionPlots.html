153     protected void process(EventHeader event) {
154 
155         List<Track> tracks;
156         if (event.hasCollection(Track.class, trackCollectionName)) {
157             tracks = event.get(Track.class, trackCollectionName);
158         } else {
159             LOGGER.warning("no seed track collection");
160             tracks = new ArrayList<Track>();
161         }
162         List<Track> gblTracks;
163         if (event.hasCollection(Track.class, gblTrackCollectionName)) {
164             gblTracks = event.get(Track.class, gblTrackCollectionName);
165         } else {
166             LOGGER.warning("no gbl track collection");
167             gblTracks = new ArrayList<Track>();
168         }
169 
170         List<MCParticle> mcparticles;
171         List<MCParticle> fsParticles;
172         if (event.hasCollection(MCParticle.class)) {
173             mcparticles = event.get(MCParticle.class).get(0);
174             fsParticles = HPSMCParticlePlotsDriver.makeGenFSParticleList(mcparticles);
175         } else {
176             LOGGER.warning("no gbl track collection");
177             mcparticles = new ArrayList<MCParticle>();
178             fsParticles = new ArrayList<MCParticle>();
179         }
180 
181         LOGGER.info("Number of Tracks = " + tracks.size());
182         LOGGER.info("Number of GBL Tracks = " + gblTracks.size());
183         LOGGER.info("Number of MC particles = " + mcparticles.size());
184         LOGGER.info("Number of FS MC particles = " + fsParticles.size());
185 
186         Map<Track, MCParticle> trackTruthMatch = new HashMap<Track, MCParticle>();
187 
188         for (Track track : gblTracks) {
189             MCParticle part = TrackUtils.getMatchedTruthParticle(track);
190             trackTruthMatch.put(track, part);
191             if (part != null) {
192                 LOGGER.info("Match track with q " + track.getCharge() + " p " + track.getMomentum()[0] + ","
193                         + track.getMomentum()[1] + "," + track.getMomentum()[2]);
194             } else {
195                 LOGGER.info("no match for track with q " + track.getCharge() + " p " + track.getMomentum()[0] + ","
196                         + track.getMomentum()[1] + "," + track.getMomentum()[2]);
197             }
198         }
199 
200         for (Track track : gblTracks) {
201 
202             LOGGER.info("Track:");
203             SeedTrack st = (SeedTrack) track;
204             SeedCandidate seed = st.getSeedCandidate();
205             HelicalTrackFit htf = seed.getHelix();
206             LOGGER.info(htf.toString());
207             HelicalTrackFit pHTF = null;
208             double pTruth = -1.;
209             double pTrackTruth = -1.;
210             if (trackTruthMatch.get(track) == null) {
211                 LOGGER.info("no truth mc particle for this track");
212             } else {
213                 MCParticle part = trackTruthMatch.get(track);
214                 pTruth = part.getMomentum().magnitude();
215                 pHTF = TrackUtils.getHTF(part, Math.abs(_bfield));
216                 pTrackTruth = pHTF.p(Math.abs(_bfield));
217                 LOGGER.info("part: " + trackTruthMatch.get(track).getPDGID());
218                 LOGGER.info("pHTF:");
219                 LOGGER.info(pHTF.toString());
220                 LOGGER.info("pTruth=" + pTruth + " pTrackTruth=" + pTrackTruth);
221             }
222 
223             double d0 = htf.dca();
224             double z0 = htf.z0();
225             double C = htf.curvature();
226             double phi = htf.phi0();
227             double slope = htf.slope();
228             double p = htf.p(Math.abs(_bfield));
229             double d0Gbl = track.getTrackStates().get(0).getD0();
230             double z0Gbl = track.getTrackStates().get(0).getZ0();
231             double CGbl = track.getTrackStates().get(0).getOmega();
232             double phiGbl = track.getTrackStates().get(0).getPhi();
233             double slopeGbl = track.getTrackStates().get(0).getTanLambda();
234             double pGbl = getMag(track.getTrackStates().get(0).getMomentum());
235             LOGGER.info("pGbl=" + pGbl);
236 
237             if (pHTF != null) {
238                 double d0Truth = pHTF.dca();
239                 double z0Truth = pHTF.z0();
240                 double CTruth = pHTF.curvature();
241                 double phiTruth = pHTF.phi0();
242                 double slopeTruth = pHTF.slope();
243                 LOGGER.info("d0 " + d0 + " d0 trugh " + d0Truth);
244                 d0Diff.fill(d0 - d0Truth);
245                 z0Diff.fill(z0 - z0Truth);
246                 phiDiff.fill(phi - phiTruth);
247                 rDiff.fill(C - CTruth);
248                 slopeDiff.fill(slope - slopeTruth);
249                 pDiff.fill(p - pTruth);
250 
251                 d0DiffGbl.fill(d0Gbl - d0Truth);
252                 z0DiffGbl.fill(z0Gbl - z0Truth);
253                 phiDiffGbl.fill(phiGbl - phiTruth);
254                 rDiffGbl.fill(CGbl - CTruth);
255                 slopeDiffGbl.fill(slopeGbl - slopeTruth);
256                 pDiffGbl.fill(pGbl - pTruth);
257             }
258 
259             d0Diff2.fill(d0 - d0Gbl);
260             z0Diff2.fill(z0 - z0Gbl);
261             phiDiff2.fill(phi - phiGbl);
262             rDiff2.fill(C - CGbl);
263             slopeDiff2.fill(slope - slopeGbl);
264             pDiff2.fill(p - pGbl);
265 
266         }
267 
268         nTracks.fill(tracks.size());
269         nTracksGbl.fill(gblTracks.size());
270         nTracksDiff.fill(tracks.size() - gblTracks.size());
271 
272     }

274     private double getMag(double p[]) {
275         return Math.sqrt(p[0] * p[0] + p[1] * p[1] + p[2] * p[2]);
276     }
