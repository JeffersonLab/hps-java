140     private void resetPlots() {
141 
142         // Reset all hit maps
143         this.clearHitMaps();
144 
145         // Since all plots are mapped to the name of a sensor, loop
146         // through the sensors, get the corresponding plots and clear them.
147         for (HpsSiSensor sensor : sensors) {
148             hitsPerSensorPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
149             firstSamplePlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
150             // firstSamplePlotsNoise.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
151             if (doPerChannelSamplePlots) {
152                 firstSamplePlotsNoisePerChannel.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
153             }
154         }
155 
156         for (IHistogram1D histogram : layersHitPlots.values()) {
157             histogram.reset();
158         }
159 
160         for (IHistogram1D histogram : hitCountPlots.values()) {
161             histogram.reset();
162         }

358     @Override
359     public void process(EventHeader event
360     ) {
361 
362         if (!event.hasCollection(RawTrackerHit.class, rawTrackerHitCollectionName)) {
363             return;
364         }
365 
366         if (!event.hasCollection(LCRelation.class, fittedHitsCollectioName)) {
367             return;
368         }
369         if (removeRandomEvents && event.hasCollection(GenericObject.class, "TSBank")) {
370             List<GenericObject> triggerList = event.get(GenericObject.class, "TSBank");
371             for (GenericObject data : triggerList) {
372                 if (AbstractIntData.getTag(data) == TSData2019.BANK_TAG) {
373                     TSData2019 triggerData = new TSData2019(data);
374                     if (triggerData.isPulserTrigger() || triggerData.isFaradayCupTrigger()) {
375                         return;
376                     }
377                 }
378             }
379         }
380         if (debug && ((int) eventCount % 100 == 0)) {
381             System.out.println(this.getClass().getSimpleName() + ": processed " + String.valueOf(eventCount)
382                     + " events");
383         }
384 
385         eventCount++;
386 
387         if (outputRootFilename.isEmpty()) {
388             outputRootFilename = "run" + String.valueOf(event.getRunNumber());
389         }
390 
391         // Get RawTrackerHit collection from event.
392         List<RawTrackerHit> rawHits = event.get(RawTrackerHit.class, rawTrackerHitCollectionName);
393         List<LCRelation> fittedHits = event.get(LCRelation.class, fittedHitsCollectioName);
394 
395         if (dropSmallHitEvents && SvtPlotUtils.countSmallHits(rawHits) > 3) {
396             return;
397         }
398 
399         this.clearHitMaps();
400         for (RawTrackerHit rawHit : rawHits) {
401             HpsSiSensor sensor = (HpsSiSensor) rawHit.getDetectorElement();
402             int channel = (int) rawHit.getIdentifierFieldValue("strip");
403             double pedestal = sensor.getPedestal(channel, 0);
404             // Find the sample with maximum ADC count
405             int maxSample = 0;
406             double maxSampleValue = 0;
407             for (int s = 0; s < 6; ++s) {
408                 if (((double) rawHit.getADCValues()[s] - pedestal) > maxSampleValue) {
409                     maxSample = s;
410                     maxSampleValue = ((double) rawHit.getADCValues()[s]) - pedestal;
411                 }
412             }
413 
414             hitsPerSensor.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))[0]++;
415             firstSamplePlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(rawHit.getADCValues()[0] - pedestal);
416             // if (maxSampleCutForNoise >= 0 && maxSample >= maxSampleCutForNoise) {
417             // firstSamplePlotsNoise.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(rawHit.getADCValues()[0] - pedestal);
418             if (doPerChannelSamplePlots) {
419                 firstSamplePlotsNoisePerChannel.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))
420                         .fill(channel, rawHit.getADCValues()[0] - pedestal);
421             }
422             // } else {
423             // firstSamplePlotsNoise.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(rawHit.getADCValues()[0] - pedestal);
424             // if (doPerChannelSamplePlots) {
425             // firstSamplePlotsNoisePerChannel.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(channel, rawHit.getADCValues()[0] - pedestal);
426             // }
427             // }
428         }
429 
430         for (LCRelation fittedHit : fittedHits) {
431             // Obtain the SVT raw hit associated with this relation
432             RawTrackerHit rawHit = (RawTrackerHit) fittedHit.getFrom();
433             int channel = (int) rawHit.getIdentifierFieldValue("strip");
434             // Obtain the HpsSiSensor associated with the raw hit
435             HpsSiSensor sensor = (HpsSiSensor) rawHit.getDetectorElement();
436             double t0 = FittedRawTrackerHit.getT0(fittedHit);
437             double amplitude = FittedRawTrackerHit.getAmp(fittedHit);
438             double chi2Prob = ShapeFitParameters.getChiProb(FittedRawTrackerHit.getShapeFitParameters(fittedHit));
439             if (cutOutLowChargeHits) {
440                 if (amplitude / DopedSilicon.ENERGY_EHPAIR < hitChargeCut) {
441                     continue;
442                 }
443             }
444             t0Plots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(t0);
445             double trigPhase = (((event.getTimeStamp() - 4 * timingConstants.getOffsetPhase()) % 24) - 12);
446             t0VsTriggerTime.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(t0, trigPhase);
447 //            System.out.println( triggerData.getIntVal(1)*0.0000001);
448 //            t0VsTriggerBank.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(t0, triggerData.getIntVal(1)*0.0000001);
449             t0VsChannel.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(t0, channel);
450 //            amplitudePlots.get(sensor).fill(amplitude);
451             //           chi2Plots.get(sensor).fill(chi2Prob);
452         }
453 
454         int[] topLayersHit = new int[14];
455         int[] botLayersHit = new int[14];
456         int eventHitCount = 0;
457         int topEventHitCount = 0;
458         int botEventHitCount = 0;
459         for (HpsSiSensor sensor : sensors) {
460             int hitCount = hitsPerSensor.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))[0];
461             hitsPerSensorPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(hitCount);
462 
463             eventHitCount += hitCount;
464 
465             if (hitsPerSensor.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))[0] > 0) {
466                 if (sensor.isTopLayer()) {
467                     topLayersHit[sensor.getLayerNumber() - 1]++;
468                     topEventHitCount += hitCount;
469                 } else {
470                     botLayersHit[sensor.getLayerNumber() - 1]++;
471                     botEventHitCount += hitCount;
472                 }
473             }
474         }
475 
476         totalHitCount += eventHitCount;
477         totalTopHitCount += topEventHitCount;
478         totalBotHitCount += botEventHitCount;
479 
480         hitCountPlots.get("Raw hit counts/Event").fill(eventHitCount);
481         hitCountPlots.get("SVT top raw hit counts/Event").fill(topEventHitCount);
482         hitCountPlots.get("SVT bottom raw hit counts/Event").fill(botEventHitCount);
483 
484         int totalTopLayersHit = 0;
485         int totalBotLayersHit = 0;
486 
487         for (int layerN = 0; layerN < 14; layerN++) {
488             if (topLayersHit[layerN] > 0) {
489                 totalTopLayersHit++;
490             }
491             if (botLayersHit[layerN] > 0) {
492                 totalBotLayersHit++;
493             }
494         }
495 
496         layersHitPlots.get("Top").fill(totalTopLayersHit);
497         layersHitPlots.get("Bottom").fill(totalBotLayersHit);
498 
499     }
