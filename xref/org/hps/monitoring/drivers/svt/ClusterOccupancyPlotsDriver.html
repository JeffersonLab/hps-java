163     public static Map<Integer, Hep3Vector> createStripPositionMap(HpsSiSensor sensor) {
164         Map<Integer, Hep3Vector> positionMap = new HashMap<Integer, Hep3Vector>();
165         for (ChargeCarrier carrier : ChargeCarrier.values())
166             if (sensor.hasElectrodesOnSide(carrier)) {
167                 SiSensorElectrodes strips = (SiSensorElectrodes) sensor.getReadoutElectrodes(carrier);
168                 ITransform3D parentToLocal = sensor.getReadoutElectrodes(carrier).getParentToLocal();
169                 ITransform3D localToGlobal = sensor.getReadoutElectrodes(carrier).getLocalToGlobal();
170                 for (int physicalChannel = 0; physicalChannel < 640; physicalChannel++) {
171                     Hep3Vector localStripPosition = strips.getCellPosition(physicalChannel);
172                     Hep3Vector stripPosition = parentToLocal.transformed(localStripPosition);
173                     Hep3Vector globalStripPosition = localToGlobal.transformed(stripPosition);
174                     positionMap.put(physicalChannel, globalStripPosition);
175                 }
176             }
177         return positionMap;
178     }

181      * Create a plotter style.
182      *
183      * @param xAxisTitle : Title of the x axis
184      * @param sensor : HpsSiSensor associated with the plot. This is used to set
185      * certain attributes based on the position of the sensor.

188     // TODO: Move this to a utilities class
189     IPlotterStyle createOccupancyPlotStyle(String xAxisTitle, HpsSiSensor sensor, boolean isAlarming) {
190         // Create a default style
191         IPlotterStyle style = this.plotterFactory.createPlotterStyle();
192 
193         // Set the style of the X axis
194         style.xAxisStyle().setLabel(xAxisTitle);
195         style.xAxisStyle().labelStyle().setFontSize(14);
196         style.xAxisStyle().setVisible(true);
197 
198         // Set the style of the Y axis
199         style.yAxisStyle().setLabel("Occupancy");
200         style.yAxisStyle().labelStyle().setFontSize(14);
201         style.yAxisStyle().setVisible(true);
202 
203         // Turn off the histogram grid
204         style.gridStyle().setVisible(false);
205 
206         // Set the style of the data
207         style.dataStyle().lineStyle().setVisible(false);
208         style.dataStyle().outlineStyle().setVisible(false);
209         style.dataStyle().outlineStyle().setThickness(3);
210         style.dataStyle().fillStyle().setVisible(true);
211         style.dataStyle().fillStyle().setOpacity(.30);
212         if (sensor.isTopLayer()) {
213             style.dataStyle().fillStyle().setColor("31, 137, 229, 1");
214             style.dataStyle().outlineStyle().setColor("31, 137, 229, 1");
215         } else {
216             style.dataStyle().fillStyle().setColor("93, 228, 47, 1");
217             style.dataStyle().outlineStyle().setColor("93, 228, 47, 1");
218         }
219         style.dataStyle().errorBarStyle().setVisible(false);
220 
221         // Turn off the legend
222         style.legendBoxStyle().setVisible(false);
223 
224         style.regionBoxStyle().backgroundStyle().setOpacity(.20);
225         setBackgroundColor(style, sensor.isAxial(), isAlarming);
226 
227         return style;
228     }

239     /**
240      * Clear all histograms of it's current data.
241      */
242     private void resetPlots() {
243 
244         // Clear the hit counter map of all previously stored data.
245         occupancyMap.clear();
246 
247         // Since all plots are mapped to the name of a sensor, loop
248         // through the sensors, get the corresponding plots and clear them.
249         for (HpsSiSensor sensor : sensors) {
250 
251             // Clear the occupancy plots.
252             occupancyPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
253             // Reset the hit counters.
254             occupancyMap.put(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()), new int[640]);
255         }
256     }

262     @Override
263     protected void detectorChanged(Detector detector) {
264 
265         // Get the HpsSiSensor objects from the geometry
266         sensors = detector.getSubdetector(SUBDETECTOR_NAME).getDetectorElement().findDescendants(HpsSiSensor.class);
267 
268         // If there were no sensors found, throw an exception
269         if (sensors.isEmpty())
270             throw new RuntimeException("There are no sensors associated with this detector");
271 
272         // For each sensor, create a mapping between a physical channel number
273         // and the global strip position
274         this.createStripPositionMap();
275 
276         // // If the tree already exist, clear all existing plots of any old data
277         // // they might contain.
278         // if (tree != null) {
279         // this.resetPlots();
280         // return;
281         // }
282         // tree = analysisFactory.createTreeFactory().create();
283         tree = AIDA.defaultInstance().tree();
284         tree.cd("/");// aida.tree().cd("/");
285         histogramFactory = analysisFactory.createHistogramFactory(tree);
286 
287         plotters.put("Occupancy: L0-L3", plotterFactory.create("Occupancy: L0-L3"));
288         plotters.get("Occupancy: L0-L3").createRegions(4, 4);
289         plotters.put("Occupancy: L4-L6", plotterFactory.create("Occupancy: L4-L6"));
290         plotters.get("Occupancy: L4-L6").createRegions(6, 4);
291 
292         for (HpsSiSensor sensor : sensors) {
293 
294             occupancyPlots.put(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()),
295                     histogramFactory.createHistogram1D(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()) + " - Occupancy", 640, 0, 640));
296 
297             if (sensor.getLayerNumber() < 9)
298                 plotters.get("Occupancy: L0-L3")
299                         .region(SvtPlotUtils.computePlotterRegionSvtUpgrade(sensor))
300                         .plot(occupancyPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())),
301                                 this.createOccupancyPlotStyle("Physical Channel", sensor, false));
302             else
303                 plotters.get("Occupancy: L4-L6")
304                         .region(SvtPlotUtils.computePlotterRegionSvtUpgrade(sensor))
305                         .plot(occupancyPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())),
306                                 this.createOccupancyPlotStyle("Physical Channel", sensor, false));
307 
308             occupancyMap.put(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()), new int[640]);
309 
310         }
311 
312         for (IPlotter plotter : plotters.values())
313             plotter.show();
314     }

316     private boolean passTriggerFilter(List<GenericObject> triggerBanks) {
317 
318         // Loop through the collection of banks and get the TI banks.
319         for (GenericObject triggerBank : triggerBanks)
320 
321             // If the bank contains TI data, process it
322             if (AbstractIntData.getTag(triggerBank) == TIData.BANK_TAG) {
323 
324                 TIData tiData = new TIData(triggerBank);
325 
326                 if (filterPulserTriggers && tiData.isPulserTrigger())
327                     return false;
328                 else if (filterSingle0Triggers && tiData.isSingle0Trigger())
329                     return false;
330                 else if (filterSingle1Triggers && tiData.isSingle1Trigger())
331                     return false;
332                 else if (filterPair0Triggers && tiData.isPair0Trigger())
333                     return false;
334                 else if (filterPair1Triggers && tiData.isPair1Trigger())
335                     return false;
336             }
337         return true;
338     }

340     @Override
341     public void process(EventHeader event) {
342 
343         // Get the run number from the event and store it. This will be used
344         // when writing the plots out to a ROOT file
345         if (runNumber == -1)
346             runNumber = event.getRunNumber();
347         if (enableTriggerFilter && event.hasCollection(GenericObject.class, triggerBankCollectionName)) {
348             System.out.println("SensorOccupancyPlotsDriver::  Filtering Event");
349             // Get the list of trigger banks from the event
350             List<GenericObject> triggerBanks = event.get(GenericObject.class, triggerBankCollectionName);
351 
352             // Apply the trigger filter
353             if (!passTriggerFilter(triggerBanks))
354                 return;
355         }
356 
357         // If the event doesn't have a collection of RawTrackerHit's, skip it.
358         if (!event.hasCollection(RawTrackerHit.class, rawTrackerHitCollectionName)) {
359             System.out.println("No SVT RawTrackerHits in this event???");
360             return;
361         }
362         // Get RawTrackerHit collection from event.
363         List<RawTrackerHit> rawHits = event.get(RawTrackerHit.class, rawTrackerHitCollectionName);
364 
365 //        System.out.println("Number of SVT RawTrackerHts = " + rawHits.size());
366         if (dropSmallHitEvents && SvtPlotUtils.countSmallHits(rawHits) > 3)
367             return;
368 
369         if (resetPeriod > 0 && eventCount > resetPeriod) { // reset occupancy numbers after resetPeriod events
370             eventCount = 0;
371             resetPlots();
372         }
373 
374         eventCount++;
375 
376         // Fill the strip cluster counts if available
377         if (event.hasCollection(SiTrackerHitStrip1D.class, stripClusterCollectionName)) {
378             List<SiTrackerHitStrip1D> stripHits1D = event.get(SiTrackerHitStrip1D.class, stripClusterCollectionName);
379             for (SiTrackerHitStrip1D h : stripHits1D) {
380                 if (enableClusterChargeCut && h.getdEdx() / DopedSilicon.ENERGY_EHPAIR < clusterChargeCut)
381                     continue;
382                 RawTrackerHit rth = h.getRawHits().get(0);
383                 if ((h.getTime() < clusterTimeCutMax && h.getTime() > clusterTimeCutMin) || !enableClusterTimeCuts)
384                     occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(((HpsSiSensor) rth.getDetectorElement()).getName()))[rth
385                             .getIdentifierFieldValue("strip")]++; //   
386             }
387         }
388 
389         // Plot strip occupancies.
390         if (eventCount % eventRefreshRate == 0)
391             for (HpsSiSensor sensor : sensors) {
392                 int[] strips = occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()));
393                 occupancyPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
394 
395                 for (int channel = 0; channel < strips.length; channel++) {
396                     double stripOccupancy = (double) strips[channel] / (double) eventCount;
397 
398                     stripOccupancy /= this.timeWindowWeight;
399                     //                  System.out.println("channel " + channel + " occupancy = " + stripOccupancy);
400                     occupancyPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(channel, stripOccupancy);
401                 }
402             }
403 
404         if (plotters.get("Occupancy") != null)
405             plotters.get("Occupancy").refresh();
406 
407     }

409     @Override
410     public void endOfData() {
411 
412         if (saveRootFile) {
413             String rootFile = "run" + runNumber + "_occupancy.root";
414             RootFileStore store = new RootFileStore(rootFile);
415             try {
416                 store.open();
417                 store.add(tree);
418                 store.close();
419             } catch (IOException e) {
420                 e.printStackTrace();
421             }
422         }
423 
424         System.out.println("%===============================================================================%");
425         System.out.println("%======================== Active Edge Sensor Occupancies =======================%");
426         System.out.println("%===============================================================================%");
427         System.out.println("% Total Events: " + eventCount);
428         // Calculate the occupancies at the sensor edge
429 
430         int[] topActiveEdgeStripOccupancy = new int[7];
431         int[] bottomActiveEdgeStripOccupancy = new int[7];
432         for (HpsSiSensor sensor : sensors)
433             if (sensor.isTopLayer() && sensor.isAxial())
434 
435                 if (sensor.getSide().equals(HpsSiSensor.ELECTRON_SIDE)) {
436                     System.out.println("% Top Layer " + getLayerNumber(sensor) + " Hit Counts: "
437                             + occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))[1]);
438                     topActiveEdgeStripOccupancy[getLayerNumber(sensor) - 1] += occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))[1];
439                 } else {
440                     System.out.println("% Top Layer " + getLayerNumber(sensor) + " Hit Counts: "
441                             + occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))[638]);
442                     topActiveEdgeStripOccupancy[getLayerNumber(sensor) - 1] += occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))[638];
443                 }
444             else if (sensor.isBottomLayer() && sensor.isAxial())
445                 if (sensor.getSide().equals(HpsSiSensor.ELECTRON_SIDE)) {
446                     System.out.println("% Bottom Layer " + getLayerNumber(sensor) + " Hit Counts: "
447                             + occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))[1]);
448                     bottomActiveEdgeStripOccupancy[getLayerNumber(sensor) - 1] += occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))[1];
449                 } else {
450                     System.out.println("% Bottom Layer " + getLayerNumber(sensor) + " Hit Counts: "
451                             + occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))[638]);
452                     bottomActiveEdgeStripOccupancy[getLayerNumber(sensor) - 1] += occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))[638];
453                 }
454 
455         for (int layerN = 0; layerN < 6; layerN++) {
456             double topStripOccupancy = (double) topActiveEdgeStripOccupancy[layerN] / (double) eventCount;
457             topStripOccupancy /= this.timeWindowWeight;
458             System.out.println("% Top Layer " + (layerN + 1) + ": Occupancy in " + (24 / this.timeWindowWeight)
459                     + " ns window: " + topStripOccupancy);
460             double botStripOccupancy = (double) bottomActiveEdgeStripOccupancy[layerN] / (double) eventCount;
461             botStripOccupancy /= this.timeWindowWeight;
462             System.out.println("% Bottom Layer " + (layerN + 1) + ": Occupancy in " + (24 / this.timeWindowWeight)
463                     + " ns window: " + botStripOccupancy);
464         }
465         System.out.println("%===============================================================================%");
466         System.out.println("%===============================================================================%");
467     }
