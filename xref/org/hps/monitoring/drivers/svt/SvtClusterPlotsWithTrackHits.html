123     IPlotterStyle createStyle(HpsSiSensor sensor, String xAxisTitle, String yAxisTitle) {
124 
125         // Create a default style
126         IPlotterStyle style = this.plotterFactory.createPlotterStyle();
127 
128         // Set the style of the X axis
129         style.xAxisStyle().setLabel(xAxisTitle);
130         style.xAxisStyle().labelStyle().setFontSize(14);
131         style.xAxisStyle().setVisible(true);
132 
133         // Set the style of the Y axis
134         style.yAxisStyle().setLabel(yAxisTitle);
135         style.yAxisStyle().labelStyle().setFontSize(14);
136         style.yAxisStyle().setVisible(true);
137 
138         // Turn off the histogram grid
139         style.gridStyle().setVisible(false);
140 
141         // Set the style of the data
142         style.dataStyle().lineStyle().setVisible(false);
143         style.dataStyle().outlineStyle().setVisible(false);
144         style.dataStyle().outlineStyle().setThickness(4);
145         style.dataStyle().fillStyle().setVisible(true);
146         style.dataStyle().fillStyle().setOpacity(.30);
147 
148         if (sensor == null) {
149             style.dataStyle().fillStyle().setColor("255, 38, 38, 1");
150             style.dataStyle().outlineStyle().setColor("255, 38, 38, 1");
151             style.dataStyle().fillStyle().setOpacity(.70);
152         } else if (sensor.isTopLayer()) {
153             style.dataStyle().fillStyle().setColor("31, 137, 229, 1");
154             style.dataStyle().outlineStyle().setColor("31, 137, 229, 1");
155         } else if (sensor.isBottomLayer()) {
156             style.dataStyle().fillStyle().setColor("93, 228, 47, 1");
157             style.dataStyle().outlineStyle().setColor("93, 228, 47, 1");
158         }
159         style.dataStyle().errorBarStyle().setVisible(false);
160 
161         style.regionBoxStyle().backgroundStyle().setOpacity(.20);
162         if (sensor != null && sensor.isAxial())
163             style.regionBoxStyle().backgroundStyle().setColor("246, 246, 34, 1");
164 
165         // Turn off the legend
166         style.legendBoxStyle().setVisible(false);
167 
168         return style;
169     }

404     public void process(EventHeader event) {
405 
406         if (runNumber == -1)
407             runNumber = event.getRunNumber();
408 
409         // // If the event doesn't contain fitted raw hits, skip it
410         // if (!event.hasCollection(FittedRawTrackerHit.class, fittedHitsCollectionName)) {
411         // return;
412         // }
413         // Get the list of fitted hits from the event
414         // List<FittedRawTrackerHit> fittedHits = event.get(FittedRawTrackerHit.class, fittedHitsCollectionName);
415         //
416         // // Map the fitted hits to their corresponding raw hits
417         // this.mapFittedRawHits(fittedHits);
418         // If the event doesn't contain any clusters, skip it
419         if (!event.hasCollection(SiTrackerHitStrip1D.class, clusterCollectionName))
420             return;
421 
422         if (event.hasCollection(RawTrackerHit.class, "SVTRawTrackerHits")) {
423             // Get RawTrackerHit collection from event.
424             List<RawTrackerHit> rawHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
425 
426             if (dropSmallHitEvents && SvtPlotUtils.countSmallHits(rawHits) > 3)
427                 return;
428         }
429 
430         // Get the list of clusters in the event
431         List<SiTrackerHitStrip1D> clusters = event.get(SiTrackerHitStrip1D.class, clusterCollectionName);
432 
433         for (SiTrackerHitStrip1D cluster : clusters) {
434 
435             // Get the sensor associated with this cluster
436             HpsSiSensor sensor = (HpsSiSensor) cluster.getSensor();
437             double absClY = Math.abs(cluster.getPosition()[1]);
438             clusterYPlots.get(sensor.getName()).fill(absClY);
439             // Fill all plots
440             clusterChargePlots.get(sensor.getName()).fill(cluster.getdEdx() / DopedSilicon.ENERGY_EHPAIR);
441 
442             if (cluster.getRawHits().size() == 1)
443                 singleHitClusterChargePlots.get(sensor.getName()).fill(cluster.getdEdx() / DopedSilicon.ENERGY_EHPAIR);
444             double trigPhase = (((event.getTimeStamp() - 4 * timingConstants.getOffsetPhase()) % 24) - 12);
445             clusterTimePlots.get(sensor.getName()).fill(cluster.getTime());
446             if (sensor.isTopLayer()) {
447                 hitTimeTrigTimePlots1D[(int) ((event.getTimeStamp() / 4) % 6)][TOP].fill(cluster.getTime());
448                 hitTimeTrigTimePlots2D[(int) ((event.getTimeStamp() / 4) % 6)][TOP].fill(cluster.getTime(),
449                         cluster.getdEdx() / DopedSilicon.ENERGY_EHPAIR);
450                 hitTimeTrigTimePlots.get("Top").fill(cluster.getTime(), trigPhase);
451             } else {
452                 hitTimeTrigTimePlots1D[(int) ((event.getTimeStamp() / 4) % 6)][BOTTOM].fill(cluster.getTime());
453                 hitTimeTrigTimePlots2D[(int) ((event.getTimeStamp() / 4) % 6)][BOTTOM].fill(cluster.getTime(),
454                         cluster.getdEdx() / DopedSilicon.ENERGY_EHPAIR);
455                 hitTimeTrigTimePlots.get("Bottom").fill(cluster.getTime(), trigPhase);
456             }
457         }
458 
459         List<Cluster> ecalClusters;
460         if (event.hasCollection(Cluster.class, eCalClusterCollectionName))
461             ecalClusters = event.get(Cluster.class, eCalClusterCollectionName);
462         else {
463             System.out.println(this.getName() + "::  No ECal Clusters found");
464             return;
465         }
466 
467         // find the positron-side cluster above a certain energy
468         // take max energy cluster
469         Cluster positronCluster = null;
470         double maxEnergy = -999;
471         for (Cluster cluster : ecalClusters) {
472             double clE = cluster.getEnergy();
473             double clX = cluster.getPosition()[0];
474             double clY = cluster.getPosition()[1];
475 //            System.out.println("Cluster y = " + clY + "; Cluster E = " + clE);
476             if (clE > clusterEnergyCut && clX > clusterXCut && clE > maxEnergy) {
477                 maxEnergy = clE;
478                 positronCluster = cluster;
479             }
480         }
481 
482 //        if (positronCluster == null)
483 //            System.out.println(this.getName() + "::  No Positron Cluster Found"); //            return;
484         if (positronCluster != null) {
485             //get positron cluster y and make a line from target (0,0,0)...all I care about is Y
486             double cluYSlope = positronCluster.getPosition()[1] / positronCluster.getPosition()[2];
487             //loop over siClusters and pick out ones that are in delta of projection
488             for (SiTrackerHitStrip1D siCl : clusters) {
489                 double clY = siCl.getPosition()[1];
490                 double clZ = siCl.getPosition()[2];
491                 //projection of ecal cluster "track" onto sensor
492                 double ecalProjY = cluYSlope * clZ;
493                 double deltaY = clY - ecalProjY;
494                 // Get the sensor associated with this cluster
495                 if (clY * ecalProjY > 0) {
496                     HpsSiSensor sensor = (HpsSiSensor) siCl.getSensor();
497                     deltaYPlots.get(sensor.getName()).fill(deltaY);
498                     deltaYVsYPlots.get(sensor.getName()).fill(deltaY, ecalProjY);
499                 }
500             }
501             //loop over tracks and match up (roughly) ECal hit, then plot 
502             // deltaY for just  hits on this track
503             // purpose of this is to see if method makes sense
504             if (event.hasCollection(Track.class, trackCollectionName)) {
505                 List<Track> tracks = event.get(Track.class, trackCollectionName);
506                 //               System.out.println("Number of Tracks = " + tracks.size());
507                 for (Track trk : tracks) {
508                     if (trk.getTrackStates().get(0).getOmega() > 0)
509                         continue;//only pick positrons (-ive omega)
510                     double trkSlope = trk.getTrackStates().get(0).getTanLambda();
511                     //                  System.out.println("track slope  = " + trkSlope);
512                     if (trkSlope * cluYSlope < 0)
513                         continue;//make sure it's in the same half
514                     double projToEcal = trkSlope * positronCluster.getPosition()[2];
515                     //difference in track projection and ECal cluster... maybe use this as a cut at some point...
516                     double deltaTrackProjToCluster = projToEcal - positronCluster.getPosition()[1];
517                     List<TrackerHit> hitsOnTrack = trk.getTrackerHits();
518                     for (TrackerHit hth : hitsOnTrack) {
519                         HelicalTrackHit htc = (HelicalTrackHit) hth;
520                         HelicalTrackCross cross = (HelicalTrackCross) htc;
521                         List<HelicalTrackStrip> clusterlist = cross.getStrips();
522                         for (HelicalTrackStrip hts : clusterlist) {
523                             SiTrackerHitStrip1D siCl = findSiClusterFromHelicalTrackStrip(hts, clusters);
524                             if (siCl != null) {
525                                 List<RawTrackerHit> rthList = hts.rawhits();
526                                 HpsSiSensor sensor = (HpsSiSensor) rthList.get(0).getDetectorElement();
527                                 double siclY = siCl.getPosition()[1];
528                                 double siclZ = siCl.getPosition()[2];
529                                 double ecalProjY = cluYSlope * siclZ;
530                                 double deltaY = siclY - ecalProjY;
531                                 deltaYHOTPlots.get(sensor.getName()).fill(deltaY);
532                                 deltaYHOTVsYPlots.get(sensor.getName()).fill(deltaY, ecalProjY);
533                             }
534                         }
535                     }
536                 }
537             }
538         }
539 
540         return;
