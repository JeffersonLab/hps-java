148 
149         // ConditionsRecordCollection col_svt_align_constants =
150         // DatabaseConditionsManager.getInstance().findConditionsRecords("svt_alignments");
151         // if (col_svt_align_constants == null) {
152         // logger.info("svt_alignments collection wasn't found");
153         // }
154         // col_svt_align_constants.
155         svtBiasConstants = DatabaseConditionsManager.getInstance()
156                 .getCachedConditions(SvtBiasConstant.SvtBiasConstantCollection.class, "svt_bias_constants")
157                 .getCachedData();
158         System.out.println("found " + svtBiasConstants.size() + " bias ON ranges");
159 
160         // timingConstants =
161         // DatabaseConditionsManager.getInstance().getCachedConditions(SvtTimingConstants.SvtTimingConstantsCollection.class,
162         // "svt_timing_constants").getCachedData().get(0);
163         try {
164             fWriter = new FileWriter(fileName);
165         } catch (IOException e) {
166             throw new RuntimeException("Cannot open file " + fileName, e);
167         }
168         pWriter = new PrintWriter(fWriter);
169 
170         tree = IAnalysisFactory.create().createTreeFactory().create();
171         tree.cd("");
172         histogramFactory = IAnalysisFactory.create().createHistogramFactory(tree);
173 
174         hists_rawadc = new HashMap<HpsSiSensor, IHistogram1D>();
175         hists_rawadcnoise = new HashMap<HpsSiSensor, IHistogram1D>();
176         hists_rawadcnoiseON = new HashMap<HpsSiSensor, IHistogram1D>();
177         hists_rawadcnoiseOFF = new HashMap<HpsSiSensor, IHistogram1D>();
178 
179         sensors = detector.getSubdetector(subdetectorName).getDetectorElement().findDescendants(HpsSiSensor.class);
180 
181         plotter1 = plotterFactory.create("Pedestal subtracted zero Sample ADC");
182         plotter1.createRegions(6, 6);
183         plotter2 = plotterFactory.create("Pedestal subtracted zero Sample ADC MaxSample>4");
184         plotter2.createRegions(6, 6);
185         plotter3 = plotterFactory.create("Pedestal subtracted zero Sample ADC MaxSample>4 ON");
186         plotter3.createRegions(6, 6);
187         plotter4 = plotterFactory.create("Pedestal subtracted zero Sample ADC MaxSample>4 OFF");
188         plotter4.createRegions(6, 6);
189 
190         allHitCount = AIDA.defaultInstance().histogram1D("all hit count", 200, 0, 200);
191         allHitCountON = AIDA.defaultInstance().histogram1D("all hit count ON", 200, 0, 200);
192         allHitCountOFF = AIDA.defaultInstance().histogram1D("all hit count OFF", 200, 0, 200);
193         allHitCountVsTime = AIDA.defaultInstance().histogram2D("all hit count vs. elapsed time", timeRange, 0,
194                 timeRange, 200, 0, 200);
195         biasVsTime = AIDA.defaultInstance().histogram2D("bias vs. elapsed time", timeRange, 0, timeRange, 200, 0, 200);
196 
197         for (HpsSiSensor sensor : sensors) {
198             AIDA aida = AIDA.defaultInstance();
199             hists_rawadc.put(sensor, aida.histogram1D(sensor.getName() + " raw adc - ped", 100, -1000.0, 5000.0));
200             plotter1.region(SvtPlotUtils.computePlotterRegion(sensor)).plot(hists_rawadc.get(sensor));
201             hists_rawadcnoise.put(sensor,
202                     aida.histogram1D(sensor.getName() + " raw adc - ped maxSample>4", 100, -1000.0, 1000.0));
203             plotter2.region(SvtPlotUtils.computePlotterRegion(sensor)).plot(hists_rawadcnoise.get(sensor));
204             hists_rawadcnoiseON.put(sensor,
205                     aida.histogram1D(sensor.getName() + " raw adc - ped maxSample>4 ON", 100, -1000.0, 1000.0));
206             plotter3.region(SvtPlotUtils.computePlotterRegion(sensor)).plot(hists_rawadcnoiseON.get(sensor));
207             hists_rawadcnoiseOFF.put(sensor,
208                     aida.histogram1D(sensor.getName() + " raw adc - ped maxSample>4 OFF", 100, -1000.0, 1000.0));
209             plotter4.region(SvtPlotUtils.computePlotterRegion(sensor)).plot(hists_rawadcnoiseOFF.get(sensor));
210             hists_hitCounts.put(sensor, aida.histogram1D(sensor.getName() + " hit count", 100, 0, 100));
211             hists_hitCountsON.put(sensor, aida.histogram1D(sensor.getName() + " hit count ON", 100, 0, 100));
212             hists_hitCountsOFF.put(sensor, aida.histogram1D(sensor.getName() + " hit count OFF", 100, 0, 100));
213         }
214 
215         if (showPlots) {
216             plotter1.show();
217             plotter2.show();
218             plotter3.show();
219             plotter4.show();
220         }
221 
222         List<RunSpreadsheet.RunData> runmap;
223         if (useRunTableFormat) {
224             runmap = SvtBiasMyaDataReader.readRunTable(new File(runSpreadSheetPath));
225         } else {
226             runmap = SvtBiasConditionsLoader.getRunListFromSpreadSheet(runSpreadSheetPath);
227         }
228         List<SvtBiasMyaRange> ranges = SvtBiasMyaDataReader.readMyaData(new File(myaDumpPath), 178.0, 2000,
229                 discardMyaDataHeader);
230 
231         // SvtBiasConditionsLoader.setTimeOffset(Calendar.)

237             }
238             pWriter.println(r.toString());
239         }

243     private Date getEventTimeStamp(EventHeader event) {
244         List<GenericObject> intDataCollection = event.get(GenericObject.class, triggerBankCollectionName);
245         for (GenericObject data : intDataCollection) {
246             if (AbstractIntData.getTag(data) == HeadBankData.BANK_TAG) {
247                 Date date = HeadBankData.getDate(data);
248                 if (date != null) {
249                     return date;
250                 }
251             }
252         }
253         return null;
254     }

256     @Override
257     public void process(EventHeader event) {
258 
259         Map<String, int[]> params = event.getIntegerParameters();
260 
261         int[] biasGood = params.get("svt_bias_good");
262         // int[] positionGood = params.get("svt_position_good");
263         // int[] burstmodeNoiseGood = params.get("svt_burstmode_noise_good");
264 
265         // System.out.format("%d %d %d\n", biasGood[0], positionGood[0], burstmodeNoiseGood[0]);
266         if (biasGood == null) {
267             hvOnEventFlag = false;
268         } else {
269             hvOnEventFlag = biasGood[0] == 1;
270         }
271 
272         if (allHitCountVsNum == null) {
273             allHitCountVsNum = AIDA.defaultInstance().histogram2D("all hit count vs. event num", 1000,
274                     event.getEventNumber(), event.getEventNumber() + eventNumRange, 50, 0, 200);
275         }
276         if (numVsTime == null) {
277             numVsTime = AIDA.defaultInstance().histogram2D("event num vs. elapsed time", timeRange, 0, timeRange, 1000,
278                     event.getEventNumber(), event.getEventNumber() + eventNumRange);
279         }
280 
281         // Read EPICS data if available
282         epicsData = EpicsData.read(event);
283 
284         if (epicsData != null) {
285             LOGGER.info(epicsData.toString());
286             if (epicsData.getKeys().contains("SVT:bias:top:0:v_sens")) {
287 
288                 epicsBiasValue = epicsData.getValue("SVT:bias:top:0:v_sens");
289                 LOGGER.info("epicsBiasValue = " + Double.toString(epicsBiasValue));
290 
291                 if (epicsBiasValue > 178.0) {
292                     hvOnEpics = true;
293                 } else {
294                     hvOnEpics = false;
295                 }
296             }
297         } else {
298             LOGGER.fine("no epics information in this event");
299         }
300 
301         // Read the timestamp for the event
302         // It comes in on block level so not every event has it, use the latest one throughout a block
303         Date newEventDate = TriggerDataUtils.getEventTimeStamp(event, triggerBankCollectionName);
304         if (newEventDate != null) {
305             if (firstDate == null) {
306                 firstDate = newEventDate;
307             }
308             hvOnConditions = svtBiasConstants.find(newEventDate) != null;
309             if (eventDate == null || !eventDate.equals(newEventDate)) {
310                 System.out.format("event %d with new timestamp %s\n", event.getEventNumber(), newEventDate.toString());
311                 System.out.println("hvOnMya is " + (hvOnMya ? "ON" : "OFF") + " hvOnEpics "
312                         + (hvOnEpics ? "ON" : "OFF") + " hvOnConditions " + (hvOnConditions ? "ON" : "OFF")
313                         + " hvOnEventFlag " + (hvOnEventFlag ? "ON" : "OFF") + " for Run " + event.getRunNumber()
314                         + " Event " + event.getEventNumber() + " date " + newEventDate.toString() + " epoch "
315                         + newEventDate.getTime());
316                 // check what the DB has
317                 if (svtBiasConstants != null) {
318                     LOGGER.info("there are " + svtBiasConstants.size() + " constants to search");
319                     for (SvtBiasConstant constant : svtBiasConstants) {
320                         LOGGER.info("start " + constant.getStart() + " end " + constant.getEnd() + " value "
321                                 + constant.getValue());
322                     }
323 
324                     SvtBiasConstant constant = svtBiasConstants.find(newEventDate);
325 
326                     LOGGER.info(constant == null ? "No constant found!" : ("Found constant " + "start "
327                             + constant.getStart() + " end " + constant.getEnd() + " value " + constant.getValue()));
328 
329                 }
330             }
331             eventDate = newEventDate;
332         }
333         if (eventDate != null) {
334             numVsTime.fill((eventDate.getTime() - firstDate.getTime()) / 1000, event.getEventNumber());
335             biasVsTime.fill((eventDate.getTime() - firstDate.getTime()) / 1000, epicsBiasValue);
336         }
337 
338         // only do this analysis where there is a date availabe.
339         if (eventDate != null) {
340             if (debug) {
341                 LOGGER.info("eventDate " + eventDate.toString());
342             }
343 
344             eventCount++;
345 
346             if (runRange == null) {
347                 for (SvtBiasRunRange r : runRanges) {
348                     if (r.getRun().getRun() == event.getRunNumber()) {
349                         runRange = r;
350                     }
351                 }
352             }
353 
354             hvOnMya = runRange.includes(eventDate);
355 
356             // print the cases where epics and run range do not agree
357             if (hvOnMya != hvOnEpics && epicsBiasValue > 0.) {
358                 if (debug) {
359                     LOGGER.warning("hvOnMya is " + (hvOnMya ? "ON" : "OFF") + " hvOnEpics "
360                             + (hvOnEpics ? "ON" : "OFF") + " for Run " + event.getRunNumber() + " Event "
361                             + event.getEventNumber() + " date " + eventDate.toString() + " epoch "
362                             + eventDate.getTime() + " hvOn " + (hvOnMya ? "YES" : "NO") + " hvOnEpics "
363                             + (hvOnEpics ? "YES" : "NO"));
364                 }
365                 pWriter.println("Run " + event.getRunNumber() + " Event " + event.getEventNumber() + " date "
366                         + eventDate.toString() + " epoch " + eventDate.getTime() + " hvOn " + (hvOnMya ? "YES" : "NO"));
367                 eventCountEpicsDisagree++;
368             }
369 
370             // print the cases where the HV is OFF
371             if (!hvOnMya) {
372                 if (debug) {
373                     LOGGER.info("Run " + event.getRunNumber() + " Event " + event.getEventNumber() + " date "
374                             + eventDate.toString() + " epoch " + eventDate.getTime() + " hvOnMya "
375                             + (hvOnMya ? "YES" : "NO") + " hvOnEpics " + (hvOnEpics ? "YES" : "NO"));
376                 }
377                 pWriter.println("Run " + event.getRunNumber() + " Event " + event.getEventNumber() + " date "
378                         + eventDate.toString() + " epoch " + eventDate.getTime() + " hvOnMya "
379                         + (hvOnMya ? "YES" : "NO") + " hvOnEpics " + (hvOnEpics ? "YES" : "NO"));
380                 eventCountHvOff++;
381             }
382             if (event.hasCollection(RawTrackerHit.class, rawTrackerHitCollectionName)) {
383                 Map<HpsSiSensor, Integer> hitCountMap = new HashMap<HpsSiSensor, Integer>();
384 
385                 // Get RawTrackerHit collection from event.
386                 List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, rawTrackerHitCollectionName);
387 
388                 for (RawTrackerHit hit : rawTrackerHits) {
389                     HpsSiSensor sensor = (HpsSiSensor) hit.getDetectorElement();
390                     Integer count = hitCountMap.get(sensor);
391                     if (count == null) {
392                         count = 0;
393                     }
394                     hitCountMap.put(sensor, count + 1);
395 
396                     int strip = hit.getIdentifierFieldValue("strip");
397                     double pedestal = sensor.getPedestal(strip, 0);
398                     hists_rawadc.get(sensor).fill(hit.getADCValues()[0] - pedestal);
399 
400                     int maxSample = 0;
401                     double maxSampleValue = 0;
402                     for (int s = 0; s < 6; ++s) {
403                         if (((double) hit.getADCValues()[s] - pedestal) > maxSampleValue) {
404                             maxSample = s;
405                             maxSampleValue = ((double) hit.getADCValues()[s]) - pedestal;
406                         }
407                     }
408                     if (maxSample >= 4) {
409                         hists_rawadcnoise.get(sensor).fill(hit.getADCValues()[0] - pedestal);
410                         if (hvOnMya) {
411                             hists_rawadcnoiseON.get(sensor).fill(hit.getADCValues()[0] - pedestal);
412                         } else {
413                             hists_rawadcnoiseOFF.get(sensor).fill(hit.getADCValues()[0] - pedestal);
414                         }
415                     }
416                 }
417 
418                 allHitCount.fill(rawTrackerHits.size());
419 
420                 // if (dropSmallHitEvents && SvtPlotUtils.countSmallHits(rawTrackerHits) > 3) {
421                 // return;
422                 // }
423                 if (hvOnMya) {
424                     allHitCountON.fill(rawTrackerHits.size());
425                 } else {
426                     allHitCountOFF.fill(rawTrackerHits.size());
427                 }
428                 allHitCountVsNum.fill(event.getEventNumber(), rawTrackerHits.size());
429                 if (eventDate != null) {
430                     allHitCountVsTime.fill((eventDate.getTime() - firstDate.getTime()) / 1000, rawTrackerHits.size());
431                 }
432                 for (HpsSiSensor sensor : sensors) {
433                     Integer count = hitCountMap.get(sensor);
434                     if (count == null) {
435                         count = 0;
436                     }
437                     hists_hitCounts.get(sensor).fill(count);
438                     if (hvOnMya) {
439                         hists_hitCountsON.get(sensor).fill(count);
440                     } else {
441                         hists_hitCountsOFF.get(sensor).fill(count);
442                     }
443                 }
444             }
445         }
446     }

452                 + Integer.toString(eventCountHvOff) + " eventCountEpicsDisagree "

455                 + Integer.toString(eventCountHvOff) + " eventCountEpicsDisagree "
456                 + Integer.toString(eventCountEpicsDisagree));

460             fWriter.close();
461         } catch (IOException ex) {
462             LOGGER.log(Level.SEVERE, null, ex);
463         }
464 
465     }
