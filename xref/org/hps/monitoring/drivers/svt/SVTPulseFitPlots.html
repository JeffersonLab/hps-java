67          timingConstants = DatabaseConditionsManager.getInstance()
68                  .getCachedConditions(SvtTimingConstants.SvtTimingConstantsCollection.class, "svt_timing_constants")
69                  .getCachedData().get(0);
70  
71          List<HpsSiSensor> sensors = detector.getSubdetector(subdetectorName).getDetectorElement()
72                  .findDescendants(HpsSiSensor.class);
73          tree = AIDA.defaultInstance().tree();
74          tree.cd("/");
75          boolean dirExists = false;
76          String dirName = "/SVTPulseFits";
77          for (String st : tree.listObjectNames()) {
78              System.out.println(st);
79              if (st.contains(dirName)) {
80                  dirExists = true;
81              }
82          }
83          tree.setOverwrite(true);
84          if (!dirExists) {
85              tree.mkdir(dirName);
86          }
87          tree.cd(dirName);
88  
89          plotters.put("Timing:  L0-L3", plotterFactory.create("3a Timing"));
90          plotters.get("Timing:  L0-L3").createRegions(4, 4);
91          plotters.put("Timing:  L4-L6", plotterFactory.create("3b Timing"));
92          plotters.get("Timing:  L4-L6").createRegions(6, 4);

94          plotters.put("Amplitude:  L0-L3", plotterFactory.create("3c Amplitude"));

182     IPlotterStyle createStyle(HpsSiSensor sensor, String xAxisTitle, String yAxisTitle) {
183         IPlotterStyle style = SvtPlotUtils.createStyle(plotterFactory, xAxisTitle, yAxisTitle);
184 
185         if (sensor.isTopLayer()) {
186             style.dataStyle().fillStyle().setColor("31, 137, 229, 1");
187             style.dataStyle().outlineStyle().setColor("31, 137, 229, 1");
188         } else {
189             style.dataStyle().fillStyle().setColor("93, 228, 47, 1");
190             style.dataStyle().outlineStyle().setColor("93, 228, 47, 1");
191         }
192 
193         return style;

203     @Override
204     public void process(EventHeader event) {
205         if (removeRandomEvents && event.hasCollection(GenericObject.class, "TSBank")) {
206             List<GenericObject> triggerList = event.get(GenericObject.class, "TSBank");
207             for (GenericObject data : triggerList) {
208                 if (AbstractIntData.getTag(data) == TSData2019.BANK_TAG) {
209                     TSData2019 triggerData = new TSData2019(data);
210                     if (triggerData.isPulserTrigger() || triggerData.isFaradayCupTrigger()) {
211                         return;
212                     }
213                 }
214             }
215         }
216 
217         List<FittedRawTrackerHit> fittedrawHits = event.get(FittedRawTrackerHit.class, fittedTrackerHitCollectionName);
218         for (FittedRawTrackerHit fit : fittedrawHits) {
219             HpsSiSensor sensor = (HpsSiSensor) fit.getRawTrackerHit().getDetectorElement();
220             int strip = fit.getRawTrackerHit().getIdentifierFieldValue("strip");
221             double fittedAmp = fit.getAmp();
222             double fittedT0 = fit.getT0();
223 
224             double tof = fit.getRawTrackerHit().getDetectorElement().getGeometry().getPosition().magnitude()
225                     / (Const.SPEED_OF_LIGHT * Const.nanosecond);
226             double offset = timingConstants.getOffsetTime()
227                     + (((event.getTimeStamp() - 4 * timingConstants.getOffsetPhase()) % 24) - 12)
228                     + sensor.getShapeFitParameters(strip)[HpsSiSensor.T0_INDEX] + sensor.getT0Shift() + tof;
229             if (!correctForT0Offset) {
230                 offset = 0.0;
231             }
232 
233             t0Plots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(fittedT0);
234             ampPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(fittedAmp);
235             chiprobPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()))
236                     .fill(fit.getShapeFitParameters().getChiProb());
237 
238             t0aPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(fit.getT0(), fit.getAmp());
239             if (this.doShapePlots) {
240                 if (fit.getAmp() > 4 * sensor.getNoise(strip, 0)) {
241                     for (int i = 0; i < fit.getRawTrackerHit().getADCValues().length; i++) {
242                         shapePlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(
243                                 (i * HPSSVTConstants.SAMPLING_INTERVAL - fit.getT0() - offset),
244                                 (fit.getRawTrackerHit().getADCValues()[i] - sensor.getPedestal(strip, i))
245                                 / fit.getAmp());
246                     }
247                 }
248             }
249         }
250     }
