226     public static Map<Integer, Hep3Vector> createStripPositionMap(HpsSiSensor sensor) {
227         Map<Integer, Hep3Vector> positionMap = new HashMap<Integer, Hep3Vector>();
228         for (ChargeCarrier carrier : ChargeCarrier.values())
229             if (sensor.hasElectrodesOnSide(carrier)) {
230                 //                SiSensorElectrodes electrodes = sensor.getReadoutElectrodes();                 
231                 SiSensorElectrodes strips = (SiSensorElectrodes) sensor.getReadoutElectrodes(carrier);
232                 ITransform3D parentToLocal = sensor.getReadoutElectrodes(carrier).getParentToLocal();
233                 ITransform3D localToGlobal = sensor.getReadoutElectrodes(carrier).getLocalToGlobal();
234                 for (int physicalChannel = 0; physicalChannel < 640; physicalChannel++) {
235                     Hep3Vector localStripPosition = strips.getCellPosition(physicalChannel);
236                     Hep3Vector stripPosition = parentToLocal.transformed(localStripPosition);
237                     Hep3Vector globalStripPosition = localToGlobal.transformed(stripPosition);
238                     positionMap.put(physicalChannel, globalStripPosition);
239                 }
240             }
241         return positionMap;
242     }

307

309         occupancyMap.clear();
310 
311         // Since all plots are mapped to the name of a sensor, loop
312         // through the sensors, get the corresponding plots and clear them.
313         for (HpsSiSensor sensor : sensors) {
314 
315             // Clear the occupancy plots.
316             occupancyPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
317 
318             if (enableMaxSamplePlots)
319                 maxSamplePositionPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
320 
321             // Reset the hit counters.
322             occupancyMap.put(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()), new int[640]);
323         }
324     }

369             maxSampleStatus.setStatus(StatusCode.UNKNOWN, "Monitor disabled in steering file.");

371         for (HpsSiSensor sensor : sensors) {
372

374                     histogramFactory.createHistogram1D(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()) + " - Occupancy", 640, 0, 640));

379                         .plot(occupancyPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())),
380                                 this.createOccupancyPlotStyle("Physical Channel", sensor, false));
381             else
382                 plotters.get("Occupancy: L4-L6")
383                         .region(SvtPlotUtils.computePlotterRegionSvtUpgrade(sensor))
384                         .plot(occupancyPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())),
385                                 this.createOccupancyPlotStyle("Physical Channel", sensor, false));
386 
387             occupancyMap.put(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()), new int[640]);
388 
389             if (enableMaxSamplePlots) {
390                 maxSamplePositionPlots.put(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()),
391                         histogramFactory.createHistogram1D(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()) + " - Max Sample Number", 6, -0.5, 5.5));
392                 if (sensor.getLayerNumber() < 9)
393                     plotters.get("Max Sample Number: L0-L3")
394                             .region(SvtPlotUtils.computePlotterRegionSvtUpgrade(sensor))
395                             .plot(maxSamplePositionPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())),
396                                     this.createOccupancyPlotStyle("Max Sample Number", sensor, false));

400                             .plot(maxSamplePositionPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())),
401                                     this.createOccupancyPlotStyle("Max Sample Number", sensor, false));

403         }

406         for (int i = 0; i < plotter.numberOfRegions(); i++)
407             plotter.region(i).setYLimits(occupancyYRange1);

411             plotter.region(i).setYLimits(occupancyYRange2);

413         for (IPlotter plotterShow : plotters.values())
414             plotterShow.show();
415     }

417     private boolean passTriggerFilter(List<GenericObject> triggerBanks) {
418 
419         // Loop through the collection of banks and get the TI banks.
420         for (GenericObject triggerBank : triggerBanks)
421 
422             // If the bank contains TI data, process it
423             if (AbstractIntData.getTag(triggerBank) == TIData.BANK_TAG) {
424 
425                 TIData tiData = new TIData(triggerBank);
426 
427                 if (filterPulserTriggers && tiData.isPulserTrigger())
428                     return false;
429                 else if (filterSingle0Triggers && tiData.isSingle0Trigger())
430                     return false;
431                 else if (filterSingle1Triggers && tiData.isSingle1Trigger())
432                     return false;
433                 else if (filterPair0Triggers && tiData.isPair0Trigger())
434                     return false;
435                 else if (filterPair1Triggers && tiData.isPair1Trigger())
436                     return false;
437             }
438         return true;
439     }

441     @Override
442     public void process(EventHeader event) {
443 
444         // Get the run number from the event and store it. This will be used
445         // when writing the plots out to a ROOT file
446         if (runNumber == -1)
447             runNumber = event.getRunNumber();
448         if (enableTriggerFilter && event.hasCollection(GenericObject.class, triggerBankCollectionName)) {
449             LOGGER.info("Filtering Event");
450             // Get the list of trigger banks from the event
451             List<GenericObject> triggerBanks = event.get(GenericObject.class, triggerBankCollectionName);
452 
453             // Apply the trigger filter
454             if (!passTriggerFilter(triggerBanks))
455                 return;
456         }
457 
458         // If the event doesn't have a collection of RawTrackerHit's, skip it.
459         if (!event.hasCollection(RawTrackerHit.class, rawTrackerHitCollectionName)) {
460             LOGGER.warning("No SVT RawTrackerHits in this event.");
461             return;
462         }
463         // Get RawTrackerHit collection from event.
464         List<RawTrackerHit> rawHits = event.get(RawTrackerHit.class, rawTrackerHitCollectionName);
465 
466 //        System.out.println("Number of SVT RawTrackerHts = " + rawHits.size());
467         if (dropSmallHitEvents && SvtPlotUtils.countSmallHits(rawHits) > 3)
468             return;
469 
470         if (resetPeriod > 0 && eventCount > resetPeriod) { // reset occupancy numbers after resetPeriod events
471             eventCount = 0;
472             resetPlots();
473         }
474 
475         eventCount++;
476 
477         // Increment strip hit count.
478         for (RawTrackerHit rawHit : rawHits) {
479             // Obtain the raw ADC samples for each of the six samples readout
480             short[] adcValues = rawHit.getADCValues();
481 
482             // Find the sample that has the largest amplitude. This should
483             // correspond to the peak of the shaper signal if the SVT is timed
484             // in correctly. Otherwise, the maximum sample value will default
485             // to 0.
486             int maxAmplitude = 0;
487             int maxSamplePositionFound = -1;
488             for (int sampleN = 0; sampleN < 6; sampleN++)
489                 if (adcValues[sampleN] > maxAmplitude) {
490                     maxAmplitude = adcValues[sampleN];
491                     maxSamplePositionFound = sampleN;
492                 }
493             if (maxSamplePosition == -1 || maxSamplePosition == maxSamplePositionFound)
494                 occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(((HpsSiSensor) rawHit.getDetectorElement()).getName()))[rawHit
495                         .getIdentifierFieldValue("strip")]++; //                System.out.println("Filling occupancy");
496 
497             if (enableMaxSamplePlots)
498                 maxSamplePositionPlots.get(SvtPlotUtils.fixSensorNumberLabel(((HpsSiSensor) rawHit.getDetectorElement()).getName())).fill(
499                         maxSamplePositionFound);
500         }
501 
502         if (enableMaxSamplePlots && eventCount > maxSampleMonitorStart && eventCount % maxSampleMonitorPeriod == 0)
503             checkMaxSample();
504 
505         if (eventCount > occupancyMonitorStart && eventCount % occupancyMonitorPeriod == 0)
506             checkOccupancy();
507 
508         // Plot strip occupancies.
509         if (eventCount % eventRefreshRate == 0)
510             for (HpsSiSensor sensor : sensors) {
511                 int[] strips = occupancyMap.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()));
512                 occupancyPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).reset();
513 
514                 for (int channel = 0; channel < strips.length; channel++) {
515                     double stripOccupancy = (double) strips[channel] / (double) eventCount;
516 
517                     stripOccupancy /= this.timeWindowWeight;
518                     //                  System.out.println("channel " + channel + " occupancy = " + stripOccupancy);
519                     occupancyPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName())).fill(channel, stripOccupancy);
520 
521                 }
522 
523             }
524 
525         if (plotters.get("Occupancy") != null)
526             plotters.get("Occupancy").refresh();
527 
528     }

530     private void checkMaxSample() {
531         StatusCode oldStatus = maxSampleStatus.getStatusCode();
532         boolean isSystemOK = true;
533 
534         String label = "foo";
535         for (HpsSiSensor sensor : sensors) {
536

541             IHistogram1D maxSamplePlot = maxSamplePositionPlots.get(SvtPlotUtils.fixSensorNumberLabel(sensor.getName()));
542 //            IPlotterRegion region = plotters.get("Max Sample Number").region(SvtPlotUtils.computePlotterRegion(sensor));
543             IPlotterRegion region = plotters.get(label).region(SvtPlotUtils.computePlotterRegionSvtUpgrade(sensor));
544

546                     .binEntries(maxSamplePosition - 1)
547                     && maxSamplePlot.binEntries(maxSamplePosition) > maxSamplePlot.binEntries(maxSamplePosition + 1);
548             if (!isSensorOK) {
549                 isSystemOK = false;
550                 if (oldStatus != StatusCode.ALARM)
551                     maxSampleStatus.setStatus(StatusCode.ALARM, "Sensor " + SvtPlotUtils.fixSensorNumberLabel(sensor.getName()) + " looks out of time.");
552                 IPlotterStyle plotterStyle = createOccupancyPlotStyle("Max Sample Number", sensor, true);
553                 // region.clear();
554                 // region.plot(maxSamplePlot, plotterStyle);
555                 region.applyStyle(plotterStyle);
556                 // region.style().regionBoxStyle().backgroundStyle().setColor("246, 34, 34, 1");
557                 // setBackgroundColor(region.style(),sensor.isAxial(),true);
558 
559             } else {
560                 IPlotterStyle plotterStyle = createOccupancyPlotStyle("Max Sample Number", sensor, false);
561                 // region.clear();
562                 // region.plot(maxSamplePlot, plotterStyle);
563                 region.applyStyle(plotterStyle);
564                 // setBackgroundColor(region.style(),sensor.isAxial(),false);
565             }
566         }
567         if (isSystemOK)
568             if (oldStatus != StatusCode.OKAY)
569                 maxSampleStatus.setStatus(StatusCode.OKAY, "All sensors are timed in.");
570     }

572     private void checkOccupancy() {
573         StatusCode oldStatus = occupancyStatus.getStatusCode();
574         boolean isSystemOK = true;
575         String label = "foo";
576         for (HpsSiSensor sensor : sensors) {
577             if (sensor.getLayerNumber() < 9)

583             IPlotterRegion region = plotters.get(label).region(SvtPlotUtils.computePlotterRegionSvtUpgrade(sensor));

585             double apvOccupancy[] = new double[5];
586             for (int i = 0; i < occupancyPlot.axis().bins(); i++)
587                 apvOccupancy[i / 128] += occupancyPlot.binHeight(i);
588             for (int i = 0; i < 5; i++)
589                 apvOccupancy[i] /= 128.0;
590 
591             boolean isSensorOK = isOccupancyOK(apvOccupancy);
592             if (!isSensorOK) {
593                 LOGGER.info(String.format("%s: %f %f %f %f %f\n",
594                         SvtPlotUtils.fixSensorNumberLabel(sensor.getName()), apvOccupancy[0], apvOccupancy[1],
595                         apvOccupancy[2], apvOccupancy[3], apvOccupancy[4]));
596                 isSystemOK = false;
597                 if (oldStatus != StatusCode.ALARM)
598                     occupancyStatus.setStatus(StatusCode.ALARM, "Sensor " + SvtPlotUtils.fixSensorNumberLabel(sensor.getName()) + " occupancy abnormal.");
599                 IPlotterStyle plotterStyle = createOccupancyPlotStyle("Max Sample Number", sensor, true);
600                 // region.clear();
601                 // region.plot(occupancyPlot, plotterStyle);
602                 region.applyStyle(plotterStyle);
603 
604             } else {
605                 IPlotterStyle plotterStyle = createOccupancyPlotStyle("Max Sample Number", sensor, false);
606                 // region.clear();
607                 // region.plot(occupancyPlot, plotterStyle);
608                 region.applyStyle(plotterStyle);
609             }
610         }
611         if (isSystemOK)
612             if (oldStatus != StatusCode.OKAY)
613                 occupancyStatus.setStatus(StatusCode.OKAY, "Occupancy looks OK.");
614     }

616     private boolean isOccupancyOK(double[] apvOccupancy) {
617         double peakOccupancy = 0;
618         int highestApv = -1;
619         for (int i = 0; i < 5; i++)
620             if (apvOccupancy[i] > peakOccupancy) {
621                 peakOccupancy = apvOccupancy[i];
622                 highestApv = i;
623             }
624         if (highestApv != 0 && highestApv != 4) {
625             LOGGER.warning("peak occupancy not at edge");
626             return false;
627         }
628         if (peakOccupancy > maxPeakOccupancy || peakOccupancy < minPeakOccupancy) {
629             LOGGER.warning("peak occupancy out of range");
630             return false;
631         }
632         if (highestApv == 0)
633             for (int i = 4; i > 0; i--) {
634                 if (apvOccupancy[i] < 0.1 * peakOccupancy || apvOccupancy[i] < minPeakOccupancy)
635                     continue; // skip through the tail end of the sensor
636                 if (0.9 * apvOccupancy[i] > apvOccupancy[i - 1]) {
637                     LOGGER.warning("occupancy not monotonic");
638                     return false;
639                 }
640             }
641         else if (highestApv == 4)
642             for (int i = 0; i < 4; i++) {
643                 if (apvOccupancy[i] < 0.1 * peakOccupancy || apvOccupancy[i] < minPeakOccupancy)
644                     continue; // skip through the tail end of the sensor
645                 if (0.9 * apvOccupancy[i] > apvOccupancy[i + 1]) {
646                     LOGGER.warning("occupancy not monotonic");
647                     return false;
648                 }
649             }
650 
651         return true;
652     }
