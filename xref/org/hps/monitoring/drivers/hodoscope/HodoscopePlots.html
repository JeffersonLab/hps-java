313     @Override
314     public void process(EventHeader event) {
315 
316         // Get the run number from the event and store it. This will be used
317         // when writing the plots out to a ROOT file
318         if (runNumber == -1) {
319             runNumber = event.getRunNumber();
320         }
321 
322         if (resetPeriod > 0 && eventCount > resetPeriod) { // reset occupancy numbers after resetPeriod events
323             eventCount = 0;
324             resetPlots();
325         }
326 
327         boolean has_cl = event.hasCollection(Cluster.class, ecalClusterName);
328 
329         List<Cluster> clusters = null;
330         if (has_cl) {
331             clusters = event.get(Cluster.class, ecalClusterName);
332         }
333 
334         // Get all of the collections you need //
335         ////e.g.
336         if (!event.hasCollection(RawTrackerHit.class, rawCollectionName)) {
337             return;
338         }
339 
340         // =================== Hodo Hit 'Energies' =====================
341         // Get all of the collections you need //
342         ////e.g.
343         if (!event.hasCollection(RawTrackerHit.class, rawCollectionName)) {
344             return;
345         }
346         // Get RawTrackerHit collection from event.
347         //List<SimpleGenericObject> reconHits = event.get(SimpleGenericObject.class, hodoHitsCollectionName);
348         List<CalorimeterHit> reconHits = event.get(CalorimeterHit.class, hodoHitsCollectionName);
349         //System.out.println("Size of reconHits is " + reconHits.size());
350         //int n_hits = reconHits.get(0).getNInt();
351 
352         int n_hits = reconHits.size();
353 
354         // ======= Loop over hits, and fill corresponding histogram =======
355         for (int ihit = 0; ihit < n_hits; ihit++) {
356 
357             int ix = reconHits.get(ihit).getIdentifierFieldValue("ix");
358             int iy = reconHits.get(ihit).getIdentifierFieldValue("iy");
359             int layer = reconHits.get(ihit).getIdentifierFieldValue("layer");
360             int hole = reconHits.get(ihit).getIdentifierFieldValue("hole");
361             double Energy = reconHits.get(ihit).getRawEnergy();
362             double hit_time = reconHits.get(ihit).getTime();
363 
364             //if( iy < 0 ){Energy = 4.*Energy;}
365             String histname = hitEnergyHistName(ix, iy, layer, hole);
366 
367             //System.out.println(hit_time);
368             if (hit_time > utils.hodo_t_min && hit_time < utils.hodo_t_max) {
369                 hitEnergy.get(histname).fill(Energy);
370 
371                 //int index = hitEnergy.get(histname).coordToIndex(Energy);
372                 //System.out.println("Energy = " + Energy + "     index = " + index);
373                 m_hit_Energies.put(hodoConditions.getChannels().findChannel(ix, iy, layer, hole).getName(), Energy);
374 
375                 String tileHistName = this.tileEnergyHistName(ix, iy, layer);
376 
377                 double EnergyTile = Energy;
378                 if (ix != 0 && ix != 4) {
379                     EnergyTile = 1.25 * Energy / 2.;
380                 }
381 
382                 //HodoUtils.HodoTileIdentifier tile_id = utils.new HodoTileIdentifier(ix, iy, layer);
383                 HodoUtils.HodoTileIdentifier tile_id = utils.new HodoTileIdentifier(ix, iy, layer);
384 
385                 if (m_tile_Energies.get(tile_id) == null) {
386                     m_tile_Energies.put(tile_id, EnergyTile);
387                 } else {
388                     double Etmp = m_tile_Energies.get(tile_id);
389                     m_tile_Energies.put(tile_id, EnergyTile + Etmp);
390                 }
391 
392             }
393         }
394 
395         ArrayList<HodoUtils.HodoTileIdentifier> lvl2_hits = new ArrayList();
396 
397         for (Map.Entry<HodoUtils.HodoTileIdentifier, Double> cur_tile : m_tile_Energies.entrySet()) {
398 
399             HodoUtils.HodoTileIdentifier cur_tile_id = cur_tile.getKey();
400             String tileHistName = tileEnergyHistName(cur_tile_id.ix, cur_tile_id.iy, cur_tile_id.ilayer);
401 
402             tileEnergy.get(tileHistName).fill(cur_tile.getValue());
403 
404             if (cur_tile_id.iy > 0) {
405                 if (cur_tile_id.ilayer == 0) {
406                     occupancyLayerPlots.get("L1 Top Occupancy").fill(cur_tile_id.ix);
407                 } else {
408                     occupancyLayerPlots.get("L2 Top Occupancy").fill(cur_tile_id.ix);
409                 }
410             } else {
411                 if (cur_tile_id.ilayer == 0) {
412                     occupancyLayerPlots.get("L1 Bottom Occupancy").fill(cur_tile_id.ix);
413                 } else {
414                     occupancyLayerPlots.get("L2 Bottom Occupancy").fill(cur_tile_id.ix);
415                 }
416             }
417 
418             if (utils.GoodTileHit(cur_tile_id, m_tile_Energies)) {
419                 tileEnergy2.get(tileHistName).fill(cur_tile.getValue());
420 
421                 boolean hasRightCluster = false;
422                 if (has_cl) {
423 
424                     // =====  ====== ====== Now loop over clusters, if there is a matching cluster, then fill the tileE histogram
425                     for (Cluster cur_clust : clusters) {
426 
427                         double cl_x = cur_clust.getPosition()[0];
428                         double cl_E = cur_clust.getEnergy();
429 
430                         int cl_x_index = hECalCoordtoIndex.coordToIndex(cl_x) + 1;
431 
432                         //if (utils.hasRightClust(cur_tile_id, cl_x, cl_E)) {
433                         if (utils.hasRightClust(cur_tile_id, cl_x_index, cl_E)) {
434                             hasRightCluster = true;
435                         }
436                     }
437 
438                 }
439 
440                 if (hasRightCluster) {
441                     tileEnergy3.get(tileHistName).fill(cur_tile.getValue());
442                     lvl2_hits.add(cur_tile_id);
443                 }
444             }
445 
446         }
447 
448         // Get RawTrackerHit collection from event.
449         List<RawTrackerHit> rawHits = event.get(RawTrackerHit.class, rawCollectionName);
450 
451         ////
452         // ========== fADC plots ===========
453         for (RawTrackerHit cur_hit : rawHits) {
454 
455             short[] adc_Values = cur_hit.getADCValues();
456 
457             long cellID = cur_hit.getCellID();
458 
459             int PMT_ch = hodoConditions.getChannels().findGeometric(cellID).getChannel();
460             int iy = hodoConditions.getChannels().findGeometric(cellID).getIY();
461 
462             HodoUtils.HodoTileIdentifier cur_tile_identifier = utils.new HodoTileIdentifier(hodoConditions.getChannels().findGeometric(cellID).getIX(), hodoConditions.getChannels().findGeometric(cellID).getIY(),
463                     hodoConditions.getChannels().findGeometric(cellID).getLayer());
464 
465             boolean islvl2hit = lvl2_hits.contains(cur_tile_identifier);
466 
467             //hodoConditions
468             //System.out.println("iY = " + iy + "    PMT_ch = " + PMT_ch);
469             if (iy == 1) {
470                 for (int is = 0; is < adc_Values.length; is++) {
471                     fADC_Spectrum_Top.get(String.format("Top fADC %d", PMT_ch)).fill(is, adc_Values[is]);
472                     if (islvl2hit) {
473                         fADC_Spectrum_Top_lvl2.get(String.format("Top fADC %d", PMT_ch)).fill(is, adc_Values[is]);
474                     }
475                 }
476             } else if (iy == -1) {
477                 for (int is = 0; is < adc_Values.length; is++) {
478                     fADC_Spectrum_Bot.get(String.format("Bot fADC %d", PMT_ch)).fill(is, adc_Values[is]);
479                     if (islvl2hit) {
480                         fADC_Spectrum_Bot_lvl2.get(String.format("Bot fADC %d", PMT_ch)).fill(is, adc_Values[is]);
481                     }
482                 }
483             }
484 
485         }
486 
487         // Now Loop over hits is finished, let's look into some matched tile distrivutions
488         double E1_Top = 0;
489 
490         double E7 = 0;
491         double E3 = 0;
492         double E2 = 0;
493         double E8 = 0;
494         double E12 = 0;
495         double E13 = 0;
496         double E10 = 0;
497         double E11 = 0;
498         double E4 = 0;
499         double E5 = 0;
500 
501         if (m_hit_Energies.containsKey(hodoConditions.getChannels().findChannel(0, 1, 0, 0).getName())) {
502             E7 = m_hit_Energies.get(hodoConditions.getChannels().findChannel(0, 1, 0, 0).getName());
503         }
504         if (m_hit_Energies.containsKey(hodoConditions.getChannels().findChannel(2, 1, 0, -1).getName())) {
505             E2 = m_hit_Energies.get(hodoConditions.getChannels().findChannel(2, 1, 0, -1).getName());
506         }
507         if (m_hit_Energies.containsKey(hodoConditions.getChannels().findChannel(2, 1, 0, 1).getName())) {
508             E3 = m_hit_Energies.get(hodoConditions.getChannels().findChannel(2, 1, 0, 1).getName());
509         }
510 
511         if (m_hit_Energies.containsKey(hodoConditions.getChannels().findChannel(1, 1, 1, -1).getName())) {
512             E11 = m_hit_Energies.get(hodoConditions.getChannels().findChannel(1, 1, 1, -1).getName());
