1   package org.hps.monitoring.plotting;
2   
3   import hep.aida.IBaseHistogram;
4   import hep.aida.IDataPointSet;
5   import hep.aida.IFunction;
6   import hep.aida.IPlotter;
7   import hep.aida.IPlotterStyle;
8   import hep.aida.jfree.plotter.PlotterRegion;
9   import hep.aida.jfree.plotter.StandalonePlotter;
10  
11  public class ChartPopup {
12      
13      private IPlotter plotter = null;
14      private PlotterRegion targetRegion = null;
15  
16      public void update(PlotterRegion region) {
17          if (plotter == null) {
18              plotter = new StandalonePlotter();
19              plotter.createRegion(0);
20              targetRegion = (PlotterRegion) plotter.region(0);
21          } else {
22              plotter.hide();
23              targetRegion.clear();
24          }
25          targetRegion.setStyle(region.style());
26          for (Object object : region.getPlottedObjects()) {
27              IPlotterStyle style = region.getState().findPlotterStyle(object);
28              if (object instanceof IBaseHistogram) {
29                  targetRegion.plot((IBaseHistogram) object, style);
30              } else if (object instanceof IDataPointSet) {
31                  targetRegion.plot((IDataPointSet) object, style);
32              } else if (object instanceof IFunction) {
33                  targetRegion.plot((IFunction) object, style);
34              }
35          }
36          plotter.show();
37      }
38  }
