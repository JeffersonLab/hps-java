1   package org.hps.monitoring.plotting;
2   
3   import java.util.Date;
4   import java.util.Timer;
5   import java.util.TimerTask;
6   
7   import org.jfree.chart.JFreeChart;
8   import org.jfree.chart.axis.DateAxis;
9   import org.jfree.data.time.DynamicTimeSeriesCollection;
10  import org.jfree.data.time.RegularTimePeriod;
11  
12  /**
13   * An abstract <tt>TimerTask</tt> to update a strip chart at a regular interval.
14   */
15  public class StripChartUpdater {
16      
17      final JFreeChart chart;
18      final DateAxis domainAxis;
19      final DynamicTimeSeriesCollection dataset;
20      final Timer timer;
21      final TimerTask task;
22      final Long rangeMillis;
23      final ValueProvider valueProvider;
24      long updateInterval;
25          
26      StripChartUpdater(JFreeChart chart, ValueProvider valueProvider, Long rangeMillis, RegularTimePeriod timeBase) {
27          this.chart = chart;                        
28          this.domainAxis = (DateAxis) chart.getXYPlot().getDomainAxis();
29          this.dataset = (DynamicTimeSeriesCollection) chart.getXYPlot().getDataset();
30          this.rangeMillis = rangeMillis;
31          this.updateInterval = timeBase.getLastMillisecond() - timeBase.getFirstMillisecond();            
32          this.valueProvider = valueProvider;
33          timer = new Timer(chart.getTitle().getText() + " Timer");
34          task = new TimerTask() {
35      
36              @Override
37              public void run() {
38                  StripChartUpdater.this.chart.setNotify(false);
39          
40                  dataset.advanceTime();
41                  long time = dataset.getNewestTime().getEnd().getTime();
42                  
43                  float values[] = StripChartUpdater.this.valueProvider.getValues();
44                  dataset.appendData(values);
45          
46                  domainAxis.setRange(
47                          new Date(time - StripChartUpdater.this.rangeMillis), 
48                          new Date(time + updateInterval));
49  
50                  StripChartUpdater.this.chart.setNotify(true);
51                  StripChartUpdater.this.chart.fireChartChanged();                  
52              }            
53          };
54      }
55      
56      JFreeChart getChart() {
57          return chart;
58      }
59       
60      void start() {
61          timer.scheduleAtFixedRate(task, 0, updateInterval);
62      }        
63      
64      public void stop() {
65          timer.cancel();
66      }
67  }
