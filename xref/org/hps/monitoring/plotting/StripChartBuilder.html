12  /**
13   * Utility methods for building strip charts using JFreeChart back end.
14   */
15  public final class StripChartBuilder {
16  
17      private StripChartBuilder() {
18      }
19      
20      /**
21       * Create a strip chart with simple parameter settings.
22       * @param name The title of the strip chart.
23       * @param seriesCount The number of series in the data set.
24       * @param timeBase The time interval for updating.
25       * @param valueProvider The interface for providing values.
26       * @return The StripChartUpdater for the chart.
27       */
28      static StripChartUpdater createDynamicTimeSeriesChart(
29              String name, 
30              int seriesCount, 
31              RegularTimePeriod timeBase,
32              ValueProvider valueProvider) {
33          return createDynamicTimeSeriesChart(name, "Values", seriesCount, null, 9999, timeBase, valueProvider, 10000L);
34      }
35      
36      /**
37       * This creates a strip chart with full parameter settings, which will automatically
38       * update at a certain time interval.
39       * @param name The title of the chart.
40       * @param rangeLabel The range axis label text.
41       * @param seriesCount The number of series in the data set.
42       * @param seriesNames The names of the series (if non-null the length must match seriesCount).
43       * @param itemCount The maximum number of items in the series.
44       * @param timeBase The time unit for updates.
45       * @param valueProvider The interface for providing the series values.
46       * @param rangeView The view in the domain axis around the current data point (milliseconds).
47       * @return The StripChartUpdater for the chart.
48       */
49      static StripChartUpdater createDynamicTimeSeriesChart(
50              String name, 
51              String rangeLabel,
52              int seriesCount, 
53              String[] seriesNames,
54              int itemCount,
55              RegularTimePeriod timeBase,
56              ValueProvider valueProvider,
57              long rangeView) {
58                  
59          if (seriesNames != null && seriesCount != seriesNames.length) {
60              throw new IllegalArgumentException("seriesNames is wrong length");
61          }
62          final DynamicTimeSeriesCollection dataset = new DynamicTimeSeriesCollection(seriesCount, itemCount, timeBase);
63          dataset.setTimeBase(timeBase);
64          for (int series = 0; series < seriesCount; series++) {
65              String seriesName = name + " " + series;
66              if (seriesNames != null) {
67                  seriesName = seriesNames[series];
68              }
69              dataset.addSeries(new float[] {}, series, seriesName);
70          }
71          
72          final JFreeChart chart = ChartFactory.createTimeSeriesChart(name, "hh:mm:ss", rangeLabel, dataset, false, false, false);
73          
74          chart.getXYPlot().getRangeAxis().setAutoRange(true);
75  
76          StripChartUpdater updater = new StripChartUpdater(
77                  chart, 
78                  valueProvider,
79                  rangeView,
80                  timeBase
81                  );
82          
83          updater.start();
84          
85          return updater;
86      }
87          
88      /**
89       * This should be used when the time period for updating is variable.
90       * 
91       * To update a chart of this type:
92       * 
93       * <code>sensorSeries.add(new Minute(new Date()), newData);</code>
94       * 
95       * @param title The title of the chart.
96       * @param yAxisLabel The range axis label.
97       * @param maxAge The maximum age of an item.
98       * @param maxCount The maximum count of items in the single data set series.
99       * @return The chart that was created.
100      */
101     static JFreeChart createTimeSeriesChart(
102             String title, 
103             String yAxisLabel, 
104             int maxAge, 
105             int maxCount,
106             int rangeSize) {
107 
108         TimeSeriesCollection dataset = new TimeSeriesCollection();
109         TimeSeries timeSeries = new TimeSeries("Default Dataset");
110         timeSeries.setMaximumItemAge(maxAge);
111         timeSeries.setMaximumItemCount(maxCount);
112         dataset.addSeries(timeSeries);
113 
114         final JFreeChart result = ChartFactory.createTimeSeriesChart(
115                 title, 
116                 "hh:mm:ss", 
117                 yAxisLabel, 
118                 dataset, 
119                 true, 
120                 false, 
121                 false);
122         final XYPlot plot = result.getXYPlot();
123         plot.getDomainAxis().setAutoRange(true);
124         plot.getDomainAxis().setAutoRangeMinimumSize(rangeSize);
125         NumberAxis rangeAxis = (NumberAxis) plot.getRangeAxis();
126         rangeAxis.setAutoRange(true);
127         rangeAxis.setAutoRangeIncludesZero(true);
128         return result;
129     }

131     /**
132      * <p>
133      * This can be used to create a strip chart with multiple <code>TimeSeries</code> 
134      * in the data set.
135      * <p>
136      * To update a chart of this type, use the following types of method calls:
137      * <pre>
138      * dataset.getSeries(0).add(new Second(time), value1);
139      * dataset.getSeries(1).add(new Second(time), value2);
140      * </pre>
141      * <p>
142      * It is not updated manually but will refresh will values are added to the backing dataset.
143      * 
144      * @param title The title of the chart.
145      * @param yAxisLabel The range axis label.
146      * @param seriesCount The number of series in the dataset.
147      * @param datasetNames The names of the datasets (can be null to use defaults).
148      * @param rangeSize The range of values to show for auto-ranging in domain axis (in milliseconds).
149      * @return The chart that was created.
150      */
151     static JFreeChart createTimeSeriesChart(
152             String title, 
153             String yAxisLabel, 
154             int seriesCount,
155             String[] datasetNames,
156             double rangeSize) {
157 
158         // If dataset names are given, the length must match the number of series requested.
159         if (datasetNames != null && seriesCount != datasetNames.length) {
160             throw new IllegalArgumentException("datasetNames has wrong length: " + datasetNames.length);
161         }
162         
163         // Create the dataset and add empty series to it.
164         TimeSeriesCollection dataset = new TimeSeriesCollection();
165         for (int i = 0; i < seriesCount; i++) {
166             
167             // Uses title for dataset names if none given explicitly.            
168             String datasetName = title;
169             
170             if (datasetNames != null) {
171                 // Use the explicitly given dataset names.
172                 datasetName = datasetNames[i];
173             }
174             TimeSeries timeSeries = new TimeSeries(datasetName);
175             dataset.addSeries(timeSeries);
176         }
177                
178         // Create the chart.
179         final JFreeChart result = ChartFactory.createTimeSeriesChart(
180                 title, 
181                 "hh:mm:ss", 
182                 yAxisLabel, 
183                 dataset, 
184                 true, 
185                 false, 
186                 false);
187         final XYPlot plot = result.getXYPlot();
188 
189         // Configure range axis.
190         NumberAxis rangeAxis = (NumberAxis) plot.getRangeAxis();
191         rangeAxis.setAutoRange(true);
192         rangeAxis.setAutoRangeIncludesZero(true);
193         
194         // Configure domain axis.
195         plot.getDomainAxis().setAutoRange(true);
196         plot.getDomainAxis().setAutoRangeMinimumSize(rangeSize);
197         
198         return result;
199     }
