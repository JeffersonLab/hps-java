1   package org.hps.monitoring.plotting;
2   
3   import hep.aida.IPlotter;
4   import hep.aida.IPlotterRegion;
5   
6   import java.util.ArrayList;
7   import java.util.Collections;
8   import java.util.HashMap;
9   import java.util.List;
10  import java.util.Map.Entry;
11  
12  /**
13   * This is a global registry of plotters used by the monitoring plot factory.
14   */
15  public class PlotterRegistry {
16  
17      HashMap<IPlotter, int[]> plotterMap = new HashMap<IPlotter, int[]>();
18  
19      /**
20       * Clear the list of plotters.
21       */
22      public void clear() {
23          System.out.println("clearing PlotterRegistry");
24          plotterMap.clear();
25      }
26  
27      /**
28       * Find a plotter that contains the region object.
29       * 
30       * @param region The plotter region object.
31       * @return The plotter that contains this region or null if none.
32       */
33      public IPlotter find(IPlotterRegion region) {
34          for (IPlotter plotter : plotterMap.keySet()) {
35              for (int i = 0; i < plotter.numberOfRegions(); i++) {
36                  if (plotter.region(i) == region) {
37                      return plotter;
38                  }
39              }
40          }
41          return null;
42      }
43  
44      /**
45       * Get the tab indices of the plotter.
46       * 
47       * @param plotter the plotter to lookup
48       * @return the tab indices of the plotter or <code>null</code> if it doesn't exist
49       */
50      public int[] getTabIndices(IPlotter plotter) {
51          return plotterMap.get(plotter);
52      }
53  
54      /**
55       * Register a plotter along with its tab indices.
56       * 
57       * @param plotter The plotter to register.
58       * @param index1 The top tab index.
59       * @param index2 The sub-tab index.
60       */
61      public void register(IPlotter plotter, int index1, int index2) {
62          plotterMap.put(plotter, new int[] {index1, index2});
63      }
64  
65      /**
66       * Find a plotter by its tab indices e.g. those that are currently selected in an application.
67       * 
68       * @param index1 The top tab index.
69       * @param index2 The sub-tab index.
70       * @return The plotter or null if none found.
71       */
72      public IPlotter find(int index1, int index2) {
73          for (Entry<IPlotter, int[]> entry : plotterMap.entrySet()) {
74              int[] indices = entry.getValue();
75              if (indices[0] == index1 && indices[1] == index2) {
76                  return entry.getKey();
77              }
78          }
79          return null;
80      }
81  
82      /**
83       * Get the current collection of plotters as an unmodifiable collection.
84       * 
85       * @return The current collection of plotters.
86       */
87      public List<IPlotter> getPlotters() {
88          return Collections.unmodifiableList(new ArrayList<IPlotter>(plotterMap.keySet()));
89      }
90  }
