18  /**
19   * This class implements an AIDA <code>IPlotterFactory</code> for the monitoring application. It
20   * extends the JFree plotter by putting plots into tabs. Each plotter factory is given its own
21   * top-level tab in a root tabbed pane, under which are separate tabs for each plotter. The root
22   * pane is static and shared across all plotter factories. The top level component is set externally
23   * by the MonitoringApplication before any calls to AIDA are made from Drivers.
24   */
25  public class MonitoringPlotFactory extends PlotterFactory {
26      
27      // Global plotter registry.
28      static PlotterRegistry plotters = new PlotterRegistry();
29      
30      // The name of the factory which will be used in naming tabs in the monitoring app.
31      String name = null;
32  
33      // The GUI tabs for this factory's plots.
34      private JTabbedPane tabs = new JTabbedPane();
35  
36      // Root pane where this factory's top-level tab will be inserted.
37      private static JTabbedPane rootPane = null;
38  
39      // The region listener for handling mouse clicks in a region.
40      private static PlotterRegionListener regionListener;
41      
42      // The current tab index.
43      int tabIndex;
44  
45      /**
46       * Set the plot region listener.
47       * @param regionListener The plot region listener.
48       */
49      public static void setPlotterRegionListener(PlotterRegionListener regionListener) {
50          MonitoringPlotFactory.regionListener = regionListener;
51      }
52  
53      /**
54       * Class constructor for unnamed factory.
55       */
56      MonitoringPlotFactory() {
57          super();
58          setIsEmbedded(true);
59          setupRootPane("  ");
60          if (regionListener != null)
61              addPlotterRegionListener(regionListener);
62      }
63  
64      /**
65       * Class constructor for named factory.
66       * @param name The name of the factory.
67       */
68      MonitoringPlotFactory(String name) {
69          super();
70          this.name = name;
71          setIsEmbedded(true);
72          setupRootPane(name);
73          if (regionListener != null)
74              addPlotterRegionListener(regionListener);
75      }
76  
77      /**
78       * Setup the root GUI pane of this factory for display of plots in tabs.
79       * @param name The tab's label.
80       */
81      private void setupRootPane(String name) {
82          // FIXME: Hack to disregard call from an AIDA related class.
83          if (!(new RuntimeException()).getStackTrace()[2].getClassName().equals("hep.aida.ref.plotter.style.registry.StyleStoreXMLReader")) {
84              rootPane.addTab(name, tabs);
85              tabIndex = rootPane.getTabCount() - 1;
86              rootPane.setTabComponentAt(tabIndex, new JLabel(name));
87          }
88      }
89  
90      /**
91       * Create a named plotter.
92       * @param plotterName The name of the plotter.
93       * @return The plotter.
94       */
95      public IPlotter create(String plotterName) {
96          IPlotter plotter = super.create(plotterName);
97          setupPlotterTab(plotterName, plotter);
98          return plotter;
99      }
100

102      * Create an unnamed plotter.
103      * @return The plotter.

105     public IPlotter create() {
106         return create((String) null);
107     }

113     public static void setRootPane(JTabbedPane rootPane) {
114         MonitoringPlotFactory.rootPane = rootPane;
115     }

117     /**
118      * Setup a plotter tab.
119      * @param plotterName The name of the plotter.
120      * @param plotter The IPlotter which will plot into the tab.
121      */
122     private void setupPlotterTab(String plotterName, IPlotter plotter) {
123         
124         // Setup the plotter's GUI pane and tab.
125         JPanel plotterPanel = new JPanel(new BorderLayout());
126         plotterPanel.add(PlotterUtilities.componentForPlotter(plotter), BorderLayout.CENTER);
127         tabs.addTab(plotterName, plotterPanel);
128         int plotterIndex = tabs.getTabCount() - 1;
129         tabs.setTabComponentAt(plotterIndex, new JLabel(plotterName));
130         
131         // Register plotter globally with its tab indices.
132         plotters.register(plotter, tabIndex, plotterIndex);
133     }
134 
135     /**
136      * Add a <code>JFreeChart</code> to one of the tabs.
137      * @param chart The JFreeChart object to add.
138      */
139     private void addChart(JFreeChart chart) {
140         ChartPanel panel = new ChartPanel(chart);
141         tabs.addTab(chart.getTitle().getText(), panel);
142         tabs.setTabComponentAt(tabs.getTabCount() - 1, new JLabel(chart.getTitle().getText()));
143     }

145     /**
146      * This creates a strip chart with full parameter settings, which will automatically
147      * update at a certain time interval.
148      * @param name The title of the chart.
149      * @param rangeLabel The range axis label text.
150      * @param seriesCount The number of series in the data set.
151      * @param seriesNames The names of the series (if non-null the length must match seriesCount).
152      * @param itemCount The maximum number of items in the series.
153      * @param timeBase The time unit for updates.
154      * @param valueProvider The interface for providing the series values.
155      * @param rangeView The view in the domain axis around the current data point (applied to plus and minus).
156      * @return The StripChartUpdater for the chart.
157      */
158     public StripChartUpdater createStripChart(
159             String name, 
160             String rangeLabel,
161             int seriesCount, 
162             String[] seriesNames,
163             int itemCount,
164             RegularTimePeriod timeBase,
165             ValueProvider valueProvider,
166             long rangeView) {
167         StripChartUpdater updater = StripChartBuilder.createDynamicTimeSeriesChart(
168                 name, 
169                 rangeLabel, 
170                 seriesCount, 
171                 seriesNames,
172                 itemCount, 
173                 timeBase, 
174                 valueProvider, 
175                 rangeView);
176         addChart(updater.getChart());
177         return updater;
178     }
179     
180     /**
181      * Create a strip chart with simple parameter settings.
182      * @param name The title of the strip chart.
183      * @param seriesCount The number of series in the data set.
184      * @param timeBase The time interval for updating.
185      * @param valueProvider The interface for providing values.
186      * @return The StripChartUpdater for the chart.
187      */
188     public StripChartUpdater createStripChart(
189             String name, 
190             int seriesCount, 
191             RegularTimePeriod timeBase,
192             ValueProvider valueProvider) {
193         return createStripChart(
194                 name, "Values", 
195                 seriesCount, 
196                 null, 
197                 9999, 
198                 timeBase, 
199                 valueProvider, 
200                 10000L);
201     }
202 
203     /**
204      * Create a strip chart which must be updated manually.
205      * @param title The title of the chart.
206      * @param yAxisLabel The y axis label.
207      * @param maxAge The maximum age of items to keep.
208      * @param maxCount The maximum count of items.
209      * @param rangeSize The size of the data range.
210      * @return The strip chart as a <code>JFreeChart</code>.
211      */
212     public JFreeChart createStripChart(String title, String yAxisLabel, int maxAge, int maxCount, int rangeSize) {
213         JFreeChart stripChart = StripChartBuilder.createTimeSeriesChart(title, yAxisLabel, maxAge, maxCount, rangeSize);
214         stripChart.getLegend().setVisible(false); /* Legend turned off for now. */
215         addChart(stripChart);
216         return stripChart;
217     }       
218     
219     public JFreeChart createTimeSeriesChart(
220             String title, 
221             String yAxisLabel, 
222             int seriesCount,
223             String[] datasetNames,
224             double rangeSize) {
225         JFreeChart chart = StripChartBuilder.createTimeSeriesChart(title, yAxisLabel, seriesCount, datasetNames, rangeSize);
226         addChart(chart);
227         return chart;
228     }
