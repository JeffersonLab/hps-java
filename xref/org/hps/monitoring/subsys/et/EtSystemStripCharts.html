3   import java.util.ArrayList;
4   import java.util.Date;
5   import java.util.List;
6   
7   import org.hps.monitoring.plotting.MonitoringPlotFactory;
8   import org.hps.monitoring.subsys.SystemStatistics;
9   import org.hps.monitoring.subsys.SystemStatisticsImpl;
10  import org.hps.monitoring.subsys.SystemStatisticsListener;
11  import org.hps.record.et.EtEventProcessor;
12  import org.jfree.chart.JFreeChart;
13  import org.jfree.data.time.Second;
14  import org.jfree.data.time.TimeSeriesCollection;
15  import org.jlab.coda.et.EtEvent;
16  import org.lcsim.util.aida.AIDA;
17  
18  /**
19   * This will show a series of strip charts from ET system performance statistics
20   * such as event and data rates.
21   */
22  public class EtSystemStripCharts extends EtEventProcessor implements SystemStatisticsListener {
23  
24      // The system statistics.
25      SystemStatisticsImpl stats = new SystemStatisticsImpl();
26      
27      // Plotting API.
28      MonitoringPlotFactory plotFactory = 
29              (MonitoringPlotFactory) AIDA.defaultInstance().analysisFactory().createPlotterFactory("ET System Monitoring");
30      
31      // List of charts.
32      List<JFreeChart> charts = new ArrayList<JFreeChart>();
33      
34      // Range size in milliseconds.
35      static final double RANGE_SIZE = 200000;
36      
37      // Chart collection indices.
38      static final int DATA_RATE_COLLECTION_INDEX = 0;    
39      static final int TOTAL_DATA_COLLECTION_INDEX = 1;    
40      static final int EVENT_RATE_COLLECTION_INDEX = 2;
41      static final int TOTAL_EVENTS_COLLECTION_INDEX = 3;
42          
43      public EtSystemStripCharts() {          
44          // Set 2 seconds between statistics updates.
45          stats.setNominalTickLengthMillis(1000);
46      }
47      
48      /**
49       * Setup the strip charts for ET system monitoring and start accumulating statistics.
50       */
51      @Override
52      public void startJob() {
53             
54          // Register this class as a listener to activate update at end of statistics clock tick.
55          stats.addSystemStatisticsListener(this);
56  
57          // Start systems statistics task.
58          stats.start();
59      }
60  
61      /**
62       * Create the strip charts for plotting the basic ET system statistics.
63       */
64      private void createStripCharts() {
65          
66          // Data rate and average data reate in megabytes per second.
67          charts.add(plotFactory.createTimeSeriesChart(
68                  "Data Rate", 
69                  "MB / second",
70                  2, 
71                  new String[] { "Data Rate", "Average Data Rate" },
72                  RANGE_SIZE));
73                  
74          // Total megabytes received.
75          charts.add(plotFactory.createTimeSeriesChart("Total Data", "Megabytes", 1, null, RANGE_SIZE));
76          
77          // Event rate and average event rate in hertz.
78          charts.add(plotFactory.createTimeSeriesChart(
79                  "Event Rate", 
80                  "Hz", 
81                  2, 
82                  new String[] { "Event Rate", "Average Event Rate" }, 
83                  RANGE_SIZE));
84          
85          // Total number of events received.
86          charts.add(plotFactory.createTimeSeriesChart("Total Events", "Number of Events", 1, null, RANGE_SIZE));
87                
88      }
89  
90      @Override
91      public void process(EtEvent event) {
92          stats.update(event.getLength());
93      }
94  
95      public void endJob() {
96          // Stop system statistics task.
97          stats.stop();
98      }
99      
100     TimeSeriesCollection getTimeSeriesCollection(int chartIndex) {
101         return (TimeSeriesCollection) charts.get(chartIndex).getXYPlot().getDataset();
102     }
103 
104     /**
105      * Hook for updating the charts at end of statistics clock tick.
106      * @param stats The statistics with the system information.
107      */
108     @Override
109     public void endTick(SystemStatistics stats) {
110         
111         Date now = new Date(stats.getTickEndTimeMillis());
112                 
113         getTimeSeriesCollection(DATA_RATE_COLLECTION_INDEX).getSeries(0).addOrUpdate(
114                 new Second(now), stats.getMegabytesPerSecond());
115         getTimeSeriesCollection(DATA_RATE_COLLECTION_INDEX).getSeries(1).addOrUpdate(
116                 new Second(now), stats.getAverageMegabytesPerSecond());
117         
118         getTimeSeriesCollection(TOTAL_DATA_COLLECTION_INDEX).getSeries(0).addOrUpdate(
119                 new Second(now), stats.getTotalMegabytes());
120         
121         getTimeSeriesCollection(EVENT_RATE_COLLECTION_INDEX).getSeries(0).addOrUpdate(
122                 new Second(now), stats.getEventsPerSecond());
123         getTimeSeriesCollection(EVENT_RATE_COLLECTION_INDEX).getSeries(1).addOrUpdate(
124                 new Second(now), stats.getAverageEventsPerSecond());
125         
126         getTimeSeriesCollection(TOTAL_EVENTS_COLLECTION_INDEX).getSeries(0).addOrUpdate(
127                 new Second(now), stats.getTotalEvents());
128     }
129      
130     @Override
131     public void started(SystemStatistics stats) {
132         createStripCharts();
133     }
134    
135     @Override
136     public void stopped(SystemStatistics stats) {
137     }
138 }
