1   package org.hps.monitoring.subsys.et;
2   
3   import java.util.Timer;
4   import java.util.TimerTask;
5   
6   import org.hps.monitoring.subsys.StatusCode;
7   import org.hps.monitoring.subsys.Subsystem;
8   import org.hps.monitoring.subsys.SystemStatus;
9   import org.hps.monitoring.subsys.SystemStatusImpl;
10  import org.hps.record.et.EtEventProcessor;
11  import org.jlab.coda.et.EtEvent;
12  
13  /**
14   * This is a class for monitoring the ET system.
15   */
16  public final class EtSystemMonitor extends EtEventProcessor {
17  
18      SystemStatus systemStatus;
19      int events = 0;
20      long eventReceivedMillis = 0;
21      long warningIntervalMillis = 10000;
22      int timerInterval = 10000;
23      Timer timer = new Timer("ET Event Monitor");
24  
25      public EtSystemMonitor() {
26          systemStatus = new SystemStatusImpl(Subsystem.ET, "ET System Monitor", false);
27          systemStatus.setStatus(StatusCode.UNKNOWN, "System is not active yet.");
28      }
29  
30      public void setWarningIntervalMillis(long warningIntervalMillis) {
31          this.warningIntervalMillis = warningIntervalMillis;
32      }
33      
34      public void setTimerInterval(int timerInterval) {
35          this.timerInterval = timerInterval;
36      }
37  
38      public void startJob() {
39          systemStatus.setStatus(StatusCode.OKAY, "ET job started.");
40          TimerTask task = new TimerTask() {
41              long startedMillis = 0;
42  
43              public void run() {
44                  if (startedMillis == 0)
45                      startedMillis = System.currentTimeMillis();
46                  long elapsedMillis = 0;
47                  if (eventReceivedMillis == 0)
48                      elapsedMillis = System.currentTimeMillis() - startedMillis;
49                  else
50                      elapsedMillis = System.currentTimeMillis() - eventReceivedMillis;
51                  if (elapsedMillis > warningIntervalMillis)
52                      systemStatus.setStatus(StatusCode.WARNING, "No ET events received for " + elapsedMillis + " millis.");
53                  else if (systemStatus.getStatusCode()!=StatusCode.OKAY)
54                      systemStatus.setStatus(StatusCode.OKAY, "ET events received.");
55              }
56          };
57  
58          timer.schedule(task, 0, timerInterval);
59      }
60  
61      public void process(EtEvent event) {
62          eventReceivedMillis = System.currentTimeMillis();
63      }
64  
65      public void endJob() {
66          timer.cancel();
67          timer.purge();
68          systemStatus.setStatus(StatusCode.OFFLINE, "ET job ended.");
69      }
70  }
