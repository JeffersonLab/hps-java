22  /**
23   * Reads output of org.hps.recon.ecal.RunningPedestalDriver and makes strip charts.
24   */
25  // FIXME: This class doesn't belong in this package because it is not a sub-system monitor.
26  public class EcalPedestalMonitor extends Driver {
27  
28      static final int REFRESH_RATE = 10 * 1000; // units = ms
29      static final double DOMAIN_SIZE = 4 * 60 * 60 * 1000; // x-axis range (ms)
30      static final int crates[] = {1, 2};
31      static final int slots[] = {3, 4, 5, 6, 7, 8, 9, 14, 15, 16, 17, 18, 19, 20};
32      static final String slotNames[] = {"Sl3", "Sl4", "Sl5", "Sl6", "Sl7", "Sl8", "Sl9", "Sl14", "Sl15", "Sl16", "Sl17",
33              "Sl18", "Sl19", "Sl20"};
34      static final String collectionName = "EcalRunningPedestals";
35  
36      long currentTime;
37      long previousTime = 0;
38      int nDetectorChanges = 0;
39      private EcalConditions ecalConditions = null;
40      List<JFreeChart> charts = new ArrayList<JFreeChart>();
41  
42      public void detectorChanged(Detector detector) {
43          if (nDetectorChanges++ > 0)
44              return;
45          ecalConditions = DatabaseConditionsManager.getInstance().getEcalConditions();
46          MonitoringPlotFactory plotFactory = (MonitoringPlotFactory) AIDA.defaultInstance().analysisFactory()
47                  .createPlotterFactory("ECal Pedestal Monitoring");
48          for (int crate : crates) {
49              charts.add(plotFactory.createTimeSeriesChart("Crate " + crate, "Offset Pedestal (ADC)", slots.length,
50                      slotNames, DOMAIN_SIZE));
51          }
52      }
53  
54      @Override
55      public void process(EventHeader event) {
56  
57          if (!event.hasItem(collectionName))
58              return;
59  
60          currentTime = System.currentTimeMillis();
61          if (currentTime - previousTime < REFRESH_RATE)
62              return;
63          previousTime = currentTime;
64  
65          // get the running pedestals:
66          Map<EcalChannel, Double> peds = (Map<EcalChannel, Double>) event.get(collectionName);
67  
68          // tally slot pedestals:
69          Map<Integer, Map<Integer, Double>> pedsum = new HashMap<Integer, Map<Integer, Double>>();
70          Map<Integer, Map<Integer, Integer>> npedsum = new HashMap<Integer, Map<Integer, Integer>>();
71  
72          for (EcalChannel cc : peds.keySet()) {
73              final Double ped = peds.get(cc);
74              final int crate = cc.getCrate();
75              final int slot = cc.getSlot();
76              if (!pedsum.containsKey(crate)) {
77                  pedsum.put(crate, new HashMap<Integer, Double>());
78              }
79              if (!npedsum.containsKey(crate)) {
80                  npedsum.put(crate, new HashMap<Integer, Integer>());
81              }
82              if (!pedsum.get(crate).containsKey(slot)) {
83                  pedsum.get(crate).put(slot, 0.0);
84              }
85              if (!npedsum.get(crate).containsKey(slot)) {
86                  npedsum.get(crate).put(slot, 0);
87              }
88              npedsum.get(crate).put(slot, npedsum.get(crate).get(slot) + 1);
89              pedsum.get(crate).put(slot, pedsum.get(crate).get(slot) + ped);
90          }
91  
92          // fill strip charts:
93          long now = System.currentTimeMillis();
94          for (int crate : pedsum.keySet()) {
95              TimeSeriesCollection cc = getTimeSeriesCollection(crate - 1);
96              JFreeChart chart = charts.get(crate - 1);
97              DateAxis ax = (DateAxis) chart.getXYPlot().getDomainAxis();
98              ax.setRange(now - DOMAIN_SIZE, now);
99              for (int slot = 0; slot < slots.length; slot++) {
100                 double ped = pedsum.get(crate).get(slots[slot]) / npedsum.get(crate).get(slots[slot]);
101                 ped -= getAveragePedestal(crate, slots[slot]) - slot + 9;
102                 cc.getSeries(slot).addOrUpdate(new Second(new Date()), ped);
103             }
104         }
105     }
106 
107     TimeSeriesCollection getTimeSeriesCollection(int chartIndex) {
108         return (TimeSeriesCollection) charts.get(chartIndex).getXYPlot().getDataset();
109     }
110 
111     public EcalChannel findChannel(int crate, int slot, int chan) {
112         for (EcalChannel cc : ecalConditions.getChannelCollection()) {
113             if (crate == cc.getCrate() && slot == cc.getSlot() && chan == cc.getChannel()) {
114                 return cc;
115             }
116         }
117         throw new RuntimeException(String.format("Could not find channel:  (crate,slot,channel)=(%d,%d,%d)", crate,
118                 slot, chan));
119     }
120 
121     public double getAveragePedestal(int crate, int slot) {
122         int nsum = 0;
123         double sum = 0;
124         for (int chan = 0; chan < 16; chan++) {
125             for (EcalChannel cc : ecalConditions.getChannelCollection()) {
126                 if (cc.getCrate() != crate || cc.getSlot() != slot || cc.getChannel() != chan)
127                     continue;
128                 sum += getStaticPedestal(crate, slot, chan);
129                 nsum++;
130             }
131         }
132         return (nsum > 0 ? sum / nsum : sum);
133     }
134 
135     public double getStaticPedestal(int crate, int slot, int chan) {
136         return ecalConditions.getChannelConstants(findChannel(crate, slot, chan)).getCalibration().getPedestal();
137     }
