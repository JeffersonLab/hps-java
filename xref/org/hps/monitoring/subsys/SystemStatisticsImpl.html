93      @Override
94      public double getEventsPerSecond() {
95          if (eventsInTick > 0 && getTickElapsedMillis() > 0) {
96              return (double) eventsInTick / millisToSeconds(getTickElapsedMillis());
97          } else {
98              return 0.;
99          }

102     @Override
103     public double getAverageEventsPerSecond() {
104         try {
105             return Double.parseDouble(decimalFormat.format(totalEvents / millisToSeconds(getTotalElapsedMillis())));
106         } catch (NumberFormatException e) {
107             return 0;
108         }
109     }

121     public double getTotalMegabytes() {

125     @Override
126     public double getAverageMegabytesPerSecond() {
127         try {
128             return Double.parseDouble(decimalFormat.format(bytesToMb(totalBytes) / millisToSeconds(getTotalElapsedMillis())));
129         } catch (NumberFormatException e) {
130             return Double.NaN;
131         }
132     }

134     @Override
135     public double getBytesPerSecond() {
136         if (bytesInTick > 0 && getTickElapsedMillis() > 0)
137             return (double) bytesInTick / millisToSeconds(getTickElapsedMillis());
138         else
139             return 0.;
140     }
141     
142     @Override
143     public double getMegabytesPerSecond() {
144         double bytes = getBytesPerSecond();
145         if (bytes > 0) {
146             return bytesToMb(bytes);
147         } else {
148             return 0;
149         }
150     }

152     @Override
153     public void start() {
154 
155         // Set session start time variables.
156         long currentTimeMillis = System.currentTimeMillis();
157         startTimeMillis = currentTimeMillis;
158         tickStartTimeMillis = currentTimeMillis;
159         
160         // Notify listeners of start.
161         for (SystemStatisticsListener listener : listeners) {
162             listener.started(this);
163         }
164         
165         // Start timer task which executes at the nominal tick length to calculate statistics periodically.
166         TimerTask task = new TimerTask() {
167             public void run() {
168                 
169                 // End the current tick.
170                 endTick();
171                
172                 // Start the new tick.
173                 nextTick();
174             }
175         };
176         timer = new Timer();
177         timer.schedule(task, 0, nominalTickLengthMillis);
178     }
179     
180     void endTick() {
181  
182         // Set absolute end time of current tick.
183         this.tickEndTimeMillis = System.currentTimeMillis();
184         
185         // Activate listeners.
186         for (SystemStatisticsListener listener : listeners) {
187             listener.endTick(this);
188         }
189     }
190 
191     @Override
192     public void stop() {
193         // Kill the Timer.
194         if (timer != null) {
195             timer.cancel();
196             timer.purge();
197         }
198 
199         // Set stop time.
200         stopTimeMillis = System.currentTimeMillis();
201         
202         // Notify listeners of stop.
203         for (SystemStatisticsListener listener : listeners) {
204             listener.stopped(this);
205         }
206     }
207     
208     public void addSystemStatisticsListener(SystemStatisticsListener listener) {
209         listeners.add(listener);
210     }

212     @Override
213     public void printSession(PrintStream ps) {
214         ps.println("session statistics ...");
215         ps.println("  timeElapsedMillis = " + this.getTotalElapsedMillis());
216         ps.println("  cumulativeEvents = " + this.getTotalEvents());
217         ps.println("  averageEventsPerSecond = " + this.getAverageEventsPerSecond());
218         ps.println("  averageMegaBytesPerSecond = " + this.getAverageMegabytesPerSecond());
219 
220     }
221 
222     @Override
223     public void printTick(PrintStream ps) {
224         ps.println("tick statistics ...");
225         ps.println("  tickElapsedMillis = " + this.getTickElapsedMillis());
226         ps.println("  eventsSinceTick = " + this.getEventsReceived());
227         ps.println("  bytesSinceTick = " + this.getBytesReceived());
228     }
229    
230     void addEvent() {
231         eventsInTick += 1;
232         totalEvents += 1;
233     }
234 
235     void addData(int size) {
236         bytesInTick += size;
237         totalBytes += size;
238     }
239 
240     // Bytes to megabytes to 2 decimal places.
241     static final double bytesToMb(long size) {
242         return Double.parseDouble(decimalFormat.format((double) size / Mb));
243     }
244     
245     // Bytes to megabytes to 2 decimal places.
246     static final double bytesToMb(double size) {
247         return Double.parseDouble(decimalFormat.format(size / Mb));
248     }
249 
250     static final double millisToSeconds(long millis) {
251         return ((double) millis) / 1000.;
252     }
253 
254     synchronized void nextTick() {
255         eventsInTick = 0;
256         bytesInTick = 0;
257         tickStartTimeMillis = System.currentTimeMillis();
258     }
