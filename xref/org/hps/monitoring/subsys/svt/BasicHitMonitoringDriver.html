1   package org.hps.monitoring.subsys.svt;
2   
3   import java.util.Timer;
4   import java.util.TimerTask;
5   
6   import org.hps.monitoring.subsys.StatusCode;
7   import org.hps.monitoring.subsys.Subsystem;
8   import org.hps.monitoring.subsys.SystemStatus;
9   import org.hps.monitoring.subsys.SystemStatusImpl;
10  import org.lcsim.event.EventHeader;
11  import org.lcsim.event.RawTrackerHit;
12  import org.lcsim.util.Driver;
13  
14  /**
15   * This is a basic example of monitoring hits using an updateable <code>SystemStatus</code>. It
16   * checks from a <code>TimerTask</code> once per second whether <code>RawTrackerHit</code> objects
17   * are being received by the {@link #process(EventHeader)} method.
18   */
19  public class BasicHitMonitoringDriver extends Driver {
20  
21      SystemStatus status;
22      long hitsReceivedMillis;
23      long warningIntervalMillis = 1000;
24      Timer timer;
25      static final String hitsCollectionName = "SVTRawTrackerHits";
26  
27      public BasicHitMonitoringDriver() {
28          status = new SystemStatusImpl(Subsystem.SVT, "Checks that SVT hits are received.", true);
29          status.setStatus(StatusCode.UNKNOWN, "Status is unknown.");
30      }
31  
32      public void setWarningIntervalMillis(long warningIntervalMillis) {
33          this.warningIntervalMillis = warningIntervalMillis;
34      }
35  
36      public void startOfData() {
37          if (hitsCollectionName == null)
38              throw new RuntimeException("The hitsCollectionName was never set.");
39          status.setStatus(StatusCode.OKAY, "SVT hit monitor started.");
40          timer = new Timer("SVT Hit Monitor");
41          TimerTask task = new TimerTask() {
42              long startedMillis = 0;
43  
44              public void run() {
45                  if (startedMillis == 0)
46                      startedMillis = System.currentTimeMillis();
47                  long elapsedMillis = 0;
48                  if (hitsReceivedMillis == 0)
49                      elapsedMillis = System.currentTimeMillis() - startedMillis;
50                  else
51                      elapsedMillis = System.currentTimeMillis() - hitsReceivedMillis;
52                  if (elapsedMillis > warningIntervalMillis)
53                      status.setStatus(StatusCode.WARNING, "No SVT hits received for " + elapsedMillis + " millis.");
54                  else
55                      status.setStatus(StatusCode.OKAY, "SVT hits received.");
56              }
57          };
58          // Task will run once per second.
59          timer.schedule(task, 0, 1000);
60      }
61  
62      public void process(EventHeader event) {
63          if (event.hasCollection(RawTrackerHit.class, hitsCollectionName))
64              if (event.get(RawTrackerHit.class, hitsCollectionName).size() > 0)
65                  hitsReceivedMillis = System.currentTimeMillis();
66      }
67  
68      public void endOfData() {
69          timer.cancel();
70          timer.purge();
71          status.setStatus(StatusCode.OFFLINE, "SVT hit monitor went offline.");
72          status.setActive(false);
73      }
74  }
