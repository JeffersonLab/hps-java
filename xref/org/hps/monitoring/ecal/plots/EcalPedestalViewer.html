23  /*
24   * Display histograms created by org.hps.recon.ecal.EcalPedestalCalculator
25   * 
26   * When user clicks on crystal in Kyle's event viewer, the corresponding channel's
27   * pedestal histogram is drawn.
28   */
29  public class EcalPedestalViewer extends Driver implements CrystalListener, ActionListener {
30  
31      // this has to match the one in EcalPedstalCalculator:
32      private String histoNameFormat = "Ecal/Pedestals/Mode7/ped%03d";
33  
34      private AIDA aida = AIDA.defaultInstance();
35      private IPlotter plotter;
36      private IPlotterFactory plotterFactory;
37      private IPlotterStyle pstyle;
38      private PEventViewer viewer;
39  
40      static final String[] colors = {"red", "black", "blue", "green", "yellow", "pink", "cyan", "magenta", "brown"};
41      static final int nRows = 3;
42      static final int nColumns = 3;
43      private int theRegion = 0;
44  
45      @Override
46      public void detectorChanged(Detector detector) {
47          plotterFactory = aida.analysisFactory().createPlotterFactory("ECal Peds");
48          plotter = plotterFactory.create("ECal Peds");
49          plotter.createRegions(nColumns, nRows);
50          // Plot dummmy histos, else null plotter regions later:
51          for (int ii = 0; ii < nColumns * nRows; ii++) {
52              plotter.region(ii).plot(aida.histogram1D("ASDF" + ii, 100, 11e9, 11e11));
53          }
54          plotter.show();
55  
56          pstyle = plotterFactory.createPlotterStyle();
57          pstyle.xAxisStyle().labelStyle().setBold(true);
58          pstyle.yAxisStyle().labelStyle().setBold(true);
59          pstyle.xAxisStyle().tickLabelStyle().setBold(true);
60          pstyle.yAxisStyle().tickLabelStyle().setBold(true);
61          pstyle.xAxisStyle().lineStyle().setColor("black");
62          pstyle.yAxisStyle().lineStyle().setColor("black");
63          pstyle.xAxisStyle().lineStyle().setThickness(2);
64          pstyle.yAxisStyle().lineStyle().setThickness(2);
65          pstyle.dataStyle().errorBarStyle().setThickness(0);
66          pstyle.legendBoxStyle().setVisible(false);
67      }
68  
69      @Override
70      public void startOfData() {
71          File config = new File("ecal-mapping-config.csv");
72          if (config.exists() && config.canRead()) {
73              try {
74                  viewer = new PDataEventViewer(config.getAbsolutePath());
75              } catch (IOException e) {
76                  viewer = new PEventViewer();
77              }
78          } else {
79              viewer = new PEventViewer();
80          }
81          viewer.addCrystalListener(this);
82          viewer.setVisible(true);
83      }
84  
85      @Override
86      public void actionPerformed(ActionEvent ae) {
87      }
88  
89      @Override
90      public void crystalActivated(CrystalEvent e) {
91      }
92  
93      @Override
94      public void crystalDeactivated(CrystalEvent e) {
95      }
96  
97      @Override
98      public void crystalClicked(CrystalEvent e) {
99          aida.tree().cd("/");
100         Point ecalPoint = Viewer.toEcalPoint(e.getCrystalID());
101         if (ecalPoint.x == 0 || ecalPoint.y == 0)
102             return;
103         if (EcalMonitoringUtilities.isInHole(ecalPoint.y, ecalPoint.x))
104             return;
105         final int cid = EcalMonitoringUtilities.getChannelIdFromRowColumn(ecalPoint.y, ecalPoint.x);
106         IHistogram1D hist = aida.histogram1D(String.format(histoNameFormat, cid));
107         if (hist == null) {
108             System.err.println("Running the Driver?");
109         } else {
110             hist.setTitle(String.format("(%d,%d)", ecalPoint.x, ecalPoint.y));
111             pstyle.dataStyle().lineStyle().setParameter("color", colors[theRegion % colors.length]);
112             plotter.region(theRegion).clear();
113             plotter.region(theRegion).plot(hist, pstyle);
114             plotter.region(theRegion).refresh();
115             theRegion = (theRegion + 1) % (nColumns * nRows);
116         }
117     }
118 
119 }
