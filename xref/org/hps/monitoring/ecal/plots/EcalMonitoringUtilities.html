64      public static int getChannelIdFromRowColumn(final int row, final int col) {
65          final int ix = col + XOFFSET + (col > 0 ? -1 : 0);
66          final int iy = row + YOFFSET + (row > 0 ? -1 : 0);
67          int cid = ix + 2 * XOFFSET * (2 * YOFFSET - iy - 1) + 1;
68          if (row == 1 && col >= XHOLESTART) {
69              cid -= XHOLEWIDTH;
70          } else if (row == -1 && col < XHOLESTART) {
71              cid -= XHOLEWIDTH;
72          } else if (row < 0) {
73              cid -= 2 * XHOLEWIDTH;
74          }
75          return cid;
76      }

89      static IPlotterStyle getPlotterStyle(final IBaseHistogram histogram) {
90          final StyleRegistry styleRegistry = StyleRegistry.getStyleRegistry();
91          final IStyleStore store = styleRegistry.getStore("DefaultStyleStore");
92          IPlotterStyle style = null;
93          if (histogram instanceof IHistogram1D || histogram instanceof ICloud1D || histogram instanceof IProfile1D) {
94              style = store.getStyle("DefaultHistogram1DStyle");
95          } else if (histogram instanceof IHistogram2D || histogram instanceof IProfile2D) {
96              style = store.getStyle("DefaultColorMapStyle");
97          }
98          if (style == null) {
99              throw new RuntimeException("A default style could not be found for " + histogram.title());
100         }
101         return style;
102     }
103 
104     public static int getRowFromHistoID(final int id) {
105         return YOFFSET - id % (2 * YOFFSET + 1);
106     }
107 
108     public static Boolean isInHole(final int row, final int column) {
109         if (row == 1 || row == -1) {
110             if ((column < XHOLESTART + XHOLEWIDTH) && (column >= XHOLESTART)) {
111                 return true;
112             }
113         }
114 
115         if (row == 0) {
116             return true;
117         }
118 
119         if (column == 0) {
120             return true;
121         }
122         return false;
123     }
124 
125     static IPlotter plot(final IPlotterFactory plotterFactory, final IBaseHistogram histogram, IPlotterStyle style,
126             final boolean show) {
127         if (style == null) {
128             style = getPlotterStyle(histogram);
129         }
130         final IPlotter plotter = plotterFactory.create(histogram.title());
131         plotter.createRegion();
132         plotter.region(0).plot(histogram, style);
133         if (show) {
134             plotter.show();
135         }
136         return plotter;
137     }
