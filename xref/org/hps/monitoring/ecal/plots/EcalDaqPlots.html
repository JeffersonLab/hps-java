21  /**
22   * The driver <code>EcalDaqPlots</code> implements the histogram shown to the
23   * user in the fourth tab of the
24   * Monitoring Application, when using the Ecal monitoring lcsim file. It
25   * contains only a sub-tab, showing
26   * the number of hits recorded by the different FADC channels. It is a very
27   * preliminary driver to monitor
28   * the DAQ status. These plots are updated continuously.
29   */
30  public class EcalDaqPlots extends Driver {
31  
32      private String subdetectorName = "Ecal";
33      private String inputCollection = "EcalCalHits";
34      private IPlotter plotter;
35      private AIDA aida;
36      private static ITree tree = AIDA.defaultInstance().tree();
37  
38      private List<IHistogram1D> plots;
39  
40      private List<Integer> slotsT, slotsB, crates;
41  
42      private EcalChannel.EcalChannelCollection channels;
43      private DatabaseConditionsManager manager;
44  
45      // private ConditionsManager manager;
46      public EcalDaqPlots() {
47          manager = DatabaseConditionsManager.getInstance();
48          // manager = ConditionsManager.defaultInstance();
49      }
50  
51      public void setSubdetectorName(String subdetectorName) {
52          this.subdetectorName = subdetectorName;
53      }
54  
55      public void setInputCollection(String inputCollection) {
56          this.inputCollection = inputCollection;
57      }
58  
59      public void detectorChanged(Detector detector) {
60  
61          if (subdetectorName == null) {
62              throw new RuntimeException("The subdetectorName parameter was not set.");
63          }
64  
65          if (inputCollection == null) {
66              throw new RuntimeException("The inputCollection parameter was not set.");
67          }
68  
69          // Get the channel information from the database.
70          channels = manager.getCachedConditions(EcalChannel.EcalChannelCollection.class, "ecal_channels")
71                  .getCachedData();
72  
73          /*
74           * I do not want the ECAL Crates and Slots to be hard-coded. It is fine to assume that the FADC channels are
75           * from 0 to 15:
76           * This is determined by JLAB FADC architecture.
77           * It is also fine to say that there are 14 slots occupied by FADCs in each crate: 14*16=224, number of channel
78           * in each Ecal sector
79           * (apart from the hole).
80           */
81          slotsT = new ArrayList<Integer>();
82          slotsB = new ArrayList<Integer>();
83          crates = new ArrayList<Integer>();
84  
85          // Loop over channels and get the list of slots-crates
86          for (EcalChannel channel : channels) {
87  
88              // y>0 means TOP, y<0 means BOTTOM
89              int y = channel.getY();
90              int slot = channel.getSlot();
91              int crate = channel.getCrate();
92  
93              if (y > 0) {
94                  if (!slotsT.contains(slot)) {
95                      slotsT.add(slot);
96                  }
97              } else if (y < 0) {
98                  if (!slotsB.contains(slot)) {
99                      slotsB.add(slot);
100                 }
101             }
102             if (!crates.contains(crate)) {
103                 crates.add(crate);
104             }
105         }
106         /* Order the slots in increasing order */
107         Collections.sort(slotsB);
108         Collections.sort(slotsT);
109 
110         /*
111          * System.out.println("These DAQ slots found:"); System.out.println("TOP: ");
112          * for (int slot : slotsT){ System.out.print(slot+" "); }
113          * System.out.println(""); System.out.println("BOTTOM: "); for
114          * (int slot : slotsB){ System.out.print(slot+" "); } System.out.println("");
115          */
116         aida = AIDA.defaultInstance();
117         tree.cd("/");
118         boolean dirExists = false;
119         String dirName = "/ECalDaq";
120         for (String st : tree.listObjectNames()) {
121             System.out.println(st);
122             if (st.contains(dirName)) {
123                 dirExists = true;
124             }
125         }
126         tree.setOverwrite(true);
127         if (!dirExists) {
128             tree.mkdir(dirName);
129         }
130         tree.cd(dirName);
131 
132         plots = new ArrayList<IHistogram1D>();
133 
134         for (int j = 0; j < 14; j++) { // TOP slot
135             plots.add(aida.histogram1D("ECAL: Top Crate Slot " + slotsT.get(j), 16, 0, 16));
136         }
137 
138         for (int j = 0; j < 14; j++) { // BOTTOM slot
139             plots.add(aida.histogram1D("ECAL: Bottom Crate Slot " + slotsB.get(j), 16, 0, 16));
140         }
141 
142         IPlotterFactory factory = aida.analysisFactory().createPlotterFactory("ECAL DAQ Plots");
143         plotter = factory.create("Crates");
144         IPlotterStyle pstyle = plotter.style();
145         pstyle.dataStyle().fillStyle().setColor("orange");
146         // pstyle.dataStyle().markerStyle().setColor("orange");
147         pstyle.dataStyle().errorBarStyle().setVisible(false);
148         pstyle.legendBoxStyle().setVisible(false);
149         plotter.createRegions(7, 4);
150 
151         int id, plot_id;
152         for (int i = 0; i < 2; i++) { // crate
153             for (int j = 0; j < 14; j++) { // slot
154                 id = i * 14 + j;
155                 plot_id = 0;
156                 if (i == 0) { // first-crate
157                     if (j % 2 == 0) {
158                         plot_id = j * 2;
159                     } else {
160                         plot_id = (j - 1) * 2 + 1;
161                     }
162                 } else if (i == 1) { // second-crate
163                     if (j % 2 == 0) {
164                         plot_id = j * 2 + 2;
165                     } else {
166                         plot_id = (j - 1) * 2 + 3;
167                     }
168                 }
169                 // System.out.println("Plot in region " + plot_id + " the plot " + plots.get(id).title() + "(index: " +
170                 // id + ")");
171                 plotter.region(plot_id).plot(plots.get(id));
172             }
173         }
174         plotter.show();
175     }
176 
177     public void process(EventHeader event) {
178         if (event.hasCollection(CalorimeterHit.class, inputCollection)) {
179             List<CalorimeterHit> hits = event.get(CalorimeterHit.class, inputCollection);
180             for (CalorimeterHit hit : hits) {
181 
182                 // Make an ID to find.
183                 // EcalChannel daq = new EcalChannel.DaqId();
184                 // daq.crate = 1;
185                 // daq.slot = 2;
186                 // daq.channel = 3;
187                 // Find the matching channel.
188                 EcalChannel channel = channels.findGeometric(hit.getCellID());
189                 int row = hit.getIdentifierFieldValue("iy");
190                 int column = hit.getIdentifierFieldValue("ix");
191 
192                 int crateN = channel.getCrate();
193                 int slotN = channel.getSlot();
194                 int channelN = channel.getChannel();
195 
196                 // System.out.println("found channel at " + column + " " + row +
197                 // " corresponding to DAQ crate/slot/channel " + crateN + " "+slotN+" "+channelN);
198                 // Top CRATE
199                 if (row > 0) {
200                     int index = slotsT.indexOf(slotN);
201                     plots.get(index).fill(channelN);
202                 } else if (row < 0) {
203                     int index = slotsB.indexOf(slotN);
204                     plots.get(index + 14).fill(channelN);
205                 }
206             }
207         }
208     }
