17  /**
18   * The driver <code>EcalMonitoringPlots</code> implements the histogram shown to
19   * the user in the
20   * first tab of the Monitoring Application, when using the Ecal monitoring lcsim
21   * file. It contains
22   * only a sub-tab, with 3 histograms. - Hit counts by channel (Histogram2D),
23   * Occupancy by channel
24   * (Histogram2D), Cluster counts by channel (Histogram2D) Each cluster is
25   * associated with the seed
26   * crystal.
27   *
28   * These plots are updated regularly, according to the event refresh rate.
29   */

75      protected void detectorChanged(Detector detector) {
76  
77          tree = AIDA.defaultInstance().tree();
78          tree.cd("/");
79          boolean dirExists = false;
80          String dirName = "/EcalMon";
81          for (String st : tree.listObjectNames()) {
82              System.out.println(st);
83              if (st.contains(dirName)) {
84                  dirExists = true;
85              }
86          }
87          tree.setOverwrite(true);
88          if (!dirExists) {
89              tree.mkdir(dirName);
90          }
91          tree.cd(dirName);
92  
93          // System.out.println("EcalMonitoringPlots:: detector changed was called");
94          // Setup the plotter.
95          plotter = aida.analysisFactory().createPlotterFactory("Ecal Monitoring Plots")
96                  .create("HPS ECal Monitoring Plots");
97          // Setup plots.       
98  
99          String hitCountDrawPlotTitle;
100         hitCountDrawPlotTitle = detector.getDetectorName() + " : " + inputCollection + " : Hit Rate KHz";
101 
102         hitCountDrawPlot = aida.histogram2D(hitCountDrawPlotTitle, 47, -23.5, 23.5, 11, -5.5, 5.5);
103         hitCountFillPlot = makeCopy(hitCountDrawPlot);
104         occupancyDrawPlot = aida.histogram2D(detector.getDetectorName() + " : " + inputCollection + " : Occupancy", 47,
105                 -23.5, 23.5, 11, -5.5, 5.5);
106         clusterCountDrawPlot = aida.histogram2D(detector.getDetectorName() + " : " + clusterCollection
107                 + " : Cluster Rate KHz", 47, -23.5, 23.5, 11, -5.5, 5.5);
108         clusterCountFillPlot = makeCopy(clusterCountDrawPlot);
109 
110         NoccupancyFill = 1; // to avoid a "NaN" at beginning
111         for (int ii = 0; ii < (11 * 47); ii++) {
112             int row = EcalMonitoringUtilities.getRowFromHistoID(ii);
113             int column = EcalMonitoringUtilities.getColumnFromHistoID(ii);
114             occupancyFill[ii] = 0.;
115         }
116 
117         // Create the plotter regions.
118         plotter.createRegions(2, 2);
119         plotter.style().statisticsBoxStyle().setVisible(false);

123         style.dataStyle().fillStyle().setParameter("showZeroHeightBins", Boolean.FALSE.toString());
124         plotter.region(0).plot(hitCountDrawPlot);
125         style = plotter.region(1).style();
126         style.setParameter("hist2DStyle", "colorMap");
127         style.dataStyle().fillStyle().setParameter("colorMapScheme", "rainbow");
128         style.dataStyle().fillStyle().setParameter("showZeroHeightBins", Boolean.FALSE.toString());
129         plotter.region(1).plot(clusterCountDrawPlot);
130         style = plotter.region(2).style();
131         style.setParameter("hist2DStyle", "colorMap");
132         style.dataStyle().fillStyle().setParameter("colorMapScheme", "rainbow");
133         style.dataStyle().fillStyle().setParameter("showZeroHeightBins", Boolean.FALSE.toString());
134         // style.zAxisStyle().setParameter("scale", "log");
135         plotter.region(2).plot(occupancyDrawPlot);
136 
137         plotter.show();
138 
139         prevTime = 0; // init the time
140         thisTime = 0; // init the time
141 
142         thisEventN = 0;
143         prevEventN = 0;
144 
145         thisEventTime = 0;
146         prevEventTime = 0;
147     }

149     public void process(EventHeader event) {
150         int nhits = 0;
151         int chits[] = new int[11 * 47];
152 
153         if (event.hasCollection(CalorimeterHit.class, inputCollection)) {
154             List<CalorimeterHit> hits = event.get(CalorimeterHit.class, inputCollection);
155             for (CalorimeterHit hit : hits) {
156                 int column = hit.getIdentifierFieldValue("ix");
157                 int row = hit.getIdentifierFieldValue("iy");
158                 int id = EcalMonitoringUtilities.getHistoIDFromRowColumn(row, column);
159                 hitCountFillPlot.fill(column, row);
160                 {
161                     chits[id]++;
162                     nhits++;
163                 }
164             }
165         }
166 
167         if (nhits > 0) {
168             for (int ii = 0; ii < (11 * 47); ii++) {
169                 occupancyFill[ii] += (1. * chits[ii]) / nhits;
170             }
171         }
172 
173         if (event.hasCollection(Cluster.class, clusterCollection)) {
174             List<Cluster> clusters = event.get(Cluster.class, clusterCollection);
175             for (Cluster cluster : clusters) {
176                 clusterCountFillPlot.fill(cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("ix"), cluster
177                         .getCalorimeterHits().get(0).getIdentifierFieldValue("iy"));
178             }
179         }
180 
181         thisTime = System.currentTimeMillis() / 1000;
182         thisEventN = event.getEventNumber();
183         thisEventTime = event.getTimeStamp() / 1E9;
184         if ((thisTime - prevTime) > eventRefreshRate) {
185             double scale = 1.;
186 
187             if (NoccupancyFill > 0) {
188                 scale = (thisEventN - prevEventN) / NoccupancyFill;
189                 scale = scale / (thisEventTime - prevEventTime);
190                 scale /= 1000.; // do KHz
191             }
192             // System.out.println("Event: "+thisEventN+" "+prevEventN);
193             // System.out.println("Time: "+thisEventTime+" "+prevEventTime);
194             // System.out.println("Monitor: "+thisTime+" "+prevTime+" "+NoccupancyFill);
195             if (scale > 0) {
196                 hitCountFillPlot.scale(scale);
197                 clusterCountFillPlot.scale(scale);
198                 redraw();
199             }
200             prevTime = thisTime;
201             prevEventN = thisEventN;
202             prevEventTime = thisEventTime;
203             NoccupancyFill = 0;
204         } else {
205             NoccupancyFill++;
206         }
207     }

212     void redraw() {
213         hitCountDrawPlot.reset();
214         hitCountDrawPlot.add(hitCountFillPlot);
215         plotter.region(0).clear();
216         plotter.region(0).plot(hitCountDrawPlot);
217         // plotter.region(0).refresh();
218         hitCountFillPlot.reset();
219 
220         clusterCountDrawPlot.reset();
221         clusterCountDrawPlot.add(clusterCountFillPlot);
222         plotter.region(1).clear();
223         plotter.region(1).plot(clusterCountDrawPlot);
224         // plotter.region(1).refresh();
225         clusterCountFillPlot.reset();
226 
227         occupancyDrawPlot.reset();
228         for (int id = 0; id < (47 * 11); id++) {
229             int row = EcalMonitoringUtilities.getRowFromHistoID(id);
230             int column = EcalMonitoringUtilities.getColumnFromHistoID(id);
231             double mean = occupancyFill[id] / NoccupancyFill;
232 
233             occupancyFill[id] = 0;
234             if ((row != 0) && (column != 0) && (!EcalMonitoringUtilities.isInHole(row, column))) {
235                 occupancyDrawPlot.fill(column, row, mean);
236             }
237         }
238         plotter.region(2).clear();
239         if (occupancyDrawPlot.sumAllBinHeights() > 0) {
240             plotter.region(2).plot(occupancyDrawPlot);
241         }
242         // plotter.region(2).refresh();
243     }

245     private IHistogram2D makeCopy(IHistogram2D hist) {
246         return aida.histogram2D(hist.title() + "_copy", hist.xAxis().bins(), hist.xAxis().lowerEdge(), hist.xAxis()
247                 .upperEdge(), hist.yAxis().bins(), hist.yAxis().lowerEdge(), hist.yAxis().upperEdge());
248     }
