314         runNumber = event.getRunNumber();
315         eventN++;
316         if (event.hasCollection(CalorimeterHit.class, inputCollection)) {
317             // List<BaseRawCalorimeterHit> hits = event.get(BaseRawCalorimeterHit.class, inputCollectionRaw);
318             List<CalorimeterHit> hits = event.get(CalorimeterHit.class, inputCollection);
319             for (CalorimeterHit hit : hits) {
320 
321                 column = hit.getIdentifierFieldValue("ix");
322                 row = hit.getIdentifierFieldValue("iy");
323                 id = EcalMonitoringUtilities.getHistoIDFromRowColumn(row, column);
324                 cellID = hit.getCellID();
325                 chid = ChannelCollection.findGeometric(cellID).getChannelId();
326 
327                 energy = hit.getCorrectedEnergy();
328 
329                 if (useRawEnergy) {
330                     fillEnergy = getRawADCSum(energy, cellID);
331                 } else {
332                     fillEnergy = energy;
333                 }
334                 fillTime = hit.getTime();
335 
336                 // find the LED
337                 if (row > 0) {
338                     ledId = LedTopMap.get(chid);
339                 } else if (row < 0) {
340                     ledId = LedBotMap.get(chid);
341                 }
342                 driverId = getDriver(ledId);
343                 if (row < 0)
344                     driverId += 4;
345 
346                 /* Skip the events under thr */
347                 if (energy < energyCut)
348                     continue;
349 
350                 /* First, check if this led is the one in the NEXT step. Therefore, increment by 1 the step */
351                 /*
352                  * if (iStep[driverId]==0){
353                  * 
354                  * actualSequence[driverId][iStep[driverId]]=ledId;
355                  * iStep[driverId]=1;
356                  * }
357                  * else if ((iStep[driverId]==1)&&(ledId!=actualSequence[driverId][0])){

359                  * " . Led ID: "+ledId+" Column: "+column+" Row: "+row);
360                  * if (iStep[driverId]>0) drawProfiles(actualSequence[driverId][iStep[driverId]-1],driverId);
361                  * actualSequence[driverId][iStep[driverId]]=ledId;
362                  * iStep[driverId]++;
363                  * }
364                  * else if
365                  * ((iStep[driverId]>1)&&(ledId!=actualSequence[driverId][iStep[driverId]-1])&&(ledId!=actualSequence
366                  * [driverId][iStep[driverId]-2])){
367                  * System.out.println("LedAnalysis:: increment step ("+iStep[driverId]+") for driver "+driverId+
368                  * " . Led ID: "+ledId+" Column: "+column+" Row: "+row);
369                  * if (iStep[driverId]>0) drawProfiles(actualSequence[driverId][iStep[driverId]-1],driverId);
370                  * actualSequence[driverId][iStep[driverId]]=ledId;
371                  * iStep[driverId]++;
372                  * }
373                  * 
374                  * 
375                  * // if (iStep[driverId]==-1) continue;
376                  */
377 
378                 if (iStep[driverId] == -1)
379                     continue; /* Not yet data */
380 
381                 /* Put this code here, since we want to always fill the ntuple */

400     /*
401      * The endOfData() method analises each ntuple to find the LED response.
402      * We cannot simply fit a gaussian to the energy distribution, since there is a high-energy tail due to the LED
403      * being turned on:
404      * When the LED turns on, it is "cold", and emits more light. Immediately, it starts to heat, and due to temperature
405      * effects the
406      * emitted light is less. This is clearly visible if one plots the charge VS the event number: the trend is
407      * decreasing, toward a
408      * plateau, that corresponds to the value at thermal equilibrium.
409      * 
410      * For (few) channels, the first charge values are close to 0, then charge grows rapidly, then it returns back to
411      * the plateau.
412      * To handle these, I always cut the first 10% events
413      * To properly handle this:
414      * 
415      * 1) First create a profile histogram, charge VS event number.
416      * 2) Fit it with something like "A*exp(-event_number/N0)+C. The function does not need to be extra-accurate at this
417      * stage
418      * 3) Cut the events with event_number < 5*N0.
419      * 4) Fit the remaining events with a gaussian.
420      */
421     @Override
422     public void endOfData() {
423         System.out.println("LedAnalysis::end of data");
424         System.out.println("LedAnalysis:: nEventsMin is: " + nEventsMin);
425 
426         double e, eMin, eMax;
427         double t;
428         int n, nBins, nSkip;
429 
430         int row, column;
431 
432         String hName;
433         IFunctionFactory fFactory = aida.analysisFactory().createFunctionFactory(aida.tree());
434 
435         IFitResult fResult;
436         IFitter fFitter;
437 
438         for (int id = 0; id < 11 * 47; id++) {
439 
440             eMin = 9999;
441             eMax = -9999;
442             row = EcalMonitoringUtilities.getRowFromHistoID(id);
443             column = EcalMonitoringUtilities.getColumnFromHistoID(id);
444             System.out.println("");
445             System.out.println("Doing channel: X= " + column + " Y= " + row + "id= " + id);
446             System.out.println("Number of entries in analysis ntuple: " + iTuple.get(id).rows());
447             System.out.println("Number of recognized events: " + nEvents[id]);
448             /* Create the profile. Create it for all the channels, to keep sync. */
449             nBins = nEvents[id] / 100;
450             if (nBins <= 0)
451                 nBins = 1;
452 
453             /* Clear previous */
454 
455             if (id > 0) {
456                 aida.tree().rm("strip");
457                 aida.tree().rm("fun0");
458                 aida.tree().rm("fun1");
459             }
460             /* Create the profile. */
461             cProfile = aida.profile1D("strip", nBins, -0.5, nEvents[id] * (1 - skipInitial) + 0.5);
462             cProfile.reset();
463             /* Create the function for the profile fit and the gaus fit */
464             fFunction = fFactory.createFunctionFromScript("fun0", 1, "A*exp(-x[0]/tau)+B", "A,tau,B", "", null);
465             fFunction1 = fFactory.createFunctionByName("fun1", "G");
466 
467             if (EcalMonitoringUtilities.isInHole(row, column) == true) {
468                 System.out.println("Channel X= " + column + " Y= " + row + " is in hole. Skip");
469                 hCharge.add(aida.histogram1D("charge_" + id, 200, 0., 1.)); // create here the histogram to keep sync
470                 System.out.println("In hole, skip");
471                 continue;
472             } else if (nEvents[id] < nEventsMin) {
473                 hCharge.add(aida.histogram1D("charge_" + id, 200, 0., 1.)); // create here the histogram to keep sync
474                 System.err.println("LedAnalysis:: the channel X= " + column + " Y= " + row + " has not enough events "
475                         + nEvents[id] + " " + nEventsMin);
476 
477                 continue;
478             }
479 
480             // Fill the profile*/
481             nSkip = (int) (nEvents[id] * skipInitial);
482             if (nSkip > iTuple.get(id).rows()) {
483                 System.out.println("Can't skip initial events?");
484                 nSkip = 0;
485             }
486             iTuple.get(id).start();
487             iTuple.get(id).skip(nSkip); // This is the work-around for those channels with charge starting from 0 and
488                                         // rapidly growing//
489             n = 0;
490             iTuple.get(id).next();
491             while (iTuple.get(id).next()) {
492                 e = iTuple.get(id).getDouble(1);
493                 if (e < eMin)
494                     eMin = e;
495                 if (e > eMax)
496                     eMax = e;
497                 cProfile.fill(1. * n, e);
498                 n++;
499             }
500             fFitter = aida.analysisFactory().createFitFactory().createFitter("chi2", "", "v");
501 
502             if (doFullAnalysis) {
503                 // Init function parameters
504                 double[] initialPars = {eMax - eMin, nEvents[id] / 10., eMin};
505                 if (initialPars[0] < 0)
506                     initialPars[0] = 0;
507                 fFunction.setParameters(initialPars);
508 
509                 // Do the fit
510                 System.out.println("LedAnalysis:: do profile fit " + id + " " + fFitter.engineName() + " "
511                         + fFitter.fitMethodName());
512                 System.out.println("LedAnalysis:: initial parameters " + initialPars[0] + " " + initialPars[1] + " "
513                         + initialPars[2]);
514                 fResult = fFitter.fit(cProfile, fFunction);
515                 fPars = fResult.fittedParameters();
516                 fParErrs = fResult.errors();
517                 fParNames = fResult.fittedParameterNames();
518                 System.out.println("LedAnalysis:: Status= " + fResult.fitStatus() + " " + fResult.isValid()
519                         + " Chi2 = " + fResult.quality() + " NDF: " + fResult.ndf());
520                 for (int i = 0; i < fResult.fittedFunction().numberOfParameters(); i++) {
521                     System.out.println(fParNames[i] + " : " + fPars[i] + " +- " + fParErrs[i]);
522                 }
523                 fFunction.setParameters(fPars);
524 
525                 // if fit failed, revert to simpler case
526                 if ((fResult.isValid() == false) || Double.isNaN(fParErrs[0]) || Double.isNaN(fParErrs[1])
527                         || Double.isNaN(fParErrs[2])) {
528                     System.out.println("LedAnalysis:: fit failed. Reverting to simpler case");
529                     nSkip = (int) (nEvents[id] * (skipMin + skipInitial));
530                 } else {
531                     // Now we have the tau parameter. Take ONLY the events that are with N>5*tau/
532                     // As a cross-check, also verify that tau > Nevents/10, otherwise skip the first Nevents/2
533                     // and emit warning
534                     nSkip = (int) (fPars[1] * 5);
535                     if (nSkip < (nEvents[id] * skipMin)) {
536                         System.out.println("LedAnalysis:: Skip number too low: " + nSkip + " Increment it to "
537                                 + nEvents[id] / 2);
538                         nSkip = (int) (nEvents[id] * (skipMin + skipInitial));
539                     }
540                     if (nSkip > nEvents[id]) {
541                         System.out.println("LedAnalysis:: Skip number too high, reduce it");
542                         nSkip = (int) (nEvents[id] * (skipMin + skipInitial));
543                     }
544                 }
545             } else {
546                 nSkip = (int) (nEvents[id] * (skipMin + skipInitial));
547             }
548 
549             System.out.println("LedAnalysis:: gaus fit :: Going to skip " + nSkip + " out of " + nEvents[id]);
550             System.out.println("eMin is: " + eMin + " eMax is: " + eMax);
551             hCharge.add(aida.histogram1D("charge_" + id, 200, eMin * 0.9, eMax * 1.1));
552 
553             iTuple.get(id).start();
554             iTuple.get(id).skip(nSkip);
555             n = 0;
556             while (iTuple.get(id).next()) {
557                 e = iTuple.get(id).getDouble(1);
558                 t = iTuple.get(id).getDouble(2);
559                 hCharge.get(id).fill(e);
560                 n++;
561             }
562 
563             /* Finally do the fit with the gaussian */
564             double[] initialPars1 = {hCharge.get(id).maxBinHeight(), hCharge.get(id).mean(), hCharge.get(id).rms()};
565 
566             System.out.println("LedAnalysis:: Gaus fit");
567             System.out.println("LedAnalysis:: initial parameters " + initialPars1[0] + " " + initialPars1[1] + " "
568                     + initialPars1[2]);
569 
570             fFunction1.setParameters(initialPars1);
571             fResult = fFitter.fit(hCharge.get(id), fFunction1);
572             fPars = fResult.fittedParameters();
573             fParErrs = fResult.errors();
574             fParNames = fResult.fittedParameterNames();
575             System.out.println("Status= " + fResult.fitStatus() + " " + fResult.isValid() + " Chi2 = "
576                     + fResult.quality() + " NDF: " + fResult.ndf());
577             for (int i = 0; i < fResult.fittedFunction().numberOfParameters(); i++) {
578                 System.out.println(fParNames[i] + " : " + fPars[i] + " +- " + fParErrs[i]);
579             }
580             fFunction1.setParameters(fPars);
581             mMean[id] = fPars[1];
582             mRMS[id] = fPars[2];
583

716     /**
717      * This function returns the driver number (from 0 to 3) given the LED id.
718      * 
719      * @param led
720      * @return the driver number from the LED id
721      */
722     public int getDriver(int led) {
723         int ret = -1;
724         if ((led >= 2) && (led < 56))
725             ret = 0;
726         else if ((led >= 56) && (led < 112))
727             ret = 1;
728         else if ((led >= 112) && (led < 168))
729             ret = 2;
730         else if ((led >= 168) && (led < 224))
731             ret = 3;
732         return ret;
733     }

735     /**
736      * Very simple method to retrieve the pedestal-subtracted raw Energy.
737      * If the gain changes (because we do a re-calibration), I do not want to include this in the LED analysis
738      * 
739      * @param energy
740      * @param cellID
741      * @return the pedestal-subtracted raw energy
742      */
743     public double getRawADCSum(double energy, long cellID) {
744         EcalChannelConstants channelData = ecalConditions.getChannelConstants(ecalConditions.getChannelCollection()
745                 .findGeometric(cellID));
746         double RawSum = energy / EcalUtils.GeV;
747         double gain = channelData.getGain().getGain();
748         double ret = RawSum / gain;
749         // System.out.println("A:C "+RawSum+" "+ret);
750         return ret;
751     }
