25  /**
26   * The driver <code>EcalCluster</code> implements the histograms shown
27   * to the user in the third tab of the Monitoring Application, when using
28   * the Ecal monitoring application.<br/>
29   * <br/>
30   * These histograms show the single-channel distributions:
31   * <ul>
32   * <li>The first sub-tab shows the cluster counts per event (Histogram1D), the
33   * number of hits in a cluster
34   * (Histogram1D), the cluster centers distribution (Histogram2D), and the
35   * maximum cluster energy in an event
36   * (Histogram1D).</li>
37   * <li>The second sub-tab shows the energy distribution of the cluster
38   * (Histogram1D) and the maximum cluster energy in
39   * each event (Histogram1D)</li>
40   * <li>The third sub-tab shows the time distribution of the cluster
41   * (Histogram1D), taken from the mean of the times
42   * forming the cluster, as well as the RMS (Histogram1D).</li>
43   * <li>The fourth tab displays the cluster pair distribution for the energy sum,
44   * energy difference, energy slope, and
45   * coplanarity cuts for all top/bottom pairs received. It also displays the
46   * average x- and y-coordinates for the pairs.</li>
47   * </ul>
48   * All histograms are updated continuously.<br/>
49   * <br/>
50   * The cluster center is calculated from the average of the positions
51   * of the constituent hits, weighted by the hit energies.<br/>
52   * <br/>
53   */
54  public class EcalClusterPlots extends Driver {

96       * Resets all of the plots for the new detector.
97       *
98       * @param detector - The <code>Detector</code> object representing
99       * the new detector.
100      */
101     @Override
102     protected void detectorChanged(Detector detector) {
103         // Re-instantiate all of the histograms.      
104 
105         tree = AIDA.defaultInstance().tree();
106         tree.cd("/");
107         boolean dirExists = false;
108         String dirName = "/EcalClusters";
109         for (String st : tree.listObjectNames()) {
110             System.out.println(st);
111             if (st.contains(dirName)) {
112                 dirExists = true;
113             }
114         }
115         tree.setOverwrite(true);
116         if (!dirExists) {
117             tree.mkdir(dirName);
118         }
119         tree.cd(dirName);
120 
121         clusterCountPlot = aida.histogram1D(detector.getDetectorName() + " : " + clusterCollectionName
122                 + " : Cluster Count per Event", 10, -0.5, 9.5);
123         clusterSizePlot = aida.histogram1D(detector.getDetectorName() + " : " + clusterCollectionName
124                 + " : Cluster Size", 15, -0.5, 15.5);
125         clusterEnergyPlot = aida.histogram1D(detector.getDetectorName() + " : " + clusterCollectionName
126                 + " : Cluster Energy", 100, -0.1, maxE);
127         clusterMaxEnergyPlot = aida.histogram1D(detector.getDetectorName() + " : " + clusterCollectionName
128                 + " : Maximum Cluster Energy in Event", 100, -0.1, maxE);
129         edgePlot = aida.histogram2D(detector.getDetectorName() + " : " + clusterCollectionName + " : Weighted Cluster",
130                 47, -23.25, 23.25, 11, -5.25, 5.25);
131         clusterTimes = aida.histogram1D(detector.getDetectorName() + " : " + clusterCollectionName
132                 + " : Cluster Time Mean", 400, 0, 4.0 * 100);
133         clusterTimeSigma = aida.histogram1D(detector.getDetectorName() + " : " + clusterCollectionName
134                 + " : Cluster Time Sigma", 100, 0, 40);
135 
136         pairEnergySum = aida.histogram1D("Pair Energy Sum Distribution", 176, 0.0, maxE);
137         pairEnergyDifference = aida.histogram1D("Pair Energy Difference Distribution", 176, 0.0, 2.2);
138         pairCoplanarity = aida.histogram1D("Pair Coplanarity Distribution", 90, 0.0, 180.0);
139         pairEnergySlope = aida.histogram1D("Pair Energy Slope Distribution", 150, 0.0, 3.0);
140         pairEnergyPositionMeanX = aida.histogram1D("Cluster Pair Weighted Energy Position (x-Index)", 100, -250, 250);
141         pairEnergyPositionMeanY = aida.histogram1D("Cluster Pair Weighted Energy Position (y-Index)", 100, -100, 100);
142         pairMassSameHalf = aida.histogram1D("Cluster Pair Mass: Same Half", 100, 0.0, 0.35);
143         pairMassOppositeHalf = aida.histogram1D("Cluster Pair Mass: Top Bottom", 100, 0.0, 0.35);
144         pairMassSameHalfFid = aida.histogram1D("Cluster Pair Mass: Same Half Fiducial", 100, 0.0, 0.35);
145         pairMassOppositeHalfFid = aida.histogram1D("Cluster Pair Mass: Top Bottom Fiducial", 100, 0.0, 0.35);
146         // Setup the plotter factory.
147         IPlotterFactory plotterFactory = aida.analysisFactory().createPlotterFactory("Ecal Cluster Plots");
148 
149         // Apply formatting that is constant across all tabs.
150         for (int tabIndex = 0; tabIndex < plotter.length; tabIndex++) {
151             plotter[tabIndex] = plotterFactory.create(TAB_NAMES[tabIndex]);
152             plotter[tabIndex].setTitle(TAB_NAMES[tabIndex]);
153             plotter[tabIndex].style().dataStyle().errorBarStyle().setVisible(false);
154             plotter[tabIndex].style().dataStyle().fillStyle()
155                     .setParameter("showZeroHeightBins", Boolean.FALSE.toString());
156             if (logScale) {
157                 plotter[tabIndex].style().yAxisStyle().setParameter("scale", "log");
158             }
159         }
160 
161         // Define the Cluster Counts tab.
162         plotter[TAB_CLUSTER_COUNT].createRegions(2, 2);
163         plotter[TAB_CLUSTER_COUNT].region(0).plot(clusterCountPlot);
164         plotter[TAB_CLUSTER_COUNT].region(1).plot(clusterSizePlot);
165         plotter[TAB_CLUSTER_COUNT].region(2).plot(edgePlot);
166         plotter[TAB_CLUSTER_COUNT].region(3).plot(clusterMaxEnergyPlot);
167 
168         // Define the Cluster Energy tab.
169         plotter[TAB_CLUSTER_ENERGY].createRegions(1, 2);
170         plotter[TAB_CLUSTER_ENERGY].region(0).plot(clusterEnergyPlot);
171         plotter[TAB_CLUSTER_ENERGY].region(1).plot(clusterMaxEnergyPlot);
172 
173         // Define the Cluster Time tab.
174         plotter[TAB_CLUSTER_TIME].createRegions(1, 2);
175         plotter[TAB_CLUSTER_TIME].region(0).plot(clusterTimes);
176         plotter[TAB_CLUSTER_TIME].region(1).plot(clusterTimeSigma);
177 
178         // Create the Cluster Pair tab.
179         plotter[TAB_CLUSTER_PAIR].createRegions(2, 3);
180         plotter[TAB_CLUSTER_PAIR].region(0).plot(pairEnergySum);
181         plotter[TAB_CLUSTER_PAIR].region(3).plot(pairEnergyDifference);
182         plotter[TAB_CLUSTER_PAIR].region(1).plot(pairEnergySlope);
183         plotter[TAB_CLUSTER_PAIR].region(4).plot(pairCoplanarity);
184         plotter[TAB_CLUSTER_PAIR].region(2).plot(pairEnergyPositionMeanX);
185         plotter[TAB_CLUSTER_PAIR].region(5).plot(pairEnergyPositionMeanY);
186         // Define the Cluster Counts tab.
187         plotter[TAB_CLUSTER_MASS].createRegions(2, 2);
188         plotter[TAB_CLUSTER_MASS].region(0).plot(pairMassSameHalf);
189         plotter[TAB_CLUSTER_MASS].region(1).plot(pairMassOppositeHalf);
190         plotter[TAB_CLUSTER_MASS].region(2).plot(pairMassSameHalfFid);
191         plotter[TAB_CLUSTER_MASS].region(3).plot(pairMassOppositeHalfFid);
192         // If they should not be hidden, display the tabs.
193         if (!hide) {
194             for (IPlotter tab : plotter) {
195                 tab.show();
196             }
197         }

200     /**
201      * Populates all histograms from event clusters, if they are present.
202      *
203      * @param event - The event from which to draw clusters.
204      */
205     @Override
206     public void process(EventHeader event) {
207         // Check whether the event has clusters or not.
208         if (event.hasCollection(Cluster.class, clusterCollectionName)) {
209             // Get the list of clusters.
210             List<Cluster> clusterList = event.get(Cluster.class, clusterCollectionName);
211 
212             // Create lists to store the clusters from the top of the
213             // calorimeter and the bottom.
214             List<Cluster> topList = new ArrayList<Cluster>();
215             List<Cluster> bottomList = new ArrayList<Cluster>();
216 
217             // Track the highest energy cluster in the event.
218             double maxEnergy = 0.0;
219 
220             // Process each of the clusters.
221             for (Cluster cluster : clusterList) {
222                 // If this cluster has a higher energy then was seen
223                 // previously, it is now the highest energy cluster.
224                 if (cluster.getEnergy() > maxEnergy) {
225                     maxEnergy = cluster.getEnergy();
226                 }
227 
228                 // Get the list of calorimeter hits and its size.
229                 List<CalorimeterHit> hitList = cluster.getCalorimeterHits();
230                 int hitCount = hitList.size();
231 
232                 // Track cluster statistics.
233                 double xEnergyWeight = 0.0;
234                 double yEnergyWeight = 0.0;
235                 double[] hitTimes = new double[hitCount];
236                 double totalHitEnergy = 0.0;
237 
238                 // Iterate over the hits and extract statistics from them.
239                 for (int hitIndex = 0; hitIndex < hitCount; hitIndex++) {
240                     hitTimes[hitIndex] = hitList.get(hitIndex).getTime();
241                     totalHitEnergy += hitList.get(hitIndex).getRawEnergy();
242                     xEnergyWeight += (hitList.get(hitIndex).getRawEnergy() * hitList.get(hitIndex)
243                             .getIdentifierFieldValue("ix"));
244                     yEnergyWeight += (hitList.get(hitIndex).getRawEnergy() * hitList.get(hitIndex)
245                             .getIdentifierFieldValue("iy"));
246                 }
247 
248                 // If the cluster energy exceeds zero, plot the cluster
249                 // statistics.
250                 if (cluster.getEnergy() > 0) {
251                     clusterSizePlot.fill(hitCount);
252                     clusterTimes.fill(StatUtils.mean(hitTimes, 0, hitCount));
253                     clusterTimeSigma.fill(Math.sqrt(StatUtils.variance(hitTimes, 0, hitCount)));
254                     edgePlot.fill(xEnergyWeight / totalHitEnergy, yEnergyWeight / totalHitEnergy);
255                 }
256 
257                 // Fill the single cluster plots.
258                 clusterEnergyPlot.fill(cluster.getEnergy());
259 
260                 // Cluster pairs are formed from all top/bottom cluster
261                 // combinations. To create these pairs, separate the
262                 // clusters into two lists based on their y-indices.
263                 if (cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("ix") > 0) {
264                     topList.add(cluster);
265                 } else {
266                     bottomList.add(cluster);
267                 }
268             }
269 
270             // Populate the event plots.
271             clusterCountPlot.fill(clusterList.size());
272             if (maxEnergy > 0) {
273                 clusterMaxEnergyPlot.fill(maxEnergy);
274             }
275 
276             // Create a list to store cluster pairs.
277             List<Cluster[]> pairList = new ArrayList<Cluster[]>(topList.size() * bottomList.size());
278 
279             // Form pairs from all possible combinations of clusters
280             // from the top and bottom lists.
281             for (Cluster topCluster : topList) {
282                 for (Cluster bottomCluster : bottomList) {
283                     // Make a cluster pair array.
284                     Cluster[] pair = new Cluster[2];
285 
286                     // The lower energy cluster goes in the second slot.
287                     if (topCluster.getEnergy() > bottomCluster.getEnergy()) {
288                         pair[0] = topCluster;
289                         pair[1] = bottomCluster;
290                     } else {
291                         pair[0] = bottomCluster;
292                         pair[1] = topCluster;
293                     }
294 
295                     // Add the pair to the pair list.
296                     pairList.add(pair);
297                 }
298             }
299 
300             // Iterate over each pair and calculate the pair cut values.
301             for (Cluster[] pair : pairList) {
302                 // Get the energy slope value.
303                 double energySumValue = TriggerModule.getValueEnergySum(pair);
304                 double energyDifferenceValue = TriggerModule.getValueEnergyDifference(pair);
305                 double energySlopeValue = TriggerModule.getValueEnergySlope(pair, 0.005500);
306                 double coplanarityValue = TriggerModule.getValueCoplanarity(pair);
307                 double xMean = ((pair[0].getEnergy() * pair[0].getPosition()[0]) + (pair[1].getEnergy() * pair[1]
308                         .getPosition()[0])) / energySumValue;
309                 double yMean = ((pair[0].getEnergy() * pair[0].getPosition()[1]) + (pair[1].getEnergy() * pair[1]
310                         .getPosition()[1])) / energySumValue;
311 
312                 // Populate the cluster pair plots.
313                 pairEnergySum.fill(energySumValue, 1);
314                 ;
315                 pairEnergyDifference.fill(energyDifferenceValue, 1);
316                 pairEnergySlope.fill(energySlopeValue, 1);
317                 pairCoplanarity.fill(coplanarityValue, 1);
318                 pairEnergyPositionMeanX.fill(xMean);
319                 pairEnergyPositionMeanY.fill(yMean);
320                 pairMassOppositeHalf.fill(getClusterPairMass(pair));
321 
322                 int idx0 = pair[0].getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
323                 int idx1 = pair[1].getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
324                 if (Math.abs(idx0) > 2 && Math.abs(idx1) > 2) {
325                     pairMassOppositeHalfFid.fill(getClusterPairMass(pair));
326                 }
327 
328             }
329             // Create a list to store cluster pairs.
330             List<Cluster[]> pairListSameHalf = new ArrayList<Cluster[]>();
331 
332             // Form pairs from all possible combinations of clusters
333             // from the top and bottom lists.
334             for (Cluster topCluster : topList) {
335                 for (Cluster topCluster2 : topList) {
336                     // Make a cluster pair array.
337                     Cluster[] pair = new Cluster[2];
338 
339                     // The lower energy cluster goes in the second slot.
340                     if (topCluster.getEnergy() > topCluster2.getEnergy()) {
341                         pair[0] = topCluster;
342                         pair[1] = topCluster2;
343                     } else {
344                         pair[0] = topCluster2;
345                         pair[1] = topCluster;
346                     }
347 
348                     // Add the pair to the pair list.
349                     pairListSameHalf.add(pair);
350                 }
351             }
352 
353             for (Cluster cluster : bottomList) {
354                 for (Cluster cluster2 : bottomList) {
355                     // Make a cluster pair array.
356                     Cluster[] pair = new Cluster[2];
357 
358                     // The lower energy cluster goes in the second slot.
359                     if (cluster.getEnergy() > cluster2.getEnergy()) {
360                         pair[0] = cluster;
361                         pair[1] = cluster2;
362                     } else {
363                         pair[0] = cluster2;
364                         pair[1] = cluster;
365                     }
366 
367                     // Add the pair to the pair list.
368                     pairListSameHalf.add(pair);
369                 }
370             }
371             // Iterate over each pair and calculate the pair cut values.
372             for (Cluster[] pair : pairListSameHalf) {
373                 // Get the energy slope value.
374                 pairMassSameHalf.fill(getClusterPairMass(pair));
375                 int idx0 = pair[0].getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
376                 int idx1 = pair[1].getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
377                 if (Math.abs(idx0) > 2 && Math.abs(idx1) > 2) {
378                     pairMassSameHalfFid.fill(getClusterPairMass(pair));
379                 }
380             }
381 
382         } // If the event does not contain clusters, update the "Event
383         // Clusters" plot accordingly.
384         else {
385             clusterCountPlot.fill(0);
386         }
387     }

429     public double getClusterPairMass(Cluster[] pair) {
430         double x0 = pair[0].getPosition()[0];
431         double y0 = pair[0].getPosition()[1];
432         double z0 = pair[0].getPosition()[2];
433         double x1 = pair[1].getPosition()[0];
434         double y1 = pair[1].getPosition()[1];
435         double z1 = pair[1].getPosition()[2];
436         double e0 = pair[0].getEnergy();
437         double e1 = pair[1].getEnergy();
438         double xlen0 = sqrt(x0 * x0 + y0 * y0 + z0 * z0);
439         double xlen1 = sqrt(x1 * x1 + y1 * y1 + z1 * z1);
440         Hep3Vector p0 = new BasicHep3Vector(x0 / xlen0 * e0, y0 / xlen0 * e0, z0 / xlen0 * e0);
441         Hep3Vector p1 = new BasicHep3Vector(x1 / xlen1 * e1, y1 / xlen1 * e1, z1 / xlen1 * e1);
442 
443         double esum = sqrt(p1.magnitudeSquared()) + sqrt(p0.magnitudeSquared());
444         double pxsum = p1.x() + p0.x();
445         double pysum = p1.y() + p0.y();
446         double pzsum = p1.z() + p0.z();
447 
448         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
449         double evtmass = esum * esum - psum * psum;
450 
451         if (evtmass > 0) {
452             return Math.sqrt(evtmass);
453         } else {
454             return -99;
455         }
