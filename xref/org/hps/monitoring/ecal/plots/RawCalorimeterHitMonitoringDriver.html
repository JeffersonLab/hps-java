33  /**
34   * This is a set of detailed plots for monitoring ECAL raw data collections in integral mode (modes 3 and 7).
35   */
36  public class RawCalorimeterHitMonitoringDriver extends Driver {
37  
38      private int accumulatedEvents = 0;
39      private final AIDA aida = AIDA.defaultInstance();
40      private IHistogram1D amplitudeH1D;
41      private ICloud1D amplitudeHighC1D;
42      private IProfile2D amplitudeHighP2D;
43      private ICloud1D amplitudeLowC1D;
44      private IProfile2D amplitudeLowP2D;
45      private EcalConditions conditions;
46      private IIdentifierHelper helper;
47      private IHistogram2D hitMapH2D;
48      private IHistogram1D hitsPerEventH1D;
49  
50      private long lastUpdatedMillis = 0;
51      private IHistogram2D occupancyH2D;
52      private final Map<Long, Integer> occupancyMap = new HashMap<Long, Integer>();
53      private long occupancyUpdateMillis = 5000; // Refresh occupancy plots every 5 seconds.
54  
55      private ICloud1D pedSubAmplitudeC1D;
56  
57      private String rawCalorimeterHitCollectionName = "EcalReadoutHits";
58  
59      private Subdetector subdetector;
60      private ICloud1D timestampC1D;
61      private IProfile2D timestampP2D;
62  
63      @Override
64      public void detectorChanged(final Detector detector) {
65          final DatabaseConditionsManager manager = DatabaseConditionsManager.getInstance();
66          this.conditions = manager.getCachedConditions(EcalConditions.class, "ecal_conditions").getCachedData();
67          this.subdetector = detector.getSubdetector("Ecal");
68          this.helper = this.subdetector.getDetectorElement().getIdentifierHelper();
69      }
70  
71      @Override
72      public void process(final EventHeader event) {
73  
74          if (event.hasCollection(RawCalorimeterHit.class, this.rawCalorimeterHitCollectionName)) {
75  
76              final List<RawCalorimeterHit> rawHits = event.get(RawCalorimeterHit.class,
77                      this.rawCalorimeterHitCollectionName);
78  
79              /*
80               * Do basic plots of RawCalorimeterHit data.
81               */
82              this.hitsPerEventH1D.fill(rawHits.size());
83              final Set<RawCalorimeterHit> uniqueRawHits = new HashSet<RawCalorimeterHit>();
84              for (final RawCalorimeterHit rawHit : rawHits) {
85                  this.amplitudeH1D.fill(rawHit.getAmplitude());
86                  this.timestampC1D.fill(rawHit.getTimeStamp());
87                  final EcalChannel channel = this.conditions.getChannelCollection().findGeometric(rawHit.getCellID());
88                  final EcalChannelConstants constants = this.conditions.getChannelConstants(channel);
89                  this.pedSubAmplitudeC1D.fill(rawHit.getAmplitude() - constants.getCalibration().getPedestal()
90                          * EcalMonitoringUtilities.INTEGRAL_WINDOW);
91  
92                  final int ix = this.helper.getValue(new Identifier(rawHit.getCellID()), "ix");
93                  final int iy = this.helper.getValue(new Identifier(rawHit.getCellID()), "iy");
94                  this.hitMapH2D.fill(ix, iy);
95  
96                  this.timestampP2D.fill(ix, iy, rawHit.getTimeStamp());
97  
98                  uniqueRawHits.add(rawHit);
99              }
100 
101             /*
102              * Update and plot occupancies based on unique hits in the event.
103              */
104             for (final RawCalorimeterHit rawHit : uniqueRawHits) {
105 
106                 // Add entry if does not exist.
107                 if (this.occupancyMap.get(rawHit.getCellID()) == null) {
108                     this.occupancyMap.put(rawHit.getCellID(), 0);
109                 }
110 
111                 // Increment hit count by one for this hit.
112                 final int nHits = this.occupancyMap.get(rawHit.getCellID()) + 1;
113                 this.occupancyMap.put(rawHit.getCellID(), nHits);
114             }
115             ++this.accumulatedEvents;
116             final long elapsed = System.currentTimeMillis() - this.lastUpdatedMillis;
117             if (elapsed >= this.occupancyUpdateMillis) {
118                 this.occupancyH2D.reset();
119                 for (final Entry<Long, Integer> entry : this.occupancyMap.entrySet()) {
120                     final IIdentifier hitId = new Identifier(entry.getKey());
121                     final int ix = this.helper.getValue(hitId, "ix");
122                     final int iy = this.helper.getValue(hitId, "iy");
123                     this.occupancyH2D.fill(ix, iy, (double) entry.getValue() / (double) this.accumulatedEvents);
124                 }
125                 this.lastUpdatedMillis = System.currentTimeMillis();
126                 this.accumulatedEvents = 0;
127                 this.occupancyMap.clear();
128             }
129 
130             /*
131              * Plot from mode 7 data if it exists.
132              */
133             if (event.hasCollection(LCRelation.class, EcalMonitoringUtilities.EXTRA_DATA_RELATIONS_NAME)) {
134                 final List<LCRelation> extraDataRelations = event.get(LCRelation.class,
135                         EcalMonitoringUtilities.EXTRA_DATA_RELATIONS_NAME);
136                 if (extraDataRelations != null) {
137                     for (final LCRelation rel : event.get(LCRelation.class,
138                             EcalMonitoringUtilities.EXTRA_DATA_RELATIONS_NAME)) {
139                         final GenericObject extraData = (GenericObject) rel.getTo();
140                         if (extraData instanceof HitExtraData.Mode7Data) {
141                             final HitExtraData.Mode7Data mode7Data = (HitExtraData.Mode7Data) extraData;
142                             this.amplitudeHighC1D.fill(mode7Data.getAmplHigh());
143                             this.amplitudeLowC1D.fill(mode7Data.getAmplLow());
144 
145                             final RawCalorimeterHit rawHit = (RawCalorimeterHit) rel.getFrom();
146                             // FIXME: Duplicated from RawTrackerHit loop from above.
147                             final int ix = this.helper.getValue(new Identifier(rawHit.getCellID()), "ix");
148                             final int iy = this.helper.getValue(new Identifier(rawHit.getCellID()), "iy");
149 
150                             // Average amplitude high.
151                             this.amplitudeHighP2D.fill(ix, iy, mode7Data.getAmplHigh());
152 
153                             // Average amplitude low.
154                             this.amplitudeLowP2D.fill(ix, iy, mode7Data.getAmplLow());
155                         }
156                     }
157                 }
158             }
159         }
160     }

163         this.occupancyUpdateMillis = occupancyUpdateMillis;
164     }
165 
166     public void setRawCalorimeterHitCollectionName(final String rawCalorimeterHitCollectionName) {
167         this.rawCalorimeterHitCollectionName = rawCalorimeterHitCollectionName;
168     }

170     @Override
171     public void startOfData() {
172         this.pedSubAmplitudeC1D = this.aida.cloud1D("Pedestal Sub Amplitude");
173         this.amplitudeH1D = this.aida.histogram1D("Amplitude", 150, 0.5, 14999.5);
174         this.timestampC1D = this.aida.cloud1D("Timestamp");
175         this.amplitudeLowC1D = this.aida.cloud1D("Mode 7 Amplitude Low");
176         this.amplitudeHighC1D = this.aida.cloud1D("Mode 7 Amplitude High");
177         this.hitsPerEventH1D = this.aida.histogram1D("Hits Per Event", 100, -0.5, 99.5);
178         this.hitMapH2D = EcalMonitoringUtilities.createChannelHistogram2D("Hit Map");
179         this.occupancyH2D = EcalMonitoringUtilities.createChannelHistogram2D("Hit Occupancy");
180         this.amplitudeHighP2D = EcalMonitoringUtilities.createChannelProfile2D("Mode 7 Average Amplitude High");
181         this.amplitudeLowP2D = EcalMonitoringUtilities.createChannelProfile2D("Mode 7 Average Amplitude Low");
182         this.timestampP2D = EcalMonitoringUtilities.createChannelProfile2D("Average Timestamp");
183 
184         final IPlotterFactory plotterFactory = this.aida.analysisFactory().createPlotterFactory(
185                 "ECAL - " + this.rawCalorimeterHitCollectionName);
186 
187         EcalMonitoringUtilities.plot(plotterFactory, this.pedSubAmplitudeC1D, null, true);
188         EcalMonitoringUtilities.plot(plotterFactory, this.amplitudeH1D, null, true);
189         EcalMonitoringUtilities.plot(plotterFactory, this.timestampC1D, null, true);
190         EcalMonitoringUtilities.plot(plotterFactory, this.amplitudeLowC1D, null, true);
191         EcalMonitoringUtilities.plot(plotterFactory, this.amplitudeHighC1D, null, true);
192         EcalMonitoringUtilities.plot(plotterFactory, this.hitsPerEventH1D, null, true);
193         EcalMonitoringUtilities.plot(plotterFactory, this.hitMapH2D, null, true);
194         EcalMonitoringUtilities.plot(plotterFactory, this.occupancyH2D, null, true);
195         EcalMonitoringUtilities.plot(plotterFactory, this.amplitudeHighP2D, null, true);
196         EcalMonitoringUtilities.plot(plotterFactory, this.amplitudeLowP2D, null, true);
197         EcalMonitoringUtilities.plot(plotterFactory, this.timestampP2D, null, true);
198 
199         this.lastUpdatedMillis = System.currentTimeMillis();
200     }
