1   package org.hps.monitoring.ecal.plots;
2   
3   import hep.aida.IAnalysisFactory;
4   import hep.aida.IHistogram1D;
5   import hep.aida.IHistogram2D;
6   import hep.aida.IHistogramFactory;
7   import hep.aida.IPlotter;
8   import hep.aida.IPlotterFactory;
9   
10  import org.lcsim.event.CalorimeterHit;
11  import org.lcsim.event.Cluster;
12  import org.lcsim.event.EventHeader;
13  import org.lcsim.event.RawCalorimeterHit;
14  import org.lcsim.util.Driver;
15  
16  /**
17   * Basic ECal monitoring plots that work on Test Run data.
18   */
19  public class BasicMonitoringPlotsDriver extends Driver {
20  
21      private String calHitsCollectionName = "EcalCalHits";
22      private String rawHitsCollectionName = "EcalReadoutHits";
23      private String clustersCollectionName = "EcalClusters";
24  
25      private IHistogram1D calHitEnergyH1D;
26      private IHistogram1D clusterEnergyH1D;
27      private IHistogram1D rawHitAmplitudeH1D;
28      private IHistogram2D calHitEnergyMapH2D;
29  
30      public BasicMonitoringPlotsDriver() {
31      }
32  
33      public void startOfData() {
34  
35          IAnalysisFactory.create().createHistogramFactory(null);
36          IPlotterFactory plotFactory = IAnalysisFactory.create().createPlotterFactory("ECAL Monitoring");
37          IHistogramFactory histogramFactory = IAnalysisFactory.create().createHistogramFactory(null);
38  
39          calHitEnergyH1D = histogramFactory.createHistogram1D(calHitsCollectionName + ": Energy", calHitsCollectionName
40                  + ": Energy", 200, 0.0, 2.0);
41          calHitEnergyH1D.annotation().addItem("xAxisLabel", "GeV");
42          calHitEnergyH1D.annotation().addItem("yAxisLabel", "Count");
43          IPlotter plotter = plotFactory.create("CalorimeterHits");
44          plotter.createRegion();
45          plotter.style().gridStyle().setVisible(false);
46          plotter.style().dataStyle().errorBarStyle().setVisible(false);
47          plotter.region(0).plot(calHitEnergyH1D);
48          plotter.show();
49  
50          rawHitAmplitudeH1D = histogramFactory.createHistogram1D(rawHitsCollectionName + ": Amplitude",
51                  rawHitsCollectionName + ": Amplitude", 150, 0.0, 15000.0);
52          rawHitAmplitudeH1D.annotation().addItem("xAxisLabel", "ADC Value");
53          rawHitAmplitudeH1D.annotation().addItem("yAxisLabel", "Count");
54          plotter = plotFactory.create("RawCalorimeterHits");
55          plotter.createRegion();
56          plotter.style().gridStyle().setVisible(false);
57          plotter.style().dataStyle().errorBarStyle().setVisible(false);
58          plotter.region(0).plot(rawHitAmplitudeH1D);
59          plotter.show();
60  
61          clusterEnergyH1D = histogramFactory.createHistogram1D(clustersCollectionName + ": Energy",
62                  clustersCollectionName + ": Energy", 100, 0.0, 3.0);
63          clusterEnergyH1D.annotation().addItem("xAxisLabel", "GeV");
64          clusterEnergyH1D.annotation().addItem("yAxisLabel", "Count");
65          plotter = plotFactory.create("Clusters");
66          plotter.createRegion();
67          plotter.style().gridStyle().setVisible(false);
68          plotter.style().dataStyle().errorBarStyle().setVisible(false);
69          plotter.region(0).plot(clusterEnergyH1D);
70          plotter.show();
71  
72          calHitEnergyMapH2D = histogramFactory.createHistogram2D(calHitsCollectionName + ": Energy Map",
73                  calHitsCollectionName + ": Energy Map", 47, -23.5, 23.5, 11, -5.5, 5.5);
74          plotter = plotFactory.create("CalorimeterHit Energy Map");
75          plotter.createRegion();
76          plotter.style().setParameter("hist2DStyle", "colorMap");
77          plotter.style().gridStyle().setVisible(false);
78          plotter.region(0).plot(calHitEnergyMapH2D);
79          plotter.show();
80      }
81  
82      public void process(EventHeader event) {
83  
84          if (event.hasCollection(CalorimeterHit.class, calHitsCollectionName)) {
85              for (CalorimeterHit hit : event.get(CalorimeterHit.class, calHitsCollectionName)) {
86                  calHitEnergyH1D.fill(hit.getCorrectedEnergy());
87                  calHitEnergyMapH2D.fill(hit.getIdentifierFieldValue("ix"), hit.getIdentifierFieldValue("iy"),
88                          hit.getCorrectedEnergy());
89              }
90          }
91  
92          if (event.hasCollection(Cluster.class, clustersCollectionName)) {
93              for (Cluster cluster : event.get(Cluster.class, clustersCollectionName)) {
94                  clusterEnergyH1D.fill(cluster.getEnergy());
95              }
96          }
97  
98          if (event.hasCollection(RawCalorimeterHit.class, rawHitsCollectionName)) {
99              for (RawCalorimeterHit hit : event.get(RawCalorimeterHit.class, rawHitsCollectionName)) {
100                 rawHitAmplitudeH1D.fill(hit.getAmplitude());
101             }
102         }
103     }
104 }
