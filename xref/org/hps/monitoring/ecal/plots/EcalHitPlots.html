20  /**
21   * The driver <code>EcalHitPlots</code> implements the histogram shown to the
22   * user
23   * in the second tab of the Monitoring Application, when using the Ecal
24   * monitoring lcsim file.
25   * These histograms shows single-channels distributions:
26   * - First sub-tab shows the hits distribution* (Histogram2D), the occupancy*
27   * (Histogram2D), the number of hits per
28   * event (Histogram1D),
29   * the time distribution of the hits (Histogram1D)
30   * The first two histograms are defined in <code>EcalMonitoringPlots</code>.
31   * - Second sub-tab shows the energy distribution of the hits (Histogram1D), and
32   * the maximum energy in each event
33   * (Histogram1D)
34   * - Third sub-tab shows the time distribution of the first hit per event, for
35   * the Ecal top (Histogram1D), for the Ecal
36   * bottom (Histogram1D), for both for the Ecal top (Histogram1D).
37   *
38   * Histograms are updated continously, expect those marked with *, that are
39   * updated regularly depending on the event
40   * refresh rate configured in the <code> EcalMonitoringPlots </code> driver
41   */

86          plotterFactory = aida.analysisFactory().createPlotterFactory("Ecal Hit Plots");

88          // Setup plots.

97          orTimePlot = aida.histogram1D(detector.getDetectorName() + " : " + inputCollection + " : First Hit Time, Or",

99

104         botTrigTimePlot = aida.histogram1D(detector.getDetectorName() + " : " + inputCollection

106         orTrigTimePlot = aida.histogram1D(detector.getDetectorName() + " : " + inputCollection + " : Trigger Time, Or",

193     @Override
194     public void process(EventHeader event) {
195 
196         int orTrigTime = 4097;
197         int topTrigTime = 4097;
198         int botTrigTime = 4097;
199 
200         if (event.hasCollection(GenericObject.class, "TriggerBank")) {
201             List<GenericObject> triggerList = event.get(GenericObject.class, "TriggerBank");
202             if (!triggerList.isEmpty()) {
203                 GenericObject triggerData = triggerList.get(0);
204 
205                 if (triggerData instanceof SSPData) {
206                     // TODO: TOP, BOTTOM, OR, and AND triggers are test
207                     // run specific parameters and are not supported
208                     // by SSPData.
209                     orTrigTime = 0; // ((SSPData)triggerData).getOrTrig();
210                     topTrigTime = 0; // ((SSPData)triggerData).getTopTrig();
211                     botTrigTime = 0; // ((SSPData)triggerData).getBotTrig();
212 
213                     orTrigTimePlot.fill(orTrigTime);
214                     topTrigTimePlot.fill(topTrigTime);
215                     botTrigTimePlot.fill(botTrigTime);
216 
217                 }
218 
219             }// end if triggerList isEmpty
220         }
221 
222         if (event.hasCollection(CalorimeterHit.class, inputCollection)) {
223             List<CalorimeterHit> hits = event.get(CalorimeterHit.class, inputCollection);
224             hitCountPlot.fill(hits.size());
225             int id = 0;
226             int row = 0;
227             int column = 0;
228             double maxEnergy = 0;
229             double topTime = Double.POSITIVE_INFINITY;
230             double botTime = Double.POSITIVE_INFINITY;
231             double orTime = Double.POSITIVE_INFINITY;
232             for (CalorimeterHit hit : hits) {
233 
234                 hitEnergyPlot.fill(hit.getRawEnergy());
235                 hitTimePlot.fill(hit.getTime());
236 
237                 if (hit.getTime() < orTime) {
238                     orTime = hit.getTime();
239                 }
240                 if (hit.getIdentifierFieldValue("iy") > 0 && hit.getTime() < topTime) {
241                     topTime = hit.getTime();
242                 }
243                 if (hit.getIdentifierFieldValue("iy") < 0 && hit.getTime() < botTime) {
244                     botTime = hit.getTime();
245                 }
246                 if (hit.getRawEnergy() > maxEnergy) {
247                     maxEnergy = hit.getRawEnergy();
248                 }
249             }
250 
251             if (orTime != Double.POSITIVE_INFINITY) {
252                 orTimePlot.fill(orTime);
253                 orTimePlot2D.fill(orTime, orTrigTime);
254             }
255             if (topTime != Double.POSITIVE_INFINITY) {
256                 topTimePlot.fill(topTime);
257                 topTimePlot2D.fill(topTime, topTrigTime);
258             }
259             if (botTime != Double.POSITIVE_INFINITY) {
260                 botTimePlot.fill(botTime);
261                 botTimePlot2D.fill(botTime, botTrigTime);
262             }
263             hitMaxEnergyPlot.fill(maxEnergy);
264         } else {
265             hitCountPlot.fill(0);
266         }
267     }

278     /**
279      * Initializes the default style for plots.
280      *
281      * @return Returns an <code>IPlotterStyle</code> object that
282      * represents the default style for plots.
283      */
284     public IPlotterStyle createDefaultStyle() {
285         IPlotterStyle pstyle = plotterFactory.createPlotterStyle();
286         // Set the appearance of the axes.
287         pstyle.xAxisStyle().labelStyle().setBold(true);
288         pstyle.yAxisStyle().labelStyle().setBold(true);
289         pstyle.xAxisStyle().tickLabelStyle().setBold(true);
290         pstyle.yAxisStyle().tickLabelStyle().setBold(true);
291         pstyle.xAxisStyle().lineStyle().setColor("black");
292         pstyle.yAxisStyle().lineStyle().setColor("black");
293         pstyle.xAxisStyle().lineStyle().setThickness(2);
294         pstyle.yAxisStyle().lineStyle().setThickness(2);
295 
296         // Set color settings.
297         pstyle.dataStyle().fillStyle().setParameter("colorMapScheme", "rainbow");
298         pstyle.dataStyle().fillStyle().setParameter("showZeroHeightBins", Boolean.FALSE.toString());
299         pstyle.dataStyle().errorBarStyle().setVisible(false);
300         pstyle.setParameter("hist2DStyle", "colorMap");
301 
302         // Force auto range to zero.
303         pstyle.yAxisStyle().setParameter("allowZeroSuppression", "false");
304         pstyle.xAxisStyle().setParameter("allowZeroSuppression", "false");
305 
306         // Set the title style.
307         pstyle.titleStyle().textStyle().setFontSize(20);
308 
309         // Draw caps on error bars.
310         pstyle.dataStyle().errorBarStyle().setParameter("errorBarDecoration", (new Float(1.0f)).toString());
311 
312         // Turn off grid lines until explicitly enabled.
313         pstyle.gridStyle().setVisible(false);
314 
315         // Return the style.
316         return pstyle;
317     }
