123     public void process(EventHeader event) {
124         if (event.hasCollection(RawTrackerHit.class, inputCollection)) {
125             List<RawTrackerHit> hits = event.get(RawTrackerHit.class, inputCollection);
126             for (RawTrackerHit hit : hits) {
127                 dec.setID(hit.getCellID());
128                 int x = dec.getValue("ix");
129                 int y = dec.getValue("iy");
130                 // System.out.println("got hit: x= " + x + ", y= " + y);
131                 if (isFirst) {
132                     System.out.println("FIRST!!!");
133                     isFirst = false;
134                     window = hit.getADCValues().length;
135                     windowPlot = aida.histogram1D(detector.getDetectorName() + " : " + inputCollection
136                             + " : Window Mode Data", window, -0.5, window - 0.5);
137                     plotter.region(0).clear();
138                     plotter.region(0).plot(windowPlot);
139                     plotter.region(0).refresh();
140 
141                 }
142                 if (testX && x != plotX) {
143                     continue;
144                 }
145                 if (testY && y != plotY) {
146                     continue;
147                 }
148                 windowPlot.reset();
149                 for (int i = 0; i < window; i++) {
150                     windowPlot.fill(i, hit.getADCValues()[i] * EcalUtils.adcResolution);
151 
152                 }
153             }
154         }
155     }

157     @Override

159         String selItem;
160         selItem = (String) xCombo.getSelectedItem();
161         if (selItem.equals("all")) {
162             testX = false;
163         } else {
164             testX = true;
165             plotX = Integer.decode(selItem);
166         }
167 
168         selItem = (String) yCombo.getSelectedItem();
169         if (selItem.equals("all")) {
170             testY = false;
171         } else {
172             testY = true;
173             plotY = Integer.decode(selItem);
174         }
175     }
