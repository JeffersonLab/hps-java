6    * Class <code>CrystalDataSet</code> contains all of the hardware data
7    * for a single calorimeter crystal as defined in the crystal hardware
8    * reference sheet.
9    */
10  public class CrystalDataSet {    
11      // Data points.
12      private final Point crystalIndex;
13      private final short apd;
14      private final Preamplifier preamp;
15      private final short ledChannel;
16      private final byte[] ledDriver;
17      private final byte fadcSlot;
18      private final byte fadcChannel;
19      private final byte splitterNum;
20      private final byte hvGroup;
21      private final byte jout;
22      private final Motherboard motherboard;
23      private final short channel;
24      private final int gain;
25      
26      /**
27       * Defines the data set.
28       * @param ix - The crystal's x-index in the LCSimcoordinate system.
29       * @param iy - The crystal's y-index in the LCSimcoordinate system.
30       * @param apd - The number of the APD attached to the crystal.
31       * @param preamp - The number of the crystal's preamplifier. This
32       * may also include a reference wire color.
33       * @param ledChannel - The channel number for the crystal's LED.
34       * @param ledDriver
35       * @param fadcSlot - The FADC slot the crystal occupies.
36       * @param fadcChannel - The channel number through which the crystal
37       * communicates with the FADC.
38       * @param splitter - The number of the crystal's splitter.
39       * @param hvGroup - The high voltage group for which the crystal
40       * is a member.
41       * @param jout - The crystal's signal channel group.
42       * @param mb - The position of the motherboard associated with the
43       * crystal.
44       * @param channel - The channel associated with the crystal.
45       * @param gain - The crystal's gain.
46       */
47      public CrystalDataSet(int ix, int iy, int apd, String preamp, int ledChannel,
48              double ledDriver, int fadcSlot, int fadcChannel, int splitter, int hvGroup,
49              int jout, String mb, int channel, int gain) {
50          // Define crystal indices.
51          crystalIndex = new Point(ix, iy);
52          
53          // Define the general properties.
54          this.apd = (short) apd;
55          this.ledChannel = (short) ledChannel;
56          this.fadcSlot = (byte) fadcSlot;
57          this.fadcChannel = (byte) fadcChannel;
58          this.splitterNum = (byte) splitter;
59          this.hvGroup = (byte) hvGroup;
60          this.jout = (byte) jout;
61          this.channel = (short) channel;
62          this.gain = gain;
63          this.motherboard = Motherboard.getMotherboard(mb);
64          
65          // Define the LED driver.
66          this.ledDriver = new byte[2];
67          this.ledDriver[0] = (byte) Math.floor(ledDriver);
68          this.ledDriver[1] = (byte) ((ledDriver - this.ledDriver[0]) * 10);
69          
70          // Handle the preamplifier number.
71          StringBuffer num = new StringBuffer();
72          StringBuffer col = new StringBuffer();
73          for(char c : preamp.toCharArray()) {
74              if(Character.isDigit(c)) { num.append(c); }
75              else { col.append(c); }
76          }
77          int number = Integer.parseInt(num.toString());
78          String color = null;
79          if(col.length() != 0) { color = col.toString(); }
80          this.preamp = new Preamplifier(number, color);
81      }

182     /**
183      * Class <code>Preamplifier</code> represents the number
184      * of a crystal's preamplifier. It can also contain a reference
185      * wire color if necessary.
186      * 
187      */
188     public class Preamplifier implements Comparable<Preamplifier> {
189         private final short number;
190         private final String color;
191         
192         /**
193          * Initializes a preamplifier with no reference wire color.
194          * @param number - The preamplifier's number.
195          */
196         public Preamplifier(int number) { this(number, null); }
197         
198         /**
199          * Initializes a preamplifier with the specified reference
200          * wire color.
201          * @param number - The preamplifier's number.
202          * @param color - The reference wire color.
203          */
204         public Preamplifier(int number, String color) {
205             this.number = (short) number;
206             this.color = color;
207         }
208         
209         /**
210          * Gets the number of the preamplifier.
211          * @return Returns the preamplifier number as an <code>int
212          * </code> primitive.
213          */
214         public int getNumber() { return number; }
215         
216         /**
217          * Gets the reference wire color associated with the crystal's
218          * preamplifier if it exists.
219          * @return Returns the reference wire color as a <code>String
220          * </code> object or <code>null</code> if it does not exist.
221          */
222         public String getColor() { return color; }
223         
224         @Override
225         public String toString() {
226             if(color == null) { return "" + number; }
227             else { return number + " (" + color + ")"; }
228         }
229         
230         @Override
231         public int compareTo(Preamplifier preamp) {
232             // Compare the preamplifier numbers.
233             int numCompare = Short.compare(number, preamp.number);
234             
235             // If they are different, return this value.
236             if(numCompare != 0) { return numCompare; }
237             
238             // The color string must now be compared. If both are null,
239             // then they are the same.
240             if(color == null && preamp.color == null) { return 0; }
241             
242             // If one is null and the other is not, the null one is
243             // always ordered first.
244             if(color == null && preamp.color != null) { return -1; }
245             else if(color != null && preamp.color == null) { return 1; }
246             
247             // If neither color is null, compare them traditionally.
248             return color.compareTo(preamp.color);
249         }
250     }

252     /**
253      * Enumerable <code>Motherboard</code> contains convenience methods
254      * for defining the motherboard location for a particular crystal.
255      * 
256      */
257     public enum Motherboard {
258         /** The motherboard on the upper, left-hand side. */
259         TOP_LEFT(true, true),
260         /** The motherboard on the upper, right-hand side. */
261         TOP_RIGHT(true, false),
262         /** The motherboard on the lower, left-hand side. */
263         BOTTOM_LEFT(false, true),
264         /** The motherboard on the lower, right-hand side. */
265         BOTTOM_RIGHT(false, false);
266         
267         private final boolean isTop;
268         private final boolean isLeft;
269         
270         /**
271          * Instantiates a new motherboard enumerable.
272          * @param isTop - Indicates whether the motherboard is on the
273          * upper side of the detector.
274          * @param isLeft - Indicates whether the motherboard is on the
275          * left-hand side of the detector.
276          */
277         private Motherboard(boolean isTop, boolean isLeft) {
278             this.isTop = isTop;
279             this.isLeft = isLeft;
280         }
281         
282         /**
283          * Indicates whether the motherboard is on the top of the
284          * detector or not.
285          * @return Returns <code>true</code> if the motherboard is on
286          * the upper half of the detector and <code>false</code> if it
287          * is not.
288          */
289         public boolean isTop() { return isTop; }
290         
291         /**
292          * Indicates whether the motherboard is on the bottom of the
293          * detector or not.
294          * @return Returns <code>true</code> if the motherboard is on
295          * the lower half of the detector and <code>false</code> if it
296          * is not.
297          */
298         public boolean isBottom() { return !isTop; }
299         
300         /**
301          * Indicates whether the motherboard is on the left-hand side
302          * of the detector or not.
303          * @return Returns <code>true</code> if the motherboard is on
304          * the left-hand side of the detector and <code>false</code>
305          * if it is not.
306          */
307         public boolean isLeft() { return isLeft; }
308         
309         /**
310          * Indicates whether the motherboard is on the right-hand side
311          * of the detector or not.
312          * @return Returns <code>true</code> if the motherboard is on
313          * the right-hand side of the detector and <code>false</code>
314          * if it is not.
315          */
316         public boolean isRight() { return !isLeft; }
317         
318         /**
319          * Gets the <code>Motherboard</code> enumerable associated
320          * with the given textual abbreviation. Valid arguments include
321          * "LT", "RT", "LB", and "RB".
322          * @param abbreviation - The textual abbreviation for the
323          * location of the motherboard.
324          * @return Returns the appropriate <code>Motherboard</code>
325          * enumerable if given a valid abbreviation and <code>null
326          * </code> otherwise.
327          */
328         public static final Motherboard getMotherboard(String abbreviation) {
329             // Abbreviations must be 2 characters in length.
330             if(abbreviation.length() != 2) { return null; }
331             
332             // The first character must be either 'R' or 'L'.
333             boolean isLeft;
334             if(abbreviation.charAt(0) == 'L') { isLeft = true; }
335             else if(abbreviation.charAt(0) == 'R') { isLeft = false; }
336             else { return null; }
337             
338             // The second character must be either 'T' or 'B'.
339             boolean isTop;
340             if(abbreviation.charAt(1) == 'T') { isTop = true; }
341             else if(abbreviation.charAt(1) == 'B') { isTop = false; }
342             else { return null; }
343             
344             // Return the appropriate motherboard enumerable.
345             return getMotherboard(isTop, isLeft);
346         }
347         
348         /**
349          * Gets the <code>Motherboard</code> enumerable associated
350          * with the given position.
351          * @param isTop - <code>true</code> indicates the motherboard
352          * on the top half of the detector and <code>false</code> the
353          * motherboard on the bottom half.
354          * @param isLeft - <code>true</code> indicates the motherboard
355          * on the left-hand side of the detector and <code>false</code>
356          * the motherboard on the right-hand side.
357          * @return Returns the appropriate <code>Motherboard</code>
358          * enumerable.
359          */
360         public static final Motherboard getMotherboard(boolean isTop, boolean isLeft) {
361             if(isTop) {
362                 if(isLeft) { return TOP_LEFT; }
363                 else { return TOP_RIGHT; }
364             }
365             else {
366                 if(isLeft) { return BOTTOM_LEFT; }
367                 else { return BOTTOM_RIGHT; }
368             }
369         }
370         
371         @Override
372         public String toString() {
373             if(isTop) {
374                 if(isLeft) { return "LT"; }
375                 else { return "RT"; }
376             }
377             else {
378                 if(isLeft) { return "LB"; }
379                 else { return "RB"; }
380             }
381         }
