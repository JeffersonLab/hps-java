1   package org.hps.monitoring.ecal.eventdisplay.util;
2   import java.awt.Color;
3   
4   /**
5    * The class <code>GradientScale</code> is an implementation of the abstract
6    * class <code>ColorScale</code> which represents a simple gradient from one
7    * color to another. It will map any argument value that exceeds its maximum
8    * value to the hot color and any argument value that is below its minimum value
9    * to its cold color. All other argument values will be mapped somewhere between
10   * the cold and hot colors using either a linear or logarithmic scale.
11   **/
12  public final class GradientScale extends ColorScale {
13      // The color associated with the maximum value.
14      private Color hotColor = Color.WHITE;
15      // The color associated with the minimum value.
16      private Color coldColor = Color.BLACK;
17      // Efficiency variable holding the rgb difference between the two
18      // colors. This is used to prevent recalculation when mapping values.
19      private int[] drgb = { 255, 255, 255 };
20      
21      public Color getColor(Double value) {
22          // If the argument is null, treat it as zero.
23          if(value == null) { value = 0.0; }
24          
25          // If the value is less than the minimum, return the cold color.
26          if (value < min) { return coldColor; }
27          
28          // If the value is greater than the maximum, return the hot color.
29          if (value > max) { return hotColor; }
30          
31          // Otherwise, calculate how far along the gradient the value is.
32          double percent;
33          if (linear) { percent = (value - min) / (max - min); }
34          else {
35              double lValue = Math.log10(scale * value);
36              percent = (lValue - lMin) / (lMax - lMin);
37          }
38          
39          // Scale the color.
40          int dr = (int) Math.round(percent * drgb[0]);
41          int dg = (int) Math.round(percent * drgb[1]);
42          int db = (int) Math.round(percent * drgb[2]);
43          
44          // Return the result.
45          return new Color(coldColor.getRed() + dr, coldColor.getGreen() + dg, coldColor.getBlue() + db);
46      }
47      
48      /**
49       * <b>setColdColor</b><br/><br/>
50       * <code>public void <b>setColdColor</b>(Color c)</code><br/><br/>
51       * Sets the color associated with the minimum value.
52       * @param c - The color to use.
53       **/
54      public void setColdColor(Color c) {
55          coldColor = c;
56          revalidateColor();
57      }
58      
59      /**
60       * <b>setHotColor</b><br/><br/>
61       * <code>public void <b>setHotColor</b>(Color c)</code><br/><br/>
62       * Sets the color associated with the maximum value.
63       * @param c - The new color to use.
64       **/
65      public void setHotColor(Color c) {
66          hotColor = c;
67          revalidateColor();
68      }
69      
70      /**
71       * <b>makeGreyScale</b><br/><br/>
72       * <code>public static GradientScale <b>makeGreyScale</b>(double minimum, double maximum)</code><br/><br/>
73       * Creates a color scale that ranges from black (cold) to white (hot) with
74       * the indicated maximum and minimum.
75       * @param minimum - The lowest value for color scaling.
76       * @param maximum - The highest value for color scaling.
77       * @return Returns a <code>GradientScale</code> that maps to grey scale over
78       * the indicated range.
79       **/
80      public static GradientScale makeGreyScale(double minimum, double maximum) {
81          GradientScale gs = new GradientScale();
82          gs.setMinimum(minimum);
83          gs.setMaximum(maximum);
84          
85          return gs;
86      }
87      
88      /**
89       * <b>makeHeatScale</b><br/><br>
90       * <code>public static GradientScale <b>makeHeatScale</b>(double minimum, double maximum)</code><br/><br/>
91       * Creates a color scale that ranges from black (cold) to red (hot) with the
92       * indicated maximum and minimum.
93       * @param minimum - The lowest value for color scaling.
94       * @param maximum - The highest value for color scaling.
95       * @return Returns a <code>GradientScale</code> that maps to a black- to-red
96       * gradient over the indicated range.
97       **/
98      public static GradientScale makeHeatScale(double minimum, double maximum) {
99          GradientScale hs = new GradientScale();
100         hs.setHotColor(Color.RED);
101         hs.setColdColor(Color.BLACK);
102         hs.setMinimum(minimum);
103         hs.setMaximum(maximum);
104         
105         return hs;
106     }
107     
108     /**
109      * <b>revalidateColor</b><br/><br/>
110      * <code>private void <b>revalidateColor</b>()</code><br/><br/>
111      * Calculates the differences between the hot and cold colors and sets the
112      * class related class variables.
113      **/
114     private void revalidateColor() {
115         drgb[0] = hotColor.getRed() - coldColor.getRed();
116         drgb[1] = hotColor.getGreen() - coldColor.getGreen();
117         drgb[2] = hotColor.getBlue() - coldColor.getBlue();
118     }
119 }
