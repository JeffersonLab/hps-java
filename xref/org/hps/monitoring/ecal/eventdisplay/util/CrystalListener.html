1   package org.hps.monitoring.ecal.eventdisplay.util;
2   
3   import java.util.EventListener;
4   
5   /**
6    * Interface <code>CrystalListener</code> receives events from a <code>
7    * Viewer</code> component regarding crystals. These include whenever
8    * a crystal is activated (i.e. it becomes highlighted), deactivated
9    * (i.e. it is no longer highlighted), and clicked. 
10   */
11  public interface CrystalListener extends EventListener {
12      /**
13       * <b>crystalActivated</b><br/><br/>
14       * <code>public void <b>crystalActivated</b>(CrystalEvent e)</code><br/><br/>
15       * Invoked when a crystal becomes highlighted.
16       * @param e - An object describing the event.
17       */
18      public void crystalActivated(CrystalEvent e);
19      
20      /**
21       * <b>crystalDeactivated</b><br/><br/>
22       * <code>public void <b>crystalDeactivated</b>(CrystalEvent e)</code><br/><br/>
23       * Invoked when a crystal ceases to be highlighted.
24       * @param e - An object describing the event.
25       */
26      public void crystalDeactivated(CrystalEvent e);
27      
28      /**
29       * <b>crystalClicked</b><br/><br/>
30       * <code>public void <b>crystalClicked</b>(CrystalEvent e)</code><br/><br/>
31       * Invoked when a crystal is clicked
32       * @param e - An object describing the event.
33       */
34      public void crystalClicked(CrystalEvent e);
35  }
