1   package org.hps.monitoring.ecal.eventdisplay.util;
2   
3   import java.awt.AWTEvent;
4   import java.awt.Point;
5   
6   import org.hps.monitoring.ecal.eventdisplay.ui.Viewer;
7   
8   /**
9    * Class <code>CrystalEvent</code> represents some event that occurred
10   * with respect to a crystal. It is thrown when a crystal either gains
11   * or loses focus or is clicked. Crystal ID indices are always with
12   * respect to the panel coordinate system.
13   */
14  public class CrystalEvent extends AWTEvent {
15      private static final long serialVersionUID = 77198267255387212L;
16      // Stores the location of the triggering crystal.
17      private final Point crystal;
18      // The AWTEvent id for this event.
19      private static final int AWT_ID = AWTEvent.RESERVED_ID_MAX + 10;
20      
21      /**
22       * <b>CrystalEvent</b><br/><br/>
23       * <code>public <b>CrystalEvent</b>(Viewer parent, Point triggerCrystal)</code><br/><br/>
24       * Creates a crystal event for the indicated crystal and triggering
25       * component.
26       * @param source - The triggering component.
27       * @param triggerCrystal - The crystal associated with the event.
28       * @throws IllegalArgumentException Occurs if the associated crystal
29       * is <code>null</code>.
30       */
31      public CrystalEvent(Viewer source, Point triggerCrystal) throws IllegalArgumentException {
32          // Run the superclass constructor.
33          super(source, AWT_ID);
34          
35          // Make sure that the trigger crystal is not null.
36          if(triggerCrystal == null) {
37              throw new IllegalArgumentException("Crystal events can not occur with respect to non-exstant crystals.");
38          }
39          
40          // Define the event parameters.
41          crystal = triggerCrystal;
42      }
43      
44      /**
45       * <b>getCrystalID</b><br/><br/>
46       * <code>public Point <b>getCrystalID</b>()</code><br/><br/>
47       * Indicates the panel indices at which the crystal is located.
48       * @return Returns the crystal's panel indices as a <code>Point
49       * </code> object.
50       */
51      public Point getCrystalID() { return crystal; }
52  }
