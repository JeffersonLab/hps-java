13  /**
14   * Abstract class <code>ActiveViewer</code> describes a <code>Viewer
15   * </code> object that is connected to a static data source. It is
16   * designed to instruct the data source when to provide new events.
17   * <br/><br/>
18   * <code>ActiveViewer</code> is now superseded by <code>FileViewer
19   * </code>, which functions on its own. Any additional file types that
20   * should be supported should be added directly to <code>FileViewer
21   * </code> or a subclass. <code>ActiveViewer</code> will be removed
22   * from coming releases.
23   */
24  @Deprecated
25  public abstract class ActiveViewer extends Viewer {
26      private static final long serialVersionUID = 2L;
27      // Gets events from some file.
28      protected EventManager em;
29      
30      /**
31       * Creates an active-type <code>Viewer</code> window which draws
32       * events from the indicated data source with additional status
33       * fields defined by the <code>fieldNames</code> argument.
34       * @param em - The data source event manager.
35       */
36      public ActiveViewer(EventManager em) {
37          // Pass any additional field values to the super class.
38          super();
39          
40          // Set the data source.
41          this.em = em;
42          
43          // Add an exit command to the file menu.
44          JMenuItem menuExit = new JMenuItem("Exit", KeyEvent.VK_X);
45          menuExit.addActionListener(new ActionListener() {
46              @Override
47              public void actionPerformed(ActionEvent e) { dispose(); }
48          });
49          menu[MENU_FILE].addSeparator();
50          menu[MENU_FILE].add(menuExit);
51          
52          // Make a key listener to change events.
53          addKeyListener(new EcalKeyListener());
54      }
55      
56      /**
57       * Feeds the calorimeter panel the data from the next event.
58       * @throws IOException Occurs when there is an issue with reading the data file.
59       **/
60      public abstract void displayNextEvent() throws IOException;
61      
62      /**
63       * Feeds the calorimeter panel the data from the previous event.
64       * @throws IOException Occurs when there is an issue with reading the data file.
65       **/
66      public abstract void displayPreviousEvent() throws IOException;
67      
68      /**
69       * The <code>EcalListener</code> class binds keys to actions.
70       * Bound actions include:
71       * [Right Arrow] :: Next event
72       * [Left Arrow ] :: Previous event
73       **/
74      private class EcalKeyListener extends KeyAdapter {
75          @Override
76          public void keyReleased(KeyEvent e) {
77              // If right-arrow was pressed, go to the next event.
78              if (e.getKeyCode() == 39) {
79                  try { displayNextEvent(); }
80                  catch (IOException ex) {
81                      System.err.println(ex.getMessage());
82                      System.exit(1);
83                  }
84              }
85              
86              // If left-arrow was pressed, go to the next event.
87              else if (e.getKeyCode() == 37) {
88                  try { displayPreviousEvent(); }
89                  catch (IOException ex) {
90                      System.err.println(ex.getMessage());
91                      System.exit(1);
92                  }
93              }
94              
95              // Otherwise, print out the key code for the pressed key.
96              else { System.out.printf("Key Code: %d%n", e.getKeyCode()); }
97          }
98      }
99  }
