31  /**
32   * Class <code>CrystalFilterPanel</code> is an extension of <code>
33   * JPanel</code> that allows users to select filters for crystals from
34   * the properties stored in a <code>CrystalDataSet</code> object and
35   * filter all the crystals present in an <code>EcalWiringManager</code>
36   * object by said properties.<br/>
37   * <br/>
38   * <code>CrystalFilterPanel</code> alerts other classes that a filter
39   * has been applied through <code>ActionListener</code> objects.
40   * 
41   * @see JPanel
42   * @see CrystalDataSet
43   * @see EcalWiringManager
44   */
45  public final class CrystalFilterPanel extends JPanel {
46      // Internal variables.
47      private Dimension preferredSize = null;
48      private final EcalWiringManager manager;
49      private List<Point> filteredList = null;

51      private CDSComparator cdsCompare = new CDSComparator();
52      private List<ActionListener> alList = new ArrayList<ActionListener>();
53      public static final int EVENT_FILTER_APPLIED = ActionEvent.RESERVED_ID_MAX + 10;
54      public static final int EVENT_FILTER_REMOVED = ActionEvent.RESERVED_ID_MAX + 11;

92      public CrystalFilterPanel(EcalWiringManager dataManager) {
93          // Store the manager.
94          manager = dataManager;
95          
96          // Create the panel layout.
97          SpringLayout layout = new SpringLayout();
98          setLayout(layout);
99          
100         // Instantiate the check boxes.
101         JCheckBox largestCheck = null;
102         checkActive = new JCheckBox[INDICES];
103         for(int index = 0; index < INDICES; index++) {
104             // Instantiate the check box.
105             checkActive[index] = new JCheckBox(FIELD_NAME[index]);
106             checkActive[index].setSelected(false);
107             checkActive[index].setForeground(COLOR_CHECK_FONT_INACTIVE);
108             checkActive[index].addItemListener(new CheckListener(index));
109             add(checkActive[index]);
110             
111             // Get the largest check box.
112             if(largestCheck == null) { largestCheck = checkActive[index]; }
113             else {
114                 if(largestCheck.getPreferredSize().width < checkActive[index].getPreferredSize().width) {
115                     largestCheck = checkActive[index];
116                 }
117             }
118         }
119         
120         // Create sets to store all possible entries for each crystal
121         // data field.
122         Set<Integer>      xIndexSet      = new HashSet<Integer>();
123         Set<Integer>      yIndexSet      = new HashSet<Integer>();
124         Set<Integer>      apdSet         = new HashSet<Integer>();
125         Set<Preamplifier> preampSet      = new HashSet<Preamplifier>();
126         Set<Integer>      ledChannelSet  = new HashSet<Integer>();
127         Set<Double>       ledDriverSet   = new HashSet<Double>();
128         Set<Integer>      fadcSlotSet    = new HashSet<Integer>();
129         Set<Integer>      fadcChannelSet = new HashSet<Integer>();
130         Set<Integer>      splitterSet    = new HashSet<Integer>();
131         Set<Integer>      hvGroupSet     = new HashSet<Integer>();
132         Set<Integer>      joutSet        = new HashSet<Integer>();
133         Set<Integer>      channelSet     = new HashSet<Integer>();
134         Set<Integer>      gainSet        = new HashSet<Integer>();
135         
136         // Iterate over all crystal data fields to obtain the set of
137         // all possible entries.
138         for(CrystalDataSet cds : manager) {
139             xIndexSet.add(cds.getCrystalXIndex());
140             yIndexSet.add(cds.getCrystalYIndex());
141             apdSet.add(cds.getAPDNumber());
142             preampSet.add(cds.getPreamplifierNumber());
143             ledChannelSet.add(cds.getLEDChannel());
144             ledDriverSet.add(cds.getLEDDriver());
145             fadcSlotSet.add(cds.getFADCSlot());
146             fadcChannelSet.add(cds.getFADCChannel());
147             splitterSet.add(cds.getSplitterNumber());
148             hvGroupSet.add(cds.getHighVoltageGroup());
149             joutSet.add(cds.getJout());
150             channelSet.add(cds.getChannel());
151             gainSet.add(cds.getGain());
152         }
153         
154         // Dump all of the set data into a list for each field.
155         Integer[]      xIndex      = xIndexSet.toArray(new Integer[xIndexSet.size()]);
156         Integer[]      yIndex      = yIndexSet.toArray(new Integer[yIndexSet.size()]);
157         Integer[]      apd         = apdSet.toArray(new Integer[apdSet.size()]);
158         Preamplifier[] preamp      = preampSet.toArray(new Preamplifier[preampSet.size()]);
159         Integer[]      ledChannel  = ledChannelSet.toArray(new Integer[ledChannelSet.size()]);
160         Double[]       ledDriver   = ledDriverSet.toArray(new Double[ledDriverSet.size()]);
161         Integer[]      fadcSlot    = fadcSlotSet.toArray(new Integer[fadcSlotSet.size()]);
162         Integer[]      fadcChannel = fadcChannelSet.toArray(new Integer[fadcChannelSet.size()]);
163         Integer[]      splitter    = splitterSet.toArray(new Integer[splitterSet.size()]);
164         Integer[]      hvGroup     = hvGroupSet.toArray(new Integer[hvGroupSet.size()]);
165         Integer[]      jout        = joutSet.toArray(new Integer[joutSet.size()]);
166         Integer[]      channel     = channelSet.toArray(new Integer[channelSet.size()]);
167         Integer[]      gain        = gainSet.toArray(new Integer[gainSet.size()]);
168         
169         // Generate the list of entries for the motherboard spinner.
170         //String[] motherboard = { "Top", "Bottom", "Left", "Right", "Top-Left", "Top-Right",
171         //        "Bottom-Left", "Bottom-Right" };
172         
173         // Sort all the lists in their natural order.
174         Arrays.sort(xIndex);
175         Arrays.sort(yIndex);
176         Arrays.sort(apd);
177         Arrays.sort(preamp);
178         Arrays.sort(ledChannel);
179         Arrays.sort(ledDriver);
180         Arrays.sort(fadcSlot);
181         Arrays.sort(fadcChannel);
182         Arrays.sort(splitter);
183         Arrays.sort(hvGroup);
184         Arrays.sort(jout);
185         //Arrays.sort(motherboard);
186         Arrays.sort(channel);
187         Arrays.sort(gain);
188         
189         // Instantiate each of the field combo boxes.
190         comboField = new JComboBox[INDICES];
191         comboField[INDEX_X]            = new JComboBox<Integer>(xIndex);
192         comboField[INDEX_Y]            = new JComboBox<Integer>(yIndex);
193         comboField[INDEX_APD]          = new JComboBox<Integer>(apd);
194         comboField[INDEX_PREAMP]       = new JComboBox<Preamplifier>(preamp);
195         comboField[INDEX_LED_CHANNEL]  = new JComboBox<Integer>(ledChannel);
196         comboField[INDEX_LED_DRIVER]   = new JComboBox<Double>(ledDriver);
197         comboField[INDEX_FADC_SLOT]    = new JComboBox<Integer>(fadcSlot);
198         comboField[INDEX_FADC_CHANNEL] = new JComboBox<Integer>(fadcChannel);
199         comboField[INDEX_SPLITTER]     = new JComboBox<Integer>(splitter);
200         comboField[INDEX_HV_GROUP]     = new JComboBox<Integer>(hvGroup);
201         comboField[INDEX_JOUT]         = new JComboBox<Integer>(jout);
202         comboField[INDEX_MOTHERBOARD]  = new JComboBox<Position>(Position.values());
203         comboField[INDEX_CHANNEL]      = new JComboBox<Integer>(channel);
204         comboField[INDEX_GAIN]         = new JComboBox<Integer>(gain);
205         
206         // Set the properties of the combo boxes.
207         for(JComboBox<?> combo : comboField) {
208             combo.setEnabled(false);
209             combo.setEditable(false);
210             add(combo);
211         }
212         
213         // Instantiate and set the properties of the apply button.
214         buttonApply = new JButton("Apply Filter");
215         buttonApply.setToolTipText("Highlight all crystals that match the selected filter.");
216         buttonApply.setEnabled(false);
217         add(buttonApply);
218         buttonApply.addActionListener(new ActionListener() {
219             @Override
220             public void actionPerformed(ActionEvent e) {
221                 // Apply the active filter.
222                 applyFilter();
223                 
224                 // Enable the remove filter button.
225                 buttonRemove.setEnabled(true);
226             }
227         });
228         
229         // Instantiate and set the properties of the remove button.
230         buttonRemove = new JButton("Remove Filter");
231         buttonRemove.setToolTipText("Clear all filter-related crystal highlighting.");
232         buttonRemove.setEnabled(false);
233         add(buttonRemove);
234         buttonRemove.addActionListener(new ActionListener() {
235             @Override
236             public void actionPerformed(ActionEvent e) {
237                 // Remove the list of filtered crystals.
238                 removeFilter();
239                 
240                 // Disable the remove filter button.
241                 buttonRemove.setEnabled(false);
242             }
243         });
244         
245         // Instantiate and set the properties of the remove button.
246         buttonClear = new JButton("Clear Filters");
247         buttonClear.setToolTipText("Reset all filters to the default and disable them.");
248         add(buttonClear);
249         buttonClear.addActionListener(new ActionListener() {
250             @Override
251             public void actionPerformed(ActionEvent e) {
252                 // Remove the list of filtered crystals.
253                 removeFilter();
254                 
255                 // Disable the remove filter button.
256                 buttonRemove.setEnabled(false);
257                 
258                 // Disable all of the filter check boxes.
259                 for(JCheckBox check : checkActive) {
260                     check.setSelected(false);
261                 }
262                 
263                 // Set the selected filter value for each filter to
264                 // the original value.
265                 for(JComboBox<?> combo : comboField) {
266                     combo.setSelectedIndex(0);
267                 }
268             }
269         });
270         
271         // Instantiate the filter status text area.
272         textStatus = new JTextArea();
273         textStatus.setEditable(false);
274         textStatus.setText("Active Filters: None");
275         textStatus.setFont(new Font(Font.MONOSPACED, getFont().getStyle(), getFont().getSize()));
276         JScrollPane textScroller = new JScrollPane(textStatus);
277         add(textScroller);
278         
279         // Determine if the check boxes or the spinners have a larger
280         // preferred height.
281         boolean checkLarger = true;
282         if(checkActive[INDEX_X].getPreferredSize().height < comboField[INDEX_X].getPreferredSize().height) {
283             checkLarger = false;
284         }
285         
286         // Place the filter components onto the layout.
287         for(int index = 0; index < INDICES; index++) {
288             // The first component needs to use the top of the panel
289             // for its upper border.
290             if(index == 0) {
291                 layout.putConstraint(SpringLayout.NORTH, checkActive[index], vexternal, SpringLayout.NORTH, this);
292                 layout.putConstraint(SpringLayout.NORTH, comboField[index], vexternal, SpringLayout.NORTH, this);
293             }
294             // Subsequent panels are locked to the previous entry's
295             // lower border.
296             else {
297                 if(checkLarger) {
298                     layout.putConstraint(SpringLayout.NORTH, checkActive[index], vinternal, SpringLayout.SOUTH, checkActive[index - 1]);
299                     layout.putConstraint(SpringLayout.NORTH, comboField[index], vinternal, SpringLayout.SOUTH, checkActive[index - 1]);
300                 }
301                 else {
302                     layout.putConstraint(SpringLayout.NORTH, checkActive[index], vinternal, SpringLayout.SOUTH, comboField[index - 1]);
303                     layout.putConstraint(SpringLayout.NORTH, comboField[index], vinternal, SpringLayout.SOUTH, comboField[index - 1]);
304                 }
305             }
306             
307             // All check boxes have left borders locked to the left side
308             // of the panel. All spinners have left borders locked to
309             // the right border of the widest check box. Finally, all of
310             // the spinners have right borders locked the right side of
311             // the panel.
312             layout.putConstraint(SpringLayout.WEST, checkActive[index], hexternal, SpringLayout.WEST, this);
313             layout.putConstraint(SpringLayout.WEST, comboField[index], hinternal, SpringLayout.EAST, largestCheck);
314             layout.putConstraint(SpringLayout.EAST, comboField[index], -hexternal, SpringLayout.EAST, this);
315         }
316         
317         // Place the status text area.
318         JComponent previous = comboField[INDICES - 1];
319         if(checkLarger) { previous = checkActive[INDICES - 1]; }
320         layout.putConstraint(SpringLayout.NORTH, textScroller, vexternal, SpringLayout.SOUTH, previous);
321         layout.putConstraint(SpringLayout.WEST, textScroller, hexternal, SpringLayout.WEST, this);
322         layout.putConstraint(SpringLayout.EAST, textScroller, -hexternal, SpringLayout.EAST, this);
323         layout.putConstraint(SpringLayout.SOUTH, textScroller, -hexternal, SpringLayout.NORTH, buttonApply);
324         
325         // Place the command buttons onto the layout.
326         layout.putConstraint(SpringLayout.SOUTH, buttonApply, -vexternal, SpringLayout.SOUTH, this);
327         layout.putConstraint(SpringLayout.WEST, buttonApply, hexternal, SpringLayout.WEST, this);
328         layout.putConstraint(SpringLayout.SOUTH, buttonRemove, -vexternal, SpringLayout.SOUTH, this);
329         layout.putConstraint(SpringLayout.WEST, buttonRemove, hinternal, SpringLayout.EAST, buttonApply);
330         layout.putConstraint(SpringLayout.SOUTH, buttonClear, -vexternal, SpringLayout.SOUTH, this);
331         layout.putConstraint(SpringLayout.WEST, buttonClear, hinternal, SpringLayout.EAST, buttonRemove);
332     }

361     @Override
362     public Dimension getPreferredSize() {
363         // If there is a user defined preferred size, use that.
364         if(preferredSize != null) { return preferredSize; }
365         
366         // Track the minimum size needed to display the panel.
367         int preferredHeight =  4 * vexternal;
368         
369         // Determine if the combo boxes or check boxes are larger.
370         int checkHeight = checkActive[INDEX_X].getPreferredSize().height;
371         int comboHeight = comboField[INDEX_X].getPreferredSize().height;
372         boolean checkLarger = checkHeight > comboHeight;
373         int compHeight = checkLarger ? checkHeight : comboHeight;
374         
375         // Get the preferred height of the filters.
376         preferredHeight += (INDICES - 1) * (vinternal + compHeight);
377         
378         // Add the button preferred height.
379         preferredHeight += buttonApply.getPreferredSize().height;
380         
381         // Add 50 for the text area.
382         preferredHeight += 150;
383         
384         // The preferred width is the width needed by the buttons.
385         int preferredWidth = hexternal + hexternal + hinternal + hinternal;
386         preferredWidth += buttonApply.getPreferredSize().width;
387         preferredWidth += buttonRemove.getPreferredSize().width;
388         preferredWidth += buttonClear.getPreferredSize().width;
389         
390         // Return the result.
391         return new Dimension(preferredWidth, preferredHeight);
392     }

415     /**
416      * Applies the filter defined by the currently selected filter
417      * options. If no filter options are selected, this does nothing.
418      */
419     private void applyFilter() {
420         // Make sure that at least one filter option is selected.
421         boolean isEnabled = false;
422         activeLoop:
423         for(JCheckBox check : checkActive) {
424             if(check.isEnabled()) {
425                 isEnabled = true;
426                 break activeLoop;
427             }
428         }
429         if(!isEnabled) { return; }
430         
431         // Clear the filter parameters.
432         cdsCompare.clearFilters();
433         
434         // Store the textual form of the active filters.
435         StringBuffer filterBuffer = new StringBuffer("Active Filters:\n");
436         
437         // Set the filter parameters.
438         if(checkActive[INDEX_X].isSelected()) {
439             int val = (Integer) comboField[INDEX_X].getSelectedItem();
440             cdsCompare.addXIndexFilter(val);
441             filterBuffer.append(String.format("    %-20s :: %d%n", "x-Index", val));
442         }
443         if(checkActive[INDEX_Y].isSelected()) {
444             int val = (Integer) comboField[INDEX_Y].getSelectedItem();
445             cdsCompare.addYIndexFilter(val);
446             filterBuffer.append(String.format("    %-20s :: %d%n", "y-Index", val));
447         }
448         if(checkActive[INDEX_APD].isSelected()) {
449             int val = (Integer) comboField[INDEX_APD].getSelectedItem();
450             cdsCompare.addAPDFilter(val);
451             filterBuffer.append(String.format("    %-20s :: %d%n", "APD Number", val));
452         }
453         if(checkActive[INDEX_PREAMP].isSelected()) {
454             Preamplifier val = (Preamplifier) comboField[INDEX_PREAMP].getSelectedItem();
455             cdsCompare.addPreampFilter((Preamplifier) comboField[INDEX_PREAMP].getSelectedItem());
456             filterBuffer.append(String.format("    %-20s :: %s%n", "APD Number", val.toString()));
457         }
458         if(checkActive[INDEX_LED_CHANNEL].isSelected()) {
459             int val = (Integer) comboField[INDEX_LED_CHANNEL].getSelectedItem();
460             cdsCompare.addLEDChannelFilter(val);
461             filterBuffer.append(String.format("    %-20s :: %d%n", "LED Channel Number", val));
462         }
463         if(checkActive[INDEX_LED_DRIVER].isSelected()) {
464             double val = (Double) comboField[INDEX_LED_DRIVER].getSelectedItem();
465             cdsCompare.addLEDDriverFilter(val);
466             filterBuffer.append(String.format("    %-20s :: %.1f%n", "LED Driver", val));
467         }
468         if(checkActive[INDEX_FADC_SLOT].isSelected()) {
469             int val = (Integer) comboField[INDEX_FADC_SLOT].getSelectedItem();
470             cdsCompare.addFADCSlotFilter(val);
471             filterBuffer.append(String.format("    %-20s :: %d%n", "FADC Slot Number", val));
472         }
473         if(checkActive[INDEX_FADC_CHANNEL].isSelected()) {
474             int val = (Integer) comboField[INDEX_FADC_CHANNEL].getSelectedItem();
475             cdsCompare.addFADCChannelFilter(val);
476             filterBuffer.append(String.format("    %-20s :: %d%n", "FADC Channel Number", val));
477         }
478         if(checkActive[INDEX_SPLITTER].isSelected()) {
479             int val = (Integer) comboField[INDEX_SPLITTER].getSelectedItem();
480             cdsCompare.addSplitterFilter(val);
481             filterBuffer.append(String.format("    %-20s :: %d%n", "Splitter Number", val));
482         }
483         if(checkActive[INDEX_HV_GROUP].isSelected()) {
484             int val = (Integer) comboField[INDEX_HV_GROUP].getSelectedItem();
485             cdsCompare.addHVGroupFilter(val);
486             filterBuffer.append(String.format("    %-20s :: %d%n", "High Voltage Group", val));
487         }
488         if(checkActive[INDEX_JOUT].isSelected()) {
489             int val = (Integer) comboField[INDEX_JOUT].getSelectedItem();
490             cdsCompare.addJoutFilter(val);
491             filterBuffer.append(String.format("    %-20s :: %d%n", "Jout", val));
492         }
493         if(checkActive[INDEX_MOTHERBOARD].isSelected()) {
494             Position val = (Position) comboField[INDEX_MOTHERBOARD].getSelectedItem();
495             cdsCompare.addMotherboardFilter((Position) comboField[INDEX_MOTHERBOARD].getSelectedItem());
496             filterBuffer.append(String.format("    %-20s :: %s%n", "Motherboard Position", val.toString()));
497         }
498         if(checkActive[INDEX_CHANNEL].isSelected()) {
499             int val = (Integer) comboField[INDEX_CHANNEL].getSelectedItem();
500             cdsCompare.addChannelFilter(val);
501             filterBuffer.append(String.format("    %-20s :: %d%n", "Channel Number", val));
502         }
503         if(checkActive[INDEX_GAIN].isSelected()) {
504             int val = (Integer) comboField[INDEX_GAIN].getSelectedItem();
505             cdsCompare.addGainFilter(val);
506             filterBuffer.append(String.format("    %-20s :: %d%n", "Gain", val));
507         }
508         
509         // Update the status pane.
510         textStatus.setText(filterBuffer.toString());
511         
512         // Populate the filtered crystals list.
513         filteredList = new ArrayList<Point>();
514         for(CrystalDataSet cds : manager) {
515             if(cdsCompare.passesFilter(cds)) { filteredList.add(cds.getCrystalIndex()); }
516         }
517         
518         // Throw an event to indicate that a filter was applied.
519         ActionEvent event = new ActionEvent(this, EVENT_FILTER_APPLIED, "Filter Applied");
520         for(ActionListener lst : alList) {
521             lst.actionPerformed(event);
522         }
523     }

525     /**
526      * Removes any active filters.
527      */
528     private void removeFilter() {
529         // Make sure that there is an active filter to remove.
530         if(filteredList == null) { return; }
531         
532         // Clear the filter list.
533         filteredList = null;
534         
535         // Update the status panel text.
536         textStatus.setText("Active Filters: None");
537         
538         // Throw an event to indicate that a filter was removed.
539         ActionEvent event = new ActionEvent(this, EVENT_FILTER_REMOVED, "Filter Removed");
540         for(ActionListener lst : alList) {
541             lst.actionPerformed(event);
542         }
543     }

551         TOP(true, null), BOTTOM(false, null), LEFT(null, true), RIGHT(null, false),
552         TOP_LEFT(true, true), TOP_RIGHT(true, false), BOTTOM_LEFT(false, true), BOTTOM_RIGHT(false, false);

554         // Internal variables.
555         private String name = null;
556         private final Boolean isTop;
557         private final Boolean isLeft;

559         /**
560          * Instantiates a new position.
561          * @param isTop - <code>true</code> if the position is on the
562          * top and <code>false</code> if it is on the bottom. <code>
563          * null</code> indicates that it does not differentiate between
564          * this axis.
565          * @param isLeft - <code>true</code> if the position is on the
566          * left and <code>false</code> if it is on the right. <code>
567          * null</code> indicates that it does not differentiate between
568          * this axis.
569          */
570         private Position(Boolean isTop, Boolean isLeft) {
571             this.isTop = isTop;
572             this.isLeft = isLeft;
573         }
574         
575         /**
576          * Checks whether a given motherboard position is consistent
577          * with the position enumerable.
578          * @param mb - The motherboard to check.
579          * @return Returns <code>true</code> if the motherboard position
580          * is consistent and <code>false</code> otherwise.
581          */
582         public boolean matchesPosition(Motherboard mb) {
583             // If this position needs to check top or bottom status,
584             // do so. Return false if it fails.
585             if(isTop != null && isTop != mb.isTop()) {
586                 return false;
587             }
588             
589             // If this position needs to check left or right status,
590             // do so. Return false if it fails.
591             if(isLeft != null && isLeft != mb.isLeft()) {
592                 return false;
593             }
594             
595             // If it passes all checks, return true.
596             return true;
597         }
598         
599         @Override
600         public String toString() {
601             // If the name has already been defined, return it.
602             if(name != null) { return name; }
603             
604             // Otherwise, the name must be parsed from the system name.
605             // Create a string buffer to contain the parsed characters.
606             StringBuffer nameBuffer = new StringBuffer();
607             
608             // Iterate over the characters in the name and process them.
609             boolean nextUpperCase = true;
610             for(char c : name().toCharArray()) {
611                 if(c == '_') {
612                     nameBuffer.append(' ');
613                     nextUpperCase = true;
614                 }
615                 else if(nextUpperCase) {
616                     nameBuffer.append(c);
617                     nextUpperCase = false;
618                 }
619                 else { nameBuffer.append(Character.toLowerCase(c)); }
620             }
621             
622             // Store the result.
623             name = nameBuffer.toString();
624             
625             // Return the result.
626             return name;
627         }

631      * Class <code>CheckListener</code> is an implementation of <code>
632      * ItemListener</code> that is linked to a specific component index
633      * and handles the enabling and disabling of filters.
634      * 
635      * @see ItemListener
636      */
637     private class CheckListener implements ItemListener {
638         // Store the index of the components associated with the listener.
639         private final int index;
640         
641         /**
642          * Instantiates a new <code>CheckListener</code> that is linked
643          * to the specified index.
644          * @param componentIndex - The index of the component.
645          */
646         public CheckListener(int componentIndex) { index = componentIndex; }
647         
648         @Override
649         public void itemStateChanged(ItemEvent e) {
650             // Set the component properties according the selection
651             // status of the check box.
652             if(checkActive[index].isSelected()) {
653                 checkActive[index].setForeground(COLOR_CHECK_FONT_ACTIVE);
654                 comboField[index].setEnabled(true);
655             }
656             else {
657                 checkActive[index].setForeground(COLOR_CHECK_FONT_INACTIVE);
658                 comboField[index].setEnabled(false);
659             }
660             
661             // If any of the filter check boxes are active, the active
662             // filter button should be clickable.
663             boolean active = false;
664             applyCheckLoop:
665             for(JCheckBox check : checkActive) {
666                 if(check.isSelected()) {
667                     active = true;
668                     break applyCheckLoop;
669                 }
670             }
671             
672             // Set the apply button to the appropriate setting.
673             buttonApply.setEnabled(active);
674         }

805         /**
806          * Checks to see if a given <code>CrystalDataSet</code> object
807          * matches all of the defined filter parameters.
808          * @param cds - The data set to check.
809          * @return Returns <code>true</code> if the <code>CrystalDataSet
810          * </code> matches the defined filter parameters and <code>false
811          * </code> otherwise.
812          */
813         public boolean passesFilter(CrystalDataSet cds) {
814             // Apply the filter for the crystal x-index.
815             if(filterX != null) {
816                 if(cds.getCrystalXIndex() != filterX) { return false; }
817             }
818             
819             // Apply the filter for the crystal y-index.
820             if(filterY != null) {
821                 if(cds.getCrystalYIndex() != filterY) { return false; }
822             }
823             
824             // Apply the filter for the crystal APD number.
825             if(filterAPD != null) {
826                 if(cds.getAPDNumber() != filterAPD) { return false; }
827             }
828             
829             // Apply the filter for the associated preamplifier.
830             if(filterPreamp != null) {
831                 if(cds.getPreamplifierNumber() != filterPreamp) { return false; }
832             }
833             
834             // Apply the filter for the crystal LED channel number.
835             if(filterLEDChannel != null) {
836                 if(cds.getLEDChannel() != filterLEDChannel) { return false; }
837             }
838             
839             // Apply the filter for the crystal LED driver.
840             if(filterLEDDriver != null) {
841                 if(cds.getLEDDriver() != filterLEDDriver) { return false; }
842             }
843             
844             // Apply the filter for the crystal FADC slot.
845             if(filterFADCSlot != null) {
846                 if(cds.getFADCSlot() != filterFADCSlot) { return false; }
847             }
848             
849             // Apply the filter for the crystal FADC channel.
850             if(filterFADCChannel != null) {
851                 if(cds.getFADCChannel() != filterFADCChannel) { return false; }
852             }
853             
854             // Apply the filter for the crystal splitter number.
855             if(filterSplitter != null) {
856                 if(cds.getSplitterNumber() != filterSplitter) { return false; }
857             }
858             
859             // Apply the filter for the crystal high voltage group.
860             if(filterHVGroup != null) {
861                 if(cds.getHighVoltageGroup() != filterHVGroup) { return false; }
862             }
863             
864             // Apply the filter for the Jout property.
865             if(filterJout != null) {
866                 if(cds.getJout() != filterJout) { return false; }
867             }
868             
869             // Apply the filter for the motherboard position.
870             if(motherboard != null) {
871                 if(!motherboard.matchesPosition(cds.getMotherboard())) { return false; }
872             }
873             
874             // Apply the filter for the crystal channel number.
875             if(filterChannel != null) {
876                 if(cds.getChannel() != filterChannel) { return false; }
877             }
878             
879             // Apply the filter for the crystal gain.
880             if(filterGain != null) {
881                 if(cds.getGain() != filterGain) { return false; }
882             }
883             
884             // If all filters succeeded, return true.
885             return true;
886         }
