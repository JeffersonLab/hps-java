3   import java.awt.Component;

5   import java.awt.Graphics;

7   import javax.swing.JLabel;

10  /**
11   * Class <code>StatusPanel</code> displays text in a set of fields.
12   * This class is being phased out in favor of <code>ResizableFieldPanel
13   * </code> and should no longer be used. It will be removed next update.
14   */
15  @Deprecated
16  public class StatusPanel extends JPanel {
17      private static final long serialVersionUID = -8353479383875379010L;
18      // The panel that displays behind the status field.
19      private BackPanel background = new BackPanel();
20      // The status fields. The first array index represents which status
21      // field and the second is always of size two, with index 0 mapping
22      // to the label that displays the field name and index 1 mapping to
23      // the label displaying the field value.
24      private JLabel[][] field;
25      // Spacing variables for panel layout.
26      private int leftBuffer = 10;
27      private int upperBuffer = 10;
28      
29      /**
30       * A <code>String</code> representing the default value to be
31       * displayed on the status panel whenever there is no value for
32       * that field.
33       */
34      static final String NULL_VALUE = "---";
35      
36      /**
37       * Creates a new status panel with display fields with the indicated
38       * names. They will be assigned a field index in the order that they
39       * are given starting with zero.
40       * @param fieldName - The names of the fields to display.
41       */
42      public StatusPanel(String... fieldName) {
43          // Initialize the component.
44          super();
45          
46          // Set the layout manager to manual.
47          setLayout(null);
48          
49          // Build the text fields.
50          int curZ = 0;
51          field = new JLabel[fieldName.length][2];
52          for(int i = 0; i < field.length; i++) {
53              for(int j = 0; j < field[i].length; j++) {
54                  field[i][j] = new JLabel();
55                  field[i][j].setOpaque(true);
56                  field[i][j].setBackground(Color.WHITE);
57                  add(field[i][j]);
58                  setComponentZOrder(field[i][j], curZ);
59                  curZ++;
60              }
61              field[i][0].setText(fieldName[i] + ":   ");
62              field[i][0].setHorizontalAlignment(JLabel.RIGHT);
63          }
64          
65          // Start the fields as null by default.
66          clearValues();
67          
68          // Build the background panel.
69          add(background);
70          setComponentZOrder(background, curZ);
71      }
72      
73      /**
74       * Sets all of the fields on the status display to the null value.
75       */
76      public void clearValues() {
77          for(int i = 0; i < field.length; i++) {
78              field[i][1].setText(NULL_VALUE);
79          }
80      }
81      
82      /**
83       * Sets the value of the indicated field.
84       * @param index - The field's index.
85       * @param value - The new value to display.
86       * @throws IndexOutOfBoundsException Occurs when the field index
87       * is neither more than the existing number of fields or is negative.
88       */
89      public void setFieldValue(int index, String value) throws IndexOutOfBoundsException {
90          if(index >= 0 && index < field.length) {
91              if(value == null) { field[index][1].setText(NULL_VALUE); }
92              else  { field[index][1].setText(value); }
93          }
94          else { throw new IndexOutOfBoundsException("Invalid field index."); }
95      }
96      
97      @Override
98      public void setSize(int width, int height) {
99          super.setSize(width, height);
100         resize();
101     }
102     
103     @Override
104     public void setSize(Dimension d) {
105         super.setSize(d);
106         resize();
107     }
108     
109     /**
110      * Finds the x-coordinate immediately after the component.
111      * @param c - The component of which to find the end.
112      * @return Returns the x-coordinate at the end of the component. 
113      */
114     private final static int getNextX(Component c) { return getNextX(c, 0); }
115     
116     /**
117     /**
118      * Finds the x-coordinate after the component with a given buffer.
119      * @param c - The component of which to find the end.
120      * @param buffer - The extra space after the component to be included.
121      * @return Returns the x-coordinate at the end of the component,
122      * with a buffer length.
123      */
124     private final static int getNextX(Component c, int buffer) {
125         return c.getX() + c.getWidth() + buffer;
126     }
127     
128     /**
129      * Updates the layout of the component to the panel's current size.
130      */
131     private void resize() {
132         // Define the width an height as convenience variables.
133         int width = getWidth();
134         int height = getHeight();
135         
136         // Size the background panel.
137         background.setBounds(0, 0, width, height);
138         
139         // Size and place the text labels.
140         if(field.length != 0) {
141             int labelHeight = (height - (int)(upperBuffer + 5)) / 3;
142             int labelRem = (height - upperBuffer - 8) % field.length;
143             int curX = leftBuffer;
144             int curY = (int)(upperBuffer + 2);
145             for(int i = 0; i < field.length; i++) {
146                 // Determine the appropriate field height.
147                 int thisHeight = labelHeight;
148                 if(labelRem > 0) {
149                     thisHeight++;
150                     labelRem--;
151                 }
152                 
153                 // Place the field.
154                 field[i][0].setBounds(curX, curY, 130, thisHeight);
155                 field[i][1].setBounds(getNextX(field[i][0]), curY, 75, thisHeight);
156                 
157                 // If we have written three labels, then start a new column.
158                 if(i % 3 == 2) {
159                     curX = getNextX(field[i][1], 10);
160                     curY = (int)(upperBuffer + 2);
161                 }
162                 
163                 // Otherwise just increment the current height.
164                 else { curY += thisHeight; }
165             }
166         }
167     }
168     
169     /**
170      * Class <code>BackPanel</code> simply renders the background panel
171      * for the status panel.
172      */
173     private class BackPanel extends JPanel {
174         private static final long serialVersionUID = 4997805650267243080L;
175 
176         public void paint(Graphics g) {
177             // Render the panel background.
178             g.setColor(Color.WHITE);
179             g.fillRect(0, upperBuffer, getWidth(), getHeight() - upperBuffer);
180             g.setColor(Color.GRAY);
181             g.drawRect(0, upperBuffer, getWidth() - 1, getHeight() - upperBuffer - 1);
182             g.setColor(Color.LIGHT_GRAY);
183             g.drawRect(1, upperBuffer + 1, getWidth() - 3, getHeight() - upperBuffer - 3);
184         }
185     }
