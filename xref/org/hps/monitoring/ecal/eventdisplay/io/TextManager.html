3   import java.io.File;
4   import java.io.IOException;
5   import java.util.ArrayList;
6   import java.util.StringTokenizer;
7   
8   import org.hps.monitoring.ecal.eventdisplay.event.Cluster;
9   import org.hps.monitoring.ecal.eventdisplay.event.EcalHit;
10  
11  /**
12   * The class <code>TextManager</code> handles loading hits and clusters from a
13   * text file to populate the calorimeter panel. Input should be of the form
14   * Event [Number]
15   * Indicates the start of a new event.
16   * 
17   * EcalHit [X] [Y] [Energy]
18   * Represents a calorimeter hit at coordinates ([X], [Y]) and with energy
19   * [Energy]. Coordinates should be in calorimeter form (x = [-23, 23] and
20   * y = [-5, 5]) and must integers. Energy can be a decimal value. Brackets
21   * should not be included in the line.
22   * 
23   * Cluster [X] [Y] [Energy]
24   * Represents the location of a cluster at coordinates ([X], [Y]) with
25   * [Energy] total cluster energy. Note that the [Energy] field is not
26   * required. Brackets should not be included in the line.
27   * 
28   * CompHit [X] [Y]
29   * Represents a component hit of the previously declared cluster and located
30   * at coordinates ([X], [Y]). Brackets should not be included in the line.
31   * 
32   * SharHit [X] [Y]
33   * Represents a hit that is shared between two or more clusters which is
34   * located at coordinates ([X], [Y]). Brackets should not be included in
35   * the line.
36   * 
37   * EndEvent
38   * Indicates that the event has ended.
39   */
40  public final class TextManager implements EventManager {
41      // File reader for reading the input.
42      private AdvancedReader reader;
43      // List for storing the hits from the current event.
44      private ArrayList<EcalHit> hitList = new ArrayList<EcalHit>();
45      // List for storing the clusters from the current hit.
46      private ArrayList<Cluster> clusterList = new ArrayList<Cluster>();
47      // Whether the event manager has an open file.
48      private boolean open = true;
49      // Track the current event number.
50      private int curEvent = 0;
51      
52      /**
53       * Initializes an event manager that will read from the indicated file.
54       * @param file - The path to the file containing hit information.
55       */
56      public TextManager(File file) throws IOException {
57          reader = new AdvancedReader(file);
58      }
59      
60      /**
61       * Initializes an event manager that will read from the indicated file.
62       * @param filename - The path to the file containing hit information.
63       */
64      public TextManager(String filename) throws IOException {
65          reader = new AdvancedReader(filename);
66      }
67      
68      @Override
69      public void close() throws IOException {
70          reader.close();
71          open = false;
72      }
73      
74      @Override
75      public int getEventNumber() {
76          return curEvent;
77      }
78      
79      @Override
80      public ArrayList<Cluster> getClusters() {
81          if (!open) { return null; }
82          else { return clusterList; }
83      }
84      
85      @Override
86      public ArrayList<EcalHit> getHits() {
87          if (!open) { return null; }
88          else { return hitList; }
89      }
90      
91      @Override
92      public boolean nextEvent() throws IOException {
93          // We can only read of the reader is open.
94          if (!open) { return false; }
95          
96          // Store the current line.
97          String curLine = reader.readNextLine();
98          
99          // Keep sorting until we hit a null or an event header.
100         while (curLine != null && !curLine.contains("Event")) {
101             curLine = reader.readNextLine();
102         }
103         
104         // If we hit a null, we are at the end of the file.
105         if (curLine == null) { return false; }
106         
107         // Clear the data lists.
108         hitList = new ArrayList<EcalHit>();
109         clusterList = new ArrayList<Cluster>();
110         
111         // Get the current event.
112         StringTokenizer et = new StringTokenizer(curLine);
113         et.nextToken();
114         curEvent = Integer.parseInt(et.nextToken());
115         
116         // Otherwise, we have read an event header and must populate
117         // the data lists.
118         curLine = reader.readNextLine();
119         while (curLine != null && curLine.compareTo("EndEvent") != 0) {
120             // Break apart the line.
121             StringTokenizer st = new StringTokenizer(curLine);
122             String name = st.nextToken();
123             int ix = Integer.parseInt(st.nextToken());
124             int iy = Integer.parseInt(st.nextToken());
125             
126             // If this is a cluster, add a new cluster object.
127             if (name.compareTo("Cluster") == 0) {
128                 // Get the cluster energy, if it is given.
129                 double clusterEnergy = Double.NaN;
130                 if(st.hasMoreTokens()) { clusterEnergy = Double.parseDouble(st.nextToken()); }
131                 
132                 // Get the cluster time, if it is defined.
133                 double clusterTime = Double.NaN;
134                 if(st.hasMoreTokens()) { clusterTime = Double.parseDouble(st.nextToken()); }
135                 
136                 // Add a new cluster.
137                 clusterList.add(new Cluster(ix, iy, clusterEnergy, clusterTime));
138             }
139             
140             // If this is a calorimeter hit, add a new calorimeter hit object.
141             else if (name.compareTo("EcalHit") == 0) {
142                 double energy = Double.parseDouble(st.nextToken());
143                 double time = Double.parseDouble(st.nextToken());
144                 hitList.add(new EcalHit(ix, iy, energy, time));
145             }
146             
147             // If this is a cluster component hit, add it to the last cluster.
148             else if(name.compareTo("CompHit") == 0) {
149                 // There must be a last cluster to process this hit type.
150                 if(clusterList.size() == 0) {
151                     System.err.println("File Format Error: A cluster component hit was read, but" +
152                             " no cluster has been declared. Terminating.");
153                     System.exit(1);
154                 }
155                 else { clusterList.get(clusterList.size() - 1).addComponentHit(ix, iy); }
156             }
157             
158             // If this is a cluster shared hit, add it to the last cluster.
159             else if(name.compareTo("SharHit") == 0) {
160                 // There must be a last cluster to process this hit type.
161                 if(clusterList.size() == 0) {
162                     System.err.println("File Format Error: A cluster shared hit was read, but" +
163                             " no cluster has been declared. Terminating.");
164                     System.exit(1);
165                 }
166                 else { clusterList.get(clusterList.size() - 1).addSharedHit(ix, iy); }
167             }
168             
169             // Get the next line.
170             curLine = reader.readNextLine();
171         }
172         
173         // Indicate that an event was processed.
174         return true;
175     }
176     
177     @Override
178     public boolean previousEvent() throws IOException {
179         // If we are at the first event, do nothing. There is no
180         // previous event to display.
181         if(curEvent == 1) { return false; }
182         
183         // Otherwise, loop backward until we find the previous event header.
184         String curLine;
185         while(true) {
186             // Get the previous line.
187             curLine = reader.readPreviousLine();
188             
189             // Otherwise, if it is null, we've reached the start of the file.
190             if(curLine == null) {
191                 System.err.println("Error: Unexpectedly reached SOF.");
192                 System.exit(1);
193             }
194             
195             // If the previous line is an event, note it.
196             if(curLine.substring(0, 5).compareTo("Event") == 0) {
197                 // Get the event number of the current event.
198                 StringTokenizer et = new StringTokenizer(curLine);
199                 et.nextToken();
200                 int readEvent = Integer.parseInt(et.nextToken());
201                 
202                 // If the read event number is one back from the current
203                 // event, jump back a step and read the event.
204                 if(readEvent == (curEvent - 1)) {
205                     reader.readPreviousLine();
206                     return nextEvent();
207                 }
208             }
209         }
210     }
