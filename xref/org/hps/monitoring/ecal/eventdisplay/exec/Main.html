1   package org.hps.monitoring.ecal.eventdisplay.exec;

3   import java.awt.GraphicsDevice;
4   import java.awt.GraphicsEnvironment;
5   import java.io.File;
6   import java.io.IOException;
7   
8   import org.hps.monitoring.ecal.eventdisplay.ui.FileViewer;
9   import org.hps.monitoring.ecal.eventdisplay.ui.DataFileViewer;
10  
11  /**
12   * The class <code>Main</code> can be used to create an event display
13   * that reads from file. The input file can be set with the "-i" command,
14   * though any file can be loaded from the file menu after initialization.
15   * Calorimeter wiring information can be loaded with the"-w" command.
16   */
17  public class Main {
18      public static void main(String[] args) throws IOException {
19          // If "-h" was given as a command line argument, print the
20          // help text.
21          for(String s : args) {
22              if(s.compareTo("-h") == 0 || s.compareTo("--help") == 0) {
23                  System.out.println("HPS Event Display");
24                  System.out.println("Options:");
25                  System.out.printf("%-4s%-12s%s%n", "-h", "--help", "Display help text.");
26                  System.out.printf("%-4s%-12s%s%n", "-i", "--input", "Defines the input file.");
27                  System.out.printf("%-4s%-12s%s%n", "-w", "--wiring", "Defines the wiring data file.");
28                  System.exit(0);
29              }
30          }
31          
32          // If -i or --input was given, set the input file.
33          File dataSource = null;
34          String filepath = null;
35          for(int i = 0; i < args.length; i++) {
36              if(args[i].compareTo("-i") == 0 || args[i].compareTo("--input") == 0) {
37                  if(args.length <= i + 1) { System.out.println("Error: Expected additional argument."); }
38                  else { filepath = args[i + 1]; }
39              }
40          }
41          if(filepath != null) { dataSource = new File(filepath); }
42          
43          // If -w or --wiring was given, set the input file.
44          String wiringPath = null;
45          for(int i = 0; i < args.length; i++) {
46              if(args[i].compareTo("-w") == 0 || args[i].compareTo("--witing") == 0) {
47                  if(args.length <= i + 1) { System.out.println("Error: Expected additional argument."); }
48                  else { wiringPath = args[i + 1]; }
49              }
50          }
51          if(filepath != null) { dataSource = new File(filepath); }
52          
53          // Define the viewer.
54          FileViewer window;
55          if(wiringPath != null) { window = new DataFileViewer(dataSource, wiringPath); }
56          else { window = new FileViewer(dataSource); }
57          
58          // Get screen size of primary monitor
59          GraphicsDevice gd = GraphicsEnvironment.getLocalGraphicsEnvironment().getDefaultScreenDevice();
60          int screenWidth = gd.getDisplayMode().getWidth();
61          int screenHeight = gd.getDisplayMode().getHeight();
62          
63          // Set the viewer window location and make it visible.
64          window.setLocation((screenWidth - window.getPreferredSize().width) / 2,
65                  (screenHeight - window.getPreferredSize().height) / 2);
66          window.setVisible(true);
67      }
68  }
