12  /**
13   * Class <code>LCIOBridgeDriver</code> displays LCIO events on the
14   * event display.
15   *
16   */
17  public class LCIOBridgeDriver extends Driver {
18  
19      // The identification name for getting the calorimeter object.
20      String ecalName;
21      // The collection name for the calorimeter hits.
22      String ecalCollectionName;
23      // The collection name in which to store clusters.
24      String clusterCollectionName = "EcalClusters";
25      // Sets how many events should pass before the display updates.
26      int displayInterval = 0;
27      // Whether the event display is currently processing an event or not.
28      private boolean updating = false;
29      // How many events where processed from the last displayed event.
30      private int eventsProcessed = 0;
31      // The event display.
32      private PEventViewer eventDisplay = new PEventViewer();
33      
34      /**
35       * <b>process</b><br/><br/>
36       * <code>public void <b>process</b>(EventHeader event)</code><br/><br/>
37       * Converts an LCIO event into an event display compatible format.
38       * Additionally handles updating the event display, if appropriate.
39       * @param event - The LCIO event.
40       */
41      public void process(EventHeader event) {
42          // If we are still updating the display, skip this event.
43          if(updating) { return; }
44          
45          // Make sure that this event has calorimeter hits.
46          if (event.hasCollection(CalorimeterHit.class, ecalCollectionName)) {
47              // Get the list of calorimeter hits from the event.
48              List<CalorimeterHit> hits = event.get(CalorimeterHit.class, ecalCollectionName);
49              
50              // Define a list of clusters from the event.
51              List<org.lcsim.event.Cluster> clusters = event.get(org.lcsim.event.Cluster.class, clusterCollectionName);
52              
53              // Increment the number of events we have seen.
54              eventsProcessed++;
55              
56              // If this is the correct place to update, do so.
57              if(eventsProcessed >= displayInterval) {
58                  // Lock the update method for the duration of the update.
59                  updating = true;
60                  
61                  // Clear the event display.
62                  eventDisplay.resetDisplay();
63                  
64                  // Add all of the hits.
65                  for(CalorimeterHit hit : hits) {
66                      // Get the hit's location and energy.
67                      int ix = hit.getIdentifierFieldValue("ix");
68                      int iy = hit.getIdentifierFieldValue("iy");
69                      double energy = hit.getRawEnergy();
70                      
71                      // Add the hit energy to the event display.
72                      eventDisplay.addHit(new EcalHit(ix, iy, energy));
73                  }
74                  
75                  // Add all the clusters.
76                  for(org.lcsim.event.Cluster cluster : clusters) {
77                      // Get the seed hit.
78                      CalorimeterHit seed = cluster.getCalorimeterHits().get(0);
79                      int ix = seed.getIdentifierFieldValue("ix");
80                      int iy = seed.getIdentifierFieldValue("iy");
81                      double energy = seed.getRawEnergy(); // FIXME: Should this be getCorrectedEnergy() instead? --JM
82                      
83                      // Add the cluster center to the event display.
84                      Cluster cc = new Cluster(ix, iy, energy);
85                      eventDisplay.addCluster(cc);
86                  }
87                  
88                  // Update the display.
89                  eventDisplay.updateDisplay();
90                  
91                  // Reset the number of events we've seen since the last update.
92                  eventsProcessed = 0;
93                  
94                  // Unlock the update method so that more events can be processed.
95                  updating = false;
96              }
97          }
98      }

100     /**
101      * <b>setClusterCollectionName</b><br/><br/>
102      * <code>public void <b>setClusterCollectionName</b>(String clusterCollectionName)</code><br/><br/>
103      * Sets the name of the LCIO collection that contains calorimeter
104      * cluster information.
105      * @param clusterCollectionName - The name of the LCIO collection. 
106      */
107     public void setClusterCollectionName(String clusterCollectionName) {
108         this.clusterCollectionName = clusterCollectionName;
109     }
110     
111     /**
112      * <b>setDisplayInterval</b><br/><br/>
113      * <code>public void <b>setDisplayInterval</b>(String displayInterval)</code><br/><br/>
114      * Sets the rate at which events are displayed. The driver will
115      * render a new event when <code>displayInterval</code> events
116      * have occurred since the last one. Note that e value of 0 or
117      * 1 will display events as quickly as they can be displayed.
118      * @param displayInterval - The number of events to skip before
119      * a new event is displayed.
120      */
121     public void setDisplayInterval(String displayInterval) {
122         // Convert the argument to an integer.
123         int disp = Integer.parseInt(displayInterval);
124         
125         // If it is negative, make it zero.
126         if(disp < 0) { disp = 0; }
127         
128         // Set the display interval.
129         this.displayInterval = disp;
130     }
131     
132     /**
133      * <b>setEcalCollectionName</b><br/><br/>
134      * <code>public void <b>setEcalCollectionName</b>(String ecalCollectionName)</code><br/><br/>
135      * Sets the name of the LCIO collection that contains calorimeter
136      * hit information.
137      * @param ecalCollectionName - The name of the LCIO collection. 
138      */
139     public void setEcalCollectionName(String ecalCollectionName) {
140         this.ecalCollectionName = ecalCollectionName;
141     }
142     
143     /**
144      * <b>setEcalName</b><br/><br/>
145      * <code>public void <b>setEcalName</b>(String ecalName)</code><br/><br/>
146      * Sets which detector configuration should be used.
147      * @param ecalName - The name of the detector configuration.
148      */
149     public void setEcalName(String ecalName) { this.ecalName = ecalName; }
150     
151     /**
152      * <b>startOfData</b><br/><br/>
153      * <code>public void <b>startOfData</b>()</code><br/><br/>
154      * Ensures that critical collection names are defined.
155      */
156     public void startOfData() {
157         // Make sure that there is a cluster collection name into which clusters may be placed.
158         if (ecalCollectionName == null) {
159             throw new RuntimeException("The parameter ecalCollectionName was not set!");
160         }
161         
162         // Make sure that there is a calorimeter detector.
163         if (ecalName == null) {
164             throw new RuntimeException("The parameter ecalName was not set!");
165         }
166         
167         // Set the events passed so that the first event will display.
168         eventsProcessed = displayInterval - 1;
169     }
170 }
