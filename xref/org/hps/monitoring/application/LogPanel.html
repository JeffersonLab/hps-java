1   package org.hps.monitoring.application;
2   
3   import java.awt.BorderLayout;
4   import java.awt.Dimension;
5   import java.awt.FlowLayout;
6   import java.awt.event.ActionListener;
7   
8   import javax.swing.BoxLayout;
9   import javax.swing.JButton;
10  import javax.swing.JLabel;
11  import javax.swing.JPanel;
12  import javax.swing.JScrollPane;
13  
14  import org.hps.monitoring.application.model.ConfigurationModel;
15  
16  /**
17   * This is a simple GUI component for the log table and its controls.
18   */
19  @SuppressWarnings("serial")
20  final class LogPanel extends JPanel {
21  
22      /**
23       * The combo box for filtering which messages are currently showing in the table.
24       */
25      private final LogLevelFilterComboBox logFilterComboBox;
26  
27      /**
28       * The table containing the log messages.
29       */
30      private final LogTable logTable;
31  
32      /**
33       * Class constructor.
34       *
35       * @param configurationModel the {@link org.hps.monitoring.application.model.ConfigurationModel} providing the data
36       * model
37       * @param listener an {@link java.awt.event.ActionListener} to register on certain components
38       */
39      LogPanel(final ConfigurationModel configurationModel, final ActionListener listener) {
40  
41          setLayout(new BoxLayout(this, BoxLayout.PAGE_AXIS));
42  
43          this.logTable = new LogTable(configurationModel);
44  
45          final JPanel controlsPanel = new JPanel();
46          controlsPanel.setLayout(new FlowLayout(FlowLayout.LEFT, 10, 5));
47  
48          final JLabel label = new JLabel("Log Level Filter");
49          this.logFilterComboBox = new LogLevelFilterComboBox(configurationModel);
50          this.logFilterComboBox.setToolTipText("Messages below this level will be filtered out.");
51          controlsPanel.add(label);
52          controlsPanel.add(this.logFilterComboBox);
53  
54          final JButton exportButton = new JButton("Export ...");
55          exportButton.setActionCommand(Commands.SAVE_LOG_TABLE);
56          exportButton.addActionListener(listener);
57          controlsPanel.add(exportButton);
58  
59          final JButton clearButton = new JButton("Clear");
60          clearButton.setActionCommand(Commands.CLEAR_LOG_TABLE);
61          clearButton.addActionListener(listener);
62          controlsPanel.add(clearButton);
63  
64          final JScrollPane tablePane = new JScrollPane(this.logTable);
65          controlsPanel.setMaximumSize(new Dimension(tablePane.getPreferredSize().width, 200));
66  
67          add(controlsPanel, BorderLayout.PAGE_START);
68          add(tablePane, BorderLayout.PAGE_END);
69      }
70  
71      /**
72       * Get the log table.
73       *
74       * @return the log table
75       */
76      LogTable getLogTable() {
77          return this.logTable;
78      }
79  }
