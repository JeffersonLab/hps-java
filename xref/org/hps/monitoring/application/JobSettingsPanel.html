41       * This filter will accept only files called compact.xml which should be an LCSim detector description file.
42       */
43      static class CompactFileFilter extends FileFilter {
44  
45          /**
46           * Class constructor.
47           */
48          public CompactFileFilter() {
49          }
50  
51          /**
52           * Returns <code>true</code> if the path passes the filter.
53           *
54           * @return <code>true</code> if the path passes the filter
55           */
56          @Override
57          public boolean accept(final File pathname) {
58              return "compact.xml".equals(pathname.getName());
59          }
60  
61          /**
62           * Get the description of the filter.
63           *
64           * @return the description of the filter
65           */
66          @Override
67          public String getDescription() {
68              return "Compact XML files";
69          }

72      /**
73       * Update the GUI from changes in the underlying model. The changes are distinguishable by their property name.
74       */
75      private class JobSettingsChangeListener implements PropertyChangeListener {
76  
77          /**
78           * Handle {@link java.beans.PropertyChangeEvent} by updating the GUI from changes to the model.
79           *
80           * @param evt the {@link java.beans.PropertyChangeEvent} to handle
81           */
82          @Override
83          public void propertyChange(final PropertyChangeEvent evt) {
84              if (evt.getSource() instanceof ConfigurationModel) {
85                  final Object value = evt.getNewValue();
86                  final String property = evt.getPropertyName();
87                  JobSettingsPanel.this.getConfigurationModel().removePropertyChangeListener(this);
88                  try {
89                      if (property.equals(ConfigurationModel.DETECTOR_NAME_PROPERTY)) {
90                          JobSettingsPanel.this.detectorNameComboBox.setSelectedItem(value);
91                      } else if (property.equals(ConfigurationModel.DETECTOR_ALIAS_PROPERTY)) {
92                          JobSettingsPanel.this.detectorAliasField.setText((String) value);
93                      } else if (property.equals(ConfigurationModel.DISCONNECT_ON_ERROR_PROPERTY)) {
94                          JobSettingsPanel.this.disconnectOnErrorCheckBox.setSelected((Boolean) value);
95                      } else if (property.equals(ConfigurationModel.DISCONNECT_ON_END_RUN_PROPERTY)) {
96                          JobSettingsPanel.this.disconnectOnEndRunCheckBox.setSelected((Boolean) value);
97                      } else if (property.equals(ConfigurationModel.EVENT_BUILDER_PROPERTY)) {
98                          JobSettingsPanel.this.eventBuilderComboBox.setSelectedItem(value);
99                      } else if (property.equals(ConfigurationModel.LOG_FILE_NAME_PROPERTY)) {
100                         JobSettingsPanel.this.logFileNameField.setText((String) value);
101                     } else if (property.equals(ConfigurationModel.LOG_LEVEL_PROPERTY)) {
102                         JobSettingsPanel.this.logLevelComboBox.setSelectedItem(value.toString());
103                     } else if (property.equals(ConfigurationModel.LOG_TO_FILE_PROPERTY)) {
104                         JobSettingsPanel.this.logToFileCheckbox.setSelected((Boolean) value);
105                     } else if (property.equals(ConfigurationModel.STEERING_TYPE_PROPERTY)) {
106                         JobSettingsPanel.this.steeringTypeComboBox.setSelectedItem(value);
107                     } else if (property.equals(ConfigurationModel.STEERING_FILE_PROPERTY)) {
108                         if (value != null) {
109                             JobSettingsPanel.this.steeringFileField.setText((String) evt.getNewValue());
110                         } else {
111                             // A null value here is actually okay and means this field should be reset to have no value.
112                             JobSettingsPanel.this.steeringFileField.setText(null);
113                         }
114                     } else if (property.equals(ConfigurationModel.STEERING_RESOURCE_PROPERTY)) {
115                         JobSettingsPanel.this.steeringResourcesComboBox.setSelectedItem(value);
116                     } else if (property.equals(ConfigurationModel.USER_RUN_NUMBER_PROPERTY)) {
117                         if (value != null) {
118                             JobSettingsPanel.this.userRunNumberField.setText(Integer.toString((int) value));
119                         } else {
120                             JobSettingsPanel.this.userRunNumberField.setText(null);
121                         }
122                     } else if (property.equals(ConfigurationModel.FREEZE_CONDITIONS_PROPERTY)) {
123                         if (value != null) {
124                             JobSettingsPanel.this.freezeConditionsCheckBox.setSelected((Boolean) value);
125                         }
126                     } else if (property.equals(ConfigurationModel.MAX_EVENTS_PROPERTY)) {
127                         if (value != null) {
128                             JobSettingsPanel.this.maxEventsField.setText(value.toString());
129                         }
130                     } else if (property.equals(ConfigurationModel.PROCESSING_STAGE_PROPERTY)) {
131                         JobSettingsPanel.this.processingStageComboBox.setSelectedItem(evt.getNewValue());
132                     }
133                 } finally {
134                     JobSettingsPanel.this.getConfigurationModel().addPropertyChangeListener(this);
135                 }
136             }
137         }
138     }

337     /**
338      * Handle {@link java.awt.event.ActionEvent} objects by pushing changes from the GUI into the model.
339      *
340      * @param event the {@link java.awt.event.ActionEvent} to handle
341      */
342     @Override
343     public void actionPerformed(final ActionEvent event) {
344         try {
345             this.getConfigurationModel().removePropertyChangeListener(this);
346             final String command = event.getActionCommand();
347             if (event.getActionCommand().equals(Commands.CHOOSE_STEERING_FILE)) {
348                 this.chooseSteeringFile();
349             } else if (event.getActionCommand().equals(Commands.CHOOSE_COMPACT_FILE)) {
350                 this.chooseCompactFile();
351             } else if (Commands.DISCONNECT_ON_ERROR_CHANGED.equals(command)) {
352                 this.getConfigurationModel().setDisconnectOnError(this.disconnectOnErrorCheckBox.isSelected());
353             } else if (Commands.DISCONNECT_ON_END_RUN_CHANGED.equals(command)) {
354                 this.getConfigurationModel().setDisconnectOnEndRun(this.disconnectOnEndRunCheckBox.isSelected());
355             } else if (Commands.STEERING_TYPE_CHANGED.equals(command)) {
356                 this.getConfigurationModel().setSteeringType(this.steeringTypeComboBox.getSelectedItem().toString());
357             } else if (Commands.STEERING_RESOURCE_CHANGED.equals(command)) {
358                 this.getConfigurationModel().setSteeringResource(
359                         this.steeringResourcesComboBox.getSelectedItem().toString());
360             } else if (Commands.LOG_LEVEL_CHANGED.equals(command)) {
361                 this.getConfigurationModel().setLogLevel(Level.parse((String) this.logLevelComboBox.getSelectedItem()));
362             } else if (Commands.EVENT_BUILDER_CHANGED.equals(command)) {
363                 this.getConfigurationModel().setEventBuilderClassName(
364                         (String) this.eventBuilderComboBox.getSelectedItem());
365             } else if (Commands.DETECTOR_NAME_CHANGED.equals(command)) {
366                 try {
367                     this.getConfigurationModel().setDetectorName((String) this.detectorNameComboBox.getSelectedItem());
368                 } catch (final Exception exception) {
369                     exception.printStackTrace();
370                 }
371             } else if (Commands.FREEZE_CONDITIONS_CHANGED.equals(command)) {
372                 if (this.getConfigurationModel().hasPropertyKey(ConfigurationModel.USER_RUN_NUMBER_PROPERTY)
373                         && this.getConfigurationModel().getUserRunNumber() != null) {
374                     this.getConfigurationModel().setFreezeConditions(this.freezeConditionsCheckBox.isSelected());
375                 } else {
376                     throw new IllegalArgumentException(
377                             "Conditions system may only be frozen if there is a valid user run number.");
378                 }
379             } else if (Commands.DETECTOR_ALIAS_CHANGED.equals(command)) {
380                 this.getConfigurationModel().setDetectorName(this.detectorAliasField.getText());
381             } else if (Commands.CONDITIONS_TAG_CHANGED.equals(command)) {
382                 this.getConfigurationModel().setConditionsTag((String) this.conditionsTagComboBox.getSelectedItem());
383             } else if (Commands.PROCESSING_STAGE_CHANGED.equals(command)) {
384                 this.getConfigurationModel().setProcessingStage(
385                         this.processingStageComboBox.getSelectedItem().toString());
386             }
387         } finally {
388             this.getConfigurationModel().addPropertyChangeListener(this);
389         }
390     }

403     /**
404      * Parse the LCSim XML steering file to see if it looks valid.
405      *
406      * @param file the input steering file
407      * @throws IOException if there is a basic IO problem like reading the file
408      * @throws JDOMException if the XML is not valid
409      */
410     private void checkSteeringFile(final File file) throws IOException, JDOMException {
411         final SAXBuilder builder = new SAXBuilder();
412         builder.setExpandEntities(false);
413 
414         final Document document = builder.build(file);
415         final Element rootNode = document.getRootElement();
416         if (!"lcsim".equals(rootNode.getName())) {
417             throw new IOException("not an LCSim XML file: " + file.getPath());
418         }
419     }

421     /**
422      * Choose a compact XML file to override the one embedded in the jar as a resource.
423      */
424     private void chooseCompactFile() {
425         final JFileChooser fc = new JFileChooser();
426         fc.setDialogTitle("Choose a Compact XML File");
427         fc.setCurrentDirectory(new File("."));
428         fc.setFileFilter(new CompactFileFilter());
429         final int r = fc.showDialog(this, "Select ...");
430         if (r == JFileChooser.APPROVE_OPTION) {
431             final File file = fc.getSelectedFile();
432             this.getConfigurationModel().setDetectorAlias(file.getParent());
433         }
434     }

436     /**
437      * Choose an LCSim steering file.
438      */
439     private void chooseSteeringFile() {
440         final JFileChooser fc = new JFileChooser();
441         fc.setDialogTitle("Choose an LCSim Steering File");
442         fc.setCurrentDirectory(new File("."));
443         final int r = fc.showDialog(this, "Select ...");
444         if (r == JFileChooser.APPROVE_OPTION) {
445             final File file = fc.getSelectedFile();
446             try {
447                 this.checkSteeringFile(file);
448                 this.getConfigurationModel().setSteeringFile(file.getCanonicalPath());
449                 this.getConfigurationModel().setSteeringType(SteeringType.FILE.toString());
450             } catch (IOException | JDOMException e) {
451                 throw new RuntimeException("Error parsing the selected steering file.", e);
452             }
453         }
454     }

456     /**
457      * Updates the configuration with changes from the GUI component values. The changes from the GUI are
458      * distinguishable by their component object.
459      */
460     @Override
461     public void propertyChange(final PropertyChangeEvent evt) {
462         this.getConfigurationModel().removePropertyChangeListener(this);
463         try {
464             final Object source = evt.getSource();
465             if (source == this.steeringFileField) {
466                 this.getConfigurationModel().setSteeringFile(this.steeringFileField.getText());
467             } else if (source == this.userRunNumberField) {
468                 // Is run number being reset to null or empty?
469                 if (this.userRunNumberField.getText() == null || this.userRunNumberField.getText().isEmpty()) {
470                     // Update the model to null user run number and do not freeze the conditions system.
471                     this.getConfigurationModel().setUserRunNumber(null);
472                     this.getConfigurationModel().setFreezeConditions(false);
473                 } else {
474                     try {
475                         // Parse the run number. Need to catch errors because it might be an invalid string.
476                         final int userRunNumber = Integer.parseInt(this.userRunNumberField.getText());
477                         this.getConfigurationModel().setUserRunNumber(userRunNumber);
478                         this.getConfigurationModel().setFreezeConditions(true);
479                     } catch (final NumberFormatException e) {
480                         System.out.println("bad number format so ignoring user run number " + evt.getNewValue());
481                         this.userRunNumberField.setText((String) evt.getOldValue());
482                         // throw new IllegalArgumentException("The value " + evt.getNewValue() +
483                         // " is not a valid run number.");
484                     }
485                 }
486             } else if (source == this.maxEventsField) {
487                 this.getConfigurationModel().setMaxEvents(Long.parseLong(this.maxEventsField.getText()));
488                 // System.out.println("setMaxEvents - " + configurationModel.getMaxEvents());
489             } else if (evt.getPropertyName().equals(ConfigurationModel.LOG_TO_FILE_PROPERTY)) {
490                 // This is getting the log to file prop change from the ConfigurationModel to update a read only
491                 // component.
492                 final Boolean logToFile = (Boolean) evt.getNewValue();
493                 if (logToFile != null) {
494                     this.logToFileCheckbox.setSelected(logToFile);
495                 }
496             } else if (evt.getPropertyName().equals(ConfigurationModel.LOG_FILE_NAME_PROPERTY)) {
497                 // This is getting the log file name prop change from the ConfigurationModel to update a read only
498                 // component.
499                 final String logFileName = (String) evt.getNewValue();
500                 if (logFileName != null && logFileName.length() > 0) {
501                     this.logFileNameField.setText(logFileName);
502                 } else {
503                     this.logFileNameField.setText("");
504                 }
505             } else if (evt.getPropertyName().equals(ConfigurationModel.CONDITIONS_TAG_PROPERTY)) {
506                 this.conditionsTagComboBox.setSelectedItem(evt.getNewValue());
507             }
508         } finally {
509             this.getConfigurationModel().addPropertyChangeListener(this);
510         }
511     }
