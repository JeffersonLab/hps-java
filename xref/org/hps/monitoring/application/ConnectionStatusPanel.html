1   package org.hps.monitoring.application;

3   import java.awt.Color;
4   import java.awt.FlowLayout;
5   import java.awt.Font;
6   import java.beans.PropertyChangeEvent;
7   import java.beans.PropertyChangeListener;
8   import java.text.SimpleDateFormat;
9   import java.util.Date;
10  
11  import javax.swing.JLabel;
12  import javax.swing.JPanel;
13  import javax.swing.JTextField;
14  import javax.swing.SwingConstants;
15  
16  import org.hps.monitoring.application.model.ConnectionStatus;
17  import org.hps.monitoring.application.model.ConnectionStatusModel;
18  
19  /**
20   * This is the panel for showing the current connection status (connected, disconnected, etc.) in the tool bar.
21   */
22  @SuppressWarnings("serial")
23  final class ConnectionStatusPanel extends JPanel implements PropertyChangeListener {
24  
25      /**
26       * Date formatting.
27       */
28      private static final SimpleDateFormat DATE_FORMAT = new SimpleDateFormat("MMMM-dd-yyyy HH:mm:ss");
29  
30      /**
31       * Field for date when status changed (read only).
32       */
33      private final JTextField dateField;
34  
35      /**
36       * The model for getting connection status changes.
37       */
38      private final ConnectionStatusModel model;
39  
40      /**
41       * The field showing the current connection status (read only).
42       */
43      private final JTextField statusField;
44  
45      /**
46       * Class constructor.
47       *
48       * @param model the model which notifies this component of connection status changes
49       */
50      ConnectionStatusPanel(final ConnectionStatusModel model) {
51  
52          this.model = model;
53          this.model.addPropertyChangeListener(this);
54  
55          setLayout(new FlowLayout(FlowLayout.LEFT, 5, 0));
56  
57          this.statusField = new JTextField("", 10);
58          this.statusField.setHorizontalAlignment(SwingConstants.LEFT);
59          this.statusField.setEditable(false);
60          this.statusField.setBackground(Color.WHITE);
61          this.statusField.setFont(new Font("Arial", Font.BOLD, 16));
62          this.statusField.setForeground(model.getConnectionStatus().getColor());
63          this.statusField.setText(model.getConnectionStatus().name());
64          add(this.statusField);
65  
66          add(new JLabel("@"));
67  
68          this.dateField = new JTextField("", 21);
69          this.dateField.setEditable(false);
70          this.dateField.setBackground(Color.WHITE);
71          this.dateField.setHorizontalAlignment(SwingConstants.LEFT);
72          this.dateField.setFont(new Font("Arial", Font.PLAIN, 14));
73          add(this.dateField);
74      }
75  
76      /**
77       * Handle a property change event coming from the model.
78       *
79       * @param evt the property change event
80       */
81      @Override
82      public void propertyChange(final PropertyChangeEvent evt) {
83          if (evt.getPropertyName().equals(ConnectionStatusModel.CONNECTION_STATUS_PROPERTY)) {
84              final ConnectionStatus status = (ConnectionStatus) evt.getNewValue();
85              this.statusField.setForeground(status.getColor());
86              this.statusField.setText(status.name());
87              this.dateField.setText(ConnectionStatusPanel.DATE_FORMAT.format(new Date()));
88          }
89      }
90  }
