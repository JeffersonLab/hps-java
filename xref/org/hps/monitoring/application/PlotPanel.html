31      /**
32       * Class constructor.
33       */
34      public PlotPanel() {
35          this.setLayout(new BorderLayout());
36          this.plotPane = new JTabbedPane();
37          this.plotPane.setPreferredSize(this.getPreferredSize());
38          this.add(this.plotPane, BorderLayout.CENTER);
39      }

41      /**
42       * Handle an {@link java.awt.event.ActionEvent}.
43       *
44       * @param event the {@link java.awt.event.ActionEvent} to handle
45       */
46      @Override
47      public void actionPerformed(final ActionEvent event) {
48          if (event.getActionCommand().equals(Commands.SAVE_SELECTED_PLOTS)) {
49              saveCurrentPlot();
50          }
51      }

93      /**
94       * Save the plotter from a tab to a graphics file.
95       *
96       * @param plotter the plotter to save
97       */
98      private void saveCurrentPlot() {
99          final JFileChooser fc = new JFileChooser();
100         fc.setAcceptAllFileFilterUsed(false);
101         fc.setDialogTitle("Save Plot");
102         fc.setCurrentDirectory(new File("."));
103         fc.setAcceptAllFileFilterUsed(false);
104         fc.setFileFilter(new FileNameExtensionFilter("PNG file", "png"));
105         fc.addChoosableFileFilter(new FileNameExtensionFilter("JPG file", "jpg"));
106         fc.addChoosableFileFilter(new FileNameExtensionFilter("GIF file", "gif"));
107         final int r = fc.showSaveDialog(this);
108         if (r == JFileChooser.APPROVE_OPTION) {
109             String path = fc.getSelectedFile().getPath();
110             final FileNameExtensionFilter filter = (FileNameExtensionFilter) fc.getFileFilter();
111             if (!path.endsWith("." + filter.getExtensions()[0])) {
112                 path += "." + filter.getExtensions()[0];
113             }
114             try {
115                 final BufferedImage image = ExportPdf.getImage(this.getSelectedComponent());
116                 ImageIO.write(image, filter.getExtensions()[0], new File(path));
117                 DialogUtil.showInfoDialog(this, "Plots Saved", "Plots from panel were saved to" + '\n' + path);
118             } catch (final IOException e) {
119                 e.printStackTrace();
120                 DialogUtil.showErrorDialog(this, "Error Saving Plots", "There was an error saving the plots.");
121             }
122         }
123     }
