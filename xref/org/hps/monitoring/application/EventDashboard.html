33          /**
34           * Task to periodically update the fields as events are processed.
35           */
36          class RunTimerTask extends TimerTask {
37  
38              /**
39               * Run the timer task to update the GUI from the current values in the model.
40               */
41              @Override
42              public void run() {
43  
44                  final double tickLengthSeconds = (System.currentTimeMillis() - EventDashboardUpdater.this.lastTickMillis)
45                          / (double) MILLIS;
46                  final int elapsedTime = (int) ((System.currentTimeMillis() - EventDashboardUpdater.this.jobStartMillis) / (double) MILLIS);
47                  final double megaBytesReceived = EventDashboardUpdater.this.bytesReceived / 1000000;
48                  EventDashboardUpdater.this.totalEvents += EventDashboardUpdater.this.eventsReceived;
49  
50                  // Print to System.out if debugging the processor (by default this is off).
51                  if (DEBUG) {
52                      System.out.println("tickLengthSeconds = " + tickLengthSeconds);
53                      System.out.println("elapsedTime = " + elapsedTime);
54                      System.out.println("eventsReceived = " + EventDashboardUpdater.this.eventsReceived);
55                      System.out.println("dataRate = " + megaBytesReceived / tickLengthSeconds);
56                      System.out.println("eventNumber = " + EventDashboardUpdater.this.eventNumber);
57                      System.out.println("eventRate = " + EventDashboardUpdater.this.eventsReceived / tickLengthSeconds);
58                      System.out.println("totalEvents = " + EventDashboardUpdater.this.totalEvents);
59                      System.out.println("megaBytesReceived = " + megaBytesReceived);
60  
61                  }
62  
63                  EventDashboard.this.runModel.setElapsedTime(elapsedTime);
64                  EventDashboard.this.runModel.setEventsReceived(EventDashboardUpdater.this.totalEvents);
65                  EventDashboard.this.runModel.setDataRate(megaBytesReceived / tickLengthSeconds);
66                  EventDashboard.this.runModel.addDataReceived(megaBytesReceived);
67                  EventDashboard.this.runModel.setEventNumber(EventDashboardUpdater.this.eventNumber);
68                  EventDashboard.this.runModel
69                          .setEventRate(EventDashboardUpdater.this.eventsReceived / tickLengthSeconds);
70  
71                  EventDashboardUpdater.this.eventsReceived = 0;
72                  EventDashboardUpdater.this.bytesReceived = 0;
73                  EventDashboardUpdater.this.eventNumber = 0;
74  
75                  EventDashboardUpdater.this.lastTickMillis = System.currentTimeMillis();
76  
77                  // System.out.println();
78              }
79          }

136         /**
137          * Check for head bank and update the run info if necessary.
138          *
139          * @param evioEvent the EVIO event
140          */
141         private void checkHeadBank(final EvioEvent evioEvent) {
142             final BaseStructure headBank = EvioEventUtilities.getHeadBank(evioEvent);
143             if (headBank != null) {
144                 final int headBankRun = headBank.getIntData()[1];
145                 if (headBankRun != this.runNumber) {
146                     this.runNumber = headBankRun;
147                     EventDashboard.this.runModel.setRunNumber(headBankRun);
148                     EventDashboard.this.runModel.setStartDate(new Date(headBank.getIntData()[3] * MILLIS));
149                 }
150             }
151         }

153         /**
154          * Perform end of job hook, which will cancel the update timer.
155          */
156         @Override
157         public void endJob() {
158             this.timer.cancel();
159 
160             // Push final values into GUI.
161             this.timer = new Timer("RunModelUpdaterEndJobTimer");
162             this.timer.schedule(new RunTimerTask(), 0);
163         }

165         /**
166          * Handle an EVIO END event.
167          *
168          * @param evioEvent the EVIO END event
169          */
170         private void endRun(final EvioEvent evioEvent) {
171             // Get end run data.
172             final int[] data = EvioEventUtilities.getControlEventData(evioEvent);
173             if (data != null) {
174                 final int seconds = data[0];
175                 final int eventCount = data[2];
176                 final long endMillis = (long) seconds * 1000;
177 
178                 // Update the GUI.
179                 EventDashboard.this.runModel.setEndDate(new Date(endMillis));
180                 EventDashboard.this.runModel.computeRunLength();
181                 EventDashboard.this.runModel.setTotalEvents(eventCount);
182             }
183         }

185         /**
186          * Process a {@link org.hps.record.composite.CompositeRecord} to extract information from available event
187          * sources and update the running values.
188          */
189         @Override
190         public void process(final CompositeRecord event) {
191             // FIXME: CompositeRecord number is always -1 here.
192             if (event.getEvioEvent() != null) {
193                 final EvioEvent evioEvent = event.getEvioEvent();
194                 this.bytesReceived += evioEvent.getTotalBytes();
195                 if (EvioEventUtilities.isPreStartEvent(evioEvent)) {
196                     // Get run start info from pre start event.
197                     startRun(evioEvent);
198                 } else if (EvioEventUtilities.isEndEvent(evioEvent)) {
199                     // Get end run info from end event.
200                     endRun(evioEvent);
201                 } else if (EvioEventUtilities.isPhysicsEvent(evioEvent)) {
202                     // Check for run info in head bank.
203                     checkHeadBank(evioEvent);
204                     this.eventNumber = evioEvent.getEventNumber();
205                     this.eventsReceived += 1;
206                 }
207             } else if (event.getEtEvent() != null) {
208                 this.bytesReceived += event.getEtEvent().getData().length;
209                 this.eventNumber = event.getEtEvent().getId();
210                 this.eventsReceived += 1;
211             } else if (event.getLcioEvent() != null) {
212                 final EventHeader lcioEvent = event.getLcioEvent();
213                 this.eventNumber = lcioEvent.getEventNumber();
214                 if (lcioEvent.getRunNumber() != this.runNumber) {
215                     this.runNumber = lcioEvent.getRunNumber();
216                     startRun(lcioEvent);
217                 }
218                 this.eventsReceived += 1;
219             }
220         }

222         /**
223          * Perform start of job hook which initializes this processor.
224          */
225         @Override
226         public void startJob() {
227             EventDashboard.this.runModel.reset();
228             this.jobStartMillis = System.currentTimeMillis();
229 
230             // Start the timer to update GUI components about once per second.
231             this.timer = new Timer("RunModelUpdaterTimer");
232             this.lastTickMillis = System.currentTimeMillis();
233             this.timer.scheduleAtFixedRate(new RunTimerTask(), 0, MILLIS);
234         }

236         /**
237          * Handle start of run using an LCIO event.
238          *
239          * @param lcioEvent the LCIO event
240          */
241         private void startRun(final EventHeader lcioEvent) {
242             EventDashboard.this.runModel.setRunNumber(lcioEvent.getRunNumber());
243             final long seconds = lcioEvent.getTimeStamp() / NANOS;
244             EventDashboard.this.runModel.setStartDate(new Date((int) seconds));
245         }
246 
247         /**
248          * Handle start of run using an EVIO START event.
249          *
250          * @param evioEvent the EVIO START event
251          */
252         private void startRun(final EvioEvent evioEvent) {
253             // Get start of run data.
254             final int[] data = EvioEventUtilities.getControlEventData(evioEvent);
255             if (data != null) {
256                 final int seconds = data[0];
257                 this.runNumber = data[1];
258 
259                 // Update the GUI.
260                 EventDashboard.this.runModel.setRunNumber(this.runNumber);
261                 EventDashboard.this.runModel.setStartDate(new Date(seconds * MILLIS));
262             }
263         }

363     /**
364      * Update the GUI from changes to the backing {@link org.hps.monitoring.application.model.RunModel} object.
365      *
366      * @param evt the {@link java.beans.PropertyChangeEvent} to handle
367      */
368     @Override
369     public void propertyChange(final PropertyChangeEvent evt) {
370         // System.out.println("RunPanel.propertyChange - " + evt.getPropertyName());
371         final Object value = evt.getNewValue();
372         if (RunModel.RUN_NUMBER_PROPERTY.equals(evt.getPropertyName())) {
373             this.runNumberField.setValue((Integer) value);
374         } else if (RunModel.START_DATE_PROPERTY.equals(evt.getPropertyName())) {
375             if (value != null) {
376                 this.startDateField.setValue((Date) value);
377             } else {
378                 this.startDateField.setValue("");
379             }
380         } else if (RunModel.END_DATE_PROPERTY.equals(evt.getPropertyName())) {
381             if (value != null) {
382                 this.endDateField.setValue((Date) value);
383             } else {
384                 this.endDateField.setValue("");
385             }
386         } else if (RunModel.RUN_LENGTH_PROPERTY.equals(evt.getPropertyName())) {
387             this.lengthField.setValue((Integer) value);
388         } else if (RunModel.TOTAL_EVENTS_PROPERTY.equals(evt.getPropertyName())) {
389             this.totalEventsField.setValue((Integer) value);
390         } else if (RunModel.EVENTS_RECEIVED_PROPERTY.equals(evt.getPropertyName())) {
391             this.eventsReceivedField.setValue((Integer) value);
392         } else if (RunModel.ELAPSED_TIME_PROPERTY.equals(evt.getPropertyName())) {
393             this.elapsedTimeField.setValue((Integer) value);
394         } else if (RunModel.DATA_RECEIVED_PROPERTY.equals(evt.getPropertyName())) {
395             this.dataReceivedField.setValue(DECIMAL_FORMAT.format(value));
396         } else if (RunModel.EVENT_NUMBER_PROPERTY.equals(evt.getPropertyName())) {
397             this.eventNumberField.setValue((Integer) value);
398         } else if (RunModel.DATA_RATE_PROPERTY.equals(evt.getPropertyName())) {
399             this.dataRateField.setValue(DECIMAL_FORMAT.format(value));
400         } else if (RunModel.EVENT_RATE_PROPERTY.equals(evt.getPropertyName())) {
401             this.eventRateField.setValue(DECIMAL_FORMAT.format(value));
402         }
