3   import java.util.HashSet;
4   import java.util.Set;
5   
6   import org.hps.conditions.database.DatabaseConditionsManager;
7   import org.hps.record.LCSimEventBuilder;
8   import org.reflections.Reflections;
9   
10  /**
11   * This is a set of utility methods for getting jar resources at runtime.
12   */
13  public final class ResourceUtil {
14  
15      /**
16       * Find all classes that implement {@link org.hps.record.LCSimEventBuilder} and return a list of their canonical
17       * names.
18       *
19       * @return the list of fully qualified class names that implement LCSimEventBuilder
20       */
21      public static String[] findEventBuilderClassNames() {
22          final Reflections reflections = new Reflections("org.hps");
23          final Set<Class<? extends LCSimEventBuilder>> subTypes = reflections.getSubTypesOf(LCSimEventBuilder.class);
24          final Set<String> classNames = new HashSet<String>();
25          for (final Class<? extends LCSimEventBuilder> type : subTypes) {
26              classNames.add(type.getCanonicalName());
27          }
28          return classNames.toArray(new String[classNames.size()]);
29      }
30  
31      /**
32       * Get the list of available conditions tags from the conditions system.
33       *
34       * @return the list of available conditions tags
35       */
36      public static String[] getConditionsTags() {
37          // FIXME: New database manager should probably not be instantiated here.
38          DatabaseConditionsManager mgr = DatabaseConditionsManager.getInstance();
39          String[] tags = mgr.getAvailableTags().toArray(new String[]{});
40          try {
41              mgr.getConnection().close();
42          } catch (Exception e) {
43              e.printStackTrace();
44          }
45          return tags;
46      }
47  
48      /**
49       * Do not allow class instantiation.
50       */
51      private ResourceUtil() {
52          throw new UnsupportedOperationException("Do not instantiate this class.");
53      }
54  }
