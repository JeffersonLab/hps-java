1   package org.hps.monitoring.application.util;
2   
3   import java.io.BufferedWriter;
4   import java.io.FileWriter;
5   import java.io.IOException;
6   
7   import javax.swing.JTable;
8   import javax.swing.table.TableModel;
9   
10  /**
11   * This is a utility for exporting a JTable's model data to a text file.
12   * <p>
13   * Non-numeric fields such as strings are delimited by double quotes.
14   */
15  public final class TableExporter {
16  
17      /**
18       * Export the given table to a text file.
19       *
20       * @param table the JTable component
21       * @param path the output file path
22       * @param fieldDelimiter the field delimiter to use
23       * @throws IOException if there are errors writing the file
24       */
25      public static void export(final JTable table, final String path, final char fieldDelimiter) throws IOException {
26  
27          final StringBuffer buffer = new StringBuffer();
28          final TableModel model = table.getModel();
29          final int rowCount = model.getRowCount();
30          final int columnCount = model.getColumnCount();
31  
32          // Column headers.
33          for (int columnIndex = 0; columnIndex < columnCount; columnIndex++) {
34              buffer.append("\"" + model.getColumnName(columnIndex) + "\"" + fieldDelimiter + ",");
35          }
36          buffer.setLength(buffer.length() - 1);
37          buffer.append('\n');
38  
39          // Row data.
40          for (int rowIndex = 0; rowIndex < rowCount; rowIndex++) {
41              for (int columnIndex = 0; columnIndex < columnCount; columnIndex++) {
42                  final Object value = model.getValueAt(rowIndex, columnIndex);
43                  if (Number.class.isAssignableFrom(model.getColumnClass(columnIndex))) {
44                      buffer.append(value);
45                  } else {
46                      buffer.append("\"" + value + "\"" + fieldDelimiter);
47                  }
48                  buffer.append(",");
49              }
50              buffer.setLength(buffer.length() - 1);
51              buffer.append('\n');
52          }
53  
54          // Write string buffer to file.
55          final BufferedWriter out = new BufferedWriter(new FileWriter(path));
56          out.write(buffer.toString());
57          out.flush();
58          out.close();
59      }
60  
61      /**
62       * Do not allow class instantiation.
63       */
64      private TableExporter() {
65          throw new UnsupportedOperationException("Do not instantiate this class.");
66      }
67  }
