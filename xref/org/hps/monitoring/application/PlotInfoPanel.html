178         // Is there a new item selected in the combo box?
179         if (Commands.PLOT_SELECTED.equals(e.getActionCommand())) {
180             if (this.plotComboBox.getSelectedItem() != null) {

183             }

197     /**
198      * Add this object as an <code>AIDAListener</code> on the current <code>AIDAObservable</code>.
199      */
200     private void addListener() {
201         if (this.currentObject instanceof AIDAObservable && !(this.currentObject instanceof FunctionDispatcher)) {
202             // Setup a listener on the current AIDA object.
203             final AIDAObservable observable = (AIDAObservable) this.currentObject;
204             observable.addListener(this);
205             observable.setValid(this);
206             observable.setConnected(true);
207         } else if (this.currentObject instanceof IFunction) {
208             if (this.currentObject instanceof FunctionDispatcher) {
209                 ((FunctionDispatcher) this.currentObject).addFunctionListener(this);
210             }
211         }
212     }

224     /**
225      * Add rows to the info table from the state of a 2D cloud.
226      *
227      * @param cloud the AIDA object
228      */
229     private void addRows(final ICloud2D cloud) {
230         this.addRow("title", cloud.title());
231         this.addRow("entries", cloud.entries());
232         this.addRow("max entries", cloud.maxEntries());
233         this.addRow("x lower edge", cloud.lowerEdgeX());
234         this.addRow("x upper edge", cloud.upperEdgeX());
235         this.addRow("y lower edge", cloud.lowerEdgeY());
236         this.addRow("y upper edge", cloud.upperEdgeY());
237         this.addRow("x mean", String.format("%.10f%n", cloud.meanX()));
238         this.addRow("y mean", String.format("%.10f%n", cloud.meanY()));
239         this.addRow("x rms", String.format("%.10f%n", cloud.rmsX()));
240         this.addRow("y rms", String.format("%.10f%n", cloud.rmsY()));
241     }

244      * Add rows to the info table from the state of a function.
245      *
246      * @param function the AIDA function object

248     private void addRows(final IFunction function) {

310     /**
311      * Get the title of an AIDA object. Unfortunately there is no base type with this information, so it is read
312      * manually from each possible type.
313      *
314      * @param object the AIDA object
315      * @return the title of the object from its title method or value of its <code>toString</code> method, if none
316      * exists
317      */
318     private String getObjectTitle(final Object object) {
319         if (object instanceof IBaseHistogram) {
320             return ((IBaseHistogram) object).title();
321         } else if (object instanceof IDataPointSet) {
322             return ((IDataPointSet) object).title();
323         } else if (object instanceof IFunction) {
324             return ((IFunction) object).title();
325         } else {
326             return object.toString();
327         }
328     }

330     /**
331      * Return <code>true</code> if the object is a valid AIDA object.
332      *
333      * @param object the object
334      * @return <code>true</code> if object is a valid AIDA object
335      */
336     private boolean isValidObject(final Object object) {
337         if (object == null) {
338             return false;
339         }
340         if (object instanceof IBaseHistogram || object instanceof IFunction || object instanceof IDataPointSet) {
341             return true;
342         }
343         return false;
344     }

346     /**
347      * Remove this as a listener on the current AIDA object.
348      */
349     private void removeListener() {
350         if (this.currentObject != null) {
351             if (this.currentObject instanceof AIDAObservable && !(this.currentObject instanceof IFunction)) {
352                 // Remove this object as an listener on the AIDA observable.
353                 ((AIDAObservable) this.currentObject).removeListener(this);
354             } else if (this.currentObject instanceof FunctionDispatcher) {
355                 // Remove this object as function listener.
356                 ((FunctionDispatcher) this.currentObject).removeFunctionListener(this);
357             }
358         }
359     }

361     /**
362      * Run the {@link #updateTable()} method on the Swing EDT.
363      */
364     private void runUpdateTable() {
365         SwingUtilities.invokeLater(new Runnable() {
366 
367             @Override
368             public void run() {
369                 PlotInfoPanel.this.updateTable();
370             }
371         });
372     }

374     /**
375      * Set the current AIDA object that backs this GUI, i.e. an IHistogram1D or other plot object etc.
376      *
377      * @param object the backing AIDA object
378      */
379     private synchronized void setCurrentObject(final Object object) {
380 
381         if (object == null) {
382             throw new IllegalArgumentException("The object arg is null!");
383         }
384 
385         if (object == this.currentObject) {
386             return;
387         }
388 
389         // Remove the AIDAListener from the previous object.
390         this.removeListener();
391 
392         // Set the current object reference.
393         this.currentObject = object;
394 
395         // Update the table immediately with information from the current object.
396         // We need to wait for this the first time, so we know the preferred size
397         // of the table GUI component when resizing the content pane.
398         this.updateTable();
399 
400         // Add an AIDAListener to the AIDA object via the AIDAObservable API.
401         this.addListener();
402     }

404     /**
405      * Set the current plotter region, which will rebuild the GUI accordingly.
406      *
407      * @param region The current plotter region.
408      */
409     synchronized void setCurrentRegion(final PlotterRegion region) {
410         if (region != this.currentRegion) {
411             this.currentRegion = region;
412             this.updateComboBox();
413             this.setCurrentObject(this.plotComboBox.getSelectedItem());
414             this.setupContentPane();
415         }
416     }

427     /**
428      * This method will be called when the backing AIDA object is updated and a state change is fired via the
429      * <code>AIDAObservable</code> API. The table is updated to reflect the new state of the object.
430      *
431      * @param evt the EventObject pointing to the backing AIDA object
432      */
433     @Override
434     public void stateChanged(final EventObject evt) {
435 
436         // Make a timer task for running the update.
437         final TimerTask task = new TimerTask() {
438 
439             @Override
440             public void run() {
441 
442                 // Is the state change from the current AIDAObservable?
443                 if (evt.getSource() != PlotInfoPanel.this.currentObject) {
444                     // Assume this means that a different AIDAObservable was selected in the GUI.
445                     return;
446                 }
447 
448                 // Update the table values on the Swing EDT.
449                 PlotInfoPanel.this.runUpdateTable();
450 
451                 // Set the observable to valid so subsequent state changes are received.
452                 ((AIDAObservable) PlotInfoPanel.this.currentObject).setValid(PlotInfoPanel.this);
453             }
454         };
455 
456         /*
457          * Schedule the task to run in ~0.5 seconds. If the Runnable runs immediately, somehow the observable state gets
458          * permanently set to invalid and additional state changes will not be received!
459          */
460         this.timer.schedule(task, 500);
461     }

463     /**
464      * Update the combo box contents with the plots from the current region.
465      */
466     private void updateComboBox() {
467         this.plotComboBox.removeAllItems();
468         final List<Object> objects = this.currentRegion.getPlottedObjects();
469         for (final Object object : objects) {
470             if (this.isValidObject(object)) {
471                 this.plotComboBox.addItem(object);
472             }
473         }
474     }
475 
476     /**
477      * Update the info table from the state of the current AIDA object.
478      */
479     private void updateTable() {
480         this.model.setRowCount(0);
481         if (this.currentObject instanceof IHistogram1D) {
482             this.addRows((IHistogram1D) this.currentObject);
483         } else if (this.currentObject instanceof IHistogram2D) {
484             this.addRows((IHistogram2D) this.currentObject);
485         } else if (this.currentObject instanceof ICloud2D) {
486             this.addRows((ICloud2D) this.currentObject);
487         } else if (this.currentObject instanceof ICloud1D) {
488             if (((ICloud1D) this.currentObject).isConverted()) {
489                 this.addRows(((ICloud1D) this.currentObject).histogram());
490             }
491         } else if (this.currentObject instanceof IFunction) {
492             this.addRows((IFunction) this.currentObject);
493         }
494     }
