3   import java.io.File;
4   
5   import org.apache.commons.cli.CommandLine;
6   import org.apache.commons.cli.CommandLineParser;
7   import org.apache.commons.cli.HelpFormatter;
8   import org.apache.commons.cli.Option;
9   import org.apache.commons.cli.Options;
10  import org.apache.commons.cli.ParseException;
11  import org.apache.commons.cli.PosixParser;
12  import org.hps.monitoring.application.model.Configuration;
13  
14  /**
15   * This is the front-end for running the monitoring application via a {@link #main(String[])} method.
16   */
17  public final class Main {
18  
19      /**
20       * The main method which starts the monitoring application, which is how users should start the application.
21       *
22       * @param args the command line arguments
23       */
24      public static void main(final String[] args) {
25  
26          // Set up command line parsing.
27          final Options options = new Options();
28          options.addOption(new Option("h", false, "Print help."));
29          options.addOption(new Option("c", true, "Load a properties file with configuration parameters."));
30          final CommandLineParser parser = new PosixParser();
31  
32          // Parse command line arguments.
33          final CommandLine cl;
34          try {
35              cl = parser.parse(options, args);
36          } catch (final ParseException e) {
37              throw new RuntimeException("Problem parsing command line options.", e);
38          }
39  
40          // Print help and exit.
41          if (cl.hasOption("h")) {
42              final HelpFormatter help = new HelpFormatter();
43              help.printHelp(" ", options);
44              System.exit(1);
45          }
46  
47          // Load the connection settings.
48          Configuration configuration = null;
49          if (cl.hasOption("c")) {
50              configuration = new Configuration(new File(cl.getOptionValue("c")));
51          }
52  
53          MonitoringApplication.create(configuration);
54      }
55  
56      /**
57       * Class constructor, which should not be used.
58       * <p>
59       * Call the {@link #main(String[])} method instead.
60       */
61      private Main() {
62          throw new UnsupportedOperationException("Do not instantiate this class.");
63      }
64  }
