114     /**

119 
120         /**
121          * A reference to the current {{@link #EventProcessing(Thread)}.
122          */
123         private final Thread processingThread;
124 
125         /**
126          * Class constructor.
127          *
128          * @param processingThread the current {{@link #EventProcessing(Thread)}
129          */
130         private SessionWatchdogThread(final Thread processingThread) {
131             this.processingThread = processingThread;
132         }

134         /**
135          * Run this thread, which will disconnect from the current session if the event processing ends for any reason.
136          */
137         @Override
138         public void run() {
139             try {
140                 // This thread waits on the event processing thread to die.
141                 this.processingThread.join();
142 
143                 // Activate a disconnect using the ActionEvent which is used by the disconnect button.
144                 EventProcessing.this.logger.finest("processing thread ended so automatic disconnect is happening");
145                 EventProcessing.this.application.actionPerformed(new ActionEvent(Thread.currentThread(), 0,
146                         Commands.DISCONNECT));
147 
148             } catch (final InterruptedException e) {
149                 EventProcessing.this.logger.finest("SessionWatchdogThread got interrupted");
150                 // This happens when the thread is interrupted by the user pressing the disconnect button.
151             }
152         }
153     }

309         try {
310             // Create a new instance of the builder class.

369     /**
370      * Interrupt and join the processing watchdog thread.
371      * <p>
372      * This will happen if there is a user requested disconnect from pushing the button in the GUI.
373      */
374     synchronized void killWatchdogThread() {
375         // Is the session watchdog thread not null?
376         if (this.sessionState.sessionWatchdogThread != null) {
377             this.logger.finest("killing watchdog thread ...");
378             // Is the thread still alive?
379             if (this.sessionState.sessionWatchdogThread.isAlive()) {
380                 // Interrupt the thread which should cause it to stop.
381                 this.sessionState.sessionWatchdogThread.interrupt();
382                 try {
383                     // This should always work once the thread is interrupted.
384                     this.sessionState.sessionWatchdogThread.join();
385                 } catch (final InterruptedException e) {
386                 }
387             }
388             // Set the thread object to null.
389             this.sessionState.sessionWatchdogThread = null;
390             this.logger.finest("watchdog thread killed");
391         }
392     }

446     /**
447      * Setup LCSim event processing from the global {@link org.hps.monitoring.model.ConfigurationModel} object.
448      *
449      * @param configurationModel the global @link org.hps.monitoring.model.ConfigurationModel} object
450      */
451     private void setupLcsim(final ConfigurationModel configurationModel) {
452         this.logger.info("setting up lcsim");
453 
454         // Get steering resource or file as a String parameter.
455         String steering = null;
456         final SteeringType steeringType = configurationModel.getSteeringType();
457         if (steeringType.equals(SteeringType.FILE)) {
458             steering = configurationModel.getSteeringFile();
459         } else {
460             steering = configurationModel.getSteeringResource();
461             if (!steering.startsWith("/")) {
462                 steering = "/" + steering;
463             }
464         }
465 
466         this.logger.config("set steering " + steering + " with type "
467                 + (steeringType == SteeringType.RESOURCE ? "RESOURCE" : "FILE"));
468 
469         try {
470             // Create the job manager. A new conditions manager is instantiated from this call but not configured.
471             this.sessionState.jobManager = new JobManager();
472 
473             // Reset the conditions system because if run and detector name are the same in multiple sessions, it
474             // will not be activated in subsequent sessions after the first one.
475             this.logger.config("Resetting the conditions system for new session");
476             DatabaseConditionsManager.reset();
477             
478             // Setup class for conditions system.
479             DatabaseConditionsManagerSetup conditions = new DatabaseConditionsManagerSetup();
480 
481             // Setup the event builder to translate from EVIO to LCIO.
482             LCSimEventBuilder eventBuilder = this.createEventBuilder(configurationModel);
483             conditions.addConditionsListener(eventBuilder);
484 
485             // Add extra conditions listeners.
486             for (final ConditionsListener conditionsListener : this.sessionState.conditionsListeners) {
487                 this.logger.config("Adding conditions listener " + conditionsListener.getClass().getName());
488                 conditions.addConditionsListener(conditionsListener);
489             }
490 
491             // Add detector alias.
492             if (configurationModel.hasValidProperty(ConfigurationModel.DETECTOR_ALIAS_PROPERTY)) {
493                 conditions.addAlias(configurationModel.getDetectorName(),
494                         "file://" + configurationModel.getDetectorAlias());
495                 this.logger.config("Added detector alias " + configurationModel.getDetectorAlias() + " for "
496                         + configurationModel.getDetectorName());
497             }
498 
499             // Add conditions tag.
500             if (configurationModel.hasValidProperty(ConfigurationModel.CONDITIONS_TAG_PROPERTY)
501                     && !configurationModel.getConditionsTag().equals("")) {
502                 Set<String> tags = new HashSet<String>();
503                 tags.add(configurationModel.getConditionsTag());
504                 this.logger.config("Added conditions tag " + configurationModel.getConditionsTag());
505                 conditions.setTags(tags);
506             }
507 
508             // Set user specified job number.
509             if (configurationModel.hasValidProperty(ConfigurationModel.USER_RUN_NUMBER_PROPERTY)) {
510                 final int userRun = configurationModel.getUserRunNumber();
511                 this.logger.config("User run number set to " + userRun);
512                 conditions.setRun(userRun);
513             }
514 
515             // Set detector name.
516             conditions.setDetectorName(configurationModel.getDetectorName());
517 
518             // Freeze the conditions system to ignore run numbers from event data.
519             if (configurationModel.hasPropertyKey(ConfigurationModel.FREEZE_CONDITIONS_PROPERTY)) {
520                 this.logger.config("user configured to freeze conditions system");
521                 conditions.setFreeze(configurationModel.getFreezeConditions());
522             }
523 
524             // Register the configured conditions settings with the job manager.
525             this.sessionState.jobManager.setConditionsSetup(conditions);
526 
527             // Configure the job manager for the XML steering.
528             this.sessionState.jobManager.setDryRun(true);
529             if (steeringType == SteeringType.RESOURCE) {
530                 this.setupSteeringResource(steering);
531             } else if (steeringType.equals(SteeringType.FILE)) {
532                 this.setupSteeringFile(steering);
533             }
534             
535             // Activate conditions system if run number was provided as hard-coded value.
536             if (configurationModel.hasValidProperty(ConfigurationModel.USER_RUN_NUMBER_PROPERTY)) {
537                 
538                 // Configure the conditions manager.
539                 conditions.configure();
540                 
541                 // Activate conditions system with supplied detector and run number.
542                 this.logger.config("Activating conditions with detector <" 
543                         + configurationModel.getDetectorName() + "> and run "
544                         + configurationModel.getUserRunNumber());
545                 conditions.setup();
546 
547                 // Post-initialize  the conditions system, which may freeze it.
548                 this.sessionState.jobManager.getConditionsSetup().postInitialize();
549             } else {
550                 this.logger.config("Conditions system not activated. No user run number was supplied.");
551             }
552 
553             this.logger.info("lcsim setup was successful");
554 
555         } catch (final Throwable t) {
556             throw new RuntimeException("Error setting up LCSim.", t);
557         }
558     }

560     /**
561      * Setup the {@link org.hps.record.composite.CompositeLoop} from the global
562      * {@link org.hps.monitoring.model.ConfigurationModel} object.
563      *
564      * @param configurationModel the global {@link org.hps.monitoring.model.ConfigurationModel} object
565      */
566     private void setupLoop(final ConfigurationModel configurationModel) {
567 
568         this.logger.config("setting up record loop ...");
569 
570         // Initialize the loop from the ConfigurationModel.
571         final CompositeLoopConfiguration loopConfig = new CompositeLoopConfiguration()
572                 .setStopOnEndRun(configurationModel.getDisconnectOnEndRun())
573                 .setStopOnErrors(configurationModel.getDisconnectOnError())
574                 .setDataSourceType(configurationModel.getDataSourceType())
575                 .setProcessingStage(configurationModel.getProcessingStage())
576                 .setEtConnection(this.sessionState.connection).setFilePath(configurationModel.getDataSourcePath())
577                 .setLCSimEventBuilder(this.sessionState.eventBuilder);
578 
579         this.logger.config("data src path " + configurationModel.getDataSourcePath() + " and type "
580                 + configurationModel.getDataSourceType());
581 
582         // Set the max events.
583         if (configurationModel.hasValidProperty(ConfigurationModel.MAX_EVENTS_PROPERTY)) {
584             final long maxEvents = configurationModel.getMaxEvents();
585             if (maxEvents > 0L) {
586                 loopConfig.setMaxRecords(maxEvents);
587             }
588         }
589 
590         // Add all Drivers from the JobManager.
591         for (final Driver driver : this.sessionState.jobManager.getDriverExecList()) {
592             loopConfig.add(driver);
593             this.logger.config("added Driver " + driver.getName());
594         }
595 
596         // Using ET server?
597         if (configurationModel.getDataSourceType().equals(DataSourceType.ET_SERVER)) {
598 
599             // ET system monitor.
600             this.logger.config("added EtSystemMonitor");
601             loopConfig.add(new EtSystemMonitor());
602 
603             // ET system strip charts.
604             this.logger.config("added EtSystemStripCharts");
605             loopConfig.add(new EtSystemStripCharts());
606         }
607 
608         // Add extra CompositeRecordProcessors to the loop config.
609         for (final CompositeRecordProcessor processor : this.sessionState.processors) {
610             loopConfig.add(processor);
611             this.logger.config("added extra processor " + processor.getClass().getSimpleName());
612         }
613 
614         // Add extra Drivers to the loop config.
615         for (final Driver driver : this.sessionState.drivers) {
616             loopConfig.add(driver);
617             this.logger.config("added extra Driver " + driver.getName());
618         }
619 
620         // Enable conditions activation from EVIO; not needed if conditions are frozen for the job.
621         if (!DatabaseConditionsManager.getInstance().isFrozen()) {
622             loopConfig.add(new EvioDetectorConditionsProcessor(configurationModel.getDetectorName()));
623             this.logger.config("added EvioDetectorConditionsProcessor to job with detector "
624                     + configurationModel.getDetectorName());
625         } else {
626             this.logger.config("Conditions activation from EVIO is disabled.");
627         }
628 
629         // Create the CompositeLoop with the configuration.
630         this.sessionState.loop = new CompositeLoop(loopConfig);
631 
632         this.logger.config("Record loop is setup.");
633     }

661      * <p>

664     synchronized void start() {

675 
676         this.logger.fine("started event processing threads");

679     /**
680      * Stop the current session, which will bring down the current ET client connection and activate end-of-job and
681      * end-of-run hooks on all registered event processors.
682      * <p>
683      * This method is called externally by the app to stop an event processing session e.g. from action event handling.
684      */
685     synchronized void stop() {
686 
687         // Kill session watchdog thread.
688         this.killWatchdogThread();
689 
690         // Wake up all ET stations to unblock the system and make sure stations are detached properly.
691         if (this.usingEtServer()) {
692             this.wakeUpEtStations();
693         }
694 
695         // Stop the event processing now that ET system should be unblocked.
696         this.logger.finer("sending STOP command to loop ...");
697         this.sessionState.loop.execute(Command.STOP);
698         this.logger.finer("loop got STOP command");
699 
700         this.logger.finer("processing thread is alive: " + this.sessionState.processingThread.isAlive());
701 
702         try {
703             // Give the event processing thread a chance to end cleanly.
704             this.logger.finer("waiting for event processing thread to end ...");
705             this.sessionState.processingThread.join(5000);
706             this.logger.finer("processing thread is alive: " + this.sessionState.processingThread.isAlive());
707             // this.logger.finer("event processing thread ended cleanly");
708         } catch (final InterruptedException e) {
709             e.printStackTrace();
710         }
711 
712         try {
713             this.logger.finer("processing thread is alive: " + this.sessionState.processingThread.isAlive());
714             // In this case the thread needs to be interrupted and then joined.
715             this.logger.finer("interrupting event processing thread");
716             this.sessionState.processingThread.interrupt();
717             this.sessionState.processingThread.join();
718             this.logger.finer("event processing thread ended after interrupt");
719         } catch (final InterruptedException e) {
720             e.printStackTrace();
721         }
722 
723         // Notify of last error that occurred in event processing.
724         if (this.sessionState.loop.getLastError() != null) {
725             // Log the error.
726             this.errorHandler.setError(this.sessionState.loop.getLastError()).log();
727         }
728 
729         // Invalidate the loop.
730         this.sessionState.loop = null;
731 
732         // Disconnect from the ET system.
733         this.disconnect();
734 
735         // Invalidate the event processing object so it is unusable now.
736         this.invalidate();
737     }

741      *
742      * @return <code>true</code> if using an ET server in the current session
743      */
744     private boolean usingEtServer() {
745         return this.sessionState.usingEtServer;

748     /**
749      * Wake up all ET stations associated with the event processing.
750      */
751     private void wakeUpEtStations() {
752         if (this.sessionState.connection != null) {
753             this.logger.fine("waking up ET stations ...");
754 
755             // Wake up secondary ET stations.
756             for (final EtStationThread station : this.sessionState.stations) {
757 
758                 // First unblock if in ET call.
759                 station.wakeUp();
760 
761                 // Next interrupt so that it will definitely stop.
762                 station.interrupt();
763             }
764 
765             // Wait for station threads to die after being woken up.
766             while (this.sessionState.stationThreadGroup.activeCount() != 0) {
767                 this.logger.finest("waiting for station threads to die ...");
768                 final Object lock = new Object();
769                 synchronized (lock) {
770                     try {
771                         lock.wait(500);
772                     } catch (final InterruptedException e) {
773                         e.printStackTrace();
774                     }
775                 }
776             }
777 
778             this.sessionState.stationThreadGroup.destroy();
779 
780             this.logger.finest("station threads destroyed");
781 
782             // Wake up the primary ET station doing the event processing.
783             this.logger.finest("waking up event processing station ...");
784             try {
785                 this.sessionState.connection.getEtSystem().wakeUpAll(this.sessionState.connection.getEtStation());
786                 this.logger.finest("event processing station was woken up");
787             } catch (IOException | EtException | EtClosedException e) {
788                 e.printStackTrace();
789             }
790 
791             this.logger.finest("ET stations all woken up");
792         }
793     }
