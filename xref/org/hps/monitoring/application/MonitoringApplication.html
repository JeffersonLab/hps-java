243     /**
244      * Instantiate and show the monitoring application with the given
245      * configuration.
246      *
247      * @param userConfiguration the Configuration object containing application
248      * settings
249      */
250     MonitoringApplication(final Configuration userConfiguration) {
251 
252         try {
253 
254             // Setup the main GUI component.
255             this.frame = new MonitoringApplicationFrame(this);
256 
257             // Add window listener to perform clean shutdown.
258             this.frame.addWindowListener(new WindowAdapter() {
259 
260                 /**
261                  * Activate cleanup when window closes.
262                  *
263                  * @param e the window event
264                  */
265                 @Override
266                 public void windowClosed(final WindowEvent e) {
267                     exit();
268                 }
269             });
270 
271             // Setup the error handler.
272             this.errorHandler = new ErrorHandler(this.frame, LOGGER);
273 
274             // Add this class as a listener on the configuration model.
275             this.configurationModel.addPropertyChangeListener(this);
276 
277             // Setup the logger.
278             setupLogger();
279 
280             // Setup AIDA plotting and connect it to the GUI.
281             setupAida();
282 
283             // Load the default configuration.
284             loadConfiguration(new Configuration(DEFAULT_CONFIGURATION), false);
285 
286             if (userConfiguration != null) {
287                 // Load user configuration to supplement default settings.
288                 loadConfiguration(userConfiguration, true);
289             }
290 
291             // Enable the GUI after initialization.
292             this.frame.setEnabled(true);
293 
294             LOGGER.info("Monitoring app initialized successfully.");
295 
296         } catch (final Exception e) {
297             // Initialization failed so print info and die.
298             System.err.println("ERROR: MonitoringApplication failed to initialize!");
299             DialogUtil.showErrorDialog(null, "Error Starting Monitoring Application",
300                     "Monitoring application failed to initialize.");
301             e.printStackTrace();
302             System.exit(1);
303         }
304     }

306     /**
307      * The primary action handler for the application.
308      *
309      * @param e the {@link java.awt.ActionEvent} to handle
310      */
311     @Override
312     public void actionPerformed(final ActionEvent e) {
313         final String command = e.getActionCommand();
314         if (Commands.CONNECT.equals(command)) {
315             startSession();
316         } else if (Commands.DISCONNECT.equals(command)) {
317             runDisconnectThread();
318         } else if (Commands.SAVE_PLOTS.equals(command)) {
319             savePlots();
320         } else if (Commands.SAVE_ROOT_AND_PDF.equals(command)) {
321             savePlotsRootPdf();
322         } else if (Commands.EXIT.equals(command)) {
323             // This will trigger the window closing action that cleans everything up.
324             this.frame.dispose();
325         } else if (Commands.PAUSE.equals(command)) {
326             this.processing.pause();
327         } else if (Commands.NEXT.equals(command)) {
328             this.processing.next();
329         } else if (Commands.RESUME.equals(command)) {
330             this.processing.resume();
331         } else if (Commands.SHOW_SETTINGS.equals(command)) {
332             showSettingsDialog();
333         } else if (Commands.LOAD_SETTINGS.equals(command)) {
334             loadSettings();
335         } else if (Commands.SAVE_SETTINGS.equals(command)) {
336             saveSettings();
337         } else if (Commands.CLEAR_PLOTS.equals(command)) {
338             clearPlots();
339         } else if (Commands.LOAD_DEFAULT_SETTINGS.equals(command)) {
340             loadDefaultSettings();
341         } else if (Commands.OPEN_FILE.equals(command)) {
342             openFile();
343         } else if (Commands.DEFAULT_WINDOW.equals(command)) {
344             restoreDefaultWindow();
345         } else if (Commands.MAXIMIZE_WINDOW.equals(command)) {
346             maximizeWindow();
347         } else if (Commands.MINIMIZE_WINDOW.equals(command)) {
348             minimizeWindow();
349         } else if (Commands.CLOSE_FILE.equals(command)) {
350             closeFile();
351         } else if (Commands.SAVE_SCREENSHOT.equals(command)) {
352             saveScreenshot();
353         } else if (Commands.SAVE_LOG_TABLE.equals(command)) {
354             saveLogTable();
355         } else if (Commands.CLEAR_LOG_TABLE.equals(command)) {
356             getLogRecordModel().clear();
357         } else if (Commands.LOG_TO_FILE.equals(command)) {
358             chooseLogFile();
359         } else if (Commands.LOG_TO_TERMINAL.equals(command)) {
360             logToTerminal();
361         } else if (Commands.PLOT_POPUP.equals(command)) {
362             this.plotPopupEnabled = ((JCheckBox) e.getSource()).isSelected();
363         }

533         }
534     }

561     /**
562      * Redirect <code>System.out</code> and <code>System.err</code> to a file.
563      *
564      * @param file the output log file
565      * @throws FileNotFoundException if the file does not exist
566      */
567     private void logToFile(final File file) {
568         try {
569 
570             // Create the output file stream.
571             final PrintStream fileStream = new PrintStream(new FileOutputStream(file.getPath()));
572             System.setOut(fileStream);
573             System.setErr(fileStream);
574 
575             // Flush the current handler, but do NOT close here or System.out gets clobbered!
576             this.streamHandler.flush();
577 
578             // Replace the current handler with one using the file stream.
579             LOGGER.removeHandler(this.streamHandler);
580             this.streamHandler = new MonitoringApplicationStreamHandler(fileStream);
581             this.streamHandler.setLevel(LOGGER.getLevel());
582             LOGGER.addHandler(this.streamHandler);
583 
584             // Set the properties on the model.
585             this.configurationModel.setLogFileName(file.getPath());
586             this.configurationModel.setLogToFile(true);
587 
588             LOGGER.info("Saving log messages to " + this.configurationModel.getLogFileName());
589             DialogUtil.showInfoDialog(this.frame, "Logging to File", "Log messages redirected to file" + '\n'
590                     + this.configurationModel.getLogFileName());
591 
592         } catch (final FileNotFoundException e) {
593             this.errorHandler.setError(e).log().showErrorDialog();
594         }
595     }

597     /**
598      * Send <code>System.out</code> and <code>System.err</code> back to the
599      * terminal, e.g. if they were previously sent
600      * to a file.
601      */
602     private void logToTerminal() {
603 
604         // Reset System.out and err back to original streams.
605         System.setOut(MonitoringApplication.SYS_OUT);
606         System.setErr(MonitoringApplication.SYS_ERR);
607 
608         // Flush and close the current handler, which is using a file stream.
609         this.streamHandler.flush();
610         this.streamHandler.close();
611 
612         // Replace the handler with the one printing to the terminal.
613         LOGGER.removeHandler(this.streamHandler);
614         this.streamHandler = new MonitoringApplicationStreamHandler(System.out);
615         this.streamHandler.setLevel(LOGGER.getLevel());
616         LOGGER.addHandler(this.streamHandler);
617 
618         LOGGER.log(Level.INFO, "log messages redirected to terminal");
619 
620         // Update the model to indicate logging to file has been disabled.
621         this.configurationModel.setLogToFile(false);
622 
623         DialogUtil.showInfoDialog(this.frame, "Log to Terminal", "Log messages will be sent to the terminal.");
624     }

640     /**
641      * Open a file data source using a <code>JFileChooser</code>.
642      */
643     private void openFile() {
644         final JFileChooser fc = new JFileChooser(System.getProperty("user.dir"));
645         fc.setAcceptAllFileFilterUsed(false);
646         fc.addChoosableFileFilter(EVIO_FILTER);
647         fc.addChoosableFileFilter(LCIO_FILTER);
648         fc.setDialogTitle("Select Data File");
649         final int r = fc.showDialog(this.frame, "Select ...");
650         if (r == JFileChooser.APPROVE_OPTION) {
651 
652             // Set data source path.
653             final String filePath = fc.getSelectedFile().getPath();
654 
655             // Set data source type.
656             final FileFilter filter = fc.getFileFilter();
657             DataSourceType type = null;
658             if (filter == LCIO_FILTER) {
659                 type = DataSourceType.LCIO_FILE;
660             } else if (filter == EVIO_FILTER) {
661                 type = DataSourceType.EVIO_FILE;
662             } else {
663                 // This should never happen.
664                 throw new RuntimeException();
665             }
666 
667             this.configurationModel.setDataSourcePath(filePath);
668             this.configurationModel.setDataSourceType(type);
669 
670             this.configurationModel.addRecentFile(filePath);
671 
672             LOGGER.config("set new data source " + filePath + " with type " + type);
673         }
674     }

736     /**
737      * Save plots to an AIDA, ROOT or PDF file using a file chooser.
738      */
739     private void savePlots() {
740         final JFileChooser fc = new JFileChooser();
741         fc.addChoosableFileFilter(new FileNameExtensionFilter("ROOT file", "root"));
742         final FileFilter filter = new FileNameExtensionFilter("PDF file", "pdf");
743         fc.addChoosableFileFilter(filter);
744         fc.addChoosableFileFilter(new FileNameExtensionFilter("AIDA file", "aida"));
745         fc.setAcceptAllFileFilterUsed(false);
746         fc.setFileSelectionMode(JFileChooser.FILES_ONLY);
747         fc.setFileFilter(filter);

766                     LOGGER.info("saved plots to " + fileName);
767                     DialogUtil.showInfoDialog(this.frame, "Plots Saved", "Plots were successfully saved to " + '\n'
768                             + fileName);
769                 } catch (final IOException e) {
770                     this.errorHandler.setError(e).setMessage("Error Saving Plots").printStackTrace().log()
771                             .showErrorDialog();
772                 }

979     /**
980      * Start a new monitoring session.
981      */
982     private synchronized void startSession() {
983 
984         LOGGER.info("starting new session");
985 
986         try {
987 
988             // Reset the plot panel and global AIDA state.
989             resetPlots();
990 
991             // The system status registry is cleared here before any event processors
992             // which might create a SystemStatus are added to the event processing chain
993             // e.g. an LCSim Driver, etc.
994             SystemStatusRegistry.getSystemStatusRegistery().clear();
995 
996             // List of extra composite record processors including the updater for the RunPanel.
997             final List<CompositeRecordProcessor> processors = new ArrayList<CompositeRecordProcessor>();
998             processors.add(this.frame.getEventDashboard().new EventDashboardUpdater());
999 
1000             // Add Driver to update the trigger diagnostics tables.
1001             final List<Driver> drivers = new ArrayList<Driver>();
1002 
1003             // Add listener to push conditions changes to conditions panel.
1004             final List<ConditionsListener> conditionsListeners = new ArrayList<ConditionsListener>();
1005 
1006             // Instantiate the event processing wrapper.
1007             this.processing = new EventProcessing(this, processors, drivers, conditionsListeners);
1008 
1009             // Connect to the ET system, if applicable.
1010             this.processing.connect();
1011 
1012             // Configure event processing from the global application settings, including setup of record loop.
1013             LOGGER.info("setting up event processing on source " + this.configurationModel.getDataSourcePath()
1014                     + " with type " + this.configurationModel.getDataSourceType());
1015             this.processing.setup(this.configurationModel);
1016 
1017             // Setup the system status monitor table.
1018             setupSystemStatusMonitor();
1019 
1020             // Start the event processing thread.
1021             this.processing.start();
1022 
1023             LOGGER.info("Event processing session started.");
1024 
1025         } catch (final Exception e) {
1026 
1027             // Disconnect from the ET system.
1028             this.processing.disconnect();
1029 
1030             // Log the error that occurred and show a pop up dialog.
1031             this.errorHandler
1032                     .setError(e)
1033                     .log()
1034                     .printStackTrace()
1035                     .showErrorDialog(
1036                             "There was an error while starting the session." + '\n' + "See the log for details.",
1037                             "Session Error");
1038 
1039             LOGGER.severe("Failed to start event processing.");
1040         }
1041     }
