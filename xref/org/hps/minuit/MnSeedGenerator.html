9   class MnSeedGenerator implements MinimumSeedGenerator {
10  
11      @Override
12      public MinimumSeed generate(MnFcn fcn, GradientCalculator gc, MnUserParameterState st, MnStrategy stra) {
13          int n = st.variableParameters();
14          MnMachinePrecision prec = st.precision();
15  
16          // initial starting values
17          MnAlgebraicVector x = new MnAlgebraicVector(n);
18          for (int i = 0; i < n; i++) {
19              x.set(i, st.intParameters().get(i));
20          }
21          double fcnmin = fcn.valueOf(x);
22          MinimumParameters pa = new MinimumParameters(x, fcnmin);
23  
24          FunctionGradient dgrad;
25          if (gc instanceof AnalyticalGradientCalculator) {
26              InitialGradientCalculator igc = new InitialGradientCalculator(fcn, st.trafo(), stra);
27              FunctionGradient tmp = igc.gradient(pa);
28              FunctionGradient grd = gc.gradient(pa);
29              dgrad = new FunctionGradient(grd.grad(), tmp.g2(), tmp.gstep());
30  
31              if (((AnalyticalGradientCalculator) gc).checkGradient()) {
32                  boolean good = true;
33                  HessianGradientCalculator hgc = new HessianGradientCalculator(fcn, st.trafo(), new MnStrategy(2));
34                  Pair<FunctionGradient, MnAlgebraicVector> hgrd = hgc.deltaGradient(pa, dgrad);
35                  for (int i = 0; i < n; i++) {
36                      if (Math.abs(hgrd.first.grad().get(i) - grd.grad().get(i)) > hgrd.second.get(i)) {
37                          logger.log(Level.INFO, "gradient discrepancy of external parameter {0} (internal parameter {1}) too large.", new Object[]{st.trafo().extOfInt(i), i});
38                          good = false;
39                      }
40                  }
41                  if (!good) {
42                      logger.info("Minuit does not accept user specified gradient. To force acceptance, override 'virtual bool checkGradient() const' of FCNGradientBase.h in the derived class.");
43                      assert (good);
44                  }
45              }
46          } else {
47              dgrad = gc.gradient(pa);
48          }
49          MnAlgebraicSymMatrix mat = new MnAlgebraicSymMatrix(n);
50          double dcovar = 1.;
51          if (st.hasCovariance()) {
52              for (int i = 0; i < n; i++) {
53                  for (int j = i; j < n; j++) {
54                      mat.set(i, j, st.intCovariance().get(i, j));
55                  }
56              }
57              dcovar = 0.;
58          } else {
59              for (int i = 0; i < n; i++) {
60                  mat.set(i, i, (Math.abs(dgrad.g2().get(i)) > prec.eps2() ? 1. / dgrad.g2().get(i) : 1.));
61              }
62          }
63          MinimumError err = new MinimumError(mat, dcovar);
64          double edm = new VariableMetricEDMEstimator().estimate(dgrad, err);
65          MinimumState state = new MinimumState(pa, err, dgrad, edm, fcn.numOfCalls());
66  
67          if (NegativeG2LineSearch.hasNegativeG2(dgrad, prec)) {
68              if (gc instanceof AnalyticalGradientCalculator) {
69                  Numerical2PGradientCalculator ngc = new Numerical2PGradientCalculator(fcn, st.trafo(), stra);
70                  state = NegativeG2LineSearch.search(fcn, state, ngc, prec);
71              } else {
72                  state = NegativeG2LineSearch.search(fcn, state, gc, prec);
73              }
74          }
75  
76          if (stra.strategy() == 2 && !st.hasCovariance()) {
77              //calculate full 2nd derivative
78              MinimumState tmp = new MnHesse(stra).calculate(fcn, state, st.trafo(), 0);
79              return new MinimumSeed(tmp, st.trafo());
80          }
81  
82          return new MinimumSeed(state, st.trafo());
83      }
84      private static final Logger logger = Logger.getLogger(MnSeedGenerator.class.getName());
85  }
