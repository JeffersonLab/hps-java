6   /**
7    * Result of the minimization.
8    * <p>
9    * The FunctionMinimum is the output of the minimizers and contains the
10   * minimization result. The methods
11   * <ul>
12   * <li>userState(),
13   * <li>userParameters() and
14   * <li>userCovariance()
15   * </ul>
16   * are provided. These can be used as new input to a new minimization after some
17   * manipulation. The parameters and/or the FunctionMinimum can be printed using
18   * the toString() method or the MnPrint class.
19   *
20   * @see #userState()
21   * @see #userParameters()
22   * @see #userCovariance()
23   * @see MnPrint
24   */
25  public class FunctionMinimum {
26  
27      FunctionMinimum(MinimumSeed seed, double up) {
28          theSeed = seed;
29          theStates = new ArrayList<MinimumState>();
30          theStates.add(new MinimumState(seed.parameters(), seed.error(), seed.gradient(), seed.parameters().fval(), seed.nfcn()));
31          theErrorDef = up;
32          theUserState = new MnUserParameterState();
33      }
34  
35      FunctionMinimum(MinimumSeed seed, List<MinimumState> states, double up) {
36          theSeed = seed;
37          theStates = states;
38          theErrorDef = up;
39          theUserState = new MnUserParameterState();
40      }
41  
42      FunctionMinimum(MinimumSeed seed, List<MinimumState> states, double up, MnReachedCallLimit x) {
43          theSeed = seed;
44          theStates = states;
45          theErrorDef = up;
46          theReachedCallLimit = true;
47          theUserState = new MnUserParameterState();
48      }
49  
50      FunctionMinimum(MinimumSeed seed, List<MinimumState> states, double up, MnAboveMaxEdm x) {
51          theSeed = seed;
52          theStates = states;
53          theErrorDef = up;
54          theAboveMaxEdm = true;
55          theReachedCallLimit = false;
56          theUserState = new MnUserParameterState();
57      }
58  
59      // why not
60      void add(MinimumState state) {
61          theStates.add(state);
62      }
63  
64      MinimumSeed seed() {
65          return theSeed;
66      }
67  
68      List<MinimumState> states() {
69          return theStates;
70      }
71  
72      /**
73       * user representation of state at minimum
74       */
75      public MnUserParameterState userState() {
76          if (!theUserState.isValid()) {
77              theUserState = new MnUserParameterState(state(), errorDef(), seed().trafo());
78          }
79          return theUserState;
80      }
81  
82      public MnUserParameters userParameters() {
83          if (!theUserState.isValid()) {
84              theUserState = new MnUserParameterState(state(), errorDef(), seed().trafo());
85          }
86          return theUserState.parameters();
87      }
88  
89      public MnUserCovariance userCovariance() {
90          if (!theUserState.isValid()) {
91              theUserState = new MnUserParameterState(state(), errorDef(), seed().trafo());
92          }
93          return theUserState.covariance();
94      }
95  
96      private MinimumState lastState() {
97          return theStates.get(theStates.size() - 1);
98      }
99  
100     // forward interface of last state
101     MinimumState state() {
102         return lastState();
103     }
104 
105     MinimumParameters parameters() {
106         return lastState().parameters();
107     }
108 
109     MinimumError error() {
110         return lastState().error();
111     }
112 
113     FunctionGradient grad() {
114         return lastState().gradient();
115     }
116 
117     /**
118      * Returns the function value at the minimum.
119      */
120     public double fval() {
121         return lastState().fval();
122     }
123 
124     /**
125      * returns the expected vertical distance to the minimum (EDM)
126      */
127     public double edm() {
128         return lastState().edm();
129     }
130 
131     /**
132      * returns the total number of function calls during the minimization.
133      */
134     public int nfcn() {
135         return lastState().nfcn();
136     }

138     public double errorDef() {
139         return theErrorDef;
140     }
141 
142     /**
143      * In general, if this returns <CODE>true</CODE>, the minimizer did find a
144      * minimum without running into troubles. However, in some cases a minimum
145      * cannot be found, then the return value will be <CODE>false</CODE>.
146      * Reasons for the minimization to fail are
147      * <ul>
148      * <li>the number of allowed function calls has been exhausted</ul>
149      * <li>the minimizer could not improve the values of the parameters (and
150      * knowing that it has not converged yet)</ul>
151      * <li>a problem with the calculation of the covariance matrix</ul>
152      * </ul>
153      * Additional methods for the analysis of the state at the minimum are
154      * provided.
155      */
156     public boolean isValid() {
157         return state().isValid() && !isAboveMaxEdm() && !hasReachedCallLimit();
158     }
159 
160     boolean hasValidParameters() {
161         return state().parameters().isValid();
162     }
163 
164     boolean hasValidCovariance() {
165         return state().error().isValid();
166     }
167 
168     boolean hasAccurateCovar() {
169         return state().error().isAccurate();
170     }
171 
172     boolean hasPosDefCovar() {
173         return state().error().isPosDef();
174     }
175 
176     boolean hasMadePosDefCovar() {
177         return state().error().isMadePosDef();
178     }
179 
180     boolean hesseFailed() {
181         return state().error().hesseFailed();
182     }
183 
184     boolean hasCovariance() {
185         return state().error().isAvailable();
186     }
187 
188     boolean isAboveMaxEdm() {
189         return theAboveMaxEdm;
190     }
191 
192     boolean hasReachedCallLimit() {
193         return theReachedCallLimit;
194     }
195 
196     public String toString() {
197         return MnPrint.toString(this);
198     }
