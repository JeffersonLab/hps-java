3   /**
4    * Calculates and the eigenvalues of the user covariance matrix
5    * MnUserCovariance.
6    *
7    */
8   public class MnEigen {
9   
10      /* Calculate eigenvalues of the covariance matrix.
11       * Will perform the calculation of the eigenvalues of the covariance matrix
12       * and return the result in the form of a double array.
13       * The eigenvalues are ordered from the smallest to the largest eigenvalue.
14       */
15      public static double[] eigenvalues(MnUserCovariance covar) {
16          MnAlgebraicSymMatrix cov = new MnAlgebraicSymMatrix(covar.nrow());
17          for (int i = 0; i < covar.nrow(); i++) {
18              for (int j = i; j < covar.nrow(); j++) {
19                  cov.set(i, j, covar.get(i, j));
20              }
21          }
22  
23          MnAlgebraicVector eigen = cov.eigenvalues();
24          return eigen.data();
25      }
26  }
