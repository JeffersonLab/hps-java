8   /**
9    * API class for Contours error analysis (2-dim errors). Minimization has to be
10   * done before and minimum must be valid. Possibility to ask only for the points
11   * or the points and associated Minos errors.
12   *
13   */
14  public class MnContours {
15  
16      /**
17       * construct from FCN + minimum
18       */
19      public MnContours(FCNBase fcn, FunctionMinimum min) {
20          this(fcn, min, MnApplication.DEFAULT_STRATEGY);
21      }
22  
23      /**
24       * construct from FCN + minimum + strategy
25       */
26      public MnContours(FCNBase fcn, FunctionMinimum min, int stra) {
27          this(fcn, min, new MnStrategy(stra));
28      }

30      /**
31       * construct from FCN + minimum + strategy
32       */
33      public MnContours(FCNBase fcn, FunctionMinimum min, MnStrategy stra) {
34          theFCN = fcn;
35          theMinimum = min;
36          theStrategy = stra;
37      }
38  
39      public List<Point> points(int px, int py) {
40          return points(px, py, 1);
41      }
42  
43      public List<Point> points(int px, int py, double errDef) {
44          return points(px, py, errDef, 20);
45      }
46  
47      /**
48       * Calculates one function contour of FCN with respect to parameters parx
49       * and pary. The return value is a list of (x,y) points. FCN minimized
50       * always with respect to all other n - 2 variable parameters (if any).
51       * MINUIT will try to find n points on the contour (default 20). To
52       * calculate more than one contour, the user needs to set the error
53       * definition in its FCN to the appropriate value for the desired confidence
54       * level and call this method for each contour.
55       */
56      public List<Point> points(int px, int py, double errDef, int npoints) {
57          ContoursError cont = contour(px, py, errDef, npoints);
58          return cont.points();
59      }
60  
61      public ContoursError contour(int px, int py) {
62          return contour(px, py, 1);
63      }
64  
65      public ContoursError contour(int px, int py, double errDef) {
66          return contour(px, py, errDef, 20);
67      }
68  
69      /**
70       * Causes a CONTOURS error analysis and returns the result in form of
71       * ContoursError. As a by-product ContoursError keeps the MinosError
72       * information of parameters parx and pary. The result ContoursError can be
73       * easily printed using MnPrint or toString().
74       */
75      public ContoursError contour(int px, int py, double errDef, int npoints) {
76          errDef *= theMinimum.errorDef();
77          assert (npoints > 3);
78          int maxcalls = 100 * (npoints + 5) * (theMinimum.userState().variableParameters() + 1);
79          int nfcn = 0;
80  
81          List<Point> result = new ArrayList<Point>(npoints);
82          List<MnUserParameterState> states = new ArrayList<MnUserParameterState>();
83          double toler = 0.05;
84  
85          //get first four points
86          MnMinos minos = new MnMinos(theFCN, theMinimum, theStrategy);
87  
88          double valx = theMinimum.userState().value(px);
89          double valy = theMinimum.userState().value(py);
90  
91          MinosError mex = minos.minos(px, errDef);
92          nfcn += mex.nfcn();
93          if (!mex.isValid()) {
94              logger.info("MnContours is unable to find first two points.");
95              return new ContoursError(px, py, result, mex, mex, nfcn);
96          }
97          Point ex = mex.range();
98  
99          MinosError mey = minos.minos(py, errDef);
100         nfcn += mey.nfcn();
101         if (!mey.isValid()) {
102             logger.info("MnContours is unable to find second two points.");
103             return new ContoursError(px, py, result, mex, mey, nfcn);
104         }
105         Point ey = mey.range();
106 
107         MnMigrad migrad = new MnMigrad(theFCN, theMinimum.userState().clone(), new MnStrategy(Math.max(0, theStrategy.strategy() - 1)));
108 
109         migrad.fix(px);
110         migrad.setValue(px, valx + ex.second);
111         FunctionMinimum exy_up = migrad.minimize();
112         nfcn += exy_up.nfcn();
113         if (!exy_up.isValid()) {
114             logger.log(Level.INFO, "MnContours is unable to find upper y value for x parameter {0}.", px);
115             return new ContoursError(px, py, result, mex, mey, nfcn);
116         }
117 
118         migrad.setValue(px, valx + ex.first);
119         FunctionMinimum exy_lo = migrad.minimize();
120         nfcn += exy_lo.nfcn();
121         if (!exy_lo.isValid()) {
122             logger.log(Level.INFO, "MnContours is unable to find lower y value for x parameter {0}.", px);
123             return new ContoursError(px, py, result, mex, mey, nfcn);
124         }
125 
126         MnMigrad migrad1 = new MnMigrad(theFCN, theMinimum.userState().clone(), new MnStrategy(Math.max(0, theStrategy.strategy() - 1)));
127         migrad1.fix(py);
128         migrad1.setValue(py, valy + ey.second);
129         FunctionMinimum eyx_up = migrad1.minimize();
130         nfcn += eyx_up.nfcn();
131         if (!eyx_up.isValid()) {
132             logger.log(Level.INFO, "MnContours is unable to find upper x value for y parameter {0}.", py);
133             return new ContoursError(px, py, result, mex, mey, nfcn);
134         }
135 
136         migrad1.setValue(py, valy + ey.first);
137         FunctionMinimum eyx_lo = migrad1.minimize();
138         nfcn += eyx_lo.nfcn();
139         if (!eyx_lo.isValid()) {
140             logger.log(Level.INFO, "MnContours is unable to find lower x value for y parameter {0}.", py);
141             return new ContoursError(px, py, result, mex, mey, nfcn);
142         }
143 
144         double scalx = 1. / (ex.second - ex.first);
145         double scaly = 1. / (ey.second - ey.first);
146 
147         result.add(new Point(valx + ex.first, exy_lo.userState().value(py)));
148         result.add(new Point(eyx_lo.userState().value(px), valy + ey.first));
149         result.add(new Point(valx + ex.second, exy_up.userState().value(py)));
150         result.add(new Point(eyx_up.userState().value(px), valy + ey.second));
151 
152         MnUserParameterState upar = theMinimum.userState().clone();
153         upar.fix(px);
154         upar.fix(py);
155 
156         int[] par = {px, py};
157         MnFunctionCross cross = new MnFunctionCross(theFCN, upar, theMinimum.fval(), theStrategy, errDef);
158 
159         for (int i = 4; i < npoints; i++) {
160             Point idist1 = result.get(result.size() - 1);
161             Point idist2 = result.get(0);
162             int pos2 = 0;
163             double distx = idist1.first - idist2.first;
164             double disty = idist1.second - idist2.second;
165             double bigdis = scalx * scalx * distx * distx + scaly * scaly * disty * disty;
166 
167             for (int j = 0; j < result.size() - 1; j++) {
168                 Point ipair = result.get(j);
169                 double distx2 = ipair.first - result.get(j + 1).first;
170                 double disty2 = ipair.second - result.get(j + 1).second;
171                 double dist = scalx * scalx * distx2 * distx2 + scaly * scaly * disty2 * disty2;
172                 if (dist > bigdis) {
173                     bigdis = dist;
174                     idist1 = ipair;
175                     idist2 = result.get(j + 1);
176                     pos2 = j + 1;
177                 }
178             }
179 
180             double a1 = 0.5;
181             double a2 = 0.5;
182             double sca = 1.;
183 
184             for (;;) {
185                 if (nfcn > maxcalls) {
186                     logger.info("MnContours: maximum number of function calls exhausted.");
187                     return new ContoursError(px, py, result, mex, mey, nfcn);
188                 }
189 
190                 double xmidcr = a1 * idist1.first + a2 * idist2.first;
191                 double ymidcr = a1 * idist1.second + a2 * idist2.second;
192                 double xdir = idist2.second - idist1.second;
193                 double ydir = idist1.first - idist2.first;
194                 double scalfac = sca * Math.max(Math.abs(xdir * scalx), Math.abs(ydir * scaly));
195                 double xdircr = xdir / scalfac;
196                 double ydircr = ydir / scalfac;
197                 double[] pmid = {xmidcr, ymidcr};
198                 double[] pdir = {xdircr, ydircr};
199 
200                 MnCross opt = cross.cross(par, pmid, pdir, toler, maxcalls);
201                 nfcn += opt.nfcn();
202                 if (opt.isValid()) {
203                     double aopt = opt.value();
204                     if (pos2 == 0) {
205                         result.add(new Point(xmidcr + (aopt) * xdircr, ymidcr + (aopt) * ydircr));
206                     } else {
207                         result.add(pos2, new Point(xmidcr + (aopt) * xdircr, ymidcr + (aopt) * ydircr));
208                     }
209                     break;
210                 }
211                 if (sca < 0.) {
212                     logger.log(Level.INFO, "MnContours is unable to find point {0} on contour.", (i + 1));
213                     logger.log(Level.INFO, "MnContours finds only {0} points.", i);
214                     return new ContoursError(px, py, result, mex, mey, nfcn);
215                 }
216                 sca = -1.;
217             }
218         }
219 
220         return new ContoursError(px, py, result, mex, mey, nfcn);
221     }
222 
223     MnStrategy strategy() {
224         return theStrategy;
225     }
