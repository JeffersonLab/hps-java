3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * Performs a minimization using the simplex method of Nelder and Mead (ref.
8    * Comp. J. 7, 308 (1965)).
9    *
10   */
11  class ScanBuilder implements MinimumBuilder {
12  
13      public FunctionMinimum minimum(MnFcn mfcn, GradientCalculator gc, MinimumSeed seed, MnStrategy stra, int maxfcn, double toler) {
14          MnAlgebraicVector x = seed.parameters().vec().clone();
15          MnUserParameterState upst = new MnUserParameterState(seed.state(), mfcn.errorDef(), seed.trafo());
16          MnParameterScan scan = new MnParameterScan(mfcn.fcn(), upst.parameters(), seed.fval());
17          double amin = scan.fval();
18          int n = seed.trafo().variableParameters();
19          MnAlgebraicVector dirin = new MnAlgebraicVector(n);
20          for (int i = 0; i < n; i++) {
21              int ext = seed.trafo().extOfInt(i);
22              scan.scan(ext);
23              if (scan.fval() < amin) {
24                  amin = scan.fval();
25                  x.set(i, seed.trafo().ext2int(ext, scan.parameters().value(ext)));
26              }
27              dirin.set(i, Math.sqrt(2. * mfcn.errorDef() * seed.error().invHessian().get(i, i)));
28          }
29  
30          MinimumParameters mp = new MinimumParameters(x, dirin, amin);
31          MinimumState st = new MinimumState(mp, 0., mfcn.numOfCalls());
32  
33          List<MinimumState> states = new ArrayList<MinimumState>(1);
34          states.add(st);
35          return new FunctionMinimum(seed, states, mfcn.errorDef());
36      }
37  
38  }
