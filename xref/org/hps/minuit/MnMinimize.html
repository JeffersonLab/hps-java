3   /**
4    * Causes minimization of the function by the method of MIGRAD, as does the
5    * MnMigrad class, but switches to the SIMPLEX method if MIGRAD fails to
6    * converge. Constructor arguments, methods arguments and names of methods are
7    * the same as for MnMigrad or MnSimplex.
8    *
9    */
10  public class MnMinimize extends MnApplication {
11  
12      /**
13       * construct from FCNBase + double[] for parameters and errors with default
14       * strategy
15       */
16      public MnMinimize(FCNBase fcn, double[] par, double[] err) {
17          this(fcn, par, err, DEFAULT_STRATEGY);
18      }
19  
20      /**
21       * construct from FCNBase + double[] for parameters and errors
22       */
23      public MnMinimize(FCNBase fcn, double[] par, double[] err, int stra) {
24          this(fcn, new MnUserParameterState(par, err), new MnStrategy(stra));
25      }
26  
27      /**
28       * construct from FCNBase + double[] for parameters and MnUserCovariance
29       * with default strategy
30       */
31      public MnMinimize(FCNBase fcn, double[] par, MnUserCovariance cov) {
32          this(fcn, par, cov, DEFAULT_STRATEGY);
33      }
34  
35      /**
36       * construct from FCNBase + double[] for parameters and MnUserCovariance
37       */
38      public MnMinimize(FCNBase fcn, double[] par, MnUserCovariance cov, int stra) {
39          this(fcn, new MnUserParameterState(par, cov), new MnStrategy(stra));
40      }
41  
42      /**
43       * construct from FCNBase + MnUserParameters with default strategy
44       */
45      public MnMinimize(FCNBase fcn, MnUserParameters par) {
46          this(fcn, par, DEFAULT_STRATEGY);
47      }
48  
49      /**
50       * construct from FCNBase + MnUserParameters
51       */
52      public MnMinimize(FCNBase fcn, MnUserParameters par, int stra) {
53          this(fcn, new MnUserParameterState(par), new MnStrategy(stra));
54      }
55  
56      /**
57       * construct from FCNBase + MnUserParameters + MnUserCovariance with default
58       * strategy
59       */
60      public MnMinimize(FCNBase fcn, MnUserParameters par, MnUserCovariance cov) {
61          this(fcn, par, cov, DEFAULT_STRATEGY);
62      }
63  
64      /**
65       * construct from FCNBase + MnUserParameters + MnUserCovariance
66       */
67      public MnMinimize(FCNBase fcn, MnUserParameters par, MnUserCovariance cov, int stra) {
68          this(fcn, new MnUserParameterState(par, cov), new MnStrategy(stra));
69      }
70  
71      /**
72       * construct from FCNBase + MnUserParameterState + MnStrategy
73       */
74      public MnMinimize(FCNBase fcn, MnUserParameterState par, MnStrategy str) {
75          super(fcn, par, str);
76      }
77  
78      ModularFunctionMinimizer minimizer() {
79          return theMinimizer;
80      }
81      private CombinedMinimizer theMinimizer = new CombinedMinimizer();
82  }
