6   class DavidonErrorUpdator implements MinimumErrorUpdator {
7   
8       public MinimumError update(MinimumState s0, MinimumParameters p1, FunctionGradient g1) {
9           MnAlgebraicSymMatrix V0 = s0.error().invHessian();
10          MnAlgebraicVector dx = MnUtils.sub(p1.vec(), s0.vec());
11          MnAlgebraicVector dg = MnUtils.sub(g1.vec(), s0.gradient().vec());
12  
13          double delgam = MnUtils.innerProduct(dx, dg);
14          double gvg = MnUtils.similarity(dg, V0);
15  
16          MnAlgebraicVector vg = MnUtils.mul(V0, dg);
17  
18          MnAlgebraicSymMatrix Vupd = MnUtils.sub(MnUtils.div(MnUtils.outerProduct(dx), delgam), MnUtils.div(MnUtils.outerProduct(vg), gvg));
19  
20          if (delgam > gvg) {
21              Vupd = MnUtils.add(Vupd, MnUtils.mul(MnUtils.outerProduct(MnUtils.sub(MnUtils.div(dx, delgam), MnUtils.div(vg, gvg))), gvg));
22          }
23  
24          double sum_upd = MnUtils.absoluteSumOfElements(Vupd);
25          Vupd = MnUtils.add(Vupd, V0);
26  
27          double dcov = 0.5 * (s0.error().dcovar() + sum_upd / MnUtils.absoluteSumOfElements(Vupd));
28  
29          return new MinimumError(Vupd, dcov);
30      }
