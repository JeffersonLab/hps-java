7   class AnalyticalGradientCalculator implements GradientCalculator {
8   
9       AnalyticalGradientCalculator(FCNGradientBase fcn, MnUserTransformation state, boolean checkGradient) {
10          theGradCalc = fcn;
11          theTransformation = state;
12          theCheckGradient = checkGradient;
13      }
14  
15      public FunctionGradient gradient(MinimumParameters par) {
16          double[] grad = theGradCalc.gradient(theTransformation.transform(par.vec()).data());
17          if (grad.length != theTransformation.parameters().size()) {
18              throw new IllegalArgumentException("Invalid parameter size");
19          }
20  
21          MnAlgebraicVector v = new MnAlgebraicVector(par.vec().size());
22          for (int i = 0; i < par.vec().size(); i++) {
23              int ext = theTransformation.extOfInt(i);
24              if (theTransformation.parameter(ext).hasLimits()) {
25                  double dd = theTransformation.dInt2Ext(i, par.vec().get(i));
26                  v.set(i, dd * grad[ext]);
27              } else {
28                  v.set(i, grad[ext]);
29              }
30          }
31  
32          return new FunctionGradient(v);
33      }
34  
35      public FunctionGradient gradient(MinimumParameters par, FunctionGradient grad) {
36          return gradient(par);
37      }
38  
39      boolean checkGradient() {
40          return theCheckGradient;
41      }
42  
43      private FCNGradientBase theGradCalc;
44      private MnUserTransformation theTransformation;
45      private boolean theCheckGradient;
46  }
