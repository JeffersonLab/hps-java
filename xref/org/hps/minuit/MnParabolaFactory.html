6   abstract class MnParabolaFactory {

8       static MnParabola create(MnParabolaPoint p1, MnParabolaPoint p2, MnParabolaPoint p3) {
9           double x1 = p1.x();
10          double x2 = p2.x();
11          double x3 = p3.x();
12          double dx12 = x1 - x2;
13          double dx13 = x1 - x3;
14          double dx23 = x2 - x3;
15  
16          double xm = (x1 + x2 + x3) / 3.;
17          x1 -= xm;
18          x2 -= xm;
19          x3 -= xm;
20  
21          double y1 = p1.y();
22          double y2 = p2.y();
23          double y3 = p3.y();
24  
25          double a = y1 / (dx12 * dx13) - y2 / (dx12 * dx23) + y3 / (dx13 * dx23);
26          double b = -y1 * (x2 + x3) / (dx12 * dx13) + y2 * (x1 + x3) / (dx12 * dx23) - y3 * (x1 + x2) / (dx13 * dx23);
27          double c = y1 - a * x1 * x1 - b * x1;
28  
29          c += xm * (xm * a - b);
30          b -= 2. * xm * a;
31  
32          return new MnParabola(a, b, c);
33      }
34  
35      static MnParabola create(MnParabolaPoint p1, double dxdy1, MnParabolaPoint p2) {
36          double x1 = p1.x();
37          double xx1 = x1 * x1;
38          double x2 = p2.x();
39          double xx2 = x2 * x2;
40          double y1 = p1.y();
41          double y12 = p1.y() - p2.y();
42  
43          double det = xx1 - xx2 - 2. * x1 * (x1 - x2);
44          double a = -(y12 + (x2 - x1) * dxdy1) / det;
45          double b = -(-2. * x1 * y12 + (xx1 - xx2) * dxdy1) / det;
46          double c = y1 - a * xx1 - b * x1;
47  
48          return new MnParabola(a, b, c);
49      }
50  }
