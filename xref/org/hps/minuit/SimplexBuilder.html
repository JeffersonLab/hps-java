1   package org.hps.minuit;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.logging.Logger;
6   
7   /**
8    *
9    */
10  class SimplexBuilder implements MinimumBuilder {
11  
12      @Override
13      public FunctionMinimum minimum(MnFcn mfcn, GradientCalculator gc, MinimumSeed seed, MnStrategy strategy, int maxfcn, double minedm) {
14          MnMachinePrecision prec = seed.precision();
15          MnAlgebraicVector x = seed.parameters().vec().clone();
16          MnAlgebraicVector step = MnUtils.mul(seed.gradient().gstep(), 10.);
17  
18          int n = x.size();
19          double wg = 1. / n;
20          double alpha = 1., beta = 0.5, gamma = 2., rhomin = 4., rhomax = 8.;
21          double rho1 = 1. + alpha;
22          double rho2 = 1. + alpha * gamma;
23  
24          List<Pair<Double, MnAlgebraicVector>> simpl = new ArrayList<Pair<Double, MnAlgebraicVector>>(n + 1);
25          simpl.add(new Pair<Double, MnAlgebraicVector>(seed.fval(), x.clone()));
26  
27          int jl = 0, jh = 0;
28          double amin = seed.fval(), aming = seed.fval();
29  
30          for (int i = 0; i < n; i++) {
31              double dmin = 8. * prec.eps2() * (Math.abs(x.get(i)) + prec.eps2());
32              if (step.get(i) < dmin) {
33                  step.set(i, dmin);
34              }
35              x.set(i, x.get(i) + step.get(i));
36              double tmp = mfcn.valueOf(x);
37              if (tmp < amin) {
38                  amin = tmp;
39                  jl = i + 1;
40              }
41              if (tmp > aming) {
42                  aming = tmp;
43                  jh = i + 1;
44              }
45              simpl.add(new Pair<Double, MnAlgebraicVector>(tmp, x.clone()));
46              x.set(i, x.get(i) - step.get(i));
47          }
48          SimplexParameters simplex = new SimplexParameters(simpl, jh, jl);
49  
50          do {
51              amin = simplex.get(jl).first;
52              jl = simplex.jl();
53              jh = simplex.jh();
54              MnAlgebraicVector pbar = new MnAlgebraicVector(n);
55              for (int i = 0; i < n + 1; i++) {
56                  if (i == jh) {
57                      continue;
58                  }
59                  pbar = MnUtils.add(pbar, MnUtils.mul(simplex.get(i).second, wg));
60              }
61  
62              MnAlgebraicVector pstar = MnUtils.sub(MnUtils.mul(pbar, 1. + alpha), MnUtils.mul(simplex.get(jh).second, alpha));
63              double ystar = mfcn.valueOf(pstar);
64  
65              if (ystar > amin) {
66                  if (ystar < simplex.get(jh).first) {
67                      simplex.update(ystar, pstar);
68                      if (jh != simplex.jh()) {
69                          continue;
70                      }
71                  }
72                  MnAlgebraicVector pstst = MnUtils.add(MnUtils.mul(simplex.get(jh).second, beta), MnUtils.mul(pbar, 1. - beta));
73                  double ystst = mfcn.valueOf(pstst);
74                  if (ystst > simplex.get(jh).first) {
75                      break;
76                  }
77                  simplex.update(ystst, pstst);
78                  continue;
79              }
80  
81              MnAlgebraicVector pstst = MnUtils.add(MnUtils.mul(pstar, gamma), MnUtils.mul(pbar, 1. - gamma));
82              double ystst = mfcn.valueOf(pstst);
83  
84              double y1 = (ystar - simplex.get(jh).first) * rho2;
85              double y2 = (ystst - simplex.get(jh).first) * rho1;
86              double rho = 0.5 * (rho2 * y1 - rho1 * y2) / (y1 - y2);
87              if (rho < rhomin) {
88                  if (ystst < simplex.get(jl).first) {
89                      simplex.update(ystst, pstst);
90                  } else {
91                      simplex.update(ystar, pstar);
92                  }
93                  continue;
94              }
95              if (rho > rhomax) {
96                  rho = rhomax;
97              }
98              MnAlgebraicVector prho = MnUtils.add(MnUtils.mul(pbar, rho), MnUtils.mul(simplex.get(jh).second, 1. - rho));
99              double yrho = mfcn.valueOf(prho);
100             if (yrho < simplex.get(jl).first && yrho < ystst) {
101                 simplex.update(yrho, prho);
102                 continue;
103             }
104             if (ystst < simplex.get(jl).first) {
105                 simplex.update(ystst, pstst);
106                 continue;
107             }
108             if (yrho > simplex.get(jl).first) {
109                 if (ystst < simplex.get(jl).first) {
110                     simplex.update(ystst, pstst);
111                 } else {
112                     simplex.update(ystar, pstar);
113                 }
114                 continue;
115             }
116             if (ystar > simplex.get(jh).first) {
117                 pstst = MnUtils.add(MnUtils.mul(simplex.get(jh).second, beta), MnUtils.mul(pbar, 1 - beta));
118                 ystst = mfcn.valueOf(pstst);
119                 if (ystst > simplex.get(jh).first) {
120                     break;
121                 }
122                 simplex.update(ystst, pstst);
123             }
124         } while (simplex.edm() > minedm && mfcn.numOfCalls() < maxfcn);
125 
126         amin = simplex.get(jl).first;
127         jl = simplex.jl();
128         jh = simplex.jh();
129 
130         MnAlgebraicVector pbar = new MnAlgebraicVector(n);
131         for (int i = 0; i < n + 1; i++) {
132             if (i == jh) {
133                 continue;
134             }
135             pbar = MnUtils.add(pbar, MnUtils.mul(simplex.get(i).second, wg));
136         }
137         double ybar = mfcn.valueOf(pbar);
138         if (ybar < amin) {
139             simplex.update(ybar, pbar);
140         } else {
141             pbar = simplex.get(jl).second;
142             ybar = simplex.get(jl).first;
143         }
144 
145         MnAlgebraicVector dirin = simplex.dirin();
146         //   scale to sigmas on parameters werr^2 = dirin^2 * (up/edm)
147         dirin = MnUtils.mul(dirin, Math.sqrt(mfcn.errorDef() / simplex.edm()));
148 
149         MinimumState st = new MinimumState(new MinimumParameters(pbar, dirin, ybar), simplex.edm(), mfcn.numOfCalls());
150         List<MinimumState> states = new ArrayList<MinimumState>(1);
151         states.add(st);
152 
153         if (mfcn.numOfCalls() > maxfcn) {
154 //            logger.info("Simplex did not converge, #fcn calls exhausted.");
155             return new FunctionMinimum(seed, states, mfcn.errorDef(), new FunctionMinimum.MnReachedCallLimit());
156         }
157         if (simplex.edm() > minedm) {
158  //           logger.info("Simplex did not converge, edm > minedm.");
159             return new FunctionMinimum(seed, states, mfcn.errorDef(), new FunctionMinimum.MnAboveMaxEdm());
160         }
161 
162         return new FunctionMinimum(seed, states, mfcn.errorDef());
163     }
164     private static final Logger logger = Logger.getLogger(SimplexBuilder.class.getName());
165 }
