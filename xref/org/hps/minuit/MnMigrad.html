3   /**
4    * MnMigrad provides minimization of the function by the method of MIGRAD, the
5    * most efficient and complete single method, recommended for general functions,
6    * and the functionality for parameters interaction. It also retains the result
7    * from the last minimization in case the user may want to do subsequent
8    * minimization steps with parameter interactions in between the minimization
9    * requests. The minimization produces as a by-product the error matrix of the
10   * parameters, which is usually reliable unless warning messages are produced.
11   *
12   */
13  public class MnMigrad extends MnApplication {
14  
15      /**
16       * construct from FCNBase + double[] for parameters and errors with default
17       * strategy
18       */
19      public MnMigrad(FCNBase fcn, double[] par, double[] err) {
20          this(fcn, par, err, DEFAULT_STRATEGY);
21      }
22  
23      /**
24       * construct from FCNBase + double[] for parameters and errors
25       */
26      public MnMigrad(FCNBase fcn, double[] par, double[] err, int stra) {
27          this(fcn, new MnUserParameterState(par, err), new MnStrategy(stra));
28      }
29  
30      /**
31       * construct from FCNBase + double[] for parameters and MnUserCovariance
32       * with default strategy
33       */
34      public MnMigrad(FCNBase fcn, double[] par, MnUserCovariance cov) {
35          this(fcn, par, cov, DEFAULT_STRATEGY);
36      }
37  
38      /**
39       * construct from FCNBase + double[] for parameters and MnUserCovariance
40       */
41      public MnMigrad(FCNBase fcn, double[] par, MnUserCovariance cov, int stra) {
42          this(fcn, new MnUserParameterState(par, cov), new MnStrategy(stra));
43      }
44  
45      /**
46       * construct from FCNBase + MnUserParameters with default strategy
47       */
48      public MnMigrad(FCNBase fcn, MnUserParameters par) {
49          this(fcn, par, DEFAULT_STRATEGY);
50      }
51  
52      /**
53       * construct from FCNBase + MnUserParameters
54       */
55      public MnMigrad(FCNBase fcn, MnUserParameters par, int stra) {
56          this(fcn, new MnUserParameterState(par), new MnStrategy(stra));
57      }
58  
59      /**
60       * construct from FCNBase + MnUserParameters + MnUserCovariance with default
61       * strategy
62       */
63      public MnMigrad(FCNBase fcn, MnUserParameters par, MnUserCovariance cov) {
64          this(fcn, par, cov, DEFAULT_STRATEGY);
65      }
66  
67      /**
68       * construct from FCNBase + MnUserParameters + MnUserCovariance
69       */
70      public MnMigrad(FCNBase fcn, MnUserParameters par, MnUserCovariance cov, int stra) {
71          this(fcn, new MnUserParameterState(par, cov), new MnStrategy(stra));
72      }
73  
74      /**
75       * construct from FCNBase + MnUserParameterState + MnStrategy
76       */
77      public MnMigrad(FCNBase fcn, MnUserParameterState par, MnStrategy str) {
78          super(fcn, par, str);
79      }
80  
81      ModularFunctionMinimizer minimizer() {
82          return theMinimizer;
83      }
84  
85      private VariableMetricMinimizer theMinimizer = new VariableMetricMinimizer();
86  }
