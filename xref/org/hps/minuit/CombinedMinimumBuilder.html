8   class CombinedMinimumBuilder implements MinimumBuilder {
9   
10      @Override
11      public FunctionMinimum minimum(MnFcn fcn, GradientCalculator gc, MinimumSeed seed, MnStrategy strategy, int maxfcn, double toler) {
12          FunctionMinimum min = theVMMinimizer.minimize(fcn, gc, seed, strategy, maxfcn, toler);
13  
14          if (!min.isValid()) {
15              logger.info("CombinedMinimumBuilder: migrad method fails, will try with simplex method first.");
16              MnStrategy str = new MnStrategy(2);
17              FunctionMinimum min1 = theSimplexMinimizer.minimize(fcn, gc, seed, str, maxfcn, toler);
18              if (!min1.isValid()) {
19                  logger.info("CombinedMinimumBuilder: both migrad and simplex method fail.");
20                  return min1;
21              }
22              MinimumSeed seed1 = theVMMinimizer.seedGenerator().generate(fcn, gc, min1.userState(), str);
23  
24              FunctionMinimum min2 = theVMMinimizer.minimize(fcn, gc, seed1, str, maxfcn, toler);
25              if (!min2.isValid()) {
26                  logger.info("CombinedMinimumBuilder: both migrad and method fails also at 2nd attempt.");
27                  logger.info("CombinedMinimumBuilder: return simplex minimum.");
28                  return min1;
29              }
30  
31              return min2;
32          }
33          return min;
34      }
35  
36      private final VariableMetricMinimizer theVMMinimizer = new VariableMetricMinimizer();
37      private final SimplexMinimizer theSimplexMinimizer = new SimplexMinimizer();
38      private static final Logger logger = Logger.getLogger(CombinedMinimumBuilder.class.getName());
39  }
