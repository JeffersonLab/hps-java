6   class InitialGradientCalculator {
7   
8       InitialGradientCalculator(MnFcn fcn, MnUserTransformation par, MnStrategy stra) {
9           theFcn = fcn;
10          theTransformation = par;
11          theStrategy = stra;
12      }
13  
14      FunctionGradient gradient(MinimumParameters par) {
15          if (!par.isValid()) {
16              throw new IllegalArgumentException("Parameters are invalid");
17          }
18  
19          int n = trafo().variableParameters();
20          if (n != par.vec().size()) {
21              throw new IllegalArgumentException("Parameters have invalid size");
22          };
23  
24          MnAlgebraicVector gr = new MnAlgebraicVector(n);
25          MnAlgebraicVector gr2 = new MnAlgebraicVector(n);
26          MnAlgebraicVector gst = new MnAlgebraicVector(n);
27  
28          // initial starting values
29          for (int i = 0; i < n; i++) {
30              int exOfIn = trafo().extOfInt(i);
31  
32              double var = par.vec().get(i);
33              double werr = trafo().parameter(exOfIn).error();
34              double sav = trafo().int2ext(i, var);
35              double sav2 = sav + werr;
36              if (trafo().parameter(exOfIn).hasLimits()) {
37                  if (trafo().parameter(exOfIn).hasUpperLimit()
38                          && sav2 > trafo().parameter(exOfIn).upperLimit()) {
39                      sav2 = trafo().parameter(exOfIn).upperLimit();
40                  }
41              }
42              double var2 = trafo().ext2int(exOfIn, sav2);
43              double vplu = var2 - var;
44              sav2 = sav - werr;
45              if (trafo().parameter(exOfIn).hasLimits()) {
46                  if (trafo().parameter(exOfIn).hasLowerLimit()
47                          && sav2 < trafo().parameter(exOfIn).lowerLimit()) {
48                      sav2 = trafo().parameter(exOfIn).lowerLimit();
49                  }
50              }
51              var2 = trafo().ext2int(exOfIn, sav2);
52              double vmin = var2 - var;
53              double dirin = 0.5 * (Math.abs(vplu) + Math.abs(vmin));
54              double g2 = 2.0 * theFcn.errorDef() / (dirin * dirin);
55              double gsmin = 8. * precision().eps2() * (Math.abs(var) + precision().eps2());
56              double gstep = Math.max(gsmin, 0.1 * dirin);
57              double grd = g2 * dirin;
58              if (trafo().parameter(exOfIn).hasLimits()) {
59                  if (gstep > 0.5) {
60                      gstep = 0.5;
61                  }
62              }
63              gr.set(i, grd);
64              gr2.set(i, g2);
65              gst.set(i, gstep);
66          }
67  
68          return new FunctionGradient(gr, gr2, gst);
69      }
70  
71      FunctionGradient gradient(MinimumParameters par, FunctionGradient gra) {
72          return gradient(par);
73      }
74  
75      MnFcn fcn() {
76          return theFcn;
77      }
78  
79      MnUserTransformation trafo() {
80          return theTransformation;
81      }
82  
83      MnMachinePrecision precision() {
84          return theTransformation.precision();
85      }
86  
87      MnStrategy strategy() {
88          return theStrategy;
89      }
90  
91      int ncycle() {
92          return strategy().gradientNCycles();
93      }
94  
95      double stepTolerance() {
96          return strategy().gradientStepTolerance();
97      }
98  
99      double gradTolerance() {
100         return strategy().gradientTolerance();
101     }
102 
103     private MnFcn theFcn;
104     private MnUserTransformation theTransformation;
105     private MnStrategy theStrategy;
106 }
