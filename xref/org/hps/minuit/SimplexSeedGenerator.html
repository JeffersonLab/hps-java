6   class SimplexSeedGenerator implements MinimumSeedGenerator {
7   
8       public MinimumSeed generate(MnFcn fcn, GradientCalculator gc, MnUserParameterState st, MnStrategy stra) {
9           int n = st.variableParameters();
10          MnMachinePrecision prec = st.precision();
11  
12          // initial starting values
13          MnAlgebraicVector x = new MnAlgebraicVector(n);
14          for (int i = 0; i < n; i++) {
15              x.set(i, st.intParameters().get(i));
16          }
17          double fcnmin = fcn.valueOf(x);
18          MinimumParameters pa = new MinimumParameters(x, fcnmin);
19          InitialGradientCalculator igc = new InitialGradientCalculator(fcn, st.trafo(), stra);
20          FunctionGradient dgrad = igc.gradient(pa);
21          MnAlgebraicSymMatrix mat = new MnAlgebraicSymMatrix(n);
22          double dcovar = 1.;
23          for (int i = 0; i < n; i++) {
24              mat.set(i, i, Math.abs(dgrad.g2().get(i)) > prec.eps2() ? 1. / dgrad.g2().get(i) : 1.);
25          }
26          MinimumError err = new MinimumError(mat, dcovar);
27          double edm = new VariableMetricEDMEstimator().estimate(dgrad, err);
28          MinimumState state = new MinimumState(pa, err, dgrad, edm, fcn.numOfCalls());
29  
30          return new MinimumSeed(state, st.trafo());
31      }
32  }
