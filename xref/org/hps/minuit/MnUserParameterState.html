6   /**
7    * The class MnUserParameterState contains the MnUserParameters and the
8    * MnUserCovariance. It can be created on input by the user, or by MINUIT itself

42      private MnUserParameterState(MnUserParameterState other) {
43          theValid = other.theValid;
44          theCovarianceValid = other.theCovarianceValid;
45          theGCCValid = other.theGCCValid;

59      /**
60       * construct from user parameters (before minimization)
61       */
62      MnUserParameterState(double[] par, double[] err) {
63          theValid = true;
64          theParameters = new MnUserParameters(par, err);
65          theCovariance = new MnUserCovariance();
66          theGlobalCC = new MnGlobalCorrelationCoeff();
67          theIntParameters = new ArrayList<Double>(par.length);
68          for (int i = 0; i < par.length; i++) {
69              theIntParameters.add(par[i]);
70          }
71          theIntCovariance = new MnUserCovariance();
72      }

74      MnUserParameterState(MnUserParameters par) {
75          theValid = true;
76          theParameters = par;
77          theCovariance = new MnUserCovariance();
78          theGlobalCC = new MnGlobalCorrelationCoeff();
79          theIntParameters = new ArrayList<Double>(par.variableParameters());
80          theIntCovariance = new MnUserCovariance();
81  
82          int i = 0;
83          for (MinuitParameter ipar : par.parameters()) {
84              if (ipar.isConst() || ipar.isFixed()) {
85                  continue;
86              }
87              if (ipar.hasLimits()) {
88                  theIntParameters.add(ext2int(ipar.number(), ipar.value()));
89              } else {
90                  theIntParameters.add(ipar.value());
91              }
92          }
93      }
94  
95      /**
96       * construct from user parameters + covariance (before minimization)
97       */
98      MnUserParameterState(double[] par, double[] cov, int nrow) {
99          theValid = true;
100         theCovarianceValid = true;
101         theCovariance = new MnUserCovariance(cov, nrow);
102         theGlobalCC = new MnGlobalCorrelationCoeff();
103         theIntParameters = new ArrayList<Double>(par.length);
104         theIntCovariance = new MnUserCovariance(cov, nrow);
105 
106         double[] err = new double[par.length];
107         for (int i = 0; i < par.length; i++) {
108             assert (theCovariance.get(i, i) > 0.);
109             err[i] = Math.sqrt(theCovariance.get(i, i));
110             theIntParameters.add(par[i]);
111         }
112         theParameters = new MnUserParameters(par, err);
113         assert (theCovariance.nrow() == variableParameters());
114     }

116     MnUserParameterState(double[] par, MnUserCovariance cov) {
117         theValid = true;
118         theCovarianceValid = true;
119         theCovariance = cov;
120         theGlobalCC = new MnGlobalCorrelationCoeff();
121         theIntParameters = new ArrayList<Double>(par.length);
122         theIntCovariance = cov.clone();
123 
124         if (theCovariance.nrow() != variableParameters()) {
125             throw new IllegalArgumentException("Bad covariance size");
126         }
127         double[] err = new double[par.length];
128         for (int i = 0; i < par.length; i++) {
129             if (theCovariance.get(i, i) <= 0.) {
130                 throw new IllegalArgumentException("Bad covariance");
131             }
132             err[i] = Math.sqrt(theCovariance.get(i, i));
133             theIntParameters.add(par[i]);
134         }
135         theParameters = new MnUserParameters(par, err);
136     }

138     MnUserParameterState(MnUserParameters par, MnUserCovariance cov) {
139         theValid = true;
140         theCovarianceValid = true;
141         theParameters = par;
142         theCovariance = cov;
143         theGlobalCC = new MnGlobalCorrelationCoeff();
144         theIntParameters = new ArrayList<Double>();
145         theIntCovariance = cov.clone();
146 
147         theIntCovariance.scale(0.5);
148         int i = 0;
149         for (MinuitParameter ipar : par.parameters()) {
150             if (ipar.isConst() || ipar.isFixed()) {
151                 continue;
152             }
153             if (ipar.hasLimits()) {
154                 theIntParameters.add(ext2int(ipar.number(), ipar.value()));
155             } else {
156                 theIntParameters.add(ipar.value());
157             }
158         }
159         assert (theCovariance.nrow() == variableParameters());
160     }

162     /**
163      * construct from internal parameters (after minimization)
164      */
165     MnUserParameterState(MinimumState st, double up, MnUserTransformation trafo) {
166         theValid = st.isValid();
167         theCovarianceValid = false;
168         theGCCValid = false;
169         theFVal = st.fval();
170         theEDM = st.edm();
171         theNFcn = st.nfcn();
172         theParameters = new MnUserParameters();
173         theCovariance = new MnUserCovariance();
174         theGlobalCC = new MnGlobalCorrelationCoeff();
175         theIntParameters = new ArrayList<Double>();
176         theIntCovariance = new MnUserCovariance();
177 
178         for (MinuitParameter ipar : trafo.parameters()) {
179             if (ipar.isConst()) {
180                 add(ipar.name(), ipar.value());
181             } else if (ipar.isFixed()) {
182                 add(ipar.name(), ipar.value(), ipar.error());
183                 if (ipar.hasLimits()) {
184                     if (ipar.hasLowerLimit() && ipar.hasUpperLimit()) {
185                         setLimits(ipar.name(), ipar.lowerLimit(), ipar.upperLimit());
186                     } else if (ipar.hasLowerLimit() && !ipar.hasUpperLimit()) {
187                         setLowerLimit(ipar.name(), ipar.lowerLimit());
188                     } else {
189                         setUpperLimit(ipar.name(), ipar.upperLimit());
190                     }
191                 }
192                 fix(ipar.name());
193             } else if (ipar.hasLimits()) {
194                 int i = trafo.intOfExt(ipar.number());
195                 double err = st.hasCovariance() ? Math.sqrt(2. * up * st.error().invHessian().get(i, i)) : st.parameters().dirin().get(i);
196                 add(ipar.name(), trafo.int2ext(i, st.vec().get(i)), trafo.int2extError(i, st.vec().get(i), err));
197                 if (ipar.hasLowerLimit() && ipar.hasUpperLimit()) {
198                     setLimits(ipar.name(), ipar.lowerLimit(), ipar.upperLimit());
199                 } else if (ipar.hasLowerLimit() && !ipar.hasUpperLimit()) {
200                     setLowerLimit(ipar.name(), ipar.lowerLimit());
201                 } else {
202                     setUpperLimit(ipar.name(), ipar.upperLimit());
203                 }
204             } else {
205                 int i = trafo.intOfExt(ipar.number());
206                 double err = st.hasCovariance() ? Math.sqrt(2. * up * st.error().invHessian().get(i, i)) : st.parameters().dirin().get(i);
207                 add(ipar.name(), st.vec().get(i), err);
208             }
209         }
210 
211         theCovarianceValid = st.error().isValid();
212 
213         if (theCovarianceValid) {
214             theCovariance = trafo.int2extCovariance(st.vec(), st.error().invHessian());
215             theIntCovariance = new MnUserCovariance(st.error().invHessian().data().clone(), st.error().invHessian().nrow());
216             theCovariance.scale(2. * up);
217             theGlobalCC = new MnGlobalCorrelationCoeff(st.error().invHessian());
218             theGCCValid = true;
219 
220             assert (theCovariance.nrow() == variableParameters());
221         }
222 
223     }

350     /// interaction via external number of parameter
351     public void fix(int e) {
352         int i = intOfExt(e);
353         if (theCovarianceValid) {
354             theCovariance = MnCovarianceSqueeze.squeeze(theCovariance, i);
355             theIntCovariance = MnCovarianceSqueeze.squeeze(theIntCovariance, i);
356         }
357         theIntParameters.remove(i);
358         theParameters.fix(e);
359         theGCCValid = false;
360     }
361 
362     public void release(int e) {
363         theParameters.release(e);
364         theCovarianceValid = false;
365         theGCCValid = false;
366         int i = intOfExt(e);
367         if (parameter(e).hasLimits()) {
368             theIntParameters.add(i, ext2int(e, parameter(e).value()));
369         } else {
370             theIntParameters.add(i, parameter(e).value());
371         }
372     }

374     public void setValue(int e, double val) {
375         theParameters.setValue(e, val);
376         if (!parameter(e).isFixed() && !parameter(e).isConst()) {
377             int i = intOfExt(e);
378             if (parameter(e).hasLimits()) {
379                 theIntParameters.set(i, ext2int(e, val));
380             } else {
381                 theIntParameters.set(i, val);
382             }
383         }
384     }

390     public void setLimits(int e, double low, double up) {
391         theParameters.setLimits(e, low, up);
392         theCovarianceValid = false;
393         theGCCValid = false;
394         if (!parameter(e).isFixed() && !parameter(e).isConst()) {
395             int i = intOfExt(e);
396             if (low < theIntParameters.get(i) && theIntParameters.get(i) < up) {
397                 theIntParameters.set(i, ext2int(e, theIntParameters.get(i)));
398             } else {
399                 theIntParameters.set(i, ext2int(e, 0.5 * (low + up)));
400             }
401         }
402     }
403 
404     public void setUpperLimit(int e, double up) {
405         theParameters.setUpperLimit(e, up);
406         theCovarianceValid = false;
407         theGCCValid = false;
408         if (!parameter(e).isFixed() && !parameter(e).isConst()) {
409             int i = intOfExt(e);
410             if (theIntParameters.get(i) < up) {
411                 theIntParameters.set(i, ext2int(e, theIntParameters.get(i)));
412             } else {
413                 theIntParameters.set(i, ext2int(e, up - 0.5 * Math.abs(up + 1.)));
414             }
415         }
416     }
417 
418     public void setLowerLimit(int e, double low) {
419         theParameters.setLowerLimit(e, low);
420         theCovarianceValid = false;
421         theGCCValid = false;
422         if (!parameter(e).isFixed() && !parameter(e).isConst()) {
423             int i = intOfExt(e);
424             if (low < theIntParameters.get(i)) {
425                 theIntParameters.set(i, ext2int(e, theIntParameters.get(i)));
426             } else {
427                 theIntParameters.set(i, ext2int(e, low + 0.5 * Math.abs(low + 1.)));
428             }
429         }
430     }

432     public void removeLimits(int e) {
433         theParameters.removeLimits(e);
434         theCovarianceValid = false;
435         theGCCValid = false;
436         if (!parameter(e).isFixed() && !parameter(e).isConst()) {
437             theIntParameters.set(intOfExt(e), value(e));
438         }
