7   class SqrtUpParameterTransformation {
8   
9       // transformation from internal to external
10      double int2ext(double value, double upper) {
11          return upper + 1. - Math.sqrt(value * value + 1.);
12      }
13  
14      // transformation from external to internal
15      double ext2int(double value, double upper, MnMachinePrecision prec) {
16          double yy = upper - value + 1.;
17          double yy2 = yy * yy;
18          if (yy2 < (1. + prec.eps2())) {
19              return 8 * Math.sqrt(prec.eps2());
20          } else {
21              return Math.sqrt(yy2 - 1);
22          }
23      }
24  
25      // derivative of transformation from internal to external
26      double dInt2Ext(double value, double upper) {
27          return -value / (Math.sqrt(value * value + 1.));
28      }

30  }
