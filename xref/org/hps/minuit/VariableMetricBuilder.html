10  class VariableMetricBuilder implements MinimumBuilder {

12      VariableMetricBuilder() {
13          theEstimator = new VariableMetricEDMEstimator();
14          theErrorUpdator = new DavidonErrorUpdator();
15      }
16  
17      public FunctionMinimum minimum(MnFcn fcn, GradientCalculator gc, MinimumSeed seed, MnStrategy strategy, int maxfcn, double edmval) {
18          FunctionMinimum min = minimum(fcn, gc, seed, maxfcn, edmval);
19          if ((strategy.strategy() == 2) || (strategy.strategy() == 1 && min.error().dcovar() > 0.05)) {
20              MinimumState st = new MnHesse(strategy).calculate(fcn, min.state(), min.seed().trafo(), 0);
21              min.add(st);
22          }
23          if (!min.isValid()) {
24              logger.info("FunctionMinimum is invalid.");
25          }
26  
27          return min;
28      }
29  
30      FunctionMinimum minimum(MnFcn fcn, GradientCalculator gc, MinimumSeed seed, int maxfcn, double edmval) {
31          edmval *= 0.0001;
32  
33          if (seed.parameters().vec().size() == 0) {
34              return new FunctionMinimum(seed, fcn.errorDef());
35          }
36  
37          MnMachinePrecision prec = seed.precision();
38  
39          List<MinimumState> result = new ArrayList<MinimumState>(8);
40  
41          double edm = seed.state().edm();
42  
43          if (edm < 0.) {
44              logger.info("VariableMetricBuilder: initial matrix not pos.def.");
45              if (seed.error().isPosDef()) {
46                  throw new RuntimeException("Something is wrong!");
47              }
48              return new FunctionMinimum(seed, fcn.errorDef());
49          }
50  
51          result.add(seed.state());
52  
53          // iterate until edm is small enough or max # of iterations reached
54          edm *= (1. + 3. * seed.error().dcovar());
55          MnAlgebraicVector step = new MnAlgebraicVector(seed.gradient().vec().size());
56          do {
57              MinimumState s0 = result.get(result.size() - 1);
58  
59              step = MnUtils.mul(MnUtils.mul(s0.error().invHessian(), s0.gradient().vec()), -1);
60  
61              double gdel = MnUtils.innerProduct(step, s0.gradient().grad());
62              if (gdel > 0.) {
63                  logger.info("VariableMetricBuilder: matrix not pos.def.");
64                  logger.info("gdel > 0: " + gdel);
65                  s0 = MnPosDef.test(s0, prec);
66                  step = MnUtils.mul(MnUtils.mul(s0.error().invHessian(), s0.gradient().vec()), -1);
67                  gdel = MnUtils.innerProduct(step, s0.gradient().grad());
68                  logger.info("gdel: " + gdel);
69                  if (gdel > 0.) {
70                      result.add(s0);
71                      return new FunctionMinimum(seed, result, fcn.errorDef());
72                  }
73              }
74              MnParabolaPoint pp = MnLineSearch.search(fcn, s0.parameters(), step, gdel, prec);
75              if (Math.abs(pp.y() - s0.fval()) < prec.eps()) {
76                  logger.info("VariableMetricBuilder: no improvement");
77                  break; //no improvement
78              }
79              MinimumParameters p = new MinimumParameters(MnUtils.add(s0.vec(), MnUtils.mul(step, pp.x())), pp.y());
80              FunctionGradient g = gc.gradient(p, s0.gradient());
81  
82              edm = estimator().estimate(g, s0.error());
83              if (edm < 0.) {
84                  logger.info("VariableMetricBuilder: matrix not pos.def.");
85                  logger.info("edm < 0");
86                  s0 = MnPosDef.test(s0, prec);
87                  edm = estimator().estimate(g, s0.error());
88                  if (edm < 0.) {
89                      result.add(s0);
90                      return new FunctionMinimum(seed, result, fcn.errorDef());
91                  }
92              }
93              MinimumError e = errorUpdator().update(s0, p, g);
94              result.add(new MinimumState(p, e, g, edm, fcn.numOfCalls()));
95              //     result[0] = MinimumState(p, e, g, edm, fcn.numOfCalls());
96              edm *= (1. + 3. * e.dcovar());
97          } while (edm > edmval && fcn.numOfCalls() < maxfcn);
98  
99          if (fcn.numOfCalls() >= maxfcn) {
100             logger.info("VariableMetricBuilder: call limit exceeded.");
101             return new FunctionMinimum(seed, result, fcn.errorDef(), new FunctionMinimum.MnReachedCallLimit());
102         }
103 
104         if (edm > edmval) {
105             if (edm < Math.abs(prec.eps2() * result.get(result.size() - 1).fval())) {
106                 logger.info("VariableMetricBuilder: machine accuracy limits further improvement.");
107                 return new FunctionMinimum(seed, result, fcn.errorDef());
108             } else if (edm < 10. * edmval) {
109                 return new FunctionMinimum(seed, result, fcn.errorDef());
110             } else {
111                 logger.info("VariableMetricBuilder: finishes without convergence.");
112                 logger.info("VariableMetricBuilder: edm= " + edm + " requested: " + edmval);
113                 return new FunctionMinimum(seed, result, fcn.errorDef(), new FunctionMinimum.MnAboveMaxEdm());
114             }
115         }
116 
117         return new FunctionMinimum(seed, result, fcn.errorDef());
118     }
