6   public class MnGlobalCorrelationCoeff {
7   
8       MnGlobalCorrelationCoeff() {
9           theGlobalCC = new double[0];
10      }
11  
12      MnGlobalCorrelationCoeff(MnAlgebraicSymMatrix cov) {
13          try {
14              MnAlgebraicSymMatrix inv = cov.clone();
15              inv.invert();
16              theGlobalCC = new double[cov.nrow()];
17              for (int i = 0; i < cov.nrow(); i++) {
18                  double denom = inv.get(i, i) * cov.get(i, i);
19                  if (denom < 1. && denom > 0.) {
20                      theGlobalCC[i] = 0;
21                  } else {
22                      theGlobalCC[i] = Math.sqrt(1. - 1. / denom);
23                  }
24              }
25              theValid = true;
26          } catch (MatrixInversionException x) {
27              theValid = false;
28              theGlobalCC = new double[0];
29          }
30      }
31  
32      public double[] globalCC() {
33          return theGlobalCC;
34      }
35  
36      public boolean isValid() {
37          return theValid;
38      }
39  
40      public String toString() {
41          return MnPrint.toString(this);
42      }
43      private double[] theGlobalCC;
44      private boolean theValid;
45  }
