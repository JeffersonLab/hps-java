9   class SinParameterTransformation {
10  
11      double int2ext(double value, double upper, double lower) {
12          return lower + 0.5 * (upper - lower) * (Math.sin(value) + 1.);
13      }
14  
15      double ext2int(double value, double upper, double lower, MnMachinePrecision prec) {
16          //double piby2 = 2.*FastMath.atan(1.);
17          double piby2 = FastMath.PI / 2.;
18          double distnn = 8. * FastMath.sqrt(prec.eps2());
19          double vlimhi = piby2 - distnn;
20          double vlimlo = -piby2 + distnn;
21  
22          double yy = 2. * (value - lower) / (upper - lower) - 1.;
23          double yy2 = yy * yy;
24          if (yy2 > (1. - prec.eps2())) {
25              if (yy < 0.) {
26                  return vlimlo;
27              } else {
28                  return vlimhi;
29              }
30  
31          } else {
32              return FastMath.asin(yy);
33          }
34      }
35  
36      double dInt2Ext(double value, double upper, double lower) {
37          return 0.5 * Math.abs((upper - lower) * FastMath.cos(value));
38      }
39  }
