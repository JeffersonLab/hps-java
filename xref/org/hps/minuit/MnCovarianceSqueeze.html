10      static MnUserCovariance squeeze(MnUserCovariance cov, int n) {
11          assert (cov.nrow() > 0);
12          assert (n < cov.nrow());
13  
14          MnAlgebraicSymMatrix hess = new MnAlgebraicSymMatrix(cov.nrow());
15          for (int i = 0; i < cov.nrow(); i++) {
16              for (int j = i; j < cov.nrow(); j++) {
17                  hess.set(i, j, cov.get(i, j));
18              }
19          }
20  
21          try {
22              hess.invert();
23          } catch (MatrixInversionException x) {
24              logger.info("MnUserCovariance inversion failed; return diagonal matrix;");
25              MnUserCovariance result = new MnUserCovariance(cov.nrow() - 1);
26              for (int i = 0, j = 0; i < cov.nrow(); i++) {
27                  if (i == n) {
28                      continue;
29                  }
30                  result.set(j, j, cov.get(i, i));
31                  j++;
32              }
33              return result;
34          }
35  
36          MnAlgebraicSymMatrix squeezed = squeeze(hess, n);
37  
38          try {
39              squeezed.invert();
40          } catch (MatrixInversionException x) {
41              logger.info("MnUserCovariance back-inversion failed; return diagonal matrix;");
42              MnUserCovariance result = new MnUserCovariance(squeezed.nrow());
43              for (int i = 0; i < squeezed.nrow(); i++) {
44                  result.set(i, i, 1. / squeezed.get(i, i));
45              }
46              return result;
47          }
48  
49          return new MnUserCovariance(squeezed.data(), squeezed.nrow());
50  
51      }
52  
53      static MinimumError squeeze(MinimumError err, int n) {
54          MnAlgebraicSymMatrix hess = err.hessian();
55          MnAlgebraicSymMatrix squeezed = squeeze(hess, n);
56          try {
57              squeezed.invert();
58          } catch (MatrixInversionException x) {
59              logger.info("MnCovarianceSqueeze: MinimumError inversion fails; return diagonal matrix.");
60              MnAlgebraicSymMatrix tmp = new MnAlgebraicSymMatrix(squeezed.nrow());
61              for (int i = 0; i < squeezed.nrow(); i++) {
62                  tmp.set(i, i, 1. / squeezed.get(i, i));
63              }
64              return new MinimumError(tmp, new MinimumError.MnInvertFailed());
65          }
66  
67          return new MinimumError(squeezed, err.dcovar());
68      }
69  
70      static MnAlgebraicSymMatrix squeeze(MnAlgebraicSymMatrix hess, int n) {
71          assert (hess.nrow() > 0);
72          assert (n < hess.nrow());
73  
74          MnAlgebraicSymMatrix hs = new MnAlgebraicSymMatrix(hess.nrow() - 1);
75          for (int i = 0, j = 0; i < hess.nrow(); i++) {
76              if (i == n) {
77                  continue;
78              }
79              for (int k = i, l = j; k < hess.nrow(); k++) {
80                  if (k == n) {
81                      continue;
82                  }
83                  hs.set(j, l, hess.get(i, k));
84                  l++;
85              }
86              j++;
87          }
88  
89          return hs;
90      }
91      private static final Logger logger = Logger.getLogger(MnCovarianceSqueeze.class.getName());
92  }
