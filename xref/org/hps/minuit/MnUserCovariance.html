1   package org.hps.minuit;
2   
3   /**
4    * MnUserCovariance is the external covariance matrix designed for the
5    * interaction of the user. The result of the minimization (internal covariance
6    * matrix) is converted into the user representable format. It can also be used
7    * as input prior to the minimization. The size of the covariance matrix is
8    * according to the number of variable parameters (free and limited).
9    *
10   */
11  public class MnUserCovariance {
12  
13      private MnUserCovariance(MnUserCovariance other) {
14          theData = other.theData.clone();
15          theNRow = other.theNRow;
16      }
17  
18      MnUserCovariance() {
19          theData = new double[0];
20          theNRow = 0;
21      }
22  
23      /*
24       * covariance matrix is stored in upper triangular packed storage format,
25       * e.g. the elements in the array are arranged like
26       * {a(0,0), a(0,1), a(1,1), a(0,2), a(1,2), a(2,2), ...},
27       * the size is nrow*(nrow+1)/2.
28       */
29      MnUserCovariance(double[] data, int nrow) {
30          if (data.length != nrow * (nrow + 1) / 2) {
31              throw new IllegalArgumentException("Inconsistent arguments");
32          }
33          theData = data;
34          theNRow = nrow;
35      }
36  
37      protected MnUserCovariance clone() {
38          return new MnUserCovariance(this);
39      }
40  
41      public MnUserCovariance(int nrow) {
42          theData = new double[nrow * (nrow + 1) / 2];
43          theNRow = nrow;
44      }
45  
46      public double get(int row, int col) {
47          if (row >= theNRow || col >= theNRow) {
48              throw new IllegalArgumentException();
49          }
50          if (row > col) {
51              return theData[col + row * (row + 1) / 2];
52          } else {
53              return theData[row + col * (col + 1) / 2];
54          }
55      }
56  
57      public void set(int row, int col, double value) {
58          if (row >= theNRow || col >= theNRow) {
59              throw new IllegalArgumentException();
60          }
61          if (row > col) {
62              theData[col + row * (row + 1) / 2] = value;
63          } else {
64              theData[row + col * (col + 1) / 2] = value;
65          }
66      }
67  
68      void scale(double f) {
69          for (int i = 0; i < theData.length; i++) {
70              theData[i] *= f;
71          }
72      }
73  
74      double[] data() {
75          return theData;
76      }
77  
78      public int nrow() {
79          return theNRow;
80      }
81  
82      public int ncol() {
83          return theNRow;
84      }
85  
86      int size() {
87          return theData.length;
88      }
89  
90      public String toString() {
91          return MnPrint.toString(this);
92      }
93      private double[] theData;
94      private int theNRow;
95  }
