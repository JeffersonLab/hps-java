511     public void process(EventHeader event) {
512         aida.tree().cd("/");
513 
514         //Grab all GBL tracks in the event
515         List<Track> tracks = event.get(Track.class, GBLTrackCollectionName);
516 
517         //Grab all the clusters in the event
518         List<SiTrackerHitStrip1D> stripHits = event.get(SiTrackerHitStrip1D.class, stripHitOutputCollectionName);
519 
520         for (Track track : tracks) {
521             //Grab the unused layer on the track
522             int unusedLay = getUnusedSvtLayer(track.getTrackerHits());
523             if (unusedLay == -1)
524                 continue;
525 
526             //Get all track states for this track
527             List<TrackState> TStates = track.getTrackStates();
528 
529             TrackState tState = getTrackState(track, unusedLay);
530             if (tState == null)
531                 continue;
532 
533             //Grab covariance matrix at track states
534             double[] covAtIP = TStates.get(0).getCovMatrix();
535             SymmetricMatrix LocCovAtIP = new SymmetricMatrix(5, covAtIP, true);
536 
537             //Fill track states and errors at IP
538             D0.get(atIP).fill(TStates.get(0).getD0());            
539             Z0.get(atIP).fill(TStates.get(0).getZ0());
540             Tanlambda.get(atIP).fill(TStates.get(0).getTanLambda());
541             Phi0.get(atIP).fill(TStates.get(0).getPhi());
542             Omega.get(atIP).fill(TStates.get(0).getOmega());
543 
544             D0_err.get(atIP).fill(Math.sqrt(LocCovAtIP.e(HelicalTrackFit.dcaIndex, HelicalTrackFit.dcaIndex)));
545             Z0_err.get(atIP).fill(Math.sqrt(LocCovAtIP.e(HelicalTrackFit.z0Index, HelicalTrackFit.z0Index)));
546             Tanlambda_err.get(atIP).fill(Math.sqrt(LocCovAtIP.e(HelicalTrackFit.slopeIndex, HelicalTrackFit.slopeIndex)));
547             Phi0_err.get(atIP).fill(Math.sqrt(LocCovAtIP.e(HelicalTrackFit.phi0Index, HelicalTrackFit.phi0Index)));
548             Omega_err.get(atIP).fill(Math.sqrt(LocCovAtIP.e(HelicalTrackFit.curvatureIndex, HelicalTrackFit.curvatureIndex)));

550             Hep3Vector p = toHep3(tState.getMomentum());
551             double q = -track.getCharge();  //HelicalTrackFit flips sign of charge       
552 
553             if (cleanFEE) {
554                 // Require track to be an electron
555                 if (q < 0)
556                     continue;
557 
558                 // Select around the FEE momentum peak
559                 if (p.magnitude() < 0.75 * ebeam || p.magnitude() > 1.25 * ebeam)
560                     continue;
561             }
562 
563             //See if track is within acceptance of both the axial and stereo sensors of the unused layer
564             Pair<HpsSiSensor, Pair<Integer, Hep3Vector>> axialSensorPair = isWithinSensorAcceptance(track, tState, unusedLay, true, p, bFieldMap);
565             Pair<HpsSiSensor, Pair<Integer, Hep3Vector>> stereoSensorPair = isWithinSensorAcceptance(track, tState, unusedLay, false, p, bFieldMap);

567             //Skip track if it isn't within acceptance of both axial and stereo pairs of a given unused layer
568             if (axialSensorPair == null || stereoSensorPair == null)
569                 continue;

573             HpsSiSensor stereoSensor = stereoSensorPair.getFirst();

796     //Computes weight based on the number of sigmas (u error) the track extrapolates from the edge of the sensor
797     private double findWeight(double y, double yErr, HpsSiSensor sensor) {
798         double readoutPitch = sensor.getReadoutStripPitch();
799         int nChan = sensor.getNumberOfChannels();
800         boolean firstChan = firstChanIsEdge(sensor);
801         double height = readoutPitch * nChan;
802         double distanceToEdge = 0;
803         if (firstChan)
804             distanceToEdge = height / 2 - y;
805         else
806             distanceToEdge = height / 2 + y;
807         double nSig = distanceToEdge / yErr;
808         return computeGaussInt(nSig, 1000);
809     }

811     //Computes gaussian integral numerically from -inf to nSig
812     private double computeGaussInt(double nSig, int nSteps) {
813         double mean = 0;
814         double sigma = 1;
815         double dx = sigma * nSig / (double) nSteps;
816         double integral = 0;
817         for (int i = 0; i < nSteps; i++) {
818             double x = dx * (i + 0.5) + mean;
819             integral += dx * Gauss(x, mean, sigma);
820         }
821         return integral + 0.5;
822     }

824     //Gaussian function
825     private double Gauss(double x, double mean, double sigma) {
826         return 1 / (Math.sqrt(2 * Math.PI * Math.pow(sigma, 2))) * Math.exp(-Math.pow(x - mean, 2) / (2 * Math.pow(sigma, 2)));
827     }

949         double step = step2 - step1;
950 
951         //Grab simple jacobian in lambda phi coordinates
952         BasicMatrix jacPointToPoint = GblUtils.gblSimpleJacobianLambdaPhi(step, cosLambda, abs(bfac));
953 
954         Matrix jacobian = new Matrix(5, 5);
955         for (int i = 0; i < 5; i++)
956             for (int j = 0; j < 5; j++)
957                 jacobian.set(i, j, jacPointToPoint.e(i, j));

960         Matrix ClToPerJac = GblUtils.getCLToPerigeeJacobian(htf, new HpsHelicalTrackFit(TrackUtils.getHTF(tState)), bfield);
961         Matrix PerToClJac = ClToPerJac.inverse();

963         Matrix MsCov = jacobian.times(PerToClJac.times(locCov.times(PerToClJac.transpose())).times(jacobian.transpose()));

968         MsCov2.set(0, 0, MsCov.get(3, 3));

970         MsCov2.set(1, 0, MsCov.get(4, 3));
971         MsCov2.set(1, 1, MsCov.get(4, 4));
972 
973         //Tranform the covariance matrix into the sensor frame u,v to get the final covariance matrix
974         SiSensorElectrodes electrodes = sensor.getReadoutElectrodes(ChargeCarrier.HOLE);
975         Matrix rot = Hep3ToMatrix(electrodes.getGlobalToLocal().getRotation().getRotationMatrix());
976         Matrix measMsCov = rot.times(MsCov2.times(rot.transpose()));
977 
978         //Fill histograms of track parameter errors
979         double d0_err = Math.sqrt(helixCovariance.get(HelicalTrackFit.dcaIndex, HelicalTrackFit.dcaIndex));

982         double phi0_err = Math.sqrt(helixCovariance.get(HelicalTrackFit.phi0Index, HelicalTrackFit.phi0Index));

988         Tanlambda_err.get(sensorName).fill(tanlambda_err);
989         Phi0_err.get(sensorName).fill(phi0_err);
990         Omega_err.get(sensorName).fill(omega_err);
991 
992         //Calculate errors in the u and v directions
993         return new double[]{Math.sqrt(measMsCov.get(0, 0)), Math.sqrt(measMsCov.get(1, 1))};
994     }
