127     protected void detectorChanged(Detector detector) {
128     
129         aida.tree().cd("/");
130         tree = aida.tree();
131         histogramFactory = analysisFactory.createHistogramFactory(tree);
132         
133         if (Double.isNaN(ebeam)) {
134             try {
135                 BeamEnergy.BeamEnergyCollection beamEnergyCollection = this.getConditionsManager()
136                         .getCachedConditions(BeamEnergy.BeamEnergyCollection.class, "beam_energies").getCachedData();
137                 ebeam = beamEnergyCollection.get(0).getBeamEnergy();
138             } catch (Exception e) {
139             }
140         }
141         
142         // Get the HpsSiSensor objects from the tracker detector element
143         sensors = detector.getSubdetector(subdetectorName)
144                           .getDetectorElement().findDescendants(HpsSiSensor.class);
145    
146         // If the detector element had no sensors associated with it, throw
147         // an exception
148         if (sensors.size() == 0) {
149             throw new RuntimeException("No sensors were found in this detector.");
150         }
151         
152         for (HpsSiSensor sensor : sensors) { 
153            
154             // Get the name of the sensor
155             String sensorName = sensor.getName();
156             clusterChargePlots.put(sensor, 
157                     histogramFactory.createHistogram1D(sensorName + " - Cluster Charge", 100, 0, 5000));
158        
159             singleHitClusterChargePlots.put(sensor, 
160                     histogramFactory.createHistogram1D(sensorName + " - Single Hit Cluster Charge", 100, 0, 5000));
161 
162             multHitClusterChargePlots.put(sensor, 
163                     histogramFactory.createHistogram1D(sensorName + " - Multiple Hit Cluster Charge", 100, 0, 5000));
164 
165             signalToNoisePlots.put(sensor, 
166                     histogramFactory.createHistogram1D(sensorName + " - Signal to Noise", 50, 0, 50));
167 
168             singleHitSignalToNoisePlots.put(sensor, 
169                     histogramFactory.createHistogram1D(sensorName + " - Single Hit Signal to Noise", 50, 0, 50));
170         
171             multHitSignalToNoisePlots.put(sensor, 
172                     histogramFactory.createHistogram1D(sensorName + " - Multiple Hit Signal to Noise", 50, 0, 50));
173 
174             clusterSizePlots.put(sensor,
175                     histogramFactory.createHistogram1D(sensorName + " - Cluster Multiplicity", 10, 0, 10));
176             
177             clusterTimePlots.put(sensor,
178                     histogramFactory.createHistogram1D(sensorName + " - Cluster Time", 100, -100, 100));
179             
180             trackClusterTimePlots.put(sensor,
181                     histogramFactory.createHistogram1D(sensorName + " - Track Cluster Time", 100, -100, 100));
182             
183             trackSingleHitClusterTimePlots.put(sensor,
184                     histogramFactory.createHistogram1D(sensorName + " - Track Single Hit Cluster Time", 100, -20, 20));
185             
186             clusterChargeVsTimePlots.put(sensor,
187                     histogramFactory.createHistogram2D(sensorName + " - Cluster Amplitude vs Time", 100, 0, 5000, 100, -100, 100));
188         
189             trackClusterChargePlots.put(sensor, 
190                     histogramFactory.createHistogram1D(sensorName + " - Tracker Cluster Charge", 100, 0, 5000));
191         
192             trackHitSignalToNoisePlots.put(sensor, 
193                     histogramFactory.createHistogram1D(sensorName + " - Track Signal to Noise", 50, 0, 50));
194         
195             trackClusterChargeVsMomentum.put(sensor,
196                     histogramFactory.createHistogram2D(sensorName + " - Cluster Amplitude vs Momentum", 100, 0, 1.5, 100, 0, 5000));
197         
198             trackClusterChargeVsCosTheta.put(sensor,
199                     histogramFactory.createHistogram2D(sensorName + " - Cluster Amplitude vs cos(theta)", 100, -0.1, 0.1, 100, 0, 5000));

203         }

206     @SuppressWarnings({ "unchecked", "rawtypes" })
207     public void process(EventHeader event) { 
208      
209         if (runNumber == -1) runNumber = event.getRunNumber();
210         
211         // If the event doesn't contain fitted raw hits, skip it
212         if (!event.hasCollection(LCRelation.class, fittedHitsCollectionName)) return;
213         
214         // Get the list of fitted hits from the event
215         List<LCRelation> fittedHits = event.get(LCRelation.class, fittedHitsCollectionName);
216         
217         // Map the fitted hits to their corresponding raw hits
218         this.mapFittedRawHits(fittedHits);
219         
220         // If the event doesn't contain any clusters, skip it
221         if (!event.hasCollection(TrackerHit.class, clusterCollectionName)) return;
222         
223         // Get the list of clusters in the event
224         List<TrackerHit> clusters = event.get(TrackerHit.class, clusterCollectionName);
225         //System.out.println("Number of clusters: " + clusters.size()); 
226        
227         for (TrackerHit cluster : clusters) { 
228            
229             // Get the sensor associated with this cluster
230             HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) cluster.getRawHits().get(0)).getDetectorElement();
231             
232             Cluster clusterObject = this.calculateSignalToNoise(cluster.getRawHits());
233             
234             clusterSizePlots.get(sensor).fill(cluster.getRawHits().size());
235             
236             // Fill all plots
237             clusterChargePlots.get(sensor).fill(clusterObject.getAmplitude());
238             signalToNoisePlots.get(sensor).fill(clusterObject.getSignalToNoise());
239             clusterTimePlots.get(sensor).fill(cluster.getTime());
240             clusterChargeVsTimePlots.get(sensor).fill(clusterObject.getAmplitude(), cluster.getTime());
241             
242             if (cluster.getRawHits().size() == 1) { 
243                 singleHitClusterChargePlots.get(sensor).fill(clusterObject.getAmplitude());
244                 singleHitSignalToNoisePlots.get(sensor).fill(clusterObject.getSignalToNoise());
245             } else { 
246                 multHitClusterChargePlots.get(sensor).fill(clusterObject.getAmplitude());
247                 multHitSignalToNoisePlots.get(sensor).fill(clusterObject.getSignalToNoise());
248             }
249         }
250        
251         if (!event.hasCollection(Track.class, trackCollectionName)) return;
252         
253         List<Track> tracks = event.get(Track.class, trackCollectionName);
254         
255         // Get the collection of LCRelations between a stereo hit and the strips making it up
256         List<LCRelation> stereoHitRelations = event.get(LCRelation.class, stereoHitRelationsColName);
257         BaseRelationalTable stereoHitToClusters = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
258         for (LCRelation relation : stereoHitRelations) { 
259             if (relation != null && relation.getFrom() != null && relation.getTo() != null) {
260                 stereoHitToClusters.add(relation.getFrom(), relation.getTo());
261             }
262         }
263         
264         // Get the collection of LCRelations relating RotatedHelicalTrackHits to
265         // HelicalTrackHits
266         List<LCRelation> rotatedHthToHthRelations = event.get(LCRelation.class, rotatedHthRelationsColName);
267         BaseRelationalTable hthToRotatedHth = new BaseRelationalTable(RelationalTable.Mode.ONE_TO_ONE, RelationalTable.Weighting.UNWEIGHTED);
268         for (LCRelation relation : rotatedHthToHthRelations) {
269             if (relation != null && relation.getFrom() != null && relation.getTo() != null) {
270                 hthToRotatedHth.add(relation.getFrom(), relation.getTo());
271             }
272         }
273        
274         // Loop over all of the tracks in the event
275         for(Track track : tracks){
276 
277             boolean isGoodTrack = track.getChi2() < maxChi2 && track.getTrackerHits().size() >= 6;
278             if(trackQualityCut && !isGoodTrack)
279                 continue;
280 
281             // Calculate the momentum of the track
282             HelicalTrackFit hlc_trk_fit = TrackUtils.getHTF(track.getTrackStates().get(0));
283             double B_field = Math.abs(TrackUtils.getBField(event.getDetector()).y());
284             double p = hlc_trk_fit.p(B_field);
285             
286             boolean isFee = p > minFee * ebeam && p < maxFee * ebeam && track.getTrackStates().get(0).getOmega() > 0;
287             
288             if(feeCut && !isFee)
289                 continue;
290 
291             for (TrackerHit rotatedStereoHit : track.getTrackerHits()) { 
292             
293                 // Get the HelicalTrackHit corresponding to the RotatedHelicalTrackHit
294                 // associated with a track
295                 Set<TrackerHit> trackClusters = stereoHitToClusters.allFrom(hthToRotatedHth.from(rotatedStereoHit));
296                 
297                 for (TrackerHit trackCluster : trackClusters) { 
298                 
299                     // Get the sensor associated with this cluster
300                     HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) trackCluster.getRawHits().get(0)).getDetectorElement();
301 
302                     Cluster clusterObject = this.calculateSignalToNoise(trackCluster.getRawHits());
303                    
304                     // Fill all plots
305                     trackClusterChargePlots.get(sensor).fill(clusterObject.getAmplitude());
306                     trackHitSignalToNoisePlots.get(sensor).fill(clusterObject.getSignalToNoise());
307                     trackClusterChargeVsMomentum.get(sensor).fill(p, clusterObject.getAmplitude());
308                     trackClusterChargeVsCosTheta.get(sensor).fill(TrackUtils.getCosTheta(track), clusterObject.getAmplitude());
309                     trackClusterChargeVsSinPhi.get(sensor).fill(Math.sin(TrackUtils.getPhi0(track)), clusterObject.getAmplitude());
310                     trackClusterTimePlots.get(sensor).fill(trackCluster.getTime());
311                     if(trackCluster.getRawHits().size() == 1)
312                         trackSingleHitClusterTimePlots.get(sensor).fill(trackCluster.getTime());
313                 }
314             }
315         }
316     }

318     private Cluster calculateSignalToNoise(List<Object> rawHitObjects) {
319        
320         double amplitudeSum = 0;
321         double noiseSquared = 0;
322            
323         // Get the sensor associated with this cluster
324         HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) rawHitObjects.get(0)).getDetectorElement();
325            
326         // Loop over all of the RawTrackerHits associated with this object and
327         // calculate the signal to noise. 
328         for (Object rawHitObject : rawHitObjects) {
329 
330             // Cast the raw hit object to a RawTrackerHit
331             RawTrackerHit rawHit = (RawTrackerHit) rawHitObject; 
332             
333             // Get the channel of the raw hit
334             int channel = rawHit.getIdentifierFieldValue("strip");
335                 
336             
337             // Add the amplitude of that channel to the total amplitude
338             amplitudeSum += FittedRawTrackerHit.getAmp(this.getFittedHit(rawHit));
339                 
340             // Calculate the mean noise for the channel
341             double channelNoise = 0;
342             for (int sampleN = 0; sampleN < 6; sampleN++) { 
343                 channelNoise += sensor.getNoise(channel, sampleN);
344             }
345             channelNoise = channelNoise/6;
346            
347             noiseSquared += channelNoise*channelNoise;
348             
349         }
350         
351         // Calculate the cluster noise 
352         double noise = Math.sqrt(noiseSquared/rawHitObjects.size());
353             
354         return new Cluster(amplitudeSum, noise);
355     }

357     /**
358      *  Method that creates a map between a fitted raw hit and it's corresponding raw fit
359      *  
360      * @param fittedHits : List of fitted hits to map
361      */
362     private void mapFittedRawHits(List<LCRelation> fittedHits) { 
363         
364         // Clear the fitted raw hit map of old values
365         fittedRawTrackerHitMap.clear();
366        
367         // Loop through all fitted hits and map them to their corresponding raw hits
368         for (LCRelation fittedHit : fittedHits) { 
369             fittedRawTrackerHitMap.put(FittedRawTrackerHit.getRawTrackerHit(fittedHit), fittedHit);
370         }
371     }
372     
373     /**
374      * 
375      * @param rawHit
376      * @return
377      */
378     private LCRelation getFittedHit(RawTrackerHit rawHit) { 
379         return fittedRawTrackerHitMap.get(rawHit);
380     }

386     private class Cluster { 
387         
388         private double amplitude;
389         private double noise; 
390         
391         /** Default Constructor */ 
392         public Cluster(double amplitude, double noise) {
393             this.amplitude = amplitude; 
394             this.noise = noise; 
395         }
396         
397         public double getAmplitude() { 
398             return this.amplitude; 
399         }
400         
401         public double getNoise() { 
402             return this.noise;
403         }
404         
405         public double getSignalToNoise() { 
406             return this.getAmplitude()/this.getNoise();
407         }
