75      private static class MilleParameterSet {
76  
77          private IDetectorElement _det = null;
78          List<MilleParameter> params = new ArrayList<MilleParameter>();
79  
80          public MilleParameterSet(IDetectorElement d) {
81              setDetector(d);
82          }
83  
84          public void setDetector(IDetectorElement d) {
85              _det = d;
86          }
87  
88          public IDetectorElement getDetector() {
89              return _det;
90          }
91  
92          public void add(MilleParameter par) {
93              params.add(par);
94          }
95  
96          public Hep3Vector getLocalTranslation() {
97              Map<String, Double> m = new HashMap<String, Double>();
98              for (MilleParameter p : params) {
99                  if (p.getType() == 1) {
100                     if (p.getDim() == 1)
101                         m.put("u", p.getValue());
102                     else if (p.getDim() == 2)
103                         m.put("v", p.getValue());
104                     else
105                         m.put("w", p.getValue());
106                 }
107             }
108             if (m.size() != 3) {
109                 System.out.println("bad trans!!");
110                 System.exit(1);
111             }
112             return new BasicHep3Vector(m.get("u"), m.get("v"), m.get("w"));
113 
114         }
115 
116         public Hep3Vector getLocalRotation() {
117             Map<String, Double> m = new HashMap<String, Double>();
118             for (MilleParameter p : params) {
119                 if (p.getType() == 2) {
120                     if (p.getDim() == 1)
121                         m.put("alpha", p.getValue());
122                     else if (p.getDim() == 2)
123                         m.put("beta", p.getValue());
124                     else
125                         m.put("gamma", p.getValue());
126                 }
127             }
128             if (m.size() != 3) {
129                 System.out.println("bad rot!!");
130                 System.exit(1);
131             }
132             return new BasicHep3Vector(m.get("alpha"), m.get("beta"), m.get("gamma"));
133         }
134 
135         public Hep3Vector getGlobalTranslation() {
136             ITransform3D localToGlobal = getLocalToGlobal();
137             return localToGlobal.getRotation().rotated(getLocalTranslation());
138         }
139 
140         public double getGlobalTranslation(int d) {
141             return getGlobalTranslation().v()[d - 1];
142         }
143 
144         public Hep3Vector getGlobalRotation() {
145             ITransform3D localToGlobal = getLocalToGlobal();
146             return localToGlobal.getRotation().rotated(getLocalRotation());
147         }
148 
149         public double getGlobalRotation(int d) {
150             return getGlobalRotation().v()[d - 1];
151         }
152 
153         public ITransform3D getLocalToGlobal() {
154             ITransform3D localToGlobal = ((SiSensor) _det).getReadoutElectrodes(ChargeCarrier.HOLE).getLocalToGlobal();
155             return localToGlobal;
156         }

160     public static void main(String[] args) {
161 
162         // Setup command line input
163         Options options = createCmdLineOpts();
164         if (args.length == 0) {
165             printHelpAndExit(options);
166         }
167 
168         CommandLineParser parser = new PosixParser();
169         CommandLine cl = null;
170         try {
171             cl = parser.parse(options, args);
172         } catch (ParseException e) {
173             throw new RuntimeException("Problem parsing command line options.", e);
174         }
175 
176         String compactFilename = null;
177         if (cl.hasOption("c")) {
178             compactFilename = cl.getOptionValue("c");
179         } else {
180             printHelpAndExit(options);
181         }
182 
183         if (cl.hasOption("d")) {
184             detectorName = cl.getOptionValue("d");
185         } else {
186             printHelpAndExit(options);
187         }
188 
189         if (cl.hasOption("r")) {
190             runNumber = Integer.parseInt(cl.getOptionValue("r"));
191         } else {
192             printHelpAndExit(options);
193         }
194 
195         String compactFilenameNew = "compact_new.xml";
196         if (cl.hasOption("o")) {
197             compactFilenameNew = cl.getOptionValue("o");
198         }
199 
200         File compactFile = new File(compactFilename);
201 
202         // read XML
203         SAXBuilder builder = new SAXBuilder();
204         builder.setExpandEntities(false);
205 
206         Document compact_document = null;
207         try {
208             compact_document = (Document) builder.build(compactFile);
209         } catch (JDOMException | IOException e1) {
210             throw new RuntimeException("problem with JDOM ", e1);
211         }
212 
213         // read detector geometry
214         FileInputStream inCompact;
215         try {
216             inCompact = new FileInputStream(compactFile);
217         } catch (FileNotFoundException e) {
218             throw new RuntimeException("cannot open compact file", e);
219         }
220 
221         GeometryReader reader = new GeometryReader();
222         Detector det;
223         try {
224             det = reader.read(inCompact);
225         } catch (IOException | JDOMException | ElementCreationException e) {
226             throw new RuntimeException("problem reading compact file", e);
227         }
228 
229         // set conditions in order to be able to determine which sensors are where in the geometry
230         setConditions(detectorName, runNumber);
231 
232         // Loop over all millepede input files and match parameters with detectors
233 
234         List<MilleParameterSet> list_det = new ArrayList<MilleParameterSet>();
235 
236         FileInputStream inMille = null;
237         BufferedReader brMille = null;
238         try {
239             for (String milleFilename : cl.getArgs()) {
240                 inMille = new FileInputStream(milleFilename);
241                 brMille = new BufferedReader(new InputStreamReader(inMille));
242                 String line;
243                 while ((line = brMille.readLine()) != null) {
244                     // System.out.printf("%s\n",line);
245                     if (!line.contains("Parameter") && !line.contains("!")) {
246 
247                         MilleParameter par = new MilleParameter(line);
248                         // System.out.println(par.getXMLName() + " " + par.getValue());
249 
250                         SiSensor detEle = getTrackerDetElement(det, par);
251                         if (detEle == null) {
252                             System.out.println("Couldn't find detector for param " + par.getId());
253                             System.exit(1);
254                         }
255                         System.out.println("Found detector  " + detEle.getName());
256                         if (detEle.getClass().isInstance(SiSensor.class)) {
257                             System.out.println("yeah");
258                         }
259 
260                         // do we have it already?
261                         MilleParameterSet useSet = null;
262                         for (MilleParameterSet set : list_det) {
263                             if (set.getDetector() == detEle) {
264                                 useSet = set;
265                             }
266                         }
267 
268                         if (useSet == null) {
269                             useSet = new MilleParameterSet(detEle);
270                             list_det.add(useSet);
271                         }
272 
273                         // add the parameter
274                         useSet.add(par);
275 
276                     }
277                 }
278                 brMille.close();
279             }
280         } catch (IOException e) {
281             throw new RuntimeException("problem reading mille file", e);
282         }
283 
284         for (MilleParameterSet set : list_det) {
285             System.out.println("Detector " + set.getDetector().getName());
286             List<MilleParameter> pars = set.params;
287             for (MilleParameter p : pars) {
288                 System.out.println(p.getXMLName() + " " + p.getValue());
289                 Element node = findXMLNode(compact_document, p.getXMLName());
290                 double value = 0.0;
291                 if (p.getType() == 1) {
292                     value = set.getGlobalTranslation(p.getDim());
293                 } else if (p.getType() == 2) {
294                     value = set.getGlobalRotation(p.getDim());
295                 } else {
296                     System.out.println("This type is illdefnied " + p.getType());
297                     System.exit(1);
298                 }
299                 node.setAttribute("value", String.format("%.6f", value));
300             }
301             Hep3Vector u = getMeasuredCoordinate((SiSensor) set.getDetector());
302             System.out.println("u  " + u.toString());
303             System.out.println("t (local)  " + set.getLocalTranslation().toString());
304             System.out.println("t (global) " + set.getGlobalTranslation().toString());
305             System.out.println("r (local)  " + set.getLocalRotation().toString());
306             System.out.println("r (global) " + set.getGlobalRotation().toString());
307 
308         }
309 
310         // Save new XML file
311 
312         XMLOutputter xmlOutput = new XMLOutputter();
313         // display nice
314         // xmlOutput.setFormat(Format.getPrettyFormat());
315         try {
316             xmlOutput.output(compact_document, new FileWriter(compactFilenameNew));
317         } catch (IOException e) {
318             throw new RuntimeException("problem with xml output", e);
319         }
320 
321     }

323     private static Element findXMLNode(Document document, String name) {
324         Element rootNode = document.getRootElement();
325         List list = rootNode.getChildren("define");
326         for (int i = 0; i < list.size(); ++i) {
327             Element node = (Element) list.get(i);
328             List llist = node.getChildren("constant");
329             // System.out.println("length of list " + llist.size());
330             for (int ii = 0; ii < llist.size(); ++ii) {
331                 Element nnode = (Element) llist.get(ii);
332                 // System.out.println("node name " + nnode.getAttributeValue("name") + " " +
333                 // nnode.getAttributeValue("value") );
334                 // if(nnode.getAttributeValue("name").contains(name)) {
335                 if (nnode.getAttributeValue("name").compareTo(name) == 0) {
336                     return nnode;
337                 }
338             }
339         }
340 
341         return null;
342     }

358     private static SiSensor getTrackerDetElement(Detector det, MilleParameter par) {
359         List<SiSensor> sensors = det.getSubdetector("Tracker").getDetectorElement().findDescendants(SiSensor.class);
360         // List<SiTrackerModule> modules = det.getDetectorElement().findDescendants(SiTrackerModule.class);
361         // System.out.printf("%d sensors\n",sensors.size());
362         for (SiSensor module : sensors) {
363             // Create DAQ Maps
364             boolean isTop = ((HpsSiSensor) module).isTopLayer();
365             int h = par.getHalf();
366             if ((isTop && h == 1) || (!isTop && h == 2)) {
367                 int layer = ((HpsSiSensor) module).getLayerNumber();
368                 if (layer == par.getSensor()) {
369                     // found match
370                     return module;
371                 }
372             }
373         }
374         return null;

406     private static Hep3Vector getTrackingMeasuredCoordinate(SiSensor sensor) {
407         // p-side unit vector
408         ITransform3D electrodes_to_global = sensor.getReadoutElectrodes(ChargeCarrier.HOLE).getLocalToGlobal();
409         Hep3Vector measuredCoordinate = sensor.getReadoutElectrodes(ChargeCarrier.HOLE).getMeasuredCoordinate(0);
410         measuredCoordinate = VecOp.mult(VecOp.mult(CoordinateTransformations.getMatrix(), electrodes_to_global
411                 .getRotation().getRotationMatrix()), measuredCoordinate);
412         return measuredCoordinate;
413     }

415     private static Hep3Vector getMeasuredCoordinate(SiSensor sensor) {
416         // p-side unit vector
417         ITransform3D electrodes_to_global = sensor.getReadoutElectrodes(ChargeCarrier.HOLE).getLocalToGlobal();
418         Hep3Vector measuredCoordinate = sensor.getReadoutElectrodes(ChargeCarrier.HOLE).getMeasuredCoordinate(0);
419         measuredCoordinate = VecOp.mult(electrodes_to_global.getRotation().getRotationMatrix(), measuredCoordinate);
420         return measuredCoordinate;
421     }

423     private static class SiSensorDetAlignConstants extends DetAlignConstants {
424 
425         public SiSensorDetAlignConstants(IDetectorElement det) {
426             super(det);
427         }
428 
429         public void transform() {
430             ITransform3D localToGlobal = null;
431             if (_det.getClass().isInstance(SiSensor.class)) {
432                 localToGlobal = ((SiSensor) _det).getReadoutElectrodes(ChargeCarrier.HOLE).getLocalToGlobal();
433             }
434             // Translation
435             Hep3Vector t_local = _constants.getTranslationVector();
436             Hep3Vector t_global = localToGlobal.getRotation().rotated(t_local);
437             _constants.addGlobalTranslation(t_global);
438             // Rotation
439             Hep3Vector r_local = _constants.getRotationVector();
440             Hep3Vector r_global = localToGlobal.getRotation().rotated(r_local);
441             _constants.addGlobalRotation(r_global);
442         }
443 
444     }

475     private static class AlignConstants<K, V> extends HashMap<String, Double> {
476 
477         List<MilleParameter> _pars = new ArrayList<MilleParameter>();
478 
479         public AlignConstants() {
480             super();
481         }
482 
483         public void add(MilleParameter p) {
484             _pars.add(p);
485             if (p.getType() == 1) {
486                 if (p.getDim() == 1)
487                     this.put("u", p.getValue());
488                 else if (p.getDim() == 2)
489                     this.put("v", p.getValue());
490                 else
491                     this.put("w", p.getValue());
492             } else {
493                 if (p.getDim() == 1)
494                     this.put("alpha", p.getValue());
495                 else if (p.getDim() == 2)
496                     this.put("beta", p.getValue());
497                 else
498                     this.put("gamma", p.getValue());
499             }
500         }
501 
502         public void print() {
503             for (Entry<String, Double> e : this.entrySet()) {
504                 System.out.println(e.getKey() + " " + e.getValue());
505             }
506         }
507 
508         public Hep3Vector getTranslationVector() {
509             if (!this.containsKey("u") || !this.containsKey("v") || !this.containsKey("w")) {
510                 System.out.println("missing pars for translation");
511                 print();
512                 System.exit(1);
513             }
514             return new BasicHep3Vector(this.get("u"), this.get("v"), this.get("w"));
515         }
516 
517         public Hep3Vector getTranslationVectorGlobal() {
518             if (!this.containsKey("x") || !this.containsKey("y") || !this.containsKey("z")) {
519                 System.out.println("missing pars for global translation");
520                 print();
521                 System.exit(1);
522             }
523             return new BasicHep3Vector(this.get("x"), this.get("y"), this.get("z"));
524         }
525 
526         public Hep3Vector getRotationVector() {
527             if (!this.containsKey("alpha") || !this.containsKey("beta") || !this.containsKey("gamma")) {
528                 System.out.println("missing pars for rotation");
529                 print();
530                 System.exit(1);
531             }
532             return new BasicHep3Vector(this.get("alpha"), this.get("beta"), this.get("gamma"));
533         }
534 
535         public Hep3Vector getRotationVectorGlobal() {
536             if (!this.containsKey("rx") || !this.containsKey("ry") || !this.containsKey("rz")) {
537                 System.out.println("missing pars for global rotation");
538                 print();
539                 System.exit(1);
540             }
541             return new BasicHep3Vector(this.get("rx"), this.get("ry"), this.get("rz"));
542         }
