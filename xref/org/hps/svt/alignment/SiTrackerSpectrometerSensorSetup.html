39      public void detectorChanged(Detector detector) {
40          if (subdetectorName == null) {
41              throw new RuntimeException("The subdetectorName was not set.");
42          }
43  
44          Subdetector subdetector = detector.getSubdetector(subdetectorName);
45          if (subdetector instanceof SiTrackerSpectrometer) {
46              setupSensorDetectorElements(subdetector);
47          } else {
48              throw new RuntimeException("The subdetector " + subdetectorName
49                      + " is not an instance of SiTrackerSpectrometer.");
50          }
51      }

200     private int GetIdentifierModule(String mylayer) {
201         int gid = -1;
202         if (mylayer.contains("1") || mylayer.contains("2"))
203             gid = 1;
204         if (mylayer.contains("3") || mylayer.contains("4"))
205             gid = 2;
206         if (mylayer.contains("5") || mylayer.contains("6"))
207             gid = 3;
208         if (mylayer.contains("7") || mylayer.contains("8"))
209             gid = 4;
210         if (mylayer.contains("9") || mylayer.contains("10"))
211             gid = 5;
212 
213         return gid; // return top/bottom plates
214     }
