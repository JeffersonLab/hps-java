73      public void PrintResidualsAndDerivatives(Track track) {
74          SeedTrack st = (SeedTrack) track;
75          SeedCandidate seed = st.getSeedCandidate();
76          Map<HelicalTrackHit, MultipleScatter> msmap = seed.getMSMap();
77          _trk = seed.getHelix();
78          List<TrackerHit> hitsOnTrack = track.getTrackerHits();
79          for (TrackerHit hit : hitsOnTrack) {
80              HelicalTrackHit htc = (HelicalTrackHit) hit;
81              double msdrphi = msmap.get(htc).drphi();
82              double msdz = msmap.get(htc).dz();
83              double sHit = _trk.PathMap().get(htc);
84              HelicalTrackCross cross = (HelicalTrackCross) htc;
85              List<HelicalTrackStrip> clusterlist = cross.getStrips();
86              TrackDirection trkdir = HelixUtils.CalculateTrackDirection(_trk, sHit);
87              cross.setTrackDirection(trkdir, _trk.covariance());
88              for (HelicalTrackStrip cl : clusterlist) {
89                  CalculateLocalDerivatives(cl);
90                  CalculateGlobalDerivatives(cl);
91                  CalculateResidual(cl, msdrphi, msdz);
92                  // CalculateResidual(cl, 0,0);
93                  PrintStripResiduals(cl);
94              }
95          }
96          AddTarget(0.1, 0.02);
97      }

99      private void CalculateLocalDerivatives(HelicalTrackStrip strip) {
100         // get track parameters.
101         double d0 = _trk.dca();
102         double z0 = _trk.z0();
103         double slope = _trk.slope();
104         double phi0 = _trk.phi0();
105         double R = _trk.R();
106         // strip origin is defined in the tracking coordinate system (x=beamline)
107         double xint = strip.origin().x();
108         double s = HelixUtils.PathToXPlane(_trk, xint, smax, _nlc).get(0);
109         double phi = s / R - phi0;
110         double[][] dfdq = new double[3][5];
111         // dx/dq
112         // these are wrong for X, but for now it doesn't matter
113         dfdq[0][0] = Math.sin(phi0);
114         dfdq[0][1] = 0;
115         dfdq[0][2] = 0;
116         dfdq[0][3] = d0 * Math.cos(phi0) + R * Math.sin(phi0) - s * Math.cos(phi0);
117         dfdq[0][4] = (phi - phi0) * Math.cos(phi0);
118         double[] mydydq = dydq(R, d0, phi0, xint, s);
119         double[] mydzdq = dzdq(R, d0, phi0, xint, slope, s);
120         for (int i = 0; i < 5; i++) {
121             dfdq[1][i] = mydydq[i];
122             dfdq[2][i] = mydzdq[i];
123         }
124 
125         BasicMatrix dfdqGlobal = FillMatrix(dfdq, 3, 5);
126         Hep3Matrix trkToStrip = getTrackToStripRotation(strip);
127         _dfdq = (BasicMatrix) MatrixOp.mult(trkToStrip, dfdqGlobal);
128 
129         if (_DEBUG) {
130             double[] trackpars = {d0, z0, slope, phi0, R, s, xint};
131             System.out.println("Strip Origin: ");
132             System.out.println(strip.origin());
133             System.out.println("trkToStrip Rotation:");
134             System.out.println(trkToStrip.toString());
135             printDerivatives(trackpars, dfdq);
136         }
137     }

139     private void CalculateGlobalDerivatives(HelicalTrackStrip strip) {
140         // 1st index = alignment parameter (only u so far)
141         // 2nd index = residual coordinate (on du so far)
142 
143         double[][] dfdpLab = new double[3][1];
144         dfdpLab[0][0] = 0; // df/dx
145         dfdpLab[1][0] = 0; // df/dy
146         dfdpLab[2][0] = 1; // df/dz
147         BasicMatrix _dfdpLab = FillMatrix(dfdpLab, 3, 1);
148         Hep3Matrix trkToStrip = getTrackToStripRotation(strip);
149         _dfdp = (BasicMatrix) MatrixOp.mult(trkToStrip, _dfdpLab);
150         if (_DEBUG) {
151             System.out.printf("dfdz = %5.5f     %5.5f   %5.5f\n", _dfdp.e(0, 0), _dfdp.e(1, 0), _dfdp.e(2, 0));
152         }
153         _globalLabel[0] = GetIdentifier(strip);
154         // _globalLabel[0] = GetIdentifierModule(strip);

158     private void CalculateResidual(HelicalTrackStrip strip, double msdrdphi, double msdz) {
159 
160         Hep3Vector u = strip.u();
161         Hep3Vector v = strip.v();
162         Hep3Vector w = strip.w();
163         Hep3Vector corigin = strip.origin();
164         double phi0 = _trk.phi0();
165         double R = _trk.R();
166         double xint = strip.origin().x();
167         double s = HelixUtils.PathToXPlane(_trk, xint, smax, _nlc).get(0);
168         double phi = s / R - phi0;
169         Hep3Vector trkpos = HelixUtils.PointOnHelix(_trk, s);
170 
171         // System.out.println("trkpos = "+trkpos.toString());
172         // System.out.println("origin = "+corigin.toString());
173 
174         Hep3Vector mserr = new BasicHep3Vector(msdrdphi * Math.sin(phi), msdrdphi * Math.sin(phi), msdz);
175         Hep3Vector vdiffTrk = VecOp.sub(trkpos, corigin);
176         Hep3Matrix trkToStrip = getTrackToStripRotation(strip);
177         Hep3Vector vdiff = VecOp.mult(trkToStrip, vdiffTrk);
178         double umc = vdiff.x();
179         double vmc = vdiff.y();
180         double wmc = vdiff.z();
181         double umeas = strip.umeas();
182         double uError = strip.du();
183         double msuError = VecOp.dot(mserr, u);
184         double vmeas = 0;
185         double vError = (strip.vmax() - strip.vmin()) / Math.sqrt(12);
186         double wmeas = 0;
187         double wError = 0.001;
188         // System.out.println("strip error="+uError+"; ms error ="+msuError);
189         _resid[0] = umeas - umc;
190         _error[0] = Math.sqrt(uError * uError + msuError * msuError);
191         _resid[1] = vmeas - vmc;
192         _error[1] = vError;
193         _resid[2] = wmeas - wmc;
194         _error[2] = wError;
195         if (_DEBUG) {
196             System.out.println("Strip Origin: ");
197             System.out.println(corigin.toString());
198             System.out.println("Position on Track:");
199             System.out.println(trkpos.toString());
200             System.out.println("vdiff :");
201             System.out.println(vdiff.toString());
202             System.out.println("u :");
203             System.out.println(u.toString());
204             System.out.println("umeas = " + umeas + "; umc = " + umc);
205             System.out.println("udiff = " + _resid[0] + " +/- " + _error[0]);
206 
207         }
208 
209     }

211     public double[] getResidual(Track track, int layer) {
212         double[] res = new double[7];
213         SeedTrack st = (SeedTrack) track;
214         SeedCandidate seed = st.getSeedCandidate();
215         Map<HelicalTrackHit, MultipleScatter> msmap = seed.getMSMap();
216         _trk = seed.getHelix();
217         List<TrackerHit> hitsOnTrack = track.getTrackerHits();
218         for (TrackerHit hit : hitsOnTrack) {
219             HelicalTrackHit htc = (HelicalTrackHit) hit;
220             double sHit = _trk.PathMap().get(htc);
221             HelicalTrackCross cross = (HelicalTrackCross) htc;
222             List<HelicalTrackStrip> clusterlist = cross.getStrips();
223             TrackDirection trkdir = HelixUtils.CalculateTrackDirection(_trk, sHit);
224             double msdrphi = msmap.get(htc).drphi();
225             double msdz = msmap.get(htc).dz();
226             cross.setTrackDirection(trkdir, _trk.covariance());
227             for (HelicalTrackStrip cl : clusterlist) {
228                 if (cl.layer() == layer) {
229                     CalculateResidual(cl, msdrphi, msdz);
230                     res[0] = _resid[0];
231                     res[1] = _resid[1];
232                     res[2] = _resid[2];
233                     res[3] = _error[0];
234                     res[4] = _error[1];
235                     res[5] = _error[2];
236                     res[6] = layer;
237                     if (hit.getPosition()[2] < 0)
238                         res[6] = layer + 10;
239                 }
240             }
241         }
242         return res;
243 
244     }

246     public void AddTarget(double beamdy, double beamdz) {
247         double[][] dfdp = new double[3][1];
248         double d0 = _trk.dca();
249         double z0 = _trk.z0();
250         double slope = _trk.slope();
251         double phi0 = _trk.phi0();
252         double R = _trk.R();
253         double xint = 0; // target
254         double s = HelixUtils.PathToXPlane(_trk, xint, smax, _nlc).get(0);
255         Hep3Vector ptAtTarget = HelixUtils.PointOnHelix(_trk, s);
256         double[] mydydq = dydq(R, d0, phi0, xint, s);
257         double[] mydzdq = dzdq(R, d0, phi0, xint, slope, s);
258         _resid[0] = ptAtTarget.z();
259         _resid[1] = ptAtTarget.y();
260         _resid[2] = ptAtTarget.x();
261         _error[0] = beamdz;
262         _error[1] = beamdy;
263         _error[2] = 666;
264         dfdp[0][0] = 1;
265         dfdp[1][0] = 0;
266         dfdp[2][0] = 0;
267         _dfdp = FillMatrix(dfdp, 3, 1);
268         _globalLabel[0] = 666;
269         pWriter.printf("%4d\n", 666);
270         pWriter.printf("%5.5e %5.5e %5.5e\n", _resid[0], _resid[1], _resid[2]);
271         pWriter.printf("%5.5e %5.5e %5.5e\n", _error[0], _error[1], _error[2]);
272         for (int i = 0; i < _nlc; i++) {
273             pWriter.printf("%5.5e %5.5e -1.0\n", mydzdq[i], mydydq[i]);
274         }
275         for (int j = 0; j < _ngl; j++) {
276             pWriter.printf("%5.5e %5.5e %5.5e   %5d\n", _dfdp.e(0, j), _dfdp.e(1, j), _dfdp.e(2, j), _globalLabel[j]);
277         }

436     private double dsdR(double R, double d0, double phi0, double xint) {
437         double sqrtTerm = Sqrt(R * R - Math.pow(((d0 - R) * Sin(phi0) + xint), 2));
438 
439         double rsign = Math.signum(R);
440         double dsdr = (1 / sqrtTerm)
441                 * ((-rsign * xint) + (-rsign) * d0 * Sin(phi0) + ArcTan(R * Cos(phi0), (-R) * Sin(phi0)) * sqrtTerm - ArcTan(
442                         rsign * sqrtTerm, xint + (d0 - R) * Sin(phi0)) * sqrtTerm);
443 
444         if (_DEBUG)
445             System.out.println("xint = " + xint + "; dsdr = " + dsdr);
446         return dsdr;
447 
448     }

450     private double dsdphi(double R, double d0, double phi0, double xint) {
451         double sqrtTerm = Sqrt(R * R - Math.pow(((d0 - R) * Sin(phi0) + xint), 2));
452         double rsign = Math.signum(R);
453         double dsdphi = R * (sqrtTerm + rsign * d0 * Cos(phi0) - rsign * R * Cos(phi0)) / sqrtTerm;
454         if (_DEBUG)
455             System.out.println("xint = " + xint + "; dsdphi = " + dsdphi);
456         return dsdphi;
457     }

459     private double dsdd0(double R, double d0, double phi0, double xint) {
460         double sqrtTerm = Sqrt(R * R - Math.pow(((d0 - R) * Sin(phi0) + xint), 2));
461         double rsign = Math.signum(R);
462         double dsdd0 = rsign * (R * Sin(phi0)) / sqrtTerm;
463         if (_DEBUG)
464             System.out.println("xint = " + xint + "; dsdd0 = " + dsdd0);
465         return dsdd0;
466     }

468     private double[] dydq(double R, double d0, double phi0, double xint, double s) {
469         double[] dy = new double[5];
470         // dy[0] = Cos(phi0) + Cot(phi0 - s / R) * Csc(phi0 - s / R) * dsdd0(R, d0, phi0, xint);
471         dy[0] = Cos(phi0) - Sec(phi0 - s / R) * Tan(phi0 - s / R) * dsdd0(R, d0, phi0, xint);
472         dy[1] = 0;
473         dy[2] = 0;
474         // dy[3] = (-(d0 - R)) * Sin(phi0) - R * Cot(phi0 - s / R) * Csc(phi0 - s / R) * (1 - dsdphi(R, d0, phi0, xint)
475         // / R);
476         dy[3] = (-(d0 - R)) * Sin(phi0) + Sec(phi0 - s / R) * Tan(phi0 - s / R) * (R - dsdphi(R, d0, phi0, xint));
477         // dy[4] = -Cos(phi0) + Csc(phi0 - s / R) - R * Cot(phi0 - s / R) * Csc(phi0 - s / R) * (s / (R * R) - dsdR(R,
478         // d0, phi0, xint) / R);
479         dy[4] = -Cos(phi0) + Sec(phi0 - s / R) + (1 / R) * Sec(phi0 - s / R) * Tan(phi0 - s / R)
480                 * (s - R * dsdR(R, d0, phi0, xint));
481         return dy;
482     }

484     private double[] dzdq(double R, double d0, double phi0, double xint, double slope, double s) {
485         double[] dz = new double[5];
486         dz[0] = slope * dsdd0(R, d0, phi0, xint);
487         dz[1] = 1;
488         dz[2] = s;
489         dz[3] = slope * dsdphi(R, d0, phi0, xint);
490         dz[4] = slope * dsdR(R, d0, phi0, xint);
491         return dz;
492     }
