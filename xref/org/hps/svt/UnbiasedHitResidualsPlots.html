174     public void process(EventHeader event){
175         aida.tree().cd("/");
176         
177         //Grab all GBL tracks in the event
178         List<Track> tracks = event.get(Track.class,GBLTrackCollectionName);
179         
180         //Grab all the clusters in the event
181         List<SiTrackerHitStrip1D> stripHits = event.get(SiTrackerHitStrip1D.class, stripHitOutputCollectionName);
182     
183         for(Track track:tracks){
184             //Grab the unused layer on the track
185             int unusedLay = getUnusedSvtLayer(track.getTrackerHits());   
186             if(unusedLay == -1) continue;
187             
188             TrackState tState = getTrackState(track,unusedLay);
189             if(tState == null){
190                 continue;
191             }
192         
193             Hep3Vector p = toHep3(tState.getMomentum());
194             double q = track.getCharge();
195             
196             if(cleanFEE){
197                 // Require track to be an electron
198                 if(q < 0) continue;
199                 
200                 // Select around the FEE momentum peak
201                 if(p.magnitude() < 0.75*ebeam || p.magnitude() > 1.25*ebeam) continue;
202             }
203         
204             //See if track is within acceptance of both the axial and stereo sensors of the unused layer
205             Pair<HpsSiSensor,Pair<Integer,Hep3Vector>> axialSensorPair = isWithinSensorAcceptance(track,tState,unusedLay,true, p,bFieldMap);
206             Pair<HpsSiSensor,Pair<Integer,Hep3Vector>> stereoSensorPair = isWithinSensorAcceptance(track,tState,unusedLay,false,p,bFieldMap);           
207             
208             //Skip track if it isn't within acceptance of both axial and stereo pairs of a given unused layer
209             if(axialSensorPair == null || stereoSensorPair == null) continue;
210             
211             //Fill momentum histogram
212             momentum.fill(p.magnitude());
213            
214             //Set axial and stereo sensors of the missing layer
215             HpsSiSensor axialSensor = axialSensorPair.getFirst();
216             HpsSiSensor stereoSensor = stereoSensorPair.getFirst();
217             
218             String sensorAxialName = axialSensor.getName();
219             String sensorStereoName = stereoSensor.getName();
220             
221             //Grab the track extrapolations at each sensor
222             Hep3Vector axialExtrapPosSensor = axialSensorPair.getSecond().getSecond();
223             Hep3Vector stereoExtrapPosSensor = stereoSensorPair.getSecond().getSecond();
224             
225             double UresidualAxial = 9999;
226             double UresidualStereo = 9999;
227             
228             boolean isSingleHitAxial = true;
229             boolean isSingleHitStereo = true;
230             
231             //Loop over all reconstructed 1D hits on sensor of interest in the events
232             for(SiTrackerHitStrip1D hit:stripHits){
233                 //Get the sensor and position of the hit
234                 HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) hit.getRawHits().get(0)).getDetectorElement();
235                 double[] hitPos = hit.getPosition();
236                 //Change to sensor coordinates
237                 Hep3Vector hitPosSensor = globalToSensor(toHep3(hitPos),sensor);
238                 //Check to see if the sensor of this hit is the same sensor you expect to see an axial hit
239                 if(sensorAxialName == sensor.getName()){
240                     //Compute the residual between extrapolated track and hit position
241                     //Keep the value of the smallest residual
242                     double Uresidual = axialExtrapPosSensor.x() - hitPosSensor.x();
243                     if(Math.abs(Uresidual) < Math.abs(UresidualAxial)){
244                         UresidualAxial = Uresidual;
245                         if(hit.getRawHits().size() > 1) isSingleHitAxial = false;
246                     }
247                 }
248               //Check to see if the sensor of this hit is the same sensor you expect to see a stereo hit
249                 if(sensorStereoName == sensor.getName()){
250                     //Compute the residual between extrapolated track and hit position
251                     //Keep the value of the smallest residual
252                     double Uresidual = stereoExtrapPosSensor.x() - hitPosSensor.x();
253                     if(Math.abs(Uresidual) < Math.abs(UresidualStereo)){
254                         UresidualStereo = Uresidual;
255                         if(hit.getRawHits().size() > 1) isSingleHitAxial = false;
256                     }
257                 }
258             }
259         
260             //Fill histograms for residuals
261             residualY.get(sensorAxialName).fill(UresidualAxial);
262             residualY.get(sensorStereoName).fill(UresidualStereo);
263             residualYvsV.get(sensorAxialName).fill(axialExtrapPosSensor.y(),UresidualAxial);
264             residualYvsV.get(sensorStereoName).fill(stereoExtrapPosSensor.y(),UresidualStereo);
265             residualYvsU.get(sensorAxialName).fill(axialExtrapPosSensor.x(),UresidualAxial);
266             residualYvsU.get(sensorStereoName).fill(stereoExtrapPosSensor.x(),UresidualStereo);
267             residualYvsP.get(sensorAxialName).fill(p.magnitude(),UresidualAxial);
268             residualYvsP.get(sensorStereoName).fill(p.magnitude(),UresidualAxial);
269             
270             
271             
272             //Fill single hit and multiple hit histograms
273             if(isSingleHitAxial){
274                 residualYSingleHit.get(sensorAxialName).fill(UresidualAxial);
275                 residualYSingleHitvsV.get(sensorAxialName).fill(axialExtrapPosSensor.y(),UresidualAxial);
276                 residualYSingleHitvsU.get(sensorAxialName).fill(axialExtrapPosSensor.x(),UresidualAxial);
277             }
278             else{
279                 residualYMultipleHit.get(sensorAxialName).fill(UresidualAxial);
280                 residualYMultipleHitvsV.get(sensorAxialName).fill(axialExtrapPosSensor.y(),UresidualAxial);
281                 residualYMultipleHitvsU.get(sensorAxialName).fill(axialExtrapPosSensor.x(),UresidualAxial);
282             }
283             
284             if(isSingleHitStereo){
285                 residualYSingleHit.get(sensorStereoName).fill(UresidualStereo);
286                 residualYSingleHitvsV.get(sensorStereoName).fill(stereoExtrapPosSensor.y(),UresidualStereo);
287                 residualYSingleHitvsU.get(sensorStereoName).fill(stereoExtrapPosSensor.x(),UresidualStereo);
288             }
289             else{
290                 residualYMultipleHit.get(sensorStereoName).fill(UresidualStereo);
291                 residualYMultipleHitvsV.get(sensorStereoName).fill(stereoExtrapPosSensor.y(),UresidualStereo);
292                 residualYMultipleHitvsU.get(sensorStereoName).fill(stereoExtrapPosSensor.x(),UresidualStereo);
293             }
294             
295             if(q > 0){
296                 residualYEle.get(sensorAxialName).fill(UresidualAxial);
297                 residualYEle.get(sensorStereoName).fill(UresidualStereo);
298                 residualYvsVEle.get(sensorAxialName).fill(axialExtrapPosSensor.y(),UresidualAxial);
299                 residualYvsVEle.get(sensorStereoName).fill(stereoExtrapPosSensor.y(),UresidualStereo);
300                 residualYvsUEle.get(sensorAxialName).fill(axialExtrapPosSensor.x(),UresidualAxial);
301                 residualYvsUEle.get(sensorStereoName).fill(stereoExtrapPosSensor.x(),UresidualStereo);
302                 residualYvsPEle.get(sensorAxialName).fill(p.magnitude(),UresidualAxial);
303                 residualYvsPEle.get(sensorStereoName).fill(p.magnitude(),UresidualAxial);
304             }
305             else{
306                 residualYPos.get(sensorAxialName).fill(UresidualAxial);
307                 residualYPos.get(sensorStereoName).fill(UresidualStereo);
308                 residualYvsVPos.get(sensorAxialName).fill(axialExtrapPosSensor.y(),UresidualAxial);
309                 residualYvsVPos.get(sensorStereoName).fill(stereoExtrapPosSensor.y(),UresidualStereo);
310                 residualYvsUPos.get(sensorAxialName).fill(axialExtrapPosSensor.x(),UresidualAxial);
311                 residualYvsUPos.get(sensorStereoName).fill(stereoExtrapPosSensor.x(),UresidualStereo);
312                 residualYvsPPos.get(sensorAxialName).fill(p.magnitude(),UresidualAxial);
313                 residualYvsPPos.get(sensorStereoName).fill(p.magnitude(),UresidualAxial);
314             }
315         }
316     }

319     private Hep3Vector globalToSensor(Hep3Vector trkpos, HpsSiSensor sensor){
320         SiSensorElectrodes electrodes = sensor.getReadoutElectrodes(ChargeCarrier.HOLE);
321         if(electrodes == null){
322             electrodes = sensor.getReadoutElectrodes(ChargeCarrier.ELECTRON);
323             System.out.println("Charge Carrier is NULL");
324         }
325         return electrodes.getGlobalToLocal().transformed(trkpos);
326     }

328     //Get the track state at the previous sensor
329     private TrackState getTrackState(Track track, int unusedLay){
330         int layer = -1;
331         boolean isTop = track.getTrackStates().get(0).getTanLambda() > 0;
332         //If unused layer is L1, then get trackstate at IP
333         if(unusedLay == 1){
334             return track.getTrackStates().get(0);
335         }
336         else{
337             layer = unusedLay - 1;
338         }
339         HpsSiSensor sensorHole = getSensor(track,layer,isTop,true);
340         HpsSiSensor sensorSlot = getSensor(track,layer,isTop,false);
341         TrackState tState = TrackStateUtils.getTrackStateAtSensor(track,sensorHole.getMillepedeId());
342         if(tState == null){
343             tState = TrackStateUtils.getTrackStateAtSensor(track,sensorSlot.getMillepedeId());
344         }
345         return tState;
346     }

348     //Returns channel number of a given position in the sensor frame
349     private int getChan(Hep3Vector pos, HpsSiSensor sensor){
350         double readoutPitch = sensor.getReadoutStripPitch();
351         int nChan = sensor.getNumberOfChannels();
352         double height = readoutPitch * nChan;
353         return (int) ((height/2-pos.x())/readoutPitch);
354     }

361     //Return the HpsSiSensor for a given top/bottom track, layer, axial/stereo, and slot/hole
362     private HpsSiSensor getSensor(Track track, int layer, boolean isAxial, boolean isHole) {
363         double tanLambda = track.getTrackStates().get(0).getTanLambda();
364         for(HpsSiSensor sensor: sensors){
365             int senselayer = (sensor.getLayerNumber() + 1)/2;
366             if(senselayer != layer) continue;
367             if((tanLambda > 0 && !sensor.isTopLayer()) || (tanLambda < 0 && sensor.isTopLayer())) continue;
368             if((isAxial && !sensor.isAxial()) || (!isAxial && sensor.isAxial())) continue;
369             if(layer < 4 && layer > 0){
370                 return sensor;
371             }
372             else{
373                 if((!sensor.getSide().matches("ELECTRON") && isHole) || (sensor.getSide().matches("ELECTRON") && !isHole)) continue;
374                 return sensor;
375             }
376         }
377         return null;
378     }

380     private int getUnusedSvtLayer(List<TrackerHit> stereoHits) {      
381         int[] svtLayer = new int[6];
382         
383         // Loop over all of the stereo hits associated with the track
384         for (TrackerHit stereoHit : stereoHits) {
385             
386             // Retrieve the sensor associated with one of the hits.  This will
387             // be used to retrieve the layer number
388             HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) stereoHit.getRawHits().get(0)).getDetectorElement();
389             
390             // Retrieve the layer number by using the sensor
391             int layer = (sensor.getLayerNumber() + 1)/2;
392            
393             // If a hit is associated with that layer, increment its 
394             // corresponding counter
395             svtLayer[layer - 1]++;
396         }
397         
398         // Loop through the layer counters and find which layer has not been
399         // incremented i.e. is unused by the track
400         for(int layer = 0; layer < svtLayer.length; layer++){
401             if(svtLayer[layer] == 0) { 
402                 return (layer + 1);
403             }
404         }
405         return -1;
406     }

408     //Checks to see if track is within acceptance of both axial and stereo sensors at a given layer
409     //Also returns channel number of the intersection
410     private Pair<HpsSiSensor,Pair<Integer,Hep3Vector>> isWithinSensorAcceptance(Track track, TrackState tState, int layer, boolean axial, Hep3Vector p,FieldMap fieldMap) {
411      
412         HpsSiSensor axialSensorHole = getSensor(track,layer,true,true);
413         HpsSiSensor axialSensorSlot = getSensor(track,layer,true,false);
414         HpsSiSensor stereoSensorHole = getSensor(track,layer,false,true);
415         HpsSiSensor stereoSensorSlot = getSensor(track,layer,false,false);
416         
417         HelicalTrackFit htf = TrackUtils.getHTF(tState);
418         
419         Hep3Vector axialTrackHolePos = TrackStateUtils.getLocationAtSensor(htf,axialSensorHole,bfield);
420         Hep3Vector axialTrackSlotPos = TrackStateUtils.getLocationAtSensor(htf,axialSensorSlot,bfield);
421         Hep3Vector stereoTrackHolePos = TrackStateUtils.getLocationAtSensor(htf,stereoSensorHole,bfield);
422         Hep3Vector stereoTrackSlotPos = TrackStateUtils.getLocationAtSensor(htf,stereoSensorSlot,bfield);
423 
424         Pair<Boolean,Pair<Integer,Hep3Vector>> axialHolePair = this.sensorContainsTrack(axialTrackHolePos, axialSensorHole);
425         Pair<Boolean,Pair<Integer,Hep3Vector>> axialSlotPair = this.sensorContainsTrack(axialTrackSlotPos, axialSensorSlot);
426         Pair<Boolean,Pair<Integer,Hep3Vector>> stereoHolePair = this.sensorContainsTrack(stereoTrackHolePos, stereoSensorHole);
427         Pair<Boolean,Pair<Integer,Hep3Vector>> stereoSlotPair = this.sensorContainsTrack(stereoTrackSlotPos, stereoSensorSlot);
428         
429         if(axialHolePair.getFirst() && axial){
430             return new Pair<> (axialSensorHole,axialHolePair.getSecond());
431         }
432     
433         if(axialSlotPair.getFirst() && axial){
434             return new Pair<> (axialSensorSlot,axialSlotPair.getSecond());
435         }
436        
437         if(stereoHolePair.getFirst() && !axial){
438             return new Pair<> (stereoSensorHole,stereoHolePair.getSecond());
439         }
440         
441         if(stereoSlotPair.getFirst() && !axial){
442             return new Pair<> (stereoSensorSlot,stereoSlotPair.getSecond());
443         }
444         
445         return null;
446     }

448     //Checks to see if track is in acceptance of sensor. Computes within sensor frame
449     //Also return channel number of the position
450     public Pair<Boolean,Pair<Integer,Hep3Vector>> sensorContainsTrack(Hep3Vector trackPosition, HpsSiSensor sensor){
451         Hep3Vector pos = globalToSensor(trackPosition, sensor);
452         int nChan = sensor.getNumberOfChannels();
453         int chan = getChan(pos,sensor);
454         double width = getSensorLength(sensor);
455         Pair<Integer,Hep3Vector> pair = new Pair<>(chan,pos);
456         if(chan < -this.chanExtd || chan > (nChan + this.chanExtd)){
457             return new Pair<>(false,pair);
458         }
459         if(Math.abs(pos.y())>width/2){
460             return new Pair<>(false,pair);
461         }
462         return new Pair<>(true,pair);
463     }

465     //Returns the horizontal length of the sensor
466     protected double getSensorLength(HpsSiSensor sensor) {
467 
468         double length = 0;
469 
470         // Get the faces normal to the sensor
471         final List<Polygon3D> faces = ((Box) sensor.getGeometry().getLogicalVolume().getSolid())
472                 .getFacesNormalTo(new BasicHep3Vector(0, 0, 1));
473         for (final Polygon3D face : faces) {
474 
475             // Loop through the edges of the sensor face and find the longest one
476             final List<LineSegment3D> edges = face.getEdges();
477             for (final LineSegment3D edge : edges) {
478                 if (edge.getLength() > length) {
479                     length = edge.getLength();
480                 }
481             }
482         }
483         return length;
484     }
