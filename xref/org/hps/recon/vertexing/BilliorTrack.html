65      public BilliorTrack(HelicalTrackFit helix) {
66          double[] helixparameters = helix.parameters();
67          _parameters = convertParsToBillior(helixparameters);
68          SymmetricMatrix helixcovmatrix = helix.covariance();
69          _covmatrix = convertCovarianceToBillior(helixcovmatrix, helixparameters);
70          _chisq = helix.chisq();
71          _nhchisq = 0.;
72          _ndf = helix.ndf();
73          _smap = helix.PathMap();
74          _msmap = helix.ScatterMap();
75      }

77      public BilliorTrack(Track track) {
78          TrackState ts = track.getTrackStates().get(0);
79          double[] helixparameters = ts.getParameters();
80          _parameters = convertParsToBillior(helixparameters);
81          SymmetricMatrix helixcovmatrix = new SymmetricMatrix(5, ts.getCovMatrix(), true);
82          _covmatrix = convertCovarianceToBillior(helixcovmatrix, helixparameters);
83          _chisq[0] = track.getChi2();
84          _nhchisq = 0.;
85          _ndf[0] = track.getNDF();
86  //        _smap = helix.PathMap();
87  //        _msmap = helix.ScatterMap();
88      }

90      public BilliorTrack(TrackState ts, double chisq, int ndf) {
91          double[] helixparameters = ts.getParameters();
92          _parameters = convertParsToBillior(helixparameters);
93          SymmetricMatrix helixcovmatrix = new SymmetricMatrix(5, ts.getCovMatrix(), true);
94          _covmatrix = convertCovarianceToBillior(helixcovmatrix, helixparameters);
95          _chisq[0] = chisq;
96          _nhchisq = 0.;
97          _ndf[0] = ndf;
98  //        _smap = helix.PathMap();
99  //        _msmap = helix.ScatterMap();
100     }

102     private double[] convertParsToBillior(double[] helixpars) {
103         double[] billior = {0, 0, 0, 0, 0};
104         billior[0] = -helixpars[0];
105         billior[1] = helixpars[3];
106         billior[2] = Math.PI / 2 - Math.atan(helixpars[4]);
107         billior[3] = helixpars[1];
108         billior[4] = helixpars[2];
109         return billior;
110 
111     }
112 
113     private Matrix convertCovarianceToBillior(SymmetricMatrix helixcov, double[] helixpars) {
114         BasicMatrix deriv = new BasicMatrix(5, 5);
115 
116         deriv.setElement(epsIndex, HelicalTrackFit.dcaIndex, -1);
117         deriv.setElement(z0Index, HelicalTrackFit.z0Index, 1);
118         deriv.setElement(thetaIndex, HelicalTrackFit.slopeIndex, -1 / (1 + helixpars[HelicalTrackFit.slopeIndex] * helixpars[HelicalTrackFit.slopeIndex]));
119         deriv.setElement(phi0Index, HelicalTrackFit.phi0Index, 1);
120         deriv.setElement(curvatureIndex, HelicalTrackFit.curvatureIndex, 1);
121 
122         Matrix derivT = MatrixTranspose(deriv);
123         Matrix newcov = MatrixOp.mult(deriv, MatrixOp.mult(helixcov, derivT));
124 
125         return newcov;
126     }

380     /**
381      * Returns the transpose of the matrix (inexplicably not handled by the
382      * matrix package for non-square matrices).
383      *
384      * @param m matrix to be transposed
385      * @return transposed matrix
386      */
387     private Matrix MatrixTranspose(Matrix m) {
388         MutableMatrix mt = new BasicMatrix(m.getNColumns(), m.getNRows());
389         for (int i = 0; i < m.getNRows(); i++) {
390             for (int j = 0; j < m.getNColumns(); j++) {
391                 mt.setElement(j, i, m.e(i, j));
392             }
393         }
394         return mt;
395     }
