3   import hep.physics.vec.BasicHep3Vector;
4   import hep.physics.vec.Hep3Vector;
5   
6   import org.hps.recon.tracking.CoordinateTransformations;
7   import org.lcsim.event.MCParticle;
8   
9   /**
10   * Class that computes the vertex of two MC particles using the line vertexer class.
11   */
12  public class TwoParticleVertexer extends TwoLineVertexer {
13      
14      public TwoParticleVertexer() {
15      }
16      public void setParticle(MCParticle track1,MCParticle track2) {
17          
18          //Calculate using detector coord.
19          Hep3Vector PA1 = track1.getOrigin();
20          Hep3Vector PB1 = track2.getOrigin();
21          Hep3Vector p1 = track1.getMomentum();
22          Hep3Vector p2 = track2.getMomentum();
23          
24          //propagate to different position
25          double dz = 20.0;
26          Hep3Vector PA2 = this.propAlongLine(PA1, p1, dz);
27          Hep3Vector PB2 = this.propAlongLine(PB1, p2, dz);
28          
29          if(_debug) {
30              System.out.printf("A1 %s p1 %s B1 %s p2 %s\n", PA1.toString(), p1.toString(), PB1.toString(), p2.toString());
31              System.out.printf("A2 %s B2 %s\n", PA2.toString(), PB2.toString());
32          }
33          
34          //set the member variables
35          A1 = CoordinateTransformations.transformVectorToTracking(PA1);
36          A2 = CoordinateTransformations.transformVectorToTracking(PA2);
37          B1 = CoordinateTransformations.transformVectorToTracking(PB1);
38          B2 = CoordinateTransformations.transformVectorToTracking(PB2);
39  
40      }
41      
42      Hep3Vector propAlongLine(Hep3Vector org, Hep3Vector p, double dz) {
43          double tanPxPz = p.x() / p.z();
44          double tanPyPz = p.y() / p.z();
45          double dx = dz * tanPxPz;
46          double dy = dz * tanPyPz;
47          return new BasicHep3Vector(org.x() + dx, org.y() + dy, org.z() + dz);
48      }
49      
50      
51  }
