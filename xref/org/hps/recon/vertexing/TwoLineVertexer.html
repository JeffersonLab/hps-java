11  /**
12   * Class to calculate the POCA between two straight lines
13   */
14  public class TwoLineVertexer extends BaseSimpleVertexer {
15      protected Hep3Vector A1,A2,B1,B2;
16      
17      public TwoLineVertexer() {
18          
19      }
20      
21      public void setLines(Hep3Vector PA1, Hep3Vector PA2, Hep3Vector PB1, Hep3Vector PB2) {
22          this.A1 = PA1;
23          this.A2 = PA2;
24          this.B1 = PB1;
25          this.B2 = PB2;
26      }
27    
28      public void clear() {
29          super.clear();
30          setLines(null,null,null,null);
31      }
32      
33      public boolean isValid() {
34          if(A1==null || A2==null || B1==null || B2==null) return false;
35          else return true;
36      }
37      
38      @Override
39      public void fitVertex() {
40          assert isValid();
41          Hep3Vector vtxPosition = getPOCALineToLine();
42          if (vtxPosition!=null) {
43              _fitted_vertex = new BaseVertex(true, "Two Line Vertexer", 0, 0, new SymmetricMatrix(0), vtxPosition, null); 
44          }
45      }

47      /**
48       * Function to calculate DCA in 3D of two lines
49       * @param A1 Starting 3D space point for one of the straight lines 
50       * @param A2 End 3D space point for one of the straight lines
51       * @param B1 Starting 3D space point for one of the straight lines
52       * @param B2 End 3D space point for one of the straight lines
53       * @return position of closest approach of the two lines defined by the input parameters
54       */
55      private Hep3Vector getPOCALineToLineAlt() {
56  
57          /*
58           * Line1 defined by A1 and A2
59           * Line2 defined by B1 and B2
60           *
61           * nA = dot(cross(B2-B1,A1-B1),cross(A2-A1,B2-B1));
62           * nB = dot(cross(A2-A1,A1-B1),cross(A2-A1,B2-B1));
63           * d = dot(cross(A2-A1,B2-B1),cross(A2-A1,B2-B1));
64           * A0 = A1 + (nA/d)*(A2-A1);
65           * B0 = B1 + (nB/d)*(B2-B1);
66           */
67          if(_debug) System.out.printf("%s: A1=%s A2=%s B1=%s B2=%s\n", this.getClass().getSimpleName(), A1.toString(), A2.toString(), B1.toString(), B2.toString());
68          double nA = VecOp.dot(VecOp.cross(VecOp.sub(B2, B1), VecOp.sub(A1, B1)), VecOp.cross(VecOp.sub(A2, A1), VecOp.sub(B2, B1)));
69          double nB = VecOp.dot(VecOp.cross(VecOp.sub(A2, A1), VecOp.sub(A1, B1)), VecOp.cross(VecOp.sub(A2, A1), VecOp.sub(B2, B1)));
70          double d = VecOp.dot(VecOp.cross(VecOp.sub(A2, A1), VecOp.sub(B2, B1)), VecOp.cross(VecOp.sub(A2, A1), VecOp.sub(B2, B1)));
71          Hep3Vector A0 = VecOp.add(A1, VecOp.mult(nA / d, VecOp.sub(A2, A1)));
72          Hep3Vector B0 = VecOp.add(B1, VecOp.mult(nB / d, VecOp.sub(B2, B1)));
73          Hep3Vector diff = VecOp.sub(B0, A0);
74          Hep3Vector tmp = VecOp.mult(0.5, diff);
75          Hep3Vector vertexPos = VecOp.add(A0, tmp);
76          if(_debug) System.out.printf("%s: A0=%s B0=%s ==> vtxPos=%s (tmp=%s)\n", this.getClass().getSimpleName(), A0.toString(), B0.toString(), vertexPos.toString(), tmp.toString());
77          return vertexPos;
78      }
79      
80      /**
81       * Function to calculate DCA in 3D of two lines
82       * @param A1 Starting 3D space point for one of the straight lines 
83       * @param A2 End 3D space point for one of the straight lines
84       * @param B1 Starting 3D space point for one of the straight lines
85       * @param B2 End 3D space point for one of the straight lines
86       * @return position of closest approach of the two lines defined by the input parameters
87       */
88      private Hep3Vector getPOCALineToLine() {
89  
90          if(_debug) System.out.printf("%s: A1=%s A2=%s B1=%s B2=%s\n", this.getClass().getSimpleName(), A1.toString(), A2.toString(), B1.toString(), B2.toString());
91  
92          double ya[][] = {VecOp.mult(-1,VecOp.sub(B1, A1)).v()};
93          BasicMatrix y = (BasicMatrix)MatrixOp.transposed(new BasicMatrix(ya));
94          Hep3Vector dB = VecOp.sub(B2, B1);
95          Hep3Vector dA = VecOp.sub(A2, A1);
96          BasicMatrix X = new BasicMatrix(3,2);
97          for(int col=0;col<2;++col) {
98              if(col==0) {
99                  X.setElement(0, col, dB.x());
100                 X.setElement(1, col, dB.y());
101                 X.setElement(2, col, dB.z());
102             } else {
103                 X.setElement(0, col, -1*dA.x());
104                 X.setElement(1, col, -1*dA.y());
105                 X.setElement(2, col, -1*dA.z());
106             }
107         }
108 
109         BasicMatrix X_T = (BasicMatrix)MatrixOp.transposed(X);
110         BasicMatrix XX_T = (BasicMatrix)MatrixOp.mult(X_T, X);
111         BasicMatrix IXX_T = null;
112         try {
113             IXX_T = (BasicMatrix)MatrixOp.inverse(XX_T);
114         } 
115         catch(MatrixOp.IndeterminateMatrixException e) {
116             System.out.printf("%s: caught indeterminate exception %s\n",this.getClass().getSimpleName(),e.getMessage());
117             return null;
118         }
119         BasicMatrix X_Ty = (BasicMatrix)MatrixOp.mult(X_T,y);
120         BasicMatrix b = (BasicMatrix)MatrixOp.mult(IXX_T, X_Ty);
121         double t = b.e(0, 0);
122         double s = b.e(1, 0);
123         Hep3Vector Bpca = VecOp.add(B1, VecOp.mult(t, dB));
124         Hep3Vector Apca = VecOp.add(A1, VecOp.mult(s, dA));
125         Hep3Vector vertex = VecOp.add(Apca, VecOp.mult(0.5, VecOp.sub(Bpca, Apca)));
126         if(_debug) {
127             System.out.printf("y:\n%s\n",y.toString());
128             System.out.printf("X:\n%s\n",X.toString());
129             System.out.printf("b:\n%s\n",b.toString());
130             Hep3Vector ymin = VecOp.add(VecOp.mult(t, dB) , VecOp.mult(s, dA) );
131             Hep3Vector yminprime = VecOp.add(VecOp.sub(B1, A1), ymin);
132             System.out.printf("ymin:\n%s\n",ymin.toString());
133             System.out.printf("yminprime:\n%s\n",yminprime.toString());
134             System.out.printf("Apca:\n%s\n",Apca.toString());
135             System.out.printf("Bpca:\n%s\n",Bpca.toString());
136             System.out.printf("vertex:\n%s\n",vertex.toString());
137         }
138         return vertex;
139
