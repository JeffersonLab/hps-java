59      @Override
60      protected void detectorChanged(Detector detector) {
61          
62          if (aida == null)
63              aida = AIDA.defaultInstance();
64          
65          aida.tree().cd("/");
66          
67          // Set the magnetic field parameters to the appropriate values.
68          Hep3Vector ip = new BasicHep3Vector(0., 0., 500.0);
69          bField = detector.getFieldMap().getField(ip).y();
70          
71          List<String> volumes = new ArrayList<String>();
72          volumes.add("_top");
73          volumes.add("_bottom");
74          volumes.add("");
75          
76          for (String vol : volumes) {
77              aida.histogram1D(vtxFold+"vtx_x"+vol,200,-3,3);
78              aida.histogram1D(vtxFold+"vtx_y"+vol,200,-1,1);
79              aida.histogram1D(vtxFold+"vtx_z"+vol,400,-20,20);
80              aida.histogram2D(vtxFold+"vtx_x_y"+vol,200,-3,3,200,-1,1);
81              aida.histogram1D(vtxFold+"vtx_chi2"+vol,200,0,8000);
82              aida.histogram1D(vtxFold+"vtx_ntrks"+vol,200,0,200);
83          }
84      }

90      @Override
91      protected void process(EventHeader event) {
92          List<Track> trackCollection = null;
93          
94          if (event.hasCollection(Track.class, trackCollectionName)) {
95              trackCollection = event.get(Track.class,trackCollectionName);
96          }
97          else {
98              return;
99          }
100         
101         for (Track track : trackCollection) {
102 
103             if (_nhits < 0 || track.getTrackerHits().size() == _nhits) {
104                 
105                 //accumulatedTracks.add(track);
106                 accumulatedBTracks.add(new BilliorTrack(track));
107                 
108                 //Avoid tracks too close to the beamline to have a more reliable result on z from single volumes
109                 if (track.getTrackStates().get(0).getTanLambda() > 0.035) {
110                     //accumulatedTracksTop.add(track);
111                     accumulatedBTracksTop.add(new BilliorTrack(track));
112                 }
113                 
114                 if (track.getTrackStates().get(0).getTanLambda() < -0.035) {
115                     //accumulatedTracksBot.add(track);
116                     accumulatedBTracksBot.add(new BilliorTrack(track));
117                 }
118                 
119             }
120         }
121     }

124     @Override
125     protected void endOfData() {
126        
127         System.out.println("Size of accumulated Billior tracks: " + accumulatedBTracks.size());
128         System.out.println("Fitting Vertex");
129         
130         beamPositionToUse[0] = 0.;
131         beamPositionToUse[0] = 0.;
132         beamPositionToUse[0] = -10;
133    
134         BilliorVertexer vtxFitter = new BilliorVertexer(bField);
135         vtxFitter.setBeamSize(beamSize);
136         vtxFitter.setBeamPosition(beamPositionToUse);
137         vtxFitter.setStoreCovTrkMomList(_storeCovTrkMomList);
138         vtxFitter.setDebug(debug);
139         vtxFitter.doBeamSpotConstraint(false);
140         
141         //Randomize the tracks ? Better to do something more reproducible
142         
143         Collections.shuffle(accumulatedBTracks);
144         FitMultiVtx(accumulatedBTracks,vtxFitter,""); 
145         
146         FitMultiVtx(accumulatedBTracksTop,vtxFitter,"_top");
147         FitMultiVtx(accumulatedBTracksBot,vtxFitter,"_bottom");
148         
149         
150 
151 
152     }

154     private void FitMultiVtx(List<BilliorTrack> accTrks, BilliorVertexer vtxFitter, String vol) {
155         
156         int n_chunks = accTrks.size() / _ntrks;
157         int n_rest   = accTrks.size() % _ntrks;
158         System.out.println("n_chunks = " + n_chunks);
159         System.out.println("n_rest = "   + n_rest);
160         System.out.printf("size  = %d \n", (n_chunks*_ntrks + n_rest));
161 
162         if (accTrks.size() < 2)
163             return;
164         
165         for (int i_chunk = 0; i_chunk < n_chunks; i_chunk++) {
166             List<BilliorTrack> tracksForFit = accTrks.subList(i_chunk*_ntrks, (i_chunk+1)*_ntrks);
167             BilliorVertex vtx = vtxFitter.fitVertex(tracksForFit);
168             aida.histogram1D(vtxFold+"vtx_x"+vol).fill(vtx.getPosition().x());
169             aida.histogram1D(vtxFold+"vtx_y"+vol).fill(vtx.getPosition().y());
170             aida.histogram1D(vtxFold+"vtx_z"+vol).fill(vtx.getPosition().z());
171             aida.histogram2D(vtxFold+"vtx_x_y"+vol).fill(vtx.getPosition().x(),vtx.getPosition().y());
172             aida.histogram1D(vtxFold+"vtx_chi2"+vol).fill(vtx.getChi2());
173             aida.histogram1D(vtxFold+"vtx_ntrks"+vol).fill(_ntrks);
174             //System.out.printf("vtx  %.5f %.5f %.5f\n",vtx.getPosition().x(), vtx.getPosition().y(),vtx.getPosition().z());
175         }
176         
177         if (n_rest > 1) {
178             
179             List<BilliorTrack> tracksForFit = accTrks.subList(n_chunks*_ntrks, n_chunks*_ntrks + n_rest);
180             BilliorVertex vtx = vtxFitter.fitVertex(tracksForFit);
181             aida.histogram1D(vtxFold+"vtx_x"+vol).fill(vtx.getPosition().x());
182             aida.histogram1D(vtxFold+"vtx_y"+vol).fill(vtx.getPosition().y());
183             aida.histogram1D(vtxFold+"vtx_z"+vol).fill(vtx.getPosition().z());
184             aida.histogram2D(vtxFold+"vtx_x_y"+vol).fill(vtx.getPosition().x(),vtx.getPosition().y());
185             aida.histogram1D(vtxFold+"vtx_ntrks"+vol).fill(n_rest);
186             aida.histogram1D(vtxFold+"vtx_chi2"+vol).fill(vtx.getChi2());
187         }
188      
189         return ;
190     }
