2   package org.hps.recon.vertexing;
3   
4   import hep.physics.matrix.SymmetricMatrix;
5   import hep.physics.vec.Hep3Vector;
6   
7   import java.util.Map;
8   
9   import org.hps.recon.tracking.StraightLineTrack;
10  
11  /**
12   *  Encapsulate the results of a vertex fit
13   */
14  public class VertexFit {
15  
16      private Hep3Vector _vtx;
17      private SymmetricMatrix _cov;
18      private double _chisq;
19      private Map<StraightLineTrack, Hep3Vector> _dirmap;
20  
21      /**
22       * Fully qualified constructor for a VertexFit
23       *
24       * @param vtx vertex position
25       * @param cov covariance matrix of vertex
26       * @param chisq chisq of vertex fit
27       * @param dirmap map relating the input StraightLineTracks to the fitted track directions
28       */
29      public VertexFit(Hep3Vector vtx, SymmetricMatrix cov, double chisq,
30              Map<StraightLineTrack, Hep3Vector> dirmap) {
31  
32          _vtx = vtx;
33          _cov = cov;
34          _chisq = chisq;
35          _dirmap = dirmap;
36      }
37  
38      /**
39       * Return the vertex position.
40       *
41       * @return vertex position
42       */
43      public Hep3Vector vtx() {
44          return _vtx;
45      }
46  
47      /**
48       * Return the covariance matrix from the vertex fit.
49       *
50       * @return covariance matrix
51       */
52      public SymmetricMatrix covariance() {
53          return _cov;
54      }
55  
56      /**
57       * Return the chisq of the vertex fit.
58       *
59       * @return chisq
60       */
61      public double chisq() {
62          return _chisq;
63      }
64  
65      /**
66       * Return the fitted direction for a given StraightLineTrack that was input to the fit
67       *
68       * @param slt input StraightLineTrack
69       * @return fitted direction
70       */
71      public Hep3Vector TrackDirection(StraightLineTrack slt) {
72          return _dirmap.get(slt);
73      }
74  }
