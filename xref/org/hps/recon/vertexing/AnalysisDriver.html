26  public class AnalysisDriver extends Driver {
27  
28      private AIDA aida = AIDA.defaultInstance();
29      int nevt = 0;
30      private List<SeedStrategy> _slist = new ArrayList<SeedStrategy>();
31  
32      public AnalysisDriver() {
33      }
34  
35      public void process(EventHeader event) {
36  
37          //  Increment the event counter
38          nevt++;
39  
40          //  Set the magnetic field
41          double bfield = 1.0;
42          double xref = 50.;
43  
44          Random rn = new Random();
45  
46          for (int iloop = 0; iloop < 10000; iloop++) {
47  
48              double p0 = 0.5 + rn.nextDouble();
49              double py0 = 0.1 * p0 * rn.nextGaussian();
50              double pz0 = 0.1 * p0 * rn.nextGaussian();
51              double px0 = Math.sqrt(p0 * p0 - py0 * py0 - pz0 * pz0);
52              double len = 20. * rn.nextDouble();
53              double vx = len * px0 / p0;
54              double vy = len * py0 / p0;
55              double vz = len * pz0 / p0;
56              Hep3Vector vertex = new BasicHep3Vector(vx, vy, vz);
57  
58              double py1 = 0.5 * py0 + 0.05 * p0 * (rn.nextDouble() - 0.5);
59              double pz1 = 0.5 * pz0 + 0.05 * p0 * (rn.nextDouble() - 0.5);
60              double px1 = 0.5 * px0 + 0.05 * p0 * (rn.nextDouble() - 0.5);
61              Hep3Vector p1 = new BasicHep3Vector(px1, py1, pz1);
62              double m = 0.000511;
63              double e1 = Math.sqrt(p1.magnitudeSquared() + m * m);
64              HepLorentzVector p4v1 = new BasicHepLorentzVector(e1, p1);
65              double ox1 = xref;
66              double oy1 = vy + py1 * (ox1 - vx) / px1;
67              double oz1 = vz + pz1 * (ox1 - vx) / px1;
68              Hep3Vector origin1 = new BasicHep3Vector(ox1, oy1, oz1);
69  
70              double py2 = py0 - py1;
71              double pz2 = pz0 - pz1;
72              double px2 = px0 - px1;
73              Hep3Vector p2 = new BasicHep3Vector(px2, py2, pz2);
74              double e2 = Math.sqrt(p2.magnitudeSquared() + m * m);
75              HepLorentzVector p4v2 = new BasicHepLorentzVector(e2, p2);
76              double ox2 = xref;
77              double oy2 = vy + py2 * (ox2 - vx) / px2;
78              double oz2 = vz + pz2 * (ox2 - vx) / px2;
79              Hep3Vector origin2 = new BasicHep3Vector(ox2, oy2, oz2);
80  
81              ParticlePropertyProvider ptp = ParticlePropertyManager.getParticlePropertyProvider();
82              BaseMCParticle mcp1 = new BaseMCParticle(origin1, p4v1, ptp.get(11), BaseMCParticle.FINAL_STATE, 0.);
83              BaseMCParticle mcp2 = new BaseMCParticle(origin2, p4v2, ptp.get(-11), BaseMCParticle.FINAL_STATE, 0.);
84  
85              HelixParamCalculator hp1 = new HelixParamCalculator(mcp1, bfield);
86              HelixParamCalculator hp2 = new HelixParamCalculator(mcp2, bfield);
87  
88              double[] pars1 = new double[5];
89              double[] pars2 = new double[5];
90              double omega1 = hp1.getMCOmega();
91              double omega2 = hp2.getMCOmega();
92              double domega = 1.e-2 * (omega1 + omega2) / 2.;
93              double dDCA = 1.e-2;
94              double dphi0 = 1.e-4;
95              double dz0 = 1.e-2;
96              double dslope = 1.e-4;
97              SymmetricMatrix cov = new SymmetricMatrix(5);
98              cov.setElement(HelicalTrackFit.curvatureIndex, HelicalTrackFit.curvatureIndex, domega * domega);
99              cov.setElement(HelicalTrackFit.dcaIndex, HelicalTrackFit.dcaIndex, dDCA * dDCA);
100             cov.setElement(HelicalTrackFit.phi0Index, HelicalTrackFit.phi0Index, dphi0 * dphi0);
101             cov.setElement(HelicalTrackFit.z0Index, HelicalTrackFit.z0Index, dz0 * dz0);
102             cov.setElement(HelicalTrackFit.slopeIndex, HelicalTrackFit.slopeIndex, dslope * dslope);
103             pars1[HelicalTrackFit.curvatureIndex] = omega1 + domega * rn.nextGaussian();
104             pars1[HelicalTrackFit.dcaIndex] = hp1.getDCA() + dDCA * rn.nextGaussian();
105             pars1[HelicalTrackFit.phi0Index] = hp1.getPhi0() + dphi0 * rn.nextGaussian();
106             pars1[HelicalTrackFit.z0Index] = hp1.getZ0() + dz0 * rn.nextGaussian();
107             pars1[HelicalTrackFit.slopeIndex] = hp1.getSlopeSZPlane() + dslope * rn.nextGaussian();
108             pars2[HelicalTrackFit.curvatureIndex] = omega2 + domega * rn.nextGaussian();
109             pars2[HelicalTrackFit.dcaIndex] = hp2.getDCA() + dDCA * rn.nextGaussian();
110             pars2[HelicalTrackFit.phi0Index] = hp2.getPhi0() + dphi0 * rn.nextGaussian();
111             pars2[HelicalTrackFit.z0Index] = hp2.getZ0() + dz0 * rn.nextGaussian();
112             pars2[HelicalTrackFit.slopeIndex] = hp2.getSlopeSZPlane() + dslope * rn.nextGaussian();
113 
114             double[] chisq = new double[2];
115             int[] ndof = new int[2];
116 
117             HelicalTrackFit h1 = new HelicalTrackFit(pars1, cov, chisq, ndof, null, null);
118             HelicalTrackFit h2 = new HelicalTrackFit(pars2, cov, chisq, ndof, null, null);
119 
120             //  Create a list of straight line tracks
121             List<StraightLineTrack> sltlist = new ArrayList<StraightLineTrack>();
122 
123             HelixConverter converter = new HelixConverter(xref);
124             sltlist.add(converter.Convert(h1));
125             sltlist.add(converter.Convert(h2));
126 
127             VertexFitter vtxfit = new VertexFitter(xref);
128             boolean success = vtxfit.VertexFit(sltlist);
129             if (success) {
130                 VertexFit vfit = vtxfit.getFit();
131                 aida.histogram1D("Chi square of vertex fit", 100, 0., 10.).fill(vfit.chisq());
132                 Hep3Vector vtx = vfit.vtx();
133                 SymmetricMatrix vtxcov = vfit.covariance();
134                 double dvx = Math.sqrt(vtxcov.diagonal(0));
135                 double dvy = Math.sqrt(vtxcov.diagonal(1));
136                 double dvz = Math.sqrt(vtxcov.diagonal(2));
137                 aida.histogram1D("Vertex position x", 100, -5., 25.).fill(vtx.x());
138                 aida.histogram1D("Vertex position y", 100, -5., 5.).fill(vtx.y());
139                 aida.histogram1D("Vertex position z", 100, -5., 5.).fill(vtx.z());
140                 aida.histogram1D("Vertex fit x residual", 100, -3., 3.).fill(vtx.x() - vertex.x());
141                 aida.histogram1D("Vertex fit y residual", 100, -0.5, 0.5).fill(vtx.y() - vertex.y());
142                 aida.histogram1D("Vertex fit z residual", 100, -0.5, 0.5).fill(vtx.z() - vertex.z());
143                 aida.histogram1D("Vertex x pull", 100, -4., 4.).fill((vtx.x() - vertex.x()) / dvx);
144                 aida.histogram1D("Vertex y pull", 100, -4., 4.).fill((vtx.y() - vertex.y()) / dvy);
145                 aida.histogram1D("Vertex z pull", 100, -4., 4.).fill((vtx.z() - vertex.z()) / dvz);
146                 aida.histogram1D("Vertex position x uncertainty", 100, 0., 1.0).fill(dvx);
147                 aida.histogram1D("Vertex position y uncertainty", 100, 0., 0.1).fill(dvy);
148                 aida.histogram1D("Vertex position z uncertainty", 100, 0., 0.1).fill(dvz);
149                 aida.histogram1D("Confidence Level", 50, 0., 1.).fill(ChisqProb.gammp(1, vfit.chisq()));
150             }
151         }
152         return;
153     }
