72      public BilliorVertex fitVertex(List<BilliorTrack> tracks) {
73          _ntracks = tracks.size();
74          follow1985Paper(tracks);
75          if (_beamspotConstraint)
76              applyBSconstraint(true);
77          else if (_targetConstraint)
78              applyBSconstraint(false);
79          Map<Integer, Hep3Vector> pFitMap = new HashMap<Integer, Hep3Vector>();
80          for (int i = 0; i < tracks.size(); i++) {
81              Hep3Vector pFit = CoordinateTransformations.transformVectorToDetector(new BasicHep3Vector(this.getFittedMomentum(i)));
82              pFitMap.put(i, pFit);
83          }
84  //        Hep3Vector vert = new BasicHep3Vector(_vertexPosition.e(0, 0), _vertexPosition.e(1, 0), _vertexPosition.e(2, 0));
85  // correct for the beamspot position here instead of HPSReconParticle...
86  //        Hep3Vector vert = new BasicHep3Vector(_vertexPosition.e(0, 0)+_beamPosition[0], _vertexPosition.e(1, 0)+_beamPosition[1], _vertexPosition.e(2, 0)+_beamPosition[2]);
87          Hep3Vector vert = new BasicHep3Vector(_vertexPosition.e(0, 0) + _referencePosition[0], _vertexPosition.e(1, 0) + _referencePosition[1], _vertexPosition.e(2, 0) + _referencePosition[2]);
88          Hep3Vector vertDet = CoordinateTransformations.transformVectorToDetector(vert);
89          SymmetricMatrix covVtxDet = CoordinateTransformations.transformCovarianceToDetector(new SymmetricMatrix(_covVtx));
90          BilliorVertex vertex = new BilliorVertex(vertDet, covVtxDet, _chiSq, getInvMass(), pFitMap, _constraintType);
91          vertex.setPositionError(CoordinateTransformations.transformVectorToDetector(this.getVertexPositionErrors()));
92          vertex.setMassError(this.getInvMassUncertainty());
93          List<Matrix> pcov = new ArrayList<Matrix>();
94          List<Matrix> tcov = new ArrayList<Matrix>();
95          List<double[]> tpars = new ArrayList<double[]>();
96          pcov.add(CoordinateTransformations.transformCovarianceToDetector(new SymmetricMatrix(this.getFittedMomentumCovariance(0))));
97          pcov.add(CoordinateTransformations.transformCovarianceToDetector(new SymmetricMatrix(this.getFittedMomentumCovariance(1))));
98          pcov.add(CoordinateTransformations.transformCovarianceToDetector(new SymmetricMatrix(this.getFittedTrk1Trk2MomCovariance(0, 1))));
99          vertex.setTrackMomentumCovariances(pcov);
100         vertex.setStoreCovTrkMomList(storeCovTrkMomList);
101         vertex.setV0Momentum(CoordinateTransformations.transformVectorToDetector(getV0Momentum()), CoordinateTransformations.transformVectorToDetector(getV0MomentumError()));
102         vertex.setV0TargetXY(getV0Projection(), getV0ProjectionError());
103         tpars.add(getFittedTrackParameters(0));
104         tpars.add(getFittedTrackParameters(1));
105         tcov.add(getFittedTrackCovariance(0));
106         tcov.add(getFittedTrackCovariance(1));
107         vertex.setFittedTrackParameters(tpars);
108         vertex.setFittedTrackCovariance(tcov);
109         debugMomentumUncertainty(1);
110         return vertex;

114      *  this method is based on progressive least squares fit
115      *  using the unconstrained fit result as the (k-1) fit

117      *  all notation is taken from:
118      * W. Hulsbergen, NIM 552 (2005) 566-575

120     private void applyBSconstraint(boolean pointback) {
121         String methodName = pointback ? "constrainV0toBS" : "constrainV0toTarget";
122         BasicMatrix Ckm1 = new BasicMatrix(3 * (_ntracks + 1), 3 * (_ntracks + 1));
123         BasicMatrix Xkm1 = new BasicMatrix(3 * (_ntracks + 1), 1);
124         MatrixOp.setSubMatrix(Ckm1, _covVtx, 0, 0);
125         MatrixOp.setSubMatrix(Xkm1, _vertexPosition, 0, 0);
126         int n = 1;
127         for (Matrix covVtxMom : covVtxMomList) {
128             if (_debug)
129                 System.out.println(methodName + "::Track " + n + "  covVtxMom : " + covVtxMom.toString());
130             MatrixOp.setSubMatrix(Ckm1, covVtxMom, 0, 3 * n);
131             MatrixOp.setSubMatrix(Ckm1, MatrixOp.transposed(covVtxMom), 3 * n, 0);
132             n++;
133         }
134         for (int i = 0; i < _ntracks; i++) {
135             BasicMatrix pi = (BasicMatrix) _pFit.get(i);
136             MatrixOp.setSubMatrix(Xkm1, pi, 3 * (i + 1), 0);
137 //            if (_debug)
138 //                System.out.println(methodName + "::Track " + i + "  p : " + pi.toString());
139             for (int j = 0; j < _ntracks; j++)
140                 MatrixOp.setSubMatrix(Ckm1, covMomList[i][j], 3 * (i + 1), 3 * (j + 1));
141         }
142 
143         //  now calculate the derivative matrix for the beam constraint.
144         //  the beamspot is assumed to be at _beamPosition 
145         //  the V0 production position is Vbvec=(0,-(ptot_y)/(ptot_x)*Vx+Vy, -(ptot_z)/(ptot_x)*Vx+Vz)
146         //  where ptot=sum_i (pi)
147         //  need derivites wrt to the vertex position and momentum (theta,phi_v,rho)
148         double Vx = _vertexPosition.e(0, 0);
149         double Vy = _vertexPosition.e(1, 0);
150         double Vz = _vertexPosition.e(2, 0);
151         if (_debug) {
152             //mg 2/28/19 ... these two should be close for the refit (z reference position (track frame) = 0 is correct)
153             System.out.println(methodName + "::unconstrained vertexPosition = (" + Vx + "," + Vy + "," + Vz + ")");
154             System.out.println(methodName + "::referencePosition = (" + _referencePosition[0] + "," + _referencePosition[1] + "," + _referencePosition[2] + ")");
155 
156         }
157 //add in the reference position about which vertex position is calculated...mg 2/28/19, this was commented out and I think it should be so...good....actually, maybe should be in???  
158 //But I put it in the matrix calculations, so ok!?
159 //        double Vx = _vertexPosition.e(0, 0) + _referencePosition[0];
160 //        double Vy = _vertexPosition.e(1, 0) + _referencePosition[1];;
161 //        double Vz = _vertexPosition.e(2, 0) + _referencePosition[2];;
162         //first, get the sum of momenta...
163         double pxtot = 0;
164         double pytot = 0;
165         double pztot = 0;
166 //use getFittedMometum here???  Better, getV0Momentum
167         for (int i = 0; i < _ntracks; i++) {
168             BasicMatrix pi = (BasicMatrix) _pFit.get(i);
169             double theta = pi.e(0, 0);
170             double phiv = pi.e(1, 0);
171             double rho = pi.e(2, 0);
172             double Pt = Math.abs((1. / rho) * _bField * Constants.fieldConversion);
173             double px = Pt * Math.cos(phiv);
174             double py = Pt * Math.sin(phiv);
175             double pz = Pt * 1 / Math.tan(theta);
176             pxtot += px;
177             pytot += py;
178             pztot += pz;
179         }
180         //calculate the position of the A' at X=Target
181         BasicMatrix rk = makeRk(Vx, Vy, Vz, pxtot, pytot, pztot, pointback);
182         if (_debug)
183             System.out.println(methodName + "::rk = " + rk);
184 
185         //mg...makeHkFixed is for our debugging..replace makeHk with that when we 
186         // are sure that it's correct...mg 2/28/19...maybe do that soon? 
187         BasicMatrix Hk = makeHkFixed(Vx, pxtot, pytot, pztot, pointback);
188         //        BasicMatrix Hk = makeHk(Vx, pxtot, pytot, pztot, pointback);
189 
190         if (_debug)
191             System.out.println(methodName + "::Hk = " + Hk);
192 
193         // the beam covariance...no crossterms but easy to include
194         BasicMatrix Vk = new BasicMatrix(3, 3);
195         Vk.setElement(0, 0, _beamSize[0] * _beamSize[0]);
196         Vk.setElement(1, 1, _beamSize[1] * _beamSize[1]);
197         Vk.setElement(2, 2, _beamSize[2] * _beamSize[2]);
198 
199         //now do the matrix operations to get the constrained parameters
200         BasicMatrix Hkt = (BasicMatrix) MatrixOp.transposed(Hk);
201         if (_debug)
202             System.out.println(methodName + "::Ckm1Hk = " + MatrixOp.mult(Ckm1, Hk));
203 
204         BasicMatrix Rk = (BasicMatrix) MatrixOp.mult(Hkt, MatrixOp.mult(Ckm1, Hk));
205         if (_debug)
206             System.out.println("Pre Vk:  Rk = " + Rk.toString());
207         Rk = (BasicMatrix) MatrixOp.add(Rk, Vk);
208         if (_debug)
209             System.out.println("Post Vk:  Rk = " + Rk.toString());
210         BasicMatrix Rkinv = (BasicMatrix) MatrixOp.inverse(Rk);
211         BasicMatrix Kk = (BasicMatrix) MatrixOp.mult(Ckm1, MatrixOp.mult(Hk, Rkinv));
212 
213         Matrix _constrainedFit = MatrixOp.mult(Kk, rk);
214         _constrainedFit = MatrixOp.add(_constrainedFit, Xkm1);//Xk
215 
216         //ok, get the new covariance
217         BasicMatrix RkKkt = (BasicMatrix) MatrixOp.mult(Rk, MatrixOp.transposed(Kk));
218         BasicMatrix HkCkm1 = (BasicMatrix) MatrixOp.mult(Hkt, Ckm1);
219         RkKkt = (BasicMatrix) MatrixOp.mult(1, RkKkt);
220         HkCkm1 = (BasicMatrix) MatrixOp.mult(-2, HkCkm1);
221         BasicMatrix sumMatrix = (BasicMatrix) MatrixOp.mult(Kk, MatrixOp.add(HkCkm1, RkKkt));
222         Matrix _constrainedCov = (BasicMatrix) MatrixOp.add(Ckm1, sumMatrix);
223 
224         //update the regular parameter names to the constrained result
225         _vertexPosition = (BasicMatrix) MatrixOp.getSubMatrix(_constrainedFit, 0, 0, 3, 1);
226 
227         if (_debug)
228             System.out.println(methodName + "  Constrained vertex: " + _vertexPosition);
229 
230         //update the covariance matrices and fitted momenta
231         _covVtx = (BasicMatrix) MatrixOp.getSubMatrix(_constrainedCov, 0, 0, 3, 3);
232         for (int i = 0; i < _ntracks; i++) {
233             BasicMatrix ptmp = (BasicMatrix) MatrixOp.getSubMatrix(_constrainedFit, 3 * (i + 1), 0, 3, 1);
234             _pFit.set(i, ptmp);
235             covVtxMomList.set(i, (BasicMatrix) MatrixOp.getSubMatrix(_constrainedCov, 0, 3 * (i + 1), 3, 3));
236             for (int j = 0; j < _ntracks; j++)
237                 covMomList[i][j] = (BasicMatrix) MatrixOp.getSubMatrix(_constrainedCov, 3 * (i + 1), 3 * (j + 1), 3, 3);;
238         }
239 
240         if (_debug) {
241             System.out.println(_constraintType + "  Chisq contribution: " + MatrixOp.mult(MatrixOp.transposed(rk), MatrixOp.mult(Rkinv, rk)));
242             if (MatrixOp.mult(MatrixOp.transposed(rk), MatrixOp.mult(Rkinv, rk)).e(0, 0) > 1000 && pointback)
243                 System.out.println(" BIG CHISQ CONTRIBUTION!!!!!!");
244         }
245         _chiSq += MatrixOp.mult(MatrixOp.transposed(rk), MatrixOp.mult(Rkinv, rk)).e(0, 0);
246 
247     }

605     public double getInvMass() {
606         double esum = 0.;
607         double pxsum = 0.;
608         double pysum = 0.;
609         double pzsum = 0.;
610         double me = 0.000511;
611         // Loop over tracks
612         for (int i = 0; i < _ntracks; i++) {
613             double[] p = getFittedMomentum(i);
614             double p1x = p[0];
615             double p1y = p[1];
616             double p1z = p[2];
617             double p1mag2 = p1x * p1x + p1y * p1y + p1z * p1z;
618             double e1 = Math.sqrt(p1mag2 + me * me);
619             pxsum += p1x;
620             pysum += p1y;
621             pzsum += p1z;
622             esum += e1;
623         }
624         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
625         double evtmass = esum * esum - psum * psum;
626 
627         if (evtmass > 0)
628             return Math.sqrt(evtmass);
629         else
630             return -99;
631     }

826     /*
827     *  The method here follows the 1985 paper from 
828     *  Billoir, P., Fruhwirth, R., & Regler, M. (1985). 
829     *  "Track element merging strategy and vertex fitting 
830     *  in complex modular detectors."
831     *  Nucl. Instrum. Methods Phys. Res., A, 241115131. 42 p.
832     *  http://cds.cern.ch/record/1330744
833      */
834     private void follow1985Paper(List<BilliorTrack> tracks) {
835 
836         //initial guess for the vertex
837         BasicMatrix v0 = new BasicMatrix(3, 1);
838         v0.setElement(0, 0, _v0[0]);
839         v0.setElement(1, 0, _v0[1]);
840         v0.setElement(2, 0, _v0[2]);
841         //  make some arrays we'll need
842         List<Matrix> Gs = new ArrayList<Matrix>();
843         List<Matrix> Ds = new ArrayList<Matrix>();
844         List<Matrix> Es = new ArrayList<Matrix>();
845         List<Matrix> As = new ArrayList<Matrix>();
846         List<Matrix> Bs = new ArrayList<Matrix>();
847         List<Matrix> pis = new ArrayList<Matrix>();
848         List<Matrix> cis = new ArrayList<Matrix>();
849 
850         BasicMatrix D0 = new BasicMatrix(3, 3);
851         boolean firstTrack = true;
852 
853         for (BilliorTrack bt : tracks) {
854             double[] par = bt.parameters();
855 //          measured track parameters, but them in a matrix
856             BasicMatrix tmpPar = new BasicMatrix(5, 1);
857             tmpPar.setElement(0, 0, par[0]);
858             tmpPar.setElement(1, 0, par[1]);
859             tmpPar.setElement(2, 0, par[2]);
860             tmpPar.setElement(3, 0, par[3]);
861             tmpPar.setElement(4, 0, par[4]);
862 //          the measured quantities in terms of the  vertex, v0. 
863 //          parameterization taken from: 
864 //          Billoir & Qian, "Fast vertex fitting witha  local parameterization of tracks", 
865 //          NIM A311 (1992) 139-150. 
866 //          BE CAREFUL, this paper has a sign error in the dz/dtheta derivative 
867 //          corrected in erratum NIM A350 (1994) 624.  
868             double theta = par[2];
869             double rho = par[4];
870             double cotth = 1. / tan(par[2]);
871             double uu = v0.e(0, 0) * cos(par[3]) + v0.e(1, 0) * sin(par[3]);//Q
872             double vv = v0.e(1, 0) * cos(par[3]) - v0.e(0, 0) * sin(par[3]);//R
873             double eps = -vv - .5 * uu * uu * par[4];
874             double zp = v0.e(2, 0) - uu * (1 - vv * par[4]) * cotth;
875             // * phi at vertex with these parameters
876 //            double phiVert = par[3] + uu * par[4]; // MG--10/30/18:  is this sign right?  isn't is phi-Qrho?  
877             double phiVert = par[3] - uu * par[4];  // MG--10/30/18:  I think the (-) sign is right, but doesn't make a difference since initial vertex guess is always (0,0,0)
878 //          measured track parameters moved to v0           
879             BasicMatrix p0 = new BasicMatrix(5, 1);
880             p0.setElement(0, 0, eps);
881             p0.setElement(1, 0, zp);
882             p0.setElement(2, 0, theta);
883             p0.setElement(3, 0, phiVert);
884             p0.setElement(4, 0, rho);
885 
886 //          zeroth approximation of track direction & curvature at the vertex
887             BasicMatrix q0 = new BasicMatrix(3, 1);
888             q0.setElement(0, 0, theta);
889             q0.setElement(1, 0, phiVert);
890             q0.setElement(2, 0, rho);
891 
892             double cosf = cos(phiVert);
893             double sinf = sin(phiVert);
894             BasicMatrix tmpA = new BasicMatrix(5, 3);
895             tmpA.setElement(0, 0, sinf);
896             tmpA.setElement(0, 1, -cosf);
897             tmpA.setElement(1, 0, -cotth * cosf);
898             tmpA.setElement(1, 1, -cotth * sinf);
899             tmpA.setElement(1, 2, 1);
900             tmpA.setElement(3, 0, -par[4] * cosf);
901             tmpA.setElement(3, 1, -par[4] * sinf);
902 
903             BasicMatrix tmpB = new BasicMatrix(5, 3);
904             tmpB.setElement(0, 1, uu);    // deps/dphiv
905             tmpB.setElement(0, 2, -uu * uu / 2);  //deps/drho
906             tmpB.setElement(1, 0, uu * (1 + cotth * cotth));  //dzp/dtheta --  the sign is wrong in 1992 paper; fixed in erratum
907             tmpB.setElement(1, 1, -vv * cotth);//  dzp/dphiv
908             tmpB.setElement(1, 2, uu * vv * cotth);  //dzp/drho
909             tmpB.setElement(3, 1, 1); //dphip/dphiv
910             tmpB.setElement(3, 2, -uu); //dphip/drho
911             tmpB.setElement(2, 0, 1);  //partial(theta)/dtheta
912             tmpB.setElement(4, 2, 1); //partial (rho)/drho
913             As.add(tmpA);
914             Bs.add(tmpB);
915 
916             BasicMatrix ci = (BasicMatrix) MatrixOp.add(p0, MatrixOp.mult(-1, MatrixOp.mult(tmpA, v0)));
917             ci = (BasicMatrix) MatrixOp.add(ci, MatrixOp.mult(-1, MatrixOp.mult(tmpB, q0)));
918             cis.add(ci);
919             pis.add(MatrixOp.add(tmpPar, MatrixOp.mult(-1, ci)));
920 
921             BasicMatrix tmpG = (BasicMatrix) MatrixOp.inverse(bt.covariance());
922             Gs.add(tmpG); // Gs are the weight matrices == inverse of covariance
923 
924             // 
925             if (firstTrack)
926                 D0 = (BasicMatrix) MatrixOp.mult(MatrixOp.transposed(tmpA), MatrixOp.mult(tmpG, tmpA));
927             else
928                 D0 = (BasicMatrix) MatrixOp.add(D0, MatrixOp.mult(MatrixOp.transposed(tmpA), MatrixOp.mult(tmpG, tmpA)));
929 
930             BasicMatrix tmpDi = (BasicMatrix) MatrixOp.mult(MatrixOp.transposed(tmpA), MatrixOp.mult(tmpG, tmpB));
931             BasicMatrix tmpEi = (BasicMatrix) MatrixOp.mult(MatrixOp.transposed(tmpB), MatrixOp.mult(tmpG, tmpB));
932             Ds.add(tmpDi);
933             Es.add(tmpEi);
934             firstTrack = false;
935         }
936 
937         //ok, now compute the vertex fit.
938         BasicMatrix tmpCovVtx = D0;
939         BasicMatrix bigsum = new BasicMatrix(3, 1);
940         for (int i = 0; i < _ntracks; i++) {
941             BasicMatrix a = (BasicMatrix) As.get(i);
942             BasicMatrix b = (BasicMatrix) Bs.get(i);
943             BasicMatrix d = (BasicMatrix) Ds.get(i);
944             BasicMatrix e = (BasicMatrix) Es.get(i);
945             BasicMatrix g = (BasicMatrix) Gs.get(i);
946             BasicMatrix p = (BasicMatrix) pis.get(i);
947             BasicMatrix sub = (BasicMatrix) MatrixOp.mult(d, MatrixOp.mult(MatrixOp.inverse(e), MatrixOp.transposed(d)));
948             tmpCovVtx = (BasicMatrix) MatrixOp.add(tmpCovVtx, MatrixOp.mult(-1, sub));// calculate C00=D0 - (Di)(Ei-1)(DiT)
949 
950             BasicMatrix aTg = (BasicMatrix) MatrixOp.mult(MatrixOp.transposed(a), g);
951             BasicMatrix beIbTg = (BasicMatrix) MatrixOp.mult(b, MatrixOp.mult(MatrixOp.inverse(e), MatrixOp.mult(MatrixOp.transposed(b), g)));
952             BasicMatrix MinusaTgbeIbTg = (BasicMatrix) MatrixOp.mult(-1, MatrixOp.mult(aTg, beIbTg));
953 
954             if (firstTrack)
955                 bigsum = (BasicMatrix) MatrixOp.mult(MatrixOp.add(aTg, MinusaTgbeIbTg), p);
956             else
957                 bigsum = (BasicMatrix) MatrixOp.add(bigsum, MatrixOp.mult(MatrixOp.add(aTg, MinusaTgbeIbTg), p));
958         }
959         BasicMatrix covVtx = (BasicMatrix) MatrixOp.inverse(tmpCovVtx);
960         BasicMatrix xtilde = (BasicMatrix) MatrixOp.mult(covVtx, bigsum);
961         if (_debug)
962             System.out.println(_constraintType + "  follow1985Paper::Vertex at : \nx= " + xtilde.e(0, 0) + " +/- " + Math.sqrt(covVtx.e(0, 0)) + "\ny= " + xtilde.e(1, 0) + " +/- " + Math.sqrt(covVtx.e(1, 1)) + "\nz= " + xtilde.e(2, 0) + " +/- " + Math.sqrt(covVtx.e(2, 2)));
963 
964         //ok, now the momentum
965         //        List<Matrix> qtildes = new ArrayList<Matrix>();
966         //        List<Matrix> ptildes = new ArrayList<Matrix>();
967         List<Matrix> C0j = new ArrayList<Matrix>();
968         List<Matrix> pfit = new ArrayList<Matrix>();
969         //Matrix[][] Cij = new Matrix[2][2];//max 2 tracks...just make this bigger for more
970         Matrix[][] Cij = new Matrix[_ntracks][_ntracks];//max 2 tracks...just make this bigger for more
971         double chisq = 0;
972         for (int j = 0; j < _ntracks; j++) {
973             BasicMatrix a = (BasicMatrix) As.get(j);
974             BasicMatrix b = (BasicMatrix) Bs.get(j);
975             BasicMatrix d = (BasicMatrix) Ds.get(j);
976             BasicMatrix e = (BasicMatrix) Es.get(j);
977             BasicMatrix g = (BasicMatrix) Gs.get(j);
978             BasicMatrix p = (BasicMatrix) pis.get(j);
979             BasicMatrix c = (BasicMatrix) cis.get(j);
980             //calculate qtilde & ptilde (vertex fitted track parameters) :  equations 22b and 22d from 1985 NIM
981             BasicMatrix first = (BasicMatrix) MatrixOp.mult(-1, MatrixOp.mult(MatrixOp.inverse(e), MatrixOp.transposed(d)));
982             first = (BasicMatrix) MatrixOp.mult(first, xtilde);
983             BasicMatrix second = (BasicMatrix) MatrixOp.mult(MatrixOp.inverse(e), MatrixOp.mult(MatrixOp.transposed(b), g));
984             second = (BasicMatrix) MatrixOp.mult(second, p);
985             BasicMatrix qtilde = (BasicMatrix) MatrixOp.add(first, second);
986             BasicMatrix ptilde = (BasicMatrix) MatrixOp.add(MatrixOp.mult(a, xtilde), MatrixOp.mult(b, qtilde));
987             //caluclate unconstrained chi^2 vertex fit. 
988             chisq += MatrixOp.mult(MatrixOp.transposed(MatrixOp.add(p, MatrixOp.mult(-1, ptilde))), MatrixOp.mult(g, MatrixOp.add(p, MatrixOp.mult(-1, ptilde)))).e(0, 0);
989             if (_debug)
990                 System.out.println("\n\n" + _constraintType + "  follow1985Paper::Track #" + j);
991             if (_debug)
992                 System.out.println("eps(meas)    = " + p.e(0, 0) + "        eps(fit)   =" + ptilde.e(0, 0));
993             if (_debug)
994                 System.out.println("zp(meas)     = " + p.e(1, 0) + "        zp(fit)    =" + ptilde.e(1, 0));
995             if (_debug)
996                 System.out.println("theta(meas)  = " + p.e(2, 0) + "        theta(fit) =" + ptilde.e(2, 0));
997             if (_debug)
998                 System.out.println("phi(meas)    = " + p.e(3, 0) + "        phi(fit)   =" + ptilde.e(3, 0));
999             if (_debug)
1000                 System.out.println("rho(meas)    = " + p.e(4, 0) + "        rho(fit)   =" + ptilde.e(4, 0));
1001 
1002             BasicMatrix tmpC0j = (BasicMatrix) MatrixOp.mult(-1, MatrixOp.mult(covVtx, MatrixOp.mult(d, MatrixOp.inverse(e))));
1003             C0j.add(tmpC0j);
1004             for (int i = 0; i < _ntracks; i++) {
1005 // MG 10/30/18:    previously, only the two lines below (commented out) were used...this doesn't seem right...seems like it's missing the delta_ij x E^
1006 //                BasicMatrix tmpCij = (BasicMatrix) MatrixOp.mult(-1, MatrixOp.mult(MatrixOp.inverse(e), MatrixOp.mult(MatrixOp.transposed(d), tmpC0j)));
1007 //               Cij[i][j] = tmpCij;
1008                 BasicMatrix di = (BasicMatrix) Ds.get(i);
1009                 BasicMatrix ei = (BasicMatrix) Es.get(i);
1010                 BasicMatrix tmpCij = (BasicMatrix) MatrixOp.mult(-1, MatrixOp.mult(MatrixOp.inverse(ei), MatrixOp.mult(MatrixOp.transposed(di), tmpC0j)));
1011                 if (i == j)
1012                     tmpCij = (BasicMatrix) MatrixOp.add(MatrixOp.inverse(ei), tmpCij);
1013                 Cij[i][j] = tmpCij;
1014             }
1015             BasicMatrix tmppfit = new BasicMatrix(3, 1);
1016             tmppfit.setElement(0, 0, qtilde.e(0, 0) + c.e(2, 0));
1017             tmppfit.setElement(1, 0, qtilde.e(1, 0) + c.e(3, 0));
1018             tmppfit.setElement(2, 0, qtilde.e(2, 0) + c.e(4, 0));

1020         }
1021

1023             System.out.println(_constraintType + "  follow1985Paper::chi^2 = " + chisq);
1024

1026         _covVtx = covVtx;

1030         covVtxMomList = C0j;
