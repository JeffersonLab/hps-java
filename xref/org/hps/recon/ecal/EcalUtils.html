59      public static int getHVGroup(int x, int y) {
60          int absy = Math.abs(y);
61          if (x > 0 || x <= -8) {
62              return (23 - Math.abs(x)) / 2 + 1;
63          } else {
64              if (x == -7 && absy == 5) {
65                  return 8;
66              } else if (x >= -4) {
67                  return 12 - Math.max(x + 4, absy - 2);
68              } else {
69                  return 12 - Math.max(-5 - x, absy - 2);
70              }
71          }
72      }

74      /**
75       * This is a very basic method that, given an array with the raw-waveform (in FADC units), returns the amplitude (in mV)
76       * @param data Array with data from FADC, in fadc units
77       * @param lenght The array lenght
78       * @param pedestalSamples How many samples at the beginning of the array to use for the pedestal. Must be < lenght
79       * @return double[], 0 is the amplitude in mV, 1 is the offest in ADC counts, 2 is the RMS in adc counts
80       */
81      public static double[] computeAmplitude(short [] data, int lenght, int pedestalSamples){
82          double amplitude,pedestal,noise;
83          pedestal=0;
84          noise=0;
85          amplitude=data[0];
86          double[] ret={0.,0.,0.};
87          if (pedestalSamples>lenght){
88              return ret;
89          }
90          for (int jj = 0; jj < lenght; jj++){
91              if (jj<pedestalSamples){
92                  pedestal+=data[jj];
93                  noise+=data[jj]*data[jj];
94              }
95              if (data[jj]>amplitude) amplitude=data[jj];
96          }
97          pedestal/=pedestalSamples;
98          noise/=pedestalSamples;
99          noise=Math.sqrt(noise-pedestal*pedestal);
100         amplitude-=pedestal;
101 
102         amplitude*=adcResolution*1000;
103         ret[0]=amplitude;
104         ret[1]=pedestal;
105         ret[2]=noise;
106         return ret;
107 
108     }
