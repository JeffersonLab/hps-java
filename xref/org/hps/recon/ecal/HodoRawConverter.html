32      public ArrayList<Integer> FindThresholdCrossings(RawTrackerHit hit, double ped) {
33  
34          // Getting the cellID of the hit
35          final long cellID = hit.getCellID();
36  
37          // ADC values for this hit
38          final short samples[] = hit.getADCValues();
39  
40          // Number of samples
41          int n_samp = samples.length;
42  
43          // ==== This list will be populated with threshold crossings
44          ArrayList<Integer> thr_crossings = new ArrayList<Integer>();
45  
46          double Threshold = ped + tet;
47  
48          // special case, first sample is above threshold:
49          if (samples[0] > Threshold) {
50              thr_crossings.add(0);
51          }
52  
53          //double ped = getPedestal();
54          // ===== Loop over samples to determine Threshold Crossings =====
55          for (int is = 1; is < n_samp; is++) {
56  
57              if (samples[is] > Threshold && samples[is - 1] <= Threshold) {
58  
59                  // found one:
60                  thr_crossings.add(is);
61  
62                  // search for next threshold crossing begins at end of this
63                  // pulse:
64                  is = is + HodoConstants.nGoForNextPeak - 1;
65  
66                  // Don't find more than NMax_peak peaks
67                  if (thr_crossings.size() >= HodoConstants.NMax_peak) {
68                      break;
69                  }
70  
71              }
72  
73          }
74  
75          return thr_crossings;
76      }

78      public ArrayList<CalorimeterHit> getCaloHits(RawTrackerHit hit, ArrayList<Integer> thr_crosings, double ped) {
79  
80          // Getting the cellID of the hit
81          long cellID = hit.getCellID();
82  
83          // ADC values for this hit
84          final short samples[] = hit.getADCValues();
85  
86          int n_samp = samples.length;
87  
88          double gain = 0;
89          if (useUserGain == true) {
90              gain = this.userGains;
91          } else {
92              gain = findChannel(cellID).getGain().getGain();
93          }
94          //System.out.println("The Gains = " + findChannel(cellID).getGain().toString());
95          
96          if(isMC) {
97              int[] identifier = getHodoIdentifiers(cellID);
98              GeometryId id_geometry = new GeometryId(helper, new int[]{subDetector.getSystemID(), identifier[0], identifier[1], identifier[2], identifier[3]});
99              cellID = id_geometry.encode(); 
100         } 
101 
102         ArrayList<CalorimeterHit> curHits = new ArrayList<CalorimeterHit>();
103 
104         // ==== Will loop over threshold crossings, and for each of thr_crossing will calculate
105         // ==== the integrated signal
106         for (int crs_time : thr_crosings) {
107 
108             int samp_Left = java.lang.Math.max(0, crs_time - HodoConstants.NSB);
109             int samp_Right = java.lang.Math.min(n_samp - 1, crs_time + HodoConstants.NSA - 1);
110 
111             double ADC_Sum = 0;
112             for (int is = samp_Left; is <= samp_Right; is++) {
113                 ADC_Sum = ADC_Sum + samples[is];
114             }
115 
116             // ==== Now subtract pedestal from the ADC Sum ====
117             ADC_Sum = ADC_Sum - ped * (samp_Right - samp_Left + 1);
118 
119             double Energy = ADC_Sum * gain;
120             double time = crs_time * HodoConstants.NSPerSample;
121 
122             //System.out.println("time = " + time + "     gain = " + gain + "        Energy = " + Energy + "ADC Sum is " + ADC_Sum);                       
123             curHits.add(CalorimeterHitUtilities.create(Energy, time, cellID));
124         }
125 
126         return curHits;
127     }

129     public double getPedestal(EventHeader event, long cellid) {
130 
131         if (useRunningPedestal && event != null) {
132 
133             Map<HodoscopeChannel, Double> runningPedMap = (Map<HodoscopeChannel, Double>) event.get("HodoRunningPedestals");
134             
135             HodoscopeChannel chan;
136             if(!isMC)
137                 chan = hodoConditions.getChannels().findGeometric(cellid);
138             else
139                 chan = hodoConditions.getChannels().findChannel((int)cellid);
140             
141             return runningPedMap.get(chan);
142         } else {
143             return findChannel(cellid).getCalibration().getPedestal();
144         }
145     }

153     public HodoscopeChannelConstants findChannel(long cellID) {

160     // =========== Computed Hodoscop identifiers from cellID
161     public int[] getHodoIdentifiers(long cellID) {
162 
163         HodoscopeChannel chan;
164         if(!isMC) 
165             chan = hodoConditions.getChannels().findGeometric(cellID);
166         else
167             chan = hodoConditions.getChannels().findChannel((int)cellID);
168 
169         int[] hodo_ids = new int[4];
170         hodo_ids[0] = chan.getIX();
171         hodo_ids[1] = chan.getIY();
172         hodo_ids[2] = chan.getLayer();
173         hodo_ids[3] = chan.getHole();
174 
175         return hodo_ids;
176     }
