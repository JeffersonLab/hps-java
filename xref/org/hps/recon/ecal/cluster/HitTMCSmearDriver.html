1   package org.hps.recon.ecal.cluster;
2   
3   import java.util.List;
4   import java.util.Random;
5   
6   import org.hps.util.TruthCalorimeterHit;
7   import org.lcsim.event.CalorimeterHit;
8   import org.lcsim.event.EventHeader;
9   import org.lcsim.util.Driver;
10  import org.lcsim.event.base.BaseCalorimeterHit;
11  
12  /**
13   * Driver <code>HitTMCSmearDriver</code> is  responsible for smearing
14   * hit times to better match the time resolution as studied in the
15   * 2016 data set.
16   * 
17   */
18  public class HitTMCSmearDriver extends Driver {
19      /**
20       * This is the default class when used in readout. Recon MC
21       * requires EcalCalHits. 
22       */
23      private String inputHitCollection = "EcalCalHits";
24      //not EcalRawHits or GTPHits
25      //EcalCorrectedHits keeps increasing in time
26      
27      /**
28       * Sets the name of the hit collection that is to undergo time
29       * correction.
30       * @param inputHitCollection - The collection name.
31       */
32      public void setInputHitCollection(String inputHitCollection) {
33          this.inputHitCollection = inputHitCollection;
34      }
35      
36      /**
37       * Calculates the time smear for a given hit energy and time.
38       * @param energy - The energy of the hit.
39       * @param time - The time of the hit.
40       * @return Returns a time-corrected value for the hit.
41       */
42      private static double calcSmear(double energy, double time){
43          // Time resolution as derived for 2016 data
44          //Factor 1.98 derived after running over MC. Probably due to
45          //lack of trigger jitter in simulation.
46          Random r = new Random();
47          double sigT = r.nextGaussian() * Math.sqrt(Math.pow(0.188 / energy, 2) + Math.pow(0.152, 2)) / 1.98;
48          return time + sigT;
49      }
50      
51      /**
52       * Calculates the time correction for event hits and modifies the
53       * hits to use the corrected time value.
54       * @param event - The event containing the hits that need to be
55       * modified.
56       */
57      @Override
58      public void process(EventHeader event) {
59          if(event.hasCollection(CalorimeterHit.class, inputHitCollection)){
60              // Get the hits in the event.
61              List<CalorimeterHit> hits = event.get(CalorimeterHit.class, inputHitCollection);
62              
63              // Smear the times for each hit.
64              for(CalorimeterHit iHit : hits) {
65                  // Calculate the smeared time.
66                  double oldT = iHit.getTime();
67                  double energy = iHit.getRawEnergy();
68                  double newT = calcSmear(energy, oldT);
69                  
70                  // If truth information exists, then hits will of the
71                  // type TruthCalorimeterHit. Otherwise, they will be
72                  // of type BaseCalorimeterHit. Ascertain the hit type
73                  // and then modify the hit time.
74                  if(iHit instanceof TruthCalorimeterHit) {
75                      ((TruthCalorimeterHit) iHit).setTime(newT);
76                  } else {
77                      ((BaseCalorimeterHit) iHit).setTime(newT);
78                  }
79              }
80          } 
81      }
82  }
