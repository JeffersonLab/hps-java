21  /**
22   * <p>
23   * This class creates clusters from a CalorimeterHit input collection.
24   * <p>
25   * This clustering logic is based on that from the <a
26   * href="https://misportal.jlab.org/ul/Physics/Hall-B/clas/viewFile.cfm/2005-001.pdf?documentId=6"
27   * >CLAS-Note-2005-001</a>.
28   * <p>
29   * The analysis and position corrections are described in <a
30   * href="https://misportal.jlab.org/mis/physics/hps_notes/index.cfm?note_year=2014">HPS Note
31   * 2014-001</a>.
32   * <p>
33   * The algorithm sorts hits from highest to lowest energy and builds clusters around each local
34   * maximum/seed hit. Common hits are distributed between clusters when minimum between two clusters.
35   * There is a threshold cut for minimum hit energy, minimum cluster energy, and minimum seed hit
36   * energy. There is also a timing threshold with respect to the seed hit. All of these parameters
37   * are tunable and should be refined with more analysis. Energy corrections are applied separately.
38   * 
39   * @see AbstractClusterer
40   * @see Clusterer
41   */
42  public class ReconClusterer extends AbstractClusterer {
43  
44      // Minimum energy threshold for hits; lower energy hits will be
45      // excluded from clustering. Units in GeV.
46      double hitEnergyThreshold = 0.0075;
47  
48      // Minimum energy threshold for seed hits; if seed hit is below
49      // cluster is excluded from output. Units in GeV.
50      double seedEnergyThreshold = 0.05;
51  
52      // Minimum energy threshold for cluster hits; if total cluster
53      // energy is below, the cluster is excluded. Units in GeV.
54      double clusterEnergyThreshold = 0.1;
55  
56      // Apply time cut to hits
57      boolean useTimeCut = true;
58  
59      // Minimum time cut window range. Units in ns.
60      double minTime = 0.0;
61  
62      // Minimum time cut window range. Units in ns.  Negative means ignore this requirement.
63      double maxTime = -1;
64  
65      // Maximum time cut window range. Units in ns.
66      double timeWindow = 8.0;
67  
68      // Make a map for quick calculation of the x-y position of crystal face
69      Map<Point, double[]> correctedPositionMap = new LinkedHashMap<Point, double[]>();
70  
71      List<CalorimeterHit> rejectedHitList = new ArrayList<CalorimeterHit>();
72  
73      ReconClusterer() {
74          super(new String[] { "hitEnergyThreshold", "seedEnergyThreshold", "clusterEnergyThreshold", "minTime","maxTime","timeWindow" },
75                  new double[] { 0.0075, 0.05, 0.1, 0.0, -1.0, 8.0 });
76      }
77  
78      void setUseTimeCut(boolean useTimeCut) {
79          this.useTimeCut = useTimeCut;
80      }

82      public void initialize() {
83          hitEnergyThreshold = getCuts().getValue("hitEnergyThreshold");
84          seedEnergyThreshold = getCuts().getValue("seedEnergyThreshold");
85          if (hitEnergyThreshold > seedEnergyThreshold) {
86              throw new IllegalArgumentException("hitEnergyThreshold must be <= to seedEnergyThreshold");
87          }
88          clusterEnergyThreshold = getCuts().getValue("clusterEnergyThreshold");
89          minTime = getCuts().getValue("minTime");
90          maxTime = getCuts().getValue("maxTime");
91          timeWindow = getCuts().getValue("timeWindow");
92      }

102     public List<Cluster> createClusters(EventHeader event, List<CalorimeterHit> hitList) {
103 
104         // Clear the position map.
105         correctedPositionMap.clear();
106 
107         // Clear the rejected hit list.
108         rejectedHitList = new ArrayList<CalorimeterHit>();
109 
110         // if(!hitList.isEmpty()) { System.out.println("TrueType: " + hitList.get(0).getClass().getSimpleName() + "; Sim Hit: " + (hitList.get(0) instanceof org.lcsim.event.SimCalorimeterHit ? "SimCalorimeterHit" : "CalorimeterHit")); }
111 
112         // Create a list for the created clusters.
113         ArrayList<Cluster> clusterList = new ArrayList<Cluster>();
114 
115         // Sort the input hit list by energy.
116         ClusterUtilities.sortHitsUniqueEnergy(hitList);
117 
118       /*  System.out.println("BEFORE");
119         for (CalorimeterHit hit : hitList) {
120             System.out.println(hit.getCorrectedEnergy()+" "+hit.getTime());
121         }*/
122         
123         // Filter the hit list of any hits that fail to pass the
124         // designated threshold.
125         for (int index = hitList.size() - 1; index >= 0; index--) {
126             // If the hit is below threshold or below min time, kill it.
127             if ((hitList.get(index).getCorrectedEnergy() < hitEnergyThreshold) || (hitList.get(index).getTime() < minTime)
128                     || (maxTime > 0 && hitList.get(index).getTime() > maxTime)) {
129                 rejectedHitList.add(hitList.get(index));
130                 hitList.remove(index);
131             }
132 
133             // Since the hits are sorted by energy from highest to
134             // lowest, any hit that is above threshold means that all
135             // subsequent hits will also be above threshold. Continue through
136             // list to check in time window.
137             else {
138                 continue;
139             }
140         }
141 
142       /*  System.out.println("AFTER");
143         for (CalorimeterHit hit : hitList) {
144             System.out.println(hit.getCorrectedEnergy()+" "+hit.getTime());
145         }*/
146         
147         // Create a map to connect the cell ID of a calorimeter crystal to the hit which occurred in
148         // that crystal.
149 
150         //        Map<Long, CalorimeterHit> hitMap = ClusterUtilities.createHitMap(hitList);
151 
152         LinkedHashMap<Long, CalorimeterHit> hitMap = new LinkedHashMap<Long, CalorimeterHit>();
153 
154         //boolean multihit = false;
155         //for (int ii = hitList.size() - 1; ii >= 0; ii--) {
156         for (int ii = 0; ii <= hitList.size() - 1; ii++) {
157             CalorimeterHit hit = hitList.get(ii);
158             if (hitMap.containsKey(hit.getCellID())) {
159                 // throw new RuntimeException("Multiple CalorimeterHits found in same crystal.");
160                 //multihit = true;
161                 rejectedHitList.add(hit);
162                 hitList.remove(ii);
163                 ii--;
164             } else {
165                 hitMap.put(hit.getCellID(), hit);
166             }
167         }
168 
169         // Multiple hits in same channel occurs in the data, so no print out is needed here.  --JM
170         //if (multihit == true) {
171         //    System.err.println("Multiple CalorimeterHits found in same crystal!");            
172         //}
173 
174         // Create a map to connect a seed hit to its cluster.
175         Map<CalorimeterHit, BaseCluster> seedToCluster = new LinkedHashMap<CalorimeterHit, BaseCluster>();
176 
177         // Map a crystal to a list of all clusters in which it is a member.
178         Map<CalorimeterHit, List<CalorimeterHit>> commonHits = new LinkedHashMap<CalorimeterHit, List<CalorimeterHit>>();
179 
180         // Map a crystal to the seed of the cluster of which it is a member.
181         LinkedHashMap<CalorimeterHit, CalorimeterHit> hitToSeed = new LinkedHashMap<CalorimeterHit, CalorimeterHit>();
182 
183         // Loop through all calorimeter hits to locate seeds and perform
184         // first pass calculations for component and common hits.
185         for (int ii = 0; ii <= hitList.size() - 1; ii++) {
186             CalorimeterHit hit = hitList.get(ii);
187 
188             // Get the set of all neighboring crystals to the current hit.
189             Set<Long> neighbors = neighborMap.get(hit.getCellID());
190 
191             // Generate a list to store any neighboring hits in.
192             ArrayList<CalorimeterHit> neighborHits = new ArrayList<CalorimeterHit>();
193 
194             // Sort through the set of neighbors and, if a hit exists
195             // which corresponds to a neighbor, add it to the list of
196             // neighboring hits.
197             for (Long neighbor : neighbors) {
198                 // Get the neighboring hit.
199                 CalorimeterHit neighborHit = hitMap.get(neighbor);
200 
201                 // If it exists, add it to the list.
202                 if (neighborHit != null && hitList.contains(neighborHit)) {
203                     neighborHits.add(neighborHit);
204                 }
205             }
206 
207             // Track whether the current hit is a seed hit or not.
208             boolean isSeed = true;
209 
210             // Loops through all the neighboring hits to determine if
211             // the current hit is the local maximum within its set of
212             // neighboring hits.
213             seedHitLoop: for (CalorimeterHit neighbor : neighborHits) {
214                 if (!equalEnergies(hit, neighbor)) {
215                     isSeed = false;
216                     break seedHitLoop;
217                 }
218             }
219 
220             // Hit is a local maximum
221             if (isSeed) {
222                 // Seed must pass minimum threshold
223                 if (hit.getCorrectedEnergy() >= seedEnergyThreshold) {
224                     // Create new cluster
225                     BaseCluster cluster = createBasicCluster();
226                     clusterList.add(cluster);
227                     seedToCluster.put(hit, cluster);
228                     hitToSeed.put(hit, hit);
229 
230                 }
231                 // Seed does not pass minimum threshold
232                 else {
233                     rejectedHitList.add(hit);
234                     hitList.remove(ii);
235                     ii--;
236                 }
237             } // end if isSeed
238 
239             // If this hit is not a seed hit, see if it should be
240             // attached to any neighboring seed hits.
241             else {
242                 // Sort through the list of neighboring hits.
243                 for (CalorimeterHit neighborHit : neighborHits) {
244                     // Check whether the neighboring hit is a seed.
245                     // if (seedToCluster.containsKey(neighborHit)) {
246                     if (hitToSeed.get(neighborHit) == neighborHit) {
247 
248                         // If the neighboring hit is a seed hit and the
249                         // current hit has been associated with a cluster,
250                         // then it is a common hit between its previous
251                         // seed and the neighboring seed.
252                         if (hitToSeed.containsKey(hit)) {
253                             // Check and see if a list of common seeds
254                             // for this hit already exists or not.
255                             List<CalorimeterHit> commonHitList = commonHits.get(hit);
256 
257                             // If it does not, make a new one.
258                             if (commonHitList == null) {
259                                 commonHitList = new ArrayList<CalorimeterHit>();
260                             }
261 
262                             // Add the neighbors to the seeds to set of
263                             // common seeds.
264                             commonHitList.add(neighborHit);
265                             commonHitList.add(hitToSeed.get(hit));
266 
267                             // Put the common seed list back into the set.
268                             commonHits.put(hit, commonHitList);
269                         }
270 
271                         // If the neighboring hit is a seed hit and the
272                         // current hit has not been added to a cluster yet
273                         // associate it with the neighboring seed and note
274                         // that it has been clustered.
275                         else {
276                             hitToSeed.put(hit, neighborHit);
277                         }
278                     }
279                 }
280             }
281         } // End primary seed loop.
282 
283         // Performs second pass calculations for component hits.
284         secondaryHitsLoop: for (CalorimeterHit secondaryHit : hitList) {
285             // Look for hits that already have an associated seed/clustering.
286             if (!hitToSeed.containsKey(secondaryHit)) {
287                 continue secondaryHitsLoop;
288             }
289 
290             // Get the secondary hit's neighboring crystals.
291             Set<Long> secondaryNeighbors = neighborMap.get(secondaryHit.getCellID());
292 
293             // Make a list to store the hits associated with the
294             // neighboring crystals.
295             List<CalorimeterHit> secondaryNeighborHits = new ArrayList<CalorimeterHit>();
296 
297             // Loop through the neighboring crystals.
298             for (Long secondaryNeighbor : secondaryNeighbors) {
299                 // Get the hit associated with the neighboring crystal.
300                 CalorimeterHit secondaryNeighborHit = hitMap.get(secondaryNeighbor);
301 
302                 // If the neighboring crystal exists and is not already
303                 // in a cluster, add it to the list of neighboring hits.
304                 if (secondaryNeighborHit != null && !hitToSeed.containsKey(secondaryNeighborHit) && hitList.contains(secondaryNeighborHit)) {
305                     secondaryNeighborHits.add(secondaryNeighborHit);
306                 }
307             }
308 
309             // Loop over the secondary neighbor hits.
310             for (CalorimeterHit secondaryNeighborHit : secondaryNeighborHits) {
311                 // If the neighboring hit is of lower energy than the
312                 // current secondary hit, then associate the neighboring
313                 // hit with the current secondary hit's seed.
314                 if (!equalEnergies(secondaryNeighborHit, secondaryHit)) {
315                     hitToSeed.put(secondaryNeighborHit, hitToSeed.get(secondaryHit));
316                 } else {
317                     continue;
318                 }
319             }
320         } // End component hits loop.
321 
322         // Performs second pass calculations for common hits.
323         // TODO make this loop keySet-order independent
324         commonHitsLoop: for (CalorimeterHit clusteredHit : hitToSeed.keySet()) {
325 
326             // Get the current clustered hit's neighboring crystals.
327             Set<Long> clusteredNeighbors = neighborMap.get(clusteredHit.getCellID());
328 
329             // Store a list of all the clustered hits neighboring
330             // crystals which themselves contain hits.
331             List<CalorimeterHit> clusteredNeighborHits = new ArrayList<CalorimeterHit>();
332 
333             // Loop through the neighbors and see if they have hits.
334             for (Long neighbor : clusteredNeighbors) {
335                 // Get the hit associated with the neighbor.
336                 CalorimeterHit clusteredNeighborHit = hitMap.get(neighbor);
337 
338                 // If it exists, add it to the neighboring hit list.
339 
340                 if (clusteredNeighborHit != null && hitToSeed.get(clusteredNeighborHit) != null && hitList.contains(clusteredNeighborHit)) {
341                     clusteredNeighborHits.add(clusteredNeighborHit);
342                 }
343             }
344 
345             // Get the seed hit associated with this clustered hit.
346             CalorimeterHit clusteredHitSeed = hitToSeed.get(clusteredHit);
347 
348             // Loop over the clustered neighbor hits.
349             for (CalorimeterHit clusteredNeighborHit : clusteredNeighborHits) {
350                 // Check to make sure that the clustered neighbor hit
351                 // is not already associated with the current clustered
352                 // hit's seed.
353 
354                 if ((hitToSeed.get(clusteredNeighborHit) != clusteredHitSeed)) {
355                     // Check for lowest energy hit and that comparison hit is not already common.
356                     // If already common, this boundary is already accounted for.
357                     if (!equalEnergies(clusteredHit, clusteredNeighborHit) && !commonHits.containsKey(clusteredNeighborHit)) {
358 
359                         // Check and see if a list of common seeds
360                         // for this hit already exists or not.
361                         List<CalorimeterHit> commonHitList = commonHits.get(clusteredHit);
362 
363                         // If it does not, make a new one.
364                         if (commonHitList == null) {
365                             commonHitList = new ArrayList<CalorimeterHit>();
366                         }
367 
368                         // Add the neighbors to the seeds to set of
369                         // common seeds.
370                         commonHitList.add(clusteredHitSeed);
371                         commonHitList.add(hitToSeed.get(clusteredNeighborHit));
372 
373                         // Put the common seed list back into the set.
374                         commonHits.put(clusteredHit, commonHitList);
375 
376                     }
377                 }
378             }
379         } // End common hits loop.
380 
381         // Remove any common hits from the clustered hits collection.
382         // Note: this does not depend on the order of the commonHits key set
383         for (CalorimeterHit commonHit : commonHits.keySet()) {
384             hitToSeed.remove(commonHit);
385             hitList.remove(commonHit);
386         }
387 
388         /*
389          * All hits are sorted from above. The next part of the code is for building the output
390          * cluster collections.
391          */
392         // Add all hits except for common hits
393         for (CalorimeterHit ihit : hitList) {
394             CalorimeterHit iseed = hitToSeed.get(ihit);
395             BaseCluster icluster = seedToCluster.get(iseed);
396             // Consider time cut-is this hit in same time window as seed?
397             if (useTimeCut) {
398                 if (Math.abs(ihit.getTime() - iseed.getTime()) < timeWindow) {
399                     icluster.addHit(ihit);
400                 }
401             } // end of using time cut
402             else {
403                 icluster.addHit(ihit);
404             }
405         }
406 
407         // Add common hits
408         // Note: the order of the entry set in this loop does not affect the output.
409         for (Map.Entry<CalorimeterHit, List<CalorimeterHit>> commHit : commonHits.entrySet()) {
410             // Check that the common hit is in both time windows to their clusters
411             CalorimeterHit seedA = commHit.getValue().get(0);
412             CalorimeterHit seedB = commHit.getValue().get(1);
413 
414             boolean inTimeWithA = false;
415             boolean inTimeWithB = false;
416             // In time window with seedA?
417             if (Math.abs(commHit.getKey().getTime() - seedA.getTime()) < timeWindow) {
418                 inTimeWithA = true;
419             }
420 
421             // In time window with seedB?
422             if (Math.abs(commHit.getKey().getTime() - seedB.getTime()) < timeWindow) {
423                 inTimeWithB = true;
424             }
425 
426             double eclusterA = seedToCluster.get(seedA).getEnergy();
427             double eclusterB = seedToCluster.get(seedB).getEnergy();
428             double fractionA = eclusterA / (eclusterA + eclusterB);
429             double fractionB = eclusterB / (eclusterA + eclusterB);
430             double hitcontributionA = commHit.getKey().getCorrectedEnergy() * fractionA;
431             double hitcontributionB = commHit.getKey().getCorrectedEnergy() * fractionB;
432 
433             BaseCluster clusterA = seedToCluster.get(seedA);
434             BaseCluster clusterB = seedToCluster.get(seedB);
435 
436             if (useTimeCut) {
437                 // Do this if the hit is in both cluster's windows
438                 if (inTimeWithA && inTimeWithB) {
439                     clusterA.addHit(commHit.getKey(), hitcontributionA);
440                     clusterB.addHit(commHit.getKey(), hitcontributionB);
441                 }
442 
443                 //If the hit is only in 1 cluster's window, add the full contribution
444                 else if (inTimeWithA ^ inTimeWithB) {
445                     if (inTimeWithA) {
446                         clusterA.addHit(commHit.getKey());
447                     } else {
448                         clusterB.addHit(commHit.getKey());
449                     }
450                 }
451             } // end of using time cut
452             else {
453                 clusterA.addHit(commHit.getKey(), hitcontributionA);
454                 clusterB.addHit(commHit.getKey(), hitcontributionB);
455             }
456 
457         }
458 
459         // Remove clusters that do not pass cluster threshold and add to rejectedHitList.
460         for (int j = 0; j <= clusterList.size() - 1; j++) {
461             BaseCluster checkcluster = (BaseCluster) clusterList.get(j);
462             if (checkcluster.getEnergy() < clusterEnergyThreshold) {
463                 List<CalorimeterHit> clusterHits = checkcluster.getCalorimeterHits();
464                 for (CalorimeterHit nhit : clusterHits) {
465                     rejectedHitList.add(nhit);
466                 }
467                 clusterList.remove(checkcluster);
468                 j--;
469             } else {
470                 // Computer the position of the cluster and set it.
471                 calculatePosition(checkcluster);
472                 continue;
473             }
474         }
475         
476      /*  System.out.println("Number of clusters:"+clusterList.size());
477         for(Cluster cluster : clusterList) {
478            System.out.println(cluster.getEnergy());
479         }*/
480         return clusterList;
481     }

483     /**
484      * Handles pathological case where multiple neighboring crystals have EXACTLY the same energy.
485      * @param hit
486      * @param neighbor Neighbor to hit
487      * @return boolean value of if the hit is a seed
488      */
489     private boolean equalEnergies(CalorimeterHit hit, CalorimeterHit neighbor) {
490         boolean isSeed = true;
491 
492         int hix = hit.getIdentifierFieldValue("ix");
493         int hiy = Math.abs(hit.getIdentifierFieldValue("iy"));
494         int nix = neighbor.getIdentifierFieldValue("ix");
495         int niy = Math.abs(neighbor.getIdentifierFieldValue("iy"));
496         double hE = hit.getCorrectedEnergy();
497         double nE = neighbor.getCorrectedEnergy();
498         if (hE < nE) {
499             isSeed = false;
500         } else if ((hE == nE) && (hiy > niy)) {
501             isSeed = false;
502         } else if ((hE == nE) && (hiy == niy) && (hix < nix)) {
503             isSeed = false;
504         }
505         return isSeed;
506     }

508     /**
509      * Calculates the position of each cluster with no correction for particle type as documented in
510      * HPS Note 2014-001.
511      * @param cluster
512      */
513     private void calculatePosition(BaseCluster cluster) {
514         final double w0 = 3.1;
515         // calculated cluster x position
516         double xCl = 0.0;
517         // calculated cluster y position
518         double yCl = 0.0;
519         double eNumX = 0.0;
520         double eNumY = 0.0;
521         double eDen = 0.0;
522         List<CalorimeterHit> clusterHits = cluster.getCalorimeterHits();
523         for (CalorimeterHit hit : clusterHits) {
524             // This block fills a map with crystal to center of face of crystal
525             // Get the hit indices as a Point.
526             int ix = hit.getIdentifierFieldValue("ix");
527             int iy = hit.getIdentifierFieldValue("iy");
528             Point hitIndex = new Point(ix, iy);
529 
530             // Get the corrected position for this index pair.
531             double[] position = correctedPositionMap.get(hitIndex);
532 
533             if (position == null) {
534                 addToMap(hit);
535                 position = correctedPositionMap.get(hitIndex);
536             }
537 
538             eNumX += Math.max(0.0, (w0 + Math.log(hit.getCorrectedEnergy() / cluster.getEnergy()))) * (correctedPositionMap.get(hitIndex)[0] / 10.0);
539             eNumY += Math.max(0.0, (w0 + Math.log(hit.getCorrectedEnergy() / cluster.getEnergy()))) * (correctedPositionMap.get(hitIndex)[1] / 10.0);
540             eDen += Math.max(0.0, (w0 + Math.log(hit.getCorrectedEnergy() / cluster.getEnergy())));
541 
542         } // end for iteration through clusterHits
543 
544         xCl = eNumX / eDen;
545         yCl = eNumY / eDen;
546 
547         double[] clusterPosition = new double[3];
548         clusterPosition[0] = xCl * 10.0;// mm
549         clusterPosition[1] = yCl * 10.0;// mm
550         int ix = clusterHits.get(0).getIdentifierFieldValue("ix");
551         int iy = clusterHits.get(0).getIdentifierFieldValue("iy");
552         Point hitIndex = new Point(ix, iy);
553         clusterPosition[2] = correctedPositionMap.get(hitIndex)[2];
554 
555         cluster.setPosition(clusterPosition);
556         cluster.setNeedsPropertyCalculation(false);
557     }

559     /**
560      * This constructs a mapping of a crystal to an x,y position at the face of the ecal. This fills
561      * the correctedPositionMap with the position of a crystal's face from the index x,y of the
562      * crystal.
563      * @param hit
564      */
565     private void addToMap(CalorimeterHit hit) {
566         int ix = hit.getIdentifierFieldValue("ix");
567         int iy = hit.getIdentifierFieldValue("iy");
568         Point hitIndex = new Point(ix, iy);
569         // If the result is null, it hasn't been calculated yet.
570         // Calculate the corrected position.
571         IGeometryInfo geom = hit.getDetectorElement().getGeometry();
572         double[] pos = geom.transformLocalToGlobal(VecOp.add(geom.transformGlobalToLocal(geom.getPosition()),
573                 (Hep3Vector) new BasicHep3Vector(0, 0, -1 * ((Trd) geom.getLogicalVolume().getSolid()).getZHalfLength()))).v();
574 
575         // Store the result in the map.
576         correctedPositionMap.put(hitIndex, pos);
577     }
