3   import org.lcsim.event.EventHeader;
4   import org.lcsim.event.base.BaseCluster;
5   import org.lcsim.util.Driver;
6   
7   /**
8    * Applies a time-dependent global correction factor to the energies of all ecal clusters.
9    */
10  public abstract class TimeDependentEcalGains extends Driver{
11      private String ecalClusterCollectionName = "EcalClustersCorr";
12      public void setEcalClusterCollectionName(String name){
13          this.ecalClusterCollectionName = name;
14      }
15      public void process(EventHeader event){
16          double gain = getGain(event.getTimeStamp());
17          for(BaseCluster c : event.get(BaseCluster.class, ecalClusterCollectionName)){
18              double old_energy = c.getEnergy();
19              double new_energy = old_energy*gain;
20              c.setEnergy(new_energy);
21          }
22      }
23      
24      protected abstract double getGain(long timeStamp);
25      
26  }
