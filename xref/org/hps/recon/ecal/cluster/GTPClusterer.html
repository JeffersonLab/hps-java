17  /**
18   * Class <code>GTPClusterer</code> is an implementation of the abstract class <code>AbstractClusterer</code> that is
19   * responsible for producing clusters using the GTP algorithm employed by the hardware.<br/>
20   * <br/>
21   * The GTP algorithm produces clusters by finding hits representing local spatiotemporal energy maxima and forming a
22   * cluster from the hits within the aforementioned spatiotemporal window. A given hit is first checked to see if it
23   * exceeds some minimum energy threshold (referred to as the "seed energy threshold"). If this is the case, the
24   * algorithm looks at all hits that occurred in the same crystal as the comparison hit, or any crystal directly adjacent
25   * to it, within a programmable time window. If the hit exceeds all hits meeting these criteria in energy, the hit is
26   * considered the “seed hit” of a cluster. Then, all hits within the 3x3 spatial window which occur in the time window
27   * are added to a <code>Cluster</code> object.<br/>
28   * <br/>
29   * Note that the algorithm employs two distinct temporal windows. The first is the “verification” window. This is used
30   * to check that the potential seed hit is a local maximum in energy, and is required to be symmetric (i.e. as long
31   * before the seed time as after it) to ensure consistency. The second temporal window is the “inclusion” window, which
32   * determines which hits are included in the cluster. The inclusion window can be asymmetric, but can not exceed the
33   * verification window in length. As an example, one could choose a 12 ns verification window, meaning that the
34   * algorithm would 12 ns before and after the seed hit to check that it has the highest energy, but use a 4 ns/12 ns
35   * inclusion window, meaning that the algorithm would only include hits in 3x3 spatial window up to 4 ns before and up
36   * to 12 ns after the seed hit in the cluster. Due to the way the hardware processes hits, the higher energy parts of a
37   * cluster always occur first in time, so it is not necessarily desirable to include hits significantly before the seed.
38   * It is however, necessary to verify a hit’s status as a maximum across the full time window to ensure consistency in
39   * cluster formation. <code>GTPClusterer</code> automatically defines the inclusion window in terms of the verification
40   * window.<br/>
41   * <br/>
42   * <code>GTPClusterer</code> requires as input a collection of <code>CalorimeterHit</code> objects representing the
43   * event hits. It will then produce a collection of <code>Cluster</code> objects representing the GTP algorithm output.
44   * It is designed to be run on Monte Carlo events where each event represents a 2 ns beam bunch. If the input data is
45   * formatted in the style of hardware readout, the sister class <code>GTPOnlineClusterer</code> should be used instead.
46   * 
47   * @see Cluster
48   * @see CalorimeterHit
49   * @see AbstractClusterer
50   * @see GTPOnlineClusterer
51   */
52  public class GTPClusterer extends AbstractClusterer {
53  
54      /**
55       * The minimum energy required for a hit to be considered as a cluster center. Hits with energy less than this value
56       * will be ignored. This is the seed energy lower bound cut.
57       */
58      private double seedEnergyThreshold;
59  
60      /**
61       * Indicates the number of FADC clock cycles (each cycle is 4 ns) before and after a given cycle that should be
62       * considered when checking if a cluster is a local maximum in space-time. This is the hit verification temporal
63       * window.
64       */
65      private int clusterWindow;
66  
67      /**
68       * Stores a set of all the hits occurring in each clock cycle for the number of clock cycles that should be
69       * considered for clustering.
70       */
71      private LinkedList<Map<Long, CalorimeterHit>> hitBuffer;
72  
73      /**
74       * Whether an asymmetric or symmetric window should be used for adding hits to a cluster. This defines the hit
75       * inclusion temporal window with respect to the verification window.
76       */
77      private boolean limitClusterRange = false;
78  
79      /**
80       * Sets whether debug text should be written.
81       */
82      private boolean verbose = false;
83  
84      /**
85       * Sets whether the clusterer should store the collection of hits that are part of clusters. This needs to be true
86       * if the clusters are to be written out to LCIO.
87       */
88      private boolean writeHitCollection = true;
89  
90      /**
91       * Instantiates a new instance of a Monte Carlo GTP clustering algorithm. It will use the default seed energy
92       * threshold of 50 MeV and a default hit inclusion window of +/- 2 ns. By default the cluster inclusion and
93       * verification windows are identical.
94       */
95      GTPClusterer() {
96          super(new String[] {"seedEnergyThreshold", "clusterWindow"}, new double[] {0.050, 2});
97      }
98  
99      /**
100      * Processes the argument <code>CalorimeterHit</code> collection and forms a collection of <code>Cluster</code>
101      * objects according to the GTP clustering algorithm.
102      * 
103      * @param event - The object containing event data.
104      * @param hits - A list of <code>CalorimeterHit</code> objects from which clusters should be formed.
105      */
106     public List<Cluster> createClusters(EventHeader event, List<CalorimeterHit> hits) {
107         // Store each hit in a set by its cell ID so that it may be
108         // easily acquired later.
109         HashMap<Long, CalorimeterHit> hitMap = new HashMap<Long, CalorimeterHit>();
110         for (CalorimeterHit hit : hits) {
111             hitMap.put(hit.getCellID(), hit);
112         }
113 
114         // Remove the last event from the hit buffer and add the new one.
115         hitBuffer.removeLast();
116         hitBuffer.addFirst(hitMap);
117 
118         // Run the clustering algorithm on the buffer.
119         List<Cluster> clusterList = getClusters();
120 
121         // The MC GTP algorithm collects hits from across events; to be
122         // stored in LCIO format properly, it needs to separately store
123         // its clusters' hits in a collection.
124         if (writeHitCollection) {
125             // Create a set to store the hits so that each one may be
126             // stored only once.
127             Set<CalorimeterHit> hitSet = new HashSet<CalorimeterHit>();
128 
129             // Loop over all clusters and add their hits to the set.
130             for (Cluster cluster : clusterList) {
131                 for (CalorimeterHit hit : cluster.getCalorimeterHits()) {
132                     hitSet.add(hit);
133                 }
134             }
135 
136             // Convert the set into a List object so that it can be stored
137             // in LCIO.
138             List<CalorimeterHit> clusterHits = new ArrayList<CalorimeterHit>(hitSet.size());
139             clusterHits.addAll(hitSet);
140 
141             // Place the list of hits into the event stream.
142             event.put("GTPHits", hits, CalorimeterHit.class, 0);
143         }
144 
145         // Return the clusters.
146         return clusterList;
147     }

153      *         <code>ClusterType.GTP</code>.

155     @Override
156     public ClusterType getClusterType() {
157         return ClusterType.GTP;

160     /**
161      * Sets the clustering algorithm parameters.
162      */
163     @Override
164     public void initialize() {
165         // Set cuts.
166         setSeedEnergyThreshold(getCuts().getValue("seedEnergyThreshold"));
167         setClusterWindow((int) getCuts().getValue("clusterWindow"));
168 
169         // Initiate the hit buffer.
170         hitBuffer = new LinkedList<Map<Long, CalorimeterHit>>();
171 
172         // Populate the event buffer with (2 * clusterWindow + 1)
173         // empty events. These empty events represent the fact that
174         // the first few events will not have any events in the past
175         // portion of the buffer.
176         int bufferSize = (2 * clusterWindow) + 1;
177         for (int i = 0; i < bufferSize; i++) {
178             hitBuffer.add(new HashMap<Long, CalorimeterHit>(0));
179         }
180     }
181 
182     /**
183      * Sets the number of clock cycles before and after a given cycle that will be used when checking whether a given
184      * hit is a local maximum in both time and space. Note that a value of <code>N</code> indicates that <code>N</code>
185      * clock cycles before and <code>N</code> clock cycles after will be considered. Thusly, a total of
186      * <code>2N + 1</code> clock cycles will be used. This defines the size of the hit verification window. The
187      * inclusion window is defined as a function of this, as discussed in the method <code>setLimitClusterRange</code>.
188      * 
189      * @param clusterWindow - The number of additional clock cycles to include in the clustering checks. A negative
190      *            value will be treated as zero.
191      */
192     void setClusterWindow(int clusterWindow) {
193         // The cluster window of must always be at least zero.
194         if (clusterWindow < 0) {
195             this.clusterWindow = 0;
196         }
197 
198         // If the cluster window is non-zero, then store it.
199         else {
200             this.clusterWindow = clusterWindow;
201         }
202     }
203 
204     /**
205      * Sets the behavior of the hit inclusion and verification temporal windows. If set to <code>true</code>, the hit
206      * inclusion window will be defined as one clock-cycle before the seed hit and the regular length of the
207      * verification window after the seed hit. If <code>false</code>, the inclusion window and the verification window
208      * are set to be identical.
209      * 
210      * @param limitClusterRange - <code>true</code> indicates that the asymmetric clustering window should be used and
211      *            <code>false</code> that the symmetric window should be used.
212      */
213     void setAsymmetricWindow(boolean limitClusterRange) {
214         this.limitClusterRange = limitClusterRange;
215     }
216 
217     /**
218      * Sets the minimum energy a hit must have before it will be considered for cluster formation.
219      * 
220      * @param seedThreshold - The seed threshold in GeV.
221      */
222     void setSeedEnergyThreshold(double seedEnergyThreshold) {
223         // A negative energy threshold is non-physical. All thresholds
224         // be at least zero.
225         if (seedEnergyThreshold < 0.0) {
226             this.seedEnergyThreshold = 0.0;
227         }
228 
229         // If the energy threshold is valid, then use it.
230         else {
231             this.seedEnergyThreshold = seedEnergyThreshold;
232         }
233     }

256     /**
257      * Generates a list of clusters from the current hit buffer. The "present" event is taken to be the list of hits
258      * occurring at index <code>clusterWindow</code>, which is the middle of the buffer.
259      * 
260      * @return Returns a <code>List</code> of <code>HPSEcalCluster
261      * </code> objects generated from the current event.
262      */
263     private List<Cluster> getClusters() {
264         // Generate a list for storing clusters.
265         List<Cluster> clusters = new ArrayList<Cluster>();
266 
267         // Get the list of hits at the current time in the event buffer.
268         Map<Long, CalorimeterHit> currentHits = hitBuffer.get(clusterWindow);
269 
270         // VERBOSE :: Print the cluster window.
271         if (verbose) {
272             // Print the event header.
273             System.out.printf("%n%nEvent:%n");
274 
275             // Calculate some constants.
276             int window = (hitBuffer.size() - 1) / 2;
277             int bufferNum = 0;
278 
279             // Print out all of the hits in the event buffer.
280             for (Map<Long, CalorimeterHit> bufferMap : hitBuffer) {
281                 System.out.printf("Buffer %d:%n", hitBuffer.size() - bufferNum - window - 1);
282                 CalorimeterHit hit = null;
283 
284                 for (Entry<Long, CalorimeterHit> entry : bufferMap.entrySet()) {
285                     hit = entry.getValue();
286                     System.out.printf("\t(%3d, %3d) --> %.4f (%.4f)%n", hit.getIdentifierFieldValue("ix"),
287                             hit.getIdentifierFieldValue("iy"), hit.getCorrectedEnergy(), hit.getRawEnergy());
288                 }
289 
290                 bufferNum++;
291             }
292 
293             // If there are not hits, indicate this.
294             if (currentHits.isEmpty()) {
295                 System.out.println("\tNo hits this event!");
296             }
297         }
298 
299         // For a hit to be a cluster center, it must be a local maximum
300         // both with respect to its neighbors and itself both in the
301         // present time and at all times within the event buffer.
302         seedLoop: for (Long currentID : currentHits.keySet()) {
303             // Get the actual hit object.
304             CalorimeterHit currentHit = currentHits.get(currentID);
305 
306             // VERBOSE :: Print the current cluster.
307             if (verbose) {
308                 System.out.printf("Cluster Check:%n");
309                 System.out.printf("\t(%3d, %3d) --> %.4f%n", currentHit.getIdentifierFieldValue("ix"),
310                         currentHit.getIdentifierFieldValue("iy"), currentHit.getCorrectedEnergy());
311             }
312 
313             // Store the energy of the current hit.
314             double currentEnergy = currentHit.getCorrectedEnergy();
315 
316             // If the hit energy is lower than the minimum threshold,
317             // then we immediately reject this hit as a possible cluster.
318             if (currentEnergy < seedEnergyThreshold) {
319                 // VERBOSE :: Note the reason the potential seed was
320                 // rejected.
321                 if (verbose) {
322                     System.out.printf("\tREJECT :: Does not exceed seed threshold %.4f.%n", seedEnergyThreshold);
323                 }
324 
325                 // Skip to the next potential seed.
326                 continue seedLoop;
327             }
328 
329             // Store the crystals that are part of this potential cluster,
330             // starting with the cluster seed candidate.
331             BaseCluster cluster = createBasicCluster();
332             cluster.addHit(currentHit);
333             cluster.setPosition(currentHit.getDetectorElement().getGeometry().getPosition().v());
334             cluster.setNeedsPropertyCalculation(false);
335 
336             // Get the set of neighbors for this hit.
337             Set<Long> neighbors = neighborMap.get(currentHit.getCellID());
338 
339             // Sort through each event stored in the buffer.
340             int bufferIndex = 0;
341             for (Map<Long, CalorimeterHit> bufferHits : hitBuffer) {
342                 // Get the hit energy at the current hit's position in
343                 // the buffer, if it exists. Ignore the current seed candidate.
344                 CalorimeterHit bufferHit = bufferHits.get(currentID);
345                 if (bufferHit != null && bufferHit != currentHit) {
346                     double bufferHitEnergy = bufferHit.getRawEnergy();
347 
348                     // Check to see if the hit at this point in the buffer
349                     // is larger than then original hit. If it is, we may
350                     // stop the comparison because this is not a cluster.
351                     if (bufferHitEnergy > currentEnergy) {
352                         // VERBOSE :: Output the reason the potential
353                         // seed was rejected along with the
354                         // hit that caused it.
355                         if (verbose) {
356                             System.out.printf("\tREJECT :: Buffer hit surpasses hit energy.");
357                             System.out.printf("\tBUFFER HIT :: (%3d, %3d) --> %.4f%n",
358                                     bufferHit.getIdentifierFieldValue("ix"), bufferHit.getIdentifierFieldValue("iy"),
359                                     bufferHit.getCorrectedEnergy(), bufferHit.getRawEnergy());
360                         }
361 
362                         // Skip to the next potential seed.
363                         continue seedLoop;
364                     }
365 
366                     // If the buffer hit is smaller, then add its energy
367                     // to the cluster total energy.
368                     else {
369                         if (limitClusterRange && bufferIndex <= clusterWindow + 1) {
370                             cluster.addHit(bufferHit);
371                         } else if (!limitClusterRange) {
372                             cluster.addHit(bufferHit);
373                         }
374                     }
375                 }
376 
377                 // We must also make sure that the original hit is
378                 // larger than all of the neighboring hits at this
379                 // point in the buffer as well.
380                 for (Long neighborID : neighbors) {
381                     // Get the neighbor hit energy if it exists.
382                     CalorimeterHit neighborHit = bufferHits.get(neighborID);
383                     if (neighborHit != null) {
384                         double neighborHitEnergy = neighborHit.getRawEnergy();
385 
386                         // Check to see if the neighbor hit at this point
387                         // in the buffer is larger than then original hit.
388                         // If it is, we may stop the comparison because this
389                         // is not a cluster.
390                         if (neighborHitEnergy > currentEnergy) {
391                             // VERBOSE :: Output the reason the potential
392                             // seed was rejected along with the
393                             // hit that caused it.
394                             if (verbose) {
395                                 System.out.printf("\tREJECT :: Buffer hit surpasses hit energy.%n");
396                                 System.out.printf("\tBUFFER HIT :: (%3d, %3d) --> %.4f%n",
397                                         neighborHit.getIdentifierFieldValue("ix"),
398                                         neighborHit.getIdentifierFieldValue("iy"), neighborHit.getCorrectedEnergy(),
399                                         neighborHit.getRawEnergy());
400                             }
401 
402                             // Skip to the next potential seed.
403                             continue seedLoop;
404                         }
405 
406                         // If the buffer neighbor hit is smaller, then
407                         // add its energy to the cluster total energy.
408                         else {
409                             if (limitClusterRange && bufferIndex <= clusterWindow + 1) {
410                                 cluster.addHit(neighborHit);
411                             } else if (!limitClusterRange) {
412                                 cluster.addHit(neighborHit);
413                             }
414                         }
415                     }
416                 }
417 
418                 // Increment the buffer index.
419                 bufferIndex++;
420             }
421 
422             // Add the cluster to the list of clusters.
423             clusters.add(cluster);
424 
425             // VERBOSE :: Output the clusters generated from this event.
426             if (verbose) {
427                 System.out.printf("Cluster added.%n");
428                 System.out.printf("\t(%3d, %3d) --> %.4f GeV --> %d hits%n", cluster.getCalorimeterHits().get(0)
429                         .getIdentifierFieldValue("ix"),
430                         cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("iy"), cluster.getEnergy(), cluster
431                                 .getCalorimeterHits().size());
432                 for (CalorimeterHit hit : cluster.getCalorimeterHits()) {
433                     System.out.printf("\t\tCLUSTER HIT :: (%3d, %3d) --> %.4f%n", hit.getIdentifierFieldValue("ix"),
434                             hit.getIdentifierFieldValue("iy"), hit.getCorrectedEnergy(), hit.getRawEnergy());
435                 }
436             }
437         }
438 
439         // Return the generated list of clusters.
440         return clusters;
441     }
