3   /**
4    * Class <code>GTPClusterDriver</code> is an implementation of the
5    * <code>ClusterDriver</code> class that defines employs the readout
6    * variant of the GTP hardware clustering algorithm. Specifics on the
7    * behavior of this algorithm can be found in its documentation.<br/>
8    * <br/>
9    * <code>GTPClusterDriver</code> allows for all of the variable settings
10   * used by the GTP algorithm to be defined. It also can be set to
11   * "verbose" mode, where it will output detailed information on each
12   * event and the cluster forming process. This is disabled by default,
13   * but can be enabled for debugging purposes.<br/>
14   * <br/>
15   * <code>GTPClusterDriver</code> is designed to read from Monte Carlo
16   * data organized into 2-ns beam bunches. It can not be used for hardware
17   * readout data, or Monte Carlo formatted in this style. For this data,
18   * the <code>GTPOnlineClusterer</code> should be employed instead.
19   * 
20   * @see GTPClusterer
21   */
22  public class GTPClusterDriver extends ClusterDriver {
23      /** An instance of the clustering algorithm object for producing
24       * cluster objects. */
25      private final GTPClusterer gtp;
26      
27      /**
28       * Instantiates a new <code>GTPClusterer</code>, which will produce
29       * clusters using the GTP algorithm in the 2-ns beam bunch scheme.
30       * It will, by default, use a 50 MeV seed energy cut with +/- 2 a
31       * clock-cycle verification and inclusion window.
32       */
33      public GTPClusterDriver() {
34          clusterer = ClustererFactory.create("GTPClusterer");
35          gtp = (GTPClusterer) clusterer;
36          setWriteClusterCollection(true);
37      }
38      
39      /**
40       * Sets whether the behavior of the hit inclusion window with respect
41       * to the hit verification window. If set to <code>false</code>,
42       * both windows will be identical in size. Otherwise, the inclusion
43       * window will be equal in size after the seed hit, but encompass
44       * only one clock-cycle before the seed hit. This should be replaced
45       * by the method <code>setAsymmetricWindow</code>.
46       * @param limitClusterRange - <code>true</code> indicates that the
47       * asymmetric window should be used and <code>false</code> that it
48       * should not.
49       */
50      @Deprecated
51      public void setLimitClusterRange(boolean limitClusterRange) {
52          gtp.setAsymmetricWindow(limitClusterRange);
53      }
54      
55      /**
56       * Sets the size of the hit verification temporal window. Note
57       * that this defines the size of the window in one direction, so
58       * the full time window will be <code>(2 * clusterWindow) + 1</code>
59       * clock-cycles in length. (i.e., it will be a length of
60       * <code>clusterWindow</code> before the seed hit, a length of
61       * <code>clusterWindow</code> after the seed hit, plus the cycle
62       * that includes the seed hit.) Time length is in clock-cycles.
63       * @param clusterWindow - The number of clock-cycles around the
64       * hit in one direction.
65       */
66      public void setClusterWindow(int clusterWindow) {
67          gtp.getCuts().setValue("clusterWindow", clusterWindow);
68      }
69      
70      /**
71       * Sets the minimum seed energy needed for a hit to be considered
72       * for forming a cluster. This is the seed energy lower bound trigger
73       * cut and is in units of GeV.
74       * @param seedEnergyThreshold - The minimum cluster seed energy in
75       * GeV.
76       */
77      public void setSeedEnergyThreshold(double seedEnergyThreshold) {
78          gtp.getCuts().setValue("seedEnergyThreshold", seedEnergyThreshold);
79      }
80      
81      /**
82       * Sets whether the behavior of the hit inclusion window with respect
83       * to the hit verification window. If set to <code>false</code>,
84       * both windows will be identical in size. Otherwise, the inclusion
85       * window will be equal in size after the seed hit, but encompass
86       * only one clock-cycle before the seed hit.
87       * @param asymmetricWindow - <code>true</code> indicates that the
88       * asymmetric window should be used and <code>false</code> that it
89       * should not.
90       */
91      public void setAsymmetricWindow(boolean asymmetricWindow) {
92          gtp.setAsymmetricWindow(asymmetricWindow);
93      }
94      
95      /**
96       * Sets whether the clustering algorithm should output diagnostic
97       * text or not.
98       * @param verbose - <code>true</code> indicates that the driver should
99       * output diagnostic text and <code>false</code> that it should not.
100      */
101     public void setVerbose(boolean verbose) {
102         gtp.setVerbose(verbose);
103     }
104     
105     /**
106      * Defines whether the output of this clusterer should be persisted
107      * to LCIO or not. By default, this 
108      * @param state - <code>true</code> indicates that clusters will
109      * be persisted, and <code>false</code> that they will not.
110      */
111     @Override
112     public void setWriteClusterCollection(boolean state) {
113         // Set the flag as appropriate with the superclass.
114         super.setWriteClusterCollection(state);
115         
116         // Also tell the clusterer whether it should persist its hit
117         // collection or not.
118         gtp.setWriteHitCollection(state);
119     }
120 }
