6   /**
7    * This uses the uncorrected cluster energy to correct the position of the cluster.
8    * This should be used before the energy is corrected on the Cluster and after
9    * cluster-track matching.
10   */
11  public final class ClusterPositionCorrection {
12      //Parameterizations tested in MC using v3-fieldmap
13      //Nov 2015
14  
15      // Variables for electron position corrections.
16      static final double ELECTRON_POS_A1 = 0.004483;
17      static final double ELECTRON_POS_A2 = -0.02884;
18      static final double ELECTRON_POS_B1 = 0.6197;
19      static final double ELECTRON_POS_B2 = -2.279;
20      static final double ELECTRON_POS_B3 = 3.66;
21  
22      // Variables for positron position corrections.
23      static final double POSITRON_POS_A1 = 0.006887;
24      static final double POSITRON_POS_A2 = -0.03207;
25      static final double POSITRON_POS_B1 = -0.8048;
26      static final double POSITRON_POS_B2 = 0.9366;
27      static final double POSITRON_POS_B3 = 2.628;
28  
29      // Variables for photon position corrections.
30      static final double PHOTON_POS_A1 = 0.005385;
31      static final double PHOTON_POS_A2 = -0.03562;
32      static final double PHOTON_POS_B1 = -0.1948;
33      static final double PHOTON_POS_B2 = -0.7991;
34      static final double PHOTON_POS_B3 = 3.797;
35  
36  
37      public static double[] calculateCorrectedPosition(Cluster cluster) {
38          double clusterPosition[] = cluster.getPosition();                
39          double correctedPosition = computeCorrectedPosition(cluster.getParticleId(), clusterPosition[0], cluster.getEnergy());
40          double[] position = new double[3];
41          position[0] = correctedPosition;
42          position[1] = clusterPosition[1];
43          position[2] = clusterPosition[2];
44          return position;
45      }
46  
47      public static void setCorrectedPosition(BaseCluster cluster) {
48          cluster.setPosition(calculateCorrectedPosition(cluster));
49      }
50  
51      /**
52       * Calculates position correction based on cluster raw energy, x calculated position, 
53       * and particle type as per 
54       * <a href="https://misportal.jlab.org/mis/physics/hps_notes/index.cfm?note_year=2014">HPS Note 2014-001</a>
55       * @param pdg Particle id as per PDG
56       * @param xCl Calculated x centroid position of the cluster, uncorrected, at face
57       * @param rawEnergy Raw energy of the cluster (sum of hits with shared hit distribution)
58       * @return the corrected x position
59       */
60      private static double computeCorrectedPosition(int pdg, double xPos, double rawEnergy) {
61          //double xCl = xPos / 10.0;//convert to cm
62          double xCorr;
63          switch(pdg) {
64              case 11: //Particle is electron        
65                  xCorr = positionCorrection(xPos, rawEnergy, ELECTRON_POS_A1, ELECTRON_POS_A2, ELECTRON_POS_B1, ELECTRON_POS_B2, ELECTRON_POS_B3);
66                  return xCorr;
67              case -11:// Particle is positron       
68                  xCorr = positionCorrection(xPos, rawEnergy, POSITRON_POS_A1, POSITRON_POS_A2, POSITRON_POS_B1, POSITRON_POS_B2, POSITRON_POS_B3);
69                  return xCorr;
70              case 22: // Particle is photon      
71                  xCorr = positionCorrection(xPos, rawEnergy, PHOTON_POS_A1, PHOTON_POS_A2, PHOTON_POS_B1, PHOTON_POS_B2, PHOTON_POS_B3);
72                  return xCorr;
73              default: //Unknown 
74                  xCorr = xPos;
75                  return xCorr;
76          }
77      }
78  
79      /**
80       * Calculates the position correction in cm using the raw energy and variables associated with the fit
81       * of the particle. Prodecure described in  
82       * <a href="https://misportal.jlab.org/mis/physics/hps_notes/index.cfm?note_year=2014">HPS Note 2014-001</a>
83       * @param xCl
84       * @param rawEnergy
85       * @param varA1
86       * @param varA2
87       * @param varA3
88       * @param varB1
89       * @param varB2
90       * @param varB3
91       * @return the cluster position correction
92       */    
93      private static double positionCorrection(double xCl, double rawEnergy, double varA1, double varA2, double varB1, double varB2, double varB3) {
94          //return ((xCl - (varB1 * rawEnergy + varB2 / Math.sqrt(rawEnergy) + varB3))/(varA1 / Math.sqrt(rawEnergy) + varA2 + 1));
95          return ((xCl - (varB1 * rawEnergy + varB2 / Math.sqrt(rawEnergy) + varB3))/(varA1 / Math.sqrt(rawEnergy) + varA2 + 1));
96      }
97  }
