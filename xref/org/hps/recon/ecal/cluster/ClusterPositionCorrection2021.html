5   /**
6    * This uses the corrected cluster energy to correct the position of the
7    * cluster. This is to be used with 2021 data
8    *
9    * To determine these corrections, we simulated e+ e- and gamma at fixed
10   * energies over the ECAL acceptance, sampled the true hit position with MC
11   * scoring plane, and compared with the measured cluster position. We then
12   * considered:
13   * 
14   * dX vs X (dX = measured - true) ---> true=measured-dX dY vs Y (dY = measured -
15   * true) ---> true=measured-dY
16   * 
17   * We then performed a fit to these with dX = q + m*X dY = q1 + t*Y if x < 0 ; =
18   * q2 + t*Y if Y > 0
19   * 
20   * See also A.C. Talk at Nov. 2022 collaboration meeting
21   * 
22   * We then checked the dependency of the q,m, q1,q2, t parameters as a function
23   * of the energy
24   * 
25   * Electrons and Positrons: parameter(E) = p0 + p1*pow(E,p2) for all parameters
26   * Photons: par(E) = p0 + p1*pow(E,p2) | par = q,m par(E) = (a + b*E + c*E*E)/(d
27   * + e*E + f*E*E) | par = q1,t,q2
28   */
29  final class ClusterPosResult21 {
30      private final double X;
31      private final double Y;
32  
33      public ClusterPosResult21(double X, double Y) {
34          this.X = X;
35          this.Y = Y;
36      }
37  
38      public double getX() {
39          return X;
40      }
41  
42      public double getY() {
43          return Y;
44      }
45  
46  }

116      *                already-corrected energy. THIS IS DIFFERENT FROM 2015!
117      * @return The corrected cluster position

120     public static double[] calculateCorrectedPosition(BaseCluster cluster) {
121         double clusterPosition[] = cluster.getPosition();
122 
123         ClusterPosResult21 correctedPosition = computeCorrectedPosition(cluster.getParticleId(), clusterPosition[0],
124                 clusterPosition[1], cluster.getEnergy());
125 
126         double[] position = new double[3];
127         position[0] = correctedPosition.getX();
128         position[1] = correctedPosition.getY();
129         position[2] = clusterPosition[2];
130 
131         return position;
132     }
133 
134     public static void setCorrectedPosition(BaseCluster cluster) {
135         cluster.setPosition(calculateCorrectedPosition(cluster));
136     }

138     /**
139      * Calculates position correction based on cluster corrected energy, x
140      * calculated position, and particle type as per <a href=
141      * "https://misportal.jlab.org/mis/physics/hps_notes/index.cfm?note_year=2014">HPS
142      * Note 2014-001</a>
143      * 
144      * @param pdg    Particle id as per PDG
145      * @param xCl    Calculated x centroid position of the cluster, uncorrected, at
146      *               face
147      * @param Energy Corrected energy of the cluster
148      * @return the corrected x position
149      */
150     private static ClusterPosResult21 computeCorrectedPosition(int pdg, double xPos, double yPos, double Energy) {
151         // double xCl = xPos / 10.0;//convert to cm
152         ClusterPosResult21 res;
153         double xCorr;
154         switch (pdg) {
155             case 11: // Particle is electron
156                 res = positionCorrectionElectron(xPos, yPos, Energy);
157                 break;
158             case -11:// Particle is positron
159                 res = positionCorrectionPositron(xPos, yPos, Energy);
160                 break;
161             case 22: // Particle is photon
162                 res = positionCorrectionPhoton(xPos, yPos, Energy);
163                 break;
164             default: // Unknown
165                 res = new ClusterPosResult21(xPos, yPos);
166         }
167         return res;
168     }

170     private static ClusterPosResult21 positionCorrectionElectron(double xPos, double yPos, double Energy) {
171         double xCorr, yCorr;
172         double deltaX, deltaY;
173 
174         double q = ELECTRON_POS_Q_P0 + ELECTRON_POS_Q_P1 * Math.pow(Energy, ELECTRON_POS_Q_P2);
175         double m = ELECTRON_POS_M_P0 + ELECTRON_POS_M_P1 * Math.pow(Energy, ELECTRON_POS_M_P2);
176         double q1 = ELECTRON_POS_Q1_P0 + ELECTRON_POS_Q1_P1 * Math.pow(Energy, ELECTRON_POS_Q2_P2);
177         double q2 = ELECTRON_POS_Q2_P0 + ELECTRON_POS_Q2_P1 * Math.pow(Energy, ELECTRON_POS_Q2_P2);
178         double t = ELECTRON_POS_T_P0 + ELECTRON_POS_T_P1 * Math.pow(Energy, ELECTRON_POS_T_P2);
179 
180         deltaX = q + m * xPos;
181 
182         if (yPos < 0) {
183             deltaY = q1 + t * yPos;
184         } else {
185             deltaY = q2 + t * yPos;
186         }
187 
188         xCorr = xPos - deltaX;
189         yCorr = yPos - deltaY;
190 
191         return new ClusterPosResult21(xCorr, yCorr);
192     }
