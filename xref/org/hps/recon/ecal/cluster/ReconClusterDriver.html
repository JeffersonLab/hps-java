9   /**
10   * This is an implementation of {@link ClusterDriver} specialized for the
11   * {@link ReconClusterer}.  It currently implements optional
12   * writing of a rejected hit list to the LCSim event.
13   * 
14   * @see ReconClusterer
15   */
16  public class ReconClusterDriver extends ClusterDriver {
17      
18      // The name of the output collection of rejected hits.
19      String rejectedHitCollectionName = "ReconClusterRejectedHits";
20      
21      // Flag to persist the rejected hits to the LCIO file (off by default).
22      boolean writeRejectedHitCollection = false;
23              
24      public ReconClusterDriver() {
25          // Setup the Clusterer with the correct type.
26          clusterer = ClustererFactory.create("ReconClusterer");
27          
28          // Need this for correct seed hit position in list.
29          this.sortHits = true; 
30      }
31      
32      public void setClusterer(Clusterer clusterer) {
33          throw new UnsupportedOperationException("This method should not be used on this Driver.");
34      }
35      
36      public void setClustererName(String clustererName) {
37          throw new UnsupportedOperationException("This method should not be used on this Driver.");
38      }
39      
40      public void setHitEnergyThreshold(double hitEnergyThreshold) {
41          getClusterer().getCuts().setValue("hitEnergyThreshold", hitEnergyThreshold);
42      }
43      
44      public void setSeedEnergyThreshold(double seedEnergyThreshold) {
45          getClusterer().getCuts().setValue("seedEnergyThreshold", seedEnergyThreshold);
46      }
47      
48      public void setClusterEnergyThreshold(double clusterEnergyThreshold) {
49          getClusterer().getCuts().setValue("clusterEnergyThreshold", clusterEnergyThreshold);
50      }
51      
52      public void setMinTime(double minTime) {
53          getClusterer().getCuts().setValue("minTime", minTime);
54      }
55      
56      public void setMaxTime(double maxTime) {
57          getClusterer().getCuts().setValue("maxTime", maxTime);
58      }
59      
60      public void setTimeWindow(double timeWindow) {
61          getClusterer().getCuts().setValue("timeWindow", timeWindow);
62      }
63      
64      public void setUseTimeCut(boolean useTimeCut) {
65          ReconClusterer clusterer = getClusterer();
66          clusterer.setUseTimeCut(useTimeCut);
67      }    
68          
69      /**
70       * Perform job initialization.  
71       */
72      public void startOfData() {
73          // Perform standard start of data initialization from super-class.
74          super.startOfData();
75      }          
76      
77      /**
78       * Set the name of the list of rejected hits.
79       * By default, the name is null and so rejected hits will not be written
80       * unless this is called.
81       * @param rejectedHitCollectionName The name of the rejected hit list.
82       */
83      public void setRejectedHitCollectionName(String rejectedHitCollectionName) {
84          this.rejectedHitCollectionName = rejectedHitCollectionName;
85      }
86      
87      /**
88       * This controls whether the rejected hit collection is written out.
89       * @param writeRejectedHitCollection True to write rejected hit collection.
90       */
91      public void setWriteRejectedHitCollection(boolean writeRejectedHitCollection) {
92          this.writeRejectedHitCollection = writeRejectedHitCollection;
93      }
94      
95      /**
96       * Perform standard event processing, optionally writing the rejected hit list.
97       */
98      public void process(EventHeader event) {
99          // Do standard ClusterDriver processing.
100         super.process(event);
101         // Write rejected hit list.
102         if (this.writeRejectedHitCollection) {
103             writeRejectedHitList(event);        
104         }
105     }
106     
107     /**
108      * Write the list of rejected hits to the event, according to current Driver parameter settings.
109      */
110     void writeRejectedHitList(EventHeader event) {
111         List<CalorimeterHit> rejectedHitList = getReconClusterer().getRejectedHitList();
112         if (rejectedHitList == null) {
113             throw new RuntimeException("The rejectedHitList is null.");
114         }
115         int flag = 1 << LCIOConstants.CLBIT_HITS;
116         getLogger().finer("writing rejected hit list " + rejectedHitCollectionName + " with " + rejectedHitList.size() + " hits");
117         event.put(rejectedHitCollectionName, rejectedHitList, CalorimeterHit.class, flag, ecal.getReadout().getName());
118         // Flag the collection as a subset, because other collection's objects are being used.
119         event.getMetaData(rejectedHitList).setSubset(true);
120         // Are we writing this collection to the output LCIO file?
121         if (!this.writeRejectedHitCollection) {
122             getLogger().finest("Rejected hit list is transient and will not be persisted.");
123             // Flag as transient.
124             event.getMetaData(rejectedHitList).setTransient(true);
125         }   
126     }             
127     
128     ReconClusterer getReconClusterer() {
129         return (ReconClusterer) this.clusterer;
130     }
131 }
