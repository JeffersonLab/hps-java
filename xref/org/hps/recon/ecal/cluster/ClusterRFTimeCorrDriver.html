3   import java.util.ArrayList;
4   import java.util.Comparator;
5   import java.util.List;
6   
7   import org.hps.conditions.database.DatabaseConditionsManager;
8   import org.lcsim.event.Cluster;
9   import org.lcsim.event.EventHeader;
10  import org.lcsim.event.GenericObject;
11  import org.lcsim.geometry.Detector;
12  import org.lcsim.util.Driver;
13  import org.hps.conditions.trigger.TriggerTimeWindow.TriggerTimeWindowCollection;
14  
15  /**
16   * This driver chooses the highest energy cluster within the trigger time window in an event and uses the RF time to set
17   * the trigger time of the event.
18   */
19  public class ClusterRFTimeCorrDriver extends Driver {
20  
21      // From database as mean, min, max
22      // Defaults here are from 2015 running
23      // This is a tunable parameter from the DAQ
24      private static final double[] DEFAULT_PARAMETER = {45, 40, 50};
25  
26      private double[] parameter = DEFAULT_PARAMETER;
27  
28      // Get the ideal trigger time window for a run from the conditions db
29      public void detectorChanged(Detector detector) {
30  
31          DatabaseConditionsManager manager = DatabaseConditionsManager.getInstance();
32          TriggerTimeWindowCollection trigColl = manager.getCachedConditions(TriggerTimeWindowCollection.class,
33                  "trigger_time_windows").getCachedData();
34  
35          parameter[0] = trigColl.get(0).getOffset();
36          parameter[1] = trigColl.get(0).getOffsetMin();
37          parameter[2] = trigColl.get(0).getOffsetMax();
38  
39      }
40  
41      public void process(EventHeader event) {
42  
43          List<TriggerTime> triggerTime = new ArrayList<TriggerTime>();
44  
45          boolean foundTdiff = false;
46  
47          double clT = parameter[0];
48          double selectedClT = parameter[0];
49          double rfT = parameter[0];
50          int seedX = 0;
51          int seedY = 0;
52          int jj = 0;
53          // read in ecal clusters, choose the highest energy cluster in the trigger time window
54          if (event.hasCollection(Cluster.class, "EcalClusters")) {
55              // Get the list of clusters in the event
56              List<Cluster> cl = event.get(Cluster.class, "EcalClusters");
57              // Sort the clusters as highest to lowest energy
58              Comparator<Cluster> comparator = new ClusterUtilities.ClusterSeedComparator();
59              ClusterUtilities.sort(cl, comparator, true, true);
60              int listSize = cl.size();
61              while (foundTdiff == false) {
62                  if (listSize == 0) {
63                      break;
64                  }
65                  // choose the highest seed energy cluster in the time window
66                  for (Cluster cc : cl) {
67                      clT = ClusterUtilities.getSeedHitTime(cc);
68                      if (clT > parameter[1] - jj * 10 && clT < parameter[2] + jj * 10) {
69                          foundTdiff = true;
70                          selectedClT = clT;
71                          seedX = ClusterUtilities.findSeedHit(cc).getIdentifierFieldValue("ix");
72                          seedY = ClusterUtilities.findSeedHit(cc).getIdentifierFieldValue("iy");
73                          break;
74                      }// end if
75                  }// end for
76                  jj++;
77              }// end while
78          }// end if has clusters
79  
80          double jitter = 0;
81          // read in rf time
82          if (event.hasCollection(GenericObject.class, "RFHits")) {
83              List<GenericObject> rfTimes = event.get(GenericObject.class, "RFHits");
84  
85              rfT = rfTimes.get(0).getDoubleVal(1);
86              jitter = ((rfT - selectedClT + 400 * 2.004) % 2.004) - 1.002;
87              selectedClT += jitter;
88          }
89  
90          triggerTime.add(new TriggerTime(selectedClT, seedX, seedY));
91          event.put("TriggerTime", triggerTime, TriggerTime.class, 1);
92  
93      }// end process event
94  
95  }// end driver
