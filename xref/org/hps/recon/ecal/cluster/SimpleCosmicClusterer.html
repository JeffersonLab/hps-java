17  /**
18   * <p>
19   * This algorithm clusters an input list of CalorimeterHits using an iterative nearest neighbor algorithm.
20   * <p>
21   * There is a set of cuts applied on the initial cluster list that includes the following defaults:
22   * <ul>
23   * <li>must have at least 3 hits total in the cluster</li>
24   * <li>must have at least three rows of crystals with hits</li>
25   * <li>must have no more than 2 hits in each row of crystals</li>
26   * </ul>
27   * <p>
28   * This algorithm does not cluster across the beam gap.  Separate clusters are made for the top
29   * and bottom sets of crystals.
30   */
31  public class SimpleCosmicClusterer extends AbstractClusterer {
32  
33      protected int minClusterSize;
34      protected int minRows;
35      protected int maxHitsPerRow;
36      
37      /**
38       * Default constructor that sets cut names and default values.
39       */
40      SimpleCosmicClusterer() {
41          super(new String[] { "minClusterSize", "minRows", "maxHitsPerRow" }, new double[] { 3, 3, 2 });
42      }
43      
44      /**
45       * Initialize some instance variables from cut settings.
46       */
47      public void initialize() {
48          this.minClusterSize = (int) getCuts().getValue("minClusterSize");
49          this.minRows = (int) getCuts().getValue("minRows");
50          this.maxHitsPerRow = (int) getCuts().getValue("maxHitsPerRow");
51      }       
52          
53      /**
54       * This method implements the cosmic clustering algorithm which uses a iterative 
55       * nearest neighbor algorithm.  It uses each hit as a seed and tries to then
56       * extend the cluster by adding neighboring hits.  Clustered hits are added to a list 
57       * which is checked so that they are not reused.
58       * @param hitList The input hit list of hits.
59       * @param event The current LCSim event.
60       * @return A list of calorimeter hits that can be turned into clusters.
61       */
62      public List<Cluster> createClusters(EventHeader event, List<CalorimeterHit> hitList) {
63          
64          // Create empty list of clusters which are just lists of hits.
65          //List<List<CalorimeterHit>> clusterList = new ArrayList<List<CalorimeterHit>>();
66          List<Cluster> clusterList = new ArrayList<Cluster>();
67          
68          // Create map of IDs to hits for convenience.
69          Map<Long, CalorimeterHit> hitMap = ClusterUtilities.createHitMap(hitList);
70          
71          // Create list of hits that are clusterable, which is initially all of them.
72          Set<CalorimeterHit> clusterable = new HashSet<CalorimeterHit>();
73          clusterable.addAll(hitList);
74          
75          // Loop over all hits in the map.
76          for (CalorimeterHit hit : hitList) {
77                                                 
78              // Is hit clusterable?
79              if (!clusterable.contains(hit)) {
80                  // Continue to next hit.
81                  continue;
82              }
83              
84              // Create list for clustering this hit.
85              List<CalorimeterHit> clusterHits = new ArrayList<CalorimeterHit>();
86              
87              // Set of hits whose neighbors have not been checked yet.
88              LinkedList<CalorimeterHit> uncheckedHits = new LinkedList<CalorimeterHit>();
89              uncheckedHits.add(hit);
90  
91              // While there are still unchecked hits in the cluster.
92              while (uncheckedHits.size() > 0) {
93                                  
94                  // Get the first hit and add it to the cluster.
95                  CalorimeterHit clusterHit = uncheckedHits.removeFirst();
96                  
97                  // Add hit to the cluster.
98                  clusterHits.add(clusterHit);
99                                          
100                 // Remove the hit from the clusterable list.
101                 clusterable.remove(clusterHit);                                    
102                                 
103                 // Loop over the neighbors and add IDs with hits to the unchecked list.
104                 for (Long neighborHitID : ClusterUtilities.findNeighborHitIDs(ecal, clusterHit, hitMap)) {
105                     CalorimeterHit neighborHit = hitMap.get(neighborHitID);
106                     if (clusterable.contains(neighborHit)) {
107                         uncheckedHits.add(neighborHit);
108                     }
109                 }                                                
110             }
111             
112             if (clusterHits.size() >= this.minClusterSize) {
113                 Cluster cluster = ClusterUtilities.createBasicCluster(clusterHits);
114                 clusterList.add(cluster);
115             }
116         }
117         
118         // Apply cuts on the cluster list that was generated.
119         List<Cluster> selectedClusters = applyCuts(clusterList);
120         
121         return selectedClusters;
122     }    
123 
124     /**
125      * Apply selection cuts to cluster list and return filtered list.
126      * @param clusterList The input hit lists. 
127      * @return The hit lists that passed the cuts.
128      */
129     protected List<Cluster> applyCuts(List<Cluster> clusterList) {
130         List<Cluster> selectedClusters = new ArrayList<Cluster>();
131         for (Cluster cluster  : clusterList) {            
132             Map<Integer, Set<CalorimeterHit>> rowMap = new HashMap<Integer, Set<CalorimeterHit>>();            
133             for (CalorimeterHit hit : cluster.getCalorimeterHits()) {
134                 int row = ecal.getDetectorElement().getIdentifierHelper().getValue(new Identifier(hit.getCellID()), "iy");
135                 if (rowMap.get(row) == null) {
136                     rowMap.put(row, new HashSet<CalorimeterHit>());
137                 }
138                 rowMap.get(row).add(hit);
139             }
140             if (rowMap.size() >= this.minRows) {
141                 boolean okay = true;
142                 rowMapLoop: for (Entry<Integer, Set<CalorimeterHit>> entries : rowMap.entrySet()) {
143                     if (entries.getValue().size() > this.maxHitsPerRow) {
144                         okay = false;
145                         break rowMapLoop;
146                     }
147                 }
148                 if (okay) {
149                     selectedClusters.add(cluster);
150                 }
151             }
152         }
153         return selectedClusters;
154     }
155 
156     @Override
157     public ClusterType getClusterType() {
158         return ClusterType.SIMPLE_COSMIC;
159     }    
160 }
