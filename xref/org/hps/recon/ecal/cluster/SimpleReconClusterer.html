3   import java.util.ArrayList;
4   import java.util.Collections;
5   import java.util.HashMap;
6   import java.util.List;
7   import java.util.Map;
8   
9   import org.lcsim.event.CalorimeterHit;
10  import org.lcsim.event.Cluster;
11  import org.lcsim.event.EventHeader;
12  import org.lcsim.event.base.BaseCluster;
13  
14  /**
15   * <p>
16   * This clustering algorithm creates clusters from an input CalorimeterHit collection.
17   * <p>
18   * It uses the basic Inner Calorimeter (IC) clustering algorithm as described in 
19   * <a href="https://misportal.jlab.org/ul/Physics/Hall-B/clas/viewFile.cfm/2005-001.pdf?documentId=6">CLAS Note 2004-040</a>.
20   * <p> 
21   * Hits are assigned to a cluster with the largest seed hit energy.  Time information is not used, and multiple hits in the same 
22   * crystal are not handled correctly so an exception is throw if this occurs.  An optional cut can be applied to discard hits
23   * with a time that is too far from t0.
24   */
25  public class SimpleReconClusterer extends AbstractClusterer {
26      
27      double minEnergy;
28      double minTime;
29      double timeWindow;
30      boolean useTimeCut;
31  
32      /**
33       * Initialize the algorithm with default cuts.
34       */
35      SimpleReconClusterer() {
36          super(new String[] { "minEnergy", "minTime", "timeWindow" }, new double[] { 0.001, 0.0, 20.0 });
37      }
38  
39      public void initialize() {
40          // Setup class variables from cuts.
41          minEnergy = getCuts().getValue("minEnergy");
42          minTime = getCuts().getValue("minTime");
43          timeWindow = getCuts().getValue("timeWindow");
44      }
45  
46      public List<Cluster> createClusters(EventHeader event, List<CalorimeterHit> hitCollection) {
47  
48          // New Cluster list to be added to event.
49          List<Cluster> clusters = new ArrayList<Cluster>();
50  
51          // Create a Calorimeter hit list in each event, then sort with highest energy first
52          ArrayList<CalorimeterHit> sortedHitList = new ArrayList<CalorimeterHit>(hitCollection.size());
53          for (CalorimeterHit h : hitCollection) {
54              // reject hits below the energy cut
55              if (h.getCorrectedEnergy() < this.minEnergy) {
56                  continue;
57              }
58              // if time cut is being used, reject hits outside the time window
59              if (useTimeCut && (h.getTime() < minTime || h.getTime() > minTime + timeWindow)) {
60                  continue;
61              }
62              sortedHitList.add(h);
63          }
64  
65          // sort the list, highest energy first
66          Collections.sort(sortedHitList, Collections.reverseOrder(new CalorimeterHit.CorrectedEnergyComparator()));
67  
68          // map from seed hit to cluster
69          Map<CalorimeterHit, BaseCluster> seedToCluster = new HashMap<CalorimeterHit, BaseCluster>();
70  
71          // Quick Map to access hits from cell IDs
72          Map<Long, CalorimeterHit> idToHit = new HashMap<Long, CalorimeterHit>();
73  
74          // map from each hit to its cluster seed
75          Map<CalorimeterHit, CalorimeterHit> hitToSeed = new HashMap<CalorimeterHit, CalorimeterHit>();
76  
77          // Fill Map with cell ID and hit
78          for (CalorimeterHit hit : sortedHitList) {
79              if (idToHit.containsKey(hit.getCellID())) {
80                  //System.out.println(this.getName() + ": multiple CalorimeterHits in same crystal");
81                  // Make this an error for now.
82                  throw new RuntimeException("Multiple CalorimeterHits found in same crystal.");
83              }
84              idToHit.put(hit.getCellID(), hit);
85          }
86  
87          for (CalorimeterHit hit : sortedHitList) {
88              CalorimeterHit biggestSeed = null;
89  
90              for (Long neighbor : neighborMap.get(hit.getCellID())) {
91                  CalorimeterHit neighborHit = idToHit.get(neighbor);
92                  if (neighborHit == null) {
93                      continue;
94                  }
95  
96                  if (neighborHit.getCorrectedEnergy() > hit.getCorrectedEnergy()) {
97                      CalorimeterHit neighborSeed = hitToSeed.get(neighborHit);
98                      if (biggestSeed == null || neighborHit.getCorrectedEnergy() > biggestSeed.getCorrectedEnergy()) {
99                          biggestSeed = neighborSeed;
100                     }
101                 }
102             }
103             if (biggestSeed == null) { // if no neighbors had more energy than this hit, this hit is a seed
104                 hitToSeed.put(hit, hit);
105                 BaseCluster cluster = createBasicCluster();
106                 clusters.add(cluster);
107                 seedToCluster.put(hit, cluster);
108             } else {
109                 hitToSeed.put(hit, biggestSeed);
110             }
111         }
112 
113         // add all hits to clusters
114         for (CalorimeterHit hit : sortedHitList) {
115             CalorimeterHit seed = hitToSeed.get(hit);
116             BaseCluster cluster = seedToCluster.get(seed);
117             cluster.addHit(hit);
118         }
119         
120         return clusters;
121     }
122 
123     @Override
124     public ClusterType getClusterType() {
125         return ClusterType.SIMPLE_RECON;
126     }
127     
128     public void setUseTimeCut(boolean useTimeCut) {
129         this.useTimeCut = useTimeCut;
130     }
131 }
