1   package org.hps.recon.ecal.cluster;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.lcsim.event.Cluster;
7   import org.lcsim.event.EventHeader;
8   import org.lcsim.event.base.BaseCluster;
9   import org.lcsim.lcio.LCIOConstants;
10  import org.lcsim.util.Driver;
11  
12  /**
13   * Copy a {@link org.lcsim.event.Cluster} collection to a new collection via the 
14   * {@link org.lcsim.event.base.BaseCluster} class's copy constructor.
15   */
16  public class CopyClusterCollectionDriver extends Driver {
17  
18      /**
19       * The input collection name.
20       */
21      private String inputCollectionName = null;
22  
23      /**
24       * The output collection name.
25       */
26      private String outputCollectionName = null;
27  
28      /**
29       * Set to <code>true</code> to store hits in the output clusters.
30       */
31      private boolean storeHits = true;
32  
33      /**
34       * Basic no argument constructor.
35       */
36      public CopyClusterCollectionDriver() {
37      }
38  
39      /**
40       * Copy clusters to a new collection (list).
41       *
42       * @param clusters the input cluster list
43       * @return the output cluster collection from copying the input list
44       */
45      public List<Cluster> copyClusters(final List<Cluster> clusters) {
46          final List<Cluster> newCollection = new ArrayList<Cluster>();
47          for (final Cluster cluster : clusters) {
48              // Use the base class's copy constructor to make a new cluster from the input.
49              final BaseCluster newCluster = new BaseCluster(cluster);
50              
51              // Turn off automatic property calculation.
52              newCluster.setNeedsPropertyCalculation(false);
53  
54              // Add new cluster to output collection.
55              newCollection.add(newCluster);
56          }
57          return newCollection;
58      }
59  
60      /**
61       * Process an event, copying the input collection to the output collection.
62       *
63       * @param event the LCSim event
64       */
65      @Override
66      public void process(final EventHeader event) {
67  
68          // Check if output collection already exists in event which is an error.
69          if (event.hasItem(outputCollectionName)) {
70              throw new RuntimeException("collection " + outputCollectionName + " already exists in event");
71          }
72  
73          // Get the input collection.
74          final List<Cluster> inputClusterCollection = event.get(Cluster.class, inputCollectionName);
75  
76          // Copy to the output collection.
77          final List<Cluster> outputClusterCollection = this.copyClusters(inputClusterCollection);
78  
79          // Copy input collection's flags.
80          int flags = event.getMetaData(inputClusterCollection).getFlags();
81  
82          // Set the store hits bit from this Driver's settings.
83          if (storeHits) {
84              flags = flags | 1 << LCIOConstants.CLBIT_HITS;
85          } else {
86              flags = flags & 0 << LCIOConstants.CLBIT_HITS;
87          }
88  
89          // Put the copied collection into the event.
90          event.put(outputCollectionName, outputClusterCollection, Cluster.class, flags);
91      }
92  
93      /**
94       * Set the input collection name (source).
95       *
96       * @param inputCollectionName the input collection name
97       */
98      public void setInputCollectionName(final String inputCollectionName) {
99          this.inputCollectionName = inputCollectionName;
100     }
101 
102     /**
103      * Set the output collection name (target).
104      *
105      * @param outputCollectionName the output collection name
106      */
107     public void setOutputCollectionName(final String outputCollectionName) {
108         this.outputCollectionName = outputCollectionName;
109     }
110 
111     /**
112      * Set to <code>true</code> to store hits in the output clusters.
113      * 
114      * @param storeHits <code>true</code> to store hits; <code>false</code> to not store hits
115      */
116     public void setStoreHits(final boolean storeHits) {
117         this.storeHits = storeHits;
118     }
119 
120     /**
121      * Start of data hook which will make sure required arguments are set properly.
122      */
123     @Override
124     public void startOfData() {
125         if (inputCollectionName == null) {
126             throw new RuntimeException("inputCollectionName was never set");
127         }
128         if (outputCollectionName == null) {
129             throw new RuntimeException("outputCollectionName was never set");
130         }
131         if (inputCollectionName.equals(outputCollectionName)) {
132             throw new IllegalArgumentException("inputCollectionName and outputCollectionName are the same");
133         }
134     }
135 }
