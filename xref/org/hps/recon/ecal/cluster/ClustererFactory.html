1   package org.hps.recon.ecal.cluster;
2   
3   import org.lcsim.conditions.ConditionsManager;
4   
5   /**
6    * <p>
7    * This is a convenience class for creating specific clustering algorithms via their name
8    * in the package <code>org.hps.recon.ecal.cluster</code>.  They must implement the {@link Clusterer} 
9    * interface.
10   * <p>
11   * If the name does not match one in the clustering package, the factory will attempt to create
12   * a new class instance, assuming that the string is a canonical class name.  It then checks if 
13   * this class implements the {@link Clusterer} interface and will throw an error if it does not.
14   * 
15   * @see Clusterer
16   * @see CTPClusterer
17   * @see DualThresholdCosmicClusterer
18   * @see GTPClusterer
19   * @see GTPOnlineClusterer
20   * @see LegacyClusterer 
21   * @see NearestNeighborClusterer
22   * @see ReconClusterer
23   * @see SimpleReconClusterer
24   * @see SimpleCosmicClusterer 
25   */
26  public final class ClustererFactory {
27      
28      /**
29       * We don't want this class to be instantiated.
30       */
31      private ClustererFactory() {        
32      }
33      
34      /**
35       * Create a clustering algorithm with a set of cuts.
36       * @param name The name of the clustering algorithm.
37       * @param cuts The set of cuts (can be null).
38       * @return The clustering algorithm.
39       * @throws IllegalArgumentException if there is no Clusterer found with name.
40       */
41      public static Clusterer create(String name, double[] cuts) {
42          Clusterer clusterer;
43          if (LegacyClusterer.class.getSimpleName().equals(name)) {
44              clusterer = new LegacyClusterer();
45          } else if (SimpleReconClusterer.class.getSimpleName().equals(name)) {
46              clusterer = new SimpleReconClusterer();
47          } else if (ReconClusterer.class.getSimpleName().equals(name)) {
48              clusterer = new ReconClusterer();
49          } else if (NearestNeighborClusterer.class.getSimpleName().equals(name)) {
50              clusterer = new NearestNeighborClusterer();
51          } else if (DualThresholdCosmicClusterer.class.getSimpleName().equals(name)) {
52              clusterer = new DualThresholdCosmicClusterer();
53          } else if (SimpleCosmicClusterer.class.getSimpleName().equals(name)) {
54              clusterer = new SimpleCosmicClusterer();
55          } else if (GTPOnlineClusterer.class.getSimpleName().equals(name)) {
56              clusterer = new GTPOnlineClusterer();
57          } else if (GTPClusterer.class.getSimpleName().equals(name)) {
58              clusterer = new GTPClusterer();
59          } else if (CTPClusterer.class.getSimpleName().equals(name)) {
60              clusterer = new CTPClusterer();
61          } else {
62              // Try to instantiate a Clusterer object from the name argument, assuming it is a canonical class name.
63              try {
64                  clusterer = fromCanonicalClassName(name);
65                  if (!clusterer.getClass().isAssignableFrom(Clusterer.class)) {
66                      throw new IllegalArgumentException("The class " + name + " does not implement the Clusterer interface.");
67                  }
68              } catch (Exception e) {
69                  // Okay nothing worked, so we have a problem!
70                  throw new IllegalArgumentException("Unknown Clusterer type " + name + " cannot be instantiated.", e);
71              }
72          }
73          
74          // Register the Clusterer for notification when conditions change.
75          ConditionsManager.defaultInstance().addConditionsListener(clusterer);        
76          
77          // Set cuts if they were provided.
78          if (cuts != null) {
79              clusterer.getCuts().setValues(cuts);
80          }
81          return clusterer;
82      }
83      
84      /**
85       * Create a clustering algorithm with default cut values.
86       * This is just a simple wrapper to {@link #create(String, double[])}.
87       * @param name The name of the clustering algorithm.
88       * @return The clustering algorithm.
89       */
90      public static Clusterer create(String name) {
91          return create(name, null);
92      }
93      
94      /**
95       * Attempt to create a Clusterer object from the canonical class name.
96       * @param canonicalName
97       * @return The new Clusterer object.
98       * @throw IllegalArgumentException if the class does not implement the Clusterer interface.
99       */
100     private static Clusterer fromCanonicalClassName(String canonicalName) {
101         Clusterer clusterer = null;
102         try {
103             Class<?> clustererClass = Class.forName(canonicalName);
104             Object object = clustererClass.newInstance();
105             if (!(object instanceof Clusterer)) { 
106                 throw new IllegalArgumentException("The class " + canonicalName + " does not implement the Clusterer interface.");
107             } 
108             clusterer = (Clusterer) object;
109         } catch (Exception e) {
110             throw new RuntimeException(e);
111         }
112         return clusterer;
113     }
114 }
