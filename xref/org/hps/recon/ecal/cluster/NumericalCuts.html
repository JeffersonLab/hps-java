1   package org.hps.recon.ecal.cluster;
2   
3   import java.util.Map;
4   
5   /**
6    * <p>
7    * This is an interface for accessing the numerical cut values 
8    * of a clustering algorithm by index or name.
9    * <p>
10   * The default units are generally GeV for energy and nanoseconds 
11   * for time, though this is not in any way modeled by this class. 
12   */
13  public interface NumericalCuts {
14  
15      /**
16       * Set all the cut values from an array.
17       * @param values The cut values.
18       */
19      void setValues(double[] values);
20      
21      /**
22       * Get the cut values.
23       * @return The cut values array.
24       */
25      double[] getValues();
26      
27      /**
28       * Get a cut setting by name.
29       * @param name The name of the cut.
30       * @return The cut value.
31       */
32      double getValue(String name);
33      
34      /**
35       * Get a cut value by index.
36       * @param index The index of the cut.
37       * @return The cut value from the index.
38       */
39      double getValue(int index);
40           
41      /**
42       * Get the names of the cuts.
43       * @return The names of the cuts.
44       */
45      String[] getNames();
46      
47      /**
48       * Set the value of a cut by index.
49       * @param index The index of the cut.
50       * @param value The value of the cut.
51       */
52      void setValue(int index, double value);
53  
54      /**
55       * Set a cut value by name.
56       * @param name The name of the cut.
57       * @param value The value of the cut.
58       */
59      void setValue(String name, double value);
60      
61      /**
62       * True if using the default cuts.
63       * @return True if using default cuts.
64       */
65      boolean isDefaultValues();
66      
67      /**
68       * Get the default cuts.
69       * @return The default cuts.
70       */
71      double[] getDefaultValues();
72      
73      /**
74       * Set the cut values from a map of keys to values.
75       * @param valueMap The cut values as a map.
76       */
77      void setValues(Map<String, Double> valueMap);            
78  }
