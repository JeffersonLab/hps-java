19      /**
20       * Calculates the position of each cluster with no correction for particle type, 
21       * as documented in HPS Note 2014-001.
22       * @param cluster The input Cluster.
23       */
24      public void calculateProperties(Cluster cluster) {
25          
26          reset();
27          
28          // Calculate the position of the cluster.
29          calculatePosition(cluster);
30      }
31  
32      private void calculatePosition(Cluster cluster) {
33          final double w0 = 3.1;
34          double xCl = 0.0;
35          // calculated cluster y position
36          double yCl = 0.0;
37          double eNumX = 0.0;
38          double eNumY = 0.0;
39          double eDen = 0.0;
40          List<CalorimeterHit> clusterHits = cluster.getCalorimeterHits();
41          for (CalorimeterHit hit : clusterHits) {
42              
43              EcalCrystal crystal = (EcalCrystal) hit.getDetectorElement();
44              Hep3Vector crystalPosition = crystal.getPositionFront();
45              
46              double wi = Math.max(0.0, (w0 + Math.log(hit.getCorrectedEnergy() / cluster.getEnergy())));
47              eNumX += wi * (crystalPosition.x() / 10.0);
48              eNumY += wi * (crystalPosition.y() / 10.0);
49              eDen += wi;
50  
51          } // end for iteration through clusterHits
52  
53          xCl = eNumX / eDen;
54          yCl = eNumY / eDen;
55  
56          position[0] = xCl * 10.0;// mm
57          position[1] = yCl * 10.0;// mm        
58          CalorimeterHit seedHit = clusterHits.get(0);
59          this.position[2] = ((EcalCrystal)seedHit.getDetectorElement()).getPositionFront().z();
60      }    
61  }
