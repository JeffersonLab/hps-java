1   package org.hps.recon.ecal.cluster;
2   
3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   
6   import org.hps.record.daqconfig.ConfigurationManager;
7   import org.hps.record.daqconfig.GTPConfig;
8   import org.lcsim.event.EventHeader;
9   
10  /**
11   * Class <code>GTPOnlineClusterDriver</code> is an implementation of
12   * the <code>ClusterDriver</code> class that defines employs the readout
13   * variant of the GTP hardware clustering algorithm. Specifics on the
14   * behavior of this algorithm can be found in its documentation.<br/>
15   * <br/>
16   * <code>GTPOnlineClusterDriver</code> allows for all of the variable
17   * settings used by the GTP algorithm to be defined. It also can be
18   * set to "verbose" mode, where it will output detailed information on
19   * each event and the cluster forming process. This is disabled by
20   * default, but can be enabled for debugging purposes.<br/>
21   * <br/>
22   * Lastly, <code>GTPOnlineClusterDriver</code> can be set to draw its
23   * settings from the <code>ConfigurationManager</code> static class,
24   * which reads and stores settings extracted directly from EvIO data.
25   * This option is disabled by default, and can be activated with the
26   * method <code>setUseDAQConfig</code>. When enabled, no clusters will
27   * be generated until <code>ConfigurationManager</code> reads a config
28   * event. This requires that the driver <code>DAQConfigDriver</code>
29   * be included in the driver chain.<br/>
30   * <br/>
31   * <code>GTPOnlineClusterDriver</code> is designed for use on hardware
32   * readout data or Monte Carlo formatted in this style. It can not be
33   * used for 2-ns beam bunch formatted data. <code>GTPClusterDriver</code>
34   * should be used for this data instead.
35   * 
36   * @see GTPOnlineClusterer
37   * @see ConfigurationManager
38   * @see org.hps.record.daqconfig.DAQConfigDriver
39   */
40  public class GTPOnlineClusterDriver extends ClusterDriver {
41      /** An instance of the clustering algorithm object for producing
42       * cluster objects. */
43      private final GTPOnlineClusterer gtp;
44      /** Indicates whether the <code>ConfigurationManager</code> object
45       * should be used for clustering settings or not. */
46      private boolean useDAQConfig = false;
47      
48      /**
49       * Initializes a clustering driver. This implements the readout
50       * variant of the hardware GTP algorithm, as defined in the class
51       * <code>GTPClusterer</code>.
52       * @see GTPOnlineClusterer
53       */
54      public GTPOnlineClusterDriver() {
55          // Instantiate the clusterer.
56          clusterer = ClustererFactory.create("GTPOnlineClusterer");
57          gtp = (GTPOnlineClusterer) clusterer;
58          
59          // Track the DAQ configuration status.
60          ConfigurationManager.addActionListener(new ActionListener() {
61              @Override
62              public void actionPerformed(ActionEvent e) {
63                  // If DAQ configuration settings should be used, then
64                  // update the clusterer.
65                  if(useDAQConfig) {
66                      // Get the GTP settings.
67                      GTPConfig config = ConfigurationManager.getInstance().getGTPConfig();
68                      
69                      // Send the DAQ configuration settings to the clusterer.
70                      gtp.setSeedLowThreshold(config.getSeedEnergyCutConfig().getLowerBound());
71                      gtp.setWindowAfter(config.getTimeWindowAfter());
72                      gtp.setWindowBefore(config.getTimeWindowBefore());
73                      
74                      // Print the updated settings.
75                      logSettings();
76                  }
77              }
78          });
79      }
80      
81      /**
82       * Processes an an <code>EventHeader</code> object to generate
83       * clusters. Events will not be processed if <code>UseDAQConfig</code>
84       * is <code>true</code> unless the <code>ConfigurationManager</code>
85       * static class has received a DAQ configuration from the event
86       * stream and is initialized. Driver <code>DAQConfigDriver</code>
87       * must be in the driver chain for this to occur.
88       * @see org.hps.record.daqconfig.DAQConfigDriver
89       */
90      @Override
91      public void process(EventHeader event) {
92          // Only process an event if either the DAQ configuration is not
93          // in use or if it has been initialized.
94          if((useDAQConfig && ConfigurationManager.isInitialized()) || !useDAQConfig) {
95              super.process(event);
96          }
97      }
98      
99      /**
100      * Outputs the clusterer settings at driver initialization, assuming
101      * <code>setVerbose</code> is set to <code>true</code>.
102      */
103     @Override
104     public void startOfData() {
105         // VERBOSE :: Output the driver settings.
106         if(gtp.isVerbose()) { logSettings(); }
107     }
108     
109     /**
110      * Sets the minimum seed energy needed for a hit to be considered
111      * for forming a cluster. This is the seed energy lower bound trigger
112      * cut and is in units of GeV.
113      * @param seedEnergyThreshold - The minimum cluster seed energy in
114      * GeV.
115      */
116     public void setSeedEnergyThreshold(double seedEnergyThreshold) {
117         getClusterer().getCuts().setValue("seedThreshold", seedEnergyThreshold);
118         gtp.setSeedLowThreshold(seedEnergyThreshold);
119     }
120     
121     /**
122      * Sets the number of clock-cycles to include in the clustering
123      * window before the seed hit. One clock-cycle is four nanoseconds.
124      * This defines the first half of the temporal hit inclusion window.
125      * The temporal hit verification window is defined as
126      * <code>max(windowAfter, windowBefore) * 2) + 1</code>.
127      * @param cyclesBefore - The length of the clustering window before
128      * the seed hit in clock cycles.
129      */
130     public void setWindowBefore(int cyclesBefore) {
131         gtp.setWindowBefore(cyclesBefore);
132     }
133     
134     /**
135      * Sets the number of clock-cycles to include in the clustering
136      * window after the seed hit. One clock-cycle is four nanoseconds.
137      * This defines the latter half of the temporal hit inclusion window.
138      * The temporal hit verification window is defined as
139      * <code>max(windowAfter, windowBefore) * 2) + 1</code>.
140      * @param cyclesAfter - The length of the clustering window after
141      * the seed hit in clock cycles.
142      * @see GTPOnlineClusterer
143      */
144     public void setWindowAfter(int cyclesAfter) {
145         gtp.setWindowAfter(cyclesAfter);
146     }
147     
148     /**
149      * Sets whether the clusterer should output diagnostic text or not.
150      * @param verbose - <code>true</code> indicates that the clusterer
151      * should output diagnostic text and <code>false</code> that it
152      * should not.
153      */
154     public void setVerbose(boolean verbose) {
155         gtp.setVerbose(verbose);
156     }
157     
158     /**
159      * Sets whether GTP settings should be drawn from the EvIO data
160      * DAQ configuration or read from the steering file. If this is
161      * set to <code>true</code>, no clusters will be generated until
162      * the static class <code>ConfigurationManager</code> has received
163      * a DAQ settings event and initialized. If this class is not part
164      * of the driver chain, then no clusters will ever be created.
165      * @param state - <code>true</code> means that DAQ configuration
166      * will be used and <code>false</code> that it will not.
167      * @see org.hps.record.daqconfig.DAQConfigDriver
168      */
169     public void setUseDAQConfig(boolean state) {
170         useDAQConfig = state;
171     }
172     
173     /**
174      * Outputs the current GTP settings to the terminal.
175      */
176     private void logSettings() {
177         // Print the cluster driver header.
178         System.out.println();
179         System.out.println();
180         System.out.println("======================================================================");
181         System.out.println("=== GTP Readout Clusterer Settings ===================================");
182         System.out.println("======================================================================");
183         
184         // Output the driver settings.
185         System.out.printf("Seed Energy Threshold :: %.3f GeV%n", gtp.getSeedLowThreshold());
186         System.out.printf("Time Window (Before)  :: %.0f ns%n", gtp.getWindowBefore());
187         System.out.printf("Time Window (After)   :: %.0f ns%n", gtp.getWindowAfter());
188     }
189 }
