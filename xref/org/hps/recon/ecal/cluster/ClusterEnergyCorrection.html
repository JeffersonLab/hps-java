42      /**
43       * Calculates energy correction based on cluster raw energy and particle
44       * type as per <a href=
45       * "https://misportal.jlab.org/mis/physics/hps_notes/index.cfm?note_year=2014"
46       * >HPS Note 2014-001</a>
47       *
48       * @param ecal
49       * @param pdg Particle id as per PDG
50       * @param rawEnergy Raw Energy of the cluster (sum of hits with shared hit
51       * distribution)
52       * @param xpos
53       * @param ypos
54       * @param isMC
55       * @return Corrected Energy
56       */
57      public static double computeCorrectedEnergy(HPSEcal3 ecal, int pdg, double rawEnergy, double xpos, double ypos,
58              boolean isMC) {
59          // distance to beam gap edge
60          double r = ClusterCorrectionUtilities.computeYDistanceFromEdge(ecal, xpos, ypos);
61  
62          // Eliminates corrections at outermost edges to negative cluster energies
63          // 66 for positrons, 69 is safe for electrons and photons
64          if (r > 66) {
65              r = 66;
66          }
67  
68          if (isMC) {
69              switch (pdg) {
70                  case 11:
71                      // electron
72                      return computeCorrectedEnergy(r, rawEnergy, par0MC_em, par1MC_em, par2MC_em);
73                  case -11:
74                      // positron
75                      return computeCorrectedEnergy(r, rawEnergy, par0MC_ep, par1MC_ep, par2MC_ep);
76                  case 22:
77                      // photon
78                      return computeCorrectedEnergy(r, rawEnergy, par0MC_p, par1MC_p, par2MC_p);
79                  default:
80                      // unknown
81                      return rawEnergy;
82              }
83          } else {
84              switch (pdg) {
85                  case 11:
86                      // electron
87                      return computeCorrectedEnergy(r, rawEnergy, par0_em, par1_em, par2_em);
88                  case -11:
89                      // positron
90                      return computeCorrectedEnergy(r, rawEnergy, par0_ep, par1_ep, par2_ep);
91                  case 22:
92                      // photon
93                      return computeCorrectedEnergy(r, rawEnergy, par0_p, par1_p, par2_p);
94                  default:
95                      // unknown
96                      return rawEnergy;
97              }
98          }
99      }

101     /**
102      * Calculates the energy correction to a cluster given the variables from
103      * the fit as per <a href=
104      * "https://misportal.jlab.org/mis/physics/hps_notes/index.cfm?note_year=2014"
105      * >HPS Note 2014-001</a> Note that this is correct as there is a typo in
106      * the formula print in the note.
107      *
108      * @param rawEnergy Raw energy of the cluster
109      * @param A,B,C from fitting in note
110      * @return Corrected Energy
111      */
112     private static double computeCorrectedEnergy(double y, double rawEnergy, double varA, double varB[],
113             double varC[]) {
114         int ii = y < varB[0] ? 2 : 5;
115         double corrEnergy = rawEnergy / (varA / rawEnergy
116                 + (varB[1] - varB[ii] * Math.exp(-(y - varB[ii + 1]) * varB[ii + 2])) / (Math.sqrt(rawEnergy))
117                 + (varC[1] - varC[ii] * Math.exp(-(y - varC[ii + 1]) * varC[ii + 2])));
118         return corrEnergy;
119     }

121     /**
122      * Calculate the corrected energy for the cluster.
123      *
124      * @param cluster The input cluster.
125      * @return The corrected energy.
126      */
127     public static final double calculateCorrectedEnergy(HPSEcal3 ecal, Cluster cluster, boolean isMC) {
128         double rawE = cluster.getEnergy();
129         return computeCorrectedEnergy(ecal, cluster.getParticleId(), rawE, cluster.getPosition()[0],
130                 cluster.getPosition()[1], isMC);
131     }

145     /**
146      * Calculate the corrected energy and set on the cluster.
147      *
148      * @param cluster The input cluster.
149      */
150     public static final void setCorrectedEnergy(HPSEcal3 ecal, BaseCluster cluster, boolean isMC) {
151         double correctedEnergy = calculateCorrectedEnergy(ecal, cluster, isMC);
152         if (isMC) {
153             correctedEnergy += ClusterCorrectionUtilities.calcNoise(correctedEnergy);
154         }
155         cluster.setEnergy(correctedEnergy);
156     }
