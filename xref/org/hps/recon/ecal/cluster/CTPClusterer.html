22  /**
23   * Creates clusters from CalorimeterHits in the HPSEcal detector.
24   *
25   * The clustering algorithm is from JLab Hall B 6 GeV DVCS Trigger Design doc.
26   */
27  public class CTPClusterer extends AbstractClusterer {
28  
29      IDDecoder dec;
30      
31      Set<Long> clusterCenters = null;
32      Map<Long, Double> hitSums = null;
33      Map<Long, CalorimeterHit> hitMap = null;
34  
35      // The time period in which clusters may be formed. A negative value means that all hits
36      // will always be used in cluster finding, regardless of the time difference between them.
37      double clusterWindow = -1;
38      
39      // The minimum energy needed for a hit to be considered.
40      double addEMin = 0;
41      
42      CTPClusterer() {
43          super(new String[] { "addEMin", "clusterWindow"}, new double[] { 0., -1. });
44      }
45      
46      public void initialize() {
47          addEMin = getCuts().getValue("addEMin");
48          clusterWindow = getCuts().getValue("clusterWindow");
49      }

51      @Override
52      public void conditionsChanged(ConditionsEvent event) {
53          super.conditionsChanged(event);
54          
55          // Get the decoder for the ECal IDs.
56          dec = ecal.getIDDecoder();    
57                  
58          // Make set of valid cluster centers.
59          // Exclude edge crystals as good cluster centers.
60          clusterCenters = new HashSet<Long>();
61          for (Long cellID : neighborMap.keySet()) {
62              boolean isValidCenter = true;
63              Set<Long> neighbors = neighborMap.get(cellID);
64              for (Long neighborID : neighbors) {
65                  Set<Long> neighborneighbors = new HashSet<Long>();
66                  neighborneighbors.addAll(neighborMap.get(neighborID));
67                  neighborneighbors.add(neighborID);
68  
69                  if (neighborneighbors.containsAll(neighbors)) {
70                      isValidCenter = false;
71                      break;
72                  }
73              }
74              if (isValidCenter) {
75                  clusterCenters.add(cellID);
76              }
77          }
78      }

80      @Override
81      public List<Cluster> createClusters(EventHeader event, List<CalorimeterHit> hits) {
82  
83          // Define a list of clusters to be filled.
84          List<Cluster> clusters;
85  
86          // If there is a cluster window, run the cluster window code. A cluster window is a time
87          // period in nanoseconds during which hits can be applied to the same cluster.
88          if (clusterWindow >= 0) {
89              // Create priority queues. These are sorted by the time variable associated with each hit.
90              PriorityQueue<CalorimeterHit> futureHits = new PriorityQueue<CalorimeterHit>(10, new TimeComparator());
91              PriorityQueue<CalorimeterHit> pastHits = new PriorityQueue<CalorimeterHit>(10, new TimeComparator());
92  
93              // Initialize the cluster list variable.
94              clusters = new ArrayList<Cluster>();
95  
96              // Populate the list of unprocessed hits with the calorimeter hits. These will then be sorted
97              // by time, from first to last, automatically by the priority queue.
98              for (CalorimeterHit hit : hits) {
99                  if (hit.getRawEnergy() > addEMin) {
100                     futureHits.add(hit);
101                 }
102             }
103 
104             // We process the unprocessed hits...
105             while (!futureHits.isEmpty()) {
106                 // Move the first occurring hit from the unprocessed list to the processed list.
107                 CalorimeterHit nextHit = futureHits.poll();
108                 pastHits.add(nextHit);
109 
110                 // Add any hits that occurred at the same time as the hit we just added to the processed list.
111                 while (!futureHits.isEmpty() && futureHits.peek().getTime() == nextHit.getTime()) {
112                     pastHits.add(futureHits.poll());
113                 }
114 
115                 // Remove hits that happened earlier than the cluster window period.
116                 while (pastHits.peek().getTime() < nextHit.getTime() - clusterWindow) {
117                     pastHits.poll();
118                 }
119 
120                 // Calculate the cluster energy for each crystal. This should be the
121                 // total energy for the 3x3 crystal collection sorrounding the center
122                 // crystal.
123                 sumHits(pastHits);
124 
125                 // Choose which crystal is the appropriate cluster crystal.
126                 clusters.addAll(createClusters());
127             }
128             // If there is no cluster window, then all the hits in the event are visible simultaneously.
129         } else {
130             // Calculate the cluster energy of each crystal.
131             sumHits(hits);
132             // Generate the clusters.
133             clusters = createClusters();
134         }
135         
136         return clusters;
137     }

139     private void sumHits(Collection<CalorimeterHit> hits) {
140         // Store the latest hit on each crystal in a map for later reference in
141         // the clustering algorithm.
142         hitMap = new HashMap<Long, CalorimeterHit>();
143         // Store the cluster energy for each crystal. Cluster energy represents
144         // the total energy of the 3x3 crystal set.
145         hitSums = new HashMap<Long, Double>();
146 
147         // Loop over the active calorimeter hits to compute the cluster energies.
148         for (CalorimeterHit hit : hits) {
149             // Make a hit map for quick lookup by ID.
150             hitMap.put(hit.getCellID(), hit);
151 
152             // Get the cell ID for the current crystal's neighbors.
153             Set<Long> neighbors = neighborMap.get(hit.getCellID());
154 
155             // If there are no neighbors, something is rather wrong.
156             if (neighbors == null) {
157                 throw new RuntimeException("Oops!  Set of neighbors is null!");
158             }
159 
160             // Store the energy of the current calorimeter hit.
161             Double hitSum;
162 
163             // We are only interested in this crystal's cluster energy if it is
164             // a valid cluster crystal. Edge crystals are not allowed to be clusters,
165             // so these are ignored.
166             if (clusterCenters.contains(hit.getCellID())) {
167                 // Check if an energy has been assigned to the crystal.
168                 hitSum = hitSums.get(hit.getCellID());
169 
170                 // If not, then the crystal's cluster energy is equal to this hit's energy.
171                 if (hitSum == null) {
172                     hitSums.put(hit.getCellID(), hit.getRawEnergy());
173                 }
174                 // Otherwise, add the energy of this hit to the total crystal cluster energy.
175                 else {
176                     hitSums.put(hit.getCellID(), hitSum + hit.getRawEnergy());
177                 }
178             }
179 
180             // Loop over neighbors to add the current hit's energy to the neighbor's
181             // cluster energy.
182             for (Long neighborId : neighbors) {
183                 // If the crystal is not an edge crystal, ignore its hit energy.
184                 if (!clusterCenters.contains(neighborId)) {
185                     continue;
186                 }
187 
188                 // Get the cluster energy of the neighboring crystals.
189                 hitSum = hitSums.get(neighborId);
190 
191                 // If the neighbor crystal has no cluster energy, then set the
192                 // cluster energy to the current hit's energy.
193                 if (hitSum == null) {
194                     hitSums.put(neighborId, hit.getRawEnergy());
195                 }
196                 // Otherwise, add the hit's energy to the neighbor's cluster energy.
197                 else {
198                     hitSums.put(neighborId, hitSum + hit.getRawEnergy());
199                 }
200             }
201         }
202     }

204     private List<Cluster> createClusters() {
205         // Create a list of clusters to be added to the event,
206         List<Cluster> clusters = new ArrayList<Cluster>();
207 
208         // We examine each crystal with a non-zero cluster energy.
209         for(Long possibleCluster : hitSums.keySet()) {
210             // Get the luster energy for the crystal this hit is assocaite with.
211             Double thisSum = hitSums.get(possibleCluster);
212 
213             // Get neighboring crystals' IDs.
214             Set<Long> neighbors = neighborMap.get(possibleCluster);
215 
216             // If there are no neighbors, throw an error.
217             if (neighbors == null) {
218                 throw new RuntimeException("Oops!  Set of neighbors is null!");
219             }
220 
221             // Get the x/y position of the hit's associated crystal.
222             dec.setID(possibleCluster);
223             int x1 = dec.getValue("ix");
224             int y1 = dec.getValue("iy");
225 
226             // Store whether it is a valid cluster or not.
227             boolean isCluster = true;
228 
229             // Check to see if any of the crystal's neighbors preclude the current crystal
230             // from being a proper cluster. The cluster crystal should have the highest
231             // energy among its neighbors
232             for (Long neighborId : neighbors) {
233                 // Get the x/y position of the neighbor's associated crystal.
234                 dec.setID(neighborId);
235                 int x2 = dec.getValue("ix");
236                 int y2 = dec.getValue("iy");
237 
238                 // If the neighbor's energy value does not exist, we don't need to perform
239                 // any additional checks for this neighbor. A crystal with no energy can
240                 // not be the center of a cluster.
241                 Double neighborSum = hitSums.get(neighborId);
242                 if (neighborSum == null) {
243                     continue;
244                 }
245 
246                 // If the neighbor's energy value is greater than this crystal's value,
247                 // then this crystal is not the cluster and we may terminate the check.
248                 if (neighborSum > thisSum) {
249                     isCluster = false;
250                     break;
251                 }
252                 // If the crystals each have the same energy, we choose the crystal that
253                 // is closest to the electron side of the detector. If both crystals are
254                 // equally close, we choose the crystal closest to the beam gap. If the
255                 // neighbor fits these parameters better, this is not a crystal and we
256                 // may skip any further checks.
257                 else if (neighborSum.equals(thisSum) && (x1 > x2 || (x1 == x2 && Math.abs(y1) < Math.abs(y2)))) {
258                     isCluster = false;
259                     break;
260                 }
261             }
262 
263             // If the crystal was not invalidated by the any of the neighboring crystals,
264             // then it is a cluster crystal and should be processed.
265             if (isCluster) {
266                 // Make a list to store the hits that are part of this cluster.
267                 List<CalorimeterHit> hits = new ArrayList<CalorimeterHit>();
268 
269                 // Store the time at which the cluster occurred.
270                 double clusterTime = Double.NEGATIVE_INFINITY;
271 
272                 // Get the last hit on this crystal.
273                 CalorimeterHit hit = hitMap.get(possibleCluster);
274 
275                 // If the hit exists, add it to the list of associated hits.
276                 if (hit != null) {
277                     hits.add(hit);
278 
279                     // If the latest hit's time is later than the current cluster time,
280                     // set the cluster time to the latest hit's time.
281                     if (hit.getTime() > clusterTime) {
282                         clusterTime = hit.getTime();
283                     }
284                 }
285 
286                 // Add all of the neighboring crystals to the cluster, if they have a
287                 // hit associated with them. Crystals with no hits are not actually part
288                 // of a cluster.
289                 for (Long neighborId : neighbors) {
290                     hit = hitMap.get(neighborId);
291                     if (hit != null) {
292                         hits.add(hit);
293                         if (hit.getTime() > clusterTime) {
294                             clusterTime = hit.getTime();
295                         }
296                     }
297                 }
298 
299                 // Generate a new cluster seed hit from the above results.
300                 CalorimeterHit seedHit = (BaseCalorimeterHit)CalorimeterHitUtilities.create(0.0, clusterTime, possibleCluster, hits.get(0).getMetaData());
301 
302                 // Generate a new cluster from the seed hit.
303                 BaseCluster cluster = createBasicCluster();
304                 cluster.addHit(seedHit);
305                 cluster.setPosition(seedHit.getDetectorElement().getGeometry().getPosition().v());
306                 cluster.setNeedsPropertyCalculation(false);
307                 // Populate the cluster with each of the chosen neighbors.
308                 for (CalorimeterHit clusterHit : hits) {
309                     cluster.addHit(clusterHit);
310                 }
311                 // Add the cluster to the cluster list.
312                 clusters.add(cluster);
313             }
314         }
315 
316         // Return the list of clusters.
317         return clusters;
318     }

320     static class TimeComparator implements Comparator<CalorimeterHit> {
321         // Compare by time with the earlier coming before the later.
322         public int compare(CalorimeterHit o1, CalorimeterHit o2) {
323             if (o1.getTime() == o2.getTime()) {
324                 return 0;
325             } else {
326                 return (o1.getTime() > o2.getTime()) ? 1 : -1;
327             }
328         }
329     }
