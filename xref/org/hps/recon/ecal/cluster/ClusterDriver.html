18   * This is a {@link org.lcsim.util.Driver} that creates ECAL {@link org.lcsim.event.Cluster} 
19   * collections using the {@link Clusterer} interface.

21   * A specific clustering engine can be created with the {@link #setClustererName(String)} method
22   * which will use a factory to create it by name.  The cuts of the {@link Clusterer}
23   * can be set generically with the {@link #setCuts(double[])} method.  
24   * 
25   * @see Clusterer
26   * @see NumericalCuts
27   * @see org.lcsim.event.Cluster
28   * @see org.lcsim.util.Driver

172     public void detectorChanged(Detector detector) {
173         Subdetector subdetector = detector.getSubdetector("Ecal");
174         if (subdetector == null) {
175             throw new RuntimeException("There is no subdetector called " + ecalName + " in the detector.");
176         }
177         if (!(subdetector instanceof HPSEcal3)) {
178             throw new RuntimeException("Ther subdetector " + ecalName + " does not have the right type.");
179         }
180         ecal = (HPSEcal3) subdetector;
181         neighborMap = ecal.getNeighborMap();
182     }

184     /**
185      * Perform start of job initialization.
186      */
187     public void startOfData() {
188         if (this.clusterer == null) {
189             throw new RuntimeException("The clusterer was never initialized.");
190         }
191         if (this.cuts != null) {
192             this.clusterer.getCuts().setValues(cuts);
193         } 
194         StringBuffer sb = new StringBuffer();
195         sb.append("Clusterer has the following cuts ...");
196         sb.append('\n');
197         for (int cutIndex = 0; cutIndex < clusterer.getCuts().getValues().length; cutIndex++) {
198             sb.append(this.clusterer.getCuts().getNames()[cutIndex] + " = " + this.clusterer.getCuts().getValue(cutIndex));
199             sb.append('\n');
200         } 
201         getLogger().config(sb.toString());
202         this.clusterer.initialize();
203     }

205     /**
206      * This method implements the default clustering procedure based on input parameters.
207      */
208     public void process(EventHeader event) {    
209                        
210         if (event.hasCollection(CalorimeterHit.class, inputHitCollectionName)) {       
211             List<CalorimeterHit> hits = event.get(CalorimeterHit.class, inputHitCollectionName);
212             getLogger().fine("input hit collection " + inputHitCollectionName + " has " + hits.size() + " hits");
213             
214             // Cluster the hits, copying the list from the event in case the clustering algorithm modifies it.
215             List<Cluster> clusters = clusterer.createClusters(event, new ArrayList<CalorimeterHit>(hits));
216             
217             if (clusters == null) {
218                 throw new RuntimeException("The clusterer returned a null list from its createClusters method.");
219             }
220             if (clusters.isEmpty() && this.skipNoClusterEvents) {
221                 getLogger().finer("skipping event with no clusters");
222                 throw new NextEventException();
223             }
224             if (event.hasCollection(Cluster.class, this.outputClusterCollectionName)) {
225                 this.getLogger().severe("There is already a cluster collection called " + this.outputClusterCollectionName);
226                 throw new RuntimeException("Cluster collection already exists in event.");
227             }
228             int flags = 0;
229             if (this.storeHits) {
230                 flags = 1 << LCIOConstants.CLBIT_HITS;
231             }
232             if (!clusters.isEmpty() || this.createEmptyClusterCollection) {
233                 if (sortHits) {
234                     // Sort the hits.
235                     ClusterUtilities.sortReconClusterHits(clusters);
236                 }                                                           
237                 getLogger().finer("writing " + clusters.size() + " clusters to collection " + outputClusterCollectionName);
238                 event.put(outputClusterCollectionName, clusters, Cluster.class, flags);
239                 if (!this.writeClusterCollection) {
240                     getLogger().finer("Collection is set to transient and will not be persisted.");
241                     event.getMetaData(clusters).setTransient(true);
242                 }
243                 
244                 if (validateClusters) {
245                     // Perform basic validation checks.
246                     this.validateClusters(event);
247                 }
248             }
249         } else {
250 //            getLogger().info("The input hit collection " + this.inputHitCollectionName + " is missing from the event.");
251             if (this.raiseErrorNoHitCollection) {
252                 throw new RuntimeException("The expected input hit collection is missing from the event.");
253             }
254         }
255     }

267     /**
268      * Perform basic validation of the cluster output collection, including checking
269      * that the cluster collection was created, clusters are not null, 
270      * none of the clustered hits are null, and each hit exists in the input 
271      * hit collection.
272      * @param event The LCSim event.
273      */
274     void validateClusters(EventHeader event) {
275         if (!event.hasCollection(Cluster.class, outputClusterCollectionName)) {
276             throw new RuntimeException("Cluster collection " + outputClusterCollectionName + " is missing.");
277         }
278         List<Cluster> clusters = event.get(Cluster.class, outputClusterCollectionName);
279         List<CalorimeterHit> inputHitCollection = event.get(CalorimeterHit.class, inputHitCollectionName);
280         for (int clusterIndex = 0; clusterIndex < clusters.size(); clusterIndex++) {
281             getLogger().finest("checking cluster " + clusterIndex);
282             Cluster cluster = clusters.get(clusterIndex);
283             if (clusters.get(clusterIndex) == null) {
284                 throw new RuntimeException("The Cluster at index " + clusterIndex + " is null.");
285             }
286             List<CalorimeterHit> clusterHits = cluster.getCalorimeterHits();
287             getLogger().finest("cluster has " + clusterHits.size() + " hits");
288             for (int hitIndex = 0; hitIndex < clusterHits.size(); hitIndex++) {
289                 getLogger().finest("checking cluster hit " + hitIndex);                              
290                 CalorimeterHit clusterHit = clusterHits.get(hitIndex);
291                 if (clusterHit == null) {
292                     throw new RuntimeException("The CalorimeterHit at index " + hitIndex + " in the cluster is null.");
293                 }
294                 if (!inputHitCollection.contains(clusterHit)) {
295                     getLogger().severe("The CalorimeterHit at index " + hitIndex + " with ID " + clusterHit.getIdentifier().toHexString() + " is missing from the input hit collection.");
296                     printHitIDs(inputHitCollection);
297                     throw new RuntimeException("The CalorimeterHit at index " + hitIndex + " in the cluster is missing from the input hit collection.");
298                 }
299             }
300         }
301     }
302     
303     void printHitIDs(List<CalorimeterHit> hits) {        
304         StringBuffer buffer = new StringBuffer();
305         buffer.append("hit IDs");
306         buffer.append('\n');
307         for (CalorimeterHit hit : hits) {            
308             buffer.append(hit.getIdentifier().toHexString());
309             buffer.append('\n');
310         }
311         getLogger().finest(buffer.toString());
312     }
