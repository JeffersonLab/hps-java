3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.Map;
6   import java.util.Set;
7   
8   import org.lcsim.event.CalorimeterHit;
9   import org.lcsim.event.Cluster;
10  import org.lcsim.event.EventHeader;
11  import org.lcsim.event.base.BaseCluster;
12  
13  /**
14   * This is a simple (example) nearest-neighbor clustering algorithm.
15   */
16  public class NearestNeighborClusterer extends AbstractClusterer {
17  
18      double minHitEnergy = 0;
19      double minSize = 2;
20      
21      NearestNeighborClusterer() {
22          super(new String[] {"minHitEnergy", "minSize" }, new double[] { 0.0, 2.0 });
23      }
24      
25      public void initialize() {
26          minHitEnergy = getCuts().getValue("minHitEnergy");
27          minSize = getCuts().getValue("minSize");
28      }
29      
30      @Override
31      public List<Cluster> createClusters(EventHeader event, List<CalorimeterHit> hits) {
32          Map<Long, CalorimeterHit> hitMap = ClusterUtilities.createHitMap(hits);
33          // New Cluster list to be added to event.
34          List<Cluster> clusters = new ArrayList<Cluster>();
35  
36          while (!hitMap.isEmpty()) {
37              Long k = hitMap.keySet().iterator().next();
38              CalorimeterHit hit = hitMap.get(k);
39              BaseCluster cluster = this.createCluster(hitMap, hit);
40              //done with this cluster, let's compute some shape properties
41              if (cluster.getSize() >= minSize) {
42                  cluster.calculateProperties();
43                  clusters.add(cluster);
44              }
45          }
46          return clusters;
47      }
48         
49      /**
50       *
51       * @param map A Map of CalorimeterHit keyed on CellID, Hits added to this
52       * cluster are removed from the map
53       * @param hit A CalorimeterHit representing a hit crystal
54       * @param threshold The energy threshold below which a hit crystal should
55       * NOT be added to the cluster
56       * @param neighborMap The map of HPS crystal neighbors.
57       */    
58      BaseCluster createCluster(Map<Long, CalorimeterHit> map, CalorimeterHit hit) {
59          BaseCluster cluster = createBasicCluster();
60          // start by adding this hit to the cluster
61          cluster.addHit(hit);
62          // remove this hit from the map so it can't be used again
63          map.remove(hit.getCellID());
64          List<CalorimeterHit> hits = cluster.getCalorimeterHits();
65          //  loop over the hits in the cluster and add all its neighbors
66          // note that hits.size() grows as we add cells, so we recursively find neighbors of neighbors
67          for (int i = 0; i < hits.size(); ++i) {
68              CalorimeterHit c = hits.get(i);
69              // Get neighbor crystal IDs.
70              Set<Long> neighbors = neighborMap.get(c.getCellID());
71              // loop over all neighboring cell Ids
72              for (Long neighborId : neighbors) {
73                  // Find the neighbor hit in the event if it exists.
74                  CalorimeterHit neighborHit = map.get(neighborId);
75                  // Was this neighbor cell hit?
76                  if (neighborHit != null) {
77                      // if so, does it meet or exceed threshold?
78                      if (neighborHit.getRawEnergy() >= minHitEnergy) {
79                          cluster.addHit(neighborHit);
80                      }
81                      //remove this hit from the map so it can't be used again
82                      map.remove(neighborHit.getCellID());
83                  }
84              }
85          }
86          return cluster;
87      }
88      
89      public ClusterType getClusterType() {
90          return ClusterType.NN;
91      }
92  }
