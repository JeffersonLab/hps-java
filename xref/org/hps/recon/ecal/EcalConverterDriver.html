13  public class EcalConverterDriver extends Driver {
14      
15      Detector detector = null;
16  
17      String rawCollectionName;
18      String ecalReadoutName = "EcalHits";
19      String ecalCollectionName = "EcalCorrectedHits";
20      int flags;
21      double scale = 1.0;
22  //    double pedestal = 0.0;
23      double period = 4.0;
24      double dt = 0.0;
25  
26      public EcalConverterDriver() {
27          flags = 0;
28          flags += 1 << LCIOConstants.CHBIT_LONG; //store position
29          flags += 1 << LCIOConstants.RCHBIT_ID1; //store cell ID
30      }
31  
32  //    public void setPedestal(double pedestal) {
33  //        this.pedestal = pedestal;
34  //    }
35      public void setScale(double scale) {
36          this.scale = scale;
37      }
38  
39      public void setEcalCollectionName(String ecalCollectionName) {
40          this.ecalCollectionName = ecalCollectionName;
41      }
42  
43      public void setRawCollectionName(String rawCollectionName) {
44          this.rawCollectionName = rawCollectionName;
45      }
46  
47      @Override
48      public void startOfData() {
49          if (ecalCollectionName == null) {
50              throw new RuntimeException("The parameter ecalCollectionName was not set!");
51          }
52      }
53      
54      @Override
55      public void detectorChanged(Detector detector) {
56          this.detector = detector;
57      }
58  
59      @Override
60      public void process(EventHeader event) {
61          if (event.hasCollection(RawCalorimeterHit.class, rawCollectionName)) {
62              // Get the list of ECal hits.
63              List<RawCalorimeterHit> hits = event.get(RawCalorimeterHit.class, rawCollectionName);
64  
65              ArrayList<CalorimeterHit> newHits = new ArrayList<CalorimeterHit>();
66  
67              for (RawCalorimeterHit hit : hits) {
68                  newHits.add(HitDtoA(hit));
69              }
70  
71              event.put(ecalCollectionName, newHits, CalorimeterHit.class, flags, ecalReadoutName);
72          }
73      }
74  
75  //    private int AtoD(double amplitude, long cellID) {
76  //        return (int) Math.round(amplitude / scale);
77  //    }
78  
79      private double DtoA(int amplitude, long cellID) {
80          return scale * amplitude;
81      }
82  
83      private CalorimeterHit HitDtoA(RawCalorimeterHit hit) {
84          double energy = DtoA(hit.getAmplitude(), hit.getCellID());
85          return CalorimeterHitUtilities.create(energy, period * hit.getTimeStamp() + dt, hit.getCellID());
86      }
