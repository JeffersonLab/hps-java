97      @Override
98      public void process(EventHeader event) {
99  
100         ArrayList<CalorimeterHit> hodoHits = new ArrayList<CalorimeterHit>();
101 
102         if (event.hasCollection(RawTrackerHit.class, rawCollectionName)) {
103 
104             int flags = 0;
105             flags += 1 << LCIOConstants.RCHBIT_TIME; // store hit time
106             flags += 1 << LCIOConstants.RCHBIT_LONG; // store hit position; this flag has no effect for RawCalorimeterHits
107 
108             // ======= Getting list of Mode1 hits from the event ======
109             List<RawTrackerHit> hits = event.get(RawTrackerHit.class, rawCollectionName);
110 
111             // ======= Loop over hits, and reconstruct energies and times for all hits =====
112             for (RawTrackerHit hit : hits) {
113 
114                 // Getting the cellID of the hit
115                 final long cellID = hit.getCellID();
116 
117                 double ped = converter.getPedestal(event, cellID);
118 
119                 //System.out.println(Arrays.toString(hodo_identifiers));
120                 // ====== Get Number of Threshold Crossings ==========
121 
122                 ArrayList<Integer> thr_crosings = converter.FindThresholdCrossings(hit, ped);
123 
124                 // ===== For now we will calculate coarse time, which is the threshold crossing sample time.
125                 // ===== Later will implement the mode7 time                                
126                 ArrayList<CalorimeterHit> hits_in_this_channel = converter.getCaloHits(hit, thr_crosings, ped);
127                 
128                 // Propagate the detector element information to these found_hits so it can be used later.
129                 for(CalorimeterHit found_hit: hits_in_this_channel) {
130                     if(!isMC)
131                         found_hit.setDetectorElement(hit.getDetectorElement());
132                     else {
133                         // Set detector element for MC hits
134                         int[] identifier = converter.getHodoIdentifiers(cellID);
135                         GeometryId id_geometry = new GeometryId(helper, new int[]{subDetector.getSystemID(), identifier[0], identifier[1], identifier[2], 0});
136                         long id_det = id_geometry.encode();                                                 
137                         IDetectorElementContainer srch = subDetector.getDetectorElement().findDetectorElement(new Identifier(id_det));
138                         found_hit.setDetectorElement(srch.get(0));
139                     }                                        
140                 }
141 
142                 hodoHits.addAll(hits_in_this_channel);
143 
144             }
145 
146             // ====== Adding hodo hits to as a Calorimeter hit to the event.
147             event.put(hodoCollectionName, hodoHits, CalorimeterHit.class, flags, hodoReadoutName);
148 
149             //== == == == == == == == == == == == == == == == == == == == == == 
150             //  Now start to do a clustering, which is combining hits together from hits
151             //  that are from the same time, but are in opposite holes. They should also be 
152             //  in time coincidence
153             // =========== The EVENT::Cluster  Object is not a perfect collection for storing hodo clusters
154             // =========== since it doesn't provide ready methods for getting, ix, iy, layer
155             // =========== therefore Hodo clusters will be stored in a genertic object.
156             // === There will be 5 genericCollections for ix, iy, layer,, Energy and time.
157             // === All these collections will have same size, which is the number of all hits.
158             // === i-th value of each collection shows the corresponding value of the i-th hit
159             int[] cl_ix = new int[]{};           // This array contains ix of hits
160             int[] cl_iy = new int[]{};           // THis array contains iy of hits
161             int[] cl_layer = new int[]{};        // THis array contains layer of hits
162             double[] cl_Energy = new double[]{}; // THis array contains the energy of hits
163             double[] cl_Time = new double[]{};   // THis array contains the Time of hits
164             int[] cl_detid = new int[]{};
165             // ======== Defining GenericObjects for Hodo Hit components, ix, it etc
166             SimpleGenericObject generic_cl_ix = new SimpleGenericObject();
167             SimpleGenericObject generic_cl_iy = new SimpleGenericObject();
168             SimpleGenericObject generic_cl_layer = new SimpleGenericObject();
169             SimpleGenericObject generic_cl_energy = new SimpleGenericObject();
170             SimpleGenericObject generic_cl_time = new SimpleGenericObject();
171             SimpleGenericObject generic_cl_detid = new SimpleGenericObject();
172             ArrayList<Integer> paired = new ArrayList<Integer>();
173             
174             for (int i = 0; i < hodoHits.size(); i++) {
175 
176                 // Check if this hit is already paired, if so, then let's pass to the next hit
177                 if (paired.contains(i))
178                     continue;
179 
180                 CalorimeterHit this_hit = hodoHits.get(i);
181                
182                 // ====== These tiles are readout with a single PMT channel, and therefore for these 
183                 // ====== tiles cluster and the hit are identical 
184                 int hit_ix = this_hit.getIdentifierFieldValue("ix");
185                 int hit_iy = this_hit.getIdentifierFieldValue("iy");
186                 int hit_ilayer=this_hit.getIdentifierFieldValue("layer");
187                 int hit_ihole=this_hit.getIdentifierFieldValue("hole");
188                 
189                 if (hit_ix == 0 || hit_ix == 4) {
190 
191                     cl_ix = ArrayUtils.add(cl_ix, hit_ix);
192                     cl_iy = ArrayUtils.add(cl_iy, hit_iy);
193                     cl_layer = ArrayUtils.add(cl_layer, hit_ilayer);
194                     cl_Energy = ArrayUtils.add(cl_Energy, this_hit.getRawEnergy());
195                     cl_Time = ArrayUtils.add(cl_Time, this_hit.getTime());
196                     cl_detid = ArrayUtils.add(cl_detid, (int)this_hit.getDetectorElement().getIdentifier().getValue());
197                     
198                     continue;
199                 }
200 
201                 boolean pair_found = false;
202                 
203 
204                 for (int j = i + 1; j < hodoHits.size(); j++) {
205                     
206                     CalorimeterHit that_hit = hodoHits.get(j);
207                     // ====== These tiles are readout with a single PMT channel, and therefore for these 
208                     // ====== tiles cluster and the hit are identical 
209                     int hit_jx = that_hit.getIdentifierFieldValue("ix");
210                     int hit_jy = that_hit.getIdentifierFieldValue("iy");
211                     int hit_jlayer=that_hit.getIdentifierFieldValue("layer");
212                     int hit_jhole=that_hit.getIdentifierFieldValue("hole");
213                     
214 
215                     // Check if this hit is already paired, if so, then let's pass to the next hit
216                     if (paired.contains(j))
217                         continue;
218 
219                     if (hit_ix == hit_jx && hit_iy == hit_jy && hit_ilayer == hit_jlayer
220                             && Math.abs(this_hit.getTime() - that_hit.getTime()) < HodoConstants.cl_hit_dtMax && 
221                             ( hit_ihole * hit_jhole == -1)) {
222 
223                         pair_found = true;
224                         paired.add(j);
225 
226                         cl_ix = ArrayUtils.add(cl_ix, hit_ix);
227                         cl_iy = ArrayUtils.add(cl_iy, hit_iy);
228                         cl_layer = ArrayUtils.add(cl_layer, hit_ilayer);
229                         cl_detid = ArrayUtils.add(cl_detid, (int)this_hit.getDetectorElement().getIdentifier().getValue());
230 
231                         double energy = HodoConstants.cl_Esum_scale * (this_hit.getRawEnergy() + that_hit.getRawEnergy()) / 2.;
232 
233                         // === SOme kind of an semiarbitrary function. This makes the hit time to be closer to the one with the highest energy
234                         double time = this_hit.getTime() + ( that_hit.getTime() - this_hit.getTime()) * that_hit.getRawEnergy() / (this_hit.getRawEnergy() + that_hit.getRawEnergy());
235 
236                         cl_Energy = ArrayUtils.add(cl_Energy, energy);
237                         cl_Time = ArrayUtils.add(cl_Time, time);
238 
239                         continue;
240                     }
241 
242                 }
243 
244                 // ===== In case if no pair is found, then make a cluster, which will have identical energy and time as this hit
245                 if (!pair_found) {
246                     cl_ix = ArrayUtils.add(cl_ix, hit_ix);
247                     cl_iy = ArrayUtils.add(cl_iy, hit_iy);
248                     cl_layer = ArrayUtils.add(cl_layer, hit_ilayer);
249                     cl_Energy = ArrayUtils.add(cl_Energy, this_hit.getRawEnergy());
250                     cl_Time = ArrayUtils.add(cl_Time, this_hit.getTime());
251                     cl_detid = ArrayUtils.add(cl_detid, (int)this_hit.getDetectorElement().getIdentifier().getValue());
252                 }
253 
254             }
255 
256             // ====== At this point clustering is finished, we are ready to write them into data =====
257             generic_cl_ix.setIntValues(cl_ix);
258             generic_cl_iy.setIntValues(cl_iy);
259             generic_cl_layer.setIntValues(cl_layer);
260             generic_cl_energy.setDoubleValues(cl_Energy);
261             generic_cl_time.setDoubleValues(cl_Time);
262             generic_cl_detid.setIntValues(cl_detid);
263 
264             // ====== In order to write generic object in the file, 1st we should add generic objects
265             // ====== in the List, so we will add above genericObjects into the list below
266             List<SimpleGenericObject> hodo_cl_list = new ArrayList();
267 
268             hodo_cl_list.add(generic_cl_ix);
269             hodo_cl_list.add(generic_cl_iy);
270             hodo_cl_list.add(generic_cl_layer);
271             hodo_cl_list.add(generic_cl_energy);
272             hodo_cl_list.add(generic_cl_time);
273             hodo_cl_list.add(generic_cl_detid);
274 
275             // Writing HodoCluster data into the event
276             event.put("HodoGenericClusters", hodo_cl_list, SimpleGenericObject.class, 0);
277 
278         }
279     }
