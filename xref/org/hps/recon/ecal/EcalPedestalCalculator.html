87      @Override
88      public void endOfData() {
89          if(batch)
90              return;
91          Console cc = System.console();
92          if (cc == null) {
93              throw new IllegalStateException("Could not access system console.");
94          }
95  
96          System.err.println("\n\n\n***************************************************************\n");
97          String userInput="";
98          String outputFilePrefix="";
99          userInput=cc.readLine("Enter filename prefix, or just press RETURN ...");
100         if (userInput==null || userInput.length()==0 || userInput=="") {
101             String home=System.getenv().get("HOME");
102             outputFilePrefix = home+"/EcalPedestals/EcalPedestalCalculator_"+runNumber+"_";
103         } else {
104             outputFilePrefix = userInput;
105         }
106 
107         String date = new SimpleDateFormat("yyyy-MM-dd-hh-mm").format(new Date());
108         outputFilePrefix += date+"_";
109 
110         if (writeFileForDAQ) writeFileForDAQ(outputFilePrefix);
111         if (writeFileForDB)  writeFileForDB(outputFilePrefix);
112 
113         System.out.println("\n\n***************************************************************\n");
114         userInput=cc.readLine(String.format("Enter 'YES' to write conditions database for run range [%s,%s] ...",runNumber,runNumberMax));
115         System.out.println("***********"+userInput+"********");
116         boolean uploadToDB=false;
117         if (userInput!=null && userInput.equals("YES")) {
118             userInput=cc.readLine("Really?");
119             if (userInput!=null && userInput.equals("YES")) {
120                 uploadToDB=true;
121             }
122         }
123         if (uploadToDB) {
124             try {
125                 uploadToDB();
126             } catch (DatabaseObjectException | ConditionsObjectException | SQLException e) {
127                 throw new RuntimeException("Error uploading to database.", e);
128             }
129         } else {
130             System.out.println("!!!!!!!!!!!!!!!!!!!!!!! Not Writing Database !!!!!!!!!!!!!!!!!!!!!!!!!!");
131         }
132     }

134     @Override

136 
137         if (nDetectorChanges++ > 1) {
138             throw new RuntimeException("No Detector Change Allowed.");
139         }

142         ecalConditions = conditionsManager.getEcalConditions();
143 
144         aida.tree().cd("/");
145         for (EcalChannel cc : ecalConditions.getChannelCollection()) {
146             aida.histogram1D(getHistoName(cc),181,19.5,200.5);
147         }

155     private void uploadToDB() throws DatabaseObjectException, ConditionsObjectException, SQLException {
156         System.out.println(String.format("Uploading new pedestals to the database, runMin=%d, runMax=%d, tag=%s ....",
157                 runNumber,runNumberMax,dbTag));
158 
159         EcalCalibrationCollection calibrations = new EcalCalibrationCollection();
160         TableMetaData tableMetaData = conditionsManager.findTableMetaData(dbTableName);
161         calibrations.setTableMetaData(tableMetaData);
162 
163         for (int cid = 1; cid <= 442; cid++) {
164             EcalChannel cc = findChannel(cid);
165             IHistogram1D hh = aida.histogram1D(getHistoName(cc));
166             if (hh.entries() < minimumStats) {
167                 System.err.println("Insufficient Statistics, Not writing to database.  (channel_id="+cid+").");
168                 return;
169             }
170             calibrations.add(new EcalCalibration(cid,
171                     Double.valueOf(dbNumberFormat.format(hh.mean())),
172                     Double.valueOf(dbNumberFormat.format(hh.rms()))));
173         }
174 
175         int collectionId = -1;
176         try {
177             collectionId = conditionsManager.getCollectionId(calibrations, "loaded by EcalPedestalCalculator");
178         } catch (Exception e) {
179             throw new RuntimeException(e);
180         }
181         calibrations.setCollectionId(collectionId);
182 
183         System.err.println("CollectionID:  "+collectionId);
184 
185         calibrations.insert();
186         ConditionsRecord conditionsRecord = new ConditionsRecord(
187                 calibrations.getCollectionId(), runNumber, runNumberMax, dbTableName, dbTableName, 
188                 "Generated by EcalPedestalCalculator from Run #"+runNumber, dbTag);
189         conditionsRecord.insert();
190 
191     }

193     private void writeFileForDB(String outputFilePrefix) {
194         System.out.println("\nWriting pedestal file for HPS Conditions Database:\n"
195                 + outputFilePrefix + filenameDB);
196         FileWriter wout = null;
197         try {
198             wout = new FileWriter(outputFilePrefix + filenameDB);
199             wout.write("ecal_channel_id pedestal noise\n");
200             for (int cid = 1; cid <= 442; cid++) {
201                 EcalChannel cc = findChannel(cid);
202                 IHistogram1D hh = aida.histogram1D(getHistoName(cc));
203                 wout.write(String.format("%3d %7.3f %7.3f\n",cid,hh.mean(),
204                         hh.rms() * Math.sqrt(nSamples)));
205             }
206         } catch (IOException ee) {
207             throw new RuntimeException("Error writing file.",ee);
208         } finally {
209             if (wout != null) {
210                 try {
211                     wout.close();
212                 } catch (IOException ee) {
213                     ee.printStackTrace();
214                 }
215             }
216         }
217     }

219     private void writeFileForDAQ(String outputFilePrefix) {
220         System.out.println("\nWriting pedestal files for Hall-B DAQ:\n" + outputFilePrefix
221                 + filenamesDAQ[0] + "\n" + outputFilePrefix + filenamesDAQ[1]);
222         FileWriter[] wout = { null, null };
223         try {
224             for (int crate = 1; crate <= 2; crate++) {
225                 wout[crate - 1] = new FileWriter(outputFilePrefix + filenamesDAQ[crate - 1]);
226                 for (int slot = 3; slot <= 20; slot++) {
227                     for (int chan = 0; chan < 16; chan++) {
228                         if (slot > 9 && slot < 14)
229                             continue;
230                         if (slot == 20 && chan > 12)
231                             continue;
232                         EcalChannel cc = findChannel(crate,slot,chan);
233                         IHistogram1D hh = aida.histogram1D(getHistoName(cc));
234                         wout[crate - 1].write(String.format("%2d %2d %7.3f %7.3f\n",slot,chan,
235                                 hh.mean(),hh.rms() * Math.sqrt(nSamples)));
236                     }
237                 }
238             }
239         } catch (IOException ee) {
240             throw new RuntimeException("Error writing file.",ee);
241         } finally {
242             if (wout[0] != null || wout[1] != null) {
243                 try {
244                     if (wout[0] != null)
245                         wout[0].close();
246                     if (wout[1] != null)
247                         wout[1].close();
248                 } catch (IOException ee) {
249                     ee.printStackTrace();
250                 }
251             }
252         }
253     }

255     @Override
256     protected void process(EventHeader event) {
257         runNumber = event.getRunNumber();
258         if (event.hasCollection(RawCalorimeterHit.class,rawCollectionName)) {
259             if (event.hasCollection(LCRelation.class,extraDataRelationsName)) {
260                 for (LCRelation rel : event.get(LCRelation.class,extraDataRelationsName)) {
261                     RawCalorimeterHit hit = (RawCalorimeterHit) rel.getFrom();
262                     GenericObject extraData = (GenericObject) rel.getTo();
263                     fillHisto(event,hit,extraData);
264                 }
265             }
266         }
267         else{ 
268             for(RawTrackerHit hit : event.get(RawTrackerHit.class,rawCollectionName)){
269                 fillHisto(event, hit);
270             }
271         }
272 
273     }

275     private void fillHisto(EventHeader event, RawTrackerHit hit) {
276 
277         double ped = 0;
278         int N = 4;
279         for(int i = 0; i< N; i++){
280             ped += hit.getADCValues()[i]/N;
281         }
282         
283         EcalChannel cc = findChannel(hit);
284         if (cc == null) {
285             System.err.println("Hit doesn't correspond to ecalchannel.");
286             return;
287         }
288         aida.histogram1D(getHistoName(cc)).fill(ped);
289     }
290     
291     private void fillHisto(EventHeader event, RawCalorimeterHit hit, GenericObject mode7data) {
292 
293         int min = 0, max = 0;
294         if(mode7data instanceof HitExtraData.Mode7Data){
295             min = ((HitExtraData.Mode7Data) mode7data).getAmplLow();
296             max = ((HitExtraData.Mode7Data) mode7data).getAmplHigh();
297         }
298 
299         // ignore if pulse at beginning of window:
300         if (max <= 0)
301             return;
302         EcalChannel cc = findChannel(hit);
303         if (cc == null) {
304             System.err.println("Hit doesn't correspond to ecalchannel.");
305             return;
306         }
307         aida.histogram1D(getHistoName(cc)).fill(min);
308     }

322     public EcalChannel findChannel(int crate, int slot, int chan) {
323         for (EcalChannel cc : ecalConditions.getChannelCollection()) {
324             if (crate == cc.getCrate() && slot == cc.getSlot() && chan == cc.getChannel()) {
325                 return cc;
326             }
327         }
328         throw new RuntimeException(String.format(
329                 "Could not find channel:  (crate,slot,channel)=(%d,%d,%d)",crate,slot,chan));
330     }
