18  public class FADCConverterDriver extends Driver {
19  
20      private EcalConditions ecalConditions = null;
21      private String rawCollectionName = "EcalReadoutHits";
22      private String ecalReadoutName = "EcalHits";
23      private String ecalCollectionName = "EcalIntegralHits";
24      private int numSamplesBefore = 5;
25      private int numSamplesAfter = 30;
26      private int threshold = 50;
27

29      }

32          this.threshold = threshold;
33      }

52          this.rawCollectionName = rawCollectionName;

55      @Override
56      public void startOfData() {
57          if (ecalCollectionName == null) {
58              throw new RuntimeException("The parameter ecalCollectionName was not set!");
59          }
60      }

62      @Override
63      public void detectorChanged(Detector detector) {
64          // ECAL combined conditions object.
65          ecalConditions = DatabaseConditionsManager.getInstance().getEcalConditions();
66  
67          System.out.println("You are now using the database conditions for FADCConverterDriver");
68      }
69  
70      @Override
71      public void process(EventHeader event) {
72          ArrayList<BaseRawCalorimeterHit> readoutHits = new ArrayList<BaseRawCalorimeterHit>();
73  
74          // Get the list of ECal hits.
75          if (event.hasCollection(RawTrackerHit.class, rawCollectionName)) {
76              List<RawTrackerHit> hits = event.get(RawTrackerHit.class, rawCollectionName);
77  
78              for (RawTrackerHit hit : hits) {
79                  short[] window = hit.getADCValues();
80                  long id = hit.getCellID();
81  
82                  // Get the channel data.
83                  EcalChannelConstants channelData = findChannel(id);
84  
85                  //do DAQ readout
86                  double crystalThreshold = channelData.getCalibration().getPedestal() + threshold;
87                  int adcSum = 0;
88                  int pointerOffset = 0;
89                  int numSamplesToRead = 0;
90                  int thresholdCrossing = 0;
91                  for (int i = 0; i < window.length; i++) {
92                      if (numSamplesToRead != 0) {
93                          adcSum += window[i - pointerOffset];
94                          numSamplesToRead--;
95                          if (numSamplesToRead == 0) {
96                              readoutHits.add(new BaseRawCalorimeterHit(id, adcSum, 64 * thresholdCrossing));
97                          }
98                      } else if ((i == 0 || window[i - 1] <= crystalThreshold) && window[i] > crystalThreshold) {
99                          thresholdCrossing = i;
100                         pointerOffset = Math.min(numSamplesBefore, i);
101                         numSamplesToRead = pointerOffset + Math.min(numSamplesAfter, window.length - i - pointerOffset - 1);
102                         adcSum = 0;
103                     }
104                 }
105                 //do trigger readout
106             }
107         }
108         int flags = 0;
109         event.put(ecalCollectionName, readoutHits, BaseRawCalorimeterHit.class, flags, ecalReadoutName);
110     }
111 
112     /**
113      * Convert physical ID to gain value.
114      *
115      * @param cellID (long)
116      * @return channel constants (EcalChannelConstants)
117      */
118     private EcalChannelConstants findChannel(long cellID) {
119         return ecalConditions.getChannelConstants(ecalConditions.getChannelCollection().findGeometric(cellID));
120     }
