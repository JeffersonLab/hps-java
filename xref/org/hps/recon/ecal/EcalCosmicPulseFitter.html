171     /**
172      * Perform and return "3-pole" fit of ECAL raw waveform.
173      * 
174      * @param hit
175      *            the RawTrackerHit (Mode-1 FADC ECal readout) to be fit
176      * @param threshCross
177      *            the sample of threshold crossing, used to initialize fit
178      *            parameters
179      * @param maxADC
180      *            the ADC at pulse maximum, used to initialize fit parameters
181      * 
182      * @return fit result, which can be null if fit fails or certain conditions are
183      *         not met.
184      */
185     public IFitResult fitCosmicPulse(RawTrackerHit hit, int threshCross, double maxADC) {
186 
187         // if (debug>0)
188         // System.err.println("FITTING.....................................................");
189 
190         final short samples[] = hit.getADCValues();
191 
192         // for (int i= 0; i<samples.length ; i++ ) {
193 
194         // System.out.println("ii\t"+i+"\tfadc\t"+samples[i]);
195 
196         // }
197 
198         final double noise = findChannel(hit.getCellID()).getCalibration().getNoise();
199         final int cid = ecalConditions.getChannelCollection().findGeometric(hit.getCellID()).getChannelId();
200 
201         // don't bother with pulses far from trigger:
202         if (threshCross < threshRange[0] || threshCross > threshRange[1]) {
203 
204             // System.out.println("outside thresh range");
205             return null;
206         }
207 
208         // calculate pedestal for initializing fit parameters:
209         int nped = 0;
210         double ped = 0;
211         for (int ii = threshCross + pedRange[0]; ii < threshCross + pedRange[1]; ii++) {
212             if (ii < 0)
213                 continue;
214             if (ii >= samples.length)
215                 break;
216             ped += samples[ii];
217             nped++;
218         }
219 
220         // System.out.println("pedestal sum"+ped);
221 
222         // don't bother trying to fit:
223         if (nped == 0)
224             return null;
225         ped /= nped;
226         // System.out.println("pedestal\t"+ped);
227 
228         // choose points to fit and get starting value for pulse integral:
229         fitData.clear();
230         int nFitPoints = 0;
231         int iPeakADC = -1;
232         int sumADC = 0;
233         // int maxADC=0;
234         for (int ii = threshCross + fitRange[0]; ii < threshCross + fitRange[1]; ii++) {
235             // System.out.println("ii in fit\t"+ii);
236             // System.out.println("samples length\t"+samples.length);
237 
238             if (ii < 0)
239                 continue;
240             if (ii >= samples.length)
241                 break;
242             // if (debug>0) System.err.print(ii+":"+samples[ii]+" ");
243             // if (samples[ii] > maxADC) maxADC=samples[ii];
244             if (iPeakADC < 0 && ii < samples.length - 1 && samples[ii + 1] < samples[ii]) {
245                 iPeakADC = ii;
246 
247             }
248             // System.out.println("fit point\t"+ii);
249 
250             sumADC += samples[ii];
251             fitData.addPoint();
252             fitData.point(nFitPoints).coordinate(0).setValue(ii);
253             fitData.point(nFitPoints).coordinate(1).setValue(samples[ii]);
254             fitData.point(nFitPoints).coordinate(1).setErrorMinus(noise);
255             fitData.point(nFitPoints).coordinate(1).setErrorPlus(noise);
256             nFitPoints++;
257         }
258         // if (debug>0) System.err.print("\n");
259 
260         // don't bother trying to fit:
261         // System.out.println("nfitpoints:\t"+nFitPoints);
262 
263         if (nFitPoints < 10)
264             return null;
265         // System.out.println("maxadc:\t"+maxADC+"\tped:\t"+ped);
266 
267         if (maxADC < ped)
268             return null;
269 
270         // if (debug>0) System.err.println("------- "+ped+" "+threshCross+" "+maxADC);
271 
272         final double pulseIntegral = sumADC - ped * nFitPoints;
273 
274         // initialize parameters:
275         fitFcn3Pole.setParameter("pedestal", ped);
276         fitFcn3Pole.setParameter("time0", (double) threshCross - 2);
277         fitFcn3Pole.setParameter("integral", pulseIntegral > 0 ? pulseIntegral : 2);
278         if (globalThreePoleWidth > 0)
279             fitFcn3Pole.setParameter("width", globalThreePoleWidth);
280         else
281             fitFcn3Pole.setParameter("width", threePoleWidths[cid - 1]);
282 
283         // constrain parameters:
284         // fitter.fitParameterSettings("time0").setBounds(1,30);
285         fitter.fitParameterSettings("time0").setBounds(t0limits[0], t0limits[1]);
286         fitter.fitParameterSettings("width").setBounds(0.1, 5);
287         fitter.fitParameterSettings("integral").setBounds(0, 999999);
288         if (fixShapeParameter)
289             fitter.fitParameterSettings("width").setFixed(true);
290 
291         // did this because something else was turning it back on every event
292         // once that thing stops doing that, this can be removed.
293         Logger.getLogger("org.freehep.math.minuit").setLevel(Level.OFF);
294         if (debug && hitCount > skipHits) {
295             System.out.println("pre fit");
296             for (int i = 0; i < fitFcn3Pole.parameters().length; i++) {
297                 System.out.println(fitFcn3Pole.parameterNames()[i] + "\t" + fitFcn3Pole.parameters()[i]);
298             }
299         }
300         IFitResult fitResult = fitter.fit(fitData, fitFcn3Pole);
301         double chi = getChi2(fitResult, fitData);
302         // System.out.println("Chi2 \t"+chi);
303         // int DOF = fitResult.ndf();
304         // System.out.println("Chi2/dof\t"+chi/DOF);
305 
306         if (debug && hitCount++ > skipHits) {
307             double max = 0;
308             double min = 99999;
309             for (int i = 0; i < fitData.size(); i++) {
310                 double v = fitData.point(i).coordinate(1).value();
311                 if (v > max)
312                     max = v;
313                 if (v < min)
314                     min = v;
315             }
316             if (max - min < 100)
317                 return fitResult; // skip this hit. too small.
318             if (hitCount % displayHits == 0) {
319                 plotter1.region(0).clear();
320                 plotter1.region(0).plot(fitData);
321                 plotter1.region(0).plot(fitResult.fittedFunction());
322 
323                 System.out.println(hit.getCellID());
324                 System.out.println("after fit");
325                 for (int i = 0; i < fitFcn3Pole.parameters().length; i++) {
326 
327                     System.out.printf("%s\t%.3f\t(+-%.3f)\n", fitResult.fittedFunction().parameterNames()[i],
328                             fitResult.fittedFunction().parameters()[i], Math.sqrt(fitResult.covMatrixElement(i, i)));
329                 }
330             }
331             double chi2 = getChi2(fitResult, fitData);
332 
333             int dof = fitResult.ndf();
334             // System.out.println("Chi2/dof\t"+chi2/dof);
335 
336             if (hitCount % displayHits == 0) {
337                 System.out.println("chi2\t" + chi2);
338                 System.out.println("dof\t" + dof);
339                 System.out.println("chi2 per dof\t" + chi2 / dof);
340                 System.out.println("fit quality: " + fitResult.quality());
341             }
342 
343             chi2Hist.fill(chi2);
344             widthOffsetHist.fill(fitResult.fittedParameter("width") - threePoleWidths[cid - 1]);
345             widthOffsets[cid - 1].fill(fitResult.fittedParameter("width") - threePoleWidths[cid - 1]);
346             pedestal_prefit.fill(ped);
347             pedestal_fit.fill(fitResult.fittedFunction().parameters()[0]);
348 
349             if (hitCount % displayHits == 0) {
350                 System.out.println("Press enter to continue");
351                 try {
352                     System.in.read();
353                 } catch (Exception e) {
354                     
355                 }
356                 plotter1.region(0).clear();
357             }
358         }
359         return fitResult;
360     }

362     /**
363      * calculate chi2 for debug
364      * 
365      * @param fitResult
366      * @param fitData2
367      * @return
368      */
369     private double getChi2(IFitResult fitResult, IDataPointSet fitData) {
370         double chi2 = 0;
371         for (int i = 0; i < fitData.size(); i++) {
372             double yp = fitResult.fittedFunction().value(new double[] { fitData.point(i).coordinate(0).value() });
373             double y = fitData.point(i).coordinate(1).value();
374             double ey = fitData.point(i).coordinate(1).errorMinus();
375             chi2 += Math.pow((y - yp) / ey, 2);
376         }
377         return chi2;
378     }
