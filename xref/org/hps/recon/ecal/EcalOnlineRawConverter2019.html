17  /**
18   * <code>EcalOnlineRawConverter2019</code> handles the conversion of raw hits of
19   * all modes to energy hit <code>CalorimeterHit</code> objects. This converter
20   * will employ the runtime values for all parameters and is intended to emulate
21   * the firmware specifically.<br/>
22   * <br/>
23   * The converter requires the presence of the DAQ configuration manager, which
24   * is activated by <code>DAQConfigDriver2019</code>. <br/>
25   * <br/>
26   * This converter is primarily employed in the trigger and hardware diagnostic
27   * processes as well as the readout simulation in Monte Carlo. <br/>
28   * <br/>
29   * <code>EcalOnlineRawConverter2019</code> is developed referring to
30   * <code>EcalOnlineRawConverter</code>
31   */
32  public class EcalOnlineRawConverter2019 {
33      // Defines the maximum number of peaks that may be extracted from
34      // a single waveform.
35      private int nPeak = 3;
36      // The DAQ configuration manager for FADC parameters.
37      private FADCConfigEcal2019 config = null;
38      // Whether or not a constant integration window should be assumed
39      // for the purpose of pedestal calculations/
40      private boolean constantWindow = false;
41      // The number of nanoseconds in a clock-cycle (sample).
42      private static final int nsPerSample = 4;
43  
44      /**
45       * Instantiates the <code>EcalOnlineRawConverter</code> and connects it to the
46       * <code>ConfigurationManager</code> to receive settings from the DAQ
47       * configuration.
48       */
49      public EcalOnlineRawConverter2019() {
50          // Track changes in the DAQ configuration.
51          ConfigurationManager2019.addActionListener(new ActionListener() {
52              @Override
53              public void actionPerformed(ActionEvent e) {
54                  // Get the FADC configuration.
55                  config = ConfigurationManager2019.getInstance().getEcalFADCConfig();
56  
57                  // Get the number of peaks.
58                  if (config.getMode() == 1) {
59                      nPeak = Integer.MAX_VALUE;
60                  } else {
61                      nPeak = config.getMaxPulses();
62                  }
63              }
64          });
65      }

67      /**
68       * Gets the pedestal for a given crystal and threshold crossing.
69       * 
70       * @param cellID            - The cell ID of the crystal.
71       * @param windowSamples     - The size of the readout window. A value of
72       *                          <code>-1</code> indicates an infinite readout
73       *                          window.
74       * @param thresholdCrossing - The time of the threshold crossing in 4-nanosecond
75       *                          clock-cycles (samples).
76       * @return Returns the pedestal for the crystal and threshold crossing.
77       */
78      public int getPulsePedestal(long cellID, int windowSamples, int thresholdCrossing) {
79          // Track the starting and ending samples over which integration
80          // will occur. Only the intermediary samples need be considered
81          // for pedestal calculation.
82          int firstSample, lastSample;
83  
84          // For finite readout windows, calculate the pedestal based on
85          // the size of the full readout window in the event that the
86          // integration window is larger than the readout window.
87          if (windowSamples > 0 && (config.getNSA() + config.getNSB()) / nsPerSample >= windowSamples) {
88              firstSample = 0;
89              lastSample = windowSamples - 1;
90          }
91  
92          // Otherwise, the pedestal should be calculated based on the
93          // integration window size.
94          else {
95              // Define the sample width as equivalent to the integration
96              // window size.
97              firstSample = thresholdCrossing - config.getNSB() / nsPerSample;
98              lastSample = thresholdCrossing + config.getNSA() / nsPerSample - 1;
99  
100             // In the event of a finite readout window, ignore any
101             // samples that fall outside the readout window. Since these
102             // are clipped and will not be integrated, these pedestals
103             // do not contribute.
104             if (windowSamples > 0) {
105                 if (firstSample < 0) {
106                     firstSample = 0;
107                 }
108                 if (lastSample >= windowSamples) {
109                     lastSample = windowSamples - 1;
110                 }
111             }
112         }
113 
114         // Calculate and return the pedestal. The extra 1 MeV added to
115         // the pedestal offsets the rounding error that is incurred when
116         // converting it to an integer.
117         return (int) ((lastSample - firstSample + 1) * (config.getPedestal(cellID) + 0.001));
118     }

120     /**
121      * Converts a mode-1 digitized waveform into standard energy hit.
122      * 
123      * @param hit - The "hit" object representing the digitized waveform for a given
124      *            crystal.
125      * @return Returns a list of <code>CalorimeterHit</code> objects parsed from the
126      *         waveform.
127      */
128     public List<CalorimeterHit> HitDtoA(RawTrackerHit hit) {
129         // Get the cell ID for the crystal as well as the digitized
130         // waveform samples.
131         final long cellID = hit.getCellID();
132         final short[] waveform = hit.getADCValues();
133 
134         // If there are no samples, then there is nothing to integrate.
135         if (waveform.length == 0) {
136             return null;
137         }
138 
139         // The pulse integration threshold is defined as the combination
140         // of the pedestal and the threshold configuration parameter.
141         final int absoluteThreshold = (int) (config.getPedestal(cellID) + config.getThreshold(cellID));
142 
143         // Store each instance of a threshold crossing in that can be
144         // found within the digitized waveform.
145         List<Integer> thresholdCrossings = new ArrayList<Integer>();
146 
147         // Check for the special case of the first sample exceeding
148         // the integration threshold.
149         if (waveform[0] > absoluteThreshold) {
150             thresholdCrossings.add(0);
151         }
152 
153         // Search the remaining samples for threshold crossings.
154         // Crossing sample is over threshold, and its previous sample is not over
155         // threshold.
156         // If a sample is over threshold, then check if any sample in the past 7 for
157         // model 1 (or NSA - 1 sample for other models) is a crossing sample. If no, a
158         // pulse is recorded.
159         thresholdLoop: for (int sample = 1; sample < waveform.length; sample++) {
160             if (waveform[sample] > absoluteThreshold && waveform[sample - 1] <= absoluteThreshold) {
161                 if (config.getMode() == 1) {
162                     if (sample < 8) {
163                         if (waveform[0] > absoluteThreshold)
164                             continue thresholdLoop;
165                         for (int pastSample = 1; pastSample < sample; pastSample++) {
166                             if (waveform[pastSample] > absoluteThreshold
167                                     && waveform[pastSample - 1] <= absoluteThreshold)
168                                 continue thresholdLoop;
169                         }
170                         thresholdCrossings.add(sample);
171                     } else {
172                         for (int pastSample = sample - 7; pastSample < sample; pastSample++) {
173                             if (waveform[pastSample] > absoluteThreshold
174                                     && waveform[pastSample - 1] <= absoluteThreshold)
175                                 continue thresholdLoop;
176                         }
177                         thresholdCrossings.add(sample);
178                     }
179                 }
180 
181                 else {
182                     if (sample < config.getNSA() / nsPerSample - 1) {
183                         if (waveform[0] > absoluteThreshold)
184                             continue thresholdLoop;
185                         for (int pastSample = 1; pastSample < sample; pastSample++) {
186                             if (waveform[pastSample] > absoluteThreshold
187                                     && waveform[pastSample - 1] <= absoluteThreshold)
188                                 continue thresholdLoop;
189                         }
190                         thresholdCrossings.add(sample);
191                     } else {
192                         for (int pastSample = sample
193                                 - config.getNSA() / nsPerSample; pastSample < sample; pastSample++) {
194                             if (waveform[pastSample] > absoluteThreshold
195                                     && waveform[pastSample - 1] <= absoluteThreshold)
196                                 continue thresholdLoop;
197                         }
198                         thresholdCrossings.add(sample);
199                     }
200                 }
201 
202                 // If there is a limit defined on the maximum number
203                 // of peaks that may be processed, terminate the search
204                 // after this number of peaks have been found.
205                 if (thresholdCrossings.size() >= nPeak) {
206                     break thresholdLoop;
207                 }
208             }
209         }
210 
211         // Use the previously located threshold crossing to generate
212         // calorimeter hits.
213         List<CalorimeterHit> newHits = new ArrayList<CalorimeterHit>();
214         for (int thresholdCrossing : thresholdCrossings) {
215             // Obtain the pedestal for the pulse.
216             int pedestal = getPulsePedestal(cellID, waveform.length, thresholdCrossing);
217 
218             // Perform the pulse integral.
219             final int[] data = convertWaveformToPulse(waveform, thresholdCrossing);
220             int time = data[0];
221             int sum = data[1];
222 
223             // Perform pedestal subtraction.
224             sum -= pedestal;
225 
226             // Perform gain scaling.
227             double energy = adcToEnergy(sum, cellID);
228 
229             // Hits should not have less than zero energy.
230             if (energy < 0) {
231                 Logger.getGlobal().log(Level.INFO, "A hit was produced with " + energy + " GeV energy!");
232             }
233 
234             // Create a new hit and add it to the list.
235             newHits.add(CalorimeterHitUtilities.create(energy, time, cellID));
236         }
237 
238         // Return the list of hits.
239         return newHits;
240     }

242     /**
243      * Converts a raw mode-3 hit to a proper calorimeter hit in units of energy.
244      * 
245      * @param hit        - The raw hit that is to be converted.
246      * @param timeOffset - The time offset for the hit.
247      * @return Returns a <code>CalorimeterHit</code> hit object that represents the
248      *         raw mode-3 hit with units of energy and a correct time-stamp.
249      */
250     public CalorimeterHit HitDtoA(RawCalorimeterHit hit, double timeOffset) {
251         // Verify the validity of the time-stamp.
252         if (hit.getTimeStamp() % 64 != 0) {
253             System.out.println("Unexpected time-stamp: " + hit.getTimeStamp());
254         }
255 
256         // Get the pedestal. In the case of a constant integration window
257         // (i.e. infinite readout window size, so no risk of clipping),
258         // the window width should be given as -1. Otherwise, the real
259         // readout window size should be used.
260         long id = hit.getCellID();
261         double time = hit.getTimeStamp() / 16.0;
262         int pedestal = getPulsePedestal(id, constantWindow ? -1 : config.getWindowWidth(), (int) time / nsPerSample);
263 
264         // Calculate the total ADC value for the pulse and convert it
265         // to energy.
266         int adcSum = hit.getAmplitude() - pedestal;
267         double rawEnergy = adcToEnergy(adcSum, id);
268 
269         // Create a calorimeter hit from the result and return it.
270         return CalorimeterHitUtilities.create(rawEnergy, time + timeOffset, id);
271     }

274      * Converts a raw mode-7 hit to a proper calorimeter hit in units of energy.
275      * 
276      * @param hit        - The raw hit that is to be converted.
277      * @param mode7Data  - Additional mode-7 data object.
278      * @param timeOffset - The time offset for the hit.
279      * @return Returns a <code>CalorimeterHit</code> hit object that represents the
280      *         raw mode-7 hit with units of energy and a correct time-stamp.
281      */
282     public CalorimeterHit HitDtoA(RawCalorimeterHit hit, GenericObject mode7Data, double timeOffset) {
283         // Get the time and crystal cell ID for the hit. Note that the
284         // time-stamps use the full 62.5 ps resolution.
285         double time = hit.getTimeStamp() / 16.0;
286         long id = hit.getCellID();
287 
288         // Get the pedestal. In the case of a constant integration window
289         // (i.e. infinite readout window size, so no risk of clipping),
290         // the window width should be given as -1. Otherwise, the real
291         // readout window size should be used.
292         int pedestal = getPulsePedestal(id, constantWindow ? -1 : config.getWindowWidth(), (int) time / nsPerSample);
293 
294         // Calculate the total ADC value for the pulse and convert it
295         // to energy.
296         int adcSum = hit.getAmplitude() - pedestal;
297         double rawEnergy = adcToEnergy(adcSum, id);
298 
299         // Create a calorimeter hit from the result and return it.
300         return CalorimeterHitUtilities.create(rawEnergy, time + timeOffset, id);
301     }

303     /**
304      * Converts a value in ADC in a crystal to energy in units of GeV.
305      * 
306      * @param adcSum - The ADC value to convert.
307      * @param cellID - The cell ID of the crystal containing the value.
308      * @return Returns the ADC value as an energy in units of GeV.
309      */
310     private double adcToEnergy(int adcSum, long cellID) {
311         // Define the gain. To mimic the hardware, this is done through
312         // manipulation of integer values only. We multiply by 256 to
313         // preserve extra digits of accuracy.
314         int gain = (int) (256.0 * (config.getGain(cellID) + 0.001));
315 
316         // Multiply the gain by the pulse-subtracted energy sum. Also
317         // remove the extra factor of 256. This gives the energy in units
318         // of MeV.
319         int energy = (int) ((adcSum * gain) / 256.0);
320 
321         // Return the final energy as a double in units of GeV.
322         return energy * EcalUtils.MeV;
323     }

330     /**
331      * Converts a mode-1 digitized waveform to a mode-3/7 pulse for a a given
332      * threshold crossing.
333      * 
334      * @param waveform          - The digitized waveform. Each array value should
335      *                          correspond to a sample of the waveform in units of
336      *                          ADC.
337      * @param thresholdCrossing - The time of the threshold crossing in samples.
338      * @return Returns a <code>double</code> primitive of size 2. The first value
339      *         represents the time in nanoseconds of the pulser and the second value
340      *         the total integrated value of the pulse in ADC.
341      */
342     private int[] convertWaveformToPulse(short[] waveform, int thresholdCrossing) {
343         // Store the integration range.
344         int firstSample, lastSample;
345 
346         // If the integration range is larger than the number of samples,
347         // then all the samples are used for pulse integration.
348         if ((config.getNSA() + config.getNSB()) / nsPerSample >= waveform.length) {
349             firstSample = 0;
350             lastSample = waveform.length - 1;
351         }
352 
353         // Otherwise, the integration range covers a number of samples
354         // before and after the threshold crossing as defined by the
355         // run parameters.
356         else {
357             firstSample = thresholdCrossing - config.getNSB() / nsPerSample;
358             lastSample = thresholdCrossing + config.getNSA() / nsPerSample - 1;
359         }
360 
361         // Perform the pulse integral.
362         int sumADC = 0;
363         integrationLoop: for (int sample = firstSample; sample <= lastSample; sample++) {
364             // If the current sample occurs before the readout window,
365             // then it does not exist and can not be integrated. Skip it.
366             if (sample < 0) {
367                 continue integrationLoop;
368             }
369 
370             // Likewise, samples that occur after the readout window are
371             // not retained and must be skipped.
372             if (sample >= waveform.length) {
373                 break integrationLoop;
374             }
375 
376             // Otherwise, add the sample to the pulse total.
377             sumADC += waveform[sample];
378         }
379 
380         // Calculate the pulse time with a 4 nanosecond resolution.
381         int pulseTime = thresholdCrossing * nsPerSample;
382 
383         // Return both the pulse time and the total integrated pulse ADC.
384         return new int[] { pulseTime, sumADC };
385     }
