24  /**
25   * This class is used to convert between {@link org.lcsim.event.RawCalorimeterHit} or
26   * {@link org.lcsim.event.RawTrackerHit}, objects with ADC/sample information, and
27   * {@link org.lcsim.event.CalorimeterHit}, an object with energy+time information. At minimum this involves pedestal
28   * subtraction/addition and gain scaling. Knows how to deal with Mode-1/3/7 FADC readout formats. Can perform Mode-3/7
29   * firmware algorithms on Mode-1 data. Can alternatively call pulse-fitting on Mode-1 data. All time walk/time offset
30   * corrections are performed to this collection after gains in EcalTimeCorrectionDriver

264     /**
265      * Get pedestal for a single ADC sample. Choose whether to use static pedestal from database or running pedestal
266      * from mode-7.
267      */
268     public double getSingleSamplePedestal(EventHeader event, long cellID) {
269         if (useDAQConfig) {
270             // EcalChannel channel =
271             // ecalConditions.getChannelCollection().findGeometric(cellID);
272             return config.getPedestal(cellID);
273         }
274         if (useRunningPedestal && event != null) {
275             if (event.hasItem("EcalRunningPedestals")) {
276                 @SuppressWarnings("unchecked")
277                 Map<EcalChannel, Double> runningPedMap = (Map<EcalChannel, Double>) event.get("EcalRunningPedestals");
278                 EcalChannel chan = ecalConditions.getChannelCollection().findGeometric(cellID);
279                 if (!runningPedMap.containsKey(chan)) {
280                     System.err.println("************** Missing Pedestal");
281                 } else {
282                     return runningPedMap.get(chan);
283                 }
284             } else {
285                 System.err.println("*****************************************************************");
286                 System.err.println("**  You Requested a Running Pedestal, but it is NOT available. **");
287                 System.err.println("**     Reverting to the database. Only printing this ONCE.     **");
288                 System.err.println("*****************************************************************");
289                 useRunningPedestal = false;
290             }
291         }
292         return findChannel(cellID).getCalibration().getPedestal();
293     }

295     /**
296      * Get pedestal for entire pulse integral. Account for clipping if windowSamples is greater than zero.
297      */
298     public double getPulsePedestal(EventHeader event, long cellID, int windowSamples, int thresholdCrossing) {
299         int firstSample, lastSample;
300         if (windowSamples > 0 && (NSA + NSB) / nsPerSample >= windowSamples) {
301             // special case where firmware always integrates entire window
302             firstSample = 0;
303             lastSample = windowSamples - 1;
304         } else {
305             firstSample = thresholdCrossing - NSB / nsPerSample;
306             lastSample = thresholdCrossing + NSA / nsPerSample - 1;
307             if (windowSamples > 0) {
308                 // properly pedestal subtract pulses clipped by edge(s) of
309                 // readout window:
310                 if (firstSample < 0)
311                     firstSample = 0;
312                 if (lastSample >= windowSamples)
313                     lastSample = windowSamples - 1;
314             }
315         }
316         return (lastSample - firstSample + 1) * getSingleSamplePedestal(event, cellID);
317     }

319     /**
320      * Emulate the FADC250 firmware in conversion of Mode-1 waveform to a Mode-3/7 pulse, given a time for threshold
321      * crossing.
322      */
323     public double[] convertWaveformToPulse(RawTrackerHit hit, int thresholdCrossing, boolean mode7) {
324 
325         double fitQuality = -1;
326 
327         short samples[] = hit.getADCValues();
328         // System.out.println("NewEvent");
329         // choose integration range:
330         int firstSample, lastSample;
331         if ((NSA + NSB) / nsPerSample >= samples.length) {
332             // firmware treats this case specially:
333             firstSample = 0;
334             lastSample = samples.length - 1;
335         } else {
336             firstSample = thresholdCrossing - NSB / nsPerSample;
337             lastSample = thresholdCrossing + NSA / nsPerSample - 1;
338         }
339 
340         // mode-7's minimum/pedestal (average of first 4 samples):
341         double minADC = 0;
342         for (int jj = 0; jj < 4; jj++)
343             minADC += samples[jj];
344         // does the firmware's conversion of min to int occur before or after
345         // time calculation? undocumented.
346         // minADC=(int)(minADC/4);
347         minADC = (minADC / 4);
348 
349         // System.out.println("Avg pedestal:\t"+minADC);
350 
351         // mode-7's max pulse height:
352         double maxADC = 0;
353         // int sampleMaxADC=0;
354 
355         // mode-3/7's pulse integral:
356         double sumADC = 0;
357 
358         for (int jj = firstSample; jj <= lastSample; jj++) {
359 
360             if (jj < 0)
361                 continue;
362             if (jj >= samples.length)
363                 break;
364 
365             // integrate pulse:
366             sumADC += samples[jj];
367         }
368 
369         // find pulse maximum:
370         // if (jj>firstSample && jj<samples.length-5) { // The "5" here is a
371         // firmware constant.
372         for (int jj = thresholdCrossing; jj < samples.length - 5; jj++) { // The
373                                                                           // "5"
374                                                                           // here
375                                                                           // is
376                                                                           // a
377                                                                           // firmware
378                                                                           // constant.
379             if (samples[jj + 1] < samples[jj]) {
380                 // sampleMaxADC=jj;
381                 maxADC = samples[jj];
382                 break;
383             }
384         }
385 
386         // pulse time with 4ns resolution:
387         double pulseTime = thresholdCrossing * nsPerSample;
388 
389         // calculate Mode-7 high-resolution time:
390         if (mode7) {
391             if (thresholdCrossing < 4) {
392                 // special case where firmware sets max to zero and time to 4ns
393                 // time.
394                 maxADC = 0;
395             } else if (maxADC > 0) {
396                 // linear interpolation between threshold crossing and
397                 // pulse maximum to find time at pulse half-height:
398 
399                 final double halfMax = (maxADC + minADC) / 2;
400                 int t0 = -1;
401                 for (int ii = thresholdCrossing - 1; ii < lastSample; ii++) {
402                     if (ii >= samples.length - 1)
403                         break;
404                     if (samples[ii] <= halfMax && samples[ii + 1] > halfMax) {
405                         t0 = ii;
406                         break;
407                     }
408                 }
409                 if (t0 > 0) {
410                     final int t1 = t0 + 1;
411                     final int a0 = samples[t0];
412                     final int a1 = samples[t1];
413                     // final double slope = (a1 - a0); // units = ADC/sample
414                     // final double yint = a1 - slope * t1; // units = ADC
415                     pulseTime = ((halfMax - a0) / (a1 - a0) + t0) * nsPerSample;
416                 }
417             }
418         }
419 
420         if (useFit) {
421             IFitResult fitResult = pulseFitter.fitPulse(hit, thresholdCrossing, maxADC);
422             if (fitResult != null) {
423                 fitQuality = fitResult.quality();
424                 if (fitQuality > 0) {
425                     pulseTime = fitResult.fittedParameter("time0") * nsPerSample;
426                     sumADC = fitResult.fittedParameter("integral");
427                     minADC = fitResult.fittedParameter("pedestal");
428                     maxADC = ((Ecal3PoleFunction) fitResult.fittedFunction()).maximum();
429                 }
430             }
431         }
432 
433         return new double[] {pulseTime, sumADC, minADC, maxADC, fitQuality};
434     }

436     /**
437      * This HitDtoA is for emulating the conversion of Mode-1 readout (RawTrackerHit) into what EcalRawConverter would
438      * have created from a Mode-3 or Mode-7 readout. Clustering classes will read the resulting CalorimeterHits same as
439      * if they were directly readout from the FADCs in Mode-3/7. For Mode-3, hit time is just the time of threshold
440      * crossing, with an optional time-walk correction. For Mode-7, it is a "high-resolution" one calculated by linear
441      * interpolation between threshold crossing and pulse maximum. TODO: Generate GenericObject (and corresponding
442      * LCRelation) to store min and max to fully emulate mode-7. This is less important for now.
443      */
444     public ArrayList<CalorimeterHit> HitDtoA(EventHeader event, RawTrackerHit hit) {
445         final long cellID = hit.getCellID();
446         final short samples[] = hit.getADCValues();
447         if (samples.length == 0)
448             return null;
449 
450         // threshold is pedestal plus threshold configuration parameter:
451         final int absoluteThreshold;
452         if (useDAQConfig) {
453             // EcalChannel channel =
454             // ecalConditions.getChannelCollection().findGeometric(hit.getCellID());
455             // int leadingEdgeThreshold =
456             // ConfigurationManager.getInstance().getFADCConfig().getThreshold(channel.getChannelId());
457             int leadingEdgeThreshold = config.getThreshold(cellID);
458             absoluteThreshold = (int) (getSingleSamplePedestal(event, cellID) + leadingEdgeThreshold);
459         } else {
460             absoluteThreshold = (int) (getSingleSamplePedestal(event, cellID) + leadingEdgeThreshold);
461         }
462 
463         ArrayList<Integer> thresholdCrossings = new ArrayList<Integer>();
464 
465         // special case, first sample is above threshold:
466         if (samples[0] > absoluteThreshold) {
467             thresholdCrossings.add(0);
468         }
469 
470         // search for threshold crossings:
471         for (int ii = 1; ii < samples.length; ++ii) {
472             if (samples[ii] > absoluteThreshold && samples[ii - 1] <= absoluteThreshold) {
473 
474                 // found one:
475                 thresholdCrossings.add(ii);
476 
477                 // search for next threshold crossing begins at end of this
478                 // pulse:
479                 if (useDAQConfig && ConfigurationManager.getInstance().getFADCConfig().getMode() == 1) {
480                     // special case, emulating SSP:
481                     ii += 8;
482                 } else {
483                     // "normal" case, emulating FADC250:
484                     ii += NSA / nsPerSample - 1;
485                 }
486 
487                 // firmware limit on # of peaks:
488                 if (thresholdCrossings.size() >= nPeak)
489                     break;
490             }
491         }
492 
493         // make hits
494         ArrayList<CalorimeterHit> newHits = new ArrayList<CalorimeterHit>();
495         for (int thresholdCrossing : thresholdCrossings) {
496             // do pulse integral:
497             final double[] data = convertWaveformToPulse(hit, thresholdCrossing, mode7);
498             double time = data[0];
499             double sum = data[1];
500             // final double min = data[2]; // TODO: stick min and max in a
501             // GenericObject with an
502             // final double max = data[3]; // LCRelation to finish mode-7
503             // emulation
504             final double fitQuality = data[4];
505 
506             if (!useFit || fitQuality <= 0) {
507                 // do pedestal subtraction:
508                 sum -= getPulsePedestal(event, cellID, samples.length, thresholdCrossing);
509             }
510 
511             // do gain scaling using a dummy gain.  
512             
513 
514             newHits.add(CalorimeterHitUtilities.create(sum, time, cellID));
515         }
516 
517         return newHits;
518     }

522     /**
523      * This HitDtoA is for Mode-3 data. A time-walk correction can be applied.
524      */
525     public CalorimeterHit HitDtoA(EventHeader event, RawCalorimeterHit hit, double timeOffset) {
526         if (hit.getTimeStamp() % 64 != 0) {
527             System.out.println("unexpected timestamp " + hit.getTimeStamp());
528         }
529         double time = hit.getTimeStamp() / 16.0;
530         long id = hit.getCellID();
531         double pedestal = getPulsePedestal(event, id, windowSamples, (int) time / nsPerSample);
532         double adcSum = hit.getAmplitude() - pedestal;
533         //double rawEnergy = adcToEnergy(adcSum);
534         
535         return CalorimeterHitUtilities.create(adcSum, time + timeOffset, id);
536     }

551     /**
552      * This converts a corrected pulse integral (pedestal-subtracted and gain-scaled) back into raw pulse integral with
553      * units ADC.
554      */
555     public RawCalorimeterHit HitAtoD(CalorimeterHit hit) {
556         int time = (int) (Math.round(hit.getTime() / 4.0) * 64.0);
557         long id = hit.getCellID();
558         // Get the channel data.
559         //EcalChannelConstants channelData = findChannel(id);
560         int amplitude;
561         double pedestal = getPulsePedestal(null, id, windowSamples, (int) hit.getTime() / nsPerSample);
562         amplitude = (int) Math.round((hit.getRawEnergy() / EcalUtils.MeV) / (EcalUtils.gainFactor * EcalUtils.ecalReadoutPeriod) + pedestal);
563        
564         // time += findChannel(id).getTimeShift().getTimeShift();
565         RawCalorimeterHit h = new BaseRawCalorimeterHit(id, amplitude, time);
566         return h;
567     }

581     /**
582      * Integrate the entire window. Return pedestal-subtracted integral.
583      */
584     public int sumADC(RawTrackerHit hit) {
585         EcalChannelConstants channelData = findChannel(hit.getCellID());
586         double pedestal;
587         if (useDAQConfig) {
588             // EcalChannel channel =
589             // ecalConditions.getChannelCollection().findGeometric(hit.getCellID());
590             pedestal = config.getPedestal(hit.getCellID());
591         } else {
592             pedestal = channelData.getCalibration().getPedestal();
593         }
594 
595         int sum = 0;
596         short samples[] = hit.getADCValues();
597         for (int isample = 0; isample < samples.length; ++isample) {
598             sum += (samples[isample] - pedestal);
599         }
600         return sum;
601     }
