51      /**
52       * Converts a hit of type {@link org.lcsim.event.CalorimeterHit
53       * CalorimeterHit} to a {@link org.lcsim.event.SimCalorimeterHit
54       * SimCalorimeterHit} object that contains all of the truth data
55       * present in the attached {@link java.util.Set Set}.
56       * @param baseHit - The generic calorimeter hit object.
57       * @param truthData - A collection of all truth data associated
58       * with the calorimeter hit object.
59       * @return Returns a SimCalorimeterHit object with the same
60       * parameters as the generic hit, but with all the compiled truth
61       * information from the truth set.
62       */
63      public static final SimCalorimeterHit convertToTruthHit(CalorimeterHit baseHit, Collection<SimCalorimeterHit> truthData, LCMetaData metaData) {
64          // If there is no truth information, just don't write any.
65          // Declare a warning, though, so the user knows that this is
66          // happening.
67          if(truthData == null || truthData.isEmpty()) {
68              System.out.println("Warning: Converting hit to truth hit, but no truth data was provided. (Hit time = " + baseHit.getTime() + ".)");
69              if(truthData == null) {
70                  truthData = new java.util.ArrayList<SimCalorimeterHit>(0);
71              }
72          }
73          
74          // Extract the truth data into one, singular list.
75          List<Integer> pdgs = new ArrayList<Integer>();
76          List<Float> times = new ArrayList<Float>();
77          List<Float> energies = new ArrayList<Float>();
78          List<MCParticle> particles = new ArrayList<MCParticle>();
79          for(SimCalorimeterHit truthHit : truthData) {
80              // Iterate over the truth data...
81              int particleCount = truthHit.getMCParticleCount();
82              for(int i = 0; i < particleCount; i++) {
83                  // Append the truth information for the truth hit to
84                  // the combined truth information for the new hit.
85                  pdgs.add(truthHit.getMCParticle(i).getPDGID());
86                  energies.add((float) truthHit.getContributedEnergy(i));
87                  particles.add(truthHit.getMCParticle(i));
88                  
89                  // Note that contribution times are automatically
90                  // used to calculate a time in SimCalorimeterHit for
91                  // the overall hit. This behavior is not desired here
92                  // and thus all particles are given the "hit time"
93                  // from the base hit to bypass this.
94                  times.add((float) baseHit.getTime());
95              }
96          }
97          
98          // Convert the lists to arrays.
99          MCParticle[] particleTruthArray = particles.toArray(new MCParticle[particles.size()]);
100         int[] pdgsTruthArray = new int[particles.size()];
101         float[] timeTruthArray = new float[particles.size()];
102         float[] energyTruthArray = new float[particles.size()];
103         for(int i = 0; i < particles.size(); i++) {
104             pdgsTruthArray[i] = pdgs.get(i).intValue();
105             timeTruthArray[i] = times.get(i).floatValue();
106             energyTruthArray[i] = energies.get(i).floatValue();
107         }
108         
109         // Store the truth information in a new SimCalorimeterHit
110         // object, and give it the same properties otherwise as the
111         // base hit.
112         SimCalorimeterHit newHit = new TruthCalorimeterHit(baseHit, particleTruthArray, energyTruthArray, timeTruthArray, pdgsTruthArray);
113         
114         // Return the hit.
115         return newHit;
116     }
