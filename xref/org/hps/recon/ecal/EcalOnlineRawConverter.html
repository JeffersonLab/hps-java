17  /**
18   * <code>EcalOnlineRawConverter</code> handles the conversion of raw
19   * hits of all modes to energy hit <code>CalorimeterHit</code> objects.
20   * This converter will employ the runtime values for all parameters and
21   * is intended to emulate the firmware specifically.<br/>
22   * <br/>
23   * The converter requires the presence of the DAQ configuration manager,
24   * which is activated by either <code>DatabaseDAQConfigDriver</code>
25   * or <code>DAQConfigDriver</code> depending on from where it is to
26   * obtain the configuration.<br/>
27   * <br/>
28   * This converter is primarily employed in the trigger and hardware
29   * diagnostic processes as well as the readout simulation in Monte
30   * Carlo.
31   */
32  public class EcalOnlineRawConverter {
33      // Defines the maximum number of peaks that may be extracted from
34      // a single waveform.
35      private int nPeak = 3;
36      // The DAQ configuration manager for FADC parameters.
37      private FADCConfig config = null;
38      // Whether or not a constant integration window should be assumed
39      // for the purpose of pedestal calculations/
40      private boolean constantWindow = false;
41      // The number of nanoseconds in a clock-cycle (sample).
42      private static final int nsPerSample = 4;
43      
44      /**
45       * Instantiates the <code>EcalOnlineRawConverter</code> and connects
46       * it to the <code>ConfigurationManager</code> to receive settings
47       * from the DAQ configuration.
48       */
49      public EcalOnlineRawConverter() {
50          // Track changes in the DAQ configuration.
51          ConfigurationManager.addActionListener(new ActionListener() {
52              @Override
53              public void actionPerformed(ActionEvent e) {
54                  // Get the FADC configuration.
55                  config = ConfigurationManager.getInstance().getFADCConfig();
56                  
57                  // Get the number of peaks.
58                  if(config.getMode() == 1) { nPeak = Integer.MAX_VALUE; }
59                  else { nPeak = config.getMaxPulses(); }
60              }
61          });
62      }

64      /**
65       * Gets the pedestal for a given crystal and threshold crossing.
66       * @param cellID - The cell ID of the crystal.
67       * @param windowSamples - The size of the readout window. A value
68       * of <code>-1</code> indicates an infinite readout window.
69       * @param thresholdCrossing - The time of the threshold crossing in
70       * 4-nanosecond clock-cycles (samples).
71       * @return Returns the pedestal for the crystal and threshold
72       * crossing.
73       */
74      public int getPulsePedestal(long cellID, int windowSamples, int thresholdCrossing) {
75          // Track the starting and ending samples over which integration
76          // will occur. Only the intermediary samples need be considered
77          // for pedestal calculation.
78          int firstSample, lastSample;
79          
80          // For finite readout windows, calculate the pedestal based on
81          // the size of the full readout window in the event that the
82          // integration window is larger than the readout window.
83          if(windowSamples > 0 && (config.getNSA() + config.getNSB()) / nsPerSample >= windowSamples) {
84              firstSample = 0;
85              lastSample = windowSamples - 1;
86          }
87          
88          // Otherwise, the pedestal should be calculated based on the
89          // integration window size.
90          else {
91              // Define the sample width as equivalent to the integration
92              // window size.
93              firstSample = thresholdCrossing - config.getNSB() / nsPerSample;
94              lastSample  = thresholdCrossing + config.getNSA() / nsPerSample - 1;
95              
96              // In the event of a finite readout window, ignore any
97              // samples that fall outside the readout window. Since these
98              // are clipped and will not be integrated, these pedestals
99              // do not contribute.
100             if(windowSamples > 0) {
101                 if(firstSample < 0) { firstSample = 0; }
102                 if(lastSample >= windowSamples) { lastSample = windowSamples - 1; }
103             }
104         }
105         
106         // Calculate and return the pedestal. The extra 1 MeV added to
107         // the pedestal offsets the rounding error that is incurred when
108         // converting it to an integer.
109         return (int) ((lastSample - firstSample + 1) * (config.getPedestal(cellID) + 0.001));
110     }

112     /**
113      * Converts a mode-1 digitized waveform into standard energy hit.
114      * @param hit - The "hit" object representing the digitized waveform
115      * for a given crystal.
116      * @return Returns a list of <code>CalorimeterHit</code> objects
117      * parsed from the waveform.
118      */
119     public List<CalorimeterHit> HitDtoA(RawTrackerHit hit) {
120         // Get the cell ID for the crystal as well as the digitized
121         // waveform samples.
122         final long cellID = hit.getCellID();
123         final short[] waveform = hit.getADCValues();
124         
125         // If there are no samples, then there is nothing to integrate.
126         if(waveform.length == 0) { return null; }
127         
128         // The pulse integration threshold is defined as the combination
129         // of the pedestal and the threshold configuration parameter.
130         final int absoluteThreshold = (int) (config.getPedestal(cellID) + config.getThreshold(cellID));
131         
132         // Store each instance of a threshold crossing in that can be
133         // found within the digitized waveform.
134         List<Integer> thresholdCrossings = new ArrayList<Integer>();
135         
136         // Check for the special case of the first sample exceeding
137         // the integration threshold.
138         if(waveform[0] > absoluteThreshold) {
139             thresholdCrossings.add(0);
140         } 
141         
142         // Search the remaining samples for threshold crossings.
143         thresholdLoop:
144         for(int sample = 1; sample < waveform.length; sample++) {
145             if(waveform[sample] > absoluteThreshold && waveform[sample - 1] <= absoluteThreshold) {
146                 // Add the sample index to the list of threshold crossing.
147                 thresholdCrossings.add(sample);
148                 
149                 // No new threshold crossings can be registered within
150                 // this pulse. In the case of mode-1 data, the end of
151                 // the pulse is considered to be 8 samples past the
152                 // crossing, as the per the SSP. Otherwise, it is defined
153                 // by the integration window.
154                 if(config.getMode() == 1) { sample += 8; }
155                 else { sample += config.getNSA() / nsPerSample - 1; }
156                 
157                 // If there is a limit defined on the maximum number
158                 // of peaks that may be processed, terminate the search
159                 // after this number of peaks have been found.
160                 if(thresholdCrossings.size() >= nPeak) { break thresholdLoop; }
161             }
162         }
163         
164         // Use the previously located threshold crossing to generate
165         // calorimeter hits.
166         List<CalorimeterHit> newHits = new ArrayList<CalorimeterHit>();
167         for(int thresholdCrossing : thresholdCrossings) {
168             // Obtain the pedestal for the pulse.
169             int pedestal = getPulsePedestal(cellID, waveform.length, thresholdCrossing);
170             
171             // Perform the pulse integral.
172             final int[] data = convertWaveformToPulse(waveform, thresholdCrossing);
173             int time = data[0];
174             int sum = data[1];
175             
176             // Perform pedestal subtraction.
177             sum -= pedestal;
178             
179             // Perform gain scaling.
180             double energy = adcToEnergy(sum, cellID);
181             
182             // Hits should not have less than zero energy.
183             if(energy < 0) {
184                 Logger.getGlobal().log(Level.WARNING, "A hit was produced with " + energy + " GeV energy!");
185             }
186             
187             // Create a new hit and add it to the list.
188             newHits.add(CalorimeterHitUtilities.create(energy, time, cellID));
189         }
190         
191         // Return the list of hits.
192         return newHits;
193     }

195     /**
196      * Converts a raw mode-3 hit to a proper calorimeter hit in units
197      * of energy.
198      * @param hit - The raw hit that is to be converted.
199      * @param timeOffset - The time offset for the hit.
200      * @return Returns a <code>CalorimeterHit</code> hit object that
201      * represents the raw mode-3 hit with units of energy and a correct
202      * time-stamp.
203      */
204     public CalorimeterHit HitDtoA(RawCalorimeterHit hit, double timeOffset) {
205         // Verify the validity of the time-stamp.
206         if(hit.getTimeStamp() % 64 != 0) {
207             System.out.println("Unexpected time-stamp: " + hit.getTimeStamp());
208         }
209         
210         // Get the pedestal. In the case of a constant integration window
211         // (i.e. infinite readout window size, so no risk of clipping),
212         // the window width should be given as -1. Otherwise, the real
213         // readout window size should be used.
214         long id = hit.getCellID();
215         double time = hit.getTimeStamp() / 16.0;
216         int pedestal = getPulsePedestal(id, constantWindow ? -1 : config.getWindowWidth(), (int) time / nsPerSample);
217         
218         // Calculate the total ADC value for the pulse and convert it
219         // to energy.
220         int adcSum = hit.getAmplitude() - pedestal;
221         double rawEnergy = adcToEnergy(adcSum, id);
222         
223         // Create a calorimeter hit from the result and return it.
224         return CalorimeterHitUtilities.create(rawEnergy, time + timeOffset, id);
225     }

227     /**
228      * Converts a raw mode-7 hit to a proper calorimeter hit in units
229      * of energy.
230      * @param hit - The raw hit that is to be converted.
231      * @param mode7Data - Additional mode-7 data object.
232      * @param timeOffset - The time offset for the hit.
233      * @return Returns a <code>CalorimeterHit</code> hit object that
234      * represents the raw mode-7 hit with units of energy and a correct
235      * time-stamp.
236      */
237     public CalorimeterHit HitDtoA(RawCalorimeterHit hit, GenericObject mode7Data, double timeOffset) {
238         // Get the time and crystal cell ID for the hit. Note that the
239         // time-stamps use the full 62.5 ps resolution.
240         double time = hit.getTimeStamp() / 16.0;
241         long id = hit.getCellID();
242         
243         // Get the pedestal. In the case of a constant integration window
244         // (i.e. infinite readout window size, so no risk of clipping),
245         // the window width should be given as -1. Otherwise, the real
246         // readout window size should be used.
247         int pedestal = getPulsePedestal(id, constantWindow ? -1 : config.getWindowWidth(), (int) time / nsPerSample);
248         
249         // Calculate the total ADC value for the pulse and convert it
250         // to energy.
251         int adcSum = hit.getAmplitude() - pedestal;
252         double rawEnergy = adcToEnergy(adcSum, id);
253         
254         // Create a calorimeter hit from the result and return it.
255         return CalorimeterHitUtilities.create(rawEnergy, time + timeOffset, id);
256     }

258     /**
259      * Converts a value in ADC in a crystal to energy in units of GeV.
260      * @param adcSum - The ADC value to convert.
261      * @param cellID - The cell ID of the crystal containing the value.
262      * @return Returns the ADC value as an energy in units of GeV.
263      */
264     private double adcToEnergy(int adcSum, long cellID) {
265         // Define the gain. To mimic the hardware, this is done through
266         // manipulation of integer values only. We multiply by 256 to
267         // preserve extra digits of accuracy.
268         int gain = (int) (256.0 * (config.getGain(cellID) + 0.001));
269         
270         // Multiply the gain by the pulse-subtracted energy sum. Also
271         // remove the extra factor of 256. This gives the energy in units
272         // of MeV.
273         int energy = (int) ((adcSum * gain) / 256.0);
274         
275         // Return the final energy as a double in units of GeV.
276         return energy * EcalUtils.MeV;
277     }

284     /**
285      * Converts a mode-1 digitized waveform to a mode-3/7 pulse for a
286      * a given threshold crossing.
287      * @param waveform - The digitized waveform. Each array value should
288      * correspond to a sample of the waveform in units of ADC.
289      * @param thresholdCrossing - The time of the threshold crossing
290      * in samples.
291      * @return Returns a <code>double</code> primitive of size 2. The
292      * first value represents the time in nanoseconds of the pulser and
293      * the second value the total integrated value of the pulse in ADC.
294      */
295     private int[] convertWaveformToPulse(short[] waveform, int thresholdCrossing) {
296         // Store the integration range.
297         int firstSample, lastSample;
298         
299         // If the integration range is larger than the number of samples,
300         // then all the samples are used for pulse integration.
301         if((config.getNSA() + config.getNSB()) / nsPerSample >= waveform.length) {
302             firstSample = 0;
303             lastSample = waveform.length - 1;
304         }
305         
306         // Otherwise, the integration range covers a number of samples
307         // before and after the threshold crossing as defined by the
308         // run parameters.
309         else {
310             firstSample = thresholdCrossing - config.getNSB() / nsPerSample;
311             lastSample  = thresholdCrossing + config.getNSA() / nsPerSample - 1;
312         }
313         
314         // Perform the pulse integral.
315         int sumADC = 0;
316         integrationLoop:
317         for (int sample = firstSample; sample <= lastSample; sample++) {
318             // If the current sample occurs before the readout window,
319             // then it does not exist and can not be integrated. Skip it.
320             if(sample < 0) { continue integrationLoop; }
321             
322             // Likewise, samples that occur after the readout window are
323             // not retained and must be skipped.
324             if(sample >= waveform.length) { break integrationLoop; }
325             
326             // Otherwise, add the sample to the pulse total.
327             sumADC += waveform[sample];
328         }
329         
330         // Calculate the pulse time with a 4 nanosecond resolution.
331         int pulseTime = thresholdCrossing * nsPerSample;
332         
333         // Return both the pulse time and the total integrated pulse ADC.
334         return new int[] { pulseTime, sumADC };
335     }
