20  /**
21   * This <code>Driver</code> converts raw ECal data collections to {@link org.lcsim.event.CalorimeterHit} collections
22   * with energy and time information. The {@link EcalRawConverter} does most of the low-level work.
23   * <p>
24   * The following input collections are used:
25   * <ul>
26   * <li>EcalReadoutHits
27   * <li>
28   * <li>EcalReadoutExtraDataRelations</li>
29   * <li>EcalRunningPedestals</li>
30   * </ul>
31   * <p>
32   * The results are by default written to the <b>EcalCalHits</b> output collection.
33   */

376      * @return false if the channel is a good one, true if it is a bad one
377      * @param hit the <code>CalorimeterHit</code> pointing to the channel
378      */
379     public boolean isBadCrystal(CalorimeterHit hit) {
380         // Get the channel data.
381         EcalChannelConstants channelData = findChannel(hit.getCellID());
382         return channelData.isBadChannel();
383     }

385     /**
386      * @return false if the ADC is a good one, true if it is a bad one
387      * @param hit the <code>CalorimeterHit</code> pointing to the FADC
388      */
389     public boolean isBadFADC(CalorimeterHit hit) {
390         return (getCrate(hit.getCellID()) == 1 && getSlot(hit.getCellID()) == 3);
391     }

393     private static double getTimestamp(int system, EventHeader event) { // FIXME: copied from
394                                                                         // org.hps.readout.ecal.ReadoutTimestamp
395         if (event.hasCollection(GenericObject.class, "ReadoutTimestamps")) {
396             List<GenericObject> timestamps = event.get(GenericObject.class, "ReadoutTimestamps");
397             for (GenericObject timestamp : timestamps) {
398                 if (timestamp.getIntVal(0) == system) {
399                     return timestamp.getDoubleVal(0);
400                 }
401             }
402             return 0;
403         } else {
404             return 0;
405         }
406     }

408     @Override
409     public void process(EventHeader event) {
410         // Do not process the event if the DAQ configuration should be
411         // used for value, but is not initialized.
412         if (useDAQConfig && !ConfigurationManager.isInitialized()) {
413             return;
414         }
415 
416         final int SYSTEM_TRIGGER = 0;
417         // final int SYSTEM_TRACKER = 1;
418         final int SYSTEM_ECAL = 2;
419 
420         double timeOffset = 0.0;
421         if (useTimestamps) {
422             double t0ECal = getTimestamp(SYSTEM_ECAL, event);
423             double t0Trig = getTimestamp(SYSTEM_TRIGGER, event);
424             timeOffset += (t0ECal - t0Trig) + 200.0;
425         }
426         if (useTruthTime) {
427             double t0ECal = getTimestamp(SYSTEM_ECAL, event);
428             timeOffset += ((t0ECal + 250.0) % 500.0) - 250.0;
429         }
430 
431         int flags = 0;
432         flags += 1 << LCIOConstants.RCHBIT_TIME; // store hit time
433         flags += 1 << LCIOConstants.RCHBIT_LONG; // store hit position; this flag has no effect for RawCalorimeterHits
434         
435         if (!runBackwards) {
436             ArrayList<CalorimeterHit> newHits = new ArrayList<CalorimeterHit>();
437 
438             /*
439              * This is for FADC Mode-1 data:
440              */
441             if (event.hasCollection(RawTrackerHit.class, rawCollectionName)) {
442                 List<RawTrackerHit> hits = event.get(RawTrackerHit.class, rawCollectionName);
443 
444                 for (RawTrackerHit hit : hits) {
445 
446                     ArrayList<CalorimeterHit> newHits2 = new ArrayList<CalorimeterHit>();
447                     if (emulateFirmware) {
448                         newHits2.addAll(converter.HitDtoA(event, hit));
449                     } else {
450                         newHits2.add(converter.HitDtoA(hit));
451                     }
452 
453                     for (CalorimeterHit newHit : newHits2) {
454 
455                         // Get the channel data.
456                         EcalChannelConstants channelData = findChannel(newHit.getCellID());
457 
458                         if (applyBadCrystalMap && channelData.isBadChannel()) {
459                             continue;
460                         }
461                         if (dropBadFADC && isBadFADC(newHit)) {
462                             continue;
463                         }
464                         if (newHit.getRawEnergy() > threshold) {
465                             newHits.add(newHit);
466                         }
467                     }
468                 }
469                 event.put(ecalCollectionName, newHits, CalorimeterHit.class, flags, ecalReadoutName);
470                 event.getMetaData(newHits).setTransient(true);
471                
472             }
473 
474             /*
475              * This is for FADC pulse mode data (Mode-3 or Mode-7):
476              */
477             if (event.hasCollection(RawCalorimeterHit.class, rawCollectionName)) {
478 
479                 /*
480                  * This is for FADC Mode-7 data:
481                  */
482                 if (event.hasCollection(LCRelation.class, extraDataRelationsName)) { // extra information available from
483                                                                                      // mode 7 readout
484                     List<LCRelation> extraDataRelations = event.get(LCRelation.class, extraDataRelationsName);
485                     for (LCRelation rel : extraDataRelations) {
486                         RawCalorimeterHit hit = (RawCalorimeterHit) rel.getFrom();
487                         if (debug) {
488                             System.out.format("old hit energy %d\n", hit.getAmplitude());
489                         }
490                         GenericObject extraData = (GenericObject) rel.getTo();
491                         CalorimeterHit newHit;
492                         newHit = converter.HitDtoA(event, hit, extraData, timeOffset);
493                         if (newHit.getRawEnergy() > threshold) {
494                             if (applyBadCrystalMap && isBadCrystal(newHit)) {
495                                 continue;
496                             }
497                             if (dropBadFADC && isBadFADC(newHit)) {
498                                 continue;
499                             }
500                             if (debug) {
501                                 System.out.format("new hit energy %f\n", newHit.getRawEnergy());
502                             }
503                             newHits.add(newHit);
504                         }
505 
506                     }
507                 } else {
508                     /*
509                      * This is for FADC Mode-3 data:
510                      */
511                     List<RawCalorimeterHit> hits = event.get(RawCalorimeterHit.class, rawCollectionName);
512                     for (RawCalorimeterHit hit : hits) {
513                         if (debug) {
514                             System.out.format("old hit energy %d\n", hit.getAmplitude());
515                         }
516                         CalorimeterHit newHit;
517                         newHit = converter.HitDtoA(event, hit, timeOffset);
518                         if (newHit.getRawEnergy() > threshold) {
519                             if (applyBadCrystalMap && isBadCrystal(newHit)) {
520                                 continue;
521                             }
522                             if (dropBadFADC && isBadFADC(newHit)) {
523                                 continue;
524                             }
525                             if (debug) {
526                                 System.out.format("new hit energy %f\n", newHit.getRawEnergy());
527                             }
528                             newHits.add(newHit);
529                         }
530                     }
531                 }
532                 event.put(ecalCollectionName, newHits, CalorimeterHit.class, flags, ecalReadoutName);
533                 event.getMetaData(newHits).setTransient(true);
534             }
535         } else {
536             ArrayList<RawCalorimeterHit> newHits = new ArrayList<RawCalorimeterHit>();
537             if (event.hasCollection(CalorimeterHit.class, ecalCollectionName)) {
538                 List<CalorimeterHit> hits = event.get(CalorimeterHit.class, ecalCollectionName);
539 
540                 for (CalorimeterHit hit : hits) {
541                     if (debug) {
542                         System.out.format("old hit energy %f\n", hit.getRawEnergy());
543                     }
544                     RawCalorimeterHit newHit = converter.HitAtoD(hit);
545                     if (newHit.getAmplitude() > 0) {
546                         if (debug) {
547                             System.out.format("new hit energy %d\n", newHit.getAmplitude());
548                         }
549                         newHits.add(newHit);
550                     }
551                 }
552                 event.put(rawCollectionName, newHits, RawCalorimeterHit.class, flags, ecalReadoutName);
553                 event.getMetaData(newHits).setTransient(true);
554             }
555         }
556 
557     }

559     /**
560      * Convert physical ID to gain value.
561      *
562      * @param cellID (long)
563      * @return channel constants (EcalChannelConstants)
564      */
565     private EcalChannelConstants findChannel(long cellID) {
566         return ecalConditions.getChannelConstants(ecalConditions.getChannelCollection().findGeometric(cellID));
567     }

569     /**
570      * Return crate number from cellID
571      *
572      * @param cellID (long)
573      * @return Crate number (int)
574      */
575     private int getCrate(long cellID) {
576         // Find the ECAL channel and return the crate number.
577         return ecalConditions.getChannelCollection().findGeometric(cellID).getCrate();
578     }

586     private int getSlot(long cellID) {
587         // Find the ECAL channel and return the slot number.
588         return ecalConditions.getChannelCollection().findGeometric(cellID).getSlot();
