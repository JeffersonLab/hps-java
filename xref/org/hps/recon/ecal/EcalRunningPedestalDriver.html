20  /**
21   * This <code>Driver</code> takes Mode-1 or Mode-7 ECal data and computes a running pedestal average for every channel.
22   * Pedestals from the database will be used if this is not available.
23   * <p>
24   * The following input collections are used:
25   * <ul>
26   * <li>EcalReadoutHits</li>
27   * <li>EcalReadoutExtraDataRelations</li>
28   * </ul>
29   * <p>
30   * Results are by default written to the <b>EcalRunningPedestals</b> output collection. 
31   * 
32   * @todo Timestamps from EVIO for some runs appear to not be monotonically increasing. This interferes with minLookbackTime, 
33   * so it defaults to disabled and its setter is left private for now.
34   */
35  public class EcalRunningPedestalDriver extends Driver {
36  
37      private static final Logger LOGGER = Logger.getLogger(EcalRunningPedestalDriver.class.getPackage().getName());
38  
39      // limit array lengths:
40      private final int limitLookbackEvents = 1000;
41  
42      // minimum number of readouts for running averages:
43      // (if not satisfied, use pedestals from database)
44      private int minLookbackEvents = 5;
45  
46      // maximum number of readouts for running averages:
47      // (if too many, discard the oldest ones)
48      private int maxLookbackEvents = 40;
49  
50      // oldest allowed time for running averages:
51      // (discard older readouts ; negative = no time limit)
52      private long maxLookbackTime = -1; // units = ms
53  
54      private static final String rawCollectionName = "EcalReadoutHits";
55      private static final String extraDataRelationsName = "EcalReadoutExtraDataRelations";
56      private static final String runningPedestalsName = "EcalRunningPedestals";
57  
58      // number of samples from the beginning of the time window used to calculate the pedestal:
59      private static final int nSamples = 4;
60  
61      // TODO: Get this from somewhere else.
62      private final int nChannels = 442;
63  
64      // running pedestal averages, one for each channel:
65      private Map<EcalChannel, Double> runningPedestals = new HashMap<EcalChannel, Double>(nChannels);
66  
67      // recent event-by-event pedestals and timestamps:
68      private Map<EcalChannel, List<Double>> eventPedestals = new HashMap<EcalChannel, List<Double>>();
69      private Map<EcalChannel, List<Long>> eventTimestamps = new HashMap<EcalChannel, List<Long>>();
70  
71      private boolean debug = false;
72      private EcalConditions ecalConditions = null;
73  
74      public EcalRunningPedestalDriver() {
75      }
76  
77      @Override
78      protected void startOfData() {
79          LOGGER.config("minLookbackEvents: " + minLookbackEvents);
80          LOGGER.config("maxLookbackEvents: " + maxLookbackEvents);
81          LOGGER.config("maxLookbackTime:" + maxLookbackTime);
82      }
83  
84      @Override
85      public void detectorChanged(Detector detector) {
86          ecalConditions = DatabaseConditionsManager.getInstance().getEcalConditions();
87          for (int ii = 0; ii < nChannels; ii++) {
88              EcalChannel chan = findChannel(ii + 1);
89              runningPedestals.put(chan, getStaticPedestal(chan));
90              eventPedestals.put(chan, new ArrayList<Double>());
91              eventTimestamps.put(chan, new ArrayList<Long>());
92          }
93          if (debug) {
94              System.out.println("Running and static pedestals better match here:");
95              printPedestals();
96          }
97      }

99      public void setMinLookbackEvents(int nev) {
100         if (nev < 1) {
101             System.err.println("ECalRunningPedestalDriver:  Ignoring minLookbackEvents too small:  " + nev);
102             nev = 1;
103         }
104         minLookbackEvents = nev;
105     }

107     public void setMaxLookbackEvents(int nev) {
108         if (nev > limitLookbackEvents) {
109             System.err.println("ECalRunningPedestalDriver:  Ignoring maxLookbackEvents too big:  " + nev);
110             nev = limitLookbackEvents;
111         }
112         maxLookbackEvents = nev;
113     }

115     private void setMaxLookbackTime(int time) {
116         maxLookbackTime = time;
117     }

119     public void printPedestals() {
120         for (int ii = 1; ii <= nChannels; ii++) {
121             EcalChannel chan = findChannel(ii);
122             if (!runningPedestals.containsKey(chan)) {
123                 System.err.println("printPedestals:   Missing Channel:  " + ii);
124                 continue;
125             }
126             System.out.printf("(%d,%.2f,%.2f) ", chan.getChannelId(), runningPedestals.get(chan),
127                     getStaticPedestal(chan));
128         }
129         System.out.printf("\n");
130     }

132     private double getNSampleMinimum(short samples[]) {
133         double min = 99999999;
134         for (int ii = 0; ii < samples.length - nSamples; ii++) {
135             double tmp = 0;
136             for (int jj = ii; jj < ii + nSamples; jj++)
137                 tmp += samples[jj];
138             tmp /= nSamples;
139             if (tmp < min)
140                 min = tmp;
141         }
142         return min;
143     }

145     @Override
146     protected void process(EventHeader event) {
147 
148         // Mode-7 Input Data:
149         if (event.hasCollection(RawCalorimeterHit.class, rawCollectionName)) {
150             if (event.hasCollection(LCRelation.class, extraDataRelationsName)) {
151                 for (LCRelation rel : event.get(LCRelation.class, extraDataRelationsName)) {
152                     RawCalorimeterHit hit = (RawCalorimeterHit) rel.getFrom();
153                     GenericObject extraData = (GenericObject) rel.getTo();
154                     updatePedestal(event, hit, extraData);
155                 }
156             }
157         }
158 
159         // Mode-1 Input Data:
160         else if (event.hasCollection(RawTrackerHit.class, rawCollectionName)) {
161             List<RawTrackerHit> hits = event.get(RawTrackerHit.class, rawCollectionName);
162             for (RawTrackerHit hit : hits) {
163                 short samples[] = hit.getADCValues();
164                 if (nSamples > samples.length) {
165                     throw new IllegalStateException("Not enough samples for ECal running pedestal.");
166                 }
167 
168                 // double ped = getNSampleMinimum(samples);
169 
170                 boolean good = true;
171                 double ped = 0;
172                 for (int ii = 0; ii < nSamples; ii++) {
173                     // reject pulses from pedestal calculation:
174                     if (samples[ii] > getStaticPedestal(findChannel(hit)) + 12) {
175                         good = false;
176                         break;
177                     }
178                     ped += samples[ii];
179                 }
180                 if (good) {
181                     ped /= nSamples;
182                     updatePedestal(event, findChannel(hit), ped);
183                 }
184             }
185         }
186 
187         event.put(runningPedestalsName, runningPedestals);
188 
189         if (debug) {
190             printPedestals();
191         }
192     }

194     private void updatePedestal(EventHeader event, RawCalorimeterHit hit, GenericObject mode7data) {
195 
196         final int min = ((HitExtraData.Mode7Data) mode7data).getAmplLow();
197         final int max = ((HitExtraData.Mode7Data) mode7data).getAmplHigh();
198 
199         // ignore if pulse at beginning of window:
200         if (max <= 0)
201             return;
202 
203         EcalChannel chan = findChannel(hit);
204         if (chan == null) {
205             System.err.println("hit doesn't correspond to ecalchannel");
206             return;
207         }
208 
209         updatePedestal(event, chan, (double) min);
210     }

212     private void updatePedestal(EventHeader event, EcalChannel chan, double min) {
213 
214         final long timestamp = event.getTimeStamp();
215 
216         List<Double> peds = eventPedestals.get(chan);
217         List<Long> times = eventTimestamps.get(chan);
218 
219         if (maxLookbackTime > 0) {
220             // If new timestamp is older than previous one, restart pedestals.
221             // This should never happen unless firmware counter cycles back to zero,
222             // in which case it could be dealt with if max timestamp is known.
223             if (times.size() > 0 && times.get(0) > timestamp) {
224                 System.err.println(String.format("Event #%d, Old Timestamp:  %d < %d", event.getEventNumber(),
225                         timestamp, times.get(0)));
226                 peds.clear();
227                 times.clear();
228             }
229         }
230 
231         // add pedestal to the list:
232         peds.add(min);
233         times.add(timestamp);
234 
235         if (peds.size() > 1) {
236 
237             // remove oldest pedestal if surpassed limit on #events:
238             if (peds.size() > limitLookbackEvents || (maxLookbackEvents > 0 && peds.size() > maxLookbackEvents)) {
239                 peds.remove(0);
240                 times.remove(0);
241             }
242 
243             // remove old pedestals surpassing limit on lookback time:
244             if (maxLookbackTime > 0) {
245                 while (times.size() > 1) {
246                     if (times.get(0) < timestamp - maxLookbackTime * 1e6) {
247                         times.remove(0);
248                         peds.remove(0);
249                     } else {
250                         break;
251                     }
252                 }
253             }
254         }
255 
256         // Update running pedestal average:
257         double ped = 0;
258         if (peds.size() >= minLookbackEvents) {
259             for (int jj = 0; jj < peds.size(); jj++) {
260                 ped += peds.get(jj);
261             }
262             ped /= peds.size();
263         } else {
264             ped = getStaticPedestal(chan);
265         }
266 
267         runningPedestals.put(chan, ped);
268     }
