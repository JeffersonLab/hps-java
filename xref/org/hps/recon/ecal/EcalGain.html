55       * GAIN * ADC

57      public void setUse2014Gain(boolean use2014Gain) {
58          this.use2014Gain = use2014Gain;

65      /**
66       * return energy (units of GeV) corresponding to the ADC sum and crystal ID
67       */
68      public double adcToEnergy(double adcSum, long cellID) {
69  
70          // Get the channel data.
71          EcalChannelConstants channelData = findChannel(cellID);
72  
73          if (useDAQConfig) {
74              // float gain =
75              // ConfigurationManager.getInstance().getFADCConfig().getGain(ecalConditions.getChannelCollection().findGeometric(cellID));
76              return config.getGain(cellID) * adcSum * EcalUtils.MeV;
77          } else if (use2014Gain) {
78              if (constantGain) {
79                  return adcSum * EcalUtils.gainFactor * EcalUtils.ecalReadoutPeriod;
80              } else {
81                  return channelData.getGain().getGain() * adcSum * EcalUtils.gainFactor * EcalUtils.ecalReadoutPeriod; // should
82                                                                                                                        // not
83                                                                                                                        // be
84                                                                                                                        // used
85                                                                                                                        // for
86                                                                                                                        // the
87                                                                                                                        // moment
88                                                                                                                        // (2014/02)
89              }
90          } else {
91              if (constantGain) {
92                  return gain * adcSum * EcalUtils.MeV;
93              } else {
94                  return channelData.getGain().getGain() * adcSum * EcalUtils.MeV; // gain
95                                                                                   // is
96                                                                                   // defined
97                                                                                   // as
98                                                                                   // MeV/integrated
99                                                                                   // ADC
100             }
101         }
102     }
