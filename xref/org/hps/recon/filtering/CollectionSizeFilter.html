1   package org.hps.recon.filtering;
2   
3   import java.util.List;
4   import org.lcsim.event.EventHeader;
5   
6   /**
7    * Accept events where the specified collection exists and is of the required
8    * size range.
9    */
10  public class CollectionSizeFilter extends EventReconFilter {
11  
12      private String collectionName = "UnconstrainedV0Candidates";
13      private int minSize = 1;
14      private int maxSize = Integer.MAX_VALUE;
15  
16      public void setCollectionName(String collectionName) {
17          this.collectionName = collectionName;
18      }
19  
20      public void setMinSize(int minSize) {
21          this.minSize = minSize;
22      }
23  
24      public void setMaxSize(int maxSize) {
25          this.maxSize = maxSize;
26      }
27  
28      @Override
29      public void process(EventHeader event) {
30          incrementEventProcessed();
31          if (event.hasCollection(Object.class, collectionName)) {
32              List<Object> collection = event.get(Object.class, collectionName);
33              if (collection.size() < minSize || collection.size() > maxSize) {
34                  skipEvent();
35              }
36          } else {
37              skipEvent();
38          }
39          incrementEventPassed();
40      }
41  }
