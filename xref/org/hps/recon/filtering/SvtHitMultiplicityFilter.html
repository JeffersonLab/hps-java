1   package org.hps.recon.filtering;
2   
3   import java.util.ArrayList;
4   import java.util.LinkedHashMap;
5   import java.util.List;
6   import java.util.Map;
7   import java.util.logging.Level;
8   import java.util.logging.Logger;
9   
10  import org.lcsim.detector.tracker.silicon.HpsSiSensor;
11  import org.lcsim.event.EventHeader;
12  import org.lcsim.recon.tracking.digitization.sisim.SiTrackerHitStrip1D;
13  
14  /**
15   * Filter events based on max nr of strip hits
16   */
17  public class SvtHitMultiplicityFilter extends EventReconFilter {
18  
19      Logger logger = Logger.getLogger(SvtHitMultiplicityFilter.class.getSimpleName());
20      private final static String stripClusterCollectionName = "StripClusterer_SiTrackerHitStrip1D";
21      private int hitsPerSensor = 1;
22      private int minHitsPerHalf = 3;
23  
24      public SvtHitMultiplicityFilter() {
25          logger.setLevel(Level.WARNING);
26      }
27  
28      @Override
29      protected void process(EventHeader event) {
30  
31          incrementEventProcessed();
32  
33          // Make sure there are any strips in the event
34          if (!event.hasCollection(SiTrackerHitStrip1D.class, stripClusterCollectionName))
35              skipEvent();
36  
37          // Find all strip clusters in the events
38          List<SiTrackerHitStrip1D> stripClusters = event.get(SiTrackerHitStrip1D.class, stripClusterCollectionName);
39  
40          Map<String, List<SiTrackerHitStrip1D>> sensorHitMap = new LinkedHashMap<String, List<SiTrackerHitStrip1D>>();
41  
42          int nhits[] = {0, 0};
43  
44          for (SiTrackerHitStrip1D cluster : stripClusters) {
45              String sensorName = cluster.getRawHits().get(0).getDetectorElement().getName();
46              boolean isTop = ((HpsSiSensor) cluster.getRawHits().get(0).getDetectorElement()).isTopLayer();
47              if (isTop)
48                  nhits[0]++;
49              else
50                  nhits[1]++;
51  
52              List<SiTrackerHitStrip1D> hits;
53              if (sensorHitMap.containsKey(sensorName))
54                  hits = sensorHitMap.get(sensorName);
55              else {
56                  hits = new ArrayList<SiTrackerHitStrip1D>();
57                  sensorHitMap.put(sensorName, hits);
58              }
59              hits.add(cluster);
60          }
61  
62          // if none of the halves contains the required nr of hits, skip the event
63          if (nhits[0] < minHitsPerHalf && nhits[1] < minHitsPerHalf)
64              skipEvent();
65  
66          // go through and check that the number of hits for each layer is what's required
67          for (Map.Entry<String, List<SiTrackerHitStrip1D>> entry : sensorHitMap.entrySet()) {
68              if (entry.getValue().size() != hitsPerSensor)
69                  skipEvent();
70          }
71  
72          StringBuffer sb = new StringBuffer();
73          sb.append("Event with " + stripClusters.size() + " hits passed:\n");
74          for (SiTrackerHitStrip1D hit : stripClusters) {
75              sb.append(hit.getPositionAsVector().toString() + " "
76                      + hit.getRawHits().get(0).getDetectorElement().getName() + "\n");
77          }
78          logger.info(sb.toString());
79  
80          incrementEventPassed();
81  
82      }
83  
84      public void setHitsPerSensor(int hitsPerSensor) {
85          this.hitsPerSensor = hitsPerSensor;
86      }
87  
88  }
