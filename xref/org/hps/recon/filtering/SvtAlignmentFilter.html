221     private Cluster findClosestCluster(Hep3Vector posonhelix, List<Cluster> clusters) {
222         Cluster closest = null;
223         double minDist = 9999;
224         for (Cluster cluster : clusters) {
225             double[] clPos = cluster.getPosition();
226             double clEne = cluster.getEnergy();
227             double dist = Math.sqrt(Math.pow(clPos[0] - posonhelix.x(), 2) + Math.pow(clPos[1] - posonhelix.y(), 2)); //coordinates!!!
228             if (dist < minDist && clEne > 0.4) {
229                 closest = cluster;
230                 minDist = dist;
231             }
232         }
233         return closest;
234     }
