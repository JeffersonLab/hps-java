1   package org.hps.recon.filtering;
2   
3   import java.util.logging.Level;
4   import java.util.logging.Logger;
5   
6   import org.lcsim.event.EventHeader;
7   
8   /**
9    * Accept only events where all of the specified flags exist and have a value of 1.
10   */
11  public class EventFlagFilter extends EventReconFilter {
12  
13      private static Logger LOGGER = Logger.getLogger(EventFlagFilter.class.getPackage().getName());
14      private static Level LEVEL = Level.FINE;
15  
16      String[] flagNames = {"svt_bias_good", "svt_position_good", "svt_burstmode_noise_good", "svt_event_header_good", "svt_latency_good", "svt_readout_overlap_good"};
17  
18      public void setFlagNames(String[] flagNames) {
19          this.flagNames = flagNames;
20      }
21  
22      @Override
23      public void process(EventHeader event) {
24          incrementEventProcessed();
25          if (flagNames != null) {
26              for (String flagName : flagNames) {
27                  int[] flag = event.getIntegerParameters().get(flagName);
28                  //System.out.println("flag <" + flagName + ">");
29                  if (flag == null || flag[0] == 0) {
30                      LOGGER.log(LEVEL, "Skipping event <" + event.getEventNumber() + "> from flag <" + flagName + ">");
31                      skipEvent();
32                  }
33              }
34          }
35          LOGGER.fine("Event passed");
36          incrementEventPassed();
37      }
38  }
