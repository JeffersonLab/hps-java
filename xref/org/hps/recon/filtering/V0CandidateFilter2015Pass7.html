1   package org.hps.recon.filtering;

3   import java.util.List;
4   import org.hps.recon.particle.ReconParticleDriver;
5   import org.hps.recon.tracking.TrackType;
6   import org.hps.record.triggerbank.AbstractIntData;
7   import org.hps.record.triggerbank.TIData;
8   import org.lcsim.event.EventHeader;
9   import org.lcsim.event.GenericObject;
10  import org.lcsim.event.ReconstructedParticle;
11  
12  /**
13   * Class to strip off trident candidates for the 2015 Pass7 Reconstruction.
14   * pairs1 trigger. SVT bias on. SVT at 0.5 mm. no DAQ errors. demand top/bottom
15   * e+e- pair.
16   */
17  public class V0CandidateFilter2015Pass7 extends EventReconFilter {
18  
19      private String _V0CandidateCollectionName = "UnconstrainedV0Candidates";
20  
21      @Override
22      protected void process(EventHeader event) {
23          incrementEventProcessed();
24  
25          //is there a V0 candidate in this event
26          if (!event.hasCollection(ReconstructedParticle.class, _V0CandidateCollectionName)) {
27              skipEvent();
28          }
29          List<ReconstructedParticle> V0Candidates = event.get(ReconstructedParticle.class, _V0CandidateCollectionName);
30          if (V0Candidates.size() == 0) {
31              skipEvent();
32          }
33  
34          // only keep pair1 triggers:
35          if (!event.hasCollection(GenericObject.class, "TriggerBank")) {
36              skipEvent();
37          }
38          boolean isPairs1 = false;
39          for (GenericObject gob : event.get(GenericObject.class, "TriggerBank")) {
40              if (!(AbstractIntData.getTag(gob) == TIData.BANK_TAG)) {
41                  continue;
42              }
43              TIData tid = new TIData(gob);
44              if (tid.isPair1Trigger()) {
45                  isPairs1 = true;
46                  break;
47              }
48          }
49          if (!isPairs1) {
50              skipEvent();
51          }
52  
53          // SVT bias on
54          int[] flag = event.getIntegerParameters().get("svt_bias_good");
55          if (flag == null || flag[0] == 0) {
56              skipEvent();
57          }
58  
59          // SVT is at correct position
60          flag = event.getIntegerParameters().get("svt_position_good");
61          if (flag == null || flag[0] == 0) {
62              skipEvent();
63          }
64  
65          // SVT latency is correct
66          flag = event.getIntegerParameters().get("svt_latency_good");
67          if (flag == null || flag[0] == 0) {
68              skipEvent();
69          }
70  
71          int nV0 = 0;
72          for (ReconstructedParticle v0 : V0Candidates) {
73              ReconstructedParticle electron = v0.getParticles().get(ReconParticleDriver.ELECTRON);
74              ReconstructedParticle positron = v0.getParticles().get(ReconParticleDriver.POSITRON);
75              //double-check on GBL (should never have non-GBL tracks in pass7, but...
76              if (!TrackType.isGBL(v0.getType())) { // we only care about GBL vertices
77                  continue;
78              }
79              // require e+ and e- to be in opposite hemispheres
80              boolean isElectronTop = electron.getTracks().get(0).getTrackStates().get(0).getTanLambda() > 0;
81              boolean isPositronTop = positron.getTracks().get(0).getTrackStates().get(0).getTanLambda() > 0;
82  
83              if (isElectronTop != isPositronTop) {
84                  nV0++;
85              }
86          }
87          if (nV0 == 0) {
88              skipEvent();
89          }
90          incrementEventPassed();
91      }
92  
93      /**
94       * Name of V0 Candidate ReconstructedParticle Collection Name
95       *
96       * @param s
97       */
98      public void setV0CandidateCollectionName(String s) {
99          _V0CandidateCollectionName = s;
100     }
101 
102 }
