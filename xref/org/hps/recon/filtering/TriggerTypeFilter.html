1   package org.hps.recon.filtering;
2   
3   import org.hps.record.triggerbank.AbstractIntData;
4   import org.hps.record.triggerbank.TIData;
5   import org.lcsim.event.EventHeader;
6   import org.lcsim.event.GenericObject;
7   
8   /**
9    * Accept events with the desired trigger bit.
10   */
11  public class TriggerTypeFilter extends EventReconFilter {
12  
13      TriggerType triggerType = TriggerType.all;
14  
15      public enum TriggerType {
16  
17          all, singles0, singles1, pairs0, pairs1
18      }
19  
20      public void setTriggerType(String triggerType) {
21          this.triggerType = TriggerType.valueOf(triggerType);
22      }
23  
24      @Override
25      public void process(EventHeader event) {
26          incrementEventProcessed();
27          if (event.hasCollection(GenericObject.class, "TriggerBank")) {
28              for (GenericObject data : event.get(GenericObject.class, "TriggerBank")) {
29                  if (AbstractIntData.getTag(data) == TIData.BANK_TAG) {
30                      TIData triggerData = new TIData(data);
31                      if (!matchTriggerType(triggerData))//only process singles0 triggers...
32                      {
33                          skipEvent();
34                      }
35                  }
36              }
37          } else {
38              skipEvent();
39          }
40          incrementEventPassed();
41      }
42  
43      public boolean matchTriggerType(TIData triggerData) {
44          switch (triggerType) {
45              case all:
46                  return true;
47              case singles0:
48                  return triggerData.isSingle0Trigger();
49              case singles1:
50                  return triggerData.isSingle1Trigger();
51              case pairs0:
52                  return triggerData.isPair0Trigger();
53              case pairs1:
54                  return triggerData.isPair1Trigger();
55              default:
56                  return false;
57          }
58      }
59  }
