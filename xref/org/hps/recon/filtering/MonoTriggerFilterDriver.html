1   package org.hps.recon.filtering;
2   
3   import org.lcsim.event.EventHeader;
4   import org.lcsim.event.GenericObject;
5   import org.lcsim.util.Driver;
6   import org.hps.record.epics.EpicsData;
7   import org.hps.record.scalers.ScalerData;
8   import org.hps.record.triggerbank.AbstractIntData;
9   import org.hps.record.triggerbank.TIData;
10  
11  /**
12   * Keep events with exactly one trigger bit. Also keep EPICS events, and Scaler events. Drop all other events.
13   */
14  public class MonoTriggerFilterDriver extends Driver {
15  
16      public void process(EventHeader event) {
17  
18          // 1. keep all events with EPICS data (could also use event tag = 31):
19          if (EpicsData.read(event) != null)
20              return;
21  
22          // 2. keep all events with Scaler data:
23          if (ScalerData.read(event) != null)
24              return;
25  
26          // 3. drop event if it doesn't have a TriggerBank
27          if (!event.hasCollection(GenericObject.class, "TriggerBank"))
28              throw new Driver.NextEventException();
29  
30          int triggers = 0;
31          // 4. keep event if it only has one trigger bit set:
32          for (GenericObject gob : event.get(GenericObject.class, "TriggerBank")) {
33              if (!(AbstractIntData.getTag(gob) == TIData.BANK_TAG))
34                  continue;
35              TIData tid = new TIData(gob);
36              if (tid.isSingle0Trigger())
37                  triggers++;
38              if (tid.isSingle1Trigger())
39                  triggers++;
40              if (tid.isPair0Trigger())
41                  triggers++;
42              if (tid.isPair1Trigger())
43                  triggers++;
44              if (tid.isPulserTrigger())
45                  triggers++;
46              if (tid.isCalibTrigger())
47                  triggers++;
48          }
49          if (triggers == 1)
50              /* tab */return;
51  
52          // 5. Else, drop event:
53          throw new Driver.NextEventException();
54      }
55  }
