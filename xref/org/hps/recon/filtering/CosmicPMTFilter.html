1   package org.hps.recon.filtering;
2   
3   import org.hps.recon.ecal.FADCGenericHit;
4   import org.lcsim.event.EventHeader;
5   import org.lcsim.event.GenericObject;
6   import org.lcsim.geometry.Detector;
7   import org.lcsim.util.Driver;
8   import org.lcsim.util.aida.AIDA;
9   
10  public class CosmicPMTFilter extends Driver {
11  
12      // class storing raw PMT readout:
13      static final String CLASSNAME = "FADCGenericHits";
14  
15      // hardware location of PMTs:
16      static final int CRATE = 2; // this is hps2, crate #37 in EVIO
17      static final int SLOT = 20;
18      static final int CHANNELS[] = { 13, 14 };
19  
20      // cuts on pulse integrals for cosmic signals (units ADC):
21      static final float CUTS[] = { 2000, 4000 };
22  
23      // number of samples to use at beginning of window for pedestal:
24      static final int NPEDSAMP = 20;
25  
26      AIDA aida = AIDA.defaultInstance();
27  
28      public void detectorChanged(Detector detector) {
29          aida.tree().cd("/");
30          aida.histogram2D("CosmicPMTs", 400, 0, 10000, 400, 0, 10000);
31      }
32  
33      public void process(EventHeader event) {
34  
35          // find PMT data:
36          FADCGenericHit pmt1 = null, pmt2 = null;
37          if (!event.hasCollection(GenericObject.class, CLASSNAME)) throw new Driver.NextEventException();
38          for (GenericObject gob : event.get(GenericObject.class, CLASSNAME)) {
39              FADCGenericHit hit = (FADCGenericHit) gob;
40  
41              if (hit.getCrate() == CRATE && hit.getSlot() == SLOT) {
42                  if (hit.getChannel() == CHANNELS[0])
43                      pmt1 = hit;
44                  else if (hit.getChannel() == CHANNELS[1]) pmt2 = hit;
45              }
46          }
47          if (pmt1 == null || pmt2 == null) throw new Driver.NextEventException();
48  
49          // calculate and histogram pulse integrals:
50          float pulse1 = getPulseIntegral(pmt1);
51          float pulse2 = getPulseIntegral(pmt2);
52          // aida.histogram2D("CosmicPMTs").fill(pulse1,pulse2);
53  
54          // cut on pulse integrals:
55          if (pulse1 < CUTS[0] || pulse2 < CUTS[1]) throw new Driver.NextEventException();
56      }
57  
58      public float getPulseIntegral(FADCGenericHit hh) {
59          float sum = 0;
60          for (int samp : hh.getData())
61              sum += samp;
62          return sum - getPedestal(hh) * hh.getData().length;
63      }
64  
65      public float getPedestal(FADCGenericHit hh) {
66          if (hh.getData().length < NPEDSAMP) throw new java.lang.RuntimeException("Not enough samples for pedestal.");
67          float sum = 0;
68          for (int isamp = 0; isamp < NPEDSAMP; isamp++)
69              sum += hh.getData()[isamp];
70          return sum / NPEDSAMP;
71      }
72  }
