1   package org.hps.recon.filtering;
2   
3   import java.util.List;
4   import org.lcsim.event.EventHeader;
5   import org.lcsim.event.ReconstructedParticle;
6   
7   public class V0SkimDriver extends EventReconFilter {
8   
9       private String v0CollectionName = "UnconstrainedV0Candidates";
10  
11      /**
12       * Sets the V0 candidate collection to use.
13       *
14       * @param val
15       */
16      public void setV0CollectionName(String val) {
17          this.v0CollectionName = val;
18      }
19  
20      public void process(EventHeader event) {
21          incrementEventProcessed();
22          List<ReconstructedParticle> vertices = event.get(ReconstructedParticle.class, v0CollectionName);
23          if (vertices.size() == 0) {
24              skipEvent();
25          }
26          incrementEventPassed();
27      }
28  }
