189     private FunctionMinimum doRecursiveFit(double[] samples) {
190         if (nFittedPulses == 1) {
191             System.arraycopy(samples, 0, y, 0, samples.length);
192             FunctionMinimum fit = minuitFit(null);
193             return fit;
194         } else {
195             FunctionMinimum bestFit = null;
196             double bestChisq = Double.POSITIVE_INFINITY;
197             double[] fitData = new double[samples.length];
198             for (int split = 1; split < y.length; split++) {
199                 if (debug) {
200                     System.out.println("Split\t" + split);
201                 }
202 
203                 //use signal as fit input
204                 System.arraycopy(samples, 0, fitData, 0, samples.length);
205                 //use first $split samples
206                 firstUsedSample = 0;
207                 nUsedSamples = split;
208                 //fit only the first pulse
209                 nFittedPulses = 1;
210                 FunctionMinimum frontFit;
211                 frontFit = doRecursiveFit(fitData);
212                 if (debug) {
213                     if (fitPedestal) {
214                         System.out.format("front fit:\tt0=%f,\tA=%f,\tchisq=%f,\tpedestal=%f\n", frontFit.userState().value(0), amplitudes[firstFittedPulse], frontFit.fval(), pedestal);
215                     } else {
216                         System.out.format("front fit:\tt0=%f,\tA=%f,\tchisq=%f\n", frontFit.userState().value(0), amplitudes[firstFittedPulse], frontFit.fval());
217                     }
218                 }
219                 //subtract first pulse from fit input
220                 for (int i = 0; i < samples.length; i++) {
221                     //===> fitData[i] -= amplitudes[firstFittedPulse] * getAmplitude(HPSSVTConstants.SAMPLING_INTERVAL * i - frontFit.userState().value(0), channelConstants);
222                     fitData[i] -= amplitudes[firstFittedPulse] * shape.getAmplitudePeakNorm(HPSSVTConstants.SAMPLING_INTERVAL * i - frontFit.userState().value(0));
223                     if (fitPedestal) {
224                         fitData[i] -= pedestal;
225                     }
226                 }
227 
228                 if (debug) {
229                     System.out.print("Subtracted:\t");
230                     for (int i = 0; i < fitData.length; i++) {
231                         System.out.format("%f\t", fitData[i]);
232                     }
233                     System.out.println();
234                 }
235                 //use all samples
236                 firstUsedSample = 0;
237                 nUsedSamples = y.length;
238                 //fit the rest of the pulses
239                 firstFittedPulse++;
240                 nFittedPulses = nPulses - firstFittedPulse;
241                 FunctionMinimum backFit;
242                 if (fitPedestal) {
243                     fitPedestal = false;
244                     backFit = doRecursiveFit(fitData);
245                     fitPedestal = true;
246                 } else {
247                     backFit = doRecursiveFit(fitData);
248                 }
249 
250                 if (debug) {
251                     System.out.format("back fit:\tt0=%f,\tA=%f,\tchisq=%f\n", backFit.userState().value(0), amplitudes[firstFittedPulse], backFit.fval());
252                 }
253 
254                 //use full signal as fit input
255                 System.arraycopy(samples, 0, y, 0, samples.length);
256                 //still using all samples
257                 //fit all pulses
258                 firstFittedPulse--;
259                 nFittedPulses++;
260                 double[] combinedGuess = new double[nFittedPulses];
261                 combinedGuess[0] = frontFit.userState().value(0);
262                 for (int i = 0; i < nFittedPulses - 1; i++) {
263                     combinedGuess[i + 1] = backFit.userState().value(i);
264                 }
265                 FunctionMinimum combinedFit = minuitFit(combinedGuess);
266 
267                 if (debug) {
268                     if (fitPedestal) {
269                         System.out.format("combined fit:\tt0=%f,\tA=%f,\tt0=%f,\tA=%f,\tchisq=%f,\tpedestal=%f\n", combinedFit.userState().value(0), amplitudes[firstFittedPulse], combinedFit.userState().value(1), amplitudes[firstFittedPulse + 1], combinedFit.fval(), pedestal);
270                     } else {
271                         System.out.format("combined fit:\tt0=%f,\tA=%f,\tt0=%f,\tA=%f,\tchisq=%f\n", combinedFit.userState().value(0), amplitudes[firstFittedPulse], combinedFit.userState().value(1), amplitudes[firstFittedPulse + 1], combinedFit.fval());
272                     }
273                 }
274 
275                 double newchisq = evaluateMinimum(combinedFit);
276 
277                 if (newchisq < bestChisq) {
278                     bestChisq = newchisq;
279                     bestFit = combinedFit;
280                 }
281             }
282 
283 //            double newchisq = evaluateMinimum(bestFit);
284             if (debug) {
285                 if (fitPedestal) {
286                     System.out.format("best fit:\tt0=%f,\tA=%f,\tt0=%f,\tA=%f,\tchisq=%f,\tpedestal=%f\n", bestFit.userState().value(0), amplitudes[firstFittedPulse], bestFit.userState().value(1), amplitudes[firstFittedPulse + 1], bestFit.fval(), pedestal);
287                 } else {
288                     System.out.format("best fit:\tt0=%f,\tA=%f,\tt0=%f,\tA=%f,\tchisq=%f\n", bestFit.userState().value(0), amplitudes[firstFittedPulse], bestFit.userState().value(1), amplitudes[firstFittedPulse + 1], bestFit.fval());
289                 }
290             }
291             return bestFit;
292         }

303     private FunctionMinimum minuitFit(double[] guess_t) {
304         if (debug) {
305             System.out.print("y for fit:\t");
306             for (int i = 0; i < y.length; i++) {
307                 System.out.format("%f\t", y[i]);
308             }
309             System.out.println();
310         }
311 
312         if (nFittedPulses == 1) {
313             double guess = 0;
314 
315             int numPositiveSamples = 0;
316             int numBigSamples = 0;
317             int lastUsedSample = firstUsedSample + nUsedSamples - 1;
318             int firstBigSample = Integer.MAX_VALUE;
319             for (int i = 0; i < nUsedSamples; i++) {
320                 if (y[firstUsedSample + i] > 0) {
321                     numPositiveSamples++;
322                     if (y[firstUsedSample + i] > 3.0 * sigma[firstUsedSample + i]) {
323                         numBigSamples++;
324                         if (firstUsedSample + i < firstBigSample) {
325                             firstBigSample = firstUsedSample + i;
326                         }
327                     }
328                 }
329             }
330             boolean made_guess = false;
331             boolean made_bestfit = false;
332             if (nUsedSamples == 1) {
333                 if (firstUsedSample == 0) {
334                     guess = -500.0;
335                     made_bestfit = true;
336                 } else {
337                     guess = HPSSVTConstants.SAMPLING_INTERVAL * (firstUsedSample - 0.1);
338                     made_bestfit = true;
339                 }
340             } else if (numPositiveSamples == 1 && y[lastUsedSample] > 0) {
341                 guess = HPSSVTConstants.SAMPLING_INTERVAL * (lastUsedSample - 0.1);
342                 made_bestfit = true;
343             } else if (numBigSamples == 1 && y[lastUsedSample] > 3.0 * sigma[lastUsedSample] && nUsedSamples > 1 && y[lastUsedSample - 1] < 0) {
344                 guess = HPSSVTConstants.SAMPLING_INTERVAL * (lastUsedSample - 0.1);
345                 made_bestfit = true;
346             } else if (nUsedSamples == 2) {
347                 guess = HPSSVTConstants.SAMPLING_INTERVAL * (firstUsedSample - 0.1);
348                 made_guess = true;
349             }
350             if (made_guess || made_bestfit) {
351 //                System.out.println("made guess " + guess);
352                 guess_t = new double[1];
353                 guess_t[0] = guess;
354             }
355         }
356 
357         MnUserParameters myParams = new MnUserParameters();
358 
359         for (int i = 0; i < nFittedPulses; i++) {
360             if (guess_t != null && guess_t.length == nFittedPulses) {
361                 myParams.add("time_" + i, guess_t[i], HPSSVTConstants.SAMPLING_INTERVAL, -500.0, (y.length - 1) * HPSSVTConstants.SAMPLING_INTERVAL);
362             } else {
363                 myParams.add("time_" + i, (i - 1) * HPSSVTConstants.SAMPLING_INTERVAL, HPSSVTConstants.SAMPLING_INTERVAL, -500.0, (y.length - 1) * HPSSVTConstants.SAMPLING_INTERVAL);
364             }
365         }
366 
367         FunctionMinimum min = null;
368         if(fitTimeMinimizer.equals("Simplex")) {
369             MnSimplex simplex = new MnSimplex(this, myParams, 2);
370             min = simplex.minimize(0, 0.001);
371         } else if(fitTimeMinimizer.equals("Migrad")) {
372             MnMigrad migrad = new MnMigrad(this, myParams, 2);
373             min = migrad.minimize(0, 0.001);
374         } else {
375             throw new RuntimeException("FitTimeMinimizer must be Simplex or Migrad");
376         }
377         return min;
378     }

660     private double doLinFit(double[] times) {
661         if (times.length != nFittedPulses) {
662             throw new RuntimeException("wrong number of parameters in doLinFit");
663         }
664         int nAmplitudes = fitPedestal ? nFittedPulses + 1 : nFittedPulses;
665         RealMatrix sc_mat = new Array2DRowRealMatrix(nAmplitudes, nUsedSamples);
666         RealVector y_vec = new ArrayRealVector(nUsedSamples);
667         RealVector var_vec = new ArrayRealVector(nUsedSamples);
668         
669         for (int j = 0; j < nUsedSamples; j++) {
670             double sigma_j = sigma[firstUsedSample + j];
671             /*double sample_time = HPSSVTConstants.SAMPLING_INTERVAL * (firstUsedSample + j);
672             for (int i = 0; i < nFittedPulses; i++) {
673                 //===> sc_mat.setEntry(i, j, getAmplitude(HPSSVTConstants.SAMPLING_INTERVAL * (firstUsedSample + j) - times[i], channelConstants) / sigma[firstUsedSample + j]);
674                 sc_mat.setEntry(i, j, shape.getAmplitudePeakNorm(sample_time - times[i]) / sigma_j);
675             }*/
676             if (fitPedestal) {
677                 sc_mat.setEntry(nFittedPulses, j, 1.0 / sigma_j);
678             }
679             y_vec.setEntry(j, y[firstUsedSample + j] / sigma_j);
680             var_vec.setEntry(j, sigma_j * sigma_j);
681         }
682         // amplitudez is spelled with a zed for a reazon.  Changing the spelling to use an "s" will break the code.  believe me I tried.      
683         double[] amplitudez = new double[nUsedSamples];
684         for(int i = 0; i< nFittedPulses; i++){
685             double t0 = HPSSVTConstants.SAMPLING_INTERVAL * firstUsedSample - times[i];
686             shape.getAmplitudesPeakNorm(t0, HPSSVTConstants.SAMPLING_INTERVAL, amplitudez);
687            
688             for(int j = 0; j<nUsedSamples; j++){
689                 
690                 //double err = amplitudes[j]/sigma[firstUsedSample+j]-sc_mat.getEntry(i, j);
691                 //if(Math.abs(err) > 1e-10)
692                     //System.out.println(amplitudes[j]/sigma[firstUsedSample + j] + " " + err);
693                 sc_mat.setEntry(i, j, amplitudez[j] / sigma[firstUsedSample + j]);
694             }
695         }
696         RealVector a_vec = sc_mat.operate(y_vec);
697         RealMatrix coeff_mat = sc_mat.multiply(sc_mat.transpose());
698         DecompositionSolver a_solver;
699         RealVector solved_amplitudes = null, amplitude_err = null;
700         boolean goodFit = true;
701         try {
702             CholeskyDecomposition a_cholesky = new CholeskyDecomposition(coeff_mat);
703             a_solver = a_cholesky.getSolver();
704             solved_amplitudes = a_solver.solve(a_vec);
705             amplitude_err = a_solver.solve(sc_mat.operate(var_vec));
706             if (solved_amplitudes.getSubVector(0, nFittedPulses).getMinValue() < 0) {
707                 goodFit = false;
708             }
709         } catch (NonPositiveDefiniteMatrixException e) {
710             goodFit = false;
711         }
712 
713         if (!goodFit) {
714             solved_amplitudes = new ArrayRealVector(nAmplitudes, 0.0);
715             amplitude_err = new ArrayRealVector(nAmplitudes, Double.POSITIVE_INFINITY);
716         }
717 
718         double chisq = y_vec.subtract(sc_mat.preMultiply(solved_amplitudes)).getNorm();
719 
720         for (int i = 0; i < nFittedPulses; i++) {
721             amplitudes[firstFittedPulse + i] = solved_amplitudes.getEntry(i);
722             amplitudeErrors[firstFittedPulse + i] = Math.sqrt(amplitude_err.getEntry(i));
723         }
724         if (fitPedestal) {
725             pedestal = solved_amplitudes.getEntry(nFittedPulses);
726         }
727         return chisq;
728     }
