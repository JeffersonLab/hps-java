136         // Cache Detector object.
137         this.detector = detector;
138 
139         // Get B-field Y with no sign. Seed Tracker doesn't like signed B-field components.
140         // FIXME Is this always right?
141         this.bfield = Math.abs((detector.getFieldMap().getField(new BasicHep3Vector(0, 0, 0)).y()));
142         if (debug) {
143             System.out.println("Set B-field to " + this.bfield);
144         }
145 
146         initialize();
147 
148         super.detectorChanged(detector);
149     }

154     private void initialize() {
155         //
156         // 1) Driver to create HelicalTrackHits expected by Seedtracker.
157         //
158         // TODO Make this step its own separate Driver??? (Matt)
159 
160         // Setup default stereo pairings, which should work for even number of
161         // modules.
162         List<SiTrackerModule> modules = detector.getSubdetector(subdetectorName).getDetectorElement().findDescendants(SiTrackerModule.class);
163         if (modules.size() == 0) {
164             throw new RuntimeException("No SiTrackerModules found in detector.");
165         }
166 
167         // Create the Driver.
168         HelicalTrackHitDriver hthdriver = new HelicalTrackHitDriver();
169         hthdriver.addCollection(stripHitsCollectionName);
170         hthdriver.setOutputCollectionName(hthOutputCollectionName);
171         // hthdriver.setHitRelationName(helicalTrackHitRelationsCollectionName);
172         // hthdriver.setMCRelationName(helicalTrackMCRelationsCollectionName);
173 
174         // hthdriver.setStripMaxSeparation(stripMaxSeparation);
175         // hthdriver.setStripTolerance(stripTolerance); // user parameter?
176         hthdriver.setTransformToTracking(true);
177         hthdriver.setDebug(true);
178         add(hthdriver);
179 
180         //
181         // 2) Driver to run Seed Tracker.
182         //
183         sFinallist = StrategyXMLUtils.getStrategyListFromInputStream(this.getClass().getResourceAsStream(strategyResource));
184         // 3) Cleanup the readouts for next event.
185         //
186         List<String> readoutCleanup = new ArrayList<String>();
187         readoutCleanup.add(this.simTrackerHitCollectionName);
188         add(new ReadoutCleanupDriver(readoutCleanup));
189 
190     }

202     public void process(EventHeader event) {
203         // This call runs the track reconstruction using the sub-Drivers.
204         super.process(event);
205         double maxChi2 = 250;
206         List<HelicalTrackHit> hth = event.get(HelicalTrackHit.class, stInputCollectionName);
207 
208         System.out.println("The HelicalTrackHit collection " + hthOutputCollectionName + " has " + hth.size() + " hits.");
209         List<HelicalTrackHit> hitsLayer1 = getLayerHits(hth, 1);
210         List<HelicalTrackHit> hitsLayer3 = getLayerHits(hth, 3);
211         List<HelicalTrackHit> hitsLayer5 = getLayerHits(hth, 5);
212         List<HelicalTrackHit> hitsLayer7 = getLayerHits(hth, 7);
213         List<HelicalTrackHit> hitsLayer9 = getLayerHits(hth, 9);
214 
215         List<SeedCandidate> seedtracks = new ArrayList<SeedCandidate>();
216         for (HelicalTrackHit h1 : hitsLayer1) {
217             for (HelicalTrackHit h3 : hitsLayer3) {
218                 for (HelicalTrackHit h5 : hitsLayer5) {
219                     for (HelicalTrackHit h7 : hitsLayer7) {
220                         // for (HelicalTrackHit h9 : hitsLayer9) {
221 
222                         SeedCandidate seed = new SeedCandidate(sFinallist.get(0), bfield);
223                         seed.addHit(h1);
224                         seed.addHit(h3);
225                         seed.addHit(h5);
226                         seed.addHit(h7);
227                         // seed.addHit(h9);
228 
229                         HelicalTrackFit fitRes = fit(seed);
230                         if (fitRes != null) {
231                             seed.setHelix(fitRes);
232                             if (fitRes.chisq()[1] > maxChi2)
233                                 continue;
234 
235                             seedtracks.add(seed);
236                             // System.out.println(fitRes.toString());
237                         }
238                         // }
239                     }
240                 }
241             }
242         }
243 
244         addTracksToEvent(event, seedtracks, bfield);
245         // Debug printouts.
246         if (debug) {
247             // Check for HelicalTrackHits.
248 
249             // Check for Tracks.
250             List<Track> tracks = event.get(Track.class, trackCollectionName);
251             System.out.println("The Track collection " + trackCollectionName + " has " + tracks.size() + " tracks.");
252 
253             // Print out track info.
254             for (Track track : tracks) {
255                 System.out.println(track.toString());
256                 System.out.println("chi2 = " + track.getChi2());
257             }
258         }
259 
260         // Increment number of events.
261         ++nevents;
262 
263         // Add to tracks found.
264         ntracks += event.get(Track.class, trackCollectionName).size();
265     }

267     public HelicalTrackFit fit(SeedCandidate seed) {
268         List<HelicalTrackHit> hitcol = seed.getHits();
269         Map<HelicalTrackHit, MultipleScatter> msmap = new HashMap<HelicalTrackHit, MultipleScatter>();
270         SlopeInterceptLineFitter _lfitter = new SlopeInterceptLineFitter();
271         SlopeInterceptLineFit _lfit;
272         boolean success = false;
273         // Check if we have enough pixel hits to do a straight-line fit of s vs z
274         int npix = hitcol.size();
275         // Calculate the arc lengths from the DCA to each hit and check for backwards hits
276         Map<HelicalTrackHit, Double> smap = getPathLengths(hitcol);
277         // Create the objects that will hold the fit output
278         double[] chisq = new double[2];
279         int[] ndof = new int[2];
280         ndof[0] = 0;
281         chisq[0] = 0;
282         double[] par = new double[5];
283         SymmetricMatrix cov = new SymmetricMatrix(5);
284 
285         // Setup for the line fit
286         double[] s = new double[npix];
287         double[] z = new double[npix];
288         double[] dz = new double[npix];
289 
290         // Store the coordinates and errors for the line fit
291         for (int i = 0; i < npix; i++) {
292             HelicalTrackHit hit = hitcol.get(i);
293             z[i] = hit.z();
294             System.out.println(hit.getCorrectedCovMatrix().toString());
295             dz[i] = Math.sqrt(hit.getCorrectedCovMatrix().e(2, 2));
296             s[i] = smap.get(hit);
297             System.out.println(z[i] + " " + dz[i] + " " + s[i]);
298         }
299 
300         // Call the line fitter and check for success
301         success = _lfitter.fit(s, z, dz, npix);
302         if (!success) {
303             return null;
304         }
305 
306         // Save the line fit, chi^2, and DOF
307         _lfit = _lfitter.getFit();
308         chisq[1] = _lfit.chisquared();
309         ndof[1] = npix - 2;
310 
311         // Save the line fit parameters
312         par[HelicalTrackFit.z0Index] = _lfit.intercept();
313         par[HelicalTrackFit.slopeIndex] = _lfit.slope();
314         par[HelicalTrackFit.curvatureIndex] = 0.0001;
315         par[HelicalTrackFit.dcaIndex] = 0.0001;
316         par[HelicalTrackFit.phi0Index] = 0.000;
317 
318         // Save the line fit covariance matrix elements
319         cov.setElement(HelicalTrackFit.z0Index, HelicalTrackFit.z0Index, Math.pow(_lfit.interceptUncertainty(), 2));
320         cov.setElement(HelicalTrackFit.z0Index, HelicalTrackFit.slopeIndex, _lfit.covariance());
321         cov.setElement(HelicalTrackFit.slopeIndex, HelicalTrackFit.slopeIndex, Math.pow(_lfit.slopeUncertainty(), 2));
322         cov.setElement(HelicalTrackFit.curvatureIndex, HelicalTrackFit.curvatureIndex, 0);
323         cov.setElement(HelicalTrackFit.curvatureIndex, HelicalTrackFit.phi0Index, 0);
324         cov.setElement(HelicalTrackFit.phi0Index, HelicalTrackFit.phi0Index, 0);
325         cov.setElement(HelicalTrackFit.curvatureIndex, HelicalTrackFit.dcaIndex, 0); // fix d0 sign
326                                                                                      // convention
327         cov.setElement(HelicalTrackFit.phi0Index, HelicalTrackFit.dcaIndex, 0); // fix d0 sign
328                                                                                 // convention
329         cov.setElement(HelicalTrackFit.dcaIndex, HelicalTrackFit.dcaIndex, 0);
330 
331         // Create the HelicalTrackFit for this helix
332         return new HelicalTrackFit(par, cov, chisq, ndof, smap, msmap);
333 
334     }

336     private Map<HelicalTrackHit, Double> getPathLengths(List<HelicalTrackHit> hits) {
337 
338         // Create a map to store the arc lengths
339         Map<HelicalTrackHit, Double> smap = new HashMap<HelicalTrackHit, Double>();
340 
341         // Initialize looper tracking and iterate over ordered list of hits
342         double slast = 0.;
343         int ilast = -1;
344         double s;
345         for (int i = 0; i < hits.size(); i++) {
346 
347             // Retrieve the next hit ordered by z coordinate and check hit type
348             HelicalTrackHit hit = hits.get(i);
349             if (hit instanceof HelicalTrack2DHit) {
350 
351                 // Axial hit - measure from the DCA (can't handle loopers)
352                 // s = HelixUtils.PathLength(_cfit, hit);
353                 s = hit.getPosition()[0];
354                 // Save the arc length for this hit
355                 smap.put(hit, s);
356             }
357         }
358         return smap;
359     }

378     private void addTracksToEvent(EventHeader event, List<SeedCandidate> seedlist, double bfield) {
379         // Create a the track list
380         List<Track> tracks = new ArrayList<Track>();
381 
382         // Initialize the reference point to the origin
383         double[] ref = new double[3];
384         ref[0] = 0.;
385         ref[1] = 0.;
386         ref[2] = 0.;
387         // Loop over the SeedCandidates that have survived
388         for (SeedCandidate trackseed : seedlist) {
389 
390             // Create a new SeedTrack (SeedTrack extends BaseTrack)
391             SeedTrack trk = new SeedTrack();
392 
393             // Add the hits to the track
394             for (HelicalTrackHit hit : trackseed.getHits()) {
395                 trk.addHit((TrackerHit) hit);
396             }
397 
398             // Retrieve the helix and save the relevant bits of helix info
399             HelicalTrackFit helix = trackseed.getHelix();
400             trk.setTrackParameters(helix.parameters(), bfield);
401             trk.setCovarianceMatrix(helix.covariance());
402             trk.setChisq(helix.chisqtot());
403             trk.setNDF(helix.ndf()[0] + helix.ndf()[1]);
404 
405             // Flag that the fit was successful and set the reference point
406             trk.setFitSuccess(true);
407             trk.setReferencePoint(ref);
408             trk.setRefPointIsDCA(true);
409 
410             // Set the strategy used to find this track
411             trk.setStratetgy(trackseed.getSeedStrategy());
412 
413             // Set the SeedCandidate this track is based on
414             trk.setSeedCandidate(trackseed);
415 
416             // Add the track to the list of tracks
417             tracks.add((Track) trk);
418         }
419 
420         // Put the tracks back into the event and exit
421         int flag = 1 << LCIOConstants.TRBIT_HITS;
422         event.put(trackCollectionName, tracks, Track.class, flag);
423     }
