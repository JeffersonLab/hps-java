52      public ITransform3D GetGlobalToLocal(HelicalTrackStrip strip) {
53          // Transform from JLab frame (RawTrackerHit) to sensor frame (i.e. u,v,w)
54          RawTrackerHit rth = (RawTrackerHit) strip.rawhits().get(0);
55          IDetectorElement ide = rth.getDetectorElement();
56          SiSensor sensor = ide.findDescendants(SiSensor.class).get(0);
57          SiSensorElectrodes electrodes = sensor.getReadoutElectrodes(ChargeCarrier.HOLE);        
58          return electrodes.getGlobalToLocal();
59      }

61      public Hep3Matrix getStripToTrackRotation(HelicalTrackStrip strip) {
62          // This function transforms the hit to the tracking coordinates
63  
64          // Transform from JLab frame to sensor frame (done through the RawTrackerHit)
65          ITransform3D detToStrip = GetGlobalToLocal(strip);
66          // Get rotation matrix
67          Hep3Matrix detToStripMatrix = (BasicHep3Matrix) detToStrip.getRotation().getRotationMatrix();
68          // Transformation between the JLAB and tracking coordinate systems
69          Hep3Matrix detToTrackMatrix = (BasicHep3Matrix) CoordinateTransformations.getMatrix();
70  
71          if (_DEBUG) {
72              System.out.println("gblToLoc translation:");
73              System.out.println(detToStrip.getTranslation().toString());
74              System.out.println("gblToLoc Rotation:");
75              System.out.println(detToStrip.getRotation().toString());
76              System.out.println("detToTrack Rotation:");
77              System.out.println(detToTrackMatrix.toString());
78          }
79  
80          return (Hep3Matrix) VecOp.mult(detToTrackMatrix, VecOp.inverse(detToStripMatrix));
81      }

137     public Hep3Vector CalculateStripUncertaintyInGlobalFrame(HelicalTrackStrip strip, HelicalTrackFit trk, double msdrdphi, double msdz) {
138 
139         if (_DEBUG)
140             System.out.println("--- CalculateStripUncertainyInGlobalFrame ---");
141         if (_DEBUG)
142             System.out.println("Strip origin = " + strip.origin().toString());
143         Hep3Vector u = strip.u();
144         Hep3Vector v = strip.v();
145         Hep3Vector w = strip.w();
146         Hep3Vector corigin = strip.origin();
147 
148         double phi0 = trk.phi0();
149         double R = trk.R();
150         double xint = strip.origin().x();
151         // double xint = this.calculateHelixInterceptXPlane(_trk, strip);
152         double s = HelixUtils.PathToXPlane(trk, xint, 0, 0).get(0);
153         double phi = -s / R + phi0;
154         if (_DEBUG)
155             System.out.println("phi0 " + phi0 + " R " + R + " xint " + xint + " s " + s + " phi " + phi);
156         // if(_DEBUG) System.out.println("trkpos = "+trkpos.toString());
157         // if(_DEBUG) System.out.println("origin = "+corigin.toString());
158 
159         Hep3Vector mserr = new BasicHep3Vector(msdrdphi * Math.sin(phi), msdrdphi * Math.sin(phi), msdz);
160         if (_DEBUG)
161             System.out.println("msdrdphi = " + msdrdphi + " msdz = " + msdz);
162         if (_DEBUG)
163             System.out.println("mserr = " + mserr.toString());
164         double uHitError = strip.du();
165         double msuError = VecOp.dot(mserr, u);
166         double uError = Math.sqrt(uHitError * uHitError + msuError * msuError);
167         if (_DEBUG)
168             System.out.println("uError = " + uError + "(MS " + msuError + ",u=" + u.toString() + ")");
169         double vHitError = (strip.vmax() - strip.vmin()) / Math.sqrt(12);
170         double msvError = VecOp.dot(mserr, v);
171         double vError = Math.sqrt(vHitError * vHitError + msvError * msvError);
172         if (_DEBUG)
173             System.out.println("vError = " + vError + "(MS " + msvError + ",v=" + v.toString() + ")");
174 
175         double wHitError = 10.0 / Math.sqrt(12); // 0.001;
176         double mswError = VecOp.dot(mserr, w);
177         double wError = Math.sqrt(wHitError * wHitError + mswError * mswError);
178         if (_DEBUG)
179             System.out.println("wError = " + wError + "(MS " + mswError + ",w=" + w.toString() + ")");
180 
181         Hep3Vector dq_local = new BasicHep3Vector(uError, vError, wError);
182         if (_DEBUG)
183             System.out.println("dq_local = " + dq_local.toString());
184         Hep3Matrix trackToStripRot = getTrackToStripRotation(strip);
185         if (_DEBUG)
186             System.out.println("trackToStripRot:\n " + trackToStripRot.toString());
187         Hep3Matrix stripToTrackRot = VecOp.inverse(trackToStripRot);
188         if (_DEBUG)
189             System.out.println("stripToTrackRot:\n " + stripToTrackRot.toString());
190         Hep3Vector dq_global = VecOp.mult(stripToTrackRot, dq_local);
191         if (_DEBUG)
192             System.out.println("q_global = " + dq_global.toString());
193         return dq_global;
194     }

196     public List<SimTrackerHit> stripClusterToSimHits(HelicalTrackStrip strip, List<SimTrackerHit> simTrackerHits, boolean stripsInTrackingFrame) {
197 
198         int layer = strip.layer();
199         Hep3Vector stripPosition = this.getClusterPosition(strip, stripsInTrackingFrame);
200 
201         // Sort the SimTrackerHits by Layer
202         Map<Integer, List<SimTrackerHit>> layerToSimTrackerHit = new HashMap<Integer, List<SimTrackerHit>>();
203         for (SimTrackerHit simTrackerHit : simTrackerHits) {
204             if (!layerToSimTrackerHit.containsKey(simTrackerHit.getLayer()))
205                 layerToSimTrackerHit.put(simTrackerHit.getLayer(), new ArrayList<SimTrackerHit>());
206             layerToSimTrackerHit.get(simTrackerHit.getLayer()).add(simTrackerHit);
207         }
208 
209         //
210         List<SimTrackerHit> simhits = new ArrayList<SimTrackerHit>();
211 
212         if (layerToSimTrackerHit.get(strip.layer()) == null) {
213             System.out.println(this.getClass().getSimpleName() + ": WARNING there is a strip in layer " + strip.layer() + " but no SimTrackerHits");
214             return simhits;
215         }
216 
217         // If there is only a single SimTrackerHit on a layer and it's in the same volume as the
218         // strip hit then they likely match to each other
219         if (layerToSimTrackerHit.get(strip.layer()).size() == 1) {
220             Hep3Vector simTrackerHitPosition = layerToSimTrackerHit.get(strip.layer()).get(0).getPositionVec();
221             if (Math.signum(simTrackerHitPosition.y()) == Math.signum(stripPosition.z())) {
222                 simhits.add(layerToSimTrackerHit.get(strip.layer()).get(0));
223                 layerToSimTrackerHit.remove(strip.layer());
224                 if (_DEBUG) {
225                     System.out.println(this.getClass().getSimpleName() + ": SimTrackerHit position: " + simTrackerHitPosition.toString());
226                     System.out.println(this.getClass().getSimpleName() + ": Cluster position: " + stripPosition.toString());
227                 }
228             } else {
229                 System.out.println(this.getClass().getSimpleName() + ": Cluster and SimTrackerHit are on different volumes");
230             }
231         } else if (layerToSimTrackerHit.get(strip.layer()).size() > 1) {
232             if (_DEBUG)
233                 System.out.println(this.getClass().getSimpleName() + ": found " + layerToSimTrackerHit.get(strip.layer()).size() + " SimTrackerHits to match to strip in layer " + strip.layer());
234             // System.exit(1);
235             double deltaZ = Double.MAX_VALUE;
236             SimTrackerHit simTrackerHitMatch = null;
237             for (SimTrackerHit simTrackerHit : layerToSimTrackerHit.get(strip.layer())) {
238                 if (Math.abs(simTrackerHit.getPositionVec().y() - stripPosition.z()) < deltaZ) {
239                     deltaZ = Math.abs(simTrackerHit.getPositionVec().y() - stripPosition.z());
240                     simTrackerHitMatch = simTrackerHit;
241                 }
242             }
243             simhits.add(simTrackerHitMatch);
244             layerToSimTrackerHit.remove(strip.layer()).remove(simTrackerHitMatch);
245             if (_DEBUG) {
246                 System.out.println(this.getClass().getSimpleName() + ": SimTrackerHit position: " + simTrackerHitMatch.getPositionVec().toString());
247                 System.out.println(this.getClass().getSimpleName() + ": Cluster position: " + stripPosition);
248             }
249         }
250 
251         return simhits;
252     }

255      * Make a SimTrackerHit {@link IIdentifier} for a given layer number
256      * 
257      * @param sensor : The sensor on which the SimTrackerHit is created on
258      * @return A 32-bit SimTrackerHit identifier
259      */
260     public static IIdentifier makeSimTrackerHitId(SiSensor sensor) {
261 
262         // Get the sensors identifier
263         IExpandedIdentifier id = new ExpandedIdentifier(sensor.getExpandedIdentifier());
264 
265         // Get the helper and dictionary
266         IIdentifierHelper helper = sensor.getIdentifierHelper();
267         IIdentifierDictionary dictionary = helper.getIdentifierDictionary();
268 
269         // Fill in the layer number
270         //===> id.setValue(dictionary.getFieldIndex("layer"), SvtUtils.getInstance().getLayerNumber(sensor));
271         id.setValue(dictionary.getFieldIndex("layer"), ((HpsSiSensor) sensor).getLayerNumber());
272 
273         // Pack and return the identifier
274         return helper.pack(id);
275 
276     }
277     //gets the sensor name that hps-java assigns from the simple name used in hpstr (and saved in e.g. the timing calibration resource files)
278     //  example:  simple name = "L0T_axial" --> "module_L1b_module_axial_sensor0"  
279     public String getSensorNameFromSimpleName(String simpleName){
280         //the format of simpleName == LX[T/B]_[axial/stereo]_[[[slot/hole if necessary]]]
281         //
282         String[] splitSimpleName=simpleName.split("_"); 
283         int lenSimpleName=splitSimpleName.length;
284         String layerSimple=splitSimpleName[0]; 
285         String axSt=splitSimpleName[1]; 
286         String slHoSimple="";
287         if(lenSimpleName == 3 )
288             slHoSimple=splitSimpleName[2]; 
289         String sensorLayer;  
290         if (layerSimple.contains("T"))
291             sensorLayer=layerSimple.replace("T","t");
292         else
293             sensorLayer=layerSimple.replace("B","b");
294         String sensorSlHo=""; 
295         if(lenSimpleName==3){
296             if(slHoSimple=="pos")
297                 sensorSlHo="slot";
298             else
299                 sensorSlHo="hole";
300         }
301 
302         String sensorName="";
303         String sensorNamePre="module_"+sensorLayer+"_module_"+axSt+"_";
304         if(lenSimpleName==4)
305             sensorName=sensorNamePre+sensorSlHo+"_sensor0"; 
306         else
307             sensorName=sensorNamePre+"_sensor0"; 
308         return sensorName;   
309     }
310     // just the flip of above...get the simple name from sensor name
311     public String getSimpleNameFromSensorName(String sensorName){
312         //the format of simpleName == LX[T/B]_[axial/stereo]_[[[slot/hole if necessary]]]
313         //format for sensor name is:  "module_LX[t/b]_module_[axial/stereo]_[[[slot/hole if necessary]]]_sensor0"
314         String[] splitSensorName=sensorName.split("_"); 
315         int lenSensorName=splitSensorName.length;
316         String layerSensor=splitSensorName[1]; 
317         String axSt=splitSensorName[3]; 
318         String slHoSensor="";
319         //decrement the layer number...
320         Scanner in=new Scanner(layerSensor).useDelimiter("[^0-9]+");
321         int laySensor=in.nextInt();
322         int laySimple=laySensor-1; 
323         String layerSensorMid=layerSensor.replace(Integer.toString(laySensor),Integer.toString(laySimple)); 
324 
325         if(lenSensorName == 6 )
326             slHoSensor=splitSensorName[4]; 
327         String simpleLayer="";  
328         if (layerSensorMid.contains("t"))
329             simpleLayer=layerSensorMid.replace("t","T");
330         else
331             simpleLayer=layerSensorMid.replace("b","B");
332         String simpleSlHo=""; 
333         //        System.out.println(sensorName+" "+slHoSensor);
334         if(lenSensorName==6){
335             if(slHoSensor.contains("slot"))
336                 simpleSlHo="pos";
337             else
338                 simpleSlHo="ele";
339         }
340 
341         String simpleName="";
342         String simpleNamePre=simpleLayer+"_"+axSt;
343         if(lenSensorName==6)
344             simpleName=simpleNamePre+"_"+simpleSlHo; 
345         else
346             simpleName=simpleNamePre;
347         return simpleName;   
348     }
