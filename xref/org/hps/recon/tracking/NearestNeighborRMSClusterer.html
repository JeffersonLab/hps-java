37       * RMS noise of the channel(s).

39       * @param seed_threshold seed threshold
40       * @param neighbor_threshold neighbor threshold
41       * @param cluster_threshold cluster threshold
42       */
43      public NearestNeighborRMSClusterer(double seed_threshold, double neighbor_threshold, double cluster_threshold) {
44          _seed_threshold = seed_threshold;
45          _neighbor_threshold = neighbor_threshold;
46          _cluster_threshold = cluster_threshold;
47      }

98      /**
99       * Use a nearest neighbor algorithm to create clusters. 
100      *
101      * @param fittedHits Collection of fitted raw tracker hits to cluster.
102      * @return Collection of fitted hits
103      */
104     @Override
105     public List<List<LCRelation>> findClusters(List<LCRelation> fittedHits) {
106 
107         // Check that the seed threshold is at least as large as the neighbor 
108         // threshold
109         if (_seed_threshold < _neighbor_threshold) {
110             throw new RuntimeException("Tracker hit clustering error: seed threshold below neighbor threshold");
111         }
112 
113         // Create maps that show the channel status and relate the channel 
114         // number to the raw hit and vice versa
115         int mapsize = 2 * fittedHits.size();
116         Set<Integer> clusterableSet = new HashSet<Integer>(mapsize);
117 
118         Map<Integer, LCRelation> channel_to_hit = new HashMap<Integer, LCRelation>(mapsize);
119 
120         // Create list of channel numbers to be used as cluster seeds
121         List<Integer> cluster_seeds = new ArrayList<Integer>();
122 
123         // Loop over the raw hits and construct the maps used to relate cells and hits, initialize
124         // the
125         // clustering status map, and create a list of possible cluster seeds
126         for (LCRelation fittedHit : fittedHits) {
127 
128             RawTrackerHit rawHit = FittedRawTrackerHit.getRawTrackerHit(fittedHit); 
129             
130             // get the channel number for this hit
131             SiTrackerIdentifierHelper sid_helper = (SiTrackerIdentifierHelper) rawHit.getIdentifierHelper();
132             IIdentifier id = rawHit.getIdentifier();
133             int channel_number = sid_helper.getElectrodeValue(id);
134 
135             // Check for duplicate RawTrackerHits or channel numbers
136             if (channel_to_hit.containsKey(channel_number)) {
137                 //TODO: be smarter about this
138                 if (Math.abs(FittedRawTrackerHit.getT0(channel_to_hit.get(channel_number))) 
139                         < Math.abs(FittedRawTrackerHit.getT0(fittedHit))) {
140                     continue;
141                 }
142             }
143 
144             // Add this hit to the maps that relate channels and hits
145             channel_to_hit.put(channel_number, fittedHit);
146 
147             // Get the signal from the readout chip
148             double signal = FittedRawTrackerHit.getAmp(fittedHit); 
149             double noiseRMS = 0; 
150             for(int sampleN = 0; sampleN < HPSSVTConstants.TOTAL_NUMBER_OF_SAMPLES; sampleN++){
151                 noiseRMS += ((HpsSiSensor) rawHit.getDetectorElement()).getNoise(channel_number, sampleN);
152             }
153             noiseRMS = noiseRMS/HPSSVTConstants.TOTAL_NUMBER_OF_SAMPLES;
154             
155             // Mark this hit as available for clustering if it is above the neighbor threshold
156             if (signal / noiseRMS >= _neighbor_threshold && passChisqCut(fittedHit)) {
157                 clusterableSet.add(channel_number);
158             }
159 
160             // Add this hit to the list of seeds if it is above the seed threshold
161             if (signal / noiseRMS >= _seed_threshold && passTimingCut(fittedHit) && passChisqCut(fittedHit)) {
162                 cluster_seeds.add(channel_number);
163             }
164         }
165 
166         // Create a list of clusters
167         // TODO: Create a cluster class instead 
168         List<List<LCRelation>> cluster_list = new ArrayList<List<LCRelation>>();
169 
170         // Now loop over the cluster seeds to form clusters
171         for (int seed_channel : cluster_seeds) {
172 
173             if (!clusterableSet.contains(seed_channel)) {
174                 continue;
175             }
176 
177             // Create a new cluster
178             List<LCRelation> cluster = new ArrayList<LCRelation>();
179             double cluster_signal = 0.;
180             double cluster_noise_squared = 0.;
181             double cluster_weighted_time = 0.;
182 
183             // Create a queue to hold channels whose neighbors need to be checked for inclusion
184             LinkedList<Integer> unchecked = new LinkedList<Integer>();
185 
186             // Add the seed channel to the unchecked list and mark it as unavailable for clustering
187             unchecked.addLast(seed_channel);
188             clusterableSet.remove(seed_channel);
189 
190             // Check the neighbors of channels added to the cluster
191             while (unchecked.size() > 0) {
192 
193                 // Pull the next channel off the queue and add it's hit to the cluster
194                 int clustered_cell = unchecked.removeFirst();
195                 cluster.add(channel_to_hit.get(clustered_cell));
196                 LCRelation hit = channel_to_hit.get(clustered_cell);
197                 cluster_signal += FittedRawTrackerHit.getAmp(hit);
198                 double strip_noise = 0; 
199                 for(int sampleN = 0; sampleN < HPSSVTConstants.TOTAL_NUMBER_OF_SAMPLES; sampleN++){
200                     strip_noise += ((HpsSiSensor) FittedRawTrackerHit.getRawTrackerHit(hit).getDetectorElement()).getNoise(clustered_cell, sampleN);
201                 }
202                 strip_noise = strip_noise/HPSSVTConstants.TOTAL_NUMBER_OF_SAMPLES;
203                 cluster_noise_squared += Math.pow(strip_noise, 2); 
204                 cluster_weighted_time += FittedRawTrackerHit.getT0(hit) * FittedRawTrackerHit.getAmp(hit);
205                 Collection<Integer> neighbor_channels = getNearestNeighborCells(clustered_cell);
206 
207                 // Now loop over the neighbors and see if we can add them to the cluster
208                 for (int channel : neighbor_channels) {
209 
210                     // Check if this neighbor channel is still available for clustering
211                     if (!clusterableSet.contains(channel)) {
212                         continue;
213                     }
214 
215                     LCRelation neighbor_hit = channel_to_hit.get(channel);
216                     if (Math.abs(FittedRawTrackerHit.getT0(neighbor_hit) - cluster_weighted_time / cluster_signal) > _neighborDeltaT) {
217                         continue;
218                     }
219 
220                     // Add channel to the list of unchecked clustered channels
221                     // and mark it unavailable for clustering
222                     unchecked.addLast(channel);
223                     clusterableSet.remove(channel);
224 
225                 } // end of loop over neighbor cells
226             } // end of loop over unchecked cells
227 
228             // Finished with this cluster, check cluster threshold and add it to the list of
229             // clusters
230             if (cluster.size() > 0 && cluster_signal / Math.sqrt(cluster_noise_squared) > _cluster_threshold) {
231                 cluster_list.add(cluster);
232             }
233 
234         } // End of loop over seeds
235 
236         // Finished finding clusters
237         return cluster_list;
238     }

240     private boolean passTimingCut(LCRelation fittedHit) {
241         double time = FittedRawTrackerHit.getT0(fittedHit);
242         return (Math.abs(time - _meanTime) < _timeWindow);
243     }
244 
245     private boolean passChisqCut(LCRelation fittedHit) {
246         return FittedRawTrackerHit.getChi2Prob(fittedHit) > _minChiProb; 
247     }

249     public Collection<Integer> getNearestNeighborCells(int cell) {
250         Collection<Integer> neighbors = new ArrayList<Integer>(2);
251         for (int ineigh = -1; ineigh <= 1; ineigh = ineigh + 2) {
252             int neighbor_cell = cell + ineigh;
253             if (isValidCell(neighbor_cell)) {
254                 neighbors.add(neighbor_cell);
255             }
256         }
257         return neighbors;
258     }
259 
260     public boolean isValidCell(int cell) {
261         return (cell >= 0 && cell < HPSSVTConstants.TOTAL_STRIPS_PER_SENSOR);
262     }
