1   package org.hps.recon.tracking;
2   
3   import java.util.HashMap;
4   import java.util.Map;
5   
6   /**
7    * Enum constants for different {@link org.lcsim.event.Track} objects based on what tracking
8    * strategy was used.  The type is defined by comparing the tracking strategy
9    * name to the name of all the enum constants.
10   */
11  public enum StrategyType {
12     
13      // NOTE: The values of these enums should never be changed.  Any additional
14      //       enum constants should be simply added to the end of the list.
15      
16      /** Enum values corresponding to different tracking strategies */
17      /** 
18       * This is the default TrackType value set by the reconstruction. It  
19       * represents the following track finding algorithms for the first
20       * two passes:
21       * 
22       * pass 1 - Seed 123, confirm 4, extend 56
23       * pass 2 - Seed 345, confirm 2, extend 16
24       * 
25       * For pass 3? and beyond, this should no longer be needed since multiple 
26       * versions of the same track will no longer exist.
27       */
28      MATCHED_TRACKS(1),
29      /** Seed 345, confirm 2, extend 16 */
30      S345_C2_E16(2),
31      /** Seed 456, confirm 3, extend 21 */
32      S456_C3_E21(3), 
33      /** Seed 123, confirm 4, extend 56 */
34      S123_C4_E56(4),
35      /** Seed 123, confirm 5, extend 46 */
36      S123_C5_E46(5);
37      
38      private int type; 
39   
40      /** Map from an enum constant value to an enum constant */
41      private static Map<Integer, StrategyType> typeValueMap = new HashMap<Integer, StrategyType>();
42      
43      /** Map from an enum constant name to an enum constant */
44      private static Map<String, StrategyType> typeNameMap = new HashMap<String, StrategyType>();
45     
46      /** Static block used to fill all maps */
47      static { 
48          for (StrategyType strategyType : StrategyType.values()) { 
49              typeValueMap.put(strategyType.getType(), strategyType);
50              typeNameMap.put(strategyType.name(), strategyType);
51          }
52      }
53      
54      /** Constructor */
55      StrategyType(int type) { 
56          this.type = type; 
57      }
58     
59      /**
60       * Returns the enum constant of this enum type
61       * 
62       * @return returns the enum constant of this enum type 
63       */
64      public int getType() { 
65          return type; 
66      }
67      
68      /**
69       * Returns the {@link StrategyType} with the specified name.  If a 
70       * {@link StrategyType} with that name doesn't exist, it returns null.
71       * 
72       * @param name The name of the StrategyType enum constant to return
73       * @return The {@link StrategyType} enum constant of the given name or
74       *         null if it doesn't exist. 
75       */
76      public static StrategyType getType(String name) {
77          return typeNameMap.get(name);
78      }
79      
80      /**
81       * Returns the {@link StrategyType} with the specified type value. If a 
82       * {@link StrategyType} with that type value doesn't exist, it returns 
83       * null.
84       * 
85       * @param typeValue The type value of the StrategyType enum constant to 
86       *        return
87       * @return The {@link StrategyType} enum constant with the given type 
88       *         value or null if it doesn't exist.
89       */
90      public static StrategyType getType(int typeValue) { 
91          return typeValueMap.get(typeValue);
92      }
93   
94      /**
95       * Returns true if the specified {@link StrategyType} is equal to this enum
96       * constant.
97       * 
98       * @param strategyType enum constant to check
99       * @return true if the specified {@link StrategyType} is equal to this enum
100      *         constant
101      */
102     public boolean equals(StrategyType strategyType) { 
103         return strategyType.getType() == getType(); 
104     }
105 }
