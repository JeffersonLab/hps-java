15  /**
16   * Basic readout chip class. This class supports the minimal functions expected of a readout chip.
17   * The charge on a strip/pixel is digitized as an integer number with a simple ADC with
18   * programmable resolution and dynamic range. A chip with 1-bit ADC resolution (binary readout) is
19   * treated as a special case.
20   * 
21   * Noise is added to strips with charge and random noise hits are generated as well. Methods are
22   * provided to decode the charge and time (although the current implementation always returns a
23   * time of 0).
24   * 
25   * This implementation has thresholds that are settable in units of RMS noise of each channel to
26   * enable simluation of highly optimized readout chains. If absolute thresholds are desired,
27   * GenericReadoutChip should be used instead.
28   */
29  // FIXME: Is this a copy-paste job from SiSim? What behavior is changed from there?
30  public class NoiselessReadoutChip implements ReadoutChip {

32      private BasicChannel _channel = new BasicChannel();
33      private ADC _adc = new ADC();
34      private boolean dropBadChannels = false;

98      /**
99       * Given a collection of electrode data (i.e., charge on strips/pixels), return a map
100      * associating the channel and it's list of raw data.
101      * 
102      * @param data electrode data from the charge distribution
103      * @param electrodes strip or pixel electrodes
104      * @return map containing the ADC counts for this sensor
105      */
106     @Override
107     public SortedMap<Integer, List<Integer>> readout(SiElectrodeDataCollection data, SiSensorElectrodes electrodes) {
108 
109         // If there is no electrode data for this readout chip, create an empty
110         // electrode data collection
111         if (data == null) {
112             data = new SiElectrodeDataCollection();
113         }
114 
115         // Add noise hits to the electrode data collection
116         // addNoise(data, electrodes);
117 
118         // return the digitized charge data as a map that associates a hit
119         // channel with a list of raw data for the channel
120         return digitize(data, electrodes);
121     }

123     /**
124      * Decode the hit charge stored in the RawTrackerHit
125      * 
126      * @param hit raw hit
127      * @return hit charge in units of electrons

129     @Override

131         return getADC().decodeCharge(hit.getADCValues()[0]);

145     /**
146      * Digitizes the hit channels in a SiElectrodeDataCollection.
147      * 
148      * The SiElectrodeDataCollection is a map that associates a given channel with it's
149      * SiElectrodeData. The SiElectrodeData encapsulates the deposited charge on an strip/pixel and
150      * any associated SimTrackerHits.
151      * 
152      * The output of this class is a map that associates a channel number with a list of raw data
153      * 
154      * @param data electrode data collection
155      * @return map associating channels with a list of raw data
156      */
157     private SortedMap<Integer, List<Integer>> digitize(SiElectrodeDataCollection data, SiSensorElectrodes electrodes) {
158         // Create the map that associates a given sensor channel with it's list of raw data
159         SortedMap<Integer, List<Integer>> chip_data = new TreeMap<Integer, List<Integer>>();
160 
161         // Loop over the channels contained in the SiElectrodeDataCollection
162         for (Integer channel : data.keySet()) {
163             
164             if(dropBadChannels && ((HpsSiSensor) electrodes.getDetectorElement()).isBadChannel(channel)){
165             //===> if (dropBadChannels && HPSSVTCalibrationConstants.isBadChannel((SiSensor) electrodes.getDetectorElement(), channel)) {
166                 // System.out.format("%d bad\n", channel);
167                 continue;
168             }
169             // System.out.format("%d OK\n", channel);
170             // Fetch the electrode data for this channel
171             SiElectrodeData eldata = data.get(channel);
172 
173             // Get the charge in units of electrons
174             double charge = eldata.getCharge();
175 
176             // Calculate the ADC value for this channel and make sure it is positive
177             int adc = getADC().convert(charge);
178             if (adc <= 0) {
179                 continue;
180             }
181 
182             // Create a list containing the adc value - for the basic readout
183             // there is only 1 word of raw data
184             List<Integer> channel_data = new ArrayList<Integer>();
185             channel_data.add(adc);
186 
187             // Save the list of raw data in the chip_data map
188             chip_data.put(channel, channel_data);
189         }
190 
191         return chip_data;
192     }

194     /**
195      * BasicChannel class representing a single channel's behavior
196      * 
197      * Note that binary readout is a special case. Anything positive value passed to a binary ADC
198      * for digitization is assumed to have crossed t hreshold and is assigned a value of 1.
199      * Decoding binary readout results in either 0 or dynamic_range.
200      */
201     private class BasicChannel implements ReadoutChannel {
202 
203         private double _noise_intercept = 0.;
204         private double _noise_slope = 0.;
205 
206         /**
207          * Set the noise (in electrons) for 0 capacitance
208          * 
209          * @param noise_intercept noise intercept
210          */
211         private void setNoiseIntercept(double noise_intercept) {
212             _noise_intercept = noise_intercept;
213         }
214 
215         /**
216          * Set the capacitative noise slope (in electrons / pF)
217          * 
218          * @param noise_slope noise slope
219          */
220         private void setNoiseSlope(double noise_slope) {
221             _noise_slope = noise_slope;
222         }
223 
224         /**
225          * Return the noise in electrons for a given strip/pixel capacitance
226          * 
227          * @param capacitance capacitance in pF
228          * @return noise in electrons
229          */
230         public double computeNoise(double capacitance) {
231             return _noise_intercept + _noise_slope * capacitance;
232         }
233     }
234 
235     /**
236      * ADC class representing analog to digital converter.
237      */
238     private class ADC {
239 
240         private int _nbits = 8;
241         private double _dynamic_range = 20.;
242 
243         /**
244          * Set the ADC resolution in number of bits.
245          * 
246          * @param nbits number of bits
247          */
248         private void setNbits(int nbits) {
249             _nbits = nbits;
250         }
251 
252         /**
253          * Set the dynamic range in fC
254          * 
255          * @param dynamic range
256          */
257         private void setDynamicRange(double dynamic_range) {
258             _dynamic_range = dynamic_range;
259         }
260 
261         /**
262          * Compute the maximum ADC value
263          * 
264          * @return largest possible ADC value according to # of bits
265          */
266         private int maxADCValue() {
267             return (int) Math.pow(2, _nbits) - 1;
268         }
269 
270         /**
271          * Compute the conversion constant in ADC/fC
272          * 
273          * @return conversion constant for ADC
274          */
275         private double conversionConstant() {
276             return maxADCValue() / _dynamic_range;
277         }
278 
279         /**
280          * Perform analog to digital conversion
281          * 
282          * @return digital ADC output between 0 and maxADCValue
283          */
284         public int convert(double charge) {
285             if (_nbits != 1) {
286                 return Math.max(0, Math.min(maxADCValue(), (int) Math.floor(charge * 1.602e-4 * conversionConstant())));
287             } else {
288                 if (charge <= 0.0) {
289                     return 0;
290                 } else {
291                     return 1;
292                 }
293             }
294         }
295 
296         /**
297          * Decode charge from ADC value
298          * 
299          * @return charge specified by a given ADC value
300          */
301         public double decodeCharge(int adc_value) {
302             if (_nbits != 1) {
303                 return (adc_value + 0.5) / (1.602e-4 * conversionConstant());
304             } else {
305                 return adc_value * _dynamic_range;
306             }
307 
308         }
309     }
