13  /**
14   * Convert a helix to a straight line track at a specified reference plane normal to the x axis.
15   * This code was developed for simulating the Heavy Photon Search experiment where the target is
16   * located outside the magnetic field volume.
17   */
18  // FIXME: If this is copy-pasted from lcsim, what behavior was changed here?  Is it needed? --JM
19  public class HelixConverter {
20  
21      private double _xref;
22  
23      /**
24       * Constructor for the HelixConverter used to convert helices to StraightLineTracks at the
25       * magnetic field boundary. The reference point xref identifies the x coordinate that specifies
26       * the magnetic field boundary. The StraightLineTracks produced by this class will use xref as
27       * their reference point.
28       * 
29       * @param xref x coordinate that specifies the magnetic field boundary
30       */
31      public HelixConverter(double xref) {
32          _xref = xref;
33      }
34  
35      /**
36       * Convert a helix to a StraightLineTrack.
37       * 
38       * @param helix helix to be converted
39       * @return resulting StraightLineTrack
40       */
41      public StraightLineTrack Convert(HelicalTrackFit helix) {
42  
43          // Get helix parameters used in this calculation
44          double RC = helix.R();
45          double xc = helix.xc();
46          double yc = helix.yc();
47          double phi0 = helix.phi0();
48          double d0 = helix.dca();
49          double slope = helix.slope();
50          double z0 = helix.z0();
51  
52          // First find path length to reference point
53          double arg = (xc - _xref) / RC;
54          if (Math.abs(arg) > 1.0)
55              return null;
56          double phi = Math.asin(arg);
57          double dphi = phi0 - phi;
58          if (dphi > Math.PI)
59              dphi -= 2. * Math.PI;
60          if (dphi < -Math.PI)
61              dphi += 2. * Math.PI;
62          double s = RC * dphi;
63          double cphi = Math.cos(phi);
64          double sphi = Math.sin(phi);
65          double cphi0 = Math.cos(phi0);
66          double sphi0 = Math.sin(phi0);
67  
68          // Get the track position at the reference point
69          double xref = xc - RC * sphi;
70          if (Math.abs(xref - _xref) > 1.e-10)
71              System.out.println("Bad path length - x0: " + xref + " xref: " + _xref);
72          double yref = yc + RC * cphi;
73          double zref = z0 + s * slope;
74  
75          // Get dy/dx and dz/dx for the straight-line track
76          double dydx = sphi / cphi;
77          double dzdx = slope / cphi;
78  
79          // Calculate the Jacobian between the straight line track parameters and the helix
80          // parameters
81          MutableMatrix deriv = new BasicMatrix(4, 5);
82          double dydcurv = (cphi0 + s / RC * sphi - cphi) * RC * RC;
83          deriv.setElement(StraightLineTrack.y0Index, HelicalTrackFit.curvatureIndex, dydcurv);
84          deriv.setElement(StraightLineTrack.y0Index, HelicalTrackFit.dcaIndex, cphi0);
85          double dydphi0 = (RC - d0) * sphi0 - RC * sphi;
86          deriv.setElement(StraightLineTrack.y0Index, HelicalTrackFit.phi0Index, dydphi0);
87          deriv.setElement(StraightLineTrack.z0Index, HelicalTrackFit.z0Index, 1.);
88          deriv.setElement(StraightLineTrack.z0Index, HelicalTrackFit.slopeIndex, s);
89          deriv.setElement(StraightLineTrack.dydxIndex, HelicalTrackFit.curvatureIndex, -s / (cphi * cphi));
90          deriv.setElement(StraightLineTrack.dydxIndex, HelicalTrackFit.phi0Index, 1. / (cphi * cphi));
91          double dzslopedphi0 = sphi * slope / (cphi * cphi);
92          deriv.setElement(StraightLineTrack.dzdxIndex, HelicalTrackFit.curvatureIndex, -s * dzslopedphi0);
93          deriv.setElement(StraightLineTrack.dzdxIndex, HelicalTrackFit.phi0Index, dzslopedphi0);
94          deriv.setElement(StraightLineTrack.dzdxIndex, HelicalTrackFit.slopeIndex, 1. / cphi);
95  
96          // Calculate the covariance matrix
97          Matrix derivT = MatrixTranspose(deriv);
98          SymmetricMatrix hcov = helix.covariance();
99          Matrix cov = MatrixOp.mult(deriv, MatrixOp.mult(hcov, derivT));
100         SymmetricMatrix scov = new SymmetricMatrix(cov);
101 
102         return new StraightLineTrack(xref, yref, zref, dydx, dzdx, scov);
103     }
104 
105     /**
106      * Convert a helix to a StraightLineTrack.
107      * 
108      * @param helix helix to be converted
109      * @return resulting StraightLineTrack
110      */
111     public StraightLineTrack Convert(Helix helix) {
112 
113         Hep3Vector unitVec = helix.getUnitTangentAtLength(0);
114         Hep3Vector posVec = helix.getPointAtDistance(0);
115         double dzdx = unitVec.z() / unitVec.x();
116         double dydx = unitVec.y() / unitVec.x();
117         double zref = posVec.z() - dzdx * (posVec.x() - _xref);
118         double yref = posVec.y() - dydx * (posVec.x() - _xref);
119         SymmetricMatrix scov = null;
120         StraightLineTrack slt = new StraightLineTrack(_xref, yref, zref, dydx, dzdx, scov);
121         // System.out.printf("%s: unitVec %s posVec %s\n",this.getClass().getSimpleName(),unitVec.toString(),posVec.toString());
122         // System.out.printf("%s: dzdx=%f dydx=%s\n",this.getClass().getSimpleName(),dzdx,dydx);
123         // System.out.printf("%s: ref = %f,%f,%f\n",this.getClass().getSimpleName(),_xref,yref,zref);
124         return slt;
125     }
126 
127     /**
128      * Returns the transpose of the matrix (inexplicably not handled by the matrix package for
129      * non-square matrices).
130      * 
131      * @param m matrix to be transposed
132      * @return transposed matrix
133      */
134     private Matrix MatrixTranspose(Matrix m) {
135         MutableMatrix mt = new BasicMatrix(m.getNColumns(), m.getNRows());
136         for (int i = 0; i < m.getNRows(); i++) {
137             for (int j = 0; j < m.getNColumns(); j++) {
138                 mt.setElement(j, i, m.e(i, j));
139             }
140         }
141         return mt;
142     }
