3   import com.sun.jna.Pointer;
4   
5   /**
6    * wrapper class for MilleBinary JNA functions
7    * <p>
8    * This class re-promotes those JNA functions into
9    * class members and implements an extra constructor
10   * with more sensible HPS defaults
11   */
12  public class MilleBinaryJna {
13      
14      /**
15       * internal handle to memory created and operated on by the GBL library
16       */
17      private Pointer self;
18  
19      /**
20       * full one-to-one constructor
21       */
22      public MilleBinaryJna(String fileName, boolean doublePrec, boolean keepZeros, int aSize) {
23          self = GblInterface.INSTANCE.MilleBinaryCtor(fileName, fileName.length(), doublePrec?1:0, keepZeros?1:0, aSize);
24      }
25      
26      /**
27       * more useful constructor with HPS-sensible defaults
28       */
29      public MilleBinaryJna(String fileName) {
30          self = GblInterface.INSTANCE.MilleBinaryCtor(fileName, fileName.length(), 0, 0, 200000);
31      }
32  
33      public Pointer getPtr() {
34          return self;
35      }
36  
37      public void close() {
38          GblInterface.INSTANCE.MilleBinary_close(self);
39      }
40  
41  }
