153     @Override
154     protected void detectorChanged(Detector detector) {
155         if (aidaGBL == null)
156             aidaGBL = AIDA.defaultInstance();
157 
158         aidaGBL.tree().cd("/");
159 
160         for (HpsSiSensor s : detector.getDetectorElement().findDescendants(HpsSiSensor.class)) {
161             if (s.getName().startsWith("module_") && s.getName().endsWith("sensor0")) {
162                 sensors.add(s);
163             }
164         }
165                 
166         
167         Hep3Vector fieldInTracker = TrackUtils.getBField(detector);
168         this.bfield = Math.abs(fieldInTracker.y());
169 
170         bFieldMap = detector.getFieldMap();
171 
172         setupPlots();
173     }

175     @Override
176     public void process(EventHeader event) {
177         List<Track> tracks = event.get(Track.class, trackCollectionName);
178 
179         /*
180         System.out.print("GBLOutputDriver tracks (" + trackCollectionName + ") N hits: ");
181         for (Track trk : tracks) {
182           System.out.print(trk.getTrackerHits().size()+" ");
183         }
184         System.out.println();
185          */
186 
187         int TrackType = 0;
188         if (trackCollectionName.contains("Kalman") || trackCollectionName.contains("KF")) {
189             TrackType = 1;
190             //System.out.println("PF:: DEBUG :: Found Kalman Tracks in the event");
191         }
192 
193         //System.out.println("Running on "+trackCollectionName);
194 
195         //RelationalTable trackMatchTable = null;
196         //trackMatchTable = new BaseRelationalTable(RelationalTable.Mode.ONE_TO_ONE, RelationalTable.Weighting.UNWEIGHTED);
197         //List<LCRelation> trackMatchRelation = event.get(LCRelation.class, "MatchedToGBLTrackRelations");
198         //for (LCRelation relation : trackMatchRelation) {
199         //    if (relation != null && relation.getFrom() != null && relation.getTo() != null) {
200         //        trackMatchTable.add(relation.getFrom(), relation.getTo());
201         //    }
202         //}
203 
204         RelationalTable hitToStrips = TrackUtils.getHitToStripsTable(event);
205         RelationalTable hitToRotated = TrackUtils.getHitToRotatedTable(event);
206         
207         for (Track trk : tracks) {
208             
209             //Some Track selection
210             
211             //System.out.println("Track loop");
212             
213             if (trk.getChi2() > chi2Cut)
214                 continue;
215 
216 
217             //System.out.println("Track passed chi2");
218             
219             //Remove tracks with less than 10 hits
220             if ((TrackType == 0 && trk.getTrackerHits().size() < nHits) 
221                     || (TrackType == 1 && trk.getTrackerHits().size() < 2*nHits)) {
222                 System.out.println("WARNING:: "+trk.getClass().getSimpleName()
223                         +" got to GBLOutputDriver with "+trk.getTrackerHits().size()+" hits"
224                         +" which is below the cut that should have been already applied.");
225                 continue;
226             }
227 
228 
229             //System.out.println("Track passed hits");
230             
231             Hep3Vector momentum = new BasicHep3Vector(trk.getTrackStates().get(0).getMomentum());
232             
233             if (momentum.magnitude() < minMom)
234                 continue;
235             
236             if (momentum.magnitude() > maxMom)
237                 continue;
238             
239             //System.out.println("Track passed momentum");
240             
241             TrackState trackState = trk.getTrackStates().get(0);
242             if (Math.abs(trackState.getTanLambda()) < minTanL)
243                 continue;
244 
245             if (Math.abs(trackState.getTanLambda()) > maxTanL)
246                 continue;
247             
248             if (Math.abs(trackState.getPhi()) < minPhi)
249                 continue;
250 
251             if (Math.abs(trackState.getPhi()) > maxPhi)
252                 continue;
253             
254             //System.out.println("Track passed tanLambda");
255             
256             GenericObject gblKink = GBLKinkData.getKinkData(event, trk);
257             
258             //if (gblKink == null) {
259                 //System.out.println("Failed finding gblKink object");
260                 //System.out.println("Looked for: "+GBLKinkData.DATA_RELATION_COLLECTION);
261                 //System.out.println("Event has "+GBLKinkData.DATA_RELATION_COLLECTION+" "+event.hasCollection(LCRelation.class, GBLKinkData.DATA_RELATION_COLLECTION));
262             //}
263             
264             
265             
266 
267             //Track matchedTrack = (Track) trackMatchTable.from(trk);
268             Map<HpsSiSensor, TrackerHit> sensorHits = new HashMap<HpsSiSensor, TrackerHit>();
269             Map<HpsSiSensor, Integer> sensorNums    = new HashMap<HpsSiSensor, Integer>();
270             List<TrackerHit> hitsOnTrack = new ArrayList<TrackerHit>();
271 
272             if (hitToStrips != null && hitToRotated != null)
273                 hitsOnTrack = TrackUtils.getStripHits(trk, hitToStrips, hitToRotated);
274             
275             int i = 0;
276             for (TrackerHit hit : hitsOnTrack) {
277                 HpsSiSensor sensor = ((HpsSiSensor) ((RawTrackerHit) hit.getRawHits().get(0)).getDetectorElement());
278                 if (sensor != null) {
279                     sensorHits.put(sensor, hit);
280                     sensorNums.put(sensor, i);
281                     if (debug)
282                         System.out.printf("adding sensor %d \n", i);
283                 }
284 
285                 if (debug && sensor == null)
286                     System.out.printf("TrackerHit null sensor %s \n", hit.toString());
287                 i++;
288 
289             }
290             
291 
292             //THIS IS TEMPORARY AND NEEDED FOR FIXING THE LOOP ON THE SENSORS ON TRACK FOR KALMAN TRACKS
293             
294             if (hitsOnTrack.size() == 0) {
295                 for (TrackerHit hit : trk.getTrackerHits()) {
296                     HpsSiSensor sensor = ((HpsSiSensor) ((RawTrackerHit) hit.getRawHits().get(0)).getDetectorElement());
297                     if (sensor != null) {
298                         sensorHits.put(sensor, hit);
299                         sensorNums.put(sensor, i);
300                         if (debug)
301                             System.out.printf("adding sensor %d \n", i);
302                     }
303                     
304                     if (debug && sensor == null)
305                         System.out.printf("TrackerHit null sensor %s \n", hit.toString());
306                     i++;
307                     
308                 }
309                 
310             }
311             
312             doBasicGBLtrack(trk,sensorHits);
313             if (b_doGBLresiduals) 
314                 doGBLresiduals(trk, sensorHits,event);
315             
316             //doMTresiduals(matchedTrack, sensorHits);
317             if (b_doGBLkinks)
318                 doGBLkinks(trk,gblKink, sensorNums);
319         }
320     }

322     private void doGBLkinks(Track trk, GenericObject kink, Map<HpsSiSensor, Integer> sensorNums) {
323         
324         if (kink == null) {
325             System.out.println("WARNING::Kink object is null");
326             return;
327         }
328             
329 
330         String vol  = "_top";
331         int spacing = 0;
332         if (trk.getTrackStates().get(0).getTanLambda() < 0) {
333             vol = "_bottom";
334             spacing  = sensors.size() / 2 + mod;
335         }
336         
337         for (HpsSiSensor sensor : sensorNums.keySet()) {
338             int index = sensorNums.get(sensor);
339             double phi = kink.getDoubleVal(index);
340             float lambda = kink.getFloatVal(index);
341             
342             //(2019) For top 0-20, for bottom 25-45
343             aidaGBL.histogram2D(kinkFolder+"lambda_kink_mod").fill(sensor.getMillepedeId()+spacing,lambda);
344             aidaGBL.profile1D(kinkFolder+"lambda_kink_mod_p").fill(sensor.getMillepedeId()+spacing,lambda);
345             aidaGBL.histogram2D(kinkFolder+"phi_kink_mod").fill(sensor.getMillepedeId()+spacing,phi);
346             aidaGBL.profile1D(kinkFolder+"phi_kink_mod_p").fill(sensor.getMillepedeId()+spacing,phi);
347             aidaGBL.histogram1D(kinkFolder+"lambda_kink_" + sensor.getName()).fill(lambda);
348             aidaGBL.histogram1D(kinkFolder+"phi_kink_" + sensor.getName()).fill(phi);
349         }
350

353     private void doMTresiduals(Track trk, Map<HpsSiSensor, TrackerHit> sensorHits) {
354         TrackState trackState = trk.getTrackStates().get(0);
355         for (HpsSiSensor sensor : sensorHits.keySet()) {
356             Hep3Vector extrapPos = TrackStateUtils.getLocationAtSensor(trackState, sensor, bfield);
357             Hep3Vector hitPos = new BasicHep3Vector(sensorHits.get(sensor).getPosition());
358             if (hitPos == null || extrapPos == null)
359                 return;
360             Hep3Vector diff = VecOp.sub(extrapPos, hitPos);
361             if (debug)
362                 System.out.printf("MextrapPos %s MhitPos %s \n Mdiff %s ", extrapPos.toString(), hitPos.toString(), diff.toString());
363 
364             ITransform3D trans = sensor.getGeometry().getGlobalToLocal();
365             trans.rotate(diff);
366 
367             aidaGBL.histogram1D(resFolder+"residual_before_GBL_" + sensor.getName()).fill(diff.x());
368             if (debug)
369                 System.out.printf("MdiffSensor %s \n", diff.toString());
370 
371         }
372     }

391     private void doBasicGBLtrack(Track trk, Map<HpsSiSensor, TrackerHit> sensorHits) {
392         
393         TrackState trackState = trk.getTrackStates().get(0);
394 
395         String isTop = "_bottom";
396         //if (trk.getTrackerHits().get(0).getPosition()[2] > 0) {
397         //  isTop = "_top";
398         //}
399 
400         //if (trk.getType()==1 && trk.getTrackerHits().size() < 10) {
401         //    return;
402         //}
403 
404         //List<Integer> missingHits; 
405         //missingHits =  findMissingLayer(trk);
406         
407         if (trackState.getTanLambda() > 0) {
408             isTop = "_top";
409         }
410                 
411         //There is a sign flip in the charge
412         String charge = "_pos";
413         if (trk.getCharge()>0)
414             charge = "_neg";
415         
416         
417         //Hep3Vector mom = new BasicHep3Vector(trackState.getMomentum());
418         //System.out.println("Track momentum " + mom.toString());
419         double trackp = new BasicHep3Vector(trackState.getMomentum()).magnitude();
420         
421         
422         FillGBLTrackPlot(trkpFolder+"d0",isTop,charge,trackState.getD0());
423         FillGBLTrackPlot(trkpFolder+"z0",isTop,charge,trackState.getZ0());
424         FillGBLTrackPlot(trkpFolder+"phi",isTop,charge,trackState.getPhi());
425         FillGBLTrackPlot(trkpFolder+"tanLambda",isTop,charge,trackState.getTanLambda());
426         FillGBLTrackPlot(trkpFolder+"p",isTop,charge,trackp);
427         if (trk.getTrackerHits().size()==7)
428             FillGBLTrackPlot(trkpFolder+"p7h",isTop,charge,trackp);
429         if (trk.getTrackerHits().size()==6)
430             FillGBLTrackPlot(trkpFolder+"p6h",isTop,charge,trackp);
431         if (trk.getTrackerHits().size()==5)
432             FillGBLTrackPlot(trkpFolder+"p5h",isTop,charge,trackp);
433         
434         if (TrackUtils.isHoleTrack(trk)) 
435             FillGBLTrackPlot(trkpFolder+"p_hole",isTop,charge,trackp);
436         else 
437             FillGBLTrackPlot(trkpFolder+"p_slot",isTop,charge,trackp);
438         
439         
440         //Momentum maps
441         FillGBLTrackPlot(trkpFolder+"p_vs_phi",isTop,charge,trackState.getPhi(),trackp);
442         FillGBLTrackPlot(trkpFolder+"p_vs_tanLambda",isTop,charge,trackState.getTanLambda(),trackp);
443         FillGBLTrackPlot(trkpFolder+"p_vs_phi_tanLambda",isTop,charge,trackState.getPhi(),trackState.getTanLambda(),trackp);
444         
445         double tanLambda = trackState.getTanLambda();
446         double cosLambda = 1. / (Math.sqrt(1+tanLambda*tanLambda));
447         
448         FillGBLTrackPlot(trkpFolder+"pT_vs_phi",isTop,charge,trackState.getPhi(),trackp*cosLambda);
449         FillGBLTrackPlot(trkpFolder+"pT_vs_tanLambda",isTop,charge,trackState.getTanLambda(),trackp*cosLambda);
450         
451         
452         //if (trk.getTrackerHits().size()==6)
453         //    FillGBLTrackPlot(trkpFolder+"p_Missing1Hit",isTop,charge,missingHits.get(0),trackp);
454         
455         //if (missingHits.size()==1 && missingHits.get(0)==7) 
456         //    FillGBLTrackPlot(trkpFolder+"p_MissingLastLayer",isTop,charge,trackp);
457         
458         
459         FillGBLTrackPlot(trkpFolder+"Chi2",isTop,charge,trk.getChi2());
460         FillGBLTrackPlot(trkpFolder+"Chi2_vs_p",isTop,charge,trackp,trk.getChi2());
461 
462         // deduce multiplication factor for ST-started GBL tracks
463         int nhits = trk.getTrackerHits().size();
464         if (nhits > 0 && trk.getTrackerHits().get(0) instanceof HelicalTrackCross) {
465             // tracks created from cross hits have 2 measurments per hit instead of only
466             // one so we have to double that count for an equal comparison
467             nhits *= 2;
468         }
469         aidaGBL.histogram1D(trkpFolder+"nHits" + isTop).fill(nhits);
470         aidaGBL.histogram1D(trkpFolder+"nHits" + isTop+charge).fill(nhits);
471 
472         Hep3Vector beamspot = CoordinateTransformations.transformVectorToDetector(TrackUtils.extrapolateHelixToXPlane(trackState, 0));
473         if (debug)
474             System.out.printf("beamspot %s transformed %s \n", beamspot.toString());
475         FillGBLTrackPlot(trkpFolder+"trk_extr_or_x",isTop,charge,beamspot.x());
476         FillGBLTrackPlot(trkpFolder+"trk_extr_or_y",isTop,charge,beamspot.y());
477         
478         //Extrapolation to assumed tgt pos - helix
479         Hep3Vector trkTgt = CoordinateTransformations.transformVectorToDetector(TrackUtils.extrapolateHelixToXPlane(trackState,bsZ));
480         FillGBLTrackPlot(trkpFolder+"trk_extr_bs_x",isTop,charge,trkTgt.x());
481         FillGBLTrackPlot(trkpFolder+"trk_extr_bs_y",isTop,charge,trkTgt.y());
482         
483         //Transform z to the beamspot plane
484         //Get the PathToPlane
485         
486         BaseTrackState ts_bs = TrackUtils.getTrackExtrapAtVtxSurfRK(trackState,bFieldMap,0.,bsZ);
487         
488         
489         //Get the track parameters wrt the beamline using helix
490         double [] beamLine = new double [] {bsZ,0};
491         double [] helixParametersAtBS = TrackUtils.getParametersAtNewRefPoint(beamLine, trackState);
492 
493                   
494         FillGBLTrackPlot(trkpFolder+"trk_extr_bs_x_rk",isTop,charge,ts_bs.getReferencePoint()[1]);
495         FillGBLTrackPlot(trkpFolder+"trk_extr_bs_y_rk",isTop,charge,ts_bs.getReferencePoint()[2]);
496 
497         //Ill defined - should be defined wrt bsX and bsY
498         FillGBLTrackPlot(trkpFolder+"d0_vs_bs_rk",isTop,charge,ts_bs.getD0());
499         FillGBLTrackPlot(trkpFolder+"d0_vs_bs_extrap",isTop,charge,helixParametersAtBS[BaseTrack.D0]);
500         
501         double s = HelixUtils.PathToXPlane(TrackUtils.getHTF(trackState),bsZ,0.,0).get(0);
502         FillGBLTrackPlot(trkpFolder+"z0_vs_bs",isTop,charge,trackState.getZ0() + s*trackState.getTanLambda());
503         FillGBLTrackPlot(trkpFolder+"z0_vs_bs_rk",isTop,charge,ts_bs.getZ0());
504         FillGBLTrackPlot(trkpFolder+"z0_vs_bs_extrap",isTop,charge,helixParametersAtBS[BaseTrack.Z0]);
505         
506 
507         FillGBLTrackPlot(trkpFolder+"phi_vs_bs_extrap",isTop,charge,helixParametersAtBS[BaseTrack.PHI]);
508         
509         //TH2D - Filling
510         FillGBLTrackPlot(trkpFolder+"d0_vs_phi",isTop,charge,trackState.getPhi(),trackState.getD0());
511         FillGBLTrackPlot(trkpFolder+"d0_vs_tanLambda",isTop,charge,trackState.getTanLambda(),trackState.getD0());
512         FillGBLTrackPlot(trkpFolder+"d0_vs_p",isTop,charge,trackp,trackState.getD0());
513         
514         //Ill defined - should be defined wrt bsX and bsY
515         FillGBLTrackPlot(trkpFolder+"d0bs_vs_p",isTop,charge,trackp,helixParametersAtBS[BaseTrack.D0]);
516         
517         FillGBLTrackPlot(trkpFolder+"z0_vs_p",isTop,charge,trackp,trackState.getZ0()); 
518         FillGBLTrackPlot(trkpFolder+"z0bs_vs_p",isTop,charge,trackp,ts_bs.getZ0()); 
519         
520         //Interesting plot to get a sense where z-vtx is. 
521         //If z0 is referenced to the right BS z location, the slope of <z0> vs tanLambda is 0
522         FillGBLTrackPlot(trkpFolder+"z0_vs_tanLambda",isTop,charge,trackState.getTanLambda(),trackState.getZ0());
523         FillGBLTrackPlot(trkpFolder+"z0bs_vs_tanLambda",isTop,charge,trackState.getTanLambda(),ts_bs.getZ0());
524         
525         
526         if (b_doDetailPlots) {
527             int ibins = 15;
528             double start= -12;
529             double end = -5;
530             double step = (end-start) / (double)ibins;
531             
532             for (int ibin = 0; ibin<ibins;ibin++) {
533                 double bslocZ = start+step*ibin;
534                 double s_bslocZ = HelixUtils.PathToXPlane(TrackUtils.getHTF(trackState),bslocZ,0.,0).get(0);
535                 double z0Corr = trackState.getZ0() + s_bslocZ*trackState.getTanLambda();
536                 String ibinstr =  String.valueOf(ibin);
537                 aidaGBL.histogram2D(trkpDetailFolder+"z0_vs_tanLambda_bsZ_"+ibinstr+isTop).fill(trackState.getTanLambda(),z0Corr);
538                 aidaGBL.profile1D(trkpDetailFolder+"z0_vs_tanLambda_bsZ_p_"+ibinstr+isTop).fill(trackState.getTanLambda(),z0Corr);
539                 //System.out.printf("bslocZ %.5f s_bslocZ = %.5f z0C0rr %.5f \n", bslocZ, s_bslocZ, z0Corr);
540                 //aidaGBL.histogram3D(trkpDetailFolder+"z0_vs_tanLambda_bsZ"+isTop).fill(bslocZ,trackState.getTanLambda(),z0Corr);
541                 //aidaGBL.profile2D(trkpDetailFolder+"z0_vs_tanLambda_bsZ_p"+isTop).fill(bslocZ,trackState.getTanLambda(),z0Corr);
542             }        
543         }
544     }

546     private void doGBLresiduals(Track trk, Map<HpsSiSensor, TrackerHit> sensorHits, EventHeader event) {
547         
548         Map<Integer,HpsSiSensor> sensorMPIDs   = new HashMap<Integer,HpsSiSensor>();
549         double trackTime = 0.;
550         double trackTimeSD = 0.;
551         
552 
553         for (HpsSiSensor sensor : sensorHits.keySet()) {
554             //Also fill here the sensorMPIDs map
555             sensorMPIDs.put(sensor.getMillepedeId(),sensor);
556             ITransform3D trans = sensor.getGeometry().getGlobalToLocal();
557             
558             // position of hit (track crossing the sensor before GBL extrapolation)
559             // the hit information available on each sensor is meaningful only along the measurement direction,
560             // Hep3Vector hitPos = new BasicHep3Vector(sensorHits.get(sensor).getPosition());
561             // instead: extract the information of the hit of the track at the sensor position before GBL
562             TrackState trackState = trk.getTrackStates().get(0);
563             Hep3Vector hitTrackPos = TrackStateUtils.getLocationAtSensor(trackState, sensor, bfield);
564             
565             if (hitTrackPos == null) {
566                 if (debug) {
567                     System.out.printf("GBLOutputDriver::doGBLresiduals:: hitTrackPos is null to sensor %s\n", sensor.toString());
568                 }
569                 continue;
570             }
571             
572             Hep3Vector hitTrackPosSensor = new BasicHep3Vector(hitTrackPos.v());
573             trans.transform(hitTrackPosSensor);
574             // after the transformation x and y in the sensor frame are reversed
575             // This plot is ill defined.
576             
577             aidaGBL.histogram2D(hitFolder+"hit_u_vs_v_sensor_frame_" + sensor.getName()).fill(hitTrackPosSensor.y(), hitTrackPosSensor.x());
578             //aidaGBL.histogram2D("hit_u_vs_v_sensor_frame_" + sensor.getName()).fill(hitPos.y(), hitPos.x());
579             //aidaGBL.histogram2D("hit y vs x lab-frame " + sensor.getName()).fill(hitPos.y(), hitPos.x());
580             
581             
582             // position predicted on track after GBL
583             Hep3Vector extrapPos = null;
584             Hep3Vector extrapPosSensor = null;
585             extrapPos = TrackUtils.extrapolateTrackPositionToSensor(trk, sensor, sensors, bfield);
586             if (extrapPos == null)
587                 return;
588             extrapPosSensor = new BasicHep3Vector(extrapPos.v());
589             trans.transform(extrapPosSensor);
590             //aidaGBL.histogram2D("residual after GBL vs u predicted " + sensor.getName()).fill(extrapPosSensor.x(), res);
591             aidaGBL.histogram2D(hitFolder+"predicted_u_vs_v_sensor_frame_" + sensor.getName()).fill(extrapPosSensor.y(), extrapPosSensor.x());
592             // select track charge
593             if(trk.getCharge()>0) {
594                 aidaGBL.histogram2D(hitFolder+"predicted_u_vs_v_pos_sensor_frame_" + sensor.getName()).fill(extrapPosSensor.y(), extrapPosSensor.x());
595             }else if(trk.getCharge()<0) {
596                 aidaGBL.histogram2D(hitFolder+"predicted_u_vs_v_neg_sensor_frame_" + sensor.getName()).fill(extrapPosSensor.y(), extrapPosSensor.x());
597             }
598             
599             // post-GBL residual
600             Hep3Vector hitPos = new BasicHep3Vector(sensorHits.get(sensor).getPosition());
601             Hep3Vector hitPosSensor = new BasicHep3Vector(hitPos.v());
602             trans.transform(hitPosSensor);
603             Hep3Vector resSensor = VecOp.sub(hitPosSensor, extrapPosSensor);
604             aidaGBL.histogram2D(resFolder+"residual_after_GBL_vs_v_predicted_" + sensor.getName()).fill(extrapPosSensor.y(), resSensor.x());
605             aidaGBL.histogram2D(resFolder+"residual_after_GBL_vs_u_hit_" + sensor.getName()).fill(hitPosSensor.x(), resSensor.x());
606             aidaGBL.histogram1D(resFolder+"residual_after_GBL_" + sensor.getName()).fill(resSensor.x());
607 
608 
609             trackTime += sensorHits.get(sensor).getTime();
610             
611 
612             if (debug) {
613                 System.out.printf("hitPos %s  hitPosSensor %s \n", hitPos.toString(), hitPosSensor.toString());
614                 System.out.printf("resSensor %s \n", resSensor.toString());
615                 System.out.printf("extrapPos %s  extrapPosSensor %s \n", extrapPos.toString(), extrapPosSensor.toString());
616                 ITransform3D electrodes_to_global = sensor.getReadoutElectrodes(ChargeCarrier.HOLE).getLocalToGlobal();
617                 Hep3Vector measuredCoordinate = sensor.getReadoutElectrodes(ChargeCarrier.HOLE).getMeasuredCoordinate(0);
618                 Hep3Vector unmeasuredCoordinate = sensor.getReadoutElectrodes(ChargeCarrier.HOLE).getUnmeasuredCoordinate(0);
619                 System.out.printf("unMeasCoordOrig %s MeasCoordOrig %s \n", unmeasuredCoordinate.toString(), measuredCoordinate.toString());
620                 measuredCoordinate = VecOp.mult(VecOp.mult(CoordinateTransformations.getMatrix(), electrodes_to_global.getRotation().getRotationMatrix()), measuredCoordinate);
621                 unmeasuredCoordinate = VecOp.mult(VecOp.mult(CoordinateTransformations.getMatrix(), electrodes_to_global.getRotation().getRotationMatrix()), unmeasuredCoordinate);
622                 Hep3Vector testX = trans.inverse().rotated(new BasicHep3Vector(1, 0, 0));
623                 Hep3Vector testY = trans.inverse().rotated(new BasicHep3Vector(0, 1, 0));
624                 Hep3Vector testZ = trans.inverse().rotated(new BasicHep3Vector(0, 0, 1));
625                 System.out.printf("unMeasCoord %s MeasCoord %s \n transX %s transY %s transZ %s \n", unmeasuredCoordinate.toString(), measuredCoordinate.toString(), testX.toString(), testY.toString(), testZ.toString());
626             }
627         }//loop on sensor hits
628 
629         
630         trackTime /= (float)sensorHits.size();
631         
632         for (HpsSiSensor sensor : sensorHits.keySet()) {
633             trackTimeSD += Math.pow(trackTime - sensorHits.get(sensor).getTime(),2);
634         }
635         
636         trackTimeSD = Math.sqrt(trackTimeSD / ((float) sensorHits.size() - 1.));
637         
638 
639         
640         RelationalTable trackResidualsTable = null;
641         if (event.hasCollection(LCRelation.class, trackResidualsRelColName)) {
642             trackResidualsTable = new BaseRelationalTable(RelationalTable.Mode.ONE_TO_ONE, RelationalTable.Weighting.UNWEIGHTED);
643             List<LCRelation> trackresRelation = event.get(LCRelation.class, trackResidualsRelColName);
644             for (LCRelation relation : trackresRelation) {
645                 if (relation != null && relation.getFrom() != null && relation.getTo() != null) {
646                     trackResidualsTable.add(relation.getFrom(), relation.getTo());
647                 }
648             }
649         } else {
650             //System.out.println("null TrackResidualsGBL Data Relations.");
651             //Failed finding TrackResidualsGBL
652             return;
653         }
654         
655         GenericObject trackRes = (GenericObject) trackResidualsTable.from(trk);
656         if (trackRes == null) {
657             //System.out.println("null TrackResidualsGBL Data.");
658             return;
659         }
660         
661         //it's bias-unbias-bias-unbias-bias-unbias....
662         //TODO add in trackRes the number of hits on tracks ?
663         int nres = (trackRes.getNInt()-1);
664         
665         String vol = "_top";
666         if (trk.getTrackStates().get(0).getTanLambda() < 0)
667             vol = "_bottom";
668 
669         //get the bias first 
670         for (int i_hit =0; i_hit <= nres-1 ; i_hit+=2) {
671             if (trackRes.getIntVal(i_hit)!=-999)  {
672                 //System.out.println("PF::DEBUG:: "+ trackCollectionName+ " trackRes.getIntVal(i_hit) " + trackRes.getIntVal(i_hit));
673                 String sensorName = (sensorMPIDs.get(trackRes.getIntVal(i_hit))).getName();
674                 if (debug) {
675                     //System.out.printf("NHits %d MPID sensor:%d %s %d\n", nres,trackRes.getIntVal(i_hit), sensorName,i_hit);
676                     //System.out.printf("Track residuals: %s %.5f %.5f\n",sensorName, trackRes.getDoubleVal(i_hit),trackRes.getFloatVal(i_hit));
677                 }
678                 //General residuals Per volume
679                 aidaGBL.histogram1D(resFolder+"bresidual_GBL"+vol).fill(trackRes.getDoubleVal(i_hit));
680                                 
681                 if (trackRes.getIntVal(i_hit) < 9) 
682                     //L1L4 
683                     aidaGBL.histogram1D(resFolder+"bresidual_GBL"+vol+"_L1L4").fill(trackRes.getDoubleVal(i_hit));
684                 else 
685                     //L5L7
686                     aidaGBL.histogram1D(resFolder+"bresidual_GBL"+vol+"_L5L7").fill(trackRes.getDoubleVal(i_hit));
687                 
688                 //Top go from 0 to 20, bottom go from 25 to 45
689                 int spacing = 0;
690                 if (vol == "_bottom")
691                     spacing = sensors.size() / 2 + mod;
692                 
693                 aidaGBL.histogram2D(resFolder  +"bresidual_GBL_mod").fill(trackRes.getIntVal(i_hit)+spacing,trackRes.getDoubleVal(i_hit));
694                 aidaGBL.profile1D(resFolder  +"bresidual_GBL_mod_p").fill(trackRes.getIntVal(i_hit)+spacing,trackRes.getDoubleVal(i_hit));
695                 
696                 aidaGBL.histogram1D(resFolder  +"bresidual_GBL_" + sensorName).fill(trackRes.getDoubleVal(i_hit));
697                 aidaGBL.histogram1D(epullFolder+"breserror_GBL_" + sensorName).fill(trackRes.getFloatVal(i_hit));
698                 aidaGBL.histogram1D(epullFolder+"bres_pull_GBL_" + sensorName).fill(trackRes.getDoubleVal(i_hit) / trackRes.getFloatVal(i_hit));
699             }
700             else {
701                 System.out.printf("Track refit failed? No biased residual for %d\n", i_hit);
702             }
703         }
704         // get the unbias
705         for (int i_hit =1; i_hit <= nres-1 ; i_hit+=2) {
706             if (trackRes.getIntVal(i_hit)!=-999)  {  
707 
708 
709                 //Measured hit
710                 HpsSiSensor hps_sensor = sensorMPIDs.get(trackRes.getIntVal(i_hit));
711                 Hep3Vector hitPosG = new BasicHep3Vector(sensorHits.get(hps_sensor).getPosition());
712                 Hep3Vector hitPosSensorG = new BasicHep3Vector(hitPosG.v());
713                 ITransform3D g2l = hps_sensor.getGeometry().getGlobalToLocal();
714                 g2l.transform(hitPosSensorG);
715                 String sensorName = (sensorMPIDs.get(trackRes.getIntVal(i_hit))).getName();
716 
717                 //Predicted hit
718                 Hep3Vector extrapPos = null;
719                 Hep3Vector extrapPosSensor = null;
720                 extrapPos = TrackUtils.extrapolateTrackPositionToSensor(trk, hps_sensor, sensors, bfield);
721                 if (extrapPos == null)
722                     continue;
723                 extrapPosSensor = new BasicHep3Vector(extrapPos.v());
724                 g2l.transform(extrapPosSensor);
725                 
726                 if (debug) {
727                     System.out.printf("NHits %d MPID sensor:%d %s %d\n", nres,trackRes.getIntVal(i_hit), sensorName,i_hit);
728                     System.out.printf("Track uresiduals: %s %.5f %.5f\n",sensorName, trackRes.getDoubleVal(i_hit),trackRes.getFloatVal(i_hit));
729                 }
730 
731                 //General residuals Per volume
732                 aidaGBL.histogram1D(resFolder+"uresidual_GBL"+vol).fill(trackRes.getDoubleVal(i_hit));
733                 
734                 if (trackRes.getIntVal(i_hit) < 9) 
735                     //L1L4 
736                     aidaGBL.histogram1D(resFolder+"uresidual_GBL"+vol+"_L1L4").fill(trackRes.getDoubleVal(i_hit));
737                 else 
738                     //L5L7
739                     aidaGBL.histogram1D(resFolder+"uresidual_GBL"+vol+"_L5L7").fill(trackRes.getDoubleVal(i_hit));
740                 
741                 
742                 //Top go from 0 to 20, bottom go from 25 to 45
743                 int spacing = 0;
744                 if (vol == "_bottom")
745                     spacing = sensors.size()/2 + mod;
746                 
747                 aidaGBL.histogram2D(resFolder+"uresidual_GBL_mod").fill(trackRes.getIntVal(i_hit)+spacing,trackRes.getDoubleVal(i_hit));
748                 aidaGBL.profile1D(resFolder+"uresidual_GBL_mod_p").fill(trackRes.getIntVal(i_hit)+spacing,trackRes.getDoubleVal(i_hit));
749                 aidaGBL.histogram1D(resFolder+"uresidual_GBL_" + sensorName).fill(trackRes.getDoubleVal(i_hit));
750                 aidaGBL.histogram2D(resFolder+"uresidual_GBL_vs_u_hit_" + sensorName).fill(hitPosSensorG.x(),trackRes.getDoubleVal(i_hit));
751                 aidaGBL.histogram2D(resFolder+"uresidual_GBL_vs_v_pred_" + sensorName).fill(extrapPosSensor.y(),trackRes.getDoubleVal(i_hit));
752                 aidaGBL.histogram1D(epullFolder+"ureserror_GBL_" + sensorName).fill(trackRes.getFloatVal(i_hit));
753                 aidaGBL.histogram1D(epullFolder+"ures_pull_GBL_" + sensorName).fill(trackRes.getDoubleVal(i_hit) / trackRes.getFloatVal(i_hit));
754                 
755                 //Get the hit time
756                 double hitTime = sensorHits.get(hps_sensor).getTime();
757                 
758                 //Get the track time (it's the average of hits-on-track time)
759                 
760                 double dT_hit_track  =  hitTime - trackTime;
761                 double dT_hit_sigma  = (hitTime - trackTime) / trackTimeSD;
762                 
763                 aidaGBL.histogram2D(resFolder+"uresidual_GBL_vs_dT_hit_"+sensorName).fill(dT_hit_track,trackRes.getDoubleVal(i_hit));
764                 aidaGBL.histogram2D(resFolder+"uresidual_GBL_vs_dTs_hit_"+sensorName).fill(dT_hit_sigma,trackRes.getDoubleVal(i_hit));
765                 
766                 
767 
768 
769             }
770             else {
771                 if (debug){
772                     System.out.printf("Track refit failed? No biased residual for %d\n", i_hit);
773                 }
774             }
775         }
776     }//doGBLresiduals

778     private List<Integer> findMissingLayer(Track trk) {
779         
780         List<Integer> layers = new ArrayList<Integer>();
781         layers.add(1);
782         layers.add(2);
783         layers.add(3);
784         layers.add(4);
785         layers.add(5);
786         layers.add(6);
787         layers.add(7);
788         
789         List<Integer> LayersOnTrack = new ArrayList<Integer>();
790         List<Integer> missingHits   = new ArrayList<Integer>();
791         
792         for (TrackerHit hit : trk.getTrackerHits()) {
793             
794             int stripLayer = ((HpsSiSensor) ((RawTrackerHit) hit.getRawHits().get(0)).getDetectorElement()).getLayerNumber();
795             
796             int hpslayer = (stripLayer + 1 ) / 2;
797             LayersOnTrack.add(hpslayer);
798         }
799 
800         for (Integer layer : layers) {
801             if (!LayersOnTrack.contains(layer))
802                 missingHits.add(layer);
803         }
804         
805         return missingHits;
806     }

989                 for (int i=0; i<type.length; i++){
990                     // strip the trailing / from the object name and check if any else
991                     String namtyp = type[i].substring(1);
992                     if(namtyp.contains("/")) {
993                         continue;
994                     }else{
995                         IManagedObject obj = aidaGBL.tree().find(namtyp);
996                         if (obj instanceof IBaseHistogram) aidaGBL.tree().rm(obj.name()) ;
997                     }
998                 }
999
