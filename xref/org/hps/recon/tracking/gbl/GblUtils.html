48          private BasicMatrix _params = new BasicMatrix(1, 5);
49  
50          public ClParams(HelicalTrackFit htf, double B) {
51  
52              if (htf == null) {
53                  return;
54              }
55  
56              Hep3Matrix perToClPrj = getPerToClPrj(htf);
57  
58              double d0 = -1 * htf.dca(); // sign convention for curvilinear frame
59              double z0 = htf.z0();
60              Hep3Vector vecPer = new BasicHep3Vector(0., d0, z0);
61              // System.out.printf("%s: vecPer=%s\n",this.getClass().getSimpleName(),vecPer.toString());
62  
63              Hep3Vector vecCl = VecOp.mult(perToClPrj, vecPer);
64              // System.out.printf("%s: vecCl=%s\n",this.getClass().getSimpleName(),vecCl.toString());
65              double xT = vecCl.x();
66              double yT = vecCl.y();
67              // double zT = vecCl.z();
68  
69              double lambda = Math.atan(htf.slope());
70              double q = Math.signum(htf.R());
71              double qOverP = q / htf.p(Math.abs(B));
72              double phi = htf.phi0();
73  
74              _params.setElement(0, FittedGblTrajectory.GBLPARIDX.QOVERP.getValue(), qOverP);
75              _params.setElement(0, FittedGblTrajectory.GBLPARIDX.YTPRIME.getValue(), lambda);
76              _params.setElement(0, FittedGblTrajectory.GBLPARIDX.XTPRIME.getValue(), phi);
77              _params.setElement(0, FittedGblTrajectory.GBLPARIDX.XT.getValue(), xT);
78              _params.setElement(0, FittedGblTrajectory.GBLPARIDX.YT.getValue(), yT);
79          }
80  
81          public BasicMatrix getParams() {
82              return _params;
83          }
84  
85          double getQoverP() {
86              return _params.e(0, FittedGblTrajectory.GBLPARIDX.QOVERP.getValue());
87          }
88  
89          double getLambda() {
90              return _params.e(0, FittedGblTrajectory.GBLPARIDX.YTPRIME.getValue());
91          }
92  
93          double getPhi() {
94              return _params.e(0, FittedGblTrajectory.GBLPARIDX.XTPRIME.getValue());
95          }
96  
97          double getXt() {
98              return _params.e(0, FittedGblTrajectory.GBLPARIDX.XT.getValue());
99          }

111     public static TrackResidualsData computeGblResiduals(Track trk, FittedGblTrajectory fitGbl_traj) {
112         
113         GblTrajectory gbl_fit_trajectory = fitGbl_traj.get_traj();
114         
115         List<Double>  b_residuals = new ArrayList<Double>();
116         List<Float>   b_sigmas    = new ArrayList<Float>();
117         List<Integer> r_sensors   = new ArrayList<Integer>();
118         
119         int numData[] = new int[1];
120         //System.out.printf("Getting the residuals. Points  on trajectory: %d \n",gbl_fit_trajectory.getNpointsOnTraj());
121         //The fitted trajectory has a mapping between the MPID and the ilabel. Use that to get the MPID of the residual.
122         Integer[] sensorsFromMapArray = fitGbl_traj.getSensorMap().keySet().toArray(new Integer[0]);
123         //System.out.printf("Getting the residuals. Sensors on trajectory: %d \n",sensorsFromMapArray.length);
124         
125         //System.out.println("Check residuals of the original fit");
126         //Looping on all the sensors on track -  to get the biased residuals.
127         for (int i_s = 0; i_s < sensorsFromMapArray.length; i_s++) {       
128             //Get the point label
129             int ilabel = sensorsFromMapArray[i_s];
130             //Get the millepede ID
131             int mpid = fitGbl_traj.getSensorMap().get(ilabel);
132             List<Double> aResiduals   = new ArrayList<Double>();   
133             List<Double> aMeasErrors  = new ArrayList<Double>();
134             List<Double> aResErrors   = new ArrayList<Double>();  
135             List<Double> aDownWeights = new ArrayList<Double>();
136             gbl_fit_trajectory.getMeasResults(ilabel,numData,aResiduals,aMeasErrors,aResErrors,aDownWeights); 
137             if (numData[0]>1) { 
138                 System.out.printf("GBLRefitterDriver::WARNING::We have SCT sensors. Residuals dimensions should be <=1\n");
139             }
140             for (int i=0; i<numData[0];i++) {
141                 //System.out.printf("KalmanToGBLDriver::ilabel numDataIDX MPID aResidual aMeasError aResError\n");
142                 //System.out.printf("KalmanToGBLDriver::measResults %d %d %d %f %f %f \n",ilabel, i, mpid, aResiduals.get(i),aMeasErrors.get(i),aResErrors.get(i));
143                 
144                 r_sensors.add(mpid);
145                 b_residuals.add(aResiduals.get(i));
146                 b_sigmas.add(aResErrors.get(i).floatValue());
147             }
148             //Perform an unbiasing fit for each traj
149             
150             //System.out.println("Run the unbiased residuals!!!\n");
151             //For each sensor create a trajectory 
152             GblTrajectory gbl_fit_traj_u = new GblTrajectory(gbl_fit_trajectory.getSingleTrajPoints());
153             double[] u_dVals = new double[2];
154             int[] u_iVals    = new int[1];
155             int[] u_numData  = new int[1]; 
156             //Fit it once to have exactly the same starting point of gbl_fit_trajectory.
157             gbl_fit_traj_u.fit(u_dVals,u_iVals,"");
158             List<Double> u_aResiduals   = new ArrayList<Double>();   
159             List<Double> u_aMeasErrors  = new ArrayList<Double>();
160             List<Double> u_aResErrors   = new ArrayList<Double>();  
161             List<Double> u_aDownWeights = new ArrayList<Double>();
162             
163             try {
164                 //Fit removing the measurement
165                 gbl_fit_traj_u.fit(u_dVals,u_iVals,"",ilabel);
166                 gbl_fit_traj_u.getMeasResults(ilabel,numData,u_aResiduals,u_aMeasErrors,u_aResErrors,u_aDownWeights); 
167                 for (int i=0; i<numData[0];i++) {
168                     //System.out.printf("Example1::ilabel numDataIDX MPID aResidual aMeasError aResError\n");
169                     //System.out.printf("Example1::UmeasResults %d %d %d %f %f %f \n",ilabel, i, mpid, u_aResiduals.get(i),u_aMeasErrors.get(i),u_aResErrors.get(i));
170                     
171                     r_sensors.add(mpid);
172                     b_residuals.add(u_aResiduals.get(i));
173                     b_sigmas.add(u_aResErrors.get(i).floatValue());
174                 }
175             }
176             catch (RuntimeException e){
177                 //  e.printStackTrack();
178                 r_sensors.add(-999);
179                 b_residuals.add(-9999.);
180                 b_sigmas.add((float)-9999.);
181                 //System.out.printf("Unbiasing fit fails! For label::%d\n",ilabel);
182             }
183             
184         }//loop on sensors on track
185         
186         //Set top by default
187         int trackerVolume = 0;
188         //if tanLamda<0 set bottom
189         //System.out.printf("Residuals size %d \n", r_sensors.size());
190         
191         if (trk.getTrackStates().get(0).getTanLambda() < 0) trackerVolume = 1;
192         TrackResidualsData resData  = new TrackResidualsData(trackerVolume,r_sensors,b_residuals,b_sigmas);
193         
194         return resData;
195     }

240     /**
241      * Get corrected perigee parameters.
242      * 
243      * @param locPar - GBL local curvilinear corrections
244      * @param helicalTrackFit - helix
245      * @param bfield - B-field strength
246      * @return corrected parameters
247      */
248     public static double[] getCorrectedPerigeeParameters(Vector locPar, HelicalTrackFit helicalTrackFit, double bfield) {
249 
250         // Explicitly assign corrections to local variables
251         double qOverPCorr = locPar.get(FittedGblTrajectory.GBLPARIDX.QOVERP.getValue());
252         double xTPrimeCorr = locPar.get(FittedGblTrajectory.GBLPARIDX.XTPRIME.getValue());
253         double yTPrimeCorr = locPar.get(FittedGblTrajectory.GBLPARIDX.YTPRIME.getValue());
254         double xTCorr = locPar.get(FittedGblTrajectory.GBLPARIDX.XT.getValue());
255         double yTCorr = locPar.get(FittedGblTrajectory.GBLPARIDX.YT.getValue());
256 
257         // Get helix parameters
258         double qOverP = helicalTrackFit.curvature()
259                 / (Constants.fieldConversion * Math.abs(bfield) * Math.sqrt(1 + Math.pow(helicalTrackFit.slope(), 2)));
260         double d0 = -1.0 * helicalTrackFit.dca(); // correct for different sign convention of d0 in perigee frame
261         double z0 = helicalTrackFit.z0();
262         double phi0 = helicalTrackFit.phi0();
263         double lambda = Math.atan(helicalTrackFit.slope());
264 
265         // calculate new d0 and z0
266         Hep3Matrix perToClPrj = GblUtils.getPerToClPrj(helicalTrackFit);
267 
268         Hep3Matrix clToPerPrj = VecOp.inverse(perToClPrj);
269         Hep3Vector corrPer = VecOp.mult(clToPerPrj, new BasicHep3Vector(xTCorr, yTCorr, 0.0));
270 
271         // d0
272         double d0_corr = corrPer.y();
273         double dca_gbl = -1.0 * (d0 + d0_corr);
274 
275         // z0
276         double z0_corr = corrPer.z();
277         double z0_gbl = z0 + z0_corr;
278 
279         // calculate new slope
280         double lambda_gbl = lambda + yTPrimeCorr;
281         double slope_gbl = Math.tan(lambda_gbl);
282 
283         // calculate new curvature
284         double qOverP_gbl = qOverP + qOverPCorr;
285         double C_gbl = Constants.fieldConversion * Math.abs(bfield) * qOverP_gbl / Math.cos(lambda_gbl);
286 
287         // calculate new phi0
288         double phi0_gbl = phi0 + xTPrimeCorr - corrPer.x() * C_gbl;
289 
290         LOGGER.info("qOverP=" + qOverP + " qOverPCorr=" + qOverPCorr + " qOverP_gbl=" + qOverP_gbl + " ==> pGbl=" + 1.0
291                 / qOverP_gbl + " C_gbl=" + C_gbl);
292 
293         LOGGER.info(String.format("corrected helix: d0=%f, z0=%f, omega=%f, tanlambda=%f, phi0=%f, p=%f", dca_gbl,
294                 z0_gbl, C_gbl, slope_gbl, phi0_gbl, Math.abs(1 / qOverP_gbl)));
295 
296         double parameters_gbl[] = new double[5];
297         parameters_gbl[HelicalTrackFit.dcaIndex] = dca_gbl;
298         parameters_gbl[HelicalTrackFit.phi0Index] = phi0_gbl;
299         parameters_gbl[HelicalTrackFit.curvatureIndex] = C_gbl;
300         parameters_gbl[HelicalTrackFit.z0Index] = z0_gbl;
301         parameters_gbl[HelicalTrackFit.slopeIndex] = slope_gbl;
302 
303         return parameters_gbl;
304 
305     }

307     public static BasicMatrix gblSimpleJacobianLambdaPhi(double ds, double cosl, double bfac) {
308         /*
309          * Simple jacobian: quadratic in arc length difference. using lambda phi as directions
310          * @param ds: arc length difference
311          * @type ds: float
312          * @param cosl: cos(lambda)
313          * @type cosl: float
314          * @param bfac: Bz*c
315          * @type bfac: float
316          * @return: jacobian to move by 'ds' on trajectory
317          * @rtype: matrix(float) ajac(1,1)= 1.0D0 ajac(2,2)= 1.0D0 ajac(3,1)=-DBLE(bfac*ds) ajac(3,3)= 1.0D0
318          * ajac(4,1)=-DBLE(0.5*bfac*ds*ds*cosl) ajac(4,3)= DBLE(ds*cosl) ajac(4,4)= 1.0D0 ajac(5,2)= DBLE(ds) ajac(5,5)=
319          * 1.0D0 ''' jac = np.eye(5) jac[2, 0] = -bfac * ds jac[3, 0] = -0.5 * bfac * ds * ds * cosl jac[3, 2] = ds *
320          * cosl jac[4, 1] = ds return jac
321          */
322         BasicMatrix mat = unitMatrix(5, 5);
323         mat.setElement(2, 0, -bfac * ds);
324         mat.setElement(3, 0, -0.5 * bfac * ds * ds * cosl);
325         mat.setElement(3, 2, ds * cosl);
326         mat.setElement(4, 1, ds);
327         return mat;
328     }

330     public static BasicMatrix unitMatrix(int rows, int cols) {
331         BasicMatrix mat = new BasicMatrix(rows, cols);
332         for (int row = 0; row != mat.getNRows(); row++) {
333             for (int col = 0; col != mat.getNColumns(); col++) {
334                 if (row != col) {
335                     mat.setElement(row, col, 0);
336                 } else {
337                     mat.setElement(row, col, 1);
338                 }
339             }
340         }
341         return mat;
342     }

354     public static double estimateScatter(IDetectorElement hitElement, HelicalTrackFit htf,
355             MultipleScattering scattering, double _B) {
356         // can be edge case where helix is outside, but close to sensor, so make a new scatter point assuming the helix
357         // does pass through the sensor
358         MaterialSupervisor.DetectorPlane hitPlane = null;
359         if (MaterialSupervisor.class.isInstance(scattering.getMaterialManager())) {
360             MaterialSupervisor matSup = (MaterialSupervisor) scattering.getMaterialManager();
361             for (MaterialSupervisor.ScatteringDetectorVolume vol : matSup.getMaterialVolumes()) {
362                 if (vol.getDetectorElement() == hitElement) {
363                     hitPlane = (MaterialSupervisor.DetectorPlane) vol;
364                     break;
365                 }
366             }
367             if (hitPlane == null) {
368                 throw new RuntimeException("cannot find plane for hit!");
369             } else {
370                 // find scatterlength
371                 double s_closest = HelixUtils.PathToXPlane(htf, hitPlane.origin().x(), 0., 0).get(0);
372                 double X0 = hitPlane.getMaterialTraversedInRL(HelixUtils.Direction(htf, s_closest));
373                 ScatterAngle scatterAngle = new ScatterAngle(s_closest, scattering.msangle(htf.p(Math.abs(_B)), X0));
374                 return scatterAngle.Angle();
375             }
376         } else {
377             throw new UnsupportedOperationException(
378                 "Should not happen. This problem is only solved with the MaterialSupervisor.");
379         }
380     }

384      * Form a virtual layer at the target location
385      * Use the helix analytical extrapolation to beamspot
386      * These computations are in tracking frame
387      */
388     
389     public static GBLBeamSpotPoint gblMakeBsPoint(HelicalTrackFit htf, double [] bsLocation, double[] udir, double [] vdir, double [] error)  {
390         //Using the GBL package for prediction and change of reference
391         //Need to use -1/R and -d0 as it uses different conventions 
392             
393         GblSimpleHelix gblHelix = new GblSimpleHelix(-1/htf.R(), htf.phi0(), -htf.dca(), htf.slope(), htf.z0());
394         GblHelixPrediction gblHelixPrediction = gblHelix.getPrediction(bsLocation, udir,vdir);
395 
396         double prediction [] = new double[2];
397         gblHelixPrediction.getMeasPred(prediction);
398 
399         gblHelix.delete();
400         gblHelixPrediction.delete();
401             
402         //Curvilinear frame, UV Plane 
403         Matrix uvDir_gbl = gblHelixPrediction.getCurvilinearDirs();
404             
405         //Global to measurement transformation
406             
407         Matrix mDir = new Matrix(2,3);
408             
409         mDir.set(0,2,udir[2]);
410         mDir.set(1,0,vdir[0]);
411         mDir.set(1,1,vdir[1]);
412 
413         //measurement to global 
414         Matrix mDirT = mDir.copy().transpose();
415             
416         //measurement to curvilinear frame.
417         Matrix proM2l = uvDir_gbl.times(mDirT);
418 
419         //Curvilinear to measurement. 
420         Matrix proL2m = proM2l.copy().inverse();
421             
422         //Create the jacobian point to point
423         double sArc2D = gblHelixPrediction.getArcLength();
424             
425         Hep3Vector bsLocation_3v = new BasicHep3Vector(bsLocation[0],bsLocation[1],bsLocation[2]);
426         Vector aResidual = new Vector(2);
427         //Notice the minus sign. The residual is defined as meas - pred
428         aResidual.set(0,-prediction[0]); //Y
429         aResidual.set(1,-prediction[1]); //X
430 
431         Vector aPrecision = new Vector(2);
432         aPrecision.set(0,1./(error[0]*error[0])); 
433         aPrecision.set(1,1./(error[1]*error[1])); 
434            
435         return  new GBLBeamSpotPoint(bsLocation_3v, aResidual,
436                                      aPrecision, proL2m,sArc2D,htf.slope(),htf.phi0() - sArc2D/htf.R());
437             
438     }

459     /**
460      * Calculate the Jacobian from Curvilinear to Perigee frame.
461      * 
462      * @param helicalTrackFitAtIPCorrected - corrected helix at this point
463      * @param bfield - magnitude of B-field
464      * @return the Jacobian matrix from Curvilinear to Perigee frame
465      */
466     public static Matrix getCLToPerigeeJacobian(HelicalTrackFit helicalTrackFit, HpsHelicalTrackFit helicalTrackFitAtIPCorrected, double bfield) {
467         
468         /*
469          * This part is taken from: // Strandlie, Wittek, NIMA 566, 2006 Matrix covariance_gbl = new Matrix(5, 5);
470          * //helpers double Bz = -Constants.fieldConversion * Math.abs(bfield); // TODO sign convention and should it be
471          * it scaled from Telsa? double p = Math.abs(1 / qOverP_gbl); double q = Math.signum(qOverP_gbl); double
472          * tanLambda = Math.tan(lambda_gbl); double cosLambda = Math.cos(lambda_gbl); // Hep3Vector B = new
473          * BasicHep3Vector(0, 0, Bz); // TODO sign convention? Hep3Vector H = new BasicHep3Vector(0, 0, 1); Hep3Vector T
474          * = HelixUtils.Direction(helix, 0.); Hep3Vector HcrossT = VecOp.cross(H, T); double alpha =
475          * HcrossT.magnitude(); // this should be Bvec cross TrackDir/|B| double Q = Bz * q / p; Hep3Vector Z = new
476          * BasicHep3Vector(0, 0, 1); Hep3Vector J = VecOp.mult(1. / VecOp.cross(T, Z).magnitude(), VecOp.cross(T, Z));
477          * Hep3Vector K = Z; Hep3Vector U = VecOp.mult(-1, J); Hep3Vector V = VecOp.cross(T, U); Hep3Vector I =
478          * VecOp.cross(J, K); Hep3Vector N = VecOp.mult(1 / alpha, VecOp.cross(H, T)); double UdotI = VecOp.dot(U, I);
479          * double NdotV = VecOp.dot(N, V); double NdotU = VecOp.dot(N, U); double TdotI = VecOp.dot(T, I); double VdotI
480          * = VecOp.dot(V, I); double VdotK = VecOp.dot(V, K); covariance_gbl.set(HelicalTrackFit.dcaIndex,
481          * FittedGblTrajectory.GBLPARIDX.XT.getValue(), VdotK / TdotI); covariance_gbl.set(HelicalTrackFit.phi0Index,
482          * FittedGblTrajectory.GBLPARIDX.XTPRIME.getValue(), 1); covariance_gbl.set(HelicalTrackFit.phi0Index,
483          * FittedGblTrajectory.GBLPARIDX.XT.getValue(), -alpha * Q * UdotI * NdotU / (cosLambda * TdotI));
484          * covariance_gbl.set(HelicalTrackFit.phi0Index, FittedGblTrajectory.GBLPARIDX.YT.getValue(), -alpha * Q * VdotI
485          * * NdotU / (cosLambda * TdotI)); covariance_gbl.set(HelicalTrackFit.curvatureIndex,
486          * FittedGblTrajectory.GBLPARIDX.QOVERP.getValue(), -1 * Bz / cosLambda); //
487          * covariance_gbl.set(HelicalTrackFit.curvatureIndex, FittedGblTrajectory.GBLPARIDX.XTPRIME.getValue(), 0);
488          * covariance_gbl.set(HelicalTrackFit.curvatureIndex, FittedGblTrajectory.GBLPARIDX.YTPRIME.getValue(), -1 * q *
489          * Bz * tanLambda / (p * cosLambda)); covariance_gbl.set(HelicalTrackFit.curvatureIndex,
490          * FittedGblTrajectory.GBLPARIDX.XT.getValue(), q * Bz * alpha * Q * tanLambda * UdotI * NdotV / (p * cosLambda
491          * * TdotI)); covariance_gbl.set(HelicalTrackFit.curvatureIndex, FittedGblTrajectory.GBLPARIDX.YT.getValue(), q
492          * * Bz * alpha * Q * tanLambda * VdotI * NdotV / (p * cosLambda * TdotI));

564      * Computes the projection matrix from perigee to curvilinear frame, with reference point(0,0,0) for the perigee frame. 
565      * The projection matrix only depends on tanLambda.
566      *
567      * @param tanLambda
568      * @return 3x3 projection matrix
569      */

571     static Hep3Matrix getSimplePerToClPrj(double tanLambda) {
572         BasicHep3Matrix trans = new BasicHep3Matrix();
573         double cosLambda = 1. / Math.sqrt(1 + tanLambda*tanLambda);
574         double sinLambda = Math.sqrt(1. - cosLambda*cosLambda);
575         //Only filling non zero entries
576         
577         trans.setElement(0, 1, -1);
578         trans.setElement(1, 0, sinLambda);
579         trans.setElement(1, 2, cosLambda);
580         trans.setElement(2, 0, -cosLambda);
581         trans.setElement(2, 2, sinLambda);
582         return trans;
583     }

587      * Computes the projection matrix from the perigee XY plane variables dca and z0 into the curvilinear xT,yT,zT frame
588      * (U,V,T) with reference point (0,0,0) for the perigee frame.
589      *
590      * @param htf input helix to find the track direction
591      * @return 3x3 projection matrix
592      */
593     static Hep3Matrix getPerToClPrj(HelicalTrackFit htf) {
594         Hep3Vector Z = new BasicHep3Vector(0, 0, 1);
595         Hep3Vector T = HelixUtils.Direction(htf, 0.);
596         Hep3Vector J = VecOp.mult(1. / VecOp.cross(T, Z).magnitude(), VecOp.cross(T, Z));
597         Hep3Vector K = Z;
598         Hep3Vector U = VecOp.mult(-1, J);
599         Hep3Vector V = VecOp.cross(T, U);
600         Hep3Vector I = VecOp.cross(J, K);
601 
602         BasicHep3Matrix trans = new BasicHep3Matrix();
603         trans.setElement(0, 0, VecOp.dot(I, U));
604         trans.setElement(0, 1, VecOp.dot(J, U));
605         trans.setElement(0, 2, VecOp.dot(K, U));
606         trans.setElement(1, 0, VecOp.dot(I, V));
607         trans.setElement(1, 1, VecOp.dot(J, V));
608         trans.setElement(1, 2, VecOp.dot(K, V));
609         trans.setElement(2, 0, VecOp.dot(I, T));
610         trans.setElement(2, 1, VecOp.dot(J, T));
611         trans.setElement(2, 2, VecOp.dot(K, T));
612         return trans;

627     private static BasicMatrix getPerParVector(double kappa, double theta, double phi, double d0, double z0) {
628         BasicMatrix perPar = new BasicMatrix(1, 5);
629         perPar.setElement(0, 0, kappa);
630         perPar.setElement(0, 1, theta);
631         perPar.setElement(0, 2, phi);
632         perPar.setElement(0, 3, d0);
633         perPar.setElement(0, 4, z0);
634         return perPar;
635     }

637     private static BasicMatrix getPerParVector(HelicalTrackFit htf, double B) {
638         if (htf != null) {
639             double kappa = -1.0 * Math.signum(B) / htf.R();
640             double theta = Math.PI / 2.0 - Math.atan(htf.slope());
641             return getPerParVector(kappa, theta, htf.phi0(), htf.dca(), htf.z0());
642         }
643         return new BasicMatrix(1, 5);
644     }
