37      public void closeFile() {
38          try {
39              _pWriter.close();
40              _fWriter.close();
41          } catch (IOException ex) {
42              Logger.getLogger(RunAlignment.class.getName()).log(Level.SEVERE, null, ex);
43          }
44      }
45  
46      private void openFile(String fileName) {
47          if (fileName.equalsIgnoreCase("")) {
48              System.out.printf("%s: no file name specified \n", this.getClass().getSimpleName());
49              System.exit(1);
50          }
51          try {
52              _fWriter = new FileWriter(fileName);
53              _pWriter = new PrintWriter(_fWriter);
54          } catch (IOException ex) {
55              Logger.getLogger(RunAlignment.class.getName()).log(Level.SEVERE, null, ex);
56          }
57      }

107     void printStripJacPointToPoint(int id, int layer, double s, BasicMatrix jac) {
108         String str = String.format("Strip id <%d> layer <%d> s <%.10f> jac <", id, layer, s);
109         for (int r = 0; r < jac.getNRows(); ++r) {
110             for (int c = 0; c < jac.getNColumns(); ++c) {
111                 str += String.format("%.10f ", jac.e(r, c));
112             }
113         }
114         str += ">";
115         addLine(str);
116     }

140     void printPerTrackCov(HelicalTrackFit htf) {
141         String str = "Track perCov (idx: dca,phi0,curv,z0,slope) ";
142         SymmetricMatrix cov = htf.covariance();
143         for (int irow = 0; irow < cov.getNRows(); ++irow) {
144             for (int icol = 0; icol < cov.getNColumns(); ++icol) {
145                 str += String.format("%e ", cov.e(irow, icol));
146             }
147         }
148         addLine(str);
149     }
