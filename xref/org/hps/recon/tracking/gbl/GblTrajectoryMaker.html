30  /**
31   * This class is a converter from track to GBLTrajectory for alignment purposes.
32   * It is largely a copy of the methods used in MakeGblTracks but with the difference that then invokes:
33   * 1) ejml for matrix algebra
34   * 2) jna for full support of GBL C++ external library
35   *
36   * No Fit is performed over GBLTrajectories
37   * Constraints supported:
38   * - impact parameters
39   * - vertex constraint
40   */
41  public class GblTrajectoryMaker {

67      /**
68       * Create a list of {@link GBLStripClusterData} objects that can be used as
69       * input to the GBL fitter.
70       *
71       * @param htf
72       * @param stripHits
73       * @param _scattering
74       * @param _B
75       * @param _debug
76       * @return the list of GBL strip cluster data
77       */
78      public List<GBLStripClusterData> makeStripData(HelicalTrackFit htf, List<TrackerHit> stripHits) {
79          List<GBLStripClusterData> stripClusterDataList = new ArrayList<GBLStripClusterData>();
80  
81          // Find scatter points along the path
82          //In principle I could use this to add the hits - TODO ?
83          MultipleScattering.ScatterPoints scatters = _scattering.FindHPSScatterPoints(htf);
84          
85          //Loop over the Scatters
86          
87          //Two things can happen here at each iteration:
88          //1) Nscatters >= Nhits on tracks =>
89          //   Build GBLStripClusterData for both measurements and scatters
90          //2) Nscatters < Nhits on track =>
91          //   The hit has been associated to the track but couldn't find the scatter on the volume => create a scatter at that sensor
92          //   Does this makes sense?
93          
94          //Case (1)
95          //Check if the scatter has a measurement => then use the usual way to build the HelicalTrackStripGbl
96          
97          int nhits = 0;
98          int nscatters = 0;
99          boolean addScatters = true;
100             
101         if (scatters.getPoints().size() >= stripHits.size() && _includeMS) {
102             
103             for (MultipleScattering.ScatterPoint scatter : scatters.getPoints()) {
104             
105                 boolean MeasOnScatter = false;
106                 
107                 //This is bit inefficient as it always loop on all the stripHits
108                 //TODO: optimize and only check hit once if is in the scatters.
109                 
110                 HpsSiSensor scatter_sensor = (HpsSiSensor) scatter.getDet();
111             
112                 for (TrackerHit stripHit : stripHits) {
113                 
114                     if (MeasOnScatter)
115                         continue;
116                 
117                     IDetectorElement det_element = ((RawTrackerHit) stripHit.getRawHits().get(0)).getDetectorElement();
118                     
119                     if (det_element.equals(scatter.getDet())) {
120                         MeasOnScatter = true;
121                         HpsSiSensor hit_sensor = (HpsSiSensor) det_element;
122                         nhits+=1;
123                         HelicalTrackStripGbl gbl_strip;
124                         if (stripHit instanceof SiTrackerHitStrip1D) {
125                             gbl_strip = new HelicalTrackStripGbl(makeDigiStrip((SiTrackerHitStrip1D) stripHit),true);
126                         } else {
127                             SiTrackerHitStrip1D newHit = new SiTrackerHitStrip1D(stripHit);
128                             gbl_strip = new HelicalTrackStripGbl(makeDigiStrip(newHit), true);
129                         }
130                         //hit_sensor or scatter_sensor, they are the same here
131                         GBLStripClusterData stripData = makeStripData(hit_sensor, gbl_strip, htf, scatter);
132                         if (stripData != null) {
133                             stripClusterDataList.add(stripData); 
134                         }
135                         else {
136                             System.out.printf("WARNING::MakeGblTracks::Couldn't make stripClusterData for hps sensor. Skipping scatter");
137                         }
138                     }
139                 }
140             
141                 if (!MeasOnScatter && addScatters) {
142                     //No measurement
143                     nscatters+=1;
144                     //If the scatter has no measurement => then only build a scatter point 
145                     //Here only scatter sensor is available
146                     GBLStripClusterData stripData = makeScatterOnlyData(scatter_sensor,htf,scatter);
147                     
148                     if (stripData != null) {
149                         stripClusterDataList.add(stripData);
150                     }
151                 }
152             } // loop on scatters
153         } // more scatters than hits
154         
155         else { //more hits than scatters 
156             
157    
158             for (TrackerHit stripHit : stripHits) {
159                 HelicalTrackStripGbl strip;
160                 if (stripHit instanceof SiTrackerHitStrip1D) {
161                     
162                     strip = new HelicalTrackStripGbl(makeDigiStrip((SiTrackerHitStrip1D) stripHit), true);
163                     
164                 } else {
165                     SiTrackerHitStrip1D newHit = new SiTrackerHitStrip1D(stripHit);
166                     strip = new HelicalTrackStripGbl(makeDigiStrip(newHit),true);
167                                                             
168                 }
169                 HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) stripHit.getRawHits().get(0)).getDetectorElement();
170                 MultipleScattering.ScatterPoint temp = scatters.getScatterPoint(((RawTrackerHit) strip.getStrip().rawhits().get(0)).getDetectorElement());
171                 
172                 //This is done to correct the fact that the helical fit might not hit a volume
173                 //But hit is associated to the track => must scatter
174                 if (temp == null){
175                     temp = getScatterPointGbl(sensor, strip, htf);
176                     if(temp == null){
177                         return null;
178                     }
179                 }
180 
181                 //Making strip data 
182                 GBLStripClusterData stripData = makeStripData(sensor, strip, htf, temp);
183                 if (stripData != null)
184                     stripClusterDataList.add(stripData);
185             }
186         }
187         
188         //Make sure they are sorted by path Lenght
189         Collections.sort(stripClusterDataList,new SortByPathLength());
190         
191         return stripClusterDataList;
192     }

194     private HelicalTrackStrip makeDigiStrip(SiTrackerHitStrip1D h) {
195         
196         
197         SiTrackerHitStrip1D local  = new SiTrackerHitStrip1D (h.getTransformedHit(TrackerHitType.CoordinateSystem.SENSOR));
198         SiTrackerHitStrip1D global = new SiTrackerHitStrip1D (h.getTransformedHit(TrackerHitType.CoordinateSystem.GLOBAL));
199         
200         ITransform3D trans = local.getLocalToGlobal();
201         Hep3Vector org = trans.transformed(new BasicHep3Vector(0., 0., 0.));
202         Hep3Vector u = global.getMeasuredCoordinate();
203         Hep3Vector v = global.getUnmeasuredCoordinate();
204         
205         // rotate to tracking frame
206         Hep3Vector neworigin = CoordinateTransformations.transformVectorToTracking(org);
207         Hep3Vector newu = CoordinateTransformations.transformVectorToTracking(u);
208         Hep3Vector newv = CoordinateTransformations.transformVectorToTracking(v);
209         
210         double umeas = local.getPosition()[0];
211         double vmin = VecOp.dot(local.getUnmeasuredCoordinate(), local.getHitSegment().getStartPoint());
212         double vmax = VecOp.dot(local.getUnmeasuredCoordinate(), local.getHitSegment().getEndPoint());
213         double du = Math.sqrt(local.getCovarianceAsMatrix().diagonal(0));
214         double dEdx = h.getdEdx();
215         double time = h.getTime();
216         List<RawTrackerHit> rawhits = h.getRawHits();
217         
218         //Print the origins!
219         //System.out.printf("PF::Debug MakeDigiStrip For det element %d! \n", ((HpsSiSensor) det_element).getMillepedeId());
220         //System.out.printf("Strip  origin %s \n", ((BasicHep3Vector)org).toString());
221         //System.out.printf("Strip  origin Tracking %s \n", ((BasicHep3Vector)neworigin).toString());
222         
223         
224         
225         HelicalTrackStrip strip = new HelicalTrackStrip(neworigin, newu, newv, umeas, du, vmin, vmax, dEdx, time, rawhits, null, -1, null);
226                 
227         return strip;
228     }

231     public MultipleScattering.ScatterPoint getScatterPointGbl(HpsSiSensor sensor, HelicalTrackStripGbl strip, HelicalTrackFit htf) {
232         
233         MultipleScattering.ScatterPoint temp = null;
234         
235         Hep3Vector pos = TrackUtils.getHelixPlaneIntercept(htf, strip, Math.abs(_B));
236         if (pos == null) {
237             System.out.println("Can't find track intercept; aborting Track refit");
238             return null;
239         }
240         ScatterAngle scatAngle = new ScatterAngle((HelixUtils.PathToXPlane(htf, pos.x(), 0, 0).get(0)), GblUtils.estimateScatter(sensor, htf, _scattering, _B));
241         temp = new MultipleScattering.ScatterPoint(((RawTrackerHit) strip.getStrip().rawhits().get(0)).getDetectorElement(), scatAngle);
242         temp.setPosition(pos);
243         temp.setDirection(HelixUtils.Direction(htf, scatAngle.PathLen()));
244         
245         return temp;
246     }

248     public static GBLStripClusterData makeStripData(HpsSiSensor sensor, HelicalTrackStripGbl strip, HelicalTrackFit htf, MultipleScattering.ScatterPoint temp) {
249         if (temp == null)
250             return null;
251         
252         // find Millepede layer definition from DetectorElement
253         int millepedeId = sensor.getMillepedeId();
254         // find volume of the sensor (top or bottom)
255         int volume = sensor.isTopLayer() ? 0 : 1;
256         
257         // Center of the strip
258         Hep3Vector origin = strip.origin();
259         
260         //Get the sensor frame transformations
261         //ITransform3D sensor_l2g = sensor.getGeometry().getLocalToGlobal();
262         //Hep3Vector s_org = sensor_l2g.transformed(new BasicHep3Vector(0.,0.,0.));
263         
264         //This is the center of the sensor
265         Hep3Vector s_pos = sensor.getGeometry().getPosition();
266         
267         //Center of the sensor in tracking coordinates.
268         Hep3Vector s_neworigin = CoordinateTransformations.transformVectorToTracking(s_pos);
269         
270         // GBLDATA
271         GBLStripClusterData stripData = new GBLStripClusterData(millepedeId);
272         
273         // Add the volume
274         stripData.setVolume(volume);
275         
276         //This GBLStripData holds a measurement
277         stripData.setScatterOnly(0);
278         
279         // Add to output list
280         
281         // path length to intercept
282         double s3D = temp.getScatterAngle().PathLen() / Math.cos(Math.atan(htf.slope()));
283         
284         // GBLDATA
285         stripData.setPath(temp.getScatterAngle().PathLen());
286         stripData.setPath3D(s3D);
287         
288         //GBLDATA
289         stripData.setU(strip.u());
290         stripData.setV(strip.v());
291         stripData.setW(strip.w());
292 
293         // Print track direction at intercept
294         double phi = htf.phi0() - temp.getScatterAngle().PathLen() / htf.R();
295         double lambda = Math.atan(htf.slope());
296         
297         // GBLDATA
298         stripData.setTrackDir(temp.getDirection());
299         stripData.setTrackPhi(phi);
300         stripData.setTrackLambda(lambda);
301         
302         // Print residual in measurement system
303         // start by find the distance vector between the center and the track position
304         
305         //DEBUG PF: check if the we return the origin or the corner of the sensors. 
306         
307         // Find the rotation from tracking to measurement frame 
308         // This matrix is just 
309         //|u1 u2 u3 | 
310         //|v1 v2 v3 |
311         //|w1 w2 w3 | 
312         
313         //This is clearer and it's identical
314         Hep3Matrix trkToStripRot = VecOp.mult(sensor.getGeometry().getGlobalToLocal().getRotation().getRotationMatrix(),
315                                               CoordinateTransformations.getMatrixInverse());
316         
317         Hep3Vector vdiffTrk = VecOp.sub(temp.getPosition(), s_neworigin);
318                 
319         // then rotate that vector into the measurement frame to get the predicted measurement position
320         Hep3Vector trkpos_meas = VecOp.mult(trkToStripRot, vdiffTrk);
321                 
322         
323         // GBLDATA
324         stripData.setMeas(strip.umeas());
325         stripData.setTrackPos(trkpos_meas);
326         stripData.setMeasErr(strip.du());
327         
328         // GBLDATA
329         stripData.setScatterAngle(temp.getScatterAngle().Angle());
330         
331         return stripData;
332     }

334     public static GBLStripClusterData makeScatterOnlyData(HpsSiSensor sensor, HelicalTrackFit htf, MultipleScattering.ScatterPoint temp) {
335         
336         if (temp ==null)
337             return null;
338         
339         // find Millepede layer definition from DetectorElement
340         int millepedeId = sensor.getMillepedeId();
341         // find volume of the sensor (top or bottom)
342         int volume = sensor.isTopLayer() ? 0 : 1;
343         
344         // GBLDATA
345         GBLStripClusterData stripData = new GBLStripClusterData(millepedeId);
346         stripData.setVolume(volume);
347         
348         //This GBLStripData doesn't hold a measurement
349         stripData.setScatterOnly(1);
350         
351         double s3D = temp.getScatterAngle().PathLen() / Math.cos(Math.atan(htf.slope()));
352         stripData.setPath(temp.getScatterAngle().PathLen());
353         stripData.setPath3D(s3D);
354         
355         //Do not set U,V,W
356         
357         // Print track direction at intercept
358         double phi = htf.phi0() - temp.getScatterAngle().PathLen() / htf.R();
359         double lambda = Math.atan(htf.slope());
360         
361         stripData.setTrackDir(temp.getDirection());
362         stripData.setTrackPhi(phi);
363         stripData.setTrackLambda(lambda);
364         
365         //Do not set the measurement. 
366         //Set a negative large error
367         stripData.setMeasErr(-9999);
368         
369         stripData.setScatterAngle(temp.getScatterAngle().Angle());
370         
371         return stripData;
372     }

376 class SortByPathLength implements Comparator<GBLStripClusterData> {
377     public int compare(GBLStripClusterData a, GBLStripClusterData b) 
378     { 
379         Double as = a.getPath();
380         Double bs = b.getPath();
381         return Double.compare(as,bs);
