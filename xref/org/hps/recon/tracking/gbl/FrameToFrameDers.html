35      //Method to convert Hep3Matrices in Jama Matrices
36      public Matrix getDerivative(Hep3Matrix objectRot, Hep3Matrix composeRot, Hep3Vector objectPos, Hep3Vector composePos) {
37          
38          Vector oPos = new Vector(3);
39          Vector cPos = new Vector(3);
40          
41          for (int i = 0; i < 3; i++) {
42              oPos.set(i,(objectPos.v())[i]);
43              cPos.set(i,(composePos.v())[i]);
44          }
45          
46          
47          //The matrices returned by the getLocalToGlobal are transposed. First fill them, then transpose them. 
48          Matrix oRot = new Matrix(3,3);
49          Matrix cRot = new Matrix(3,3);
50          
51          for (int i = 0; i<3; i++) {
52              for (int j = 0 ; j<3; j++) {
53                  
54                  oRot.set(i,j,objectRot.e(i,j));
55                  cRot.set(i,j,composeRot.e(i,j));
56              }//j
57          }//i
58          
59          oRot.transposeInPlace();
60          cRot.transposeInPlace();
61          
62          if (debug) {
63              System.out.println("PF::FrameToFrame Check Derivatives!");
64              System.out.println("oPos");
65              oPos.print(3,6);
66              System.out.println("cPos");
67              cPos.print(3,6);
68              System.out.println("oRot");
69              oRot.print(3,6);
70              System.out.println("cRot");
71              cRot.print(3,6);
72          }
73          
74          return getDerivative(oRot,cRot,oPos,cPos);
75      }

84          Matrix derivative = new Matrix(6,6);

142     //Gets linear approximated euler Angles
143     private Vector linearEulerAngles(Matrix rotDelta) {
144         
145         Matrix eulerAB = new Matrix(3,3);
146         Vector aB = new Vector(3);
147         
148         eulerAB.set(0,1,1);
149         eulerAB.set(1,0,-1);
150         aB.set(2,1);
151 
152         Matrix eulerC = new Matrix(3,3);
153         Vector C = new Vector(3);
154         eulerC.set(2,0,1);
155         C.set(1,1);
156         
157         Vector eulerAngles =  new Vector(3);
158         eulerAngles = ((eulerAB.times(rotDelta)).times(aB)).plus((eulerC.times(rotDelta)).times(C));
159         return eulerAngles;
160     }

169     //Calculate the derivative DPos/DRot
170     private Matrix derivativePosRot(Matrix RotDet, Matrix RotRot, Vector S) {
171 
172         Vector dEulerA = new Vector(3);
173         Vector dEulerB = new Vector(3);
174         Vector dEulerC = new Vector(3);
175         
176         Vector tmp = new Vector(3);
177         
178         //dEulerA = RotDet * (RotRot.T() * RotDa * RotRot * S);
179         
180         tmp = (RotRot.transpose()).times(RotDa.times(RotRot.times(S)));
181         dEulerA = RotDet.times(tmp);
182         
183         tmp = (RotRot.transpose()).times(RotDb.times(RotRot.times(S)));
184         dEulerB = RotDet.times(tmp);
185 
186         tmp = (RotRot.transpose()).times(RotDc.times(RotRot.times(S)));
187         dEulerC = RotDet.times(tmp);
188         
189         Matrix eulerDeriv = new Matrix(3,3);
190         
191         eulerDeriv.set(0,0,dEulerA.get(0));
192         eulerDeriv.set(1,0,dEulerA.get(1));
193         eulerDeriv.set(2,0,dEulerA.get(2));
194         eulerDeriv.set(0,1,dEulerB.get(0));
195         eulerDeriv.set(1,1,dEulerB.get(1));
196         eulerDeriv.set(2,1,dEulerB.get(2));
197         eulerDeriv.set(0,2,dEulerC.get(0));
198         eulerDeriv.set(1,2,dEulerC.get(1));
199         eulerDeriv.set(2,2,dEulerC.get(2));
200         
201         return eulerDeriv;
202 
203     }

209
