66      /**
67       * Create a new {@link BaseTrack} from a {@link FittedGblTrajectory}.
68       *
69       * @param fittedGblTrajectory
70       * @param helicalTrackFit
71       * @param hitsOnTrack
72       * @param trackType
73       * @param bfield
74       * @return the new {@link BaseTrack} and the kinks along the
75       * {@link GblTrajectory} as a {@link Pair}.
76       */
77      public static Pair<Track, GBLKinkData> makeCorrectedTrack(FittedGblTrajectory fittedGblTrajectory, HelicalTrackFit helicalTrackFit, List<TrackerHit> hitsOnTrack, int trackType, double bfield, boolean storeTrackStates) {
78          //  Initialize the reference point to the origin
79          double[] ref = new double[] { 0., 0., 0. };
80  
81          // Create a new SeedTrack
82          BaseTrack trk = new BaseTrack();
83  
84          /*
85           * we could look into converting the points listed with the GBL
86           * trajectory into the actual TrackerHits in the GBL track.
87           * This was confusing so I didn't do it.
88          // Add the hits to the track, converting them to 1D hits
89          // that inherit from TrackerHit
90          for (GblPoint pt : fittedGblTrajectory.get_traj().getSinglePoints()) {
91             * GblPoint has
92             *  - P2P Jacobian
93             *  - Derivative matrices
94             *  - Position projected onto something?
95            SiTrackerHitStrip1D hit = new SiTrackerHitStrip1D(
96                ,// Hep3Vector position
97                ,// SymmetricMatrix covariance matrix
98                ,// double energy
99                ,// double time
100               ,// List<RawTrackerHit> raw hits
101               // TrackerHitType decoded type
102               );
103           trk.addHit(hit);
104         }
105         */
106         for (TrackerHit hit : hitsOnTrack) {
107             trk.addHit(hit);
108         }
109 
110         // Set state at IP
111         Pair<double[], SymmetricMatrix> correctedHelixParams = fittedGblTrajectory.getCorrectedPerigeeParameters(helicalTrackFit, FittedGblTrajectory.GBLPOINT.IP, bfield);
112 
113         if (correctedHelixParams == null) {
114             return null;
115         }
116         
117 
118         trk.setTrackParameters(correctedHelixParams.getFirst(), bfield);// hack to set the track charge
119         trk.getTrackStates().clear();
120         TrackState stateIP = new BaseTrackState(correctedHelixParams.getFirst(), ref, correctedHelixParams.getSecond().asPackedArray(true), TrackState.AtIP, bfield);
121         trk.getTrackStates().add(stateIP);
122 
123         if (!storeTrackStates) {
124             // just store last state
125             Pair<double[], SymmetricMatrix> correctedHelixParamsLast = fittedGblTrajectory.getCorrectedPerigeeParameters(helicalTrackFit, FittedGblTrajectory.GBLPOINT.LAST, bfield);
126             TrackState stateLast = new BaseTrackState(correctedHelixParamsLast.getFirst(), ref, correctedHelixParamsLast.getSecond().asPackedArray(true), TrackState.AtLastHit, bfield);
127             trk.getTrackStates().add(stateLast);
128         } else {
129 
130             // store states at all 18 sensors
131             int prevID = 0;
132             int dummyCounter = -1;
133             // note: SensorMap doesn't include IP
134             Integer[] sensorsFromMapArray = fittedGblTrajectory.getSensorMap().keySet().toArray(new Integer[0]);
135 
136             for (int i = 0; i < sensorsFromMapArray.length; i++) {
137                 int ilabel = sensorsFromMapArray[i];
138                 int millepedeID = fittedGblTrajectory.getSensorMap().get(ilabel);
139 
140                 // if sensors are missing from track, insert blank TrackState objects
141                 for (int k = 1; k < millepedeID - prevID; k++) {
142                     // uses new lcsim constructor
143                     BaseTrackState dummy = new BaseTrackState(dummyCounter);
144                     trk.getTrackStates().add(dummy);
145                     dummyCounter--;
146                 }
147                 prevID = millepedeID;
148                 Pair<double[], SymmetricMatrix> correctedHelixParamsSensor = fittedGblTrajectory.getCorrectedPerigeeParameters(helicalTrackFit, ilabel, bfield);
149 
150                 // set TrackState location code
151                 int loc = TrackState.AtOther;
152                 if (i == 0) {
153                     loc = TrackState.AtFirstHit;
154                 } else if (i == sensorsFromMapArray.length - 1) {
155                     loc = TrackState.AtLastHit;
156                 }
157                 // insert TrackState at sensor
158                 TrackState stateSensor = new BaseTrackState(correctedHelixParamsSensor.getFirst(), ref, correctedHelixParamsSensor.getSecond().asPackedArray(true), loc, bfield);
159                 trk.getTrackStates().add(stateSensor);
160             }
161 
162         }
163 
164         // Extract kinks from trajectory
165         GBLKinkData kinkData = fittedGblTrajectory.getKinks();
166 
167         // Set other info needed
168         trk.setChisq(fittedGblTrajectory.get_chi2());
169         trk.setNDF(fittedGblTrajectory.get_ndf());
170         trk.setFitSuccess(true);
171         trk.setRefPointIsDCA(true);
172         trk.setTrackType(TrackType.setGBL(trackType, true));
173 
174         LOGGER.fine(String.format("helix chi2 %f ndf %d gbl chi2 %f ndf %d\n", helicalTrackFit.chisqtot(), helicalTrackFit.ndf()[0] + helicalTrackFit.ndf()[1], trk.getChi2(), trk.getNDF()));
175 
176         return new Pair<Track, GBLKinkData>(trk, kinkData);
177     }

183     /**
184      * Do a GBL fit to an arbitrary set of strip hits, with a starting value of
185      * the helix parameters.
186      *
187      * @param helix Initial helix parameters. Only track parameters are used
188      * (not covariance)
189      * @param stripHits Strip hits to be used for the GBL fit. Does not need to
190      * be in sorted order.
191      * @param hth Stereo hits for the track's hit list (these are not used in
192      * the GBL fit). Does not need to be in sorted order.
193      * @param nIterations Number of times to iterate the GBL fit.
194      * @param scattering Multiple scattering manager.
195      * @param bfield B-field
196      * @return The refitted track.
197      */
198     public static Pair<Track, GBLKinkData> refitTrack(HelicalTrackFit helix, Collection<TrackerHit> stripHits, Collection<TrackerHit> hth, int nIterations, int trackType, MultipleScattering scattering, double bfield, boolean storeTrackStates) {
199         Pair<Pair<Track, GBLKinkData>, FittedGblTrajectory> refitTrack = refitTrackWithTraj(helix, stripHits, hth, nIterations, trackType, scattering, bfield, storeTrackStates,false);
200         if(refitTrack == null)
201             return null;
202         else
203             return refitTrack.getFirst();
204     }

206     public static Pair<Pair<Track, GBLKinkData>, FittedGblTrajectory> refitTrackWithTraj(HelicalTrackFit helix, Collection<TrackerHit> stripHits, Collection<TrackerHit> hth, int nIterations, int trackType, MultipleScattering scattering, double bfield, boolean storeTrackStates,boolean includeMS) {
207 
208         List<TrackerHit> allHthList = TrackUtils.sortHits(hth);
209         List<TrackerHit> sortedStripHits = TrackUtils.sortHits(stripHits);
210         FittedGblTrajectory fit = doGBLFit(helix, sortedStripHits, scattering, bfield, 0,includeMS);
211         if(fit==null) return null;
212         for (int i = 0; i < nIterations; i++) {
213             Pair<Track, GBLKinkData> newTrack = makeCorrectedTrack(fit, helix, allHthList, trackType, bfield);
214             
215             if (newTrack == null)
216                 return null;
217             
218             helix = TrackUtils.getHTF(newTrack.getFirst());
219             fit = doGBLFit(helix, sortedStripHits, scattering, bfield, 0,includeMS);
220             if (fit == null)
221                 return null;
222         }
223 
224         Pair<Track, GBLKinkData> mergedTrack = makeCorrectedTrack(fit, helix, allHthList, trackType, bfield, storeTrackStates);
225         return new Pair<Pair<Track, GBLKinkData>, FittedGblTrajectory>(mergedTrack, fit);
226     }

228     /**
229      * Do a GBL fit to a list of {@link TrackerHit}.
230      *
231      * @param htf - seed fit
232      * @param stripHits - list of {@link TrackerHit}.
233      * @param _scattering - estimation of the multiple scattering
234      * {@link MultipleScattering}.
235      * @param bfield - magnitude of B-field.
236      * @param debug - debug flag.
237      * @return the fitted GBL trajectory
238      */
239     public static FittedGblTrajectory doGBLFit(HelicalTrackFit htf, List<TrackerHit> stripHits, MultipleScattering _scattering, double bfield, int debug,boolean includeMS) {
240         List<GBLStripClusterData> stripData = makeStripData(htf, stripHits, _scattering, bfield, debug, includeMS);
241         if (stripData == null)
242             return null;
243         double bfac = Constants.fieldConversion * bfield;
244         FittedGblTrajectory fit = HpsGblRefitter.fit(stripData, bfac, debug > 0);
245         return fit;
246     }

248     /**
249      * Create a list of {@link GBLStripClusterData} objects that can be used as
250      * input to the GBL fitter.
251      *
252      * @param htf
253      * @param stripHits
254      * @param _scattering
255      * @param _B
256      * @param _debug
257      * @return the list of GBL strip cluster data
258      */
259     public static List<GBLStripClusterData> makeStripData(HelicalTrackFit htf, List<TrackerHit> stripHits, MultipleScattering _scattering, double _B, int _debug,boolean includeMS) {
260         List<GBLStripClusterData> stripClusterDataList = new ArrayList<GBLStripClusterData>();
261 
262         // Find scatter points along the path
263         //In principle I could use this to add the hits - TODO ?
264         MultipleScattering.ScatterPoints scatters = _scattering.FindHPSScatterPoints(htf);
265         
266         //Loop over the Scatters
267         
268         //Two things can happen here at each iteration:
269         //1) Nscatters >= Nhits on tracks =>
270         //   Build GBLStripClusterData for both measurements and scatters
271         //2) Nscatters < Nhits on track =>
272         //   The hit has been associated to the track but couldn't find the scatter on the volume => create a scatter at that sensor
273         //   Does this makes sense?
274         
275         //Case (1)
276         //Check if the scatter has a measurement => then use the usual way to build the HelicalTrackStripGbl
277         
278         int nhits = 0;
279         int nscatters = 0;
280         boolean addScatters = true;
281             
282         if (scatters.getPoints().size() >= stripHits.size() && includeMS) {
283             
284             for (MultipleScattering.ScatterPoint scatter : scatters.getPoints()) {
285             
286                 boolean MeasOnScatter = false;
287             
288                 //This is bit inefficient as it always loop on all the stripHits
289                 //TODO: optimize and only check hit once if is in the scatters.
290                 
291                 HpsSiSensor scatter_sensor = (HpsSiSensor) scatter.getDet();
292             
293                 for (TrackerHit stripHit : stripHits) {
294                 
295                     if (MeasOnScatter)
296                         continue;
297                 
298                     IDetectorElement det_element = ((RawTrackerHit) stripHit.getRawHits().get(0)).getDetectorElement();
299                 
300                     if (det_element.equals(scatter.getDet())) {
301                         MeasOnScatter = true;
302                         HpsSiSensor hit_sensor = (HpsSiSensor) det_element;
303                         nhits+=1;
304                         HelicalTrackStripGbl gbl_strip;
305                         if (stripHit instanceof SiTrackerHitStrip1D) {
306                             gbl_strip = new HelicalTrackStripGbl(makeDigiStrip((SiTrackerHitStrip1D) stripHit),true);
307                         } else {
308                             SiTrackerHitStrip1D newHit = new SiTrackerHitStrip1D(stripHit);
309                             gbl_strip = new HelicalTrackStripGbl(makeDigiStrip(newHit), true);
310                         }
311                         //hit_sensor or scatter_sensor, they are the same here
312                         GBLStripClusterData stripData = makeStripData(hit_sensor, gbl_strip, htf, scatter);
313                         if (stripData != null) {
314                             stripClusterDataList.add(stripData); 
315                         }
316                         else {
317                             System.out.printf("WARNING::MakeGblTracks::Couldn't make stripClusterData for hps sensor. Skipping scatter");
318                         }
319                     }
320                 }
321             
322                 if (!MeasOnScatter && addScatters) {
323                     //No measurement
324                     nscatters+=1;
325                     //If the scatter has no measurement => then only build a scatter point 
326                     //Here only scatter sensor is available
327                     GBLStripClusterData stripData = makeScatterOnlyData(scatter_sensor,htf,scatter);
328                     
329                     if (stripData != null) {
330                         stripClusterDataList.add(stripData);
331                     }
332                 }
333             } // loop on scatters
334         } // more scatters than hits
335         
336         else { //more hits than scatters 
337             
338    
339             for (TrackerHit stripHit : stripHits) {
340                 HelicalTrackStripGbl strip;
341                 if (stripHit instanceof SiTrackerHitStrip1D) {
342                     
343                     strip = new HelicalTrackStripGbl(makeDigiStrip((SiTrackerHitStrip1D) stripHit), true);
344                     
345                 } else {
346                     SiTrackerHitStrip1D newHit = new SiTrackerHitStrip1D(stripHit);
347                     strip = new HelicalTrackStripGbl(makeDigiStrip(newHit),true);
348                                                             
349                 }
350                 HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) stripHit.getRawHits().get(0)).getDetectorElement();
351                 MultipleScattering.ScatterPoint temp = scatters.getScatterPoint(((RawTrackerHit) strip.getStrip().rawhits().get(0)).getDetectorElement());
352                 
353                 //This is done to correct the fact that the helical fit might not hit a volume
354                 //But hit is associated to the track => must scatter
355                 if (temp == null){
356                     temp = getScatterPointGbl(sensor, strip, htf, _scattering, _B);
357                     if(temp == null){
358                         return null;
359                     }
360                 }
361 
362                 //Making strip data 
363                 GBLStripClusterData stripData = makeStripData(sensor, strip, htf, temp);
364                 if (stripData != null)
365                     stripClusterDataList.add(stripData);
366             }
367         }
368         
369         return stripClusterDataList;
370     }

372     public static MultipleScattering.ScatterPoint getScatterPointGbl(HpsSiSensor sensor, HelicalTrackStripGbl strip, HelicalTrackFit htf, MultipleScattering _scattering, double _B) {
373 
374         MultipleScattering.ScatterPoint temp = null;
375 
376         Hep3Vector pos = TrackUtils.getHelixPlaneIntercept(htf, strip, Math.abs(_B));
377         if (pos == null) {
378             System.out.println("Can't find track intercept; aborting Track refit");
379             return null;
380         }
381         ScatterAngle scatAngle = new ScatterAngle((HelixUtils.PathToXPlane(htf, pos.x(), 0, 0).get(0)), GblUtils.estimateScatter(sensor, htf, _scattering, _B));
382         temp = new MultipleScattering.ScatterPoint(((RawTrackerHit) strip.getStrip().rawhits().get(0)).getDetectorElement(), scatAngle);
383         temp.setPosition(pos);
384         temp.setDirection(HelixUtils.Direction(htf, scatAngle.PathLen()));
385 
386         return temp;
387     }

389     public static GBLStripClusterData makeScatterOnlyData(HpsSiSensor sensor, HelicalTrackFit htf, MultipleScattering.ScatterPoint temp) {
390         
391         if (temp ==null)
392             return null;
393         
394         // find Millepede layer definition from DetectorElement
395         int millepedeId = sensor.getMillepedeId();
396         // find volume of the sensor (top or bottom)
397         int volume = sensor.isTopLayer() ? 0 : 1;
398                                                 
399         // GBLDATA
400         GBLStripClusterData stripData = new GBLStripClusterData(millepedeId);
401         stripData.setVolume(volume);
402         
403         //This GBLStripData doesn't hold a measurement
404         stripData.setScatterOnly(1);
405 
406         double s3D = temp.getScatterAngle().PathLen() / Math.cos(Math.atan(htf.slope()));
407         stripData.setPath(temp.getScatterAngle().PathLen());
408         stripData.setPath3D(s3D);
409         
410         //Do not set U,V,W
411         
412         // Print track direction at intercept
413         double phi = htf.phi0() - temp.getScatterAngle().PathLen() / htf.R();
414         double lambda = Math.atan(htf.slope());
415         
416         stripData.setTrackDir(temp.getDirection());
417         stripData.setTrackPhi(phi);
418         stripData.setTrackLambda(lambda);
419         
420         //Do not set the measurement. 
421         //Set a negative large error
422         stripData.setMeasErr(-9999);
423 
424         stripData.setScatterAngle(temp.getScatterAngle().Angle());
425         
426         return stripData;
427     }

543     private static Hep3Matrix getTrackToStripRotation(SiSensor sensor) {
544         // This function transforms the hit to the sensor coordinates
545 
546         // Transform from JLab frame to sensor frame (done through the RawTrackerHit)
547         SiSensorElectrodes electrodes = sensor.getReadoutElectrodes(ChargeCarrier.HOLE);
548         ITransform3D detToStrip = electrodes.getGlobalToLocal();
549         // Get rotation matrix
550         Hep3Matrix detToStripMatrix = detToStrip.getRotation().getRotationMatrix();
551 
552         return VecOp.mult(detToStripMatrix, CoordinateTransformations.getMatrixInverse());
553     }

555     private static HelicalTrackStrip makeDigiStrip(SiTrackerHitStrip1D h) {
556 
557         
558         SiTrackerHitStrip1D local  = new SiTrackerHitStrip1D (h.getTransformedHit(TrackerHitType.CoordinateSystem.SENSOR));
559         SiTrackerHitStrip1D global = new SiTrackerHitStrip1D (h.getTransformedHit(TrackerHitType.CoordinateSystem.GLOBAL));
560 
561         ITransform3D trans = local.getLocalToGlobal();
562         Hep3Vector org = trans.transformed(new BasicHep3Vector(0., 0., 0.));
563         Hep3Vector u = global.getMeasuredCoordinate();
564         Hep3Vector v = global.getUnmeasuredCoordinate();
565 
566         // rotate to tracking frame
567         Hep3Vector neworigin = CoordinateTransformations.transformVectorToTracking(org);
568         Hep3Vector newu = CoordinateTransformations.transformVectorToTracking(u);
569         Hep3Vector newv = CoordinateTransformations.transformVectorToTracking(v);
570 
571         double umeas = local.getPosition()[0];
572         double vmin = VecOp.dot(local.getUnmeasuredCoordinate(), local.getHitSegment().getStartPoint());
573         double vmax = VecOp.dot(local.getUnmeasuredCoordinate(), local.getHitSegment().getEndPoint());
574         double du = Math.sqrt(local.getCovarianceAsMatrix().diagonal(0));
575         double dEdx = h.getdEdx();
576         double time = h.getTime();
577         List<RawTrackerHit> rawhits = h.getRawHits();
578 
579         //Print the origins!
580         //System.out.printf("PF::Debug MakeDigiStrip For det element %d! \n", ((HpsSiSensor) det_element).getMillepedeId());
581         //System.out.printf("Strip  origin %s \n", ((BasicHep3Vector)org).toString());
582         //System.out.printf("Strip  origin Tracking %s \n", ((BasicHep3Vector)neworigin).toString());
583         
584         
585 
586         HelicalTrackStrip strip = new HelicalTrackStrip(neworigin, newu, newv, umeas, du, vmin, vmax, dEdx, time, rawhits, null, -1, null);
587         
588 
589         return strip;
590     }
