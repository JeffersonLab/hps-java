48      public void runExample(int inTry, int inLayer,boolean idebug) {
49          
50          nTry=inTry;
51          nLayer=inLayer;
52          debug = idebug;
53          
54          setupPlots();
55          System.out.println("Running GBL Example!");
56          
57          long startTime = System.nanoTime();
58          
59  
60          double sinLambda = 0.3;
61          double cosLambda = Math.sqrt(1.0-sinLambda*sinLambda);
62          double sinPhi = 0.;
63          double cosPhi = Math.sqrt(1.0 - sinPhi * sinPhi);
64          
65          Matrix uvDir = new Matrix(2, 3);
66          uvDir.set(0, 0, -sinPhi);
67          uvDir.set(0, 1, cosPhi);
68          uvDir.set(0, 2, 0.);
69          uvDir.set(1, 0, -sinLambda * cosPhi);
70          uvDir.set(1, 1, -sinLambda * sinPhi);
71          uvDir.set(1, 2, cosLambda);
72  
73          if (debug) {
74              System.out.println("UVDIR");
75              uvDir.print(3,4);
76          }
77            
78  
79          Vector measErr = new Vector(2);
80          measErr.set(0,0.001);
81          measErr.set(1,0.001);
82          Vector measPrec = new Vector(2);
83          measPrec.set(0, 1.0 / (measErr.get(0) * measErr.get(0)));
84          measPrec.set(1, 1.0 / (measErr.get(0) * measErr.get(0)));
85  
86          if (debug) {
87              System.out.println("measPrec");
88              measPrec.print(2,4);
89          }
90  
91          Matrix measInvCov = new Matrix(2, 2);
92          //Set it to zero
93          measInvCov.times(0);
94          measInvCov.set(0,0,measPrec.get(0));
95          measInvCov.set(1,1,measPrec.get(1));
96  
97          if (debug) {
98              System.out.println("measInvCov");
99              measInvCov.print(2,4);
100         }
101         
102         Vector scatErr = new Vector(2);
103         scatErr.set(0,0.001);
104         scatErr.set(1,0.001);
105         Vector scatPrec = new Vector(2);
106         scatPrec.set(0, 1.0 / (scatErr.get(0) * scatErr.get(0)));
107         scatPrec.set(1, 1.0 / (scatErr.get(1) * scatErr.get(1)));
108 
109         if (debug) {
110             System.out.println("scatPrec");
111             scatPrec.print(2,4);
112         }
113 
114         Vector clPar = new Vector(5);
115         Vector clErr = new Vector(5);
116         clErr.set(0,0.001);
117         clErr.set(1,-0.1);
118         clErr.set(2,0.2);
119         clErr.set(3,-0.15);
120         clErr.set(4,0.25);
121 
122         if (debug) {
123             System.out.println("clErr");
124             clErr.print(2,4);
125         }
126 
127         Matrix clCov = new Matrix(5, 5);
128         
129         
130         double bfac = 0.2998;
131         double step = 1.5 / cosLambda;
132         
133         double Chi2Sum = 0.;
134         int NdfSum = 0;
135         double LostSum = 0.;
136         int numFit = 0;
137         Matrix jacPointToPoint = new Matrix(5, 5);
138         
139         //Check on factor
140         //System.out.println(norm.sample());
141         double norm_sample = 0.5;
142         
143         for (int iTry = 1; iTry<=nTry; iTry+=1) {
144             for (int i = 0; i<5; i+=1) {
145                 
146                 clPar.set(i, clErr.get(i)*norm_sample);
147                 //clPar.set(i, clErr.get(i)*norm.sample());
148                 //System.out.println("clPar " + i + " " + clPar.get(i));
149                 //aida.histogram1D("clPar_true_"+String.valueOf(i)).fill(clPar.get(i));
150             }
151             
152             clCov.times(0);
153             for (int i = 0; i<5; i+=1) {
154                 clCov.set(i,i,1.0*clErr.get(i)*clErr.get(i));
155             }
156             
157             if (debug) {
158                 System.out.println("clPar");
159                 clPar.print(2,4);
160                 System.out.println("clCov");
161                 clCov.print(5,7);
162             }
163             
164             
165             double s = 0;
166             jacPointToPoint.UnitMatrix();
167             List<GblPoint> listOfPoints = new ArrayList<GblPoint>();
168             
169             for (int iLayer = 0; iLayer<nLayer; iLayer+=1) {
170                 double sinStereo = (iLayer % 2 == 0) ? 0. : 0.1;
171                 double cosStereo = sqrt(1.0 - sinStereo * sinStereo);
172                 Matrix mDirT = new Matrix(3, 2);
173                 mDirT.times(0);
174                 mDirT.set(1,0,cosStereo);
175                 mDirT.set(2,0,sinStereo);
176                 mDirT.set(1,1,-sinStereo);
177                 mDirT.set(2,1,cosStereo);
178 
179                 if (debug) {
180                     System.out.println("mDirT");
181                     mDirT.print(3,5);
182                 }
183                 
184                 Matrix proM2l = uvDir.times(mDirT);
185 
186                 if (debug) {
187                     System.out.println("ProM2L");
188                     proM2l.print(2,3);
189                 }
190                 
191                 Matrix proL2m = proM2l.copy();
192                 proL2m = proL2m.inverse();
193                 
194                 if (debug) {
195                     System.out.println("proL2m");
196                     proL2m.print(2,3);
197                 }
198                 
199                 GblPoint pointMeas = new GblPoint(jacPointToPoint);
200                 if (debug) {
201                     System.out.println("JacPointToPoint");
202                     jacPointToPoint.print(5,5);
203                 }
204                 
205                 Vector meas = new Vector(2);
206                 Vector clPar_tail = new Vector(2);
207                 clPar_tail.set(0,clPar.get(3));
208                 clPar_tail.set(1,clPar.get(4));
209                 
210                 if (debug) {
211                     System.out.println("clPar.tail");
212                     clPar_tail.print(2,3);
213                 }
214                 
215                 meas = proL2m.times(clPar_tail);
216                 if (debug) {
217                     System.out.println("measurement before smearing");
218                     meas.print(2,3);
219                 }
220                 
221                 for (int i=0; i<2; i+=1) {
222                     meas.set(i,meas.get(i)+measErr.get(i) * norm_sample);
223                     //meas.set(i,meas.get(i)+measErr.get(i) * norm.sample());
224                 }
225                 
226                 if (debug) {
227                     System.out.println("measurement after smearing");
228                     meas.print(2,3);
229                 }
230                 
231                 pointMeas.addMeasurement(proL2m,meas,measPrec,0.);
232                 //pointMeas.printPoint(1);
233                 
234                 
235                 listOfPoints.add(pointMeas);
236 
237                 if (debug) {
238                     System.out.println("Points::" + listOfPoints.size());
239                 }
240                 
241                 jacPointToPoint = gblSimpleJacobianSvn(step, cosLambda, bfac);
242                 
243                 clPar = jacPointToPoint.times(clPar);
244                 clCov = jacPointToPoint.times(clCov.times(jacPointToPoint.transpose()));
245                 s=s+step;
246                 
247                 if (iLayer < nLayer-1) {
248                     Vector scat = new Vector(2);
249                     scat.set(0,0.);
250                     scat.set(1,0.);
251                     GblPoint pointScat = new GblPoint(jacPointToPoint);
252                     pointScat.addScatterer(scat,scatPrec);
253                     listOfPoints.add(pointScat);
254                     
255                     if (debug) {
256                         System.out.println("clPar before Scatter");
257                         clPar.print(1,3);
258                         clCov.print(5,5);
259                     }
260                     
261                     //Only change the slopes
262                     for (int i =0; i<2; i+=1) {
263                         double dslope = scatErr.get(i)*norm_sample;
264                         //double dslope = scatErr.get(i)*norm.sample();
265                         double slope  = clPar.get(i + 1);
266                         double dslope_prec = scatErr.get(i) * scatErr.get(i);
267                         double slope_prec = clCov.get(i+1, i+1);
268                         clPar.set(i + 1,slope+dslope);
269                         clCov.set(i + 1, i + 1, slope_prec + dslope_prec);
270                     }
271                     if (debug) {
272                         System.out.println("clPar after Scatter");
273                         clPar.print(1,3);
274                         clCov.print(5,5);
275                     }
276                     
277                     clPar = jacPointToPoint.times(clPar);
278                     clCov = jacPointToPoint.times(clCov.times(jacPointToPoint.transpose()));
279                     s=s+step;
280                 }
281                 
282                 if (debug) {
283                     System.out.println("--------");
284                     System.out.println("clPar/clCov/ds before next point");
285                     clPar.print(1,3);
286                     System.out.println("--------");
287                     clCov.print(5,5);
288                     System.out.println("--------");
289                     System.out.println(s);
290                     System.out.println("--------");
291                 }
292                     
293                 
294             }//layers
295 
296             GblTrajectory traj = new GblTrajectory(listOfPoints);
297             if (!traj.isValid()) {
298                 System.out.println("Example1: " + " Invalid GblTrajectory -> skip");
299             }
300             double[] dVals = new double[2];
301             int [] iVals = new int[1];
302             traj.fit(dVals,iVals,"");
303             Chi2Sum += dVals[0];
304             NdfSum += iVals[0];
305             LostSum += dVals[1];
306             numFit++;
307             aida.histogram1D("Chi2").fill(dVals[0]);
308             aida.histogram1D("Ndf").fill(iVals[0]);
309             aida.histogram1D("Chi2_Ndf").fill(dVals[0]/(double)iVals[0]);
310         
311         
312             if (debug) {
313                 traj.printTrajectory(1);
314                 traj.printPoints(1);
315             }
316             traj.printData();
317             
318             int numData[] = new int[1];
319             System.out.printf("Example1::Size list Of Points %d %d\n",listOfPoints.size(),traj.getNpointsOnTraj());
320             for (int ilabel=1; ilabel<=listOfPoints.size(); ilabel++) {
321                 //Check biased residual
322                 List<Double> aResiduals   = new ArrayList<Double>();
323                 List<Double> aMeasErrors  = new ArrayList<Double>();
324                 List<Double> aResErrors   = new ArrayList<Double>();
325                 List<Double> aDownWeights = new ArrayList<Double>();
326                 traj.getMeasResults(ilabel,numData,aResiduals,aMeasErrors,aResErrors,aDownWeights);           
327                 for (int i=0; i<numData[0];i++) {
328                     System.out.printf("Example1::measResults %d %d %f %f %f \n",ilabel, i, aResiduals.get(i),aMeasErrors.get(i),aResErrors.get(i));
329                 }
330             }
331             
332             //System.out.printf("%d %d %f %f %f %f",label,numData[1]);
333             //Check unbiased residual
334             System.out.println("Checking unbiased residual => remove label 3: Internal Measurement");
335             GblTrajectory traj_unbias = new GblTrajectory(listOfPoints);
336             if (!traj.isValid()) {
337                 System.out.println("Example1: " + " Invalid GblTrajectory -> skip");
338             }
339             double[] dVals_u = new double[2];
340             int [] iVals_u = new int[1];
341             //Remove point label 3
342             int ulabel = 3;
343             traj.fit(dVals_u,iVals_u,"",ulabel);
344             
345             
346             int numData_u[] = new int[1];
347             List<Double> aResiduals_u   = new ArrayList<Double>();
348             List<Double> aMeasErrors_u  = new ArrayList<Double>();
349             List<Double> aResErrors_u   = new ArrayList<Double>();
350             List<Double> aDownWeights_u = new ArrayList<Double>();
351             
352             traj.getMeasResults(ulabel,numData_u,aResiduals_u,aMeasErrors_u,aResErrors_u,aDownWeights_u);           
353             for (int i_u=0; i_u<numData_u[0];i_u++) {
354                 System.out.printf("Example1::measResults %d %d %f %f %f \n",ulabel, i_u, aResiduals_u.get(i_u),aMeasErrors_u.get(i_u),aResErrors_u.get(i_u));
355             }
356             
357             System.out.println("Checking unbiased residual => remove label 4: Internal Kink");
358             ulabel = 4;
359             traj.fit(dVals_u,iVals_u,"",ulabel);
360             aResiduals_u.clear();
361             aMeasErrors_u.clear();
362             aResErrors_u.clear();
363             aDownWeights_u.clear();
364             
365             traj.getMeasResults(ulabel,numData_u,aResiduals_u,aMeasErrors_u,aResErrors_u,aDownWeights_u);
366             for (int i_u=0; i_u<numData_u[0];i_u++) {
367                 System.out.printf("Example1::measResults %d %d %f %f %f \n",ulabel, i_u, aResiduals_u.get(i_u),aMeasErrors_u.get(i_u),aResErrors_u.get(i_u));
368             }
369             
370             System.out.println("Checking biased residual again");
371             traj.fit(dVals_u,iVals_u,"");
372             aResiduals_u.clear();
373             aMeasErrors_u.clear();
374             aResErrors_u.clear();
375             aDownWeights_u.clear();
376             
377             traj.getMeasResults(ulabel,numData_u,aResiduals_u,aMeasErrors_u,aResErrors_u,aDownWeights_u);
378             for (int i_u=0; i_u<numData_u[0];i_u++) {
379                 System.out.printf("Example1::measResults (b) %d %d %f %f %f \n",ulabel, i_u, aResiduals_u.get(i_u),aMeasErrors_u.get(i_u),aResErrors_u.get(i_u));
380             }
381         }
382         
383         long endTime = System.nanoTime();
384         long duration = endTime - startTime;

400     private Matrix gblSimpleJacobianSvn(double ds, double cosl, double bfac) {
401         
402         Matrix jac = new Matrix(5,5);
403         jac.UnitMatrix();
404         jac.set(1, 0, -bfac * ds * cosl);
405         jac.set(3, 0, -0.5 * bfac * ds * ds * cosl);
406         jac.set(3, 1, ds);
407         jac.set(4, 2, ds);
408         return jac;
409         
410     }
