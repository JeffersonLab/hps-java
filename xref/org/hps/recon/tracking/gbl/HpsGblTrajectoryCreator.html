35      public List<GblPointJna> MakeGblPointsList(List<GBLStripClusterData> hits, GBLBeamSpotPoint bs, double bfac, Map<Integer, Integer> sensorMap, Map<Integer, Double> pathLengthMap) {
36          
37          
38          // Save the association between strip cluster and label, and between label and path length
39          //Map<Integer, Double> pathLengthMap = new HashMap<Integer, Double>();
40          //Map<Integer, Integer> sensorMap = new HashMap<Integer, Integer>();
41          List<GblPointJna> listOfPoints = new ArrayList<GblPointJna>();
42          
43          
44  
45          int iLabel = 0;
46          double s = 0.;
47          Matrix jacPointToPoint = new Matrix(5,5);
48          jacPointToPoint.UnitMatrix();
49                  
50          if (bs != null) {
51  
52              double cosL_bs = 1. / Math.sqrt(1. + bs._tanLambda*bs._tanLambda); 
53              //double s = 0.;
54              s = bs._arcLength / cosL_bs;
55                          
56              //start trajectory at the beamspot
57              jacPointToPoint = gblSimpleJacobianLambdaPhi(bs._arcLength / cosL_bs, cosL_bs, abs(bfac));
58          
59          
60              GblPointJna bs_point = new GblPointJna(jacPointToPoint);
61              bs_point.addMeasurement(bs._projL2m,bs._aResidual,bs._aPrecision,0.);
62              listOfPoints.add(bs_point);
63              iLabel=listOfPoints.size();
64              pathLengthMap.put(iLabel, s);
65          
66              //jacPointToPoint.UnitMatrix(); 
67              //jacPointToPoint = gblSimpleJacobianLambdaPhi(0. - bs._arcLength / cosL_bs, cosL_bs, abs(bfac));
68              //GblPointJna ref_point = new GblPointJna(jacPointToPoint);
69              //listOfPoints.add(ref_point);
70              //iLabel = listOfPoints.size();
71              //pathLengthMap.put(iLabel,0.);
72              
73          } else {
74              
75              GblPointJna ref_point = new GblPointJna(jacPointToPoint);
76              listOfPoints.add(ref_point);
77              iLabel = listOfPoints.size();
78              pathLengthMap.put(iLabel,0.);
79          }
80          
81          s = 0.;
82          
83          
84          // Loop over strips
85          int n_strips = hits.size();
86          for (int istrip = 0; istrip != n_strips; ++istrip) {
87              GBLStripClusterData strip = hits.get(istrip);
88              // MG--9/18/2015--beamspot has Id=666/667...don't include it in the GBL fit
89              if (strip.getId() > 99) {
90                  continue;
91              }
92              if (_debug) {
93                  System.out.println("HpsGblFitter: Processing strip " + istrip + " with id/layer " + strip.getId());
94              }
95              // Path length step for this cluster
96              double step = strip.getPath3D() - s;
97              if (_debug) {
98                  System.out.println("HpsGblFitter: " + "Path length step " + step + " from " + s + " to " + strip.getPath3D());
99              }
100             
101             if (strip.getScatterOnly() == 1 && _debug) {
102                 System.out.println("This is scatter only on sensor: MPID " + strip.getId());
103             }
104 
105             // get measurement frame unit vectors
106             Hep3Vector u = strip.getU();
107             Hep3Vector v = strip.getV();
108             Hep3Vector w = strip.getW();
109 
110             // Measurement direction (perpendicular and parallel to strip direction)
111             Matrix mDir = new Matrix(2, 3);
112             mDir.set(0, 0, u.x());
113             mDir.set(0, 1, u.y());
114             mDir.set(0, 2, u.z());
115             mDir.set(1, 0, v.x());
116             mDir.set(1, 1, v.y());
117             mDir.set(1, 2, v.z());
118             if (_debug) {
119                 System.out.println("HpsGblFitter: " + "mDir");
120                 mDir.print(4, 6);
121             }
122             Matrix mDirT = mDir.copy().transpose();
123 
124             if (_debug) {
125                 System.out.println("HpsGblFitter: " + "mDirT");
126                 mDirT.print(4, 6);
127             }
128 
129             // Track direction
130             double sinLambda = sin(strip.getTrackLambda());// ->GetLambda());
131             double cosLambda = sqrt(1.0 - sinLambda * sinLambda);
132             double sinPhi = sin(strip.getTrackPhi());// ->GetPhi());
133             double cosPhi = sqrt(1.0 - sinPhi * sinPhi);
134 
135             if (_debug) {
136                 System.out.println("HpsGblFitter: " + "Track direction sinLambda=" + sinLambda + " sinPhi=" + sinPhi);
137             }
138 
139             // Track direction in curvilinear frame (U,V,T)
140             // U = Z x T / |Z x T|, V = T x U
141             Matrix uvDir = new Matrix(2, 3);
142             uvDir.set(0, 0, -sinPhi);
143             uvDir.set(0, 1, cosPhi);
144             uvDir.set(0, 2, 0.);
145             uvDir.set(1, 0, -sinLambda * cosPhi);
146             uvDir.set(1, 1, -sinLambda * sinPhi);
147             uvDir.set(1, 2, cosLambda);
148 
149             if (_debug) {
150                 System.out.println("HpsGblFitter: " + "uvDir");
151                 uvDir.print(6, 4);
152             }
153 
154             // projection from measurement to local (curvilinear uv) directions (duv/dm)
155             Matrix proM2l = uvDir.times(mDirT);
156 
157             // projection from local (uv) to measurement directions (dm/duv)
158             Matrix proL2m = proM2l.copy();
159             
160             //Invertible
161             if (strip.getScatterOnly() == 0)
162                 proL2m = proL2m.inverse();
163 
164             if (_debug) {
165                 System.out.println("HpsGblFitter: " + "proM2l:");
166                 proM2l.print(4, 6);
167                 System.out.println("HpsGblFitter: " + "proL2m:");
168                 proL2m.print(4, 6);
169                 System.out.println("HpsGblFitter: " + "proM2l*proL2m (should be unit matrix):");
170                 (proM2l.times(proL2m)).print(4, 6);
171             }
172 
173             // measurement/residual in the measurement system
174             // only 1D measurement in u-direction, set strip measurement direction to zero
175             Vector meas = new Vector(2);
176             // double uRes = strip->GetUmeas() - strip->GetTrackPos().x(); // how can this be correct?
177             double uRes = strip.getMeas() - strip.getTrackPos().x();
178             meas.set(0, uRes);
179             meas.set(1, 0.);
180             Vector measErr = new Vector(2);
181             measErr.set(0, strip.getMeasErr());
182             measErr.set(1, 0.);
183             Vector measPrec = new Vector(2);
184             measPrec.set(0, 1.0 / (measErr.get(0) * measErr.get(0)));
185             measPrec.set(1, 0.);
186 
187             if (_debug) {
188                 System.out.println("HpsGblFitter: " + "meas: ");
189                 meas.print(4, 6);
190                 System.out.println("HpsGblFitter: " + "measErr:");
191                 measErr.print(4, 6);
192                 System.out.println("HpsGblFitter: " + "measPrec:");
193                 measPrec.print(4, 6);
194             }
195 
196             // Find the Jacobian to be able to propagate the covariance matrix to this strip position
197             jacPointToPoint = gblSimpleJacobianLambdaPhi(step, cosLambda, abs(bfac));
198             if (_debug) {
199                 System.out.println("HpsGblFitter: " + "jacPointToPoint to extrapolate to this point:");
200                 jacPointToPoint.print(4, 6);
201             }
202 
203             GblPointJna point = new GblPointJna(jacPointToPoint);
204             // Add measurement to the point
205 
206             if (strip.getScatterOnly()==0)
207                 point.addMeasurement(proL2m, meas, measPrec, 0.);
208             // Add scatterer in curvilinear frame to the point
209             // no direction in this frame
210             Vector scat = new Vector(2);
211             // Scattering angle in the curvilinear frame
212             // Note the cosLambda to correct for the projection in the phi direction
213             Vector scatErr = new Vector(2);
214             scatErr.set(0, strip.getScatterAngle());
215             scatErr.set(1, strip.getScatterAngle() / cosLambda);
216             Vector scatPrec = new Vector(2);
217             scatPrec.set(0, 1.0 / (scatErr.get(0) * scatErr.get(0)));
218             scatPrec.set(1, 1.0 / (scatErr.get(1) * scatErr.get(1)));
219 
220             // add scatterer 
221             point.addScatterer(scat, scatPrec);
222             if (_debug) {
223                 System.out.println("HpsGblFitter: " + "adding scatError to this point:");
224                 scatErr.print(4, 6);
225             }

227             // Add this GBL point to list that will be used in fit
228             listOfPoints.add(point);
229             iLabel = listOfPoints.size();
230             // save path length and sensor-number to each point
231             pathLengthMap.put(iLabel, s + step);
232             sensorMap.put(iLabel, strip.getId());

234             // // Calculate global derivatives for this point
235             // track direction in tracking/global frame
236             Hep3Vector tDirGlobal = new BasicHep3Vector(cosPhi * cosLambda, sinPhi * cosLambda, sinLambda);
237 
238             // Cross-check that the input is consistent
239             if (VecOp.sub(tDirGlobal, strip.getTrackDirection()).magnitude() > 0.00001) {

244             Hep3Vector tDirMeas = new BasicHep3Vector(VecOp.dot(tDirGlobal, u), VecOp.dot(tDirGlobal, v), VecOp.dot(tDirGlobal, w));

248             double vmeas = 0.;
249             double wmeas = 0.;

287                 //            }
288                 //            LOGGER.info("\n" + logders);
289                 
290                 //LOGGER.info("uRes " + strip.getId() + " uRes " + uRes + " pred (" + strip.getTrackPos().x() + "," + strip.getTrackPos().y() + "," + strip.getTrackPos().z() + ") s(3D) " + strip.getPath3D());
291             }
292             // go to next point
293             s += step;
294             
295         } // strips

348     private static Matrix gblSimpleJacobianLambdaPhi(double ds, double cosl, double bfac) {
349         /**
350          * Simple jacobian: quadratic in arc length difference. using lambda phi as directions
351          *
352          * @param ds: arc length difference
353          * @type ds: float
354          * @param cosl: cos(lambda)
355          * @type cosl: float
356          * @param bfac: Bz*c
357          * @type bfac: float
358          * @return: jacobian to move by 'ds' on trajectory
359          * @rtype: matrix(float) ajac(1,1)= 1.0D0 ajac(2,2)= 1.0D0 ajac(3,1)=-DBLE(bfac*ds) ajac(3,3)= 1.0D0
360          *         ajac(4,1)=-DBLE(0.5*bfac*ds*ds*cosl) ajac(4,3)= DBLE(ds*cosl) ajac(4,4)= 1.0D0 ajac(5,2)= DBLE(ds)
361          *         ajac(5,5)= 1.0D0 ''' jac = np.eye(5) jac[2, 0] = -bfac * ds jac[3, 0] = -0.5 * bfac * ds * ds * cosl
362          *         jac[3, 2] = ds * cosl jac[4, 1] = ds return jac
363          */
364         Matrix jac = new Matrix(5, 5);
365         jac.UnitMatrix();
366         jac.set(2, 0, -bfac * ds);
367         jac.set(3, 0, -0.5 * bfac * ds * ds * cosl);
368         jac.set(3, 2, ds * cosl);
369         jac.set(4, 1, ds);
370         return jac;
371     }
