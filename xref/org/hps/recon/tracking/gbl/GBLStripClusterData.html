72      public void print() {
73          for (int i = 0; i < GBLINT.BANK_INT_SIZE; ++i) {
74              System.out.println("bank int " + i+ " "+bank_int[i]);
75          }
76          for (int i = 0; i < GBLDOUBLE.BANK_DOUBLE_SIZE; ++i) {
77              System.out.println("bank double " + i+ " "+bank_double[i]);
78          }
79          
80      }

82      /*
83       * Constructor from GenericObject TODO add size checks for backwards compatability
84       */
85      public GBLStripClusterData(GenericObject o) {
86          for (int i = 0; i < GBLINT.BANK_INT_SIZE; ++i) {
87              bank_int[i] = o.getIntVal(i);
88          }
89          for (int i = 0; i < GBLDOUBLE.BANK_DOUBLE_SIZE; ++i) {
90              bank_double[i] = o.getDoubleVal(i);
91          }
92  
93      }
