14  /**
15   * wrapper class for GblTrajectory JNA functions
16   * <p>
17   * This class re-promotes these JNA functions to be
18   * class member functions while providing some translation
19   * capabilities form java types into their C-style
20   * counterparts required by JNA.
21   * <p>
22   * <b>Note</b>: While the points are passed into this constructor,
23   * the memory owned by those points is never owned by this
24   * class. It is up to the user to delete the points when they
25   * are done using them.
26   */
27  public class GblTrajectoryJna {
28      private Pointer self;
29      
30      // copy a java-style list into a JNA C-style array that can then be copied into a std vector in GBL
31      public GblTrajectoryJna(List<GblPointJna> points, boolean flagCurv, boolean flagU1dir, boolean flagU2dir) { 
32          Pointer [] ppoints = new Pointer[points.size()];
33  
34          int ipoint=-1;
35          for (GblPointJna point : points) {
36              ipoint+=1;
37              ppoints[ipoint]  = point.getPtr();
38          }
39          
40          self = GblInterface.INSTANCE.GblTrajectoryCtorPtrArray(ppoints, points.size(), 
41              flagCurv?1:0, flagU1dir?1:0, flagU2dir?1:0);
42          if (self == Pointer.NULL)
43              System.out.println("Failed generating trajectory");
44                  
45      }
46  
47      //Simple trajectory constructor with seed 
48      public GblTrajectoryJna(List<GblPointJna> points, int aLabel, Matrix seed, boolean flagCurv, boolean flagU1dir, boolean flagU2dir) { 
49          Pointer [] ppoints = new Pointer[points.size()];
50  
51          int ipoint=-1;
52          for (GblPointJna point : points) {
53              ipoint+=1;
54              ppoints[ipoint]  = point.getPtr();
55          }
56          
57          if (seed.getRowDimension() != 5 || seed.getColumnDimension() != 5)
58              throw new RuntimeException("GBLTrajectoryJna::Seed must be 5x5 matrix.\n");
59          
60          //The seed is 5x5 matrix
61          double [] seedArray = seed.getColumnPackedCopy();
62          
63          
64          self = GblInterface.INSTANCE.GblTrajectoryCtorPtrArraySeed(ppoints, points.size(), aLabel, seedArray, flagCurv?1:0, flagU1dir?1:0, flagU2dir?1:0);
65          
66      }

68      //Composed (curved) trajectory constructor from a list of points and transformation (at inner (first) point)
69      //Only 2 tracks, supported for the moment
70      
71      public GblTrajectoryJna(List < Pair <List<GblPointJna>, Matrix> > PointsAndTransList) {
72          Pointer [] ppoints_1 = new Pointer[PointsAndTransList.get(0).getFirst().size()];
73          Pointer [] ppoints_2 = new Pointer[PointsAndTransList.get(0).getFirst().size()];
74  
75          int npoints_1 = PointsAndTransList.get(0).getFirst().size();
76          int npoints_2 = PointsAndTransList.get(1).getFirst().size();
77         
78          
79          int ipoint=-1;
80          for (GblPointJna point : PointsAndTransList.get(0).getFirst()) {
81              ipoint+=1;
82              ppoints_1[ipoint]  = point.getPtr();
83          }
84          
85          ipoint=-1;
86          for (GblPointJna point : PointsAndTransList.get(1).getFirst()) {
87              ipoint+=1;
88              ppoints_1[ipoint]  = point.getPtr();
89          }
90          
91          double [] trafo_1 = PointsAndTransList.get(0).getSecond().getColumnPackedCopy();
92          double [] trafo_2 = PointsAndTransList.get(1).getSecond().getColumnPackedCopy();
93          
94          self = GblInterface.INSTANCE.GblTrajectoryCtorPtrComposed(ppoints_1, npoints_1, trafo_1,
95                                                                              ppoints_2, npoints_2, trafo_2);
96      }

141     public void getMeasResults(int aLabel, int numData[], List<Double> aResiduals,List<Double> aMeasErrors, List<Double> aResErrors, List<Double> aDownWeights) {
142         
143         double[] d_aResiduals  = new double[2];
144         double[] d_aMeasErrors = new double[2];
145         double[] d_aResErrors  = new double[2];
146         double[] d_aDownWeights = new double[2];
147         
148         GblInterface.INSTANCE.GblTrajectory_getMeasResults(self, aLabel, numData, d_aResiduals, d_aMeasErrors, d_aResErrors, d_aDownWeights);
149 
150         for (int i=0; i<2; i++) {
151             aResiduals.add(d_aResiduals[i]);
152             aMeasErrors.add(d_aMeasErrors[i]);
153             aResErrors.add(d_aResErrors[i]);
154             aDownWeights.add(d_aDownWeights[i]);
155         }
156         
157     }

159     //!! Only 5-localPar and 5x5 local Cov for the moment
160     public int getResults(int aSignedLabel, Vector localPar, SymMatrix localCov) {
161         
162         double[] d_localPar = new double[5];
163         double[] d_localCov = new double[25];
164         int[] nLocalPar = new int[1];
165         int[] sizeLocalCov = new int[1];
166         
167         GblInterface.INSTANCE.GblTrajectory_getResults(self,aSignedLabel,d_localPar,nLocalPar,d_localCov,sizeLocalCov);
168         
169         for (int i=0; i<5;  i++) {
170             localPar.set(i,d_localPar[i]);
171         }
172         
173         for (int i=0; i<5; i++) {
174             for (int j=0; j<5; j++) {
175                 localCov.set(i,j,d_localCov[i+5*j]);
176             }
177         }
178 
179         return 0;
180     }
