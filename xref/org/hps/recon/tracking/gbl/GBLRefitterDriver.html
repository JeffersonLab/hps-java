193     @Override
194     protected void detectorChanged(Detector detector) {
195 
196         bfield = Math.abs(TrackUtils.getBField(detector).magnitude());
197         _scattering.getMaterialManager().buildModel(detector);
198         _scattering.setBField(bfield); // only absolute of B is needed as it's used for momentum calculation only
199         
200         IDetectorElement detectorElement = detector.getDetectorElement();
201         sensors = detectorElement.findDescendants(SiSensor.class);
202 
203     }

205     @Override
206     protected void process(EventHeader event) {
207         if (!event.hasCollection(Track.class, inputCollectionName))
208             return;
209         
210         setupSensors(event);
211         List<Track> tracks = event.get(Track.class, inputCollectionName);
212         // System.out.println("GBLRefitterDriver::process number of tracks = "+tracks.size());
213         //RelationalTable hitToStrips = TrackUtils.getHitToStripsTable(event);
214         //RelationalTable hitToRotated = TrackUtils.getHitToRotatedTable(event);
215         RelationalTable hitToStrips = TrackUtils.getHitToStripsTable(event,helicalTrackHitRelationsCollectionName);
216         RelationalTable hitToRotated = TrackUtils.getHitToRotatedTable(event,rotatedHelicalTrackHitRelationsCollectionName);
217 
218         List<Track> refittedTracks = new ArrayList<Track>();
219         List<LCRelation> trackRelations = new ArrayList<LCRelation>();
220         
221         List<GBLKinkData> kinkDataCollection = new ArrayList<GBLKinkData>();
222         List<LCRelation> kinkDataRelations = new ArrayList<LCRelation>();
223 
224         //Hit on Track Residuals
225         List<TrackResidualsData> trackResidualsCollection =  new ArrayList<TrackResidualsData>();
226         List<LCRelation> trackResidualsRelations          = new ArrayList<LCRelation>();
227         
228         //Map<Track, Track> inputToRefitted = new HashMap<Track, Track>();
229         for (Track track : tracks) {
230             List<TrackerHit> temp = TrackUtils.getStripHits(track, hitToStrips, hitToRotated);
231             if (temp.size() == 0)
232                 //               System.out.println("GBLRefitterDriver::process  did not find any strip hits on this track???");
233                 continue;
234             
235             Pair<Pair<Track, GBLKinkData>, FittedGblTrajectory> newTrackTraj = MakeGblTracks.refitTrackWithTraj(TrackUtils.getHTF(track), temp, track.getTrackerHits(), gblRefitIterations, track.getType(), _scattering, bfield, storeTrackStates,includeNoHitScatters);
236             if (newTrackTraj == null) {
237                 getLogger().warning("Null returned from MakeGblTracks.refitTrackWithTraj - aborting refit");
238                 continue;
239             }
240             Pair<Track, GBLKinkData> newTrack = newTrackTraj.getFirst();
241             if (newTrack == null)
242                 continue;
243             Track gblTrk = newTrack.getFirst();
244 
245             if (enableStandardCuts && gblTrk.getChi2() > cuts.getMaxTrackChisq(gblTrk.getTrackerHits().size()))
246                 continue;
247             
248             //Include trackSelector to decide which tracks to use for alignment. 
249             //Propose to use tracks with at least 6 hits in the detector. 
250             
251             
252             if (enableAlignmentCuts) {
253                 
254                 //At least 1 GeV
255                 Hep3Vector momentum = new BasicHep3Vector(gblTrk.getTrackStates().get(0).getMomentum());
256                 
257                 
258                 if (momentum.magnitude() < 1)
259                     continue;
260                 
261                 ////At least 6 hits
262                 //if (gblTrk.getTrackerHits().size() < 6) 
263                 //continue;
264                 
265                 double[] trk_prms = track.getTrackParameters();
266                 double tanLambda = trk_prms[BaseTrack.TANLAMBDA];
267                 
268                 //Align with tracks without hole on track
269                 if ((tanLambda > 0 && track.getTrackerHits().size() < 5) || (tanLambda < 0 && track.getTrackerHits().size() < 5 )) 
270                     continue;
271 
272             }
273             
274             //System.out.printf("gblTrkNDF %d  gblTrkChi2 %f  getMaxTrackChisq5 %f getMaxTrackChisq6 %f \n", gblTrk.getNDF(), gblTrk.getChi2(), cuts.getMaxTrackChisq(5), cuts.getMaxTrackChisq(6));
275             if (enableStandardCuts && (gblTrk.getChi2() > cuts.getMaxTrackChisq(gblTrk.getTrackerHits().size())))
276                 continue;
277             
278             refittedTracks.add(gblTrk);
279             trackRelations.add(new BaseLCRelation(track, gblTrk));
280             //PF :: unused
281             //inputToRefitted.put(track, gblTrk);
282             kinkDataCollection.add(newTrack.getSecond());
283             kinkDataRelations.add(new BaseLCRelation(newTrack.getSecond(), gblTrk));
284             
285 
286             if (computeGBLResiduals) { 
287                 
288                 GblTrajectory gbl_fit_trajectory =  newTrackTraj.getSecond().get_traj();
289                 
290                 List<Double>  b_residuals = new ArrayList<Double>();
291                 List<Float>   b_sigmas    = new ArrayList<Float>();
292                 //List<Double>  u_residuals = new ArrayList<Double>();
293                 //List<Float>   u_sigmas    = new ArrayList<Float>();
294                 List<Integer> r_sensors   = new ArrayList<Integer>();
295                 
296                 int numData[] = new int[1];
297                 //System.out.printf("Getting the residuals. Points  on trajectory: %d \n",gbl_fit_trajectory.getNpointsOnTraj());
298                 //The fitted trajectory has a mapping between the MPID and the ilabel. Use that to get the MPID of the residual.
299                 Integer[] sensorsFromMapArray = newTrackTraj.getSecond().getSensorMap().keySet().toArray(new Integer[0]);
300                 //System.out.printf("Getting the residuals. Sensors on trajectory: %d \n",sensorsFromMapArray.length);
301                 
302                 //System.out.println("Check residuals of the original fit");
303                 //Looping on all the sensors on track -  to get the biased residuals.
304                 for (int i_s = 0; i_s < sensorsFromMapArray.length; i_s++) {       
305                     //Get the point label
306                     int ilabel = sensorsFromMapArray[i_s];
307                     //Get the millepede ID
308                     int mpid = newTrackTraj.getSecond().getSensorMap().get(ilabel);
309                     List<Double> aResiduals   = new ArrayList<Double>();   
310                     List<Double> aMeasErrors  = new ArrayList<Double>();
311                     List<Double> aResErrors   = new ArrayList<Double>();  
312                     List<Double> aDownWeights = new ArrayList<Double>();
313                     gbl_fit_trajectory.getMeasResults(ilabel,numData,aResiduals,aMeasErrors,aResErrors,aDownWeights); 
314                     if (numData[0]>1) { 
315                         System.out.printf("GBLRefitterDriver::WARNING::We have SCT sensors. Residuals dimensions should be <=1\n");
316                     }
317                     for (int i=0; i<numData[0];i++) {
318                         //System.out.printf("Example1::ilabel numDataIDX MPID aResidual aMeasError aResError\n");
319                         //System.out.printf("Example1::measResults %d %d %d %f %f %f \n",ilabel, i, mpid, aResiduals.get(i),aMeasErrors.get(i),aResErrors.get(i));
320                         
321                         r_sensors.add(mpid);
322                         b_residuals.add(aResiduals.get(i));
323                         b_sigmas.add(aResErrors.get(i).floatValue());
324                     }
325                     //Perform an unbiasing fit for each traj
326                     
327                     //System.out.println("Run the unbiased residuals!!!\n");
328                     //For each sensor create a trajectory 
329                     GblTrajectory gbl_fit_traj_u = new GblTrajectory(gbl_fit_trajectory.getSingleTrajPoints());
330                     double[] u_dVals = new double[2];
331                     int[] u_iVals    = new int[1];
332                     int[] u_numData  = new int[1]; 
333                     //Fit it once to have exactly the same starting point of gbl_fit_trajectory.
334                     gbl_fit_traj_u.fit(u_dVals,u_iVals,"");
335                     List<Double> u_aResiduals   = new ArrayList<Double>();   
336                     List<Double> u_aMeasErrors  = new ArrayList<Double>();
337                     List<Double> u_aResErrors   = new ArrayList<Double>();  
338                     List<Double> u_aDownWeights = new ArrayList<Double>();
339                     
340                     try {
341                         //Fit removing the measurement
342                         gbl_fit_traj_u.fit(u_dVals,u_iVals,"",ilabel);
343                         gbl_fit_traj_u.getMeasResults(ilabel,numData,u_aResiduals,u_aMeasErrors,u_aResErrors,u_aDownWeights); 
344                         for (int i=0; i<numData[0];i++) {
345                             //System.out.printf("Example1::ilabel numDataIDX MPID aResidual aMeasError aResError\n");
346                             //System.out.printf("Example1::UmeasResults %d %d %d %f %f %f \n",ilabel, i, mpid, u_aResiduals.get(i),u_aMeasErrors.get(i),u_aResErrors.get(i));
347                             
348                             r_sensors.add(mpid);
349                             b_residuals.add(u_aResiduals.get(i));
350                             b_sigmas.add(u_aResErrors.get(i).floatValue());
351                         }
352                     }
353                     catch (RuntimeException e){
354                     //  e.printStackTrack();
355                         r_sensors.add(-999);
356                         b_residuals.add(-9999.);
357                         b_sigmas.add((float)-9999.);
358                         //System.out.printf("Unbiasing fit fails! For label::%d\n",ilabel);
359                     }
360                     
361                 }//loop on sensors on track
362                 
363                 //Set top by default
364                 int trackerVolume = 0;
365                 //if tanLamda<0 set bottom
366                 //System.out.printf("Residuals size %d \n", r_sensors.size());
367                 if (gblTrk.getTrackStates().get(0).getTanLambda() < 0) trackerVolume = 1;
368                 TrackResidualsData resData  = new TrackResidualsData(trackerVolume,r_sensors,b_residuals,b_sigmas);
369                 trackResidualsCollection.add(resData);
370                 trackResidualsRelations.add(new BaseLCRelation(resData,gblTrk));
371             }//computeGBLResiduals
372         }//loop on tracks
373         
374         // Put the tracks back into the event and exit
375         int flag = 1 << LCIOConstants.TRBIT_HITS;
376         event.put(outputCollectionName, refittedTracks, Track.class, flag);
377         event.put(trackRelationCollectionName, trackRelations, LCRelation.class, 0);
378 
379         if (computeGBLResiduals) {
380             event.put(trackResidualsColName,    trackResidualsCollection,  TrackResidualsData.class, 0);
381             event.put(trackResidualsRelColName, trackResidualsRelations, LCRelation.class, 0);
382         }
383         
384         event.put(GBLKinkData.DATA_COLLECTION, kinkDataCollection, GBLKinkData.class, 0);
385         event.put(GBLKinkData.DATA_RELATION_COLLECTION, kinkDataRelations, LCRelation.class, 0);
386     }

389     private void setupSensors(EventHeader event) {
390         List<RawTrackerHit> rawTrackerHits = null;
391         if (event.hasCollection(RawTrackerHit.class, rawHitCollectionName))
392             rawTrackerHits = event.get(RawTrackerHit.class, rawHitCollectionName);
393         if (event.hasCollection(RawTrackerHit.class, "RawTrackerHitMaker_RawTrackerHits"))
394             rawTrackerHits = event.get(RawTrackerHit.class, "RawTrackerHitMaker_RawTrackerHits");
395 
396         EventHeader.LCMetaData meta = event.getMetaData(rawTrackerHits);
397         // Get the ID dictionary and field information.
398         IIdentifierDictionary dict = meta.getIDDecoder().getSubdetector().getDetectorElement().getIdentifierHelper().getIdentifierDictionary();
399         int fieldIdx = dict.getFieldIndex("side");
400         int sideIdx = dict.getFieldIndex("strip");
401         for (RawTrackerHit hit : rawTrackerHits) {
402             // if sensor already has a DetectorElement, skip it
403             if (hit.getDetectorElement() != null)
404                 continue;
405 
406             // The "side" and "strip" fields needs to be stripped from the ID for sensor lookup.
407             IExpandedIdentifier expId = dict.unpack(hit.getIdentifier());
408             expId.setValue(fieldIdx, 0);
409             expId.setValue(sideIdx, 0);
410             IIdentifier strippedId = dict.pack(expId);
411             // Find the sensor DetectorElement.
412             List<IDetectorElement> des = DetectorElementStore.getInstance().find(strippedId);
413             if (des == null || des.size() == 0)
414                 throw new RuntimeException("Failed to find any DetectorElements with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
415             else if (des.size() == 1)
416                 hit.setDetectorElement((SiSensor) des.get(0));
417             else
418                 // Use first sensor found, which should work unless there are sensors with duplicate IDs.
419                 for (IDetectorElement de : des)
420                     if (de instanceof SiSensor) {
421                         hit.setDetectorElement((SiSensor) de);
422                         break;
423                     }
424             // No sensor was found.
425             if (hit.getDetectorElement() == null)
426                 throw new RuntimeException("No sensor was found for hit with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
427         }
428     }
