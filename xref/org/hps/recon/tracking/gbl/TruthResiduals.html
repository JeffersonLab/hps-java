59      public void processSim(List<MCParticle> mcParticles, List<SimTrackerHit> simTrackerHits) {
60  
61          if (res_truthsimhit == null)
62              makePlots();
63  
64          // map layer, mcparticles and sim hits
65          Map<Integer, List<SimTrackerHit>> simHitsLayerMap = new HashMap<Integer, List<SimTrackerHit>>();
66          Map<MCParticle, List<SimTrackerHit>> mcPartSimHitsMap = new HashMap<MCParticle, List<SimTrackerHit>>();
67          for (SimTrackerHit sh : simTrackerHits) {
68              Hep3Vector shpos = CoordinateTransformations.transformVectorToTracking(sh.getPositionVec());
69              if (Math.abs(shpos.x()) < 50.0) {
70                  System.out.printf("%s: Weird hit at %s (%s) in layer %d for MC part %d org %s p %s\n", this.getClass()
71                          .getSimpleName(), sh.getPositionVec().toString(), shpos.toString(), sh
72                          .getIdentifierFieldValue("layer"), sh.getMCParticle().getPDGID(), sh.getMCParticle()
73                          .getOrigin().toString(), sh.getMCParticle().getMomentum().toString());
74                  System.exit(1);
75              }
76  
77              int layer = sh.getIdentifierFieldValue("layer");
78              if (!simHitsLayerMap.containsKey(layer)) {
79                  simHitsLayerMap.put(layer, new ArrayList<SimTrackerHit>());
80              }
81              simHitsLayerMap.get(layer).add(sh);
82  
83              MCParticle part = sh.getMCParticle();
84              if (!mcPartSimHitsMap.containsKey(part)) {
85                  mcPartSimHitsMap.put(part, new ArrayList<SimTrackerHit>());
86              }
87              mcPartSimHitsMap.get(part).add(sh);
88          }
89  
90          for (MCParticle mcp : mcPartSimHitsMap.keySet()) {
91              this.h_mcp_org.fill(mcp.getOriginX(), mcp.getOriginY());
92          }
93  
94          // Find the particle responsible for the hit in each layer and compute the residual
95  
96          for (int layer = 1; layer < 13; ++layer) {
97              // System.out.printf("layer %d: \n",layer);
98  
99              List<SimTrackerHit> simHitsLayer = simHitsLayerMap.get(layer);
100 
101             if (simHitsLayer != null) {
102 
103                 if (simHitsLayer.size() == 2)
104                     continue;
105 
106                 for (SimTrackerHit simHit : simHitsLayer) {
107 
108                     // Find the MC particle
109                     MCParticle mcp = simHit.getMCParticle();
110 
111                     if (mcp.getMomentum().magnitude() < 0.5)
112                         continue;
113 
114                     // Position in tracking coord
115                     Hep3Vector simHitPosTracking = CoordinateTransformations.transformVectorToTracking(simHit
116                             .getPositionVec());
117 
118                     if (_debug > 0) {
119                         System.out.printf("%s: simHit for layer %d at %s (%s) from MC part %d org %s p %s\n", this
120                                 .getClass().getSimpleName(), layer, simHit.getPositionVec().toString(),
121                                 simHitPosTracking.toString(), simHit.getMCParticle().getPDGID(), simHit.getMCParticle()
122                                         .getOrigin().toString(), simHit.getMCParticle().getMomentum().toString());
123 
124                         if (simHitPosTracking.x() < 50.)
125                             System.exit(1);
126 
127                     }
128 
129                     // Get track parameters from MC particle
130                     // HelicalTrackFit htfTruth = TrackUtils.getHTF(mcp, -1*this._B.z());
131                     HelicalTrackFit htfTruth = TrackUtils.getHTF(mcp, -1 * this._B.z());
132 
133                     Hep3Vector trkposExtraPolator = TrackUtils.extrapolateTrack(htfTruth, simHitPosTracking.x());
134                     // System.out.printf("trkposextrapol (det) %s\n",trkposExtraPolator.toString());
135 
136                     trkposExtraPolator = CoordinateTransformations.transformVectorToTracking(trkposExtraPolator);
137 
138                     // Calculate residuals
139                     Hep3Vector res = VecOp.sub(simHitPosTracking, trkposExtraPolator);
140 
141                     // System.out.printf("trkpos %s trkposextrapol %s res %s\n",trkpos.toString(),trkposExtraPolator.toString(),res.toString());
142 
143                     // Fill residuals
144                     this.res_truthsimhit.get(layer).get(0).fill(res.y());
145                     this.res_truthsimhit.get(layer).get(1).fill(res.z());
146                     if (simHit.getPositionVec().y() > 0) {
147                         if (simHit.getMCParticle().getPDGID() < 0) {
148                             this.res_truthsimhit_top_plus.get(layer).get(0).fill(res.y());
149                             this.res_truthsimhit_top_plus.get(layer).get(1).fill(res.z());
150                         } else {
151                             this.res_truthsimhit_top_minus.get(layer).get(0).fill(res.y());
152                             this.res_truthsimhit_top_minus.get(layer).get(1).fill(res.z());
153                         }
154                     } else {
155                         if (simHit.getMCParticle().getPDGID() < 0) {
156                             this.res_truthsimhit_bot_plus.get(layer).get(0).fill(res.y());
157                             this.res_truthsimhit_bot_plus.get(layer).get(1).fill(res.z());
158                         } else {
159                             this.res_truthsimhit_bot_minus.get(layer).get(0).fill(res.y());
160                             this.res_truthsimhit_bot_minus.get(layer).get(1).fill(res.z());
161                         }
162                     }
163 
164                     if (layer == 1 && res.y() > 0.1 && this.firstWeirdTrack) {
165                         double dx = 1.0;
166                         double xpos = mcp.getOriginZ();
167                         while (xpos < 100.) {
168                             xpos += dx;
169                             trkposExtraPolator = CoordinateTransformations.transformVectorToTracking(TrackUtils
170                                     .extrapolateTrack(htfTruth, xpos));
171                             double ypos = trkposExtraPolator.y();
172                             trkpos_y_vs_x.fill(xpos, ypos);
173                         }
174 
175                         int idummy = 0;
176                         while (idummy < 2) {
177                             trkpos_y_vs_x.fill(simHitPosTracking.x(), simHitPosTracking.y());
178                             idummy++;
179                             // System.out.printf("weird simhit res pos %s \n", simHitPosTracking.toString());
180                         }
181 
182                         this.firstWeirdTrack = false;
183 
184                     }
185 
186                 }
187             }
188         }
189 
190     }

192     public IHistogram getResidual(int layer, String coord) {
193         if (!this.res_truthsimhit.containsKey(layer))
194             throw new RuntimeException("Error the layer number is not valid");
195         if (coord != "x" && coord != "y")
196             throw new RuntimeException("Error the coord is not valid");
197         IHistogram1D h = this.res_truthsimhit.get(layer).get(coord == "x" ? 0 : 1);
198         return h;
199     }

201     private void makePlots() {
202 
203         res_truthsimhit = new HashMap<Integer, List<IHistogram1D>>();
204         res_truthsimhit_top_plus = new HashMap<Integer, List<IHistogram1D>>();
205         res_truthsimhit_bot_plus = new HashMap<Integer, List<IHistogram1D>>();
206         res_truthsimhit_top_minus = new HashMap<Integer, List<IHistogram1D>>();
207         res_truthsimhit_bot_minus = new HashMap<Integer, List<IHistogram1D>>();
208 
209         IHistogramFactory hf = aida.histogramFactory();
210         double min = -1.;
211         double max = 1.;
212 
213         List<IPlotter> plotter1 = new ArrayList<IPlotter>();
214         List<IPlotter> plotter2 = new ArrayList<IPlotter>();
215         List<IPlotter> plotter3 = new ArrayList<IPlotter>();
216         List<IPlotter> plotter4 = new ArrayList<IPlotter>();
217         List<IPlotter> plotter5 = new ArrayList<IPlotter>();
218 
219         for (int idir = 0; idir < 2; ++idir) {
220             String dir = idir == 0 ? "x" : "y";
221             IPlotter pl1 = af.createPlotterFactory().create(String.format("SimHit-Truth Track Residual %s", dir));
222             pl1.createRegions(3, 4);
223             IPlotter pl2 = af.createPlotterFactory().create(String.format("SimHit-Truth Track Residual %s", dir));
224             pl2.createRegions(3, 4);
225             IPlotter pl3 = af.createPlotterFactory().create(String.format("SimHit-Truth Track Residual %s", dir));
226             pl3.createRegions(3, 4);
227             IPlotter pl4 = af.createPlotterFactory().create(String.format("SimHit-Truth Track Residual %s", dir));
228             pl4.createRegions(3, 4);
229             IPlotter pl5 = af.createPlotterFactory().create(String.format("SimHit-Truth Track Residual %s", dir));
230             pl5.createRegions(3, 4);
231 
232             for (int layer = 1; layer < 13; ++layer) {
233 
234                 if (!res_truthsimhit.containsKey(layer)) {
235                     res_truthsimhit.put(layer, new ArrayList<IHistogram1D>());
236                     res_truthsimhit_top_plus.put(layer, new ArrayList<IHistogram1D>());
237                     res_truthsimhit_bot_plus.put(layer, new ArrayList<IHistogram1D>());
238                     res_truthsimhit_top_minus.put(layer, new ArrayList<IHistogram1D>());
239                     res_truthsimhit_bot_minus.put(layer, new ArrayList<IHistogram1D>());
240                 }
241 
242                 if (layer < 2) {
243                     max = 0.05;// 0.07;
244                     min = -0.05;// -0.07;
245                 } else if (layer < 3) {
246                     max = 0.3;// 0.07;
247                     min = -0.3;// -0.07;
248                 } else {
249                     max = 0.5 * layer;
250                     min = -1.0 * max;
251                 }
252 
253                 IHistogram1D h = hf.createHistogram1D(String.format("dres_truthsimhit_layer%d_%s", layer, dir), 50,
254                         min, max);
255                 h.setTitle(String.format("L%d SimHit-Truth Track Residual in %s", layer, dir));
256                 res_truthsimhit.get(layer).add(h);
257                 pl1.region(layer - 1).plot(h);
258 
259                 h = hf.createHistogram1D(String.format("res_truthsimhit_top_plus_layer%d_%s", layer, dir), 50, min, max);
260                 h.setTitle(String.format("L%d SimHit-Truth Track (top,q=+1) Residual in %s", layer, dir));
261                 res_truthsimhit_top_plus.get(layer).add(h);
262                 pl2.region(layer - 1).plot(h);
263 
264                 h = hf.createHistogram1D(String.format("res_truthsimhit_top_minus_layer%d_%s", layer, dir), 50, min,
265                         max);
266                 h.setTitle(String.format("L%d SimHit-Truth Track (top,q=-1) Residual in %s", layer, dir));
267                 res_truthsimhit_top_minus.get(layer).add(h);
268                 pl3.region(layer - 1).plot(h);
269 
270                 h = hf.createHistogram1D(String.format("res_truthsimhit_bot_minus_layer%d_%s", layer, dir), 50, min,
271                         max);
272                 h.setTitle(String.format("L%d SimHit-Truth Track (bot,q=-1) Residual in %s", layer, dir));
273                 res_truthsimhit_bot_minus.get(layer).add(h);
274                 pl4.region(layer - 1).plot(h);
275 
276                 h = hf.createHistogram1D(String.format("res_truthsimhit_bot_plus_layer%d_%s", layer, dir), 50, min, max);
277                 h.setTitle(String.format("L%d SimHit-Truth Track (bot,q=+1) Residual in %s", layer, dir));
278                 res_truthsimhit_bot_plus.get(layer).add(h);
279                 pl5.region(layer - 1).plot(h);
280 
281             }
282             plotter1.add(pl1);
283             plotter2.add(pl2);
284             plotter3.add(pl3);
285             plotter4.add(pl4);
286             plotter5.add(pl5);
287 
288             if (!this._hideFrame) {
289                 pl1.show();
290                 pl2.show();
291                 pl3.show();
292                 pl4.show();
293                 pl5.show();
294             } else {
295                 pl1.hide();
296                 pl2.hide();
297                 pl3.hide();
298                 pl4.hide();
299                 pl5.hide();
300             }
301 
302         }
303 
304         this.h_mcp_org = hf.createHistogram2D("MC particle origin", 50, -0.2, 0.2, 50, -0.2, 0.2);
305         IPlotter pl_org = af.createPlotterFactory().create("MC particle origin");
306         pl_org.createRegions(1, 1);
307         pl_org.region(0).plot(h_mcp_org);
308         pl_org.region(0).style().setParameter("hist2DStyle", "colorMap");
309         pl_org.region(0).style().dataStyle().fillStyle().setParameter("colorMapScheme", "rainbow");
310         if (this._hideFrame)
311             pl_org.hide();
312         else
313             pl_org.show();
314 
315         trkpos_y_vs_x = hf.createHistogram2D("Track pos y vs x", 300, -150, 150, 100, -4, 4);
316         IPlotter pl_pos_y_vs_x = af.createPlotterFactory().create("Track pos y vs x");
317         pl_pos_y_vs_x.createRegions(1, 1);
318         pl_pos_y_vs_x.region(0).plot(trkpos_y_vs_x);
319         pl_pos_y_vs_x.region(0).style().setParameter("hist2DStyle", "colorMap");
320         pl_pos_y_vs_x.region(0).style().dataStyle().fillStyle().setParameter("colorMapScheme", "rainbow");
321         if (this._hideFrame)
322             pl_pos_y_vs_x.hide();
323         else
324             pl_pos_y_vs_x.show();
325 
326     }
