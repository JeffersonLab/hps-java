3   public class SymMatrix extends Matrix implements java.io.Serializable {
4   
5       public SymMatrix(int n) {
6           super(n, n);
7       }
8   
9       public SymMatrix(SymMatrix smat) {
10          super(smat.getRowDimension(), smat.getColumnDimension());
11          for (int i = 0; i < m; i++) {
12              for (int j = 0; j < n; j++) {
13                  A[i][j] = smat.get(i, j);
14              }
15          }
16      }
17  
18      public SymMatrix(Matrix mat) {
19          super(mat.getRowDimension(), mat.getColumnDimension());
20          for (int i = 0; i < m; i++) {
21              for (int j = 0; j < n; j++) {
22                  A[i][j] = mat.get(i, j);
23              }
24          }
25      }
26  
27      @Override
28      public void set(int i, int j, double s) {
29          super.set(i, j, s);
30          super.set(j, i, s);
31      }
32  
33      public void setToIdentity() {
34          for (int i = 0; i < m; i++) {
35              for (int j = 0; j < n; j++) {
36                  A[i][j] = (i == j ? 1.0 : 0.0);
37              }
38          }
39      }
40  
41      // Calculate B * (this) * B^T , final matrix will be (nrowsb x nrowsb)
42      // TODO see if this needs to be made more efficient
43      public SymMatrix Similarity(Matrix b) {
44          SymMatrix X = new SymMatrix(b.times(this.times(b.transpose())));
45          return X;
46      }
