14  /**
15   * Extension to lcsim MaterialManager to allow more flexibility in track reconstruction
16   */
17  public class MaterialManager extends org.lcsim.recon.tracking.seedtracker.MaterialManager {
18  
19      /**
20       * Get the path groups for SiTrackerEndcap2, which has modules placed directly in the tracking volume.
21       */
22      static private class HPSTracker2VolumeGrouper implements SubdetectorVolumeGrouper {
23  
24          @Override
25          public List<List<String>> getPathGroups(final Subdetector subdet, final IPhysicalVolume topVol) {
26              // System.out.println(this.getClass().getSimpleName() + ".getPathGroups()");
27              final List<List<String>> pathGroups = new ArrayList<List<String>>();
28              // Layer loop.
29              for (final IDetectorElement layer : subdet.getDetectorElement().getChildren()) {
30                  final List<String> modulePaths = new ArrayList<String>();
31  
32                  // Module loop.
33                  for (final IDetectorElement module : layer.getChildren()) {
34                      final String path = "";
35                      PhysicalVolumeNavigator.getLeafPaths(modulePaths, module.getGeometry().getPhysicalVolume(), path);
36                  }
37  
38                  // Add module paths to this layer.
39                  pathGroups.add(modulePaths);
40              }
41              return pathGroups;
42          }
43      }
44  
45      private final static List<MaterialXPlane> _emptyMaterialXPlaneList = new ArrayList<MaterialXPlane>();
46  
47      protected boolean _includeMS = true;
48  
49      public MaterialManager() {
50          super();
51  
52          // Add volume groupers for HPS tracker types.
53          final SubdetectorVolumeGrouper endcap2Grouper = new SiTrackerEndap2VolumeGrouper();
54          subdetGroups.put(HPSTracker.class, endcap2Grouper);
55          subdetGroups.put(HPSTracker2.class, new HPSTracker2VolumeGrouper());
56      }
57  
58      public MaterialManager(final boolean includeMS) {
59          super();
60          this._includeMS = includeMS;
61      }
62  
63      @Override
64      public List<MaterialXPlane> getMaterialXPlanes() {
65          return this._includeMS ? super.getMaterialXPlanes() : _emptyMaterialXPlaneList;
66      }
67  
68      @Override
69      public void setDebug(final boolean debug) {
70          super.setDebug(debug);
71      }
72  
73  }
