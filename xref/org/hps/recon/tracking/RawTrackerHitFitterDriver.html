129     public void setFitAlgorithm(String fitAlgorithm) {
130         if (fitAlgorithm.equals("Analytic"))
131             fitter = new ShaperAnalyticFitAlgorithm();
132         else if (fitAlgorithm.equals("Linear"))
133             fitter = new ShaperLinearFitAlgorithm(1);
134         else if (fitAlgorithm.equals("PileupAlways"))
135             fitter = new ShaperPileupFitAlgorithm(1.0);
136         else if (fitAlgorithm.equals("Pileup"))
137             fitter = new ShaperPileupFitAlgorithm();
138         else
139             throw new RuntimeException("Unrecognized fitAlgorithm: " + fitAlgorithm);
140     }

175     protected void detectorChanged(Detector detector) {
176         timingConstants = DatabaseConditionsManager.getInstance().getCachedConditions(SvtTimingConstants.SvtTimingConstantsCollection.class, "svt_timing_constants").getCachedData().get(0);

189     @Override
190     public void process(EventHeader event) {
191         if (!event.hasCollection(RawTrackerHit.class, rawHitCollectionName))
192             // System.out.println(rawHitCollectionName + " does not exist; skipping event");
193             return;
194         
195         if(isFirstEvent && correctPerSensorPerPhase){
196             //this loads the sensorPhaseCalibConstants or returns false if run not in range or calib file not found/broken
197             //...there should be a better way to get run number...right?  
198             correctPerSensorPerPhase=LoadPerSensorPerPhaseConstants(event);
199         }
200         isFirstEvent=false;
201         
202         jitter = -666;
203         if (subtractRFTime)
204             if (event.hasCollection(TriggerTime.class, "TriggerTime")) {
205                 if (debug)
206                     System.out.println("Getting TriggerTime Object");
207                 List<TriggerTime> jitterList = event.get(TriggerTime.class, "TriggerTime");
208                 if (debug)
209                     System.out.println("TriggerTime List Size = " + jitterList.size());
210                 TriggerTime jitterObject = jitterList.get(0);
211                 jitter = jitterObject.getDoubleVal();
212                 if (debug)
213                     System.out.println("RF time jitter " + jitter);
214 
215             } else {
216                 System.out.println("Requested RF Time correction but TriggerTime Collection doesn't exist!!!");
217                 return;
218             }
219 
220         List<RawTrackerHit> rawHits = event.get(RawTrackerHit.class, rawHitCollectionName);
221         if (rawHits == null)
222             throw new RuntimeException("Event is missing SVT hits collection!");
223         List<FittedRawTrackerHit> hits = new ArrayList<FittedRawTrackerHit>();
224         List<ShapeFitParameters> fits = new ArrayList<ShapeFitParameters>();
225 
226         // Make a fitted hit from this cluster
227         for (RawTrackerHit hit : rawHits) {
228             int strip = hit.getIdentifierFieldValue("strip");
229             HpsSiSensor sensor = (HpsSiSensor) hit.getDetectorElement();
230             //===> ChannelConstants constants = HPSSVTCalibrationConstants.getChannelConstants((SiSensor) hit.getDetectorElement(), strip);
231             for (ShapeFitParameters fit : fitter.fitShape(hit, shape)) {
232 
233                 if (correctTimeOffset) {
234                     if (debug)
235                         System.out.println("subtracting svt time offset " + timingConstants.getOffsetTime());
236                     fit.setT0(fit.getT0() - timingConstants.getOffsetTime());
237                 }
238                 if (subtractTriggerTime) {
239                     double tt = (((event.getTimeStamp() - 4 * timingConstants.getOffsetPhase()) % 24) - trigTimeOffset);
240                     if (!syncGood) tt = tt - 8;
241                     if (!syncGood && (((event.getTimeStamp() - 4 * timingConstants.getOffsetPhase()) % 24)/8 < 1)) {
242                         tt = tt + 24;
243                     }
244                     if (isMC && (((event.getTimeStamp() - 4 * timingConstants.getOffsetPhase()) % 24)/8 == 1)) {
245                         tt = tt + 24;
246                     }
247                     if (debug) {
248                         System.out.println("event time stamp " + event.getTimeStamp());
249                         System.out.println("trigger offset time " + timingConstants.getOffsetPhase());
250                         System.out.println("subtracting trigger time from event " + tt);
251                         System.out.println("T0 " + fit.getT0());
252                     }
253                     fit.setT0(fit.getT0() - tt);
254                 }
255                 if (subtractRFTime && jitter != -666) {
256                     if (debug)
257                         System.out.println("subtracting RF time jitter " + jitter);
258                     fit.setT0(fit.getT0() - jitter + trigTimeScale);
259                 }
260                 if (correctChanT0) {
261                     if (debug)
262                         System.out.println("subtracting channel t0 " + sensor.getShapeFitParameters(strip)[HpsSiSensor.T0_INDEX]);
263                     fit.setT0(fit.getT0() - sensor.getShapeFitParameters(strip)[HpsSiSensor.T0_INDEX]);
264                 }
265                 if (correctT0Shift) {
266                     if (debug)
267                         System.out.println("subtracting sensor shift " + sensor.getT0Shift());
268                     //===> fit.setT0(fit.getT0() - constants.getT0Shift());
269                     fit.setT0(fit.getT0() - sensor.getT0Shift());
270                 }
271 
272                 ////////////   mg 5/17/2023 ....  this is for reading sensor/phase dependent shifts using database...
273                 ///////////                       I should make new branch for reading from db because we may want to
274                 ///////////                       do that in future but for now remove
275                 //                if (correctPhaseDepT0Shift){
276                 //    Double phaseShifts[]=sensor.getT0PhaseShifts(); 
277                 //    int phase=(int)((event.getTimeStamp()%24)/4);
278                 //    fit.setT0(fit.getT0()-phaseShifts[phase]);
279                 //}
280                 // correct time per sensor and event phase (using the constants read in by resource file)
281                 if(correctPerSensorPerPhase){
282                     String sensorName=sensor.getName();
283                     String simpleName=tkHitUtils.getSimpleNameFromSensorName(sensorName);
284                     //                    System.out.println(sensorName+" --> "+simpleName);
285                     Long evtPhaseL=(event.getTimeStamp() % 24)/4;
286                     Integer evtPhase=evtPhaseL.intValue();
287                     Pair<String,Integer> evtPair=new Pair(simpleName,evtPhase);
288                     Double calConstant=sensorPhaseCalibConstants.get(evtPair);
289                     //                    System.out.println("shifting t0 by "+calConstant); 
290                     fit.setT0(fit.getT0()-calConstant);
291                 }
292 
293                 if (subtractTOF) {
294                     double tof = hit.getDetectorElement().getGeometry().getPosition().magnitude() / (Const.SPEED_OF_LIGHT * Const.nanosecond);
295                     fit.setT0(fit.getT0() - tof);
296                 }
297                 if (useTimestamps) {
298                     double t0Svt = ReadoutTimestamp.getTimestamp(ReadoutTimestamp.SYSTEM_TRACKER, event);
299                     double t0Trig = ReadoutTimestamp.getTimestamp(ReadoutTimestamp.SYSTEM_TRIGGERBITS, event);
300                     // double corMod = (t0Svt - t0Trig) + 200.0;///where does 200.0 come from?  for 2016 MC, looks like should be 240
301                     double corMod = (t0Svt - t0Trig) + tsCorrectionScale;
302                     fit.setT0(fit.getT0() + corMod);
303                 }
304                 if (useTruthTime) {
305                     double t0Svt = ReadoutTimestamp.getTimestamp(ReadoutTimestamp.SYSTEM_TRACKER, event);
306                     double absoluteHitTime = fit.getT0() + t0Svt;
307                     double relativeHitTime = ((absoluteHitTime + 250.0) % 500.0) - 250.0;
308 
309                     fit.setT0(relativeHitTime);
310                 }
311                 if (debug)
312                     System.out.println(fit);
313 
314                 fits.add(fit);
315                 FittedRawTrackerHit hth = new FittedRawTrackerHit(hit, fit);
316                 hits.add(hth);
317                 if (strip == HPSSVTConstants.TOTAL_STRIPS_PER_SENSOR) // drop unbonded channel
318                     continue;
319                 hit.getDetectorElement().getReadout().addHit(hth);
320             }
321         }
322         event.put(fitCollectionName, fits, ShapeFitParameters.class, genericObjectFlags);
323         event.put(fittedHitCollectionName, hits, FittedRawTrackerHit.class, relationFlags);
324     }

326     private boolean LoadPerSensorPerPhaseConstants(EventHeader event){
327         
328         // get run number from first event
329         int runNumber= event.getRunNumber();
330         String infilePreResDir = "/org/hps/recon/tracking/timingCorrections/"; 
331         String infilePre = "run";
332         String infilePost = "_calib_constants_final.txt";
333         String runString="foobar"; 
334 
335         if(runNumber>10000 && runNumber<10464){
336             runString="10377";
337         }else if(runNumber>=10464 && runNumber<10660){
338             runString="10564"; 
339         }else if(runNumber>=10660 && runNumber<11000){
340             runString="10666"; 
341         }else if(runNumber>14000 && runNumber<14566){        //2021 constants
342             runString="14495";
343         }else if(runNumber>=14566 &&runNumber<14626){
344             runString="14569"; 
345         }else if(runNumber>=14626 && runNumber<14680){
346             runString="14654";
347         }else if(runNumber>=14680 && runNumber<15000){
348             runString="14720";
349         } else {
350             getLogger().config("Run is not in range defined by correctPerSensorPerPhase ... corrections not performed");
351             return false;
352         }
353 
354         //ok, overwrite the run string for some individual funny runs
355         if(runNumber==10687 || runNumber==10711||runNumber==10713||runNumber==10714){
356             runString=Integer.toString(runNumber);
357         }
358 
359 
360 
361         String infile=infilePreResDir+infilePre+runString+infilePost; 
362         InputStream inRatios = this.getClass().getResourceAsStream(infile);
363         System.out.println("reading in per-sensor per-phase calibs from "+infile);
364         BufferedReader reader = new BufferedReader(new InputStreamReader(inRatios));
365         String line;
366         String delims = "[ ]+";// this will split strings between one or more spaces
367         try {
368             while ((line = reader.readLine()) != null) {
369                 String[] tokens = line.split(delims);
370                 System.out.println("sensor_phase = " + tokens[0] + "; constant = " + tokens[1]);
371                 String[] sensor_phase = tokens[0].split("_phase"); 
372                 //                System.out.println("Making Pair::sensor name = "+sensor_phase[0]+", phase = "+Integer.parseInt(sensor_phase[1]));
373                 Pair<String, Integer> senPhPair=new Pair(sensor_phase[0],Integer.parseInt(sensor_phase[1])); 
374                 Double constant=Double.parseDouble(tokens[1]);
375                 sensorPhaseCalibConstants.put(senPhPair, constant); 
376             }
377         } catch (IOException ex) {
378             getLogger().config("died while reading "+infile);
379             return false;
380         }        
381         return true;
382     }
