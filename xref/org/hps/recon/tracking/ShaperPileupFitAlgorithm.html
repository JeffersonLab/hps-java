3   import java.util.Collection;
4   import org.lcsim.event.RawTrackerHit;
5   
6   public class ShaperPileupFitAlgorithm implements ShaperFitAlgorithm {
7   
8       ShaperLinearFitAlgorithm onePulseFitter = new ShaperLinearFitAlgorithm(1);
9       ShaperLinearFitAlgorithm twoPulseFitter = new ShaperLinearFitAlgorithm(2);
10      private String fitTimeMinimizer = "Simplex";
11      private boolean debug = false;
12      private double refitThreshold = 0.5;
13      private int totalFits = 0;
14      private int refitAttempts = 0;
15      private int refitsAccepted = 0;
16  
17      public ShaperPileupFitAlgorithm() {
18      }
19  
20      public ShaperPileupFitAlgorithm(double threshold) {
21          refitThreshold = threshold;
22      }
23  
24      @Override
25      public void setFitTimeMinimizer(String fitTimeMinimizer) {
26          this.onePulseFitter.setFitTimeMinimizer(fitTimeMinimizer);
27          this.twoPulseFitter.setFitTimeMinimizer(fitTimeMinimizer);
28      }
29  
30      //===> public Collection<ShapeFitParameters> fitShape(RawTrackerHit rth, HPSSVTCalibrationConstants.ChannelConstants constants) {
31      public Collection<ShapeFitParameters> fitShape(RawTrackerHit rth, PulseShape shape) {
32          //===> Collection<ShapeFitParameters> fittedPulses = onePulseFitter.fitShape(rth, constants);
33          Collection<ShapeFitParameters> fittedPulses = onePulseFitter.fitShape(rth, shape);
34          double singlePulseChiProb = fittedPulses.iterator().next().getChiProb();
35          totalFits++;
36          if (singlePulseChiProb < refitThreshold) {
37              refitAttempts++;
38              //===> Collection<ShapeFitParameters> doublePulse = twoPulseFitter.fitShape(rth, constants);
39              Collection<ShapeFitParameters> doublePulse = twoPulseFitter.fitShape(rth, shape);
40              double doublePulseChiProb = doublePulse.iterator().next().getChiProb();
41              if (doublePulseChiProb > singlePulseChiProb) {
42                  refitsAccepted++;
43                  fittedPulses = doublePulse;
44              }
45          }
46          if (debug && totalFits % 10000 == 0) {
47              System.out.format("%d fits, %d refit attempts, %d refits accepted\n", totalFits, refitAttempts, refitsAccepted);
48          }
49          return fittedPulses;
50      }
51  
52      public void setDebug(boolean debug) {
53          this.debug = debug;
54          onePulseFitter.setDebug(debug);
55          twoPulseFitter.setDebug(debug);
56      }
57  
58  }
