1   package org.hps.recon.tracking.kalman;
2   
3   import java.util.ArrayList;
4   import java.util.Iterator;
5   import java.util.logging.Level;
6   import java.util.logging.Logger;
7   
8   import org.ejml.data.DMatrixRMaj;
9   import org.ejml.dense.row.CommonOps_DDRM;
10  
11  /**
12   * Driver program for executing a Kalman refit of existing tracks.  This version starts at layer N, filters to layer 0,
13   * then starts over using the fit result to start filtering from layer 0 outward. Then it smooths
14   * back to layer 0. The code assumes that the SiModules hold all the hits to be fit and only those hits.
15   * No pattern recognition is done; no hits are dropped.
16   */
17  class KalmanTrackFit2 {
18  
19      ArrayList<MeasurementSite> sites;
20      int initialSite;
21      int finalSite;
22      double chi2f, chi2s; // Filtered and smoothed chi squared values (just summed over the N measurement
23                           // sites)
24      boolean success;
25      KalTrack tkr;
26      private Logger logger;
27  
28      KalmanTrackFit2(int evtNumb, ArrayList<SiModule> data, // List of Si modules with data points to be included in the fit
29              ArrayList<Integer> hits, // Which hit to use in each SiModule. Can be null if each module has only 1 hit.
30              int start, // Starting point in the list
31              int nIterations, // Number of fit iterations requested
32              Vec pivot, // Pivot point for the starting "guess" helix
33              Vec helixParams, // 5 helix parameters for the starting "guess" helix
34              DMatrixRMaj C, // Full covariance matrix for the starting "guess" helix
35              KalmanParams kPar,
36              org.lcsim.geometry.FieldMap fM) {
37          
38          if (hits == null) {
39              hits = new ArrayList<Integer>(data.size());
40              for (int i=0; i<data.size(); ++i) {
41                  hits.add(0);
42              }
43          }
44          logger = Logger.getLogger(KalmanTrackFit2.class.getName());
45          boolean verbose = logger.getLevel() == Level.FINER;
46          success = true;
47          tkr = null;
48  
49          // Create an state vector from the input seed to initialize the Kalman filter
50          Vec Bfield = KalmanInterface.getField(pivot, fM);
51          double B = Bfield.mag();
52          Vec t = Bfield.unitVec(B);
53          StateVector sI = new StateVector(-1, helixParams, C, new Vec(0., 0., 0.), B, t, pivot);
54  
55          if (verbose) {
56              System.out.format("KalmanTrackFit2: begin Kalman fit, start=%d, number iterations=%d\n", start, nIterations);
57              pivot.print("KalmanTrackFit2 pivot point");
58              sI.print("initial state for KalmanTrackFit");
59          }
60  
61          double mxResid = 9999.;
62  
63          sites = new ArrayList<MeasurementSite>();
64          initialSite = 0;
65          finalSite = 0;
66  
67          int prevSite;
68          int thisSite;
69          MeasurementSite startSite = null;
70          if (start > 1) {
71              MeasurementSite newSite = null;
72              chi2f = 0.;
73              prevSite = -1;
74              thisSite = -1;
75              for (int idx = start; idx > -1; idx--) {
76                  finalSite = idx;
77                  SiModule m = data.get(idx);
78                  if (m.hits.size()<=0) { // && m.detector==1) {
79                      continue;
80                  }
81                  thisSite++;
82                  newSite = new MeasurementSite(idx, m, kPar);
83                  int hitNumber = hits.get(idx);
84                  if (m.hits.size() == 0) hitNumber = -1;
85                  if (idx == start) {
86                      if (newSite.makePrediction(sI, hitNumber, false, false) < 0) {
87                          logger.warning(String.format("Failed to make initial prediction at site %d, idx=%d.  Abort", thisSite, idx));
88                          success = false;
89                          break;
90                      }
91                  } else {
92                      if (newSite.makePrediction(sites.get(prevSite).aF, sites.get(prevSite).m, hitNumber, false, false) < 0) {
93                          logger.warning(String.format("Failed to make prediction at site %d, idx=%d.  Abort", thisSite, idx));
94                          success = false;
95                          break;
96                      }
97                  }
98  
99                  if (!newSite.filter()) {
100                     logger.warning(String.format("Failed to filter at site %d, idx=%d.  Ignore remaining sites", thisSite, idx));
101                     success = false;
102                     break;
103                 }
104                 ;
105 
106                 if (m.Layer >= 0 && hitNumber >= 0) chi2f += newSite.chi2inc;
107 
108                 sites.add(newSite);
109 
110                 prevSite = thisSite;
111             }
112             if (!success) { return; }
113             if (verbose) {
114                 for (MeasurementSite site : sites) site.print("initial filtering");
115                 System.out.format("KalmanTrackFit2: Fit chi^2 after initial filtering = %12.4e;  Final site = %d\n", chi2f, finalSite);
116                 newSite.aF.helix.a.print("filtered helix parameters at innermost site.");
117                 System.out.format("    The innermost site is at layer %d\n", newSite.m.Layer);
118                 int cnt = 0;
119                 for (MeasurementSite site : sites) {
120                     SiModule m = site.m;
121                     StateVector aF = site.aF;
122                     double phiF = aF.helix.planeIntersect(m.p);
123                     if (Double.isNaN(phiF)) phiF = 0.;
124                     double vPred = site.h(aF, site.m, phiF);
125                     System.out.format("   %d Lyr %d stereo=%b Hit %d chi2inc=%10.6f, vPred=%10.6f; Hits: ", cnt, m.Layer, m.isStereo,
126                             site.hitID, site.chi2inc, vPred);
127                     for (Measurement hit : m.hits) {
128                         double resid = hit.v - vPred;
129                         System.out.format(" v=%10.6f r=%10.8f #tks=%d,", hit.v, resid, hit.tracks.size());
130                     }
131                     System.out.format("\n");
132                     cnt++;
133                 }
134             }
135             startSite = newSite;
136         }
137 
138         int nHits = 0;
139         // Restart the fit at the first layer and iterate the fit if requested
140         for (int iteration = 0; iteration < nIterations; iteration++) {
141             StateVector sH = null;
142             if (startSite == null) {
143                 sH = sI;
144             } else {
145                 if (startSite.smoothed) {
146                     sH = startSite.aS.copy();
147                 } else {
148                     sH = startSite.aF.copy();
149                 }
150                 CommonOps_DDRM.scale(100., sH.helix.C); // Blow up the initial covariance matrix to avoid double counting measurements
151             }
152             if (verbose) {
153                 System.out.format("KalmanTrackFit: starting filtering for iteration %d\n", iteration);
154                 // sH.print("starting state vector for iteration");
155             }
156             sites.clear();
157 
158             Iterator<SiModule> itr = data.iterator();
159             chi2f = 0.;
160             boolean success = true;
161             MeasurementSite previousSite = null;
162             thisSite = -1;
163             while (itr.hasNext()) {
164                 SiModule m = itr.next();
165                 if (m.hits.size()<=0) { // && m.detector==1) {
166                     continue;
167                 }
168                 thisSite++;
169                 int hitNumber = hits.get(data.indexOf(m));
170                 if (m.hits.size() == 0) hitNumber = -1;
171                 MeasurementSite newSite = new MeasurementSite(thisSite, m, kPar);
172                 if (thisSite == 0) {
173                     if (newSite.makePrediction(sH, hitNumber, false, false) < 0) {
174                         System.out.format("KalmanTrackFit2: Failed to make initial prediction at site %d.  Abort\n", thisSite);
175                         success = false;
176                         break;
177                     }
178                 } else {
179                     if (newSite.makePrediction(previousSite.aF, previousSite.m, hitNumber, false, false) < 0) {
180                         System.out.format("KalmanTrackFit2: Failed to make prediction at site %d.  Abort\n", thisSite);
181                         success = false;
182                         break;
183                     }
184                 }
185 
186                 if (!newSite.filter()) {
187                     System.out.format("KalmanTrackFit2 153: Failed to filter at site %d layer %d.  Ignore remaining sites\n", thisSite, newSite.m.Layer);
188                     success = false;
189                     break;
190                 }
191                 if (verbose && previousSite != null) {
192                     System.out.format("KalmanTrackFit2: filter at layer %d detector %d\n", m.Layer, m.detector);
193                     previousSite.aF.helix.a.print("old filtered helix");
194                     newSite.aP.helix.a.print("new predicted helix");
195                     newSite.aF.helix.a.print("new filtered helix");
196                 }
197                 // if (verbose) {
198                 // newSite.print(String.format("Iteration %d: filtering", iteration));
199                 // }
200                 if (m.Layer >= 0 && hitNumber >=0) chi2f += newSite.chi2inc;
201 
202                 sites.add(newSite);
203 
204                 previousSite = newSite;
205             }
206             if (!success) { return; }
207             if (verbose) {
208                 System.out.format("KalmanTrackFit2: Fit chi^2 after first full filtering = %12.4e\n", chi2f);
209                 previousSite.aF.helix.a.print("filtered helix parameters at last site filtered");
210                 int cnt = 0;
211                 for (MeasurementSite site : sites) {
212                     SiModule m = site.m;
213                     StateVector aF = site.aF;
214                     double phiF = aF.helix.planeIntersect(m.p);
215                     if (Double.isNaN(phiF)) phiF = 0.;
216                     double vPred = site.h(aF, site.m, phiF);
217                     double phiP = site.aP.helix.planeIntersect(m.p);
218                     if (Double.isNaN(phiP)) phiP = 0.;
219                     double vPredP = site.h(site.aP, site.m, phiP);
220                     System.out.format("   %d Lyr %d stereo=%b Hit %d chi2inc=%10.6f, vPred=%10.6f; vPredP=%10.6f Hits: ", cnt, m.Layer,
221                             m.isStereo, site.hitID, site.chi2inc, vPred, vPredP);
222                     for (Measurement hit : m.hits) {
223                         double resid = hit.v - vPred;
224                         System.out.format(" v=%10.6f r=%10.8f #tks=%d,", hit.v, resid, hit.tracks.size());
225                     }
226                     System.out.format("\n");
227                     if (m.Layer == 11) site.print("for comparison");
228                     cnt++;
229                 }
230             }
231             chi2s = 0.;
232             MeasurementSite nextSite = null;
233             MeasurementSite currentSite = null;
234             for (int idx = sites.size() - 1; idx >= 0; idx--) {
235                 currentSite = sites.get(idx);
236                 if (currentSite.m.Layer < 0 && currentSite.m.hits.size() == 0) {
237                     currentSite.aS = currentSite.aF;
238                     continue;
239                 }
240 
241                 if (nextSite == null) {
242                     currentSite.aS = currentSite.aF.copy();
243                     currentSite.smoothed = true;
244                 } else {
245                     currentSite.smooth(nextSite);
246                 }
247                 if (currentSite.hitID >= 0) {
248                     chi2s += currentSite.chi2inc;
249                     if (iteration == nIterations - 1) nHits++;
250                 }
251 
252                 if (verbose) {
253                     currentSite.print(String.format("Iteration %d smoothing", iteration));
254                 }
255                 nextSite = currentSite;
256             }
257 
258             if (verbose) {
259                 System.out.format("KalmanTrackFit2: Iteration %d, Fit chi^2 after smoothing = %12.4e\n", iteration, chi2s);
260                 currentSite.aS.helix.a.print("smoothed helix parameters at the innermost site");
261                 int cnt = 0;
262                 double chi2prime = 0.;
263                 for (MeasurementSite site : sites) {
264                     double phiS = site.aS.helix.planeIntersect(site.m.p);
265                     if (Double.isNaN(phiS)) phiS = 0.;
266                     double vpred = site.h(site.aS, site.m, phiS);
267                     System.out.format("   %d Lyr %d stereo=%b Hit %d chi2inc=%10.6f r1=%10.8f err=%10.8f vPred=%10.6f; Hits: ", cnt,
268                             site.m.Layer, site.m.isStereo, site.hitID, site.chi2inc, site.aS.r, Math.sqrt(site.aS.R), vpred);
269                     for (Measurement hit : site.m.hits) {
270                         double resid = hit.v - vpred; // verified to be the same as site.aS.R
271                         System.out.format(" v=%10.6f sigma=%10.8f #tks=%d,", hit.v, hit.sigma, hit.tracks.size());
272                         chi2prime += Math.pow(resid / hit.sigma, 2);
273                     }
274                     System.out.format("\n");
275                     cnt++;
276                 }
277                 System.out.format("Kalman fit chi**2 calculated directly from residuals and hit errors: %10.4f\n", chi2prime);
278             }
279             startSite = currentSite;
280         }
281 
282         finalSite = sites.size() - 1;
283 
284         int nLyrs = 14;
285         int [] lyrPtr = new int[nLyrs+1];
286         for (int i=0; i<sites.size(); ++i) {
287             MeasurementSite site = sites.get(i);
288             //System.out.println("PF::KalmanTrackFit2 == check");
289             //site.print("PF::CHECK");
290             int lyr = site.m.Layer;
291             lyrPtr[lyr] = i;
292         }
293         ArrayList<Double> yScat = new ArrayList<Double>(nLyrs);
294         ArrayList<Double> XLscat = new ArrayList<Double>(nLyrs);
295         double rho = 2.329; // Density of silicon in g/cm^2
296         double radLen = (21.82 / rho) * 10.0; // Radiation length of silicon in millimeters
297         for (int lyr=1; lyr<=nLyrs; ++lyr) {
298             MeasurementSite site = sites.get(lyrPtr[lyr]);
299             double yLyr = site.m.p.X().v[1];
300             XLscat.add(site.m.thickness/radLen);
301             yScat.add(yLyr);
302             //System.out.format("Layer %d, y=%10.5f\n", lyr, yLyr);
303         }
304         tkr = new KalTrack(evtNumb, 0, sites, yScat, XLscat, kPar); // Store the fit information as a KalTrack object
305         for (MeasurementSite site : sites) { // Mark the hits as used
306             for (Measurement hit : site.m.hits) { // Should be only a single hit (this is not a pattern recognition routine)
307                 hit.tracks.add(tkr);
308             }
309         }
310         tkr.originHelix(); // Propagates the track to the origin region
311 
312         if (verbose) {
313             System.out.format("KalmanTrackFit2: Final fit chi^2 after smoothing = %12.4e\n", chi2s);
314             double[] hp = tkr.originHelixParms();
315             System.out.format("   At origin, drho=%10.7f, phi0=%10.7f, K=%10.7f, dz=%10.7f, tanl=%10.7f\n", hp[0], hp[1], hp[2], hp[3], hp[4]);
316             Vec afF = sites.get(sites.size() - 1).aF.helix.a;
317             Vec afC = sites.get(sites.size() - 1).aF.helix.helixErrors();
318             afF.print("KalmanFit helix parameters at final filtered site");
319             afC.print("KalmanFit helix parameter errors");
320             if (startSite != null) {
321                 startSite.aS.helix.a.print("KalmanFit helix parameters at the final smoothed site");
322                 startSite.aS.helix.helixErrors().print("KalmanFit helix parameter errors:");
323             }
324         }
325     }

327     public StateVector fittedStateBegin() {
328         return sites.get(initialSite).aS;
329     }
330 
331     public StateVector fittedStateEnd() {
332         return sites.get(sites.size() - 1).aS;
333     }

335     public void printFit(String s) {
336         System.out.format("\nKalmanTrackFit2: dump of track information for %s\n", s);
337         System.out.format("    Final fit chi^2 after filtering = %12.4e\n", chi2f);
338         StateVector fS = sites.get(finalSite).aF;
339         double[] a = fS.helix.a.v;
340         System.out.format("    Helix parameters at the outermost layer=%f10.7 %f10.7 %f10.7 %f10.7 %f10.7\n", a[0], a[1], a[2], a[3], a[4]);
341         double Bmag = fS.helix.B;
342         double[] tB = { fS.helix.Rot.M[2][0], fS.helix.Rot.M[2][1], fS.helix.Rot.M[2][2] };
343         System.out.format("        B-field at the outermost layer=%10.6f,  direction=%8.6f %8.6f %8.6f\n", Bmag, tB[0], tB[1], tB[2]);
344         System.out.format("    Final fit chi^2 after smoothing = %12.4e\n", chi2s);
345         StateVector iS = sites.get(initialSite).aS;
346         if (iS == null) return;
347         a = iS.helix.a.v;
348         System.out.format("    Helix parameters at the innermost layer=%f10.7 %f10.7 %f10.7 %f10.7 %f10.7\n", a[0], a[1], a[2], a[3], a[4]);
349         Bmag = iS.helix.B;
350         double[] tB2 = { iS.helix.Rot.M[2][0], iS.helix.Rot.M[2][1], iS.helix.Rot.M[2][2] };
351         System.out.format("        B-field at the innermost layer=%10.6f,  direction=%8.6f %8.6f %8.6f\n", Bmag, tB2[0], tB2[1], tB2[2]);
352         if (tkr != null) {
353             tkr.originHelix();
354             double[] aO = tkr.originHelixParms();
355             System.out.format("    Helix parameters at the origin=%f10.7 %f10.7 %f10.7 %f10.7 %f10.7\n", aO[0], aO[1], aO[2], aO[3], aO[4]);
356         }
357         for (MeasurementSite site : sites) {
358             if (site.hitID < 0) continue;
359             SiModule m = site.m;
360             StateVector S = site.aS;
361             System.out.format("    Layer %d, detector %d, stereo=%b, chi^2 inc.=%10.6f, predicted=%10.6f, measured=%10.6f, residual=%10.6f\n", m.Layer, m.detector, m.isStereo,
362                     site.chi2inc, S.mPred, m.hits.get(site.hitID).v, S.r);
363         }
364     }

366     public void print(String s) {
367         System.out.format("KalmanTrackFit: dump of the fitted sites, %s, chi2f=%12.4e, chi2s=%12.4e\n", s, chi2f, chi2s);
368         Iterator<MeasurementSite> itr = sites.iterator();
369         while (itr.hasNext()) {
370             MeasurementSite currentSite = itr.next();
371             currentSite.print("dump of fit");
372         }
