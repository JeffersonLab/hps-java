142     static double [] getFielD(Vec kalPos, org.lcsim.geometry.FieldMap hpsFm) {
143         // Field map for stand-alone running
144         if (FieldMap.class.isInstance(hpsFm)) return ((FieldMap) (hpsFm)).getField(kalPos);
145 
146         // Standard field map for running in hps-java
147         //System.out.format("Accessing HPS field map for position %8.3f %8.3f %8.3f\n", kalPos.v[0], kalPos.v[1], kalPos.v[2]);
148         double[] hpsPos = { kalPos.v[0], -1.0 * kalPos.v[2], kalPos.v[1] };
149         if (uniformB) {
150             hpsPos[0] = 0.;
151             hpsPos[1] = 0.;
152             hpsPos[2] = SVTcenter;
153         } else {
154             if (hpsPos[1] > 70.0) hpsPos[1] = 70.0;   // To avoid getting a field returned that is identically equal to zero
155             if (hpsPos[1] < -70.0) hpsPos[1] = -70.0;
156             if (hpsPos[0] < -225.) hpsPos[0] = -225.;
157             if (hpsPos[0] > 270.) hpsPos[0] = 270.;
158         }
159         double[] hpsField = hpsFm.getField(hpsPos);
160         if (uniformB) {
161             double [] kalField = {0., 0., -1.0 * hpsField[1]};
162             return kalField;
163         }
164         double [] kalField = {hpsField[0], hpsField[2], -1.0 * hpsField[1]};
165         return kalField;
166     }

178     public KalmanInterface(boolean uniformB, KalmanParams kPar, org.lcsim.geometry.FieldMap fM) {
179         
180         this.det = det;
181         this.sensors = sensors;
182         this.fM = fM;
183         this.kPar = kPar;
184         logger = Logger.getLogger(KalmanInterface.class.getName());
185         logger.info("Entering the KalmanInterface constructor");
186         maxHits = 0;
187         nBigEvents = 0;
188         
189         tempM = new DMatrixRMaj(5,5);
190         Ft = new DMatrixRMaj(5,5);
191         
192         KalmanInterface.uniformB = uniformB;
193         hitMap = new HashMap<Measurement, TrackerHit>();
194         simHitMap = new HashMap<Measurement, SimTrackerHit>();
195         moduleMap = new HashMap<SiModule, SiStripPlane>();
196         trackHitsKalman = new ArrayList<KalHit>();  // Used only to refit existing GBL tracks
197         SiMlist = new ArrayList<SiModule>();
198         SeedTrackLayers = new ArrayList<Integer>();
199         // SeedTrackLayers.add(2);
200         SeedTrackLayers.add(3);
201         SeedTrackLayers.add(4);
202         SeedTrackLayers.add(5);
203         
204         if (uniformB) {
205             logger.log(Level.WARNING, "KalmanInterface WARNING: the magnetic field is set to a uniform value.");
206         }
207         
208         // Transformation from HPS SVT tracking coordinates to Kalman global coordinates
209         double[][] HpsSvtToKalmanVals = { { 0, 1, 0 }, { 1, 0, 0 }, { 0, 0, -1 } };
210         HpsSvtToKalman = new RotMatrix(HpsSvtToKalmanVals);
211         HpsSvtToKalmanMatrix = new BasicHep3Matrix();
212         for (int i = 0; i < 3; i++) {
213             for (int j = 0; j < 3; j++)
214                 HpsSvtToKalmanMatrix.setElement(i, j, HpsSvtToKalmanVals[i][j]);
215         }
216         KalmanToHpsSvt = HpsSvtToKalman.invert();
217         if (debug) {
218             HpsSvtToKalman.print("HPS tracking to Kalman conversion");
219             KalmanToHpsSvt.print("Kalman to HPS tracking conversion");
220             Vec zHPS = new Vec(0.,0.,1.);
221             Vec xHPS = new Vec(1.,0.,0.);
222             Vec yHPS = new Vec(0.,1.,0.);
223             HpsSvtToKalman.rotate(xHPS).print("HPS tracking x axis in Kalman coordinates");
224             HpsSvtToKalman.rotate(yHPS).print("HPS tracking y axis in Kalman coordinates");
225             HpsSvtToKalman.rotate(zHPS).print("HPS tracking z axis in Kalman coordinates");
226         }
227         
228         // Seed the random number generator
229         long rndSeed = -3263009337738135404L;
230         rnd = new Random();
231         rnd.setSeed(rndSeed);
232         
233         kPat = new KalmanPatRecHPS(kPar);
234         
235         Vec centerB = KalmanInterface.getField(new Vec(0., SVTcenter, 0.), fM);
236         double conFac = 1.0e12 / c;
237         alphaCenter = conFac/ centerB.mag();
238     }

325     // Create an HPS TrackState from a Kalman HelixState at the location of a particular SiModule
326     public TrackState createTrackState(MeasurementSite ms, int loc, boolean useSmoothed) {
327         // Note that the helix parameters that get stored in the TrackState assume a B-field exactly oriented in the
328         // z direction and a pivot point at the origin (0,0,0). The referencePoint of the TrackState is set to the
329         // intersection point with the detector plane.
330         StateVector sv = null;
331         if (useSmoothed) {
332             if (!ms.smoothed) return null;
333             sv = ms.aS;
334         } else {   // using the filtered state is really not recommended
335             if (!ms.filtered) return null;
336             sv = ms.aF;
337         }
338 
339         return sv.helix.toTrackState(alphaCenter, ms.m.p, loc);
340     }

342     // Transform a Kalman helix to an HPS helix rotated to the global frame and with the pivot at the origin
343     // Provide covHPS with 15 elements to get the covariance as well
344     // Provide 3-vector position to get the location in HPS global coordinates of the original helix pivot
345     static double [] toHPShelix(HelixState helixState, Plane pln, double alphaCenter, double [] covHPS, double[] position) {
346         final boolean debug = false;
347         double phiInt = helixState.planeIntersect(pln);
348         if (Double.isNaN(phiInt)) {
349             Logger logger = Logger.getLogger(KalmanInterface.class.getName());
350             logger.fine(String.format("toHPShelix: no intersection with the plane at %s",pln.toString()));
351             phiInt = 0.;
352         }
353         // Transforms helix to a pivot point on the helix itself (so rho0 and z0 become zero)
354         Vec newPivot = helixState.atPhi(phiInt);
355         Vec helixParamsPivoted = helixState.pivotTransform(newPivot);
356         DMatrixRMaj F = new DMatrixRMaj(5,5);
357         helixState.makeF(helixParamsPivoted, F);
358         if (debug) {
359             System.out.format("Entering KalmanInterface.toHPShelix");
360             helixState.print("provided");
361             pln.print("provided");
362             newPivot.print("new pivot");
363             helixParamsPivoted.print("pivoted helix params");
364             System.out.format("turning angle to the plane containing the helixState origin=%10.6f\n", phiInt);
365             Vec intGlb = helixState.toGlobal(newPivot);
366             intGlb.print("global intersection with plane");
367         }
368         
369         // Then rotate the helix to the global system. This isn't quite kosher, since the B-field will not
370         // be aligned with the global system in general, but we have to do it to fit back into the HPS TrackState
371         // coordinate convention, for which the field is assumed to be uniform and aligned.
372         DMatrixRMaj fRot = new DMatrixRMaj(5,5);
373         Vec helixParamsRotated = HelixState.rotateHelix(helixParamsPivoted, helixState.Rot.invert(), fRot);
374         CommonOps_DDRM.mult(fRot, F, Ft);             
375         
376         CommonOps_DDRM.multTransB(helixState.C, Ft, tempM);
377         DMatrixRMaj covRotated = new DMatrixRMaj(5,5);
378         CommonOps_DDRM.mult(Ft, tempM, covRotated);
379         if (debug) helixParamsRotated.print("rotated helix params");
380         
381         // Transform the pivot to the global system. 
382         Vec pivotGlobal = helixState.toGlobal(newPivot);
383         if (debug) pivotGlobal.print("pivot in global system");
384         
385         // Pivot transform to the final pivot at the origin
386         Vec finalPivot = new Vec(0.,0.,0.);
387         Vec finalHelixParams = HelixState.pivotTransform(finalPivot, helixParamsRotated, pivotGlobal, alphaCenter, 0.);
388         HelixState.makeF(finalHelixParams, F, helixParamsRotated, alphaCenter);
389         CommonOps_DDRM.multTransB(covRotated, F, tempM);
390         CommonOps_DDRM.mult(F, tempM, covRotated);
391         if (debug) {
392             finalPivot.print("final pivot point");
393             finalHelixParams.print("final helix parameters");
394             HelixPlaneIntersect hpi = new HelixPlaneIntersect();
395             phiInt = hpi.planeIntersect(finalHelixParams, finalPivot, alphaCenter, pln);
396             if (!Double.isNaN(phiInt)) {
397                 Vec rInt = HelixState.atPhi(finalPivot, finalHelixParams, phiInt, alphaCenter);
398                 rInt.print("final helix intersection with given plane");
399             }
400             System.out.format("Exiting KalmanInterface.toHPShelix\n");
401         }
402         if (covHPS != null) {
403             double [] temp = KalmanInterface.getLCSimCov(covRotated, alphaCenter).asPackedArray(true);
404             for (int i=0; i<15; ++i) covHPS[i] = temp[i];
405         }
406         if (position != null) {
407             double [] temp = KalmanInterface.vectorKalmanToGlb(pivotGlobal);
408             for (int i=0; i<3; ++i) position[i] = temp[i];
409         }
410         return KalmanInterface.getLCSimParams(finalHelixParams.v, alphaCenter);
411     }

436     // Make a GBLStripClusterData object for each MeasurementSite of a Kalman track
437     public List<GBLStripClusterData> createGBLStripClusterData(KalTrack kT) {
438         List<GBLStripClusterData> rtnList = new ArrayList<GBLStripClusterData>(kT.SiteList.size());
439 
440         //Arc length from origin to first plane
441         
442         double phi_org         = kT.originHelixParms()[1];
443         double phi_1state      = kT.SiteList.get(0).aS.helix.a.v[1]; 
444         double alpha           = kT.helixAtOrigin.alpha;
445         double radius          = Math.abs(alpha/kT.originHelixParms()[2]);
446         double arcLength2D     = radius*(phi_1state-phi_org);
447         double arcLength       = arcLength2D*Math.sqrt(1.0 + kT.originHelixParms()[4] * kT.originHelixParms()[4]);
448 
449         double total_s3D = arcLength;
450         double total_s2D = arcLength2D;
451         
452         double phiLast = 9999.;
453 
454         for (MeasurementSite site : kT.SiteList) {
455             GBLStripClusterData clstr = new GBLStripClusterData(site.m.millipedeID);
456             clstr.setVolume(site.m.topBottom);
457             
458             // Sites without hits are "scatter-only"
459             if (site.hitID < 0) clstr.setScatterOnly(1);
460             else clstr.setScatterOnly(0);
461             
462             // Store the total Arc length in the GBLStripClusterData
463             total_s3D += site.arcLength;
464             clstr.setPath3D(total_s3D);
465             double tanL = site.aS.helix.a.v[4];
466             clstr.setPath(site.arcLength/FastMath.sqrt(1.+tanL*tanL));
467             
468             Hep3Vector u = new BasicHep3Vector(vectorKalmanToTrk(site.m.p.V()));
469             Hep3Vector v = new BasicHep3Vector(vectorKalmanToTrk(site.m.p.U().scale(-1.0)));
470             Hep3Vector w = new BasicHep3Vector(vectorKalmanToTrk(site.m.p.T()));
471             
472             clstr.setU(u);
473             clstr.setV(v);
474             clstr.setW(w);
475             
476             // Direction of the track in the HPS tracking coordinate system
477             // Find the momentum from the smoothed helix at the sensor location, make it a unit vector, 
478             // and then transform from the B-field frame to the Kalman global tracking frame.
479             Vec momentum = site.aS.helix.getMom(0.);
480             Vec pDir= site.aS.helix.Rot.inverseRotate(momentum.unitVec());
481             Hep3Vector trackDir = new BasicHep3Vector(vectorKalmanToTrk(pDir));
482             clstr.setTrackDir(trackDir);
483             
484             // Phi and lambda of the track (assuming standard spherical polar coordinates)
485             double phi = FastMath.atan2(trackDir.y(), trackDir.x());
486             double ct = trackDir.z()/trackDir.magnitude();
487             double tanLambda = ct/FastMath.sqrt(1-ct*ct);  // Should be very much the same as tanL above, after accounting for the field tilt
488             if (debug) {
489                 Vec tilted = site.aS.helix.Rot.inverseRotate(new Vec(0.,0.,1.));
490                 double tiltAngle = FastMath.acos(tilted.v[2]);
491                 System.out.format("KalmanInterface.createGBLStripClusterData: layer=%d det=%d tanL=%10.6f, tanLambda=%10.6f, tilt=%10.6f, sum=%10.6f\n", 
492                         site.m.Layer, site.m.detector, -tanL, tanLambda, tiltAngle, tiltAngle+tanLambda);
493             }
494             clstr.setTrackPhi(phi);
495             if (phiLast < 10.) {
496                 if (Math.abs(phi - phiLast) > 1.2) System.out.format("Big phi change in event %d\n", eventNumber);
497             }
498             phiLast = phi;
499             clstr.setTrackLambda(FastMath.atan(tanLambda));
500             
501             // Measured value in the sensor coordinates (u-value in the HPS system)
502             double uMeas, uMeasErr;
503             if (site.hitID >= 0) {
504                 uMeas = site.m.hits.get(site.hitID).v; 
505                 uMeasErr = site.m.hits.get(site.hitID).sigma;
506                 //This is the un-biased residual error (post-fit)
507                 //uMeasErr = Math.sqrt(site.aS.R);
508

636     // Convert helix parameters from Kalman to LCSim
637     static double[] getLCSimParams(double[] oldParams, double alpha) {  
638         // Note: since HPS-java has assumed a constant field for tracking, the alpha value used here should
639         // correspond to the field at the center of the SVT or magnet.   See the class variable alphaCenter.
640         double[] params = new double[5];
641         params[ParameterName.d0.ordinal()] = oldParams[0];
642         params[ParameterName.phi0.ordinal()] = -1.0 * oldParams[1];
643         params[ParameterName.omega.ordinal()] = oldParams[2] / alpha * -1.0;
644         params[ParameterName.z0.ordinal()] = oldParams[3] * -1.0;
645         params[ParameterName.tanLambda.ordinal()] = oldParams[4] * -1.0;
646 
647         return params;
648     }

663     // Convert helix parameter covariance matrix from Kalman to LCSim
664     static SymmetricMatrix getLCSimCov(DMatrixRMaj oldCov, double alpha) {
665         // Note: since HPS-java has assumed a constant field for tracking, the alpha value used here should
666         // correspond to the field at the center of the SVT or magnet.   See the class variable alphaCenter.
667         double[] d = { 1.0, -1.0, -1.0 / alpha, -1.0, -1.0 };
668         SymmetricMatrix cov = new SymmetricMatrix(5);
669         for (int i = 0; i < 5; i++) {
670             for (int j = 0; j < 5; j++) {
671                 cov.setElement(i, j, d[i] * d[j] * oldCov.unsafe_get(i, j));
672             }
673         }
674         /*
675          * for (int i = 0; i <= 2; i++) { for (int j = 0; j <= 2; j++) {
676          * cov.setElement(i, j, oldCov.M[i][j]); } } for (int i = 3; i <= 4; i++) { for
677          * (int j = 0; j <= 4; j++) { cov.setElement(i, j, oldCov.M[j][i]);
678          * cov.setElement(j, i, oldCov.M[i][j]); } }
679          */
680         return cov;
681     }

683     // Convert helix parameter covariance matrix from LCSim to Kalman
684     static double[][] ungetLCSimCov(double[] oldCov, double alpha) {
685         // Note: since HPS-java has assumed a constant field for tracking, the alpha value used here should
686         // correspond to the field at the center of the SVT or magnet.   See the class variable alphaCenter.
687         double[] d = { 1.0, -1.0, -1.0 * alpha, -1.0, -1.0 };
688         double[][] cov = new double[5][5];
689         cov[0][0] = oldCov[0] * d[0] * d[0];
690         cov[1][0] = oldCov[1] * d[1] * d[0];
691         cov[1][1] = oldCov[2] * d[1] * d[1];
692         cov[2][0] = oldCov[3] * d[2] * d[0];
693         cov[2][1] = oldCov[4] * d[2] * d[1];
694         cov[2][2] = oldCov[5] * d[2] * d[2];
695         cov[3][0] = oldCov[6] * d[3] * d[0];
696         cov[3][1] = oldCov[7] * d[3] * d[1];
697         cov[3][2] = oldCov[8] * d[3] * d[2];
698         cov[3][3] = oldCov[9] * d[3] * d[3];
699         cov[4][0] = oldCov[10] * d[4] * d[0];
700         cov[4][1] = oldCov[11] * d[4] * d[1];
701         cov[4][2] = oldCov[12] * d[4] * d[2];
702         cov[4][3] = oldCov[13] * d[4] * d[3];
703         cov[4][4] = oldCov[14] * d[4] * d[4];
704         for (int i = 0; i < 5; ++i) {
705             for (int j = i + 1; j < 5; ++j) {
706                 cov[i][j] = cov[j][i];
707             }
708         }
709         return cov;
710     }
