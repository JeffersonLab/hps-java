65      KalTrack(int evtNumb, int tkID, ArrayList<MeasurementSite> SiteList, ArrayList<Double> yScat, ArrayList<Double> XLscat, KalmanParams kPar) {
66          // System.out.format("KalTrack constructor chi2=%10.6f\n", chi2);
67          eventNumber = evtNumb;
68          bad = false;
69          this.yScat = yScat;
70          this.XLscat = XLscat;        
71          this.kPar = kPar;
72          ID = tkID;
73          arcLength = null;
74          //debug = (evtNumb == 217481);
75          
76          if (!initialized) {
77              logger = Logger.getLogger(KalTrack.class.getName());
78              tempV = new DMatrixRMaj(5,1);
79              Cinv = new DMatrixRMaj(5,5);
80              initialized = true;
81              solver = LinearSolverFactory_DDRM.symmPosDef(5);
82          }
83          
84          // Trim empty sites from the track ends
85          Collections.sort(SiteList, MeasurementSite.SiteComparatorUp);
86          int firstSite = -1;
87          for (int idx=0; idx<SiteList.size(); ++idx) {
88              firstSite = idx;
89              if (SiteList.get(idx).hitID >= 0) break;
90          }
91          int lastSite = 999;
92          for (int idx = SiteList.size()-1; idx >= 0; --idx) {
93              lastSite = idx;
94              if (SiteList.get(idx).hitID >= 0) break;
95          }
96          
97          // Make a new list of sites, without empty sites at beginning or end
98          this.SiteList = new ArrayList<MeasurementSite>(SiteList.size());
99          for (int idx=firstSite; idx<=lastSite; ++idx) { 
100             MeasurementSite site = SiteList.get(idx);
101             if (site.aS == null) {  // This should never happen
102                 logger.log(Level.SEVERE, String.format("Event %d: site of track %d is missing smoothed state vector for layer %d detector %d", 
103                         eventNumber, ID, site.m.Layer, site.m.detector));
104                 logger.log(Level.WARNING, site.toString("bad site"));
105                 bad = true;
106                 continue;
107             }
108             this.SiteList.add(site); 
109         }
110         
111         helixAtOrigin = null;
112         propagated = false;
113         MeasurementSite site0 = this.SiteList.get(0);
114         Vec B = KalmanInterface.getField(new Vec(3,kPar.beamSpot), site0.m.Bfield);
115         Bmag = B.mag();
116         tB = B.unitVec(Bmag);
117         Vec yhat = new Vec(0., 1.0, 0.);
118         Vec uB = yhat.cross(tB).unitVec();
119         Vec vB = tB.cross(uB);
120         Rot = new RotMatrix(uB, vB, tB);
121         originPoint = null;
122         originMomentum = null;
123         double c = 2.99793e8; // Speed of light in m/s
124         alpha = 1.0e12 / (c * Bmag); // Convert from pt in GeV to curvature in mm
125         Cx = null;
126         Cp = null;        
127         // Fill the maps
128         time = 0.;
129         tMin = 9.9e9;
130         tMax = -9.9e9;
131         this.chi2 = 0.;
132         this.nHits = 0;
133         if (debug) System.out.format("KalTrack: event %d, creating track %d\n", evtNumb, ID);
134         for (MeasurementSite site : this.SiteList) {
135             if (site.hitID < 0) continue;
136             nHits++;
137             time += site.m.hits.get(site.hitID).time;
138             tMin = Math.min(tMin, site.m.hits.get(site.hitID).time);
139             tMax = Math.max(tMax,  site.m.hits.get(site.hitID).time);
140             this.chi2 += site.chi2inc;
141             if (debug) System.out.format("  Layer %d, chi^2 increment=%10.5f, a=%s\n", site.m.Layer, site.chi2inc, site.aS.helix.a.toString());
142         }
143         time = time/(double)nHits; 
144         reducedChi2 = chi2/(double)nHits;
145         lyrMap = null;
146         millipedeMap = null;
147         interceptVects = null;
148         interceptMomVects = null;
149         if (nHits < 5) {  // This should never happen
150             logger.log(Level.WARNING, "KalTrack error: not enough hits ("+nHits+") on the candidate track (ID::"+ID+") for event "+eventNumber);
151             bad = true;
152             //for (MeasurementSite site : SiteList) logger.log(Level.FINE, site.toString("in KalTrack input list"));
153             //logger.log(Level.FINE, String.format("KalTrack error in event %d: not enough hits on track %d: ",evtNumb,tkID));
154             //String str="";
155             //for (MeasurementSite site : SiteList) {
156             //    str = str + String.format("(%d, %d, %d) ",site.m.Layer,site.m.detector,site.hitID);
157             //}
158             //str = str + "\n";
159             //logger.log(Level.FINER,str);
160         }
161     }

167     public Map<MeasurementSite, Vec> interceptVects() {
168         if (interceptVects == null) {
169             interceptVects = new HashMap<MeasurementSite, Vec>(nHits);
170             for (MeasurementSite site : this.SiteList) {
171                 StateVector sV = null;
172                 if (site.smoothed) sV = site.aS;
173                 else sV = site.aP;
174                 double phiS = sV.helix.planeIntersect(site.m.p);
175                 if (Double.isNaN(phiS)) phiS = 0.;
176                 interceptVects.put(site, sV.helix.toGlobal(sV.helix.atPhi(phiS)));                
177             }
178         }
179         return interceptVects;
180     }

197     // Calculate and return the intersection point of the Kaltrack with an SiModule.
198     // Local sensor coordinates (u,v) are returned. 
199     // The global intersection can be returned via rGbl if an array of length 3 is passed.
200     public double [] moduleIntercept(SiModule mod, double [] rGbl) {
201         HelixState hx = null;
202         for (MeasurementSite site : SiteList) {
203             if (site.m == mod) hx = site.aS.helix;
204         }
205         if (hx == null) {
206             int mxLayer = -1;
207             for (MeasurementSite site : SiteList) {
208                 if (site.m.Layer > mod.Layer) continue;
209                 if (site.m.Layer > mxLayer) {
210                     mxLayer = site.m.Layer;
211                     hx = site.aS.helix;
212                 }
213             }
214         }
215         if (hx == null) hx = SiteList.get(0).aS.helix;
216         double phiS = hx.planeIntersect(mod.p);
217         if (Double.isNaN(phiS)) phiS = 0.;
218         Vec intGlb = hx.toGlobal(hx.atPhi(phiS));
219         if (rGbl != null) {
220             rGbl[0] = intGlb.v[0];
221             rGbl[1] = intGlb.v[1];
222             rGbl[2] = intGlb.v[2];
223         }
224         Vec intLcl = mod.toLocal(intGlb);
225         double [] rtnArray = {intLcl.v[0], intLcl.v[1]};
226         return rtnArray;
227     }

243     // Find the change in smoothed helix angle in XY between one layer and the next
244     public double scatX(int layer) {
245         if (lyrMap == null) makeLyrMap();
246         if (!lyrMap.containsKey(layer)) return -999.;
247         int lyrNxt = layer + 1;
248         while (lyrNxt <= 13 && !lyrMap.containsKey(lyrNxt)) lyrNxt++;
249         if (lyrNxt > 13) return -999.; 
250 
251         MeasurementSite s1 = lyrMap.get(layer);
252         MeasurementSite s2 = lyrMap.get(lyrNxt);
253         if (s1.aS == null || s2.aS == null) return -999.;
254         double phiS1 = s1.aS.helix.planeIntersect(s2.m.p);
255         if (Double.isNaN(phiS1)) return -999.;
256         Vec p1 = s1.aS.helix.getMom(phiS1);
257         double t1 = FastMath.atan2(p1.v[0], p1.v[1]);
258         double phiS2 = s2.aS.helix.planeIntersect(s2.m.p);
259         if (Double.isNaN(phiS2)) return -999.;
260         Vec p2 = s2.aS.helix.getMom(phiS2);
261         double t2 = FastMath.atan2(p2.v[0], p2.v[1]);
262         return t1 - t2;
263     }

267         if (lyrMap == null) makeLyrMap();
268         if (!lyrMap.containsKey(layer)) return -999.;

272         MeasurementSite s1 = lyrMap.get(layer);
273         MeasurementSite s2 = lyrMap.get(lyrNxt);
274         if (s1.aS == null || s2.aS == null) return -999.;
275         double phiS1 = s1.aS.helix.planeIntersect(s2.m.p);
276         if (Double.isNaN(phiS1)) return -999.;
277         Vec p1 = s1.aS.helix.Rot.inverseRotate(s1.aS.helix.getMom(phiS1));
278         double t1 = FastMath.atan2(p1.v[2], p1.v[1]);
279         double phiS2 = s2.aS.helix.planeIntersect(s2.m.p);
280         if (Double.isNaN(phiS2)) return -999.;
281         Vec p2 = s2.aS.helix.Rot.inverseRotate(s2.aS.helix.getMom(phiS2));
282         double t2 = FastMath.atan2(p2.v[2], p2.v[1]);
283         return t1 - t2;

286     public double chi2prime() { // Alternative calculation of the fit chi^2, considering only residuals divided by the hit sigma
287         double c2 = 0.;
288         for (MeasurementSite S : SiteList) {
289             if (S.aS == null) continue;
290             double phiS = S.aS.helix.planeIntersect(S.m.p);
291             if (Double.isNaN(phiS)) { phiS = 0.; }
292             double vpred = S.h(S.aS, S.m, phiS);
293             for (Measurement hit : S.m.hits) {
294                 for (KalTrack tkr : hit.tracks) { 
295                     if (tkr.equals(this)) c2 += FastMath.pow((vpred - hit.v) / hit.sigma, 2);
296                 }
297             }
298         }
299         return c2;
300     }

320     // Returns the unbiased residual for the track at a given layer, together with the variance on that residual
321     public Pair<Double,Double> unbiasedResidual(MeasurementSite site) {
322         double resid = -999.;
323         double varResid = -999.;               
324         Vec aStar = null;  
325         if (site.hitID >= 0) {
326             double sigma = site.m.hits.get(site.hitID).sigma;
327             DMatrixRMaj Cstar = new DMatrixRMaj(5,5);
328             aStar = site.aS.inverseFilter(site.H, sigma*sigma, Cstar);
329             HelixPlaneIntersect hpi = new HelixPlaneIntersect();
330             Plane pTrans = site.m.p.toLocal(site.aS.helix.Rot, site.aS.helix.origin);
331             double phiInt = hpi.planeIntersect(aStar, site.aS.helix.X0, site.aS.helix.alpha, pTrans);
332             if (!Double.isNaN(phiInt)) {
333                 Vec intPnt = HelixState.atPhi(site.aS.helix.X0, aStar, phiInt, site.aS.helix.alpha);
334                 Vec globalInt = site.aS.helix.toGlobal(intPnt);
335                 Vec localInt = site.m.toLocal(globalInt);
336                 resid = site.m.hits.get(site.hitID).v - localInt.v[1];
337                 
338                 CommonOps_DDRM.mult(Cstar, site.H, tempV);               
339                 varResid = sigma*sigma + CommonOps_DDRM.dot(site.H, tempV);
340             }
341         }        
342         return new Pair<Double,Double>(resid, varResid);
343     }

436                 str=str+"\n";
437             }

439                 Vec interceptVec = interceptVects().get(site);
440                 Vec interceptMomVec = interceptMomVects().get(site);
441                 double residual = site.m.hits.get(hitID).v - site.aS.mPred;
442                 Pair<Double,Double> unBiasedResid = unbiasedResidual(site);

444                         interceptMomVec.toString(), site.m.hits.get(hitID).v, site.aS.mPred, residual, unBiasedResid.getFirstElement(), unBiasedResid.getSecondElement(), FastMath.sqrt(site.aS.R));
445             }
446         }
447         str=str+String.format("End of printing for KalTrack %s ID %d in event %d\n\n", s, ID, eventNumber);
448         return str;
449     }

803     private static double phiDOCA(Vec a, Vec v, Vec X0, double alpha) {
804

818     // Safe Newton-Raphson zero finding from Numerical Recipes in C, used here to solve for the point of closest approach.
819     private static double rtSafe(double xGuess, double x1, double x2, double xacc, Vec a, Vec v, Vec X0, double alpha) {
820         // Here xGuess is a starting guess for the phi angle of the helix DOCA point
821         // x1 and x2 give a range for the value of the solution
822         // xacc specifies the accuracy needed
823         // The output is an accurate result for the phi of the DOCA point
824         double df, dx, dxold, f, fh, fl;
825         double temp, xh, xl, rts;
826         int MAXIT = 100;
827 
828         if (xGuess <= x1 || xGuess >= x2) {
829             Logger.getLogger(KalTrack.class.getName()).log(Level.WARNING,"rtsafe: initial guess needs to be bracketed.");
830             return xGuess;
831         }
832         fl = fDOCA(x1, a, v, X0, alpha);
833         fh = fDOCA(x2, a, v, X0, alpha);
834         int nTry = 0;
835         while (fl*fh > 0.0) {
836             if (nTry == 5) {
837                 Logger.getLogger(KalTrack.class.getName()).log(Level.FINE,String.format("Root is not bracketed in zero finding, fl=%12.5e, fh=%12.5e, alpha=%10.6f, x1=%12.5f x2=%12.5f xGuess=%12.5f", 
838                         fl, fh, alpha, x1, x2, xGuess));
839                 return xGuess;
840             }
841             x1 -= 0.01;
842             x2 += 0.01;
843             fl = fDOCA(x1, a, v, X0, alpha);
844             fh = fDOCA(x2, a, v, X0, alpha);
845             nTry++;
846         }
847         //if (nTry > 0) System.out.format("KalTrack.rtsafe: %d tries needed to bracket solution.\n", nTry);
848         if (fl == 0.) return x1;
849         if (fh == 0.) return x2;
850         if (fl < 0.0) {
851             xl = x1;
852             xh = x2;
853         } else {
854             xh = x1;
855             xl = x2;
856         }
857         rts = xGuess;
858         dxold = Math.abs(x2 - x1);
859         dx = dxold;
860         f = fDOCA(rts, a, v, X0, alpha);
861         df = dfDOCAdPhi(rts,a, v, X0, alpha);
862         for (int j = 1; j <= MAXIT; j++) {
863             if ((((rts - xh) * df - f) * ((rts - xl) * df - f) > 0.0) || (Math.abs(2.0 * f) > Math.abs(dxold * df))) {
864                 dxold = dx;
865                 dx = 0.5 * (xh - xl); // Use bisection if the Newton-Raphson method is going bonkers
866                 rts = xl + dx;
867                 if (xl == rts) return rts;
868             } else {
869                 dxold = dx;
870                 dx = f / df; // Newton-Raphson method
871                 temp = rts;
872                 rts -= dx;
873                 if (temp == rts) return rts;
874             }
875             if (Math.abs(dx) < xacc) {
876                 // System.out.format("KalTrack.rtSafe: solution converged in %d iterations.\n",
877                 // j);
878                 return rts;
879             }
880             f = fDOCA(rts, a, v, X0, alpha);
881             df = dfDOCAdPhi(rts,a, v, X0, alpha);
882             if (f < 0.0) {
883                 xl = rts;
884             } else {
885                 xh = rts;
886             }
887         }
888         Logger.getLogger(KalTrack.class.getName()).log(Level.WARNING,"rtsafe: maximum number of iterations exceeded.");
889         return rts;
890     }

892     // Function that is zero when the helix turning angle phi is at the point of closest approach to v
893     private static double fDOCA(double phi, Vec a, Vec v, Vec X0, double alpha) {
894         Vec t = tangentVec(a, phi, alpha);
895         Vec x = HelixState.atPhi(X0, a, phi, alpha);
896         return (v.dif(x)).dot(t);
897     }

899     // derivative of the fDOCA function with respect to phi, for the zero-finding algorithm
900     private static double dfDOCAdPhi(double phi, Vec a, Vec v, Vec X0, double alpha) {
901         Vec x = HelixState.atPhi(X0, a, phi, alpha);
902         Vec dxdphi = dXdPhi(a, phi, alpha);
903         Vec t = tangentVec(a, phi, alpha);
904         Vec dtdphi = dTangentVecDphi(a, phi, alpha);
905         Vec dfdt = v.dif(x);
906         double dfdphi = -t.dot(dxdphi) + dfdt.dot(dtdphi);
907         return dfdphi;
908     }

910     // Derivatives of position along a helix with respect to the turning angle phi
911     private static Vec dXdPhi(Vec a, double phi, double alpha) {
912         return new Vec((alpha / a.v[2]) * FastMath.sin(a.v[1] + phi), -(alpha / a.v[2]) * FastMath.cos(a.v[1] + phi),
913                 -(alpha / a.v[2]) * a.v[4]);
914     }

916     // A vector tangent to the helix 'a' at the alpha phi
917     private static Vec tangentVec(Vec a, double phi, double alpha) {
918         return new Vec((alpha/a.v[2])*FastMath.sin(a.v[1]+phi), -(alpha/a.v[2])*FastMath.cos(a.v[1]+phi), -(alpha/a.v[2])*a.v[4]);
919     }

921     private static Vec dTangentVecDphi(Vec a, double phi, double alpha) {
922         return new Vec((alpha/a.v[2])*FastMath.cos(a.v[1]+phi), (alpha/a.v[2])*FastMath.sin(a.v[2]+phi), 0.);
923     }

925     //Derivative matrix for the helix 'a' point of closet approach to point 'v'
926     private static double [][] buildH(Vec a, Vec v, Vec X0, double phi, double alpha) {
927         // a = helix parameters
928         // X0 = pivot point of helix
929         // phi = angle along helix to the point of closet approach
930         // v = 3D point for which we are finding the DOCA (the "measurement" point)
931         // alpha = constant to convert from curvature to 1/pt
932         
933         Vec x = HelixState.atPhi(X0, a, phi, alpha);
934         Vec dxdphi = dXdPhi(a, phi, alpha);
935         Vec t = tangentVec(a, phi, alpha);
936         Vec dtdphi = dTangentVecDphi(a, phi, alpha);
937         Vec dfdt = v.dif(x);
938         double dfdphi = -t.dot(dxdphi) + dfdt.dot(dtdphi);
939         double [][] dtda = new double[3][5];
940         dtda[0][1] = (alpha/a.v[2])*FastMath.cos(a.v[1]+phi);
941         dtda[0][2] = (-alpha/(a.v[2]*a.v[2]))*FastMath.sin(a.v[1]+phi);
942         dtda[1][1] = (alpha/a.v[2])*FastMath.sin(a.v[1]+phi);
943         dtda[1][2] = (alpha/(a.v[2]*a.v[2]))*FastMath.cos(a.v[1]+phi);
944         dtda[2][2] = (alpha/(a.v[2]*a.v[2]))*a.v[4];
945         dtda[2][4] = -alpha/a.v[2];
946         
947         double [][] dxda = new double[3][5];
948         dxda[0][0] = FastMath.cos(a.v[1]);
949         dxda[1][0] = FastMath.sin(a.v[1]);
950         dxda[0][1] = -(a.v[0] + alpha / a.v[2]) * FastMath.sin(a.v[1]) + (alpha / a.v[2]) * FastMath.sin(a.v[1] + phi);
951         dxda[1][1] = (a.v[0] + alpha / a.v[2]) * FastMath.cos(a.v[1]) - (alpha / a.v[2]) * FastMath.cos(a.v[1] + phi);
952         dxda[0][2] = -(alpha / (a.v[2] * a.v[2])) * (FastMath.cos(a.v[1]) - FastMath.cos(a.v[1] + phi));
953         dxda[1][2] = -(alpha / (a.v[2] * a.v[2])) * (FastMath.sin(a.v[1]) - FastMath.sin(a.v[1] + phi));
954         dxda[2][2] = (alpha / (a.v[2] * a.v[2])) * a.v[4] * phi;
955         dxda[2][3] = 1.0;
956         dxda[2][4] = -(alpha / a.v[2]) * phi;
957         
958         Vec dfda = new Vec(5);
959         Vec dphida = new Vec(5);
960         for (int i=0; i<5; ++i) {
961             for (int j=0; j<3; ++j) {
962                 dfda.v[i] += -t.v[j]*dxda[j][i] + dfdt.v[j]*dtda[j][i];
963             }
964             dphida.v[i] = -dfda.v[i]/dfdphi;
965         }
966         
967         double [][] H = new double[3][5];
968         for (int i=0; i<3; ++i) {
969             for (int j=0; j<5; ++j) {
970                 H[i][j] = dxdphi.v[i]*dphida.v[j] + dxda[i][j];
971             }           
972         }
973         
974         return H;
975     }

1039                     eventNumber, ID, site.hitID, site.m.Layer, site.m.detector);
1040         } else {
1041             SiteList.remove(site);
1042         }
1043         return exchange;

1098                                 if (debug) System.out.format("KalTrack.addHits: event %d track %d filtered chi2inc=%8.3f\n",eventNumber,ID,newSite.chi2inc);
1099                                 if (newSite.chi2inc < mxChi2inc) {
1100                                     if (debug) System.out.format("KalTrack.addHits: event %d added hit %d with chi2inc<%8.3f to layer %d\n",
1101                                             eventNumber, newSite.hitID, newSite.chi2inc, module.Layer);
1102                                     newSites.add(newSite);
1103                                     numAdded++;
1104                                     nHits++;
1105                                     siteFrom = newSite;
1106                                     double hitTime = newSite.m.hits.get(newSite.hitID).time;
1107                                     if (hitTime > tMax) tMax = hitTime;
1108                                     else if (hitTime < tMin) tMin = hitTime;

1111                             }
1112                         }
1113                     }
1114                 }              
1115             }
1116         }
1117         if (numAdded > 0) {

1130             sortSites(true);

1137             }
1138         } else {
1139             if (debug) System.out.format("KalTrack.addHits: no hits added in event %d to track %d\n", eventNumber, ID);
1140         }

1145     // re-fit the track 
1146     public boolean fit(boolean keep) {
1147         // keep = true if there might be another recursion after dropping more hits
1148         double chi2s = 0.;
1149         if (debug) System.out.format("Entering KalTrack.fit for event %d, track %d\n", eventNumber, ID);
1150         StateVector sH = SiteList.get(0).aS.copy();
1151         boolean badC = KalmanPatRecHPS.negativeCov(sH.helix.C);
1152         if (badC) {
1153             if (debug) System.out.format("KalTrack.fit: negative starting covariance, event %d track %d\n", eventNumber, ID);
1154             KalmanPatRecHPS.setInitCov(sH.helix.C, sH.helix.a, false);
1155         } else {
1156             CommonOps_DDRM.scale(100., sH.helix.C);  // Blow up the initial covariance matrix to avoid double counting measurements 
1157         }
1158         SiModule prevMod = null;
1159         double chi2f = 0.;
1160         ArrayList<MeasurementSite> newSiteList = new ArrayList<MeasurementSite>(SiteList.size());
1161         boolean badCov = false;
1162         for (int idx = 0; idx < SiteList.size(); idx++) { // Redo all the filter steps
1163             MeasurementSite currentSite = SiteList.get(idx);
1164             MeasurementSite newSite = new MeasurementSite(currentSite.m.Layer, currentSite.m, kPar);
1165             
1166             boolean allowSharing = false;
1167             boolean pickupHits = false;
1168             boolean checkBounds = false;
1169             double [] tRange = {-999., 999.};
1170             if (newSite.makePrediction(sH, prevMod, currentSite.hitID, allowSharing, pickupHits, checkBounds, tRange, 0) < 0) {
1171                 if (debug) System.out.format("KalTrack.fit: event %d, track %d failed to make prediction at layer %d!\n", eventNumber, ID, newSite.m.Layer);
1172                 return false;
1173             }
1174             if (!newSite.filter()) {
1175                 if (debug) System.out.format("KalTrack.fit: event %d, track %d failed to filter!\n", eventNumber, ID);
1176                 return false;
1177             }
1178             if (KalmanPatRecHPS.negativeCov(currentSite.aF.helix.C)) {
1179                 if (debug) System.out.format("KalTrack: event %d, ID %d, negative covariance after filtering at layer %d\n", 
1180                                 eventNumber,ID,currentSite.m.Layer);
1181                 badCov = true;
1182                 KalmanPatRecHPS.fixCov(currentSite.aF.helix.C, currentSite.aF.helix.a);
1183             }
1184             if (debug) {
1185                 if (newSite.hitID >= 0) chi2f += Math.max(currentSite.chi2inc,0.);
1186             }
1187             newSite.hitID = currentSite.hitID;
1188             sH = newSite.aF;
1189             if (debug) System.out.format("  Layer %d hit %d filter, chi^2 increment=%10.5f, a=%s\n", 
1190                     newSite.m.Layer, newSite.hitID, newSite.chi2inc, newSite.aF.helix.a.toString());
1191             prevMod = newSite.m;
1192             if (keep) currentSite.chi2inc = newSite.chi2inc; // Residuals to cut out hits in next recursion
1193             newSiteList.add(newSite);
1194         }
1195         if (badCov) {
1196             nBadCov[0]++;
1197             bad = true;
1198         }        
1199         if (debug) System.out.format("KalTrack.fit: Track %d, Fit chi^2 after filtering = %12.4e\n", ID, chi2f);
1200         
1201         chi2s = 0.;
1202         int nNewHits = 0;
1203         badCov = false;
1204         MeasurementSite nextSite = null;
1205         for (int idx = newSiteList.size() - 1; idx >= 0; idx--) {
1206             MeasurementSite currentSite = newSiteList.get(idx);
1207             if (nextSite == null) {
1208                 currentSite.aS = currentSite.aF;
1209                 currentSite.smoothed = true;
1210             } else {
1211                 currentSite.smooth(nextSite);
1212             }
1213             if (currentSite.hitID >= 0) {
1214                 chi2s += Math.max(currentSite.chi2inc,0.);
1215                 nNewHits++;
1216             }
1217             if (KalmanPatRecHPS.negativeCov(currentSite.aS.helix.C)) {
1218                 if (debug) System.out.format("KalTrack: event %d, ID %d, negative covariance after smoothing at layer %d\n", 
1219                             eventNumber,ID,currentSite.m.Layer);
1220                 badCov = true;
1221                 KalmanPatRecHPS.fixCov(currentSite.aS.helix.C, currentSite.aS.helix.a);
1222             }
1223             if (debug) {
1224                 System.out.format("  Layer %d hit %d, smooth, chi^2 increment=%10.5f, a=%s\n", 
1225                             currentSite.m.Layer, currentSite.hitID, currentSite.chi2inc, currentSite.aS.helix.a.toString());
1226             }
1227             nextSite = currentSite;
1228         }
1229         if (badCov) {
1230             nBadCov[1]++;
1231             bad = true;
1232         }
1233         if (debug) System.out.format("KalTrack.fit: Track %d, Fit chi^2 after smoothing = %12.4e\n", ID, chi2s);
1234         if (!keep) {
1235             if (chi2s/(double)nNewHits > reducedChi2*1.2) {        
1236                 if (debug) System.out.format("KalTrack.fit event %d track %d: fit chisquared=%10.5f is not an improvement. Discard new fit.\n", 
1237                                       eventNumber, ID, chi2s);
1238                 return false;
1239             }
1240         }
1241         SiteList = newSiteList;
1242         this.chi2 = chi2s;
1243         this.nHits = nNewHits;
1244         this.reducedChi2 = chi2s/(double)nNewHits;
1245         propagated = false;
1246         if (debug) System.out.format("Exiting KalTrack.fit for event %d, track %d\n", eventNumber, ID);
1247         return true;
1248     }

1250     // Derivative matrix for propagating the covariance of the helix parameters to a covariance of momentum
1251     static double[][] DpTOa(Vec a) {
1252         double[][] M = new double[3][5];
1253         double K = Math.abs(a.v[2]);
1254         double sgn = Math.signum(a.v[2]);
1255         M[0][1] = -FastMath.cos(a.v[1]) / K;
1256         M[1][1] = -FastMath.sin(a.v[1]) / K;
1257         M[0][2] = sgn * FastMath.sin(a.v[1]) / (K * K);
1258         M[1][2] = -sgn * FastMath.sin(a.v[1]) / (K * K);
1259         M[2][4] = 1. / K;
1260         M[2][2] = -sgn * a.v[4] / (K * K);
1261         return M;
1262     }

1264     // Derivative matrix for propagating the covariance of the helix parameter to a
1265     // covariance of the point of closest approach to the origin (i.e. at phi=0)
1266     static double[][] DxTOa(Vec a) {
1267         double[][] M = new double[3][5];
1268         M[0][0] = FastMath.cos(a.v[1]);
1269         M[0][1] = -a.v[0] * FastMath.sin(a.v[1]);
1270         M[1][0] = FastMath.sin(a.v[1]);
1271         M[1][1] = a.v[0] * FastMath.cos(a.v[1]);
1272         M[2][3] = 1.0;
1273         return M;
