10  /**
11   * Fit a line/parabola approximation of a helix to a set of measurement points.
12   * The line and parabola are fit simultaneously in order to handle properly the stereo layers.
13   * The pivot of the helix is assumed to be the origin of the coordinate system, but the
14   * coordinate system in which the helix is described may be rotated slightly with respect to
15   * the global coordinates, in order to optimize its alignment with the field.
16   */

62      String toString(String s) {
63          String str;
64          if (success) {
65              str = String.format("Seed track %s: B=%10.7f helix= %10.6f, %10.6f, %10.6f, %10.6f, %10.6f\n", s, Bavg, drho, phi0, K, dz, tanl);
66              str=str+String.format("  Number of hits in the bending plane=%d; in the non-bending plane=%d\n", Nbending, Nnonbending);
67              str=str+hParm.toString("helix parameters rotated into magnetic field frame")+"\n";
68              str=str+String.format("  Note that these parameters are with respect to a pivot point 0. %10.7f 0.\n", yOrigin);
69              double yP = p0.X().v[1];
70              double[] pivot = { 0., yP, 0. };
71              double[] a = this.pivotTransform(pivot);
72              str=str+String.format("    helix parameters transformed to y=%8.2f: %10.6f, %10.6f, %10.6f, %10.6f, %10.6f\n", yP, a[0], a[1], a[2],
73                      a[3], a[4]);
74              str=str+String.format("  seed track hits:");
75              for (int j = 0; j < hits.size(); j++) str=str+String.format(" %d: %10.5f, ", hits.get(j).module.Layer, hits.get(j).hit.v);
76              str=str+"\n";
77              Vec pInt = planeIntersection(p0);
78              double xzDist = FastMath.sqrt(pInt.v[0]*pInt.v[0] + pInt.v[2]*pInt.v[2]);
79              str=str+String.format("  Distance from origin in X,Z at y= %8.2f is %10.5f. Intersection=%s\n", yP, xzDist, pInt.toString());
80              str = str + "Covariance: " + C.toString();
81          } else {
82              str = String.format("Seed track %s fit unsuccessful.\n", s);
83          }
84          return str;
85      }

109     private void SeedTracker(ArrayList<KalHit> hitList, double yOrigin, double yTarget) {
110         // yOrigin is the location along the beam about which we fit the seed helix
111         // yTarget is the location along the beam of the target itself
112         //if (nCalls < 3) debug = true;
113         //else debug = false;
114         //nCalls++;
115 
116         p0 = new Plane(new Vec(0., yTarget, 0.), new Vec(0., 1., 0.));  // create plane at the target position
117         this.yOrigin = yOrigin;
118         
119         // Store the list of hits with the instance, for comparing different instances later
120         hits = new ArrayList<KalHit>(hitList.size());
121         for (KalHit hit : hitList) {
122             hits.add(hit);
123         }
124 
125         // Fit a straight line in the non-bending plane and a parabola in the bending plane
126         // Don't allocate space for debugging arrays unless debugging is really happening
127         if (debug) {
128             System.out.format("Entering SeedTrack, yOrigin=%10.7f\n", yOrigin);
129             for (KalHit hit : hitList) hit.print(" in SeedTrack ");
130             xMC = new double[hitList.size()]; // Global x coordinates of measurements (bending plane)
131             zMC = new double[hitList.size()]; // Global z coordinates of measurements (along B field direction)
132             yMC = new double[hitList.size()];
133             mTrue = new double[hitList.size()];
134         }
135         // reallocate the static working arrays only if necessary
136         if (hitList.size() != nHits) {
137             nHits = hitList.size();
138             y = new double[nHits]; // Global y coordinates of measurements (along beam direction)
139             v = new double[nHits]; // Measurement value (i.e. position of the hit strip)
140             s = new double[nHits]; // Uncertainty in the measurement (spatial resolution of the SSD)
141             vpred = new double[nHits];
142             delta = new double[nHits][3];
143             R2 = new double[nHits][3];
144             M = new double[5][5];
145             Mint = new DMatrixRMaj(5);
146             A = new double[5][5];
147             B = new double[5];
148             a = new DMatrixRMaj(5);
149         }
150         
151         int N = 0;
152         Nbending = 0;
153         Nnonbending = 0;
154 
155         // First find the average field
156         Vec Bvec = new Vec(0., 0., 0.);
157         for (KalHit pnt : hitList) {
158             SiModule thisSi = pnt.module;
159             Vec thisB = KalmanInterface.getField(thisSi.toGlobal(new Vec(0., 0., 0.)), thisSi.Bfield);
160             Bvec = Bvec.sum(thisB);
161         }
162         double sF = 1.0 / ((double) nHits);
163         Bvec = Bvec.scale(sF);
164         Bavg = Bvec.mag();
165         if (debug) { System.out.format("*** SeedTrack: nHits=%d, Bavg=%10.5e\n", nHits, Bavg); }
166         double c = 2.99793e8; // Speed of light in m/s
167         alpha = 1000.0 * 1.0e9 / (c * Bavg); // Convert from pt in GeV to curvature in mm
168 
169         N = 0;
170         for (KalHit itr : hitList) {
171             SiModule thisSi = itr.module;
172             if (!thisSi.isStereo) {
173                 Nnonbending++;
174             } else {
175                 Nbending++;
176             }
177             Measurement m = itr.hit;
178             Vec pnt = new Vec(0., m.v, 0.);
179             pnt = thisSi.toGlobal(pnt);
180             if (debug) {
181                 if (thisSi.isStereo) {
182                     System.out.format("Layer %d detector %d, Stereo measurement %d = %10.7f\n", thisSi.Layer, thisSi.detector, N,
183                             m.v);
184                 } else {
185                     System.out.format("Layer %d detector %d, Axial measurement %d = %10.7f\n", thisSi.Layer, thisSi.detector, N,
186                             m.v);
187                 }
188                 pnt.print("point global");
189             }
190             y[N] = pnt.v[1] - yOrigin;
191             if (debug) {
192                 xMC[N] = m.rGlobal.v[0];
193                 yMC[N] = m.rGlobal.v[1] - yOrigin;
194                 zMC[N] = m.rGlobal.v[2];
195                 mTrue[N] = m.vTrue;
196             }
197             v[N] = m.v;
198             s[N] = m.sigma;
199             for (int j = 0; j < 3; j++) {
200                 if (j == 1) {
201                     delta[N][j] = thisSi.p.X().v[j] - yOrigin;
202                 } else {
203                     delta[N][j] = thisSi.p.X().v[j];
204                 }
205                 R2[N][j] = thisSi.Rinv.M[1][j];
206             }
207             N++;
208         }
209         if (Nnonbending < 2) {
210             System.out.format("SeedTrack: not enough points in the non-bending plane; N=%d\n", Nnonbending);
211             success = false;
212             return;
213         }
214         if (Nbending < 3) {
215             System.out.format("SeedTrack: not enough points in the bending plane; N=%d\n", Nbending);
216             success = false;
217             return;
218         }
219         if (debug) {
220             System.out.format("SeedTrack: data in global coordinates: y, yMC, zMC, xMC, m, mTrue, check, sigma, R2[0..2]\n");
221             for (int i = 0; i < N; i++) {
222                 double vcheck = R2[i][0] * (xMC[i] - delta[i][0]) + R2[i][2] * (zMC[i] - delta[i][2]) + R2[i][1] * (yMC[i] - delta[i][1]);
223                 System.out.format("%d  %10.6f  %10.6f  %10.6f  %10.6f   %10.6f   %10.6f   %10.6f   %10.6f %10.7f %10.7f %10.7f\n", i,
224                         y[i], yMC[i], zMC[i], xMC[i], v[i], mTrue[i], vcheck, s[i], R2[i][0], R2[i][1], R2[i][2]);
225             }
226         }
227 
228         // Here we do the 5-parameter linear fit:
229         LinearHelixFit(N, y, v, s, delta, R2);
230         if (debug) printFit(N, xMC, y, zMC, v, s);
231 
232         // Now, derive the helix parameters and covariance from the two fits
233         double sgn = -1.0; // Careful with this sign--->selects the correct half of the circle! B in +z direction only
234         Vec coef = new Vec(a.unsafe_get(2,0), a.unsafe_get(3,0), a.unsafe_get(4,0)); // Parabola coefficients
235 
236         double[] circleParams = parabolaToCircle(sgn, coef);
237         phi0 = circleParams[1];
238         K = circleParams[2];
239         drho = circleParams[0];
240         double dphi0da = (2.0 * coef.v[2] / coef.v[1]) * square(FastMath.sin(phi0));
241         double dphi0db = FastMath.sin(phi0) * FastMath.cos(phi0) / coef.v[1] - square(FastMath.sin(phi0));
242         double dphi0dc = (2.0 * coef.v[0] / coef.v[1]) * square(FastMath.sin(phi0));
243         double temp = xc * FastMath.tan(phi0) / FastMath.cos(phi0);
244 
245         double slope = a.unsafe_get(1,0);
246         double intercept = a.unsafe_get(0,0);
247         tanl = slope * FastMath.cos(phi0);
248         dz = intercept + drho * tanl * FastMath.tan(phi0);
249         if (debug) {
250             System.out.format("SeedTrack: dz=%12.5e   tanl=%12.5e\n", dz, tanl);
251             System.out.format("     R=%10.6f, xc=%10.6f, yc=%10.6f\n", R, xc, yc);
252             System.out.format("     phi0=%10.7f,  K=%10.6f,   drho=%10.6f\n", phi0, K, drho);
253         }
254         // Some derivatives to transform the covariance from line and parabola
255         // coefficients to helix parameters
256         M[0][2] = 1.0 / FastMath.cos(phi0) + temp * dphi0da;
257         M[0][3] = -(coef.v[1] / (2.0 * coef.v[2])) / FastMath.cos(phi0) + temp * dphi0db;
258         M[0][4] = -(sgn + (1.0 - 0.5 * coef.v[1] * coef.v[1]) / FastMath.cos(phi0)) / (2.0 * coef.v[2] * coef.v[2]) + temp * dphi0dc;
259         M[1][2] = dphi0da;
260         M[1][3] = dphi0db;
261         M[1][4] = dphi0dc;
262         M[2][4] = -2.0 * alpha * sgn;
263         M[3][0] = 1.0;
264         M[3][1] = drho * FastMath.sin(phi0);
265         M[4][1] = FastMath.cos(phi0);
266         DMatrixRMaj D = new DMatrixRMaj(M);
267         CommonOps_DDRM.multTransB(CovA,D,Mint);
268         C = new DMatrixRMaj(5);
269         CommonOps_DDRM.mult(D, Mint, C);
270         //boolean badC = false;
271         //for (int i=0; i<5; ++i) {
272         //    if (C.unsafe_get(i, i) < 0.) {
273         //        badC = false;
274         //        System.out.format("SeedTrack: negative covariance matrix found!\n");
275         //        break;
276         //    }
277         //}
278         //if (debug || badC) { 
279         //    System.out.println("line/parabola to helix derivatives and final covariance:");
280         //    D.print();
281         //    C.print();
282         //}
283 
284         // Note that the non-bending plane is assumed to be y,z (B field in z
285         // direction), and the track is assumed to start out more-or-less
286         // in the y direction, so that phi0 should be close to zero. phi0 at 90 degrees
287         // will give trouble here!
288 
289         // Rotate the result into the frame of the B field at the specified origin
290         KalHit itr = hitList.get(0);
291         SiModule firstSi = itr.module;
292         Vec firstB = KalmanInterface.getField(new Vec(0., yOrigin, 0.), firstSi.Bfield);
293         if (Math.abs(firstB.v[1] / firstB.v[2]) > 0.002) {
294             Vec zhat = firstB.unitVec();
295             Vec yhat = new Vec(0., 1., 0.);
296             Vec xhat = (yhat.cross(zhat)).unitVec();
297             yhat = zhat.cross(xhat);
298             RotMatrix Rot = new RotMatrix(xhat, yhat, zhat);
299 
300             hParm = rotateHelix(new Vec(drho, phi0, K, dz, tanl), Rot);
301             if (debug) {
302                 firstB.print("Seedtrack, B field");
303                 Rot.print("Seedtrack, rotation matrix");
304                 hParm.print("Seedtrack, rotated helix");
305             }
306         } else {
307             hParm = new Vec(drho, phi0, K, dz, tanl);
308             if (debug) { hParm.print("Seedtrack, rotated helix"); }
309         }
310 
311         success = true;
312     }

354     private double[] parabolaToCircle(double sgn, Vec coef) { // Utility to convert from parabola coefficients to circle
355                                                               // (i.e. helix)
356                                                               // parameters drho, phi0, and K
357         R = -sgn / (2.0 * coef.v[2]);
358         yc = sgn * R * coef.v[1];
359         xc = coef.v[0] - sgn * R * (1.0 - 0.5 * coef.v[1] * coef.v[1]);
360         double[] r = new double[3];
361         r[1] = FastMath.atan2(yc, xc);
362         if (R < 0.) { r[1] += Math.PI; }
363         if (r[1] > Math.PI) r[1] -= 2.0 * Math.PI;
364         r[2] = alpha / R;
365         r[0] = xc / FastMath.cos(r[1]) - R;
366         if (debug) {
367             System.out.format("parabolaToCircle:     R=%10.6f, xc=%10.6f, yc=%10.6f, drho=%10.7f, phi0=%10.7f, K=%10.7f\n", R, xc, yc, r[0],
368                     r[1], r[2]);
369             coef.print("parabola coefficients");
370             double phi02 = FastMath.atan(-coef.v[1] / (2.0 * coef.v[0] * coef.v[2] + (1.0 - coef.v[1] * coef.v[1])));
371             System.out.format("phi02 = %10.7f\n", phi02);
372         }
373         return r;
374     }

376     // Transformation of a helix from one B-field frame to another, by rotation R
377     private Vec rotateHelix(Vec a, RotMatrix R) {
378         // a = 5 helix parameters
379         // R = 3 by 3 rotation matrix
380         // The rotation is easily applied to the momentum vector, so first we transform
381         // from helix parameters
382         // to momentum, apply the rotation, and then transform back to helix parameters.
383         Vec p_prime = R.rotate(HelixState.aTOp(a));
384         double Q = Math.signum(a.v[2]);
385         Vec a_prime = pTOa(p_prime, Q);
386         return new Vec(a.v[0], a_prime.v[0], a_prime.v[1], a.v[3], a_prime.v[2]);
387     }
388 
389     // Transform from momentum at helix starting point back to the helix parameters
390     private static Vec pTOa(Vec p, Double Q) {
391         double phi0 = FastMath.atan2(-p.v[0], p.v[1]);
392         double K = Q / FastMath.sqrt(p.v[0] * p.v[0] + p.v[1] * p.v[1]);
393         double tanl = p.v[2] / FastMath.sqrt(p.v[0] * p.v[0] + p.v[1] * p.v[1]);
394         if (phi0 > Math.PI) phi0 = phi0 - 2.0 * Math.PI;
395         // Note: the following only makes sense when a.v[0] and a.v[3] (drho and dz) are
396         // both zero, i.e. pivot is on the helix
397         return new Vec(phi0, K, tanl);
398     }

426     //Check if two seeds are compatible given a certain relative threshold
427     boolean isCompatibleTo(SeedTrack st, double rel_eps) {
428         Vec st_hp = st.helixParams();
429         boolean compatible = true;
430         for (int i=0; i<5; i++) {
431             if (Math.abs((st_hp.v[i] - hParm.v[i])/hParm.v[i]) > rel_eps) {
432                 compatible = false;
433                 break;
434             }
435         }
436         return compatible;
437     }

439     Vec planeIntersection(Plane p) {
440         double arg = (K / alpha) * ((drho + (alpha / K)) * FastMath.sin(phi0) - (p.X().v[1] - yOrigin));
441         double phiInt = -phi0 + FastMath.asin(arg);
442         return atPhi(phiInt);
443     }

445     private Vec atPhi(double phi) { // point on the helix at the angle phi
446         double x = (drho + (alpha / K)) * FastMath.cos(phi0) - (alpha / K) * FastMath.cos(phi0 + phi);
447         double y = yOrigin + (drho + (alpha / K)) * FastMath.sin(phi0) - (alpha / K) * FastMath.sin(phi0 + phi);
448         double z = dz - (alpha / K) * phi * tanl;
449         return new Vec(x, y, z);
450     }

452     double[] pivotTransform(double[] pivot) {
453         Vec X0 = new Vec(0., yOrigin, 0.);
454         double xC = X0.v[0] + (drho + alpha / K) * FastMath.cos(phi0); // Center of the helix circle
455         double yC = X0.v[1] + (drho + alpha / K) * FastMath.sin(phi0);
456 
457         // Transformed helix parameters
458         double[] aP = new double[5];
459         aP[2] = K;
460         aP[4] = tanl;
461         if (K > 0) {
462             aP[1] = FastMath.atan2(yC - pivot[1], xC - pivot[0]);
463         } else {
464             aP[1] = FastMath.atan2(pivot[1] - yC, pivot[0] - xC);
465         }
466         aP[0] = (xC - pivot[0]) * FastMath.cos(aP[1]) + (yC - pivot[1]) * FastMath.sin(aP[1]) - alpha / K;
467         aP[3] = X0.v[2] - pivot[2] + dz - (alpha / K) * (aP[1] - phi0) * tanl;
468 
469         return aP;
470     }

472     private void LinearHelixFit(int N, double[] y, double[] v, double[] s, double[][] delta, double[][] R2) {
473         // N = number of measurement points (detector layers) to fit. This must be no less than 5, with at least 2 axial and 2 stereo
474         // y = nominal location of each measurement plane along the beam axis
475         // v = measurement value in the detector coordinate system, perpendicular to the strips
476         // s = one sigma error estimate on each measurement
477         // delta = offset of the detector coordinate system from the global system (minus the nominal y value along the beam axis)
478         // R2 = 2nd row of the general rotation from the global system to the local detector system
479 
480         A[0][0] = 0.;
481         A[0][1] = 0.;
482         A[0][2] = 0.;
483         A[0][3] = 0.;
484         A[0][4] = 0.;
485         A[1][1] = 0.;
486         A[1][4] = 0.;
487         A[2][2] = 0.;
488         A[2][3] = 0.;
489         A[2][4] = 0.;
490         A[3][4] = 0.;
491         A[4][4] = 0.;
492         B[0] = 0.;
493         B[1] = 0.;
494         B[2] = 0.;
495         B[3] = 0.;
496         B[4] = 0.;
497         for (int i = 0; i < N; i++) {
498             double R10 = R2[i][0];
499             double R11 = R2[i][1];
500             double R12 = R2[i][2];
501             double w = 1.0 / (s[i] * s[i]);
502 
503             A[0][0] += w * R12 * R12;
504             A[0][1] += w * y[i] * R12 * R12;
505             A[0][2] += w * R12 * R10;
506             A[0][3] += w * y[i] * R12 * R10;
507             A[0][4] += w * y[i] * y[i] * R12 * R10;
508             A[1][1] += w * y[i] * y[i] * R12 * R12;
509             A[1][4] += w * y[i] * y[i] * y[i] * R12 * R10;
510             A[2][2] += w * R10 * R10;
511             A[2][3] += w * y[i] * R10 * R10;
512             A[2][4] += w * y[i] * y[i] * R10 * R10;
513             A[3][4] += w * y[i] * y[i] * y[i] * R10 * R10;
514             A[4][4] += w * y[i] * y[i] * y[i] * y[i] * R10 * R10;
515             double vcorr = (v[i] + R10 * delta[i][0] + R12 * delta[i][2] + R11 * (delta[i][1] - y[i]));
516             B[0] += vcorr * R12 * w;
517             B[1] += vcorr * y[i] * R12 * w;
518             B[2] += vcorr * R10 * w;
519             B[3] += vcorr * y[i] * R10 * w;
520             B[4] += vcorr * y[i] * y[i] * R10 * w;
521         }
522         A[1][0] = A[0][1];
523         A[1][2] = A[0][3];
524         A[1][3] = A[0][4];
525         A[2][0] = A[0][2];
526         A[2][1] = A[1][2];
527         A[3][0] = A[0][3];
528         A[3][1] = A[1][3];
529         A[3][2] = A[2][3];
530         A[3][3] = A[2][4];
531         A[4][0] = A[0][4];
532         A[4][1] = A[1][4];
533         A[4][2] = A[2][4];
534         A[4][3] = A[3][4];
535 
536         CovA = new DMatrixRMaj(A);
537         DMatrixRMaj V = new DMatrixRMaj(B);
538         
539         if (debug) {
540             System.out.format("SeedTrack:LinearHelixFit, A matrix= ");
541             CovA.print();
542             System.out.format("SeedTrack:LinearHelixFit, B vector= ");
543             V.print();
544         }
545 
546         // Do the calculation
547         CommonOps_DDRM.invert(CovA);
548         CommonOps_DDRM.mult(CovA,V,a);
549 
550         // Add up the chi-squared
551         chi2 = 0.;
552         for (int i = 0; i < N; i++) {
553             double R10 = R2[i][0];
554             double R11 = R2[i][1];
555             double R12 = R2[i][2];
556             vpred[i] = R10 * (evaluateParabola(y[i]) - delta[i][0]) + R12 * (evaluateLine(y[i]) - delta[i][2]) + R11 * (y[i] - delta[i][1]);
557             //   R10 -0.000146  ePyi -4.094654  deltai0 -22.794523  R12 -0.999999 eLyi 22.174849  deltai2 32.839151  R11 -0.001407  yi 912.877681  deltai1 912.892707
558             //System.out.printf("  R10 %f  ePyi %f  deltai0 %f  R12 %f eLyi %f  deltai2 %f  R11 %f  yi %f  deltai1 %f \n", R10, evaluateParabola(y[i]), delta[i][0], R12, evaluateLine(y[i]), delta[i][2], R11, y[i], delta[i][1]);
559             double err = (v[i] - vpred[i]) / s[i];
560             chi2 += err * err;
561         }
562     }
563     
564     private double evaluateLine(double y) {
565         return a.unsafe_get(0,0) + a.unsafe_get(1,0) * y;
566     }
567 
568     private double evaluateParabola(double y) {
569         return a.unsafe_get(2,0) + (a.unsafe_get(3,0) + a.unsafe_get(4,0) * y) * y;
570     }
571     
572     private void printFit(int N, double[] x, double[] y, double[] z, double[] v, double[] s) {
573         System.out.format("LinearHelixFit: parabola a=%10.7f   b=%10.7f   c=%10.7f\n", a.get(2,0), a.get(3,0), a.get(4,0));
574         System.out.format("LinearHelixFit:     line a=%10.7f   b=%10.7f\n", a.get(0,0), a.get(1,0));
575         System.out.println("LinearHelixFit covariance:");
576         CovA.print();
577         System.out.format(
578                 "LinearHelixFit: i  xMC   xpred       y           zMC       zpred        v    v predicted   residual   sigmas       chi^2=%8.3f\n",
579                 chi2);
580         for (int i = 0; i < N; i++) {
581             double xpred = evaluateParabola(y[i]);
582             double zpred = evaluateLine(y[i]);
583             System.out.format("        %d   %10.7f %10.7f %10.7f %10.7f %10.7f %10.7f %10.7f %10.7f %10.7f\n", i, x[i], xpred, y[i], z[i],
584                     zpred, v[i], vpred[i], v[i] - vpred[i], (v[i] - vpred[i]) / s[i]);
585         }
586     }
