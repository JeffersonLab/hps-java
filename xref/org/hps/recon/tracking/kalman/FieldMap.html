14  /**
15   * This class is used for standalone testing of the Kalman code. See KalmanInterface.java for how the B-field access works
16   * when running in hps-java.
17   * Retrieve the magnetic field vector from the HPS field map.
18   * The HPS field is down, in the -y direction in the map coordinates, or the z direction in local Kalman coordinates
19   * The constructor reads the map from a text file or binary file.
20   * The field map is in coordinates different from the Kalman fitter coordinates
21   *     x map =  x Kalman
22   *     y map = -z Kalman
23   *     z map =  y Kalman
24   * These map coordinates are the HPS global coordinates (not HPS tracking coordinates)
25   */
26  public class FieldMap extends FieldOverlay {
27      private int nX, nY, nZ;
28      private double[][][] bX, bY, bZ;
29      private double[] X, Y, Z;
30      private double dX, dY, dZ;
31      private Vec offsets; // Offset of the map coordinates from the HPS coordinates
32      private boolean uniform;
33  
34      public FieldMap(String FileName, String type, boolean uniform, double xOffset, double yOffset, double zOffset) throws IOException {
35          // The offsets are in HPS coordinates and come from HPSDipoleFieldMap3D
36  
37          this.uniform = uniform; // To allow testing the effects of fitting with a uniform field
38          if (uniform) {
39              System.out.println("FieldMap.java: warning, the field map is set to be uniform!");
40          }
41          if (type == "binary") { // This is far faster than scanning the text file (and the file is ~1/3 the size)!
42              FileInputStream ifile = new FileInputStream(FileName);
43              BufferedInputStream bis = new BufferedInputStream(ifile); // This buffering is essential!
44              DataInputStream dis = new DataInputStream(bis);
45              nX = dis.readInt();
46              nY = dis.readInt();
47              nZ = dis.readInt();
48              System.out.format("FieldMap.java, buffered read of binary data: nX=%d, nY=%d, nZ=%d\n", nX, nY, nZ);
49              X = new double[nX];
50              Y = new double[nY];
51              Z = new double[nZ];
52              bX = new double[nX][nY][nZ];
53              bY = new double[nX][nY][nZ];
54              bZ = new double[nX][nY][nZ];
55              int nEcho = 0;
56              for (int ix = 0; ix < nX; ix++) {
57                  for (int iy = 0; iy < nY; iy++) {
58                      for (int iz = 0; iz < nZ; iz++) { // The field map is expected to be uniformly spaced in each
59                                                        // coordinate!
60                          X[ix] = dis.readFloat();
61                          Y[iy] = dis.readFloat();
62                          Z[iz] = dis.readFloat();
63                          bX[ix][iy][iz] = dis.readFloat();
64                          bY[ix][iy][iz] = dis.readFloat();
65                          bZ[ix][iy][iz] = dis.readFloat();
66                          if (nEcho < 10) {
67                              System.out.format("x=%12.4e, y=%12.4e, z=%12.4e, Bx=%12.4e, By=%12.4e, Bz=%12.4e\n", X[ix], Y[iy], Z[iz],
68                                      bX[ix][iy][iz], bY[ix][iy][iz], bZ[ix][iy][iz]);
69                              nEcho++;
70                          }
71                      }
72                  }
73              }
74  
75              dis.close();
76          } else {
77              Scanner scan = new Scanner(new File(FileName));
78              nX = 0;
79              boolean foundEnd = false;
80              while (scan.hasNextLine() && !foundEnd) {
81                  if (scan.hasNextInt() && nX == 0) {
82                      nX = scan.nextInt();
83                      nY = scan.nextInt();
84                      nZ = scan.nextInt();
85                      System.out.format("FieldMap.java, scanning text-format data: nX=%d, nY=%d, nZ=%d\n", nX, nY, nZ);
86                  } else {
87                      if (scan.findInLine("End of Header") != null) {
88                          System.out.format("FieldMap.java: end of header found in the field map file\n");
89                          foundEnd = true;
90                      }
91                      String s = scan.nextLine();
92                      System.out.println(s);
93                  }
94              }
95  
96              X = new double[nX];
97              Y = new double[nY];
98              Z = new double[nZ];
99              bX = new double[nX][nY][nZ];
100             bY = new double[nX][nY][nZ];
101             bZ = new double[nX][nY][nZ];
102             int nEcho = 0;
103             for (int ix = 0; ix < nX; ix++) {
104                 for (int iy = 0; iy < nY; iy++) {
105                     for (int iz = 0; iz < nZ; iz++) {
106                         X[ix] = scan.nextDouble();
107                         Y[iy] = scan.nextDouble();
108                         Z[iz] = scan.nextDouble();
109                         bX[ix][iy][iz] = scan.nextDouble();
110                         bY[ix][iy][iz] = scan.nextDouble();
111                         bZ[ix][iy][iz] = scan.nextDouble();
112                         if (!scan.hasNextLine()) {
113                             System.out.format("FieldMap.java: stopped reading the text field map at i=%d, j=%d, k=%d\n", ix, iy, iz);
114                             break;
115                         }
116                         scan.nextLine();
117                         if (nEcho < 10) {
118                             System.out.format("x=%12.4e, y=%12.4e, z=%12.4e, Bx=%12.4e, By=%12.4e, Bz=%12.4e\n", X[ix], Y[iy], Z[iz],
119                                     bX[ix][iy][iz], bY[ix][iy][iz], bZ[ix][iy][iz]);
120                             nEcho++;
121                         }
122                     }
123                 }
124             }
125 
126             scan.close();
127         }
128         dX = X[1] - X[0];
129         dY = Y[1] - Y[0];
130         dZ = Z[1] - Z[0];
131         System.out.format("FieldMap.java: dX=%10.7f, dY=%10.7f, dZ=%10.7f\n", dX, dY, dZ);
132         offsets = new Vec(xOffset, yOffset, zOffset);
133         offsets.print("field map offsets");
134     }

136     double [] getField(Vec r) { // Interpolate the 3D field map
137         Vec rHPS;
138         if (uniform) {
139             rHPS = new Vec(0., 0., 505.57);
140         } else {
141             rHPS = new Vec(r.v[0], -r.v[2], r.v[1]); // Converting from Kalman coordinates to HPS coordinates
142         }
143         Vec rMag = rHPS.dif(offsets); // Converting to magnet coordinates
144         // rMag.print("rMag original");
145         int iX = (int) Math.floor((rMag.v[0] - X[0]) / dX);
146         if (iX < 0) {
147             iX = 0;
148             rMag.v[0] = X[0];
149         }
150         if (iX > nX - 2) {
151             iX = nX - 2;
152             if (rMag.v[0] > X[nX - 1]) rMag.v[0] = X[nX - 1];
153         }
154         int iY = (int) Math.floor((rMag.v[1] - Y[0]) / dY);
155         if (iY < 0) {
156             iY = 0;
157             rMag.v[1] = Y[0];
158         }
159         if (iY > nY - 2) {
160             iY = nY - 2;
161             if (rMag.v[1] > Y[nY - 1]) rMag.v[1] = Y[nY - 1];
162         }
163         int iZ = (int) Math.floor((rMag.v[2] - Z[0]) / dZ);
164         if (iZ < 0) {
165             iZ = 0;
166             rMag.v[2] = Z[0];
167             if (rMag.v[2] > Z[nZ - 1]) rMag.v[2] = Z[nZ - 1];
168         }
169         if (iZ > nZ - 2) iZ = nZ - 2;
170 
171         // r.print("r");
172         // rHPS.print("rHPS");
173         // rMag.print("rMag");
174         // System.out.format("FieldMap.getField: iX=%d, iY=%d, iZ=%d, X=%10.5f,
175         // Y=%10.5f, Z=%10.5f\n",iX,iY,iZ,X[iX],Y[iY],Z[iZ]);
176         double xd = (rMag.v[0] - X[iX]) / dX;
177         double yd = (rMag.v[1] - Y[iY]) / dY;
178         double zd = (rMag.v[2] - Z[iZ]) / dZ;
179         double [] Bout = new double[3];
180         Bout[0] = triLinear(iX, iY, iZ, xd, yd, zd, bX) * 1000.;  // HPS Bx
181         Bout[2] = -triLinear(iX, iY, iZ, xd, yd, zd, bY) * 1000.; // HPS By
182         Bout[1] = triLinear(iX, iY, iZ, xd, yd, zd, bZ) * 1000.;  // HPS Bz
183         
184         // This transforms the field from the HPS global coordinates to Kalman global coordinates
185         if (uniform) {
186             Bout[0] = 0.; // constant field
187             Bout[1] = 0.;
188         }
189         //Bout[0] = 0.;
190         //Bout[1] = 0.;
191         //Bout[2] = 0.240241;
192         return Bout; 
193     }

195     private double triLinear(int i, int j, int k, double xd, double yd, double zd, double[][][] f) {
196         // System.out.format(" triLinear: xd=%10.7f, yd=%10.7f, zd=%10.7f\n", xd,yd,zd);
197         // System.out.format(" 000=%12.4e, 100=%12.4e\n", f[i][j][k],f[i+1][j][k]);
198         // System.out.format(" 001=%12.4e, 101=%12.4e\n", f[i][j][k+1],f[i+1][j][k+1]);
199         // System.out.format(" 010=%12.4e, 110=%12.4e\n", f[i][j+1][k],f[i+1][j+1][k]);
200         // System.out.format(" 011=%12.4e, 111=%12.4e\n",
201         // f[i][j+1][k+1],f[i+1][j+1][k+1]);
202         double c00 = f[i][j][k] * (1.0 - xd) + f[i + 1][j][k] * xd; // interpolate in x
203         double c01 = f[i][j][k + 1] * (1.0 - xd) + f[i + 1][j][k + 1] * xd;
204         double c10 = f[i][j + 1][k] * (1.0 - xd) + f[i + 1][j + 1][k] * xd;
205         double c11 = f[i][j + 1][k + 1] * (1.0 - xd) + f[i + 1][j + 1][k + 1] * xd;
206         double c0 = c00 * (1.0 - yd) + c10 * yd; // interpolate in y
207         double c1 = c01 * (1.0 - yd) + c11 * yd;
208         double c = c0 * (1.0 - zd) + c1 * zd; // interpolate in z
209         // System.out.format(" c=%12.4e\n", c);
210         return c;
211     }

213     void writeBinaryFile(String fName) { // Make a binary field map file that can be read much more quickly
214         FileOutputStream ofile;
215         try {
216             ofile = new FileOutputStream(fName);
217             DataOutputStream dos = new DataOutputStream(ofile);
218             dos.writeInt(nX);
219             dos.writeInt(nY);
220             dos.writeInt(nZ);
221             for (int i = 0; i < nX; i++) {
222                 for (int j = 0; j < nY; j++) {
223                     for (int k = 0; k < nZ; k++) {
224                         dos.writeFloat((float) X[i]);
225                         dos.writeFloat((float) Y[j]);
226                         dos.writeFloat((float) Z[k]);
227                         dos.writeFloat((float) bX[i][j][k]);
228                         dos.writeFloat((float) bY[i][j][k]);
229                         dos.writeFloat((float) bZ[i][j][k]);
230                     }
231                 }
232             }
233 
234             dos.close();
235         } catch (IOException e) {
236             System.out.println("FieldMap.writeBinaryFile: IOException : " + e);
237         }
238     }
