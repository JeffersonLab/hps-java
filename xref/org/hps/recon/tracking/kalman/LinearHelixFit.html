1   package org.hps.recon.tracking.kalman;
2   
3   import org.ejml.data.DMatrixRMaj;
4   import org.ejml.dense.row.CommonOps_DDRM;
5   
6   /**
7    * Simultaneous fit of axial and stereo measurements to a line in the non-bending plane
8    * and a parabola in the bending plane
9    * Deprecated. The LinearHelixFit code was integrated into SeedTrack as a method of that class, for better efficiency.
10   */
11  class LinearHelixFit { // 
12  
13      private DMatrixRMaj  C; // Covariance matrix of solution
14      private DMatrixRMaj  a; // Solution vector (line coefficients followed by parabola coefficients
15      private double chi2;
16      private double[] vpred;
17  
18      LinearHelixFit(int N, double[] y, double[] v, double[] s, double[][] delta, double[][] R2, boolean verbose) {
19          // N = number of measurement points (detector layers) to fit. This must be no less than 5, with at least 2 axial and 2 stereo
20          // y = nominal location of each measurement plane along the beam axis
21          // v = measurement value in the detector coordinate system, perpendicular to the strips
22          // s = one sigma error estimate on each measurement
23          // delta = offset of the detector coordinate system from the global system (minus the nominal y value along the beam axis)
24          // R2 = 2nd row of the general rotation from the global system to the local detector system
25          // verbose: set true to get lots of debug printout
26          double [][] A = new double[5][5];
27          double [] B = new double[5];
28  
29          for (int i = 0; i < N; i++) {
30              double R10 = R2[i][0];
31              double R11 = R2[i][1];
32              double R12 = R2[i][2];
33              double w = 1.0 / (s[i] * s[i]);
34  
35              A[0][0] += w * R12 * R12;
36              A[0][1] += w * y[i] * R12 * R12;
37              A[0][2] += w * R12 * R10;
38              A[0][3] += w * y[i] * R12 * R10;
39              A[0][4] += w * y[i] * y[i] * R12 * R10;
40              A[1][1] += w * y[i] * y[i] * R12 * R12;
41              A[1][4] += w * y[i] * y[i] * y[i] * R12 * R10;
42              A[2][2] += w * R10 * R10;
43              A[2][3] += w * y[i] * R10 * R10;
44              A[2][4] += w * y[i] * y[i] * R10 * R10;
45              A[3][4] += w * y[i] * y[i] * y[i] * R10 * R10;
46              A[4][4] += w * y[i] * y[i] * y[i] * y[i] * R10 * R10;
47              double vcorr = (v[i] + R10 * delta[i][0] + R12 * delta[i][2] + R11 * (delta[i][1] - y[i]));
48              B[0] += vcorr * R12 * w;
49              B[1] += vcorr * y[i] * R12 * w;
50              B[2] += vcorr * R10 * w;
51              B[3] += vcorr * y[i] * R10 * w;
52              B[4] += vcorr * y[i] * y[i] * R10 * w;
53          }
54          A[1][0] = A[0][1];
55          A[1][2] = A[0][3];
56          A[1][3] = A[0][4];
57          A[2][0] = A[0][2];
58          A[2][1] = A[1][2];
59          A[3][0] = A[0][3];
60          A[3][1] = A[1][3];
61          A[3][2] = A[2][3];
62          A[3][3] = A[2][4];
63          A[4][0] = A[0][4];
64          A[4][1] = A[1][4];
65          A[4][2] = A[2][4];
66          A[4][3] = A[3][4];
67  
68          C = new DMatrixRMaj(A);
69          DMatrixRMaj V = new DMatrixRMaj(B);
70          
71          if (verbose) C.print();
72  
73          // Do the calculation
74          CommonOps_DDRM.invert(C);
75          a = new DMatrixRMaj(5);
76          CommonOps_DDRM.mult(C,V,a);
77  
78          // Add up the chi-squared
79          vpred = new double[N];
80          chi2 = 0.;
81          for (int i = 0; i < N; i++) {
82              double R10 = R2[i][0];
83              double R11 = R2[i][1];
84              double R12 = R2[i][2];
85              vpred[i] = R10 * (evaluateParabola(y[i]) - delta[i][0]) + R12 * (evaluateLine(y[i]) - delta[i][2]) + R11 * (y[i] - delta[i][1]);
86              //   R10 -0.000146  ePyi -4.094654  deltai0 -22.794523  R12 -0.999999 eLyi 22.174849  deltai2 32.839151  R11 -0.001407  yi 912.877681  deltai1 912.892707
87              //System.out.printf("  R10 %f  ePyi %f  deltai0 %f  R12 %f eLyi %f  deltai2 %f  R11 %f  yi %f  deltai1 %f \n", R10, evaluateParabola(y[i]), delta[i][0], R12, evaluateLine(y[i]), delta[i][2], R11, y[i], delta[i][1]);
88              double err = (v[i] - vpred[i]) / s[i];
89              chi2 += err * err;
90          }
91      }
92  
93      void print(int N, double[] x, double[] y, double[] z, double[] v, double[] s) {
94          System.out.format("LinearHelixFit2: parabola a=%10.7f   b=%10.7f   c=%10.7f\n", a.get(2,0), a.get(3,0), a.get(4,0));
95          System.out.format("LinearHelixFit2:     line a=%10.7f   b=%10.7f\n", a.get(0,0), a.get(1,0));
96          System.out.println("LinearHelixFit2 covariance:");
97          C.print();
98          System.out.format(
99                  "LinearHelixFit2: i  xMC   xpred       y           zMC       zpred        v    v predicted   residual   sigmas       chi^2=%8.3f\n",
100                 chi2);
101         for (int i = 0; i < N; i++) {
102             double xpred = evaluateParabola(y[i]);
103             double zpred = evaluateLine(y[i]);
104             System.out.format("        %d   %10.7f %10.7f %10.7f %10.7f %10.7f %10.7f %10.7f %10.7f %10.7f\n", i, x[i], xpred, y[i], z[i],
105                     zpred, v[i], vpred[i], v[i] - vpred[i], (v[i] - vpred[i]) / s[i]);
106         }
107     }
108 
109     DMatrixRMaj covariance() {
110         return C;
111     }
112 
113     Vec solution() {
114         Vec aV = new Vec(a.unsafe_get(0,0),a.unsafe_get(1,0),a.unsafe_get(2,0),a.unsafe_get(3,0),a.unsafe_get(4,0));
115         return aV;
116     }
117 
118     double evaluateLine(double y) {
119         return a.unsafe_get(0,0) + a.unsafe_get(1,0) * y;
120     }
121 
122     double evaluateParabola(double y) {
123         return a.unsafe_get(2,0) + (a.unsafe_get(3,0) + a.unsafe_get(4,0) * y) * y;
124     }
125 
126     double chiSquared() {
127         return chi2;
128     }
129 }
