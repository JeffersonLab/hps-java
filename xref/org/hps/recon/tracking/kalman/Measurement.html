1   package org.hps.recon.tracking.kalman;
2   
3   import java.util.ArrayList;
4   
5   /**
6    * Holds a single silicon-strip measurement (single-sided), to interface with the Kalman fit
7    */
8   class Measurement { // 
9       double v; // Measurement value in detector frame
10      double x; // X of the center of the strip in the detector frame
11      double sigma; // Measurement uncertainty
12      double time;  // Time of the hit in ns
13      double energy; // Energy deposited in the silicon
14      double vTrue; // MC truth measurement value
15      Vec rGlobal; // Global MC truth
16      ArrayList<KalTrack> tracks;     // Tracks that this hit lies on
17      ArrayList<Integer> tksMC;       // MC tracks that contributed to this hit
18  
19      Measurement(double value, double xStrip, double resolution, double t, double E) {
20          v = value;
21          x = xStrip;
22          sigma = resolution;
23          time = t;
24          energy = E;
25          tracks = new ArrayList<KalTrack>();
26          vTrue = 0.;
27          rGlobal = null;
28          tksMC = null;
29      }
30      
31      Measurement(double value, double xStrip, double resolution, double t, double E, Vec rGlobal, double vTrue) {
32          v = value;
33          x = xStrip;
34          sigma = resolution;
35          time = t;
36          energy = E;
37          this.rGlobal = rGlobal;
38          this.vTrue = vTrue;
39          tracks = new ArrayList<KalTrack>();
40          tksMC = new ArrayList<Integer>();
41      }
42      
43      void addMC(int idx) {
44          tksMC.add(idx);
45      }
46  
47      void print(String s) {
48          System.out.format("Measurement %s: Measurement value=%10.5f+-%8.6f; xStrip=%7.2f, MC truth=%10.5f; t=%8.3f; E=%8.3f", s, v, sigma, x, vTrue, time, energy);
49          if (tracks.size() == 0) {
50              System.out.format("  Not on any track.\n");
51          } else {
52              System.out.format("  Tracks: ");
53              for (KalTrack tk : tracks) System.out.format(" %d ", tk.ID);
54              System.out.format("\n");
55          }
56          if (rGlobal != null) rGlobal.print("global location from MC truth"); 
57      }
58      
59      String toString(String s) {
60          String str = String.format("Measurement %s: Measurement value=%10.5f+-%8.6f; xStrip=%7.2f, MC truth=%10.5f; t=%8.3f; E=%8.3f", s, v, sigma, x, vTrue, time, energy);
61          if (tracks.size() == 0) {
62              str = str + String.format("  Not on any track.\n");
63          } else {
64              str = str + String.format("  Tracks: ");
65              for (KalTrack tk : tracks) str = str + String.format(" %d ", tk.ID);
66              str = str + String.format("\n");
67          }
68          if (rGlobal != null) str = str + rGlobal.toString("global location from MC truth"); 
69          return str;
70      }
71  }
