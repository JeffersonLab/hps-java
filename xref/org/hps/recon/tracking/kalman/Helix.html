4   /**
5    * This is for stand-alone testing only and is not part of the Kalman fitting code.
6    * Create a simple helix oriented along the B field axis for testing the Kalman fit.

8   class Helix { 
9       Vec p; // Helix parameters drho, phi0, K, dz, tanl
10      Vec X0; // Pivot point in the B field reference frame
11      private double alpha;
12      private double B; // Magnetic field magnitude
13      private Vec tB; // Magnetic field direction in the global system
14      private Vec uB;
15      private Vec vB;
16      private double rho;
17      private double Q;
18      private double radLen;
19      private HelixPlaneIntersect hpi;
20      RotMatrix R; // Rotation from global coordinates to the B field frame
21      Vec origin; // Origin of the B-field reference frame in global coordinates
22      private FieldMap fM;
23      private Random rndm;

25      // Construct a helix starting from a momentum vector
26      Helix(double Q, Vec Xinit, Vec Pinit, Vec origin, FieldMap fM, Random rndm) {
27          this.Q = Q;
28          this.origin = origin.copy();
29          this.fM = fM;
30          Vec Bf = new Vec(3,fM.getField(Xinit));
31          B = Bf.mag();
32          double c = 2.99793e8;
33          alpha = 1000.0 * 1.0E9 / (c * B); // Units are Tesla, mm, GeV
34          rho = 2.329; // Density of silicon in g/cm^2
35          tB = Bf.unitVec(B);
36          Vec yhat = new Vec(0., 1.0, 0.);
37          uB = yhat.cross(tB).unitVec();
38          vB = tB.cross(uB);
39          R = new RotMatrix(uB, vB, tB);
40          X0 = R.rotate(Xinit.dif(origin));
41          hpi = new HelixPlaneIntersect();
42          this.rndm = rndm;
43          radLen = (21.82 / rho) * 10.0; // Radiation length of silicon in millimeters
44          double Pmag = Pinit.mag();
45          Vec tnew = R.rotate(Pinit.unitVec(Pmag));
46          //tnew.print("Helix constructor tnew");
47          double tanl = tnew.v[2] / Math.sqrt(1.0 - tnew.v[2] * tnew.v[2]);
48          double pt = Pmag / Math.sqrt(1.0 + tanl * tanl);
49          double K = Q / pt;
50          double phi0 = Math.atan2(-tnew.v[0], tnew.v[1]);
51          //System.out.format("    Helix constructor pt=%10.5f, tanl=%10.5f, phi0=%10.6f\n", pt, tanl, phi0);
52          p = new Vec(0., phi0, K, 0., tanl); // Pivot point is on the helix, so drho and dz are zero
53      }

55      // Construct a helix from given helix parameters (given in B field frame)
56      Helix(Vec HelixParams, Vec pivotGlobal, Vec origin, FieldMap fM, Random rndm) {
57          this.origin = origin.copy();
58          this.fM = fM;
59          p = HelixParams.copy();
60          Vec Bf = new Vec(3,fM.getField(pivotGlobal));
61          B = Bf.mag();
62          double c = 2.99793e8;
63          alpha = 1000.0 * 1.0E9 / (c * B); // Units are Tesla, mm, GeV
64          rho = 2.329; // Density of silicon in g/cm^2
65          Q = Math.signum(p.v[2]);
66          radLen = (21.82 / rho) * 10.0; // Radiation length of silicon in millimeters
67          // System.out.format("helix.java: new helix with radius = %10.2f mm.\n",
68          // alpha/p[2]);
69  
70          // HelixParams.print("helix constructor helix parameters");
71          // X0.print("Helix constructor: pivot in B-field frame");
72          // Bf.print("Helix constructor: B field");
73          tB = Bf.unitVec(B);
74          Vec yhat = new Vec(0., 1.0, 0.);
75          uB = yhat.cross(tB).unitVec();
76          vB = tB.cross(uB);
77          R = new RotMatrix(uB, vB, tB);
78          // R.print("new helix rotation matrix");
79          X0 = R.rotate(pivotGlobal.dif(origin));
80          hpi = new HelixPlaneIntersect();
81          // origin.print("new helix origin");
82          // pivotGlobal.print("new helix pivot global");
83          // X0.print("new helix pivot");
84          this.rndm = rndm;
85      }
86  
87      Helix copy() {
88          return new Helix(p, R.inverseRotate(X0).sum(origin), origin, fM, rndm);
89      }

130     Vec atPhi(double phi) { // return the local coordinates on the helix at a particular phi value
131         double x = X0.v[0] + (p.v[0] + (alpha / (p.v[2]))) * Math.cos(p.v[1]) - (alpha / (p.v[2])) * Math.cos(p.v[1] + phi);
132         double y = X0.v[1] + (p.v[0] + (alpha / (p.v[2]))) * Math.sin(p.v[1]) - (alpha / (p.v[2])) * Math.sin(p.v[1] + phi);
133         double z = X0.v[2] + p.v[3] - (alpha / (p.v[2])) * phi * p.v[4];
134         return new Vec(x, y, z);
135     }

137     Vec atPhiGlobal(double phi) { // return the global coordinates on the helix at a particular phi value
138         double x = X0.v[0] + (p.v[0] + (alpha / (p.v[2]))) * Math.cos(p.v[1]) - (alpha / (p.v[2])) * Math.cos(p.v[1] + phi);
139         double y = X0.v[1] + (p.v[0] + (alpha / (p.v[2]))) * Math.sin(p.v[1]) - (alpha / (p.v[2])) * Math.sin(p.v[1] + phi);
140         double z = X0.v[2] + p.v[3] - (alpha / (p.v[2])) * phi * p.v[4];
141         return R.inverseRotate(new Vec(x, y, z)).sum(origin);
142     }

144     double planeIntersect(Plane Pin) { // phi value where the helix intersects the plane P (given in global
145                                        // coordinates)
146         Plane P = Pin.toLocal(R, origin);
147         double phi = hpi.planeIntersect(p, X0, alpha, P);
148         // System.out.format("Helix:planeIntersect: phi = %12.10f\n", phi);
149         return phi;
150     }

152     Vec getMom(double phi) { // get the particle momentum vector at a particular phi value
153         double px = -Math.sin(p.v[1] + phi) / Math.abs(p.v[2]);
154         double py = Math.cos(p.v[1] + phi) / Math.abs(p.v[2]);
155         double pz = p.v[4] / Math.abs(p.v[2]);
156         return new Vec(px, py, pz);
157     }

159     Vec getMomGlobal(double phi) { // get the particle momentum vector at a particular phi value
160         double px = -Math.sin(p.v[1] + phi) / Math.abs(p.v[2]);
161         double py = Math.cos(p.v[1] + phi) / Math.abs(p.v[2]);
162         double pz = p.v[4] / Math.abs(p.v[2]);
163         return R.inverseRotate(new Vec(px, py, pz));
164     }

166     Helix randomScat(Plane P, Vec r, Vec pmom, double X) { // Produce a new helix scattered randomly in a given plane P
167         // X is the thickness of the silicon material in meters
168         // r is the intersection point and pmom the momentum at that point
169         //double phi = this.planeIntersect(P); // Here the plane P is assumed to be given in global coordinates
170         // p.print("randomScat: helix parameters before scatter");
171 
172         // Vec r = this.atPhiGlobal(phi);
173         //double tst = r.dif(P.X()).dot(P.T());
174         // System.out.format("randomScat: test dot product %12.6e should be zero\n", tst);
175 
176         // r.print("randomScat: r global");
177         // Vec pmom = getMomGlobal(phi);
178         // pmom.print("randomScat: p global");
179         Vec t = pmom.unitVec();
180         // System.out.format("randomScat: original direction in global coordinates=%10.7f, %10.7f, %10.7f\n", t.v[0],t.v[1],t.v[2]);
181         Vec zhat = new Vec(0., 0., 1.);
182         Vec uhat = t.cross(zhat).unitVec(); // A unit vector u perpendicular to the helix direction
183         Vec vhat = t.cross(uhat);
184         RotMatrix Rp = new RotMatrix(uhat, vhat, t);
185         // t.print("initial helix direction in Helix.randomScat");
186         // Rp.print("rotation matrix in Helix.randomScat");
187         double ct = Math.abs(P.T().dot(t));
188         double theta0;
189         // Get the scattering angle
190         if (X == 0.) theta0 = 0.;
191         else theta0 = Math.sqrt((X / radLen) / ct) * (0.0136 / pmom.mag()) * (1.0 + 0.038 * Math.log((X / radLen) / ct));
192         double thetaX = rndm.nextGaussian() * theta0;
193         double thetaY = rndm.nextGaussian() * theta0;
194         // System.out.format("Helix.randomScat: X=%12.5e, ct=%12.5e, theta0=%12.5e, thetaX=%12.5e,
195         // thetaY=%12.5e\n",X,ct,theta0,thetaX,thetaY);
196         double tx = Math.sin(thetaX);
197         double ty = Math.sin(thetaY);
198         Vec tLoc = new Vec(tx, ty, Math.sqrt(1.0 - tx * tx - ty * ty));
199         // tLoc.print("tLoc in Helix.randomScat");
200         Vec tnew = Rp.inverseRotate(tLoc);
201         // tnew.print("tnew in Helix.randomScat");
202         // System.out.format("tnew dot tnew= %14.10f\n", tnew.dot(tnew));
203         // System.out.format("t dot tnew= %14.10f\n", t.dot(tnew));
204         // double check = Math.acos(Math.min(t.dot(tnew), 1.));
205         // System.out.format("recalculated scattered angle=%10.7f\n", check);
206 
207         // Rotate the direction into the frame of the new field (evaluated at the new pivot)
208         Vec Bf = new Vec(3,fM.getField(r));
209         double Bnew = Bf.mag();
210         Vec tBnew = Bf.unitVec(Bnew);
211         Vec yhat = new Vec(0., 1., 0.);
212         Vec uBnew = yhat.cross(tBnew).unitVec();
213         Vec vBnew = tBnew.cross(uBnew);
214         RotMatrix RB = new RotMatrix(uBnew, vBnew, tBnew);
215         // RB.print("randomscat: field rotation matrix");
216         tnew = RB.rotate(tnew);
217 
218         double E = pmom.mag(); // Everything is assumed electron
219         double sp = 0.0; // 0.002; // Estar collision stopping power for electrons in silicon at about a
220                          // GeV, in GeV cm2/g
221         double dEdx = 0.1 * sp * rho; // in GeV/mm
222         double eLoss = dEdx * X / ct;
223         // System.out.format("randomScat: energy=%10.7f, energy loss=%10.7f\n", E, eLoss);
224         E = E - eLoss;
225         double tanl = tnew.v[2] / Math.sqrt(1.0 - tnew.v[2] * tnew.v[2]);
226         double pt = E / Math.sqrt(1.0 + tanl * tanl);
227         double K = Q / pt;
228         double phi0 = Math.atan2(-tnew.v[0], tnew.v[1]);
229         Vec H = new Vec(0., phi0, K, 0., tanl); // Pivot point is on the helix, at the plane intersection point, so drho and dz are zero
230         // H.print("scattered helix parameters");
231 
232         return new Helix(H, r, P.X(), fM, rndm); // Create the new helix with new origin and pivot point
233     }

235     Vec pivotTransform(Vec pivot) {
236         double xC = X0.v[0] + (p.v[0] + alpha / p.v[2]) * Math.cos(p.v[1]); // Center of the helix circle
237         double yC = X0.v[1] + (p.v[0] + alpha / p.v[2]) * Math.sin(p.v[1]);
238         // System.out.format("pivotTransform center=%10.6f, %10.6f\n", xC, yC);
239 
240         // Predicted state vector
241         double[] aP = new double[5];
242         aP[2] = p.v[2];
243         aP[4] = p.v[4];
244         if (p.v[2] > 0) {
245             aP[1] = Math.atan2(yC - pivot.v[1], xC - pivot.v[0]);
246         } else {
247             aP[1] = Math.atan2(pivot.v[1] - yC, pivot.v[0] - xC);
248         }
249         aP[0] = (xC - pivot.v[0]) * Math.cos(aP[1]) + (yC - pivot.v[1]) * Math.sin(aP[1]) - alpha / p.v[2];
250         aP[3] = X0.v[2] - pivot.v[2] + p.v[3] - (alpha / p.v[2]) * (aP[1] - p.v[1]) * p.v[4];
251 
252         return new Vec(5, aP);
253     }
