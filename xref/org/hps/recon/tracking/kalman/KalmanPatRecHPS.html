21  /**
22   * Tracking pattern recognition for the HPS experiment based on an extended Kalman filter
23   * Algorithm:
24   *    1. Loop over starting strategies, each involving at least 3 stereo planes and 2 non-stereo planes
25   *    2. Make all possible seeds by selecting a hit from each of the 5 planes
26   *    3. Do a linear helix fit to the 5 hits in each seed and select helices that project near to the target origin
27   *    4. Sort the seeds by quality and then loop over them and use them to start the Kalman filter working outward toward the ECAL
28   *         Smooth back to the beginning of the seed and then filter inward toward the vertex to make a candidate track.
29   *         The seed hits always remain on the track candidate.
30   *         On other layers the track candidate picks up the best fitting hit.
31   *         Hits not from the seed can be dropped to try to get a good fit.
32   *         Only decent quality candidates are kept.
33   *         Don't keep candidates that are identical in hit content to candidates already found
34   *    5. Sort all the track candidates by quality
35   *    6. Remove hits from track candidates that are used by better candidates, unless the hits can be shared
36   *    7. Track candidates without enough hits remaining get dropped, the others get refit
37   */

66      KalmanPatRecHPS(KalmanParams kPar) {
67          startTime = (long)0.;
68          this.kPar = kPar;     
69          logger = Logger.getLogger(KalmanPatRecHPS.class.getName());
70          logger.info("KalmanPatRecHPS instance created.");
71          vtx = new double[3];
72          vtxCov = new double[3][3];
73          for (int i=0; i<3; ++i) {
74              vtx[i] = kPar.beamSpot[i];
75              vtxCov[i][i] = kPar.vtxSize[i]*kPar.vtxSize[i];
76          }
77          rho = 2.329;                   // Density of silicon in g/cm^2
78          radLen = (21.82 / rho) * 10.0; // Radiation length of silicon in millimeters
79          firstLayer = kPar.firstLayer;
80          
81          // Some arrays for making preliminary cuts on seeds
82          zh = new double[5];    // global z coordinate of axial hit
83          yh = new double[5];    // global y coordinate of axial hit
84          ah = new boolean[5];   // true if stereo layer, false for axial
85          
86          // Arrays for covariance matrix propagation
87          yScat = new ArrayList<Double>(KalmanParams.numLayers);  // List of approx. y locations where scattering in Si occurs
88          XLscat = new ArrayList<Double>(KalmanParams.numLayers); // Thicknesses of the scattering layers, in radiation lengths
89          
90          p0 = new Plane(new Vec(0., kPar.beamSpot[1], 0.), new Vec(0., 1., 0.));  // xy plane at the target position
91          candidateList = new ArrayList<TrackCandidate>();
92          eventNumber = 0;
93      }

101     ArrayList<KalTrack> kalmanPatRec(EventHeader event, Map<Measurement, TrackerHit> hitMapHPS, ArrayList<SiModule> data, int topBottom) {
102         // topBottom = 0 for the bottom tracker (z>0); 1 for the top tracker (z<0)
103         
104         if (event != null) eventNumber = event.getEventNumber();      
105         else eventNumber++;
106         //debug = (eventNumber == 16276);
107 
108         if (debug) startTime = System.nanoTime();
109         int nCandHits = 0;        
110         TkrList = new ArrayList<KalTrack>();
111         nModules = data.size();
112         int tkID = 100*topBottom + 1;
113         
114         // Make a list of hits for each tracker layer, 0 through 13
115         // This is needed because one layer can have multiple SiModules
116         // Also make a list of Si modules with hits in each layer
117         lyrHits = new ArrayList<ArrayList<KalHit>>(KalmanParams.numLayers);
118         moduleList = new ArrayList<ArrayList<SiModule>>(KalmanParams.numLayers);
119         for (int lyr = 0; lyr < KalmanParams.numLayers; lyr++) {
120             ArrayList<KalHit> hits = new ArrayList<KalHit>();
121             ArrayList<SiModule> modules = new ArrayList<SiModule>();
122             lyrHits.add(hits);
123             moduleList.add(modules);
124         }
125         hitMap = new HashMap<Measurement, KalHit>();
126         for (SiModule thisSi : data) {
127             for (Measurement m : thisSi.hits) {
128                 KalHit hit = new KalHit(thisSi, m);  // Create exactly one KalHit object for each measurement!
129                 lyrHits.get(thisSi.Layer).add(hit);
130                 hitMap.put(m, hit);
131             }
132             moduleList.get(thisSi.Layer).add(thisSi);
133         }
134 
135         if (debug) {
136             double runTime = (double)((System.nanoTime() - startTime)/1000000.);
137             System.out.format("Entering KalmanPatRecHPS for event %d, top-bottom=%d with %d modules, for %d trials at time=%10.6f ms.\n", 
138                          eventNumber, topBottom, nModules, kPar.nTrials, runTime);
139         }
140  
141         if (debug) {
142             for (int lyr = 0; lyr < KalmanParams.numLayers; lyr++) {
143                 System.out.format("Layer %d modules:  ", lyr);
144                 for (SiModule thisSi : moduleList.get(lyr)) {
145                     System.out.format("(det=%d #hits=%d loc= %5.1f %5.1f %5.1f) ", thisSi.detector, thisSi.hits.size(), thisSi.p.X().v[0], thisSi.p.X().v[1], thisSi.p.X().v[2]);
146                 }
147                 System.out.format("\n");
148             }
149         }
150         if (yScat.size() != KalmanParams.numLayers) {
151             yScat.clear();
152             XLscat.clear();
153             for (int lyr = 0; lyr < KalmanParams.numLayers; lyr++) {
154                 if (moduleList.get(lyr).size() > 0) {
155                     SiModule thisSi = moduleList.get(lyr).get(0);
156                     yScat.add(thisSi.p.X().v[1]);
157                     XLscat.add(thisSi.thickness/radLen);
158                 }
159             }
160         }
161 
162         if (debug) {
163             System.out.format("  KalmanPatRecHPS: list of the seed strategies to be applied:\n");
164             for (int[] list : kPar.lyrList[topBottom]) {
165                 for (int lyr=0; lyr<list.length; ++lyr) {
166                     System.out.format(" %3d ", list[lyr]);
167                 }
168                 System.out.format("\n");
169             }
170             System.out.format("    Layer types: ");
171             for (SiModule module : data) {
172                 if (module.isStereo) System.out.format(" %d=S ",module.Layer);
173                 else System.out.format(" %d=A ",module.Layer);
174             }
175             System.out.format("\n");
176         }
177 
178         // Loop over seed strategies, each with 2 non-stereo layers and 3 stereo layers
179         // For each strategy generate a seed track for every hit combination
180         // Keep only those pointing more-or-less back to the origin and not too curved
181         for (int trial = KalmanParams.mxTrials - kPar.nTrials; trial < KalmanParams.mxTrials; trial++) {
182             int candID = topBottom*1000 + trial*100 + 1;
183             if (debug) {
184                 double runTime = (double)((System.nanoTime() - startTime)/1000000.);
185                 System.out.format("\nKalmanPatRecHPS: start of pass %d through the algorithm. Time %10.6f ms\n", trial, runTime);
186                 for (int lyr = 0; lyr < lyrHits.size(); ++lyr) {
187                     ArrayList<KalHit> LL = lyrHits.get(lyr);
188                     System.out.format("KalmanPatRecHPS: layer %d hits:", lyr);
189                     for (KalHit ht : LL) {
190                         ht.print("short");
191                     }
192                     System.out.format("\n");
193                 }
194             }
195             
196             // Remove references from the hits to the track candidates of the previous iteration
197             if (trial != 0) {
198                 for (int lyr = 0; lyr < KalmanParams.numLayers; lyr++) {
199                     for (KalHit ht : lyrHits.get(lyr)) {
200                         ht.tkrCandidates.clear();
201                     }
202                 }
203             }
204             candidateList.clear();
205             for (int iList = 0; iList<kPar.lyrList[topBottom].size(); ++iList) {
206                 if (trial == 0 && iList > kPar.maxListIter1) break;
207                 int[] list = kPar.lyrList[topBottom].get(iList);
208                 int nLyrs = list.length;
209                 int middleLyr = 2;
210                 if (moduleList.get(list[middleLyr]).size() == 0) continue;  // Skip seed if there is no hit in the middle layer
211                 SiModule m0 = moduleList.get(list[middleLyr]).get(0);
212                 double yOrigin = m0.p.X().v[1];                     // Set the local origin to be in the middle of the seed list
213                 Vec pivot = new Vec(0, yOrigin, 0.);
214                 if (debug) {
215                     double runTime = (double)((System.nanoTime() - startTime)/1000000.);
216                     System.out.format("\n\nKalmanPatRecHPS: layer list=%d %d %d %d %d, time=%10.6f ms\n", list[0], list[1], list[2], list[3], list[4],runTime);
217                     System.out.format("KalmanPatRecHPS: yOrigin=%10.6f\n", yOrigin);
218                 }
219                 ArrayList<SeedTrack> seedList = new ArrayList<SeedTrack>();
220                 int[] idx = new int[nLyrs];
221                 for (idx[0] = 0; idx[0] < lyrHits.get(list[0]).size(); idx[0]++) {
222                     KalHit kht = lyrHits.get(list[0]).get(idx[0]);
223                     Measurement hit = kht.hit;
224                     if (hit.tracks.size() > 0) continue; // don't use hits already on KalTrack tracks
225                     if (hit.energy < kPar.minSeedE[kht.module.Layer]) continue; // don't use hits with small energy to make seeds
226                     SiModule mod = kht.module;
227                     ah[0] = mod.isStereo;
228                     if (!mod.isStereo) {
229                         zh[0] = mod.toGlobal(new Vec(0.,hit.v,0.)).v[2];
230                         yh[0] = mod.p.X().v[1];
231                     }                 
232                     for (idx[1] = 0; idx[1] < lyrHits.get(list[1]).size(); idx[1]++) {
233                         kht = lyrHits.get(list[1]).get(idx[1]);
234                         hit = kht.hit;
235                         if (hit.tracks.size() > 0) continue;
236                         if (hit.energy < kPar.minSeedE[kht.module.Layer]) continue;
237                         mod = kht.module;
238                         ah[1] = mod.isStereo;
239                         if (!mod.isStereo) {
240                             zh[1] = mod.toGlobal(new Vec(0.,hit.v,0.)).v[2];
241                             yh[1] = mod.p.X().v[1];
242                             if (seedNoGood(1, trial)) continue;
243                         }
244                         for (idx[2] = 0; idx[2] < lyrHits.get(list[2]).size(); idx[2]++) {
245                             kht = lyrHits.get(list[2]).get(idx[2]);
246                             hit = kht.hit;
247                             if (hit.tracks.size() > 0) continue;
248                             if (hit.energy < kPar.minSeedE[kht.module.Layer]) continue;
249                             mod = kht.module;
250                             ah[2] = mod.isStereo;
251                             if (!mod.isStereo) {
252                                 zh[2] = mod.toGlobal(new Vec(0.,hit.v,0.)).v[2];
253                                 yh[2] = mod.p.X().v[1];
254                                 if (seedNoGood(2, trial)) continue;
255                             }
256                             for (idx[3] = 0; idx[3] < lyrHits.get(list[3]).size(); idx[3]++) {
257                                 kht = lyrHits.get(list[3]).get(idx[3]);
258                                 hit = kht.hit;
259                                 if (hit.tracks.size() > 0) continue;
260                                 if (hit.energy < kPar.minSeedE[kht.module.Layer]) continue;
261                                 mod = kht.module;
262                                 ah[3] = mod.isStereo;
263                                 if (!mod.isStereo) {
264                                     zh[3] = mod.toGlobal(new Vec(0.,hit.v,0.)).v[2];
265                                     yh[3] = mod.p.X().v[1];
266                                     if (seedNoGood(3, trial)) continue;
267                                 }
268                                 for (idx[4] = 0; idx[4] < lyrHits.get(list[4]).size(); idx[4]++) {
269                                     kht = lyrHits.get(list[4]).get(idx[4]);
270                                     hit = kht.hit;
271                                     if (hit.tracks.size() > 0) continue;
272                                     if (hit.energy < kPar.minSeedE[kht.module.Layer]) continue;
273                                     mod = kht.module;
274                                     ah[4] = mod.isStereo;
275                                     if (!mod.isStereo) {
276                                         zh[4] = mod.toGlobal(new Vec(0.,hit.v,0.)).v[2];
277                                         yh[4] = mod.p.X().v[1];
278                                         if (seedNoGood(4, trial)) continue;
279                                     }
280                                     ArrayList<KalHit> hitList = new ArrayList<KalHit>(5);
281                                     for (int i = 0; i < nLyrs; i++) {
282                                         hitList.add(lyrHits.get(list[i]).get(idx[i]));
283                                     }
284                                     // Cut on the seed timing
285                                     double tmin = 1.e10;
286                                     double tmax = -1.e10;
287                                     for (KalHit ht : hitList) {
288                                         tmin = Math.min(tmin, ht.hit.time);
289                                         tmax = Math.max(tmax, ht.hit.time);
290                                     }
291                                     if (tmax - tmin > kPar.mxTdif) {
292                                         if (debug) {
293                                             System.out.format("KalmanPatRecHPS: skipping seed %d %d %d %d %d with tdif=%8.2f\n Hits:  ", 
294                                                     idx[0], idx[1], idx[2], idx[3], idx[4], tmax-tmin);
295                                             for (KalHit ht : hitList) ht.print("short");
296                                             System.out.format("\n");
297                                         }
298                                         continue;
299                                     }
300                                     // To avoid wasting time fitting seeds, skip seeds that are entirely contained in already found candidates
301                                     boolean redundantSeed = false;
302                                     for (TrackCandidate tkr : candidateList) {
303                                         if (tkr.contains(hitList)) {
304                                             if (debug) System.out.format("KalmanPatRecHPS: seed %d %d %d %d %d is already on candidate %d\n",
305                                                     idx[0], idx[1], idx[2], idx[3], idx[4], tkr.ID);
306                                             redundantSeed = true;
307                                             break;
308                                         }
309                                     }
310                                     if (redundantSeed) continue;
311                                     
312                                     // Fit the seed to extract helix parameters
313                                     SeedTrack seed = new SeedTrack(hitList, yOrigin, kPar.beamSpot[1]);
314                                     if (!seed.success) {
315                                         if (debug) {
316                                             System.out.format("Seed %d %d %d %d %d failed fit\n",idx[0], idx[1], idx[2], idx[3], idx[4]);
317                                         }
318                                         continue;
319                                     }
320                                     
321                                     // Cuts on the seed quality
322                                     Vec hp = seed.helixParams();                        
323                                     if (debug) {
324                                         System.out.format("Seed %d %d %d %d %d parameters for cuts: K=%10.5f (%10.5f), tanl=%10.5f (%10.5f) ",
325                                                           idx[0], idx[1], idx[2], idx[3], idx[4], hp.v[2], kPar.kMax[trial], hp.v[4], kPar.tanlMax[trial]);
326                                     }                                    
327                                     boolean seed_passes_cuts = false;                                    
328                                     if (Math.abs(hp.v[2]) < kPar.kMax[trial]) {
329                                         if (Math.abs(hp.v[4]) < kPar.tanlMax[trial]) {
330                                             Vec pInt = seed.planeIntersection(p0); 
331                                             double xzDist = Math.sqrt(pInt.v[0]*pInt.v[0] + pInt.v[2]*pInt.v[2]);
332                                             if (debug) System.out.format("dxz=%10.5f, Intersection with target plane= %s\n", xzDist, pInt.toString());                                         
333                                             if (xzDist < kPar.dRhoMax[trial]) {
334                                                 if (Math.abs(pInt.v[2]) < kPar.dzMax[trial]) seed_passes_cuts = true;
335                                             } //Check intersection with target plane
336                                         } //Check tanLambda
337                                     } //Check curvature
338                                     
339                                     //Good seed; check for approximate duplicates
340                                     if (seed_passes_cuts) {
341                                         boolean reject_seed = false;
342                                         for (SeedTrack sel_seed : seedList) {
343                                             reject_seed = seed.isCompatibleTo(sel_seed,kPar.seedCompThr);
344                                             if (reject_seed) break;
345                                         }
346                                         if (!reject_seed) seedList.add(seed);
347                                     }
348                                     if (debug) System.out.format("\n");
349                                 }
350                             }
351                         }
352                     }
353                 }
354                 
355                 if (debug) {
356                     double runTime = (double)((System.nanoTime() - startTime)/1000000.);
357                     System.out.format("KalmanPatRecHPS::SeedList size = %d at time %10.6f ms\n", seedList.size(), runTime);
358                     //for (SeedTrack seed : seedList) {
359                     //  seed.print("PF::Check Seed");
360                     //}
361                 }
362                 
363                 // Sort all of the seeds by distance from origin in x,z plane
364                 Collections.sort(seedList, SeedTrack.dRhoComparator);
365                 if (debug) {
366                     double runTime = (double)((System.nanoTime() - startTime)/1000000.);
367                     System.out.format("KalmanPatRecHPS: list of sorted seeds at time %10.6f ms\n",runTime);
368                     int cnt = 0;
369                     for (SeedTrack seed : seedList) {
370                         System.out.format("\nSorted seed %d", cnt);
371                         for (KalHit ht : seed.hits) {
372                             ht.print("short");
373                         }
374                         System.out.format("\n");
375                         seed.print("sorted seeds");
376                         cnt++;
377                     }
378                 }
379 
380                 // Kalman filter the sorted seeds
381                 Vec Bfield = KalmanInterface.getField(pivot, m0.Bfield);
382                 double Bmag = Bfield.mag();
383                 Vec tB = Bfield.unitVec(Bmag);
384                 seedLoop: for (SeedTrack seed : seedList) {
385                     if (debug) {
386                         System.out.format("\n\nStart the filter step for seed");
387                         for (KalHit ht : seed.hits) {
388                             ht.print("short");
389                         }
390                         System.out.format("\n");
391                     }
392                     // Skip seeds that are already on a good track candidate
393                     // This is not redundant with the above test, as we can catch here some candidates formed within this seed loop
394                     for (TrackCandidate tkCand : candidateList) {
395                         if (tkCand.contains(seed.hits)) {
396                             if (debug) System.out.format("KalmanPatRecHPS: skipping seed %d that is already on candidate %d\n", 
397                                     seedList.indexOf(seed),candidateList.indexOf(tkCand));
398                             continue seedLoop;
399                         }
400                     }
401                     // Skip seeds that are already on a track that has already been found and saved
402                     if (TkrList.size() > 0) {
403                         for (KalHit ht : seed.hits) {
404                             if (ht.hit.tracks.size()>0) {
405                                 if (debug) System.out.format("KalmanPatRecHPS: skipping seed %d that is already on track %d\n", 
406                                         seedList.indexOf(seed),ht.hit.tracks.get(0).ID);
407                                 continue seedLoop;
408                             }
409                         }
410                     }
411                     
412                     //DMatrixRMaj CovGuess = seed.covariance().copy();
413                     //if (negativeCov(CovGuess)) {
414                     //    System.out.format("KalmanPatRec: starting covariance matrix is negative for event %d!\n", eventNumber);
415                     //    CovGuess.print();
416                     //}
417                     //CommonOps_DDRM.scale(10., CovGuess);
418                     DMatrixRMaj CovGuess = new DMatrixRMaj(5,5);
419                     setInitCov(CovGuess, seed.helixParams(), true);
420 
421                     // Create an state vector from the input seed to initialize the Kalman filter
422                     StateVector sI = new StateVector(-1, seed.helixParams(), CovGuess, new Vec(0., 0., 0.), Bmag, tB, pivot);
423                     TrackCandidate candidateTrack = new TrackCandidate(candID, seed.hits, kPar, hitMap, eventNumber);
424                     candID++;
425                     filterTrack(candidateTrack, list[0], KalmanParams.numLayers - 1, sI, trial, true, true);
426                     if (!candidateTrack.filtered) {
427                         if (debug) {
428                             System.out.format("KalmanPatRecHPS: filtering of this seed failed. chi2=%10.5f, nHits=%d\n", candidateTrack.chi2f,
429                                     candidateTrack.hits.size());
430                         }
431                         continue seedLoop;
432                     }
433                     if (!candidateTrack.good) {
434                         if (debug) {
435                             System.out.format("KalmanPatRecHPS: candidate track is no good. chi2=%10.5f, nHits=%d\n", candidateTrack.chi2f,
436                                     candidateTrack.hits.size());
437                         }
438                         continue seedLoop;
439                     }
440                     
441                     // Go back to the seed hits on the filtered track and check the detector bounds
442                     for (MeasurementSite site : candidateTrack.sites) {
443                         if (debug) System.out.format("KalmamPatRecHPS: check seed hit layer %d boundaries for candidate %d\n", 
444                                 site.m.Layer, candidateTrack.ID);
445                         int nseedht = 0;
446                         for (int i=0; i<candidateTrack.seedLyrs.size(); ++i) {
447                             boolean seedlyr = false;
448                             if (site.m.Layer == candidateTrack.seedLyrs.get(i)) {
449                                 seedlyr = true;
450                             }
451                             if (!seedlyr) continue;
452                             HelixState hx = site.aP.helix;
453                             double phi = 0.; // The helix at this point starts from the intersection, so phi is always 0 here.
454                             if (debug) {
455                                 System.out.format("       phi at layer %d is %10.6f; measurement=%10.5f\n", site.m.Layer, phi, site.m.hits.get(site.hitID).v);
456                                 double phiChk = hx.planeIntersect(site.m.p); 
457                                 System.out.format("       phi calculated = %10.6f\n", phiChk);
458                             }
459                             //if (Double.isNaN(phi)) continue seedLoop;
460                             Vec rGlob = hx.toGlobal(hx.atPhi(phi));
461                             Vec rDet = site.m.toLocal(rGlob);
462                             if (debug) {
463                                 System.out.format("    global = %s, local = %s\n", rGlob.toString(), rDet.toString());
464                                 System.out.format("    extents: x=%8.3f->%8.3f  y=%8.3f->%8.3f, tolerance=%8.3f\n",
465                                         site.m.xExtent[0], site.m.xExtent[1], site.m.yExtent[0], site.m.yExtent[1],kPar.edgeTolerance);
466                             }
467                             if (rDet.v[0] > site.m.xExtent[1] + kPar.edgeTolerance) {
468                                 if (debug) {
469                                     System.out.format("    The x coordinate %8.3f is too large; reject!\n", rDet.v[0]);
470                                     hx.print(" the offender");
471                                     seed.print(" the originator");
472                                 }
473                                 continue seedLoop;
474                             }
475                             if (rDet.v[0] < site.m.xExtent[0] - kPar.edgeTolerance) {
476                                 if (debug) System.out.format("    The x coordinate %8.3f is too small; reject!\n", rDet.v[0]);
477                                 continue seedLoop;
478                             }
479                             if (rDet.v[1] > site.m.yExtent[1] + kPar.edgeTolerance) {
480                                 if (debug) System.out.format("    The y coordinate %8.3f is too large; reject!\n", rDet.v[1]);
481                                 continue seedLoop;
482                             }

817                                     }
818                                 }
819                             }

831                                 }

838                         } 
839                     }
840                 } // Next seed in set
841             } // Next set of seeds

882                                 }

891                     }

920                                                             }

928                                                 }
929                                             } else {
930                                                 tkr.good = true;
931                                                 if (debug) {
932                                                     System.out.format("KalmanPatRecHPS event %d: resurrecting candidate %d with chi2=%9.5f\n", 
933                                                                       eventNumber, tkr.ID, tkr.chi2s);
934                                                     tkr.print("resurrected", true);

936                                                 continue;

938                                         }

949                         ht.tkrCandidates.remove(tkr);

956             Collections.sort(candidateList,TrackCandidate.CandidateComparator);

958                 for (TrackCandidate tkr : candidateList) {

961                 }
962             }

970                         while (iterator.hasNext()) {

1377     // Remove the worst hit from lousy track candidates
1378     private boolean removeBadHits(TrackCandidate tkr, int trial) {
1379         
1380         if (tkr.chi2s/(double) tkr.hits.size() < kPar.chi2mx1[trial]) return false;
1381         if (tkr.hits.size() <= kPar.minHits1[trial]) return false;
1382         
1383         double mxChi2 = 0.;
1384         int idxBad = -1;
1385         for (int idx=0; idx<tkr.sites.size(); ++idx) {
1386             MeasurementSite site = tkr.sites.get(idx);
1387             if (tkr.seedLyrs.contains(site.m.Layer)) continue;
1388             if (site.chi2inc > mxChi2) {
1389                 mxChi2 = site.chi2inc;
1390                 idxBad = idx;
1391             }
1392         }
1393         if (debug) {
1394             MeasurementSite badSite = tkr.sites.get(idxBad);
1395             System.out.format("KalmanPatRecHPS.removeBadHits: the worst non-seed layer is %d with chi2inc=%7.2f.\n", badSite.m.Layer, badSite.chi2inc);
1396         }
1397         if (idxBad >= 0) {
1398             if (mxChi2 > kPar.minChi2IncBad) {
1399                 MeasurementSite site = tkr.sites.get(idxBad);
1400                 Measurement badOne = site.m.hits.get(site.hitID);
1401                 KalHit badHit = hitMap.get(badOne);
1402                 if (badHit != null) {
1403                     candidateList.add(tkr.copy());    // Keep a "bad" copy of the original candidate so that it isn't found and fit again
1404                     if (debug) System.out.format("KalmanPatRecHPS.removeBadHits: event %d, candidate %d, removing bad hit in layer %d with chi2=%9.3f.\n",eventNumber, 
1405                             tkr.ID, tkr.sites.get(idxBad).m.Layer,mxChi2);
1406                     tkr.removeHit(badHit, true);
1407                     return true;
1408                 }
1409             }
1410         }
1411         return false;
1412     }

1414     // Method to smooth an already filtered track candidate
1415     private void smoothTrack(TrackCandidate filteredTkr) {
1416         MeasurementSite nextSite = null;
1417         boolean badCov = false;
1418         for (int idxS = filteredTkr.sites.size() - 1; idxS >= 0; idxS--) {
1419             MeasurementSite currentSite = filteredTkr.sites.get(idxS);
1420             if (nextSite == null) {   // The outermost site with a hit is already smoothed by definition
1421                 if (currentSite.hitID < 0) continue;
1422                 currentSite.aS = currentSite.aF;
1423                 currentSite.smoothed = true;
1424             } else {
1425                 currentSite.smooth(nextSite);
1426             }
1427             filteredTkr.chi2s += Math.max(currentSite.chi2inc, 0.);
1428             if (negativeCov(currentSite.aS.helix.C)) {
1429                 badCov = true;
1430                 fixCov(currentSite.aS.helix.C, currentSite.aS.helix.a);
1431             }
1432             nextSite = currentSite;
1433             //if (debug) currentSite.print("smoothed");
1434         }
1435         if (badCov) nBadCov[1]++;
1436         filteredTkr.smoothed = true;

1439     // Execute the Kalman prediction and filter steps over a range of SVT layers
1440     private void filterTrack(TrackCandidate tkrCandidate, int lyrBegin, // layer on which to start the filtering
1441             int lyrEnd, // layer on which to end the filtering
1442             StateVector sI, // initialization state vector
1443             int trial, // trial level, for selecting cuts
1444             boolean startNew, // Start the fit over
1445             boolean pickUp    // true to allow picking up new hits
1446     ) {
1447 
1448         //if (tkrCandidate.hits.size() - tkrCandidate.nTaken < minUnique) {
1449         //    if (debug) {
1450         //        System.out.format("KalmanPatRecHPS:filterTrack: skipping for hits used. nLyrs=%d, nTaken=%d\n", tkrCandidate.hits.size(),
1451         //                tkrCandidate.nTaken);
1452         //    }
1453         //    return;
1454         //}
1455 
1456         if (debug) {
1457             Vec hprms = sI.helix.a;
1458             System.out.format("\n KalmanPatRecHPS:filterTrack: Start filtering candidate %d with drho=%10.5f phi0=%10.5f k=%10.5f dz=%10.5f tanl=%10.5f \n",
1459                     tkrCandidate.ID, hprms.v[0], hprms.v[1], hprms.v[2], hprms.v[3], hprms.v[4]);
1460             System.out.format("                    origin=%s,   pivot=%s\n", sI.helix.origin.toString(), sI.helix.X0.toString());
1461         }
1462 
1463         if (startNew) {
1464             tkrCandidate.filtered = false;
1465             tkrCandidate.chi2f = 0.;
1466             tkrCandidate.chi2s = 0.;
1467             tkrCandidate.sites.clear();
1468         }
1469         
1470         MeasurementSite newSite = null;
1471         MeasurementSite prevSite = null;
1472         Map<KalHit, MeasurementSite> siteMap = new HashMap<KalHit, MeasurementSite>(tkrCandidate.hits.size());
1473         for (KalHit hit : tkrCandidate.hits) {
1474             if (debug) {
1475                 System.out.format("    On entering filterTrack: ");
1476                 hit.print(" existing ");
1477             }
1478             siteMap.put(hit, null);
1479         }
1480 
1481         int thisSite = -1;
1482         // Filter from the start of the seed to the last downstream detector layer
1483         // loop over all layers from the seed beginning to the end of the tracker
1484         int direction;
1485         if (lyrEnd > lyrBegin) {
1486             direction = 1;
1487         } else {
1488             direction = -1;
1489         }
1490         boolean badCov = false;
1491         boolean needCleanup = false;
1492         layerLoop: for (int lyr = lyrBegin; lyr != lyrEnd + direction; lyr += direction) {
1493             SiModule mExistingHit = null;
1494             // Find the correct hit number and its module if this is one of the layers with an existing hit
1495             int hitno = -1;
1496             for (KalHit ht : tkrCandidate.hits) {
1497                 if (ht.module.Layer == lyr) {
1498                     hitno = ht.module.hits.indexOf(ht.hit);
1499                     mExistingHit = ht.module;
1500                     break;
1501                 }
1502             }
1503             if (debug) System.out.format("KalmanPatRecHPS.filterTrack: layer %d, %d modules\n", lyr, moduleList.get(lyr).size());
1504             // Loop over all of the modules in this layer
1505             for (int imod = 0; imod < moduleList.get(lyr).size(); ++imod) {
1506                 SiModule m = moduleList.get(lyr).get(imod);
1507                 // Only consider the one module with the given hit for the existing specified hits
1508                 if (mExistingHit != null) {
1509                     if (m != mExistingHit) {
1510                         if (debug) System.out.format("KalmanPatRecHPS.filterTrack: skip module at layer %d, detector %d, %d hits, given hit=%d.\n",
1511                                                        mExistingHit.Layer, mExistingHit.detector, mExistingHit.hits.size(), hitno);
1512                         continue;
1513                     }
1514                 } 
1515 
1516                 if (debug) {
1517                     System.out.format("KalmanPatRecHPS.filterTrack: try prediction at layer %d, detector %d, %d hits, given hit=%d.\n",
1518                                        m.Layer, m.detector, m.hits.size(), hitno);
1519                     //HelixState hx = null;
1520                     //if (prevSite == null) hx = sI.helix; else hx = prevSite.aF.helix;
1521                     //hx.print("predicting from ");
1522                     //Vec originB = hx.toGlobal(hx.X0);
1523                     //Vec newB = KalmanInterface.getField(originB, m.Bfield);
1524                     //System.out.format("  B=%10.4f   t=%s\n", newB.mag(), newB.unitVec().toString());
1525                 }
1526                 newSite = new MeasurementSite(lyr, m, kPar);
1527                 int rF;
1528                 double [] tRange = {tkrCandidate.tMax - kPar.mxTdif, tkrCandidate.tMin + kPar.mxTdif}; 
1529                 if (prevSite == null) { // For first layer use the initializer state vector               
1530                     boolean checkBounds = imod < moduleList.get(lyr).size() - 1;  // Note: boundary check is not made on last module of the layer
1531                     rF = newSite.makePrediction(sI, null, hitno, tkrCandidate.nTaken <= kPar.mxShared, pickUp, checkBounds, tRange, trial);
1532                     if (rF > 0) {
1533                         if (m.hits.get(newSite.hitID).tracks.size() > 0) tkrCandidate.nTaken++;
1534                         tkrCandidate.tMin = Math.min(tkrCandidate.tMin, m.hits.get(newSite.hitID).time);
1535                         tkrCandidate.tMax = Math.max(tkrCandidate.tMax, m.hits.get(newSite.hitID).time);
1536                     } else if (rF < 0) {
1537                         if (rF == -2) {  // This really shouldn't happen at the initial site
1538                             if (debug) System.out.format("KalmanPatRecHPS.filterTrack: not within detector boundary on layer %d detector %d!!\n",
1539                                     newSite.m.Layer, newSite.m.detector);
1540                             continue;
1541                         }
1542                         if (debug) {
1543                             System.out.format("KalmanPatRecHPS:filterTrack: Failed to make prediction at initial site %d, layer=%d! Abort\n",
1544                                     thisSite + 1, lyr);
1545                         }
1546                         for (KalHit hit : tkrCandidate.hits) {
1547                             hit.tkrCandidates.remove(tkrCandidate);                            
1548                         }
1549                         tkrCandidate.good = false;
1550                         tkrCandidate.hits.clear();
1551                         return;
1552                     }
1553                 } else {
1554                     boolean checkBounds = imod < moduleList.get(lyr).size() - 1;  // Note: boundary check is not made on last module of layer
1555                     rF = newSite.makePrediction(prevSite.aF, prevSite.m, hitno, tkrCandidate.nTaken <= kPar.mxShared, pickUp, checkBounds, tRange, trial);
1556                     if (rF > 0) {
1557                         if (m.hits.get(newSite.hitID).tracks.size() > 0) tkrCandidate.nTaken++;
1558                         tkrCandidate.tMin = Math.min(tkrCandidate.tMin, m.hits.get(newSite.hitID).time);
1559                         tkrCandidate.tMax = Math.max(tkrCandidate.tMax, m.hits.get(newSite.hitID).time);
1560                     } else if (rF < 0) {
1561                         if (rF == -2) {
1562                             if (debug) System.out.format("KalmanPatRecHPS.filterTrack: not within detector boundary on layer %d detector %d\n",
1563                                     newSite.m.Layer, newSite.m.detector);
1564                             continue;  // No hit associated here, so look in the next Si module of this layer.
1565                         }
1566                         if (debug) {
1567                             System.out.format("KalmanPatRecHPS:filterTrack: Failed to make prediction at site %d, layer=%d.  Exit layer loop\n",
1568                                     thisSite + 1, lyr);
1569                         }
1570                         needCleanup = true;
1571                         break layerLoop;
1572                     }
1573                 }
1574                 if (debug) {
1575                     System.out.format("KalmanPatRecHPS.filterTrack: candidate %d, completed prediction at site (%d, %d, %d)\n",
1576                             tkrCandidate.ID, newSite.m.Layer, newSite.m.detector, newSite.hitID);
1577                 }
1578                 thisSite++;
1579                 if (!newSite.filter()) {
1580                     if (debug) System.out.format("KalmanPatRecHPS:filterTrack: Failed to filter at site %d, layer=%d.  Ignore remaining sites\n", thisSite, lyr);
1581                     needCleanup = true;
1582                     break layerLoop;
1583                 }
1584                 if (negativeCov(newSite.aF.helix.C)) {
1585                     if (debug) System.out.format("KalmanPatRecHPS.filtertrack: event %d candidate %d layer %d has negative covariance after filter step.\n",
1586                             eventNumber, tkrCandidate.ID, newSite.m.Layer);
1587                     badCov = true;
1588                     fixCov(newSite.aF.helix.C, newSite.aF.helix.a);
1589                 }
1590                 if (debug) {
1591                     System.out.format("KalmanPatRecHPS.filterTrack: candidate %d, completed filter at site (%d, %d, %d), chi2-inc=%8.3f\n",
1592                             tkrCandidate.ID, newSite.m.Layer, newSite.m.detector, newSite.hitID, newSite.chi2inc);
1593                 }
1594                 KalHit theHit = null;
1595                 if (newSite.hitID >= 0) theHit = hitMap.get(m.hits.get(newSite.hitID));
1596                 if (rF == 1 && hitno < 0) {
1597                     tkrCandidate.hits.add(theHit);
1598                     if (debug) theHit.print("new");
1599                 }
1600 
1601                 // if (debug) newSite.print("initial filtering");
1602                 tkrCandidate.chi2f += Math.max(newSite.chi2inc,0.);
1603 
1604                 tkrCandidate.sites.add(newSite);
1605                 if (theHit != null) siteMap.put(theHit, newSite);
1606                 
1607                 prevSite = newSite;
1608                 break;    // A hit was found for this Si module, so skip the rest of them in this layer.
1609             }
1610         }
1611         if (needCleanup) {
1612             Iterator<KalHit> itr = tkrCandidate.hits.iterator();
1613             int nstereo = 0;
1614             while (itr.hasNext()) {   // Clean up the hit list
1615                 KalHit hit = itr.next();
1616                 MeasurementSite theSite = siteMap.get(hit);
1617                 int hitID = -1;
1618                 if (theSite != null) hitID = theSite.hitID;
1619                 if (hitID < 0) {
1620                     if (debug) System.out.format("    KalmanPatRecHPS.filterTrack: remove hit from candidate at layer %d\n", hit.module.Layer);
1621                     hit.tkrCandidates.remove(tkrCandidate);
1622                     itr.remove();
1623                 } else if (hit.module.isStereo) nstereo++;
1624             }
1625             if (tkrCandidate.hits.size() < 5) {
1626                 if (debug) System.out.format("    KalmanPatRecHPS.filterTrack: marking candidate bad; too few hits after cleanup: %d\n",tkrCandidate.hits.size());
1627                 tkrCandidate.good = false;
1628             }
1629             if (nstereo < 3) {
1630                 if (debug) System.out.format("    KalmanPatRecHPS.filterTrack: marking candidate bad; too few stereo hits after cleanup: %d\n",nstereo);
1631                 tkrCandidate.good = false;
1632             }
1633         }
1634         if (badCov) nBadCov[0]++;
1635         tkrCandidate.filtered = true;
1636         return;
1637     }

1639     boolean storeTrack(int tkID, TrackCandidate tkrCand) {
1640         if (debug) System.out.format("entering storeTrack for track %d, debug=%b\n", tkID, debug);
1641 
1642         MeasurementSite firstSite = null;
1643         for (int idx=0; idx<tkrCand.sites.size()-1; ++idx) {
1644             MeasurementSite site = tkrCand.sites.get(idx);
1645             if (site.hitID >= 0) firstSite = site;
1646             MeasurementSite nxtSite = tkrCand.sites.get(idx+1);
1647             if (nxtSite.m.Layer-site.m.Layer > 1) {
1648                 logger.warning(String.format("Event %d, Track candidate %d has missing layer %d", eventNumber, tkrCand.ID, site.m.Layer+1));
1649             }
1650         }
1651         if (firstSite == null) tkrCand.print("firstSite null", false);
1652         if (firstSite.aS == null) tkrCand.print("aS null", false);
1653         if (Math.abs(firstSite.aS.helix.a.v[2]) > kPar.kMax[1]) {
1654             if (debug) System.out.format("KalmanPatRecHPS.storeTrack: k=%10.4f is too large for candidate %d\n", 
1655                     firstSite.aS.helix.a.v[2], tkrCand.ID);
1656             return false;
1657         }
1658         if (Math.abs(firstSite.aS.helix.a.v[0]) > kPar.dRhoMax[1]) {
1659             if (debug) System.out.format("KalmanPatRecHPS.storeTrack: dRho=%10.4f is too large for candidate %d\n", 
1660                     firstSite.aS.helix.a.v[0], tkrCand.ID);
1661             return false;
1662         }
1663         if (Math.abs(firstSite.aS.helix.a.v[3]) > kPar.dzMax[1]) {
1664             if (debug) System.out.format("KalmanPatRecHPS.storeTrack: dz=%10.4f is too large for candidate %d\n", 
1665                     firstSite.aS.helix.a.v[3], tkrCand.ID);
1666             return false;
1667         }
1668         
1669         KalTrack tkr = new KalTrack(eventNumber, tkID, tkrCand.sites, yScat, XLscat, kPar);
1670         boolean redundant = false;
1671         for (KalTrack oldTkr : TkrList) {
1672             if (tkr.equals(oldTkr)) {
1673                 redundant = true;
1674                 break;
1675             }
1676         }
1677         if (redundant) {
1678             if (debug) System.out.format("KalmanPatRecHPS.storeTrack: throwing away redundant track %d\n", tkID);
1679             return false;
1680         } 
1681         for (MeasurementSite site : tkr.SiteList) {
1682             int theHit = site.hitID;
1683             if (theHit >= 0) site.m.hits.get(theHit).tracks.add(tkr); // Mark the hits as used
1684         }
1685         if (debug) {
1686             System.out.format("KalmanPatRecHPS.storeTrack: Adding track %d with %d hits and smoothed chi^2=%10.5f\n",
1687                     tkID, tkrCand.hits.size(), tkrCand.chi2s);
1688             System.out.format(" Complete list of sites on this track: ");
1689             for (MeasurementSite site : tkr.SiteList) {
1690                 System.out.format("(%d, %d, %d) ",site.m.Layer, site.m.detector, site.hitID);
1691             }
1692             System.out.format("\n");
1693         }
1694         tkrCand.kalTkrID = tkID;
1695         TkrList.add(tkr);
1696         return true;
1697     }

1736                 for (int j=0; j<5; ++j) {

1764     // Quick check on where the seed track is heading, using only the two axial layers in the seed
1765     private boolean seedNoGood(int j, int iter) {
1766         // j must point to an axial layer in the seed.
1767         // Find the previous axial layer, if there is one. . .
1768         for (int i=0; i<j; ++i) {
1769             if (!ah[i]) {
1770                 double slope = (zh[j] - zh[i]) / (yh[j] - yh[i]);
1771                 double zIntercept = zh[i] - slope * yh[i];
1772                 //System.out.format("seedNoGood: i=%d lyr=%d, j=%d, lyr=%d slope=%10.5f, zInt=%10.5f\n", i, lyrs[i], j, lyrs[j], slope, zIntercept);
1773                 if (Math.abs(zIntercept) > kPar.dzMax[iter]) {
1774                     if (debug) {
1775                         System.out.format("KalmanPatRecHPS.seedNoGood: reject z j=%d, i=%d, zInt=%10.5f vs %10.5f, slope=%10.5f vs %10.5f\n",
1776                                 j, i, zIntercept, kPar.dzMax[iter], slope, kPar.tanlMax[iter]);
1777                     }
1778                     return true;
1779                 }
1780                 if (Math.abs(slope) > kPar.tanlMax[iter]) {
1781                     if (debug) {
1782                         System.out.format("KalmanPatRecHPS.seedNoGood: reject slope j=%d, i=%d, zInt=%10.5f vs %10.5f, slope=%10.5f vs %10.5f\n",
1783                                 j, i, zIntercept, kPar.dzMax[iter], slope, kPar.tanlMax[iter]);
1784                     }
1785                     return true;
1786                 }
1787                 return false;
1788             }
1789         }
1790         return false;
