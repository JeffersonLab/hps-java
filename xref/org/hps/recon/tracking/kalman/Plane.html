1   package org.hps.recon.tracking.kalman;
2   
3   /**
4    * Description of a 2D plane in 3D space
5    */
6   class Plane { 
7   
8       private Vec x; // A point in the plane, at the origin of the local plane's coordinate system
9       private Vec t; // Unit vector perpendicular to the plane
10      private Vec u; // Local axis always lies in the world system x,y plane
11      private Vec v; // Other local axis, t,u,v forms a RH system
12  
13      Plane(Vec point, Vec direction) {
14          x = point.copy();
15          t = direction.copy();
16          Vec zhat = new Vec(0., 0., 1.);
17          u = t.cross(zhat).unitVec();
18          v = t.cross(u);
19      }
20  
21      Plane copy() {
22          return new Plane(x, t);
23      }
24  
25      Plane(Vec newX, Vec newT, Vec newU, Vec newV) {
26          x = newX;
27          t = newT;
28          u = newU;
29          v = newV;
30      }
31      
32      Plane(Vec X, Vec T, Vec U) {
33          x = X;
34          t = T;
35          u = U;
36          v = t.cross(u);        
37      }
38      
39      // strange constructor for backward compatibility of SiModule.java
40      Plane(Vec X, Vec T, double theta) {
41          x = X;
42          Vec zhat = new Vec(0., 0., 1.);
43          t = T.copy();
44          u = t.cross(zhat).unitVec();
45          v = t.cross(u);
46          RotMatrix R1 = new RotMatrix(u,v,t);
47          RotMatrix R2 = new RotMatrix(theta);
48          RotMatrix Rt = R2.multiply(R1);
49          for (int i=0; i<3; i++) {
50              u.v[i] = Rt.M[0][i];
51              v.v[i] = Rt.M[1][i];
52              t.v[i] = Rt.M[2][i];
53          }
54      }
55  
56      void print(String s) {
57          System.out.format("%s",this.toString(s));
58      }
59      
60      String toString(String s) {
61          String str = String.format("Printout of plane %s\n", s);
62          str=str+"       point="+x.toString();
63          str=str+"       direction="+t.toString()+"\n";
64          str=str+"       uhat="+u.toString();
65          str=str+"       vhat="+v.toString()+"\n";
66          return str;
67      }
68      
69      public String toString() {
70          return String.format("pnt %s dir %s", x.toString(), t.toString());
71      }
72  
73  
74      Vec U() { // unit vector in the plane perpendicular to t and the world z axis
75          return u;
76      }
77  
78      Vec V() { // another orthogonal unit vector in the plane perp to t and u
79          return v;
80      }
81  
82      Vec T() {
83          return t;
84      }
85  
86      Vec X() {
87          return x;
88      }
89  
90      // Convert the plane to a local coordinate system
91      Plane toLocal(RotMatrix R, Vec origin) {
92          return new Plane(R.rotate(x.dif(origin)), R.rotate(t), R.rotate(u), R.rotate(v));
93      }
94  }
