37      // Constructor for the initial state vector used to start the Kalman filter.
38      StateVector(int site, Vec helixParams, DMatrixRMaj Cov, Vec pivot, double B, Vec tB, Vec origin) {
39          // Here tB is the B field direction, while B is the magnitude       
40          if (debug) System.out.format("StateVector: constructing an initial state vector\n");
41          helix = new HelixState(helixParams, pivot, origin, Cov, B, tB);
42          kLow = site;
43          kUp = kLow;
44          if (!initialized) {  // Initialize the static working arrays on the first call
45              logger = Logger.getLogger(StateVector.class.getName());
46              tempV = new DMatrixRMaj(5,1);
47              tempV2 = new DMatrixRMaj(5,1);
48              tempM = new DMatrixRMaj(5,5);
49              tempA = new DMatrixRMaj(5,5);
50              Q = new DMatrixRMaj(5,5);
51              U = CommonOps_DDRM.identity(5,5);
52              Cinv = new DMatrixRMaj(5,5);
53              solver = LinearSolverFactory_DDRM.symmPosDef(5);
54              initialized = true;
55          }
56      }

93      StateVector copy() {
94          StateVector q = new StateVector(kLow);
95          q.helix = (HelixState)helix.copy();      // Deep copy
96          q.kUp = kUp;
97          q.F = F;  // Don't deep copy the F matrix
98          q.mPred = mPred;
99          q.R = R;
100         q.r = r;
101         q.K = K;
102         return q;
103     }

110     String toString(String s) {
111         String str = String.format(">>>Dump of state vector %s %d  %d\n", s, kUp, kLow);
112         str = str + helix.toString(" ");
113         if (F != null) str = str + "Propagator matrix: " + F.toString();
114         double sigmas;
115         if (R > 0.) {
116             sigmas = r / Math.sqrt(R);
117         } else {
118             sigmas = 0.;
119         }
120         str = str + String.format("Predicted measurement=%10.6f, residual=%10.7f, covariance of residual=%12.4e, std. devs. = %12.4e\n", mPred, r, R,
121                 sigmas);
122         str = str + String.format("End of dump of state vector %s %d  %d<<<\n", s, kUp, kLow);
123         return str;
124     }

126     // Create a predicted state vector by propagating a given helix to a measurement site
127     StateVector predict(int newSite, Vec pivot, double B, Vec t, Vec originPrime, double XL, double deltaE) {
128         // newSite = index of the new site
129         // pivot = pivot point of the new site in the local coordinates of this state vector (i.e. coordinates of the old site)
130         // B and t = magnitude and direction of the magnetic field at the pivot point, in global coordinates
131         // XL = thickness of the scattering material
132         // deltaE = energy loss in the scattering material
133         // originPrime = origin of the detector coordinates at the new site in global coordinates
134 
135         // This constructs a new blank state vector with pivot and helix parameters undefined as yet
136         StateVector aPrime = new StateVector(newSite, B, t, originPrime);
137         aPrime.kUp = kUp;
138         aPrime.helix.X0 = pivot; // pivot before helix rotation, in coordinate system of the previous site
139 
140         double E = helix.a.v[2] * Math.sqrt(1.0 + helix.a.v[4] * helix.a.v[4]);
141         double deltaEoE = deltaE / E;
142 
143         // Transform helix in old coordinate system to new pivot point lying on the next detector plane
144         if (deltaE == 0.) {
145             aPrime.helix.a = this.helix.pivotTransform(pivot);
146         } else {
147             aPrime.helix.a = this.helix.pivotTransform(pivot, deltaEoE);
148         }
149 
150         if (debug) { // drho and dz are indeed always zero here
151             aPrime.helix.a.print("StateVector predict: pivot transformed helix; should have zero drho and dz");
152             helix.a.print("old helix");
153             pivot.print("new pivot");
154             helix.X0.print("old pivot");
155         }
156 
157         F = new DMatrixRMaj(5,5);
158         this.helix.makeF(aPrime.helix.a, F); // Calculate derivatives of the pivot transform
159         if (deltaE != 0.) {
160             double factor = 1.0 - deltaEoE;
161             for (int i = 0; i < 5; i++) F.unsafe_set(i, 2, F.unsafe_get(i,2)*factor);  
162         }
163 
164         // Transform to the coordinate system of the field at the new site
165         // First, transform the pivot point to the new system
166         aPrime.helix.X0 = aPrime.helix.toLocal(this.helix.toGlobal(aPrime.helix.X0));
167 
168         // Calculate the matrix for the net rotation from the old site coordinates to the new site coordinates
169         RotMatrix Rt = aPrime.helix.Rot.multiply(this.helix.Rot.invert());
170         if (debug) {
171             aPrime.helix.Rot.print("aPrime rotation matrix");
172             this.helix.Rot.print("this rotation matrix");
173             Rt.print("rotation from old local frame to new local frame");
174             aPrime.helix.a.print("StateVector:predict helix before rotation");
175         }
176 
177         // Rotate the helix parameters here. 
178         // dz and drho will remain unchanged at zero
179         // phi0 and tanl(lambda) change, as does kappa (1/pt). However, |p| should be unchanged by the rotation.
180         // This call to rotateHelix also calculates the derivative matrix fRot
181         aPrime.helix.a = HelixState.rotateHelix(aPrime.helix.a, Rt, tempM);
182         if (debug) {
183             aPrime.helix.a.print("StateVector:predict helix after rotation");
184             System.out.println("fRot from StateVector:predict");
185             tempM.print();
186         }
187         CommonOps_DDRM.mult(tempM, F, tempA);
188 
189         // Test the derivatives
190         /*
191         if (debug) {
192             double daRel[] = { 0.01, 0.03, -0.02, 0.05, -0.01 };
193             StateVector aPda = this.Copy();
194             for (int i = 0; i < 5; i++) {
195                 aPda.a.v[i] = a.v[i] * (1.0 + daRel[i]);
196             }
197             Vec da = aPda.a.dif(a);
198             StateVector aPrimeNew = this.Copy();
199             aPrimeNew.a = aPda.pivotTransform(pivot);
200             RotMatrix RtTmp = Rot.invert().multiply(aPrime.Rot);
201             SquareMatrix fRotTmp = new SquareMatrix(5);
202             aPrimeNew.a = rotateHelix(aPrimeNew.a, RtTmp, fRotTmp);
203             for (int i = 0; i < 5; i++) {
204                 double deltaExact = aPrimeNew.a.v[i] - aPrime.a.v[i];
205                 double delta = 0.;
206                 for (int j = 0; j < 5; j++) {
207                     delta += F.M[i][j] * da.v[j];
208                 }
209                 System.out.format("Test of F: Helix parameter %d, deltaExact=%10.8f, delta=%10.8f\n", i, deltaExact,
210                         delta);
211             }
212         }
213         */
214 
215         aPrime.kLow = newSite;
216         aPrime.kUp = kUp;
217 
218         // Add the multiple scattering contribution for the silicon layer
219         // sigmaMS is the rms of the projected scattering angle
220         if (XL == 0.) {
221             Cinv.set(this.helix.C);
222             if (debug) System.out.format("StateVector.predict: XL=%9.6f\n", XL);
223         } else {
224             double momentum = (1.0 / helix.a.v[2]) * Math.sqrt(1.0 + helix.a.v[4] * helix.a.v[4]);
225             double sigmaMS = HelixState.projMSangle(momentum, XL);
226             if (debug) System.out.format("StateVector.predict: momentum=%12.5e, XL=%9.6f sigmaMS=%12.5e\n", momentum, XL, sigmaMS);
227             this.helix.getQ(sigmaMS, Q);
228             CommonOps_DDRM.add(this.helix.C, Q, Cinv);
229         }
230 
231         // Now propagate the multiple scattering matrix and covariance matrix to the new site
232         CommonOps_DDRM.multTransB(Cinv, tempA, tempM);
233         aPrime.helix.C = new DMatrixRMaj(5,5);
234         CommonOps_DDRM.mult(tempA, tempM, aPrime.helix.C);
235 
236         return aPrime;
237     }

239     // Create a filtered state vector from a predicted state vector
240     StateVector filter(DMatrixRMaj H, double V) {
241         // H = prediction matrix (5-vector)
242         // V = hit variance (1/sigma^2)
243 
244         StateVector aPrime = this.copy();
245         aPrime.kUp = kLow;
246 
247         CommonOps_DDRM.mult(helix.C, H, tempV);
248         double denom = V + CommonOps_DDRM.dot(H, tempV);
249         CommonOps_DDRM.scale(1.0/denom, helix.C, tempM);
250         K = new DMatrixRMaj(5,1);
251         CommonOps_DDRM.mult(tempM, H, K);  //  Kalman gain matrix
252         if (debug) {
253             System.out.format("StateVector.filter: kLow=%d\n", kLow);
254             System.out.format("StateVector.filter: V=%12.4e,  denom=%12.4e\n", V, denom);
255             System.out.format("Kalman gain matrix in StateVector.filter: ");
256             K.print();
257             System.out.format("matrix H in StateVector.filter");
258             H.print();
259             System.out.format("StateVector.filter: k dot H = %10.7f\n", CommonOps_DDRM.dot(K, H));
260             // Alternative calculation of K (sanity check that it gives the same result):
261             DMatrixRMaj D = new DMatrixRMaj(5,5);
262             CommonOps_DDRM.invert(helix.C, D);
263             DMatrixRMaj Dp = new DMatrixRMaj(5,5);
264             for (int i=0; i<5; ++i) {
265                 for (int j=0; j<5; ++j) {
266                     Dp.unsafe_set(i, j, H.unsafe_get(i, 0)*H.unsafe_get(j, 0)/V);
267                 }
268             }
269             CommonOps_DDRM.addEquals(D, Dp);
270             CommonOps_DDRM.invert(D);
271             DMatrixRMaj Hs = new DMatrixRMaj(5);
272             CommonOps_DDRM.scale(1.0/V, H, Hs);
273             DMatrixRMaj Kalt = new DMatrixRMaj(5);
274             CommonOps_DDRM.mult(D, Hs, Kalt);
275             System.out.println("Alternative Kalman gain matrix:");
276             Kalt.print();
277         }
278 
279         CommonOps_DDRM.scale(r, K, tempV);
280         aPrime.helix.a = helix.a.sum(mToVec(tempV));
281         directProd(K, H, tempM);
282         CommonOps_DDRM.scale(-1.0, tempM);
283         CommonOps_DDRM.addEquals(tempM, U);
284         CommonOps_DDRM.mult(tempM, helix.C, aPrime.helix.C);
285 
286         if (debug) {
287             System.out.format("StateVector.filter: compare covariance calculations, original one first:\n");
288             // Alternate calculation of the covariance update
289             double R = V*(1- CommonOps_DDRM.dot(H, K));
290             CommonOps_DDRM.mult(tempM, helix.C, tempV);
291             CommonOps_DDRM.multTransB(tempV, tempM, tempA);
292             directProd(K, K, tempV);
293             DMatrixRMaj tempZ = new DMatrixRMaj(5,5);
294             CommonOps_DDRM.add(tempA, R, tempV, tempZ);
295             aPrime.helix.C.print();
296             tempZ.print();
297             System.out.println("filtered covariance (gain-matrix formalism) in StateVector.filter:");
298             aPrime.helix.C.print();
299             // Alternative calculation of filtered covariance (sanity check that it gives
300             // the same result):
301             DMatrixRMaj Cinv = new DMatrixRMaj(25);
302             CommonOps_DDRM.invert(helix.C, Cinv);
303             DMatrixRMaj Hscaled = new DMatrixRMaj(5);
304             CommonOps_DDRM.scale(1.0/V, H, Hscaled);
305             directProd(Hscaled,H,tempM);
306             DMatrixRMaj D = new DMatrixRMaj(5);
307             CommonOps_DDRM.add(Cinv,tempM,D);
308             CommonOps_DDRM.invert(D);
309             System.out.println("alternate (weighted-means formalism) filtered covariance in StateVector.filter");
310             D.print();
311             helix.a.print("predicted helix parameters");
312             aPrime.helix.a.print("filtered helix parameters (gain matrix formalism)");
313         }
314 
315         return aPrime;
316     }

318     // Modify the state vector by removing the hit information
319     Vec inverseFilter(DMatrixRMaj H, double V, DMatrixRMaj Cnew) {
320         CommonOps_DDRM.mult(helix.C, H, tempV);
321         double denom = -V + CommonOps_DDRM.dot(H, tempV);
322         CommonOps_DDRM.scale(1.0/denom, helix.C, tempM);
323         DMatrixRMaj Kstar = new DMatrixRMaj(5,1);
324         CommonOps_DDRM.mult(tempM, H, Kstar);   // Kalman gain matrix
325 
326         CommonOps_DDRM.scale(r, Kstar, tempV);
327         Vec aNew = helix.a.sum(mToVec(tempV));
328         directProd(Kstar, H, tempM);
329         CommonOps_DDRM.scale(-1.0, tempM);
330         CommonOps_DDRM.addEquals(tempM, U);
331         CommonOps_DDRM.mult(tempM, helix.C, Cnew);
332         if (debug) {
333             System.out.format("StateVector.inverseFilter: V=%12.4e,  denom=%12.4e\n", V, denom);
334             helix.a.print("old helix");
335             aNew.print(" new helix in StateVector.inverseFilter");
336         }
337         return aNew;
338     }

340     // Create a smoothed state vector from the filtered state vector
341     StateVector smooth(StateVector snS, StateVector snP) {
342         if (debug) System.out.format("StateVector.smooth of filtered state %d %d, using smoothed state %d %d and predicted state %d %d\n", kLow, kUp,
343                     snS.kLow, snS.kUp, snP.kLow, snP.kUp);
344         StateVector sS = this.copy();
345 
346         // solver.setA defines the input matrix and checks whether it is singular. 
347         // A copy is needed because the input gets modified.
348         if (!solver.setA(snP.helix.C.copy())) {
349             SquareMatrix invrs = KalTrack.mToS(snP.helix.C).fastInvert();
350             if (invrs == null) {
351                 logger.warning("StateVector:smooth, inversion of the covariance matrix failed");
352                 snP.helix.C.print();
353                 for (int i=0; i<5; ++i) {      // Fill the inverse with something not too crazy and continue . . .
354                     for (int j=0; j<5; ++j) {
355                         if (i == j) {
356                             Cinv.unsafe_set(i,j,1.0/snP.helix.C.unsafe_get(i,j));
357                         } else {
358                             Cinv.unsafe_set(i, j, 0.);
359                             snP.helix.C.unsafe_set(i, j, 0.);
360                         }
361                     }
362                 }          
363             } else {
364                 if (debug) {
365                     KalTrack.mToS(snP.helix.C).print("singular covariance?");
366                     invrs.print("inverse");
367                     invrs.multiply(KalTrack.mToS(snP.helix.C)).print("unit matrix?"); 
368                 }
369                 for (int i=0; i<5; ++i) {
370                     for (int j=0; j<5; ++j) {
371                         Cinv.unsafe_set(i, j, invrs.M[i][j]);
372                     }
373                 }
374             }
375         } else {
376             solver.invert(Cinv);
377         }
378 
379         CommonOps_DDRM.multTransB(helix.C, sS.F, tempM);
380         CommonOps_DDRM.mult(tempM, Cinv, tempA);
381 
382         vecToM(snS.helix.a.dif(snP.helix.a), tempV);
383         CommonOps_DDRM.mult(tempA, tempV, tempV2);
384         sS.helix.a = helix.a.sum(mToVec(tempV2));
385         if (debug) {
386             System.out.println("StateVector:smooth, inverse of the covariance:");
387             Cinv.print("%11.6e");
388             CommonOps_DDRM.mult(snP.helix.C, Cinv, tempM);
389             System.out.format("Unit matrix?? ");
390             tempM.print();
391             System.out.format("Predicted helix covariance: ");
392             snP.helix.C.print();
393             System.out.format("This helix covariance: ");
394             helix.C.print();
395             System.out.format("Matrix F ");
396             sS.F.print();
397             System.out.format("tempM ");
398             tempM.print();
399             System.out.format("tempA ");
400             tempA.print();
401             System.out.format("Difference of helix parameters tempV: ");
402             tempV.print();
403             System.out.format("tempV2 ");
404             tempV2.print();
405             sS.helix.a.print("new helix parameters");
406         }
407 
408         CommonOps_DDRM.subtract(snS.helix.C, snP.helix.C, tempM);
409         CommonOps_DDRM.multTransB(tempM, tempA, Cinv);
410         CommonOps_DDRM.mult(tempA, Cinv, tempM);
411         CommonOps_DDRM.add(helix.C, tempM, sS.helix.C);
412         
413         if (debug) sS.print("Smoothed");
414         return sS;
415     }

417     // Return errors on the helix parameters at the global origin
418     Vec helixErrors(Vec aPrime) {
419         // aPrime are the helix parameters for a pivot at the global origin, assumed
420         // already to be calculated by pivotTransform()
421         DMatrixRMaj tC = covariancePivotTransform(aPrime);
422         return new Vec(Math.sqrt(tC.unsafe_get(0,0)), Math.sqrt(tC.unsafe_get(1,1)), Math.sqrt(tC.unsafe_get(2,2)), 
423                 Math.sqrt(tC.unsafe_get(3,3)), Math.sqrt(tC.unsafe_get(4,4)));
424     }

426     // Transform the helix covariance to new pivot point (specified in local coordinates)
427     DMatrixRMaj covariancePivotTransform(Vec aP) {
428         // aP are the helix parameters for the new pivot point, assumed already to be
429         // calculated by pivotTransform()
430         // Note that no field rotation is assumed or accounted for here
431         DMatrixRMaj mF = new DMatrixRMaj(5,5);
432         helix.makeF(aP, mF);
433         CommonOps_DDRM.multTransB(helix.C, mF, tempM);
434         CommonOps_DDRM.mult(mF, tempM, tempA);
435         return tempA;
436     }    
437     // Go to and from 1D EJML matrix for a vector Vec
438     static void vecToM(Vec a, DMatrixRMaj m) {
439         for (int i=0; i<a.N; ++i) {
440             m.unsafe_set(i, 0, a.v[i]);
441         }
442     }
443     static Vec mToVec(DMatrixRMaj M) {
444         return new Vec(M.unsafe_get(0, 0), M.unsafe_get(1, 0), M.unsafe_get(2, 0), M.unsafe_get(3, 0), M.unsafe_get(4, 0));
445     }
446     // Direct product of two row vectors to make a 2D matrix
447     private static void directProd(DMatrixRMaj a, DMatrixRMaj b, DMatrixRMaj c) {
448         for (int i=0; i<5; ++i) {
449             for (int j=0; j<5; ++j) {
450                 c.unsafe_set(i, j, a.unsafe_get(i, 0)*b.unsafe_get(j, 0));
451             }
452         }
453     }
