1   package org.hps.recon.tracking.kalman;
2   
3   import org.apache.commons.math.util.FastMath;
4   
5   /**
6    * Propagate a charged particle according to the magnetic field map by 4th order Runge Kutta integration.
7    * Note that the coordinate system is the Kalman-Filter-code system, as that is what the field-map routine
8    * called here assumes.
9    */
10  public class RungeKutta4 {
11  
12      private double h;
13      private double h2;
14      private double alpha;
15      private org.lcsim.geometry.FieldMap fM;
16  
17      public RungeKutta4(double Q, double dx, org.lcsim.geometry.FieldMap fM) {
18          alpha = Q * 2.99792458e-4; // Q is the charge in units of the proton charge
19          h = dx; // Step size in mm
20          h2 = h / 2.0;
21          this.fM = fM; // Magnetic field map
22      }
23  
24      double[] integrate(Vec r0, Vec p0, double s) {
25          // r0 is the initial point in mm
26          // p0 is the initial momentum in GeV/c
27          // s is the distance to propagate (approximate to distance dx)
28          double[] r = { r0.v[0], r0.v[1], r0.v[2], p0.v[0], p0.v[1], p0.v[2] };
29          double[] k1 = new double[6];
30          double[] k2 = new double[6];
31          double[] k3 = new double[6];
32          double[] k4 = new double[6];
33          int nStep = (int) (s / h) + 1;
34          for (int step = 0; step < nStep; step++) {
35              Vec ri = new Vec(r[0], r[1], r[2]);
36              double[] pi = { r[3], r[4], r[5] };
37              k1 = f(ri, pi);
38              Vec r1 = new Vec(r[0] + h2 * k1[0], r[1] + h2 * k1[1], r[2] + h2 * k1[2]);
39              double[] p1 = { r[3] + h2 * k1[3], r[4] + h2 * k1[4], r[5] + h2 * k1[5] };
40              k2 = f(r1, p1);
41              Vec r2 = new Vec(r[0] + h2 * k2[0], r[1] + h2 * k2[1], r[2] + h2 * k2[1]);
42              double[] p2 = { r[3] + h2 * k2[3], r[4] + h2 * k2[4], r[5] + h2 * k2[5] };
43              k3 = f(r2, p2);
44              Vec r3 = new Vec(r[0] + h * k3[0], r[1] + h * k3[1], r[2] + h * k3[2]);
45              double[] p3 = { r[3] + h * k3[3], r[4] + h * k3[4], r[5] + h * k3[5] };
46              k4 = f(r3, p3);
47              for (int i = 0; i < 6; i++) { r[i] = r[i] + h * (k1[i] / 6. + k2[i] / 3. + k3[i] / 3. + k4[i] / 6.); }
48          }
49          return r;
50      }
51  
52      private double[] f(Vec x, double[] p) { // Return all the derivatives
53          double [] B = KalmanInterface.getFielD(x, fM);  // This field routine assumes the Kalman-Filter coordinate system.
54          double[] d = new double[6];
55          double pmag = FastMath.sqrt(p[0] * p[0] + p[1] * p[1] + p[2] * p[2]);
56          // System.out.format("P magnitude = %10.7f GeV\n", pmag);
57          d[0] = p[0] / pmag; // dx/ds 
58          d[1] = p[1] / pmag;
59          d[2] = p[2] / pmag;
60          d[3] = alpha * (d[1] * B[2] - d[2] * B[1]); // dp/ds
61          d[4] = alpha * (d[2] * B[0] - d[0] * B[2]);
62          d[5] = alpha * (d[0] * B[1] - d[1] * B[0]);
63          return d;
64      }
