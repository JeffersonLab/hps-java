45      HelixState(Vec a, Vec X0, Vec origin, DMatrixRMaj C, double B, Vec tB) {
46          logger = Logger.getLogger(HelixState.class.getName());
47          this.a = a;
48          this.X0 = X0;
49          this.origin = origin;
50          this.C = C;
51          this.B = B;
52          this.tB = tB;
53          hpi = new HelixPlaneIntersect();
54          c = 2.99793e8; // Speed of light in m/s
55          alpha = 1.0e12 / (c * B); // Convert from pt in GeV to curvature in mm
56          Vec yhat = new Vec(0., 1.0, 0.);
57          Vec u = yhat.cross(tB).unitVec();
58          Vec v = tB.cross(u);
59          Rot = new RotMatrix(u, v, tB);
60      }

112     static Vec atPhi(Vec X0, Vec a, double phi, double alpha) {
113         double x = X0.v[0] + (a.v[0] + (alpha / a.v[2])) * FastMath.cos(a.v[1]) - (alpha / a.v[2]) * FastMath.cos(a.v[1] + phi);
114         double y = X0.v[1] + (a.v[0] + (alpha / a.v[2])) * FastMath.sin(a.v[1]) - (alpha / a.v[2]) * FastMath.sin(a.v[1] + phi);
115         double z = X0.v[2] + a.v[3] - (alpha / a.v[2]) * phi * a.v[4];
116         return new Vec(x, y, z);
117     }

137     static void makeF(Vec aP, DMatrixRMaj F, Vec a, double alpha) {
138         F.unsafe_set(0, 0, FastMath.cos(aP.v[1] - a.v[1]));
139         F.unsafe_set(0, 1, (a.v[0] + alpha / a.v[2]) * FastMath.sin(aP.v[1] - a.v[1]));
140         F.unsafe_set(0, 2, (alpha / (a.v[2] * a.v[2])) * (1.0 - FastMath.cos(aP.v[1] - a.v[1])));
141         F.unsafe_set(1, 0, -FastMath.sin(aP.v[1] - a.v[1]) / (aP.v[0] + alpha / a.v[2]));
142         F.unsafe_set(1, 1, (a.v[0] + alpha / a.v[2]) * FastMath.cos(aP.v[1] - a.v[1]) / (aP.v[0] + alpha / a.v[2]));
143         F.unsafe_set(1, 2, (alpha / (a.v[2] * a.v[2])) * FastMath.sin(aP.v[1] - a.v[1]) / (aP.v[0] + alpha / a.v[2]));
144         F.unsafe_set(2, 2,  1.0);
145         F.unsafe_set(3, 0, (alpha / a.v[2]) * a.v[4] * FastMath.sin(aP.v[1] - a.v[1]) / (aP.v[0] + alpha / a.v[2]));
146         F.unsafe_set(3, 1, (alpha / a.v[2]) * a.v[4] * (1.0 - (a.v[0] + alpha / a.v[2]) * FastMath.cos(aP.v[1] - a.v[1]) / (aP.v[0] + alpha / a.v[2])));
147         F.unsafe_set(3, 2, (alpha / (a.v[2] * a.v[2])) * a.v[4]
148                 * (aP.v[1] - a.v[1] - (alpha / a.v[2]) * FastMath.sin(aP.v[1] - a.v[1]) / (aP.v[0] + alpha / a.v[2])));
149         F.unsafe_set(3, 3, 1.0);
150         F.unsafe_set(3, 4, -(alpha / a.v[2]) * (aP.v[1] - a.v[1]));
151         F.unsafe_set(4, 4, 1.0);
152 
153         // All other values are always zero
154     }

162     static Vec getMom(double phi, Vec a) {
163         double px = -FastMath.sin(a.v[1] + phi) / Math.abs(a.v[2]);
164         double py = FastMath.cos(a.v[1] + phi) / Math.abs(a.v[2]);
165         double pz = a.v[4] / Math.abs(a.v[2]);
166         return new Vec(px, py, pz);
167     }

169     // Momentum at the start of the given helix (point closest to the pivot)
170     static Vec aTOp(Vec a) {
171         double px = -FastMath.sin(a.v[1]) / Math.abs(a.v[2]);
172         double py = FastMath.cos(a.v[1]) / Math.abs(a.v[2]);
173         double pz = a.v[4] / Math.abs(a.v[2]);
174         return new Vec(px, py, pz);
175     }

177     // Transform from momentum at helix starting point back to the helix parameters
178     // drho and dz are not modified
179     static Vec pTOa(Vec p, double drho, double dz, double Q) {
180         double phi0 = FastMath.atan2(-p.v[0], p.v[1]);
181         double K = Q / FastMath.sqrt(p.v[0] * p.v[0] + p.v[1] * p.v[1]);
182         double tanl = p.v[2] / FastMath.sqrt(p.v[0] * p.v[0] + p.v[1] * p.v[1]);
183 
184         return new Vec(drho, phi0, K, dz, tanl);
185     }

187     // To transform a space point from global to local field coordinates, first subtract

195     // the inverse of <Rot> and then add the <origin>.

197         Vec xGlobal = Rot.inverseRotate(xLocal).sum(origin);
198         return xGlobal;

201     // Transformation of helix parameters from one B-field frame to another, by rotation R
202     // Warning: the pivot point has to be transformed too! Here we assume that the new pivot point
203     // will be on the helix at phi=0, so drho and dz will always be returned as zero. Therefore, before
204     // calling this routine, make sure that the current pivot point is on the helix (drho=dz=0)
205     static Vec rotateHelix(Vec a, RotMatrix R, DMatrixRMaj fRot) {
206         // The rotation is easily applied to the momentum vector, so first we transform from helix parameters
207         // to momentum, apply the rotation, and then transform back to helix parameters.
208         // The values for fRot, the corresponding derivative matrix, are also calculated and returned.
209 
210         final boolean debug = false;
211 
212         Vec phlx = aTOp(a); // Momentum at point closest to helix
213         Vec p_prime = R.rotate(phlx); // Momentum in new coordinate system
214         if (debug) {
215             a.print("input helix, in rotateHelix");
216             R.print("in rotateHelix");
217             phlx.print("momentum vector in rotateHelix");
218             p_prime.print("rotated momentum vector in rotateHelix");
219             System.out.format("in rotateHelix: p=%10.7f,  prot=%10.7f\n", phlx.mag(), p_prime.mag());
220         }
221 
222         double dpda00 = -FastMath.cos(a.v[1]) / Math.abs(a.v[2]);
223         double dpda01 = FastMath.sin(a.v[1]) / (a.v[2] * Math.abs(a.v[2]));
224         double dpda10 = -FastMath.sin(a.v[1]) / Math.abs(a.v[2]);
225         double dpda11 = -FastMath.cos(a.v[1]) / (a.v[2] * Math.abs(a.v[2]));
226         double dpda21 = -a.v[4] / (a.v[2] * Math.abs(a.v[2]));
227         double dpda22 = 1. / Math.abs(a.v[2]);
228         DMatrix3x3 dpda = new DMatrix3x3(dpda00,dpda01,0., dpda10,dpda11,0., 0.,dpda21,dpda22);
229 
230         DMatrix3x3 dprimedp = new DMatrix3x3(R.M[0][0],R.M[0][1],R.M[0][2], R.M[1][0],R.M[1][1],R.M[1][2], R.M[2][0],R.M[2][1],R.M[2][2]);
231 
232         double Q = Math.signum(a.v[2]);
233         double pt2 = p_prime.v[0] * p_prime.v[0] + p_prime.v[1] * p_prime.v[1];
234         double pt = FastMath.sqrt(pt2);
235         
236         double dadp00 = -p_prime.v[1] / pt2;
237         double dadp01 = p_prime.v[0] / pt2;
238         double dadp10 = -Q * p_prime.v[0] / (pt2 * pt);
239         double dadp11 = -Q * p_prime.v[1] / (pt2 * pt);
240         double dadp20 = -p_prime.v[0] * p_prime.v[2] / (pt2 * pt);
241         double dadp21 = -p_prime.v[1] * p_prime.v[2] / (pt2 * pt);
242         double dadp22 = 1.0 / (pt);
243         DMatrix3x3 dadp = new DMatrix3x3(dadp00,dadp01,0., dadp10,dadp11,0., dadp20, dadp21, dadp22);
244 
245         DMatrix3x3 c = new DMatrix3x3();
246         CommonOps_DDF3.mult(dprimedp, dpda, c);
247         DMatrix3x3 prod = new DMatrix3x3();
248         CommonOps_DDF3.mult(dadp,c,prod);
249 
250         fRot.zero();
251         fRot.unsafe_set(0, 0, 1.0);
252         fRot.unsafe_set(1, 1, prod.unsafe_get(0, 0));
253         fRot.unsafe_set(1, 2, prod.unsafe_get(0, 1));
254         fRot.unsafe_set(1, 4, prod.unsafe_get(0, 2));
255         fRot.unsafe_set(2, 1, prod.unsafe_get(1, 0));
256         fRot.unsafe_set(2, 2, prod.unsafe_get(1, 1));
257         fRot.unsafe_set(2, 4, prod.unsafe_get(1, 2));
258         fRot.unsafe_set(3, 3, 1.0);
259         fRot.unsafe_set(4, 1, prod.unsafe_get(2, 0));
260         fRot.unsafe_set(4, 2, prod.unsafe_get(2, 1));
261         fRot.unsafe_set(4, 4, prod.unsafe_get(2, 2));
262 
263         /*
264          * if (verbose) { // derivative test Vec da = a.scale(0.005); Vec apda =
265          * a.sum(da); Vec ap = pTOa(R.rotate(aTOp(a)), a); Vec apdap =
266          * pTOa(R.rotate(aTOp(apda)), apda); Vec dap = apdap.dif(ap); Vec dap2 =
267          * da.leftMultiply(fRot);
268          * System.out.format("StateVector:rotateHelix: derivative test:\n");
269          * dap.print("actual difference in helix parameters");
270          * dap2.print("diff in helix params from derivatives"); }
271          */
272         // The parameters drho and dz are assumed not to change in the rotation. This will be correct
273         // if the rotation is about the pivot point, and the pivot is on the helix, in which case
274         // dz and drho are both zero. That is not necessarily valid if dz and drho are not zero. If the
275         // rotation were only about the z axis it would be, but any other sort of rotation will mix dz and drho.
276         //if (Math.abs(a.v[0]) > 0.001 || Math.abs(a.v[3]) > 0.001) {
277         //    System.out.format("StateVector.rotateHelix: warning, dz=%10.5f and drho=%10.5f are not zero.\n",a.v[3],a.v[0]);
278         //}
279         Vec aNew = pTOa(p_prime, a.v[0], a.v[3], Q);
280         if (debug) { aNew.print("rotated helix in rotateHelix"); }
281         return aNew;
282     }

301     static Vec pivotTransform(Vec pivot, Vec a, Vec X0, double alpha, double deltaEoE) {
302         double K = a.v[2] * (1.0 - deltaEoE); // Lose energy before propagating
303         double xC = X0.v[0] + (a.v[0] + alpha / K) * FastMath.cos(a.v[1]); // Center of the helix circle
304         double yC = X0.v[1] + (a.v[0] + alpha / K) * FastMath.sin(a.v[1]);
305         // if (verbose) System.out.format("pivotTransform center=%13.10f, %13.10f\n",
306         // xC, yC);
307 
308         // Predicted state vector
309         double[] aP = new double[5];
310         aP[2] = K;
311         aP[4] = a.v[4];
312         if (K > 0) {
313             aP[1] = FastMath.atan2(yC - pivot.v[1], xC - pivot.v[0]);
314         } else {
315             aP[1] = FastMath.atan2(pivot.v[1] - yC, pivot.v[0] - xC);
316         }
317         aP[0] = (xC - pivot.v[0]) * FastMath.cos(aP[1]) + (yC - pivot.v[1]) * FastMath.sin(aP[1]) - alpha / K;
318         aP[3] = X0.v[2] - pivot.v[2] + a.v[3] - (alpha / K) * (aP[1] - a.v[1]) * a.v[4];
319 
320         // xC = pivot.v[0] + (aP[0]+alpha/aP[2])*FastMath.cos(aP[1]);
321         // yC = pivot.v[1] + (aP[0]+alpha/aP[2])*FastMath.sin(aP[1]);
322 
323         return new Vec(5, aP);
324     }

326     // Propagate a helix by Runge-Kutta integration to an arbitrary plane
327     HelixState propagateRungeKutta(Plane pln, ArrayList<Double> yScat, ArrayList<Double> XL, org.lcsim.geometry.FieldMap fM, double [] arcLength) {
328         // pln   = plane to where the extrapolation is taking place in global coordinates.  
329         //         The origin of pln will be the new helix pivot point in global coordinates and the origin of the B-field system.
330         // yScat = input array of y values where scattering in silicon will take place. Only those between the start and finish points
331         //         will be used, so including extras will just waste a bit of CPU time. Silicon is assumed to lie in a plane
332         //         perpendicular to the beam axis at each of these yScat values.
333         // XL    = silicon thickness in radiation lengths at each of the scattering planes
334         // fM    = HPS field map
335         // return value = helix state at the new pivot. These helix parameters are valid in the B-field coordinate system with
336         //                origin at the pivot point and z axis in the direction of the B-field at the pivot.
337         
338         final boolean debug = false;
339         
340         // Take the B-field reference frame to be at the position X of the plane. The returned helix will be in this frame.
341         Vec B = KalmanInterface.getField(pln.X(), fM);
342         double Bmag = B.mag();
343         double alphatarget = 1.0e12 / (c * Bmag);
344         Vec tB = B.unitVec(Bmag);
345         Vec yhat = new Vec(0., 1.0, 0.);
346         Vec uB = yhat.cross(tB).unitVec();
347         Vec vB = tB.cross(uB);
348         RotMatrix targetRot = new RotMatrix(uB, vB, tB); // Rotation from the global system into the B-field system at the plane
349         
350         // Target plane after transforming to the B-field coordinate system 
351         Plane targetPlane = pln.toLocal(targetRot, pln.X());    
352         
353         // Point and momentum on the helix in the B-field system near the tracking layer from where we extrapolate
354         Vec x0Local = atPhi(0.);
355         Vec p0Local = getMom(0.);
356         Vec x0Global = toGlobal(x0Local);
357         Vec p0Global = Rot.inverseRotate(p0Local);
358 
359         double Q = Math.signum(a.v[2]);
360 
361         if (debug) {
362             System.out.format("\nHelixState.propagateRungeKutta, Q=%8.1f, origin=%10.5f %10.5f %10.5f:\n", Q, origin.v[0], origin.v[1],
363                     origin.v[2]);
364             System.out.format("    At final plane B=%10.5f, t=%10.6f %10.6f %10.6f\n", Bmag, tB.v[0], tB.v[1], tB.v[2]);
365             System.out.format("    alpha=%10.6f,  alpha at final plane=%10.6f\n", alpha, alphatarget);
366             targetRot.print("to Bfield system at final plane");
367             targetPlane.print("at final plane in B-field system");
368             origin.print("origin of local detector system in global coordinates");
369             X0.print("helix pivot");
370             a.print("local helix parameters at initial point");
371             x0Local.print("point on helix, local at initial point");
372             x0Global.print("point on helix, global at initial point");
373             p0Local.print(String.format("helix momentum, local initial point, p=%10.6f",p0Local.mag()));
374             p0Global.print(String.format("helix momentum, global initial point, p=%10.6f",p0Global.mag()));
375         }
376         
377         Vec pInt = new Vec(3);
378         Vec xPlane = hpi.rkIntersect(pln, x0Global, p0Global, Q, fM, pInt); // RK propagation to the target plane
379         arcLength[0] = hpi.arcLength();
380         Vec XplaneLocal = targetRot.rotate(xPlane.dif(pln.X()));
381         Vec helixAtIntersect = pTOa(targetRot.rotate(pInt), 0., 0., Q); // Helix with pivot at Xplane (in field coordinates)
382         Vec helixAtTarget = pivotTransform(targetPlane.X(), helixAtIntersect, XplaneLocal, alphatarget, 0.);
383         if (debug) {
384             xPlane.print("RK helix intersection with final plane");
385             System.out.format("    Arc length for the RK propatation=%9.4f\n", arcLength[0]);
386             XplaneLocal.print("RK helix intersection with final plane in local system");
387             pInt.print("RK momentum at helix intersection");
388             double pTanL = pInt.v[2]/FastMath.sqrt(pInt.v[0]*pInt.v[0]+pInt.v[1]*pInt.v[1]);
389             Vec pLoc = targetRot.rotate(pInt);
390             double pTanLlocal = pLoc.v[2]/FastMath.sqrt(pLoc.v[0]*pLoc.v[0]+pLoc.v[1]*pLoc.v[1]);
391             System.out.format("from momentum, global tan(lambda)=%10.6f, local tan(lambda)=%10.6f\n",pTanL,pTanLlocal);
392             helixAtIntersect.print("helix at final-plane intersection");
393             helixAtTarget.print("helix with pivot at final plane");
394             pln.X().print("origin of final field system in global system");
395             Rot.print("rotation matrix from global to final field system");
396         }
397 
398         // The covariance matrix is transformed assuming a sequence of pivot transforms (not Runge Kutta)
399         double stepSize = 20.0;
400         // Step from the origin of this StateVector to pln.X(), both in global coordinates
401         Vec transHelix = new Vec(5);
402         DMatrixRMaj newCovariance = new DMatrixRMaj(5,5);
403         if (!helixStepper(stepSize, yScat, XL, newCovariance, transHelix, pln.X(), fM)) {
404             newCovariance.set(this.C);
405         } else if (debug) {
406             transHelix.print("helixStepper helix at final plane");
407             System.out.println("    Errors: ");
408             for (int i = 0; i < 5; ++i) { System.out.format(" %10.7f", FastMath.sqrt(C.get(i,i))); }
409             System.out.println("\n");
410             newCovariance.print("transformed covariance");
411             System.out.println("    Errors: ");
412             for (int i = 0; i < 5; ++i) { System.out.format(" %10.7f", FastMath.sqrt(newCovariance.get(i,i))); }
413             System.out.println("\n");
414         }
415         
416         HelixState newHelixState = new HelixState(helixAtTarget, new Vec(0.,0.,0.), pln.X(), newCovariance, Bmag, tB);
417         newHelixState.xPlaneRK = xPlane;
418         return newHelixState;
419     }

421     // Optional interface for the case in which there are no scattering planes

431     static double projMSangle(double p, double XL) {
432         if (XL <= 0.) return 0.;
433         return (0.0136 / Math.abs(p)) * FastMath.sqrt(XL) * (1.0 + 0.038 * FastMath.log(XL));
434     }

436     boolean helixStepper(double maxStep, ArrayList<Double> yScat, ArrayList<Double> XL, DMatrixRMaj Covariance, Vec finalHelix, Vec newOrigin, org.lcsim.geometry.FieldMap fM) {
437         // The old and new origin points are in global coordinates. The old helix and old pivot are defined
438         // in a coordinate system aligned with the field and centered at the old origin. The returned
439         // helix will be in a coordinate system aligned with the local field at the new origin, and the
440         // new pivot point will be at the new origin, in global coordinates, although to use it one should transform to the field frame, which
441         // is defined with its origin at newOrigin and aligned there with the local field.        
442         // All scattering layers are assumed to be of the same thickness XL, in radiation lengths
443         // We assume that the starting StateVector is at a layer with a hit, in which case the Kalman filter has already accounted for
444         // multiple scattering at that layer.       
445                 
446         final boolean debug = false;
447         
448         double tol = 2.0;  // Tolerance in mm to determine whether a location is on a scattering plane. 
449         if (maxStep < tol) tol = maxStep/2.0;
450         int nSteps = (int)(Math.abs(newOrigin.v[1] - this.origin.v[1])/maxStep);
451         if (nSteps < 1) nSteps = 1;
452         ArrayList<Pair<Double,Double>> stepPnts = new ArrayList<Pair<Double,Double>>(nSteps+yScat.size());
453         double yDistance = newOrigin.v[1] - this.origin.v[1];
454         double yStep = yDistance/(double)nSteps;
455         
456         //stepPnts.add(new Pair<Double,Double>(this.origin.v[1],0.)); 
457         double yNext = this.origin.v[1];
458         double dir = Math.signum(yStep);
459         for (int i=0; i<nSteps; ++i) {
460             double yLast = yNext;
461             yNext += yStep;
462             double XLnext = 0.;
463             for (int j=0; j<yScat.size(); ++j) {
464                 double y = yScat.get(j);
465                 if (dir*y > dir*yLast + tol && dir*y < dir*yNext - tol) { // Add intermediate scattering layer
466                     Pair<Double,Double> newLayer = new Pair<Double,Double>(y,XL.get(j));
467                     stepPnts.add(newLayer);
468                 } else if (y >= yNext - tol && y <= yNext + tol) {
469                     XLnext = XL.get(j);
470                 }
471             }
472             stepPnts.add(new Pair<Double,Double>(yNext,XLnext));         
473         }
474         if (dir > 0.) {
475             Collections.sort(stepPnts, HelixState.pairComparator);
476         } else {
477             Collections.sort(stepPnts, HelixState.pairComparator.reversed());
478         }
479         
480         double localAlpha = alpha;
481 
482         RotMatrix RM = Rot;
483         Vec newHelix = this.a.copy();
484         Vec Pivot = this.X0.copy();
485         Vec Origin = this.origin.copy(); // In global coordinates
486         DMatrixRMaj fRot = new DMatrixRMaj(5,5);
487         DMatrixRMaj Cov = this.C.copy();
488         Vec pMom = getMom(0.,newHelix);
489         double momentum = pMom.mag();
490         double ct = pMom.v[1]/momentum;
491         double thisXL = 0.;
492         // Account for scattering in the silicon layer from where we are starting if going in the beam direction.
493         // When going toward the target, the Kalman smoother has already accounted for scattering in the first layer.
494         if (yDistance > 0.) {
495             for (int i=0; i<yScat.size(); ++i) {
496                 double y = yScat.get(i);
497                 if (Math.abs(this.origin.v[1]-y) < tol) {
498                     thisXL = XL.get(i);                    // Find the amount of material at the starting layer
499                 }
500             }
501         }
502         double sigmaMS = projMSangle(momentum, thisXL/ct);
503         DMatrixRMaj Q = new DMatrixRMaj(5,5);
504         this.getQ(sigmaMS, Q);
505         Vec yhat = new Vec(0., 1., 0.);
506         if (debug) {
507             System.out.format("Entering helixStepper for %d steps, B=%10.7f, B direction=%10.7f %10.7f %10.7f\n", 
508                     stepPnts.size(), B, RM.M[2][0],RM.M[2][1], RM.M[2][2]);
509             if (yScat.size() == 0) {
510                 System.out.format("     No scattering layers provided\n");
511             }
512             for (double y : yScat) {
513                 System.out.format("    scattering at y=%9.4f with XL=%8.4f\n", y, XL.get(yScat.indexOf(y)));
514             }
515             for (Pair<Double,Double> step : stepPnts) {
516                 double y = step.getFirstElement();
517                 double XLScat = step.getSecondElement();
518                 System.out.format("  stepping layer at y=%10.5f, R.L. scatter = %8.4f\n", y, XLScat);
519             }
520             this.origin.print("old origin");
521             this.X0.print("old pivot");
522             this.a.print("old helix");
523             newOrigin.print("new origin");
524             RM.print("to transform to the local field frame");
525             Plane pln = new Plane(newOrigin, yhat);
526             Plane plnLocal = pln.toLocal(RM, Origin);
527             double dphi = hpi.planeIntersect(newHelix, Pivot, localAlpha, plnLocal); // Find the helix intersection with the plane
528             Vec newPivot = atPhi(Pivot, newHelix, dphi, localAlpha);
529             newPivot.print("new pivot in origin plane");
530             Vec newHelix0 = pivotTransform(newPivot, newHelix, Pivot, localAlpha, 0.);
531             newHelix0.print("Pivot transform to final plane in a single step");
532         }
533         CommonOps_DDRM.addEquals(Cov,Q);
534         DMatrixRMaj cIntermediate = new DMatrixRMaj(5,5);
535         DMatrixRMaj Ft = new DMatrixRMaj(5,5);
536         DMatrixRMaj F = new DMatrixRMaj(5,5);
537         for (int step = 0; step < stepPnts.size(); ++step) {
538             Pair<Double, Double> thisStep = stepPnts.get(step);
539             double yInt = thisStep.getFirstElement();
540             thisXL = thisStep.getSecondElement();
541             Plane pln = new Plane(new Vec(0., yInt, 0.), yhat); // Make a plane in global coordinates, perpendicular to the y axis
542             Plane plnLocal = pln.toLocal(RM, Origin); // Transform the plane to local coordinates
543             double dphi = hpi.planeIntersect(newHelix, Pivot, localAlpha, plnLocal); // Find the helix intersection with the plane
544             if (Double.isNaN(dphi)) {
545                 logger.log(Level.FINE, String.format("No intersection with the plane at step=%d\n", step));
546                 return false;
547             }
548             Vec newPivot = atPhi(Pivot, newHelix, dphi, localAlpha);
549             Vec newHelixPivoted = pivotTransform(newPivot, newHelix, Pivot, localAlpha, 0.); // Transform the helix pivot to the intersection point
550             if (debug) {
551                 System.out.format("Step %d, y=%8.3f, dphi=%10.7f, alpha=%10.7e\n", step, yInt, dphi, localAlpha);
552                 pln.print("to intersect, in global coordinate");
553                 plnLocal.print("to intersect, in local field coordinates");
554                 Pivot.print("old pivot");
555                 newPivot.print("new pivot at intersection of helix with plane");
556                 // pln.print("target");
557                 // plnLocal.print("transformed to local coordinates");
558                 newHelix.print("new helix after pivot transform");
559                 dphi = hpi.planeIntersect(newHelixPivoted, newPivot, localAlpha, plnLocal);
560                 System.out.format("    New delta-phi=%13.10f; should be zero!\n", dphi);
561                 Vec newPoint = atPhi(newPivot, newHelixPivoted, dphi, localAlpha);
562                 newPoint.print("new point of intersection, should be same as the old");
563             }           
564             makeF(newHelixPivoted, F, newHelix, localAlpha);
565             newHelix = newHelixPivoted;
566 
567             // Rotate the helix into the field system at the new origin
568             Origin = RM.inverseRotate(newPivot).sum(Origin); // Make a new coordinate system with origin in global coords at the intersection point
569             Vec Bfield = null;
570             if (step == stepPnts.size()-1) {   // At the last step define the local field at the point provided by the calling routine
571                 Bfield = KalmanInterface.getField(newOrigin, fM);
572             } else {
573                 Bfield = KalmanInterface.getField(Origin, fM);
574             }     
575             double Bmag = Bfield.mag();
576             Vec tB = Bfield.unitVec(Bmag); // Local field at the new origin
577             Vec uB = yhat.cross(tB).unitVec();
578             Vec vB = tB.cross(uB);
579             RotMatrix RMnew = new RotMatrix(uB, vB, tB);
580             localAlpha = 1.0e12 / c / Bmag;
581             RotMatrix deltaRM = RMnew.multiply(RM.invert());    // New coordinate system is rotated to align with local field
582             newHelix = rotateHelix(newHelix, deltaRM, fRot);    // Rotate the helix into the new local field coordinate system
583             CommonOps_DDRM.mult(F, fRot, Ft);
584             CommonOps_DDRM.multTransB(Cov, Ft, cIntermediate);   // Similarity transformation to propagate the covariance matrix
585             CommonOps_DDRM.mult(Ft,cIntermediate,Cov);
586             // Add in multiple scattering if we are here passing through a plane with material
587             if (thisXL > 0.) {                
588                 pMom = getMom(0.,newHelix);
589                 momentum = pMom.mag();
590                 ct = pMom.v[1]/momentum;
591                 sigmaMS = projMSangle(momentum, thisXL/ct);
592                 this.getQ(sigmaMS, Q);
593                 CommonOps_DDRM.addEquals(Cov, Q);
594             }
595 
596             // In the next step the pivot will be right at the origin of the local system, except for the last step
597             Pivot.v[0] = 0.;
598             Pivot.v[1] = 0.;
599             Pivot.v[2] = 0.;
600             if (debug) {
601                 System.out.format("  helixStepper after step %d, B=%10.7f", step, Bmag);
602                 RMnew.print("new rotation to field frame");
603                 deltaRM.print("rotation to field system");
604                 Origin.print("intermediate origin in global system");
605                 newHelix.print("new helix after rotation");
606             }
607             RM = RMnew;
608         }
609         // Finally, move the pivot to the requested point in the final plane
610         Vec newOriginLocal = new Vec(0.,0.,0.);
611         Vec oldPivot = RM.rotate(Origin.dif(newOrigin));
612         Vec finalHx = pivotTransform(newOriginLocal, newHelix, oldPivot, localAlpha, 0.);
613         makeF(finalHx, F, newHelix, localAlpha);
614         CommonOps_DDRM.multTransB(Cov, F, cIntermediate); 
615         CommonOps_DDRM.mult(F,cIntermediate,Cov);
616         if (debug) {
617             Origin.print("helixStepper final origin and pivot on helix");
618             oldPivot.print("  final origin in final local coordinate system");
619             RM.print("helixStepper final rotation to field system");
620             newOriginLocal.print("final origin on helix in local coordinates");
621             newHelix.print("helix before pivot transform");
622             finalHx.print("final helix");
623         }
624         Covariance.set(Cov);
625 
626         return true;
627     }

660     // Multiple scattering matrix; assume a single thin scattering layer at the
661     // beginning of the helix propagation
662     void getQ(double sigmaMS, DMatrixRMaj Q) {
663         double V = sigmaMS * sigmaMS;
664         Q.unsafe_set(1, 1,  V * (1.0 + a.v[4] * a.v[4]));
665         Q.unsafe_set(4, 4, V * (1.0 + a.v[4] * a.v[4]) * (1.0 + a.v[4] * a.v[4]));
666         // All other elements are always zero, so don't need to be reset
667     }
