11  /**
12   * Description of a single silicon-strip module, and a container for its hits
13   */
14  class SiModule {
15      int Layer; // Tracker layer number, or a negative integer for a dummy layer added just for stepping in a
16                 // non-uniform field
17      int detector; // Detector number within the layer
18      int millipedeID; // ID used by millipede for alignment
19      ArrayList<Measurement> hits; // Hits ordered by coordinate value, from minimum to maximum
20      Plane p; // Orientation and offset of the detector measurement plane in global coordinates 
21               // The offset should be the location of the center of the detector in global
22               // coordinates
23      double[] xExtent; // Plus and minus limits on the detector active area in the x direction (along
24                        // the strips)
25      double[] yExtent; // Plus and minus limits on the detector active area in the y direction
26                        // (perpendicular to the strips)
27      boolean split;    // True if the strips are split into two channels at the middle
28      RotMatrix R; // Rotation from the detector coordinates to global coordinates (not field coordinates)
29      RotMatrix Rinv; // Rotation from global (not field) coordinates to detector coordinates (transpose of R)
30                      // The local coordinate system is u, v, t where t is more-or-less the beam direction (y-global)
31                      // and v is the measurement direction.
32      double thickness; // Silicon thickness in mm (should be 0 for a dummy layer!)
33      int topBottom;    // 0 for bottom tracker, 1 for top tracker
34      org.lcsim.geometry.FieldMap Bfield;
35      boolean isStereo;
36  
37      private boolean verbose;
38      private Logger logger;

40      SiModule(int Layer, Plane p, double stereo, double width, double height, boolean split, double thickness, org.lcsim.geometry.FieldMap Bfield) {
41          // for backwards-compatibility with old stand-alone development code: assume axial
42          // layers have stereo angle=0
43          this(Layer, p, stereo != 0.0, width, height, split, thickness, Bfield, 0, 0, 0);
44          topBottom = 1;
45          if (p.X().v[2]>0) topBottom = 0;
46      }

62      SiModule(int Layer, Plane p, boolean isStereo, double width, double height, boolean split, double thickness,
63              org.lcsim.geometry.FieldMap Bfield, int detector, int millipedeID, int topBottom) {
64          this.topBottom = topBottom;
65          logger = Logger.getLogger(SiModule.class.getName());
66          verbose = (logger.getLevel()==Level.FINE);
67          this.millipedeID = millipedeID;
68          if (verbose) { 
69              System.out.format("SiModule constructor called with layer = %d, detector module = %d, y=%8.2f\n", Layer, detector, p.X().v[1]);
70              p.print("of SiModule");
71          }
72          Vec BOnAxis = KalmanInterface.getField(new Vec(0.,p.X().v[1],0.), Bfield);
73          Vec BatCenter = KalmanInterface.getField(p.X(), Bfield);
74          if (verbose) {
75              BOnAxis.print("B field on axis");
76              BatCenter.print("B at detector center");
77          }
78          this.split = split;
79          this.Layer = Layer;
80          this.detector = detector;
81          this.Bfield = Bfield;
82          this.p = p;
83          this.isStereo = isStereo;
84          this.thickness = thickness;
85          xExtent = new double[2];
86          xExtent[0] = -width / 2.0;
87          xExtent[1] = width / 2.0;
88          yExtent = new double[2];
89          yExtent[0] = -height / 2.0;
90          yExtent[1] = height / 2.0;
91          RotMatrix R1 = new RotMatrix(p.U(), p.V(), p.T());
92          Rinv = R1; // This goes from global to local
93          R = Rinv.invert(); // This goes from local to global
94          hits = new ArrayList<Measurement>();
95      }

97      void print(String s) {
98          System.out.format("%s",this.toString(s));
99      }

101     String toString(String s) {
102         boolean top = topBottom == 1;
103         String str = String.format(
104                 "Si module %s, Layer=%2d, Detector=%2d, Top=%b, Millipede=%d, stereo=%b, thickness=%8.4f mm, x extents=%10.6f %10.6f, y extents=%10.6f %10.6f\n",
105                 s, Layer, detector, top, millipedeID, isStereo, thickness, xExtent[0], xExtent[1], yExtent[0], yExtent[1]);
106         if (isStereo) {
107             str = str + String.format("This is a stereo detector layer");
108         } else {
109             str = str + String.format("This is an axial detector layer");
110         }
111         str = str + p.X().toString("origin of Si layer coordinates in the global system");
112         if (split) str = str + "  The strips are split at the detector center.\n";
113         else str = str + "\n";
114         Pair<Integer, Integer> IDdecode = TrackUtils.getLayerSide(topBottom, millipedeID);
115         str = str + String.format("From the millipede ID, lyr=%d det=%d\n",IDdecode.getFirst()-1, IDdecode.getSecond());
116         Vec Bf = KalmanInterface.getField(p.X(), Bfield);
117         Vec tBf = Bf.unitVec();
118         str = str + String.format("      At this origin, B=%10.6f Tesla with direction = %10.7f %10.7f %10.7f\n",Bf.mag(),tBf.v[0],tBf.v[1],tBf.v[2]);
119         str = str + R.toString("from detector coordinates to global coordinates");
120         str = str + String.format("List of measurements for Si module %s:\n", s);
121         Iterator<Measurement> itr = hits.iterator();
122         while (itr.hasNext()) {
123             Measurement m = itr.next();
124             str = str + m.toString(" ");
125         }
126         return str;
127     }
128     public String toString() {
129         boolean top = topBottom == 1;
130         String str = String.format("Si Module: Lyr=%2d Det=%2d Top=%b Mpd=%d split=%b stereo=%b pnt=%8.3f %8.3f %8.3f t=%7.3f %7.3f %7.3f\n",
131                 Layer, detector, top, millipedeID, split, isStereo, p.X().v[0], p.X().v[1], p.X().v[2], p.T().v[0], p.T().v[1], p.T().v[2]);
132         str = str +  String.format("           thick=%8.4f mm, x ext=%8.4f %8.4f, y ext=%8.4f %8.4f\n", 
133                 thickness, xExtent[0], xExtent[1], yExtent[0], yExtent[1]);
134         return str;
135     }

137     // Delete all the existing hits
138     void reset() {
139         hits = new ArrayList<Measurement>();
140     }
141 
142     void addMeasurement(Measurement m) {
143         if (hits.size() == 0) hits.add(m);
144         else {
145             boolean added = false;
146             for (int i = hits.size() - 1; i >= 0; i--) { // Keep the measurements ordered by coordinate value
147                 if (m.v > hits.get(i).v) {
148                     hits.add(i, m);
149                     added = true;
150                     break;
151                 }
152             }
153             if (!added) hits.add(m);
154         }
155     }

157     Vec toGlobal(Vec vLocal) { // Convert a position vector from local detector coordinates to global
158         return p.X().sum(R.rotate(vLocal));
159     }
160 
161     Vec toLocal(Vec vGlobal) { // Convert a position vector from global coordinates to local detector
162         return Rinv.rotate(vGlobal.dif(p.X()));
163     }
