45      String toString(String s) {
46          String str;
47          if (m.Layer < 0) {
48              str = String.format("\n****Dump of dummy measurement site %d %s;  ", thisSite, s);
49          } else {
50              str = String.format("\n****Dump of measurement site %d %s;  ", thisSite, s);
51          }
52          if (smoothed) {
53              str=str+"    This site has been smoothed\n";
54          } else if (filtered) {
55              str=str+"    This site has been filtered\n";
56          } else if (predicted) { 
57              str=str+"    This site has been predicted\n"; 
58          }
59          str=str+String.format("    Hit ID=%d, maximum allowed residual=%10.5f\n", hitID, kPar.mxResid[1]);
60          str = str + m.toString("for this site");
61          double B = KalmanInterface.getField(m.p.X(), m.Bfield).mag();
62          Vec tB = KalmanInterface.getField(m.p.X(), m.Bfield).unitVec();
63          str=str+String.format("    Magnetic field strength=%10.6f;   alpha=%10.6f\n", B, alpha);
64          str = str + tB.toString("magnetic field direction") + "\n";
65          str=str+String.format("    chi^2 increment=%12.4e\n", chi2inc);
66          str=str+String.format("    x scattering angle=%10.8f, y scattering angle=%10.8f\n", scatX(), scatZ());
67          if (predicted) str = str + aP.toString("predicted");
68          if (filtered) str = str + aF.toString("filtered");
69          if (smoothed) str = str + aS.toString("smoothed");
70          if (H != null) str = str + "matrix of the transformation from state vector to measurement:" + H.toString();
71          str=str+String.format("      Assumed electron dE/dx in GeV/mm = %10.6f;  Detector thickness=%10.6f\n", dEdx, m.thickness);
72          if (m.Layer < 0) {
73              str=str+String.format("End of dump of dummy measurement site %d<<\n", thisSite);
74          } else {
75              str=str+String.format("End of dump of measurement site %d<<\n", thisSite);
76          }
77          return str;
78      }

80      MeasurementSite(int thisSite, SiModule data, KalmanParams kPar) {
81          this.thisSite = thisSite;
82          this.kPar = kPar;
83          this.m = data;       
84          hitID = -1;
85          double c = 2.99793e8; // Speed of light in m/s
86          conFac = 1.0e12 / c;
87          Vec Bfield = KalmanInterface.getField(m.p.X(), m.Bfield);
88          B = Bfield.mag();
89          alpha = conFac / B; // Convert from 1/pt in 1/GeV to curvature in mm
90          predicted = false;
91          filtered = false;
92          smoothed = false;
93          double rho = 2.329; // Density of silicon in g/cm^3
94          radLen = (21.82 / rho) * 10.0; // Radiation length of silicon in millimeters
95          double sp = 0.002; // Estar collision stopping power for electrons in silicon at about a GeV, in GeV cm2/g
96          dEdx = -0.1 * sp * rho; // in GeV/mm
97          chi2inc = 0.;
98          H = new DMatrixRMaj(5,1);
99          if (!initialized) {
100             tempV = new DMatrixRMaj(5,1);
101             logger = Logger.getLogger(MeasurementSite.class.getName());
102             initialized = true;
103         }
104     }

106     double scatX() { // scattering angle in the x,y plane for the filtered state vector
107         if (aP == null || aF == null) return -999.;
108         Vec p1 = aP.helix.getMom(0.);
109         double t1 = FastMath.atan2(p1.v[0], p1.v[1]);
110         Vec p2 = aF.helix.getMom(0.);
111         double t2 = FastMath.atan2(p2.v[0], p2.v[1]);
112         return t1 - t2;
113     }
114 
115     double scatZ() { // scattering angle in the z,y plane for the filtered state vector
116         if (aP == null || aF == null) return -999.;
117         Vec p1 = aP.helix.getMom(0.);
118         double t1 = FastMath.atan2(p1.v[2], p1.v[1]);
119         Vec p2 = aF.helix.getMom(0.);
120         double t2 = FastMath.atan2(p2.v[2], p2.v[1]);
121         return t1 - t2;
122     }

142     int makePrediction(StateVector pS, SiModule mPs, int hitNumber, boolean sharingOK, boolean pickup, boolean checkBounds, double [] tRange, int trial, boolean verbose2) { // Create predicted state vector by propagating from previous site
143         // pS = state vector that we are predicting from
144         // mPS = Si module that we are predicting from, if any
145         // tRange = allowed time range [tmin,tmax] for picking up a hit
146         // minE = minimum hit energy for it to be picked up, unless no other hit exists
147         // sharingOK = whether to allow sharing of a hit between multiple tracks
148         // pickup = whether we are doing pattern recognition here and need to pick up hits to add to the track
149         int returnFlag = 0;
150         double phi = pS.helix.planeIntersect(m.p);
151         if (debug) verbose2 = true;
152         if (Double.isNaN(phi)) { // There may be no intersection if the momentum is too low!
153             if (debug) {
154                 System.out.format("MeasurementSite.makePrediction: no intersection of helix with the plane exists. Site=%d\n", thisSite);
155                 m.p.print("of intersection");
156                 pS.print("missing plane");
157             }
158             return -1;
159         }
160 
161         Vec X0 = pS.helix.atPhi(phi); // Intersection point in local field coordinate system of pS
162         if (debug) {
163             pS.helix.a.print("helix parameters in makePrediction");
164             X0.print("intersection in local coordinates in makePrediction");
165             pS.helix.toGlobal(X0).print("intersection in global coordinates in makePrediction");
166             Plane pRot = m.p.toLocal(pS.helix.Rot, pS.helix.origin);
167             double check = (X0.dif(pRot.X())).dot(pRot.T());
168             System.out.format("MeasurementSite.makePrediction: dot product of vector in plane with plane direction=%12.8e, should be zero\n",
169                     check);
170         }
171 
172         double deltaE = 0.; // dEdx*thickness/ct;
173 
174         Vec origin = m.p.X();
175         Vec Bfield = KalmanInterface.getField(pS.helix.toGlobal(X0), m.Bfield);
176         double B = Bfield.mag();
177         Vec tB = Bfield.unitVec(B);
178         if (debug) {
179             origin.print("new origin in MeasurementSite.makePrediction");
180             Bfield.print("B field at pivot in MeasurementSite.makePrediction");
181         }
182 
183         // Move pivot point to X0 to generate the predicted helix
184         // First we need the momentum direction to calculate how much silicon we pass through
185         Vec pMom = pS.helix.Rot.inverseRotate(pS.helix.getMom(0.));
186         double XL;
187         if (mPs == null) {
188             XL = 0.;
189             arcLength = 0.;
190         } else {
191             double ct = pMom.unitVec().dot(mPs.p.T()); // cos(theta) at the **previous** site
192             double radius = alpha/pS.helix.a.v[2];
193             XL = mPs.thickness / radLen / Math.abs(ct); // Si scattering thickness at previous site
194             arcLength = -radius*phi*FastMath.sqrt(1.0 + pS.helix.a.v[4] * pS.helix.a.v[4]);
195             if (debug) {
196                 double dx = m.p.X().v[0]-mPs.p.X().v[0];
197                 double dy = m.p.X().v[1]-mPs.p.X().v[1];
198                 double dz = m.p.X().v[2]-mPs.p.X().v[2];
199                 double distance = FastMath.sqrt(dx*dx + dy*dy + dz*dz);
200                 System.out.format("MeasurementSite.predict: arc length=%10.5f, distance=%10.5f\n", arcLength, distance);
201             }
202         }
203         aP = pS.predict(thisSite, X0, B, tB, origin, XL, deltaE);
204         
205         // This calculates the H corresponding to using aP in h( , , ). It is kind of trivial, because
206         // aP are helix parameters for a pivot right at the predicted intersection point, on the helix. 
207         // Hence the prediction at that point does not depend on the helix parameters at all.
208         buildH(aP, H);
209         CommonOps_DDRM.mult(aP.helix.C, H, tempV);
210         double Rextrap = CommonOps_DDRM.dot(H, tempV);
211         
212         // Check whether the intersection is within the bounds of the detector, with some margin
213         // If not, then the pattern recognition may look in another detector in the layer.
214         // Don't do the check if the hit number is already specified. Also, the check will not
215         // be called for in the last sensor of the layer, because we need to run the Kalman prediction
216         // to this layer before moving to the next, even if there is no hit.       
217         double tol = kPar.edgeTolerance + FastMath.sqrt(Rextrap); // Tolerance on the check, in mm
218         if (debug) System.out.format("MeasurementSite.predict: edge tolerance = %10.4f mm\n", tol);
219         Vec rLocal = null;
220         if (checkBounds && hitNumber < 0) {
221             Vec rGlobal = pS.helix.toGlobal(X0); // Transform from field coordinates to global coordinates
222             rLocal = m.toLocal(rGlobal); // Rotate into the detector coordinate system
223             if (rLocal.v[0] < m.xExtent[0] - tol || rLocal.v[0] > m.xExtent[1] + tol) return -2;
224             if (rLocal.v[1] < m.yExtent[0] - tol || rLocal.v[1] > m.yExtent[1] + tol) return -2;
225         }
226         
227         if (debug) {
228             pS.helix.a.print("original helix in MeasurementSite.makePrediction");
229             pS.helix.X0.print("original helix pivot point");
230             //pS.toGlobal(pS.X0).print("original pivot in global coordinates");
231             //if (mPs != null) mPs.toLocal(pS.toGlobal(pS.X0)).print("original pivot in detector coordinates");
232             aP.helix.a.print("pivot transformed helix in MeasurementSite.makePrediction");
233             aP.helix.X0.print("transformed helix pivot point");
234             //aP.toGlobal(aP.X0).print("transformed pivot in global coordinates");
235             //m.toLocal(aP.toGlobal(aP.X0)).print("transformed pivot in detector coordinates");
236             // double phi2 = aP.planeIntersect(m.p);
237             // System.out.format("MeasurementSite.makePrediction: phi2=%12.9f\n", phi2);
238             // Vec X02 = aP.atPhi(phi2);
239             // X02.print("intersection in local coordinates from new helix");
240             // aP.toGlobal(X02).print("intersection in global coordinates from new helix");
241             System.out.format("MeasurementSite.makePrediction: old helix intersects plane at phi=%10.7f\n", phi);
242             Vec rGlobalOld = pS.helix.toGlobal(pS.helix.atPhi(phi));
243             rGlobalOld.print("global intersection with old helix from measurementSite.makePrediction");
244             double phiNew = aP.helix.planeIntersect(m.p); // This angle should always be zero
245             System.out.format("MeasurementSite.makePrediction: new helix intersects plane at phi=%10.7f\n", phiNew);
246             Vec rGlobal = aP.helix.toGlobal(aP.helix.atPhi(phiNew)); // This should be equal to rGlobalOld
247             rGlobal.print("global intersection with new helix from MeasurementSite.makePrediction");
248         }
249 
250         aP.mPred = h(pS, m, phi);
251 
252         // Test of the H vector, by comparing with a numerical difference. If this is done with the H
253         // calculated from aP, comparing with h calculated from aP, then the difference is always zero.
254         // By comparing with the less trivial case of h and H calculated from pS, we verify here that
255         // the derivatives are correct.
256         
257         if (debug) {
258             System.out.format("H corresponding to aP ");
259             H.print();
260             DMatrixRMaj H3m = new DMatrixRMaj(5,1);
261             buildH(pS, H3m);
262             Vec H3 = StateVector.mToVec(H3m);
263             H3.print("H corresponding to pS");
264             double mPredaP = h(aP, m); // This should give exactly the same result as for aP.mPred above
265             System.out.format("Predicted m: from pS=%10.5f;  from aP=%10.5f\n", mPredaP, aP.mPred);
266 
267             StateVector tS = pS.copy();
268             Vec da = new Vec(5);
269             double[] del = { 0.009, -0.005, -0.01, 0.013, 0.011 };
270             for (int i = 0; i < 5; i++) { da.v[i] = pS.helix.a.v[i] * del[i]; }
271             tS.helix.a = tS.helix.a.sum(da);
272             double dxTrue = h(tS, m) - h(pS, m);
273             double dxH = H3.dot(da);
274             System.out.format("Measurementsite.predict: dm True=%10.5f,   dm approx by H=%10.5f\n\n", dxTrue, dxH);
275         }        
276 
277         // Loop over hits and find the one that is closest, unless the hit has been specified
278         int nHits = m.hits.size();
279         if (nHits > 0) {
280             double cut = kPar.mxResid[trial];
281             int theHit = hitNumber;
282             if (theHit < 0 && pickup) {
283                 // Don't pick up a hit if the track projection is outside of the bounds of the strip
284                 // This is especially serious for stereo layers
285                 if (rLocal == null) {
286                     Vec rGlobal = pS.helix.toGlobal(X0); // Transform from field coordinates to global coordinates
287                     rLocal = m.toLocal(rGlobal);   // Rotate into the detector coordinate system
288                 }
289                 if (rLocal.v[0] > m.xExtent[0] - tol && rLocal.v[0] < m.xExtent[1] + tol) {
290                     int theHitAll = theHit;
291                     double minResid = 999.;
292                     double minResidAll = 999.;
293                     for (int i = 0; i < nHits; i++) {
294                         Measurement mHit = m.hits.get(i);
295                         if (mHit.tracks.size() > 0) continue; // Skip used hits
296                         //if (m.hits.get(i).tksMC.size()==0) {  // For cheating, using MC truth to avoid noise hits.
297                         //    continue;
298                         //}
299                         if (m.split) { // Innermost 2019 layers have strips split in the middle into separate channels
300                             if (rLocal.v[0] * mHit.x < 0.) { // The track is on the wrong side
301                                 if (Math.abs(rLocal.v[0]) > tol) continue;
302                             }
303                         }
304                         double residual = Math.abs(mHit.v - aP.mPred);
305                         double ctv = (mHit.v - aP.mPred)/mHit.sigma;
306                         if (debug) {
307                             System.out.format("  MeasurementSite.makePrediction: Found unused hit, residual=%10.5f, sigmas=%10.5f, cut=%10.5f\n", residual, ctv, kPar.mxResid[trial]); 
308                             System.out.format("        Hit energy=%10.4f, cut = %10.4f\n",m.hits.get(i).energy, kPar.minSeedE[m.Layer]);
309                         }
310                         if (residual < minResid) {
311                             if (m.hits.get(i).energy > kPar.minSeedE[m.Layer]) {
312                                 theHit = i;
313                                 minResid = residual;
314                             }
315                         }
316                         if (residual < minResidAll) {
317                             theHitAll = i;
318                             minResidAll = residual;
319                         }
320                     }
321                     double sigmas = 999.;
322                     if (theHitAll >= 0) {
323                         sigmas = minResidAll/m.hits.get(theHitAll).sigma;
324                     }
325                     if (theHitAll < 0 || sigmas > cut) {  // Look for good shared hits only if there is no other option
326                         if (sharingOK) {
327                             double minResid2 = minResidAll;
328                             for (int i = 0; i < nHits; i++) {
329                                 double residual = m.hits.get(i).v - aP.mPred;
330                                 if (debug) {                         
331                                     double ctv = residual/m.hits.get(i).sigma;
332                                     System.out.format("  MeasurementSite.makePrediction: Found used hit, residual=%10.5f, sigmas=%10.5f, cut=%10.5f\n", residual, ctv, kPar.mxResid[trial]); 
333                                 }
334                                 if (Math.abs(residual) < minResid2 && m.hits.get(i).energy > kPar.minSeedE[m.Layer]) {   // Shared hits should never be low energy
335                                     theHit = i;
336                                     minResid2 = Math.abs(residual);
337                                     cut = kPar.mxResidShare;
338                                     if (debug) System.out.format("MeasurementSite.makePrediction: shared hit candidate %d with residual %10.4f\n", theHit, residual);
339                                 }
340                             }
341                         }
342                     } else {
343                         if (theHit < 0) {   // A low-pulse-height hit is the only option
344                             theHit = theHitAll;
345                         } else if (theHit != theHitAll){
346                             double cutVal = minResid/m.hits.get(theHit).sigma;
347                             if (debug) {
348                                 System.out.format("MeasurementSite.makePrediction: hit with ph=%8.2f versus ph=%8.2f\n", m.hits.get(theHitAll).energy,m.hits.get(theHit).energy);
349                                 System.out.format("  Residuals = %10.4f for %d and %10.4f for %d, cutVal=%10.4f\n", minResidAll, theHit, minResid, theHitAll, cutVal);
350                             }
351                             if (cutVal < cut) {  // A high-ph hit and a low-ph hit are within the cut. Now we have to decide. . .
352                                 if (cutVal > kPar.mxResidShare) { // If the high-ph hit is really good, just keep it
353                                     if (minResidAll/minResid < kPar.lowPhThresh) { // the low-ph hit is attractive enough that we will take it
354                                         if (debug) {
355                                             System.out.format("MeasurementSite.makePrediction: choosing hit with ph=%8.2f over hit with ph=%8.2f\n", m.hits.get(theHitAll).energy,m.hits.get(theHit).energy);
356                                             System.out.format("  Residuals = %10.4f for %d and %10.4f for %d, ratio=%9.4f\n", minResidAll, theHit, minResid, theHitAll, minResidAll/minResid);
357                                         }
358                                         theHit = theHitAll;
359                                     }
360                                 }
361                             } else {  // Maybe the low-pulse-height hit will pass
362                                 theHit = theHitAll;
363                             }
364                         }
365                     }
366                 }
367             }
368             if (theHit >= 0) {
369                 aP.r = m.hits.get(theHit).v - aP.mPred;
370                 if (debug) {
371                     if (hitNumber >= 0) {
372                         System.out.format("MeasurementSite.makePrediction: specified hit=%d with residual=%10.7f\n", theHit, aP.r);
373                     } else {
374                         System.out.format("MeasurementSite.makePrediction: selected hit=%d with minimum residual=%10.7f\n", theHit, aP.r);
375                     }
376                     System.out.format("MeasurementSite.makePrediction: intersection with old helix is at phi=%10.7f, z=%10.7f\n", phi,
377                             aP.mPred);
378                     double phi2 = aP.helix.planeIntersect(m.p); // This should always be zero
379                     double mPred2 = h(aP, m, phi2);
380                     System.out.format("MeasurementSite.makePrediction: intersection with new helix is at phi=%10.7f, z=%10.7f\n", phi2, mPred2);
381                 }
382 
383                 aP.R = m.hits.get(theHit).sigma * m.hits.get(theHit).sigma + Rextrap;
384 
385                 chi2inc = aP.r * aP.r / aP.R;
386                 double cutVal = Math.abs(aP.r / m.hits.get(theHit).sigma);
387                 if (verbose2) {
388                     System.out.format("  MeasurementSite.makePrediction: Lyr %d det %d: do we add hit with residual=%10.5f, err=%10.5f, chi2inc=%10.5f, %10.5f<%10.5f?, t=%8.2f\n",
389                             m.Layer, m.detector, aP.r, FastMath.sqrt(aP.R), chi2inc, cutVal, cut, m.hits.get(theHit).time);
390                 }
391 
392                 if (hitNumber >= 0) { // Use the hit no matter what if it was handed to us
393                     hitID = theHit;
394                     returnFlag = 1;
395                     if (verbose2) System.out.format("  MeasurementSite.makePrediction: adding given hit number %d\n", hitID);
396                 } else {
397                     double hitTime = m.hits.get(theHit).time;
398                     if ((!Double.isNaN(chi2inc) && cutVal < cut) && hitTime >= tRange[0] && hitTime <= tRange[1]) { 
399                         hitID = theHit;
400                         returnFlag = 1;
401                         if (verbose2) System.out.format("  MeasurementSite.makePrediction: adding hit number %d with t=%8.2f\n", hitID, hitTime);
402                     } else {
403                         chi2inc = 0.;
404                         if (verbose2) System.out.format("   MeasurementSite.makePrediction: rejecting hit %d with t=%8.2f cutval=%10.6f, t-range=%9.3f-%9.3f\n", hitID, hitTime,cutVal,tRange[0],tRange[1]);
405                     }
406                 }
407                 if (debug) {
408                     System.out.format("MeasurementSite.makePrediction: chi2 increment=%12.5e, hitID=%d, theHit=%d, mxResid=%12.5e, rF=%d\n",
409                             chi2inc, hitID, theHit, cut, returnFlag);
410                 }
411             }
412         }
413 
414         predicted = true;
415 
416         return returnFlag; // -2 for extrapolation not within detector, -1 for error, 1 for a hit was used, 0 no hit used
417     }

419     boolean filter() { // Produce the filtered state vector for this site
420         if (!predicted) {
421             System.out.format("******MeasurementSite.filter: Warning, this site is not in the correct state!\n");
422         }
423 
424         // For dummy layers or layers with no hits just copy the predicted state
425         if (hitID < 0) {
426             aF = aP;
427             filtered = true;
428             chi2inc = 0.;
429             return true;
430         }
431 
432         Measurement hit = m.hits.get(hitID);
433         double V = hit.sigma * hit.sigma;
434         aF = aP.filter(H, V);
435         double phiF = aF.helix.planeIntersect(m.p);
436 
437         // double phiCheck = aF.planeIntersect(m.p);
438         // System.out.format("MeasurementSite.filter: phi = %10.7f, phi check = %10.7f\n",phiF, phiCheck);
439         final boolean verbose2 = false;
440         if (Double.isNaN(phiF)) { // There may be no intersection if the momentum is too low!
441             if (verbose2) {
442                 System.out.format("MeasurementSite.filter: no intersection of helix with the plane exists at layer %d detector %d\n", m.Layer, m.detector);
443                 aP.helix.a.print("predicted helix parameters");
444                 aF.helix.a.print("Filtered helix parameters");
445                 m.p.print("for the intersection");
446             }
447             return false;
448         }
449         aF.mPred = h(aF, m, phiF);
450         aF.r = hit.v - aF.mPred;
451 
452         // Recalculate H using the filtered state vector (this makes a minor difference)
453         buildH(aF, H);
454 
455         // Another numerical test of the derivatives in H
456         if (debug) {
457             StateVector tS = aF.copy();
458             Vec da = new Vec(5);
459             double[] del = { 0.009, -0.005, -0.01, 0.013, 0.011 };
460             for (int i = 0; i < 5; i++) { da.v[i] = aF.helix.a.v[i] * del[i]; }
461             tS.helix.a = tS.helix.a.sum(da);
462             double dxTrue = h(tS, m) - aF.mPred;
463             Vec Hv = StateVector.mToVec(H);
464             double dxH = Hv.dot(da);
465             System.out.format("Measurementsite.predict: dm True=%10.5f,   dm approx by H=%10.5f\n\n", dxTrue, dxH);
466         }
467 
468         // Calculate the filtered covariance of the residual
469         CommonOps_DDRM.mult(aF.helix.C, H, tempV);
470         aF.R = V - CommonOps_DDRM.dot(H, tempV);
471 
472         //System.out.format("MeasurmentSite.filter: R=%10.8f\n", aF.R);
473         if (aF.R < 0) {
474             if (verbose2) { System.out.format("MeasurementSite.filter: covariance of residual %12.4e is negative\n", aF.R); }
475             aF.R = V;
476         }
477 
478         chi2inc = (aF.r * aF.r) / aF.R;
479 
480         if (verbose2) {
481             System.out.format("  MeasurementSite.filter: hit=%10.6f pred=%10.6f resid=%10.7f err=%10.7f chi2inc=%10.6f\n", hit.v, aF.mPred, aF.r, FastMath.sqrt(aF.R), chi2inc); 
482         }
483         filtered = true;
484         return true;
485     }

488     boolean removeHit() {
489         if (hitID < 0) return false;
490         hitID = -1;
491         chi2inc = 0.;
492         smoothed = false;
493         filtered = false;
494         return true;

497     Measurement addHit(KalTrack tkr, double cut, double mxTdif, int oldID) {
498         if (aP == null) {
499             logger.log(Level.WARNING, "******MeasurementSite.addHit: Warning, this site is not in the correct state!");
500             // this.print("in the wrong state for hit addition");
501             return null;
502         }
503         double mPred = 0.;
504         boolean firstHit = true;
505         double chi2incMin = 9999.;
506         int theHit = -1;
507         double [] tRange = {tkr.tMax - mxTdif, tkr.tMin + mxTdif};
508         hitList: for (int hitidx = 0; hitidx < m.hits.size(); hitidx++) {
509             if (debug) System.out.format("MeasurementSite.addHit: trying hit %d.  OldID=%d\n", hitidx, oldID);
510             if (hitidx == oldID) {
511                 continue; // don't add a hit that was just removed
512             }
513             Measurement hit = m.hits.get(hitidx);
514             if (debug) hit.print("to try");
515             for (KalTrack tkOther: hit.tracks) {
516                 if (tkOther != tkr) continue hitList; // ignore already used hits
517             }
518             if (firstHit) {
519                 double phiS = aP.helix.planeIntersect(m.p);
520 
521                 // double phiCheck = aF.planeIntersect(m.p);
522                 // System.out.format("MeasurementSite.filter: phi = %10.7f, phi check = %10.7f\n",phiF, phiCheck);
523                 if (Double.isNaN(phiS)) { // There may be no intersection if the momentum is too low!
524                     break;
525                 }
526                 mPred = h(aP, m, phiS);
527                 firstHit = false;
528             }
529             double residual = hit.v - mPred;
530             double var = hit.sigma * hit.sigma;
531             double chi2inc = (residual * residual) / var;
532             if (debug) System.out.format("MeasurementSite.addHit:residual=%10.5f, variance=%10.5f, chi2inc=%10.5f\n", residual, var, chi2inc);
533             if (chi2inc < chi2incMin) {
534                 chi2incMin = chi2inc;
535                 theHit = hitidx;
536             }
537         }
538         if (theHit >= 0) {
539             Measurement hit = m.hits.get(theHit);
540             if (chi2incMin < kPar.mxResid[0]*kPar.mxResid[0] && hit.time < tRange[1] && hit.time > tRange[0]) {
541                 hitID = theHit;
542                 if (filter()) {
543                     if (debug) System.out.format("MeasurementSite.addHit: chi2inc from filter = %10.5f\n", chi2inc);
544                     if (chi2inc < cut) {
545                         if (debug) {
546                             System.out.format("MeasurementSite.addHit: success! Adding hit %d on layer %d detector %d  hit=", hitID, m.Layer, m.detector);
547                             hit.print("short");
548                             System.out.format("\n");
549                         }
550                         return m.hits.get(hitID);
551                     } else {
552                         hitID = -1;
553                     }
554                 }
555             }
556         }
557         return null;
558     }

560     // Produce the smoothed state vector for this site
561     boolean smooth(MeasurementSite nS) {
562         // nS is the next site in the filtering chain (i.e. the previous site that was smoothed)
563         if (!filtered) {
564             logger.log(Level.WARNING, "******MeasurementSite.smooth: Warning, this site is not in the correct state!");
565             return false;
566         }        
567 
568         this.aS = this.aF.smooth(nS.aS, nS.aP);
569         if (hitID < 0) { 
570             smoothed = true;
571             return true; 
572         }
573 
574         Measurement hit = this.m.hits.get(hitID);
575         double V = hit.sigma * hit.sigma;
576         double phiS = aS.helix.planeIntersect(m.p);
577 
578         if (Double.isNaN(phiS)) { // This should almost never happen!
579             logger.log(Level.FINE, "MeasurementSite.smooth: no intersection of helix with the plane exists.");
580             return false;
581         }
582         aS.mPred = h(aS, m, phiS);
583         aS.r = hit.v - aS.mPred;
584 
585         // Recalculate H using the smoothed helix parameters. Usually this makes little difference, but with the
586         // non-uniform field this seems to reduce tails significantly in residuals of the last SVT layers.
587         buildH(aS, H);
588 
589         CommonOps_DDRM.mult(aS.helix.C, H, tempV);
590         aS.R = V - CommonOps_DDRM.dot(H, tempV);
591         if (aS.R < 0) {
592             if (debug) System.out.format("MeasurementSite.smooth, measurement covariance %12.4e is negative\n", aS.R);
593             //aS.print("the smoothed state");
594             //nS.print("the next site in the chain");
595             aS.R = 0.25*V;  // A negative covariance makes no sense, hence this fudge
596         }
597 
598         chi2inc = (aS.r * aS.r) / aS.R;
599 
600         if (debug) System.out.format("  MeasurementSite.smooth: hit=%10.6f pred=%10.6f resid=%10.7f err=%10.7f chi2inc=%8.5f", 
601                 hit.v, aS.mPred, aS.r, FastMath.sqrt(aS.R), chi2inc);
602         this.smoothed = true;
603         return true;
604     }

606     double h(StateVector pS, SiModule siM) {// Predict the measurement for a helix passing through this plane
607         double phi = pS.helix.planeIntersect(siM.p);
608         if (Double.isNaN(phi)) {
609             logger.log(Level.FINE, "MeasurementSite.h: warning, no intersection of helix with the plane exists.");
610             phi = 0.;
611         }
612         return h(pS, siM, phi);
613     }

615     double h(StateVector pS, SiModule siM, double phi) { // Shortcut call in case phi is already known
616         Vec rGlobal = pS.helix.toGlobal(pS.helix.atPhi(phi));
617         Vec rLocal = siM.toLocal(rGlobal); // Rotate into the detector coordinate system
618         if (debug) {
619             rGlobal.print("MeasurementSite.h: global intersection");
620             rLocal.print("MeasurementSite.h: local intersection");
621             System.out.format("MeasurementSite.h: phi=%12.5e, detector coordinate out of the plane = %10.7f, h=%10.7f\n", phi, rLocal.v[2],
622                     rLocal.v[1]);
623         }
624         return rLocal.v[1];
625     }

627     // Create the derivative matrix for prediction of the measurement from the helix
628     private void buildH(StateVector S, DMatrixRMaj H) {
629         double phi = S.helix.planeIntersect(m.p);
630 
631         if (Double.isNaN(phi)) { // There may be no intersection if the momentum is too low, but highly unlikely here.
632             logger.log(Level.FINE, "MeasurementSite.buildH: no intersection of helix with the plane exists.");
633             return;
634         }
635         if (debug) {
636             System.out.format("MeasurementSite.buildH: phi=%10.7f\n", phi);
637             // S.print("given to buildH");
638             // R.print("in buildH");
639             // p.print("in buildH");
640         }
641         Vec dxdphi = new Vec((alpha / S.helix.a.v[2]) * FastMath.sin(S.helix.a.v[1] + phi), -(alpha / S.helix.a.v[2]) * FastMath.cos(S.helix.a.v[1] + phi),
642                 -(alpha / S.helix.a.v[2]) * S.helix.a.v[4]);
643         double[][] dxda = new double[3][5];
644         dxda[0][0] = FastMath.cos(S.helix.a.v[1]);
645         dxda[1][0] = FastMath.sin(S.helix.a.v[1]);
646         dxda[0][1] = -(S.helix.a.v[0] + alpha / S.helix.a.v[2]) * FastMath.sin(S.helix.a.v[1]) + (alpha / S.helix.a.v[2]) * FastMath.sin(S.helix.a.v[1] + phi);
647         dxda[1][1] = (S.helix.a.v[0] + alpha / S.helix.a.v[2]) * FastMath.cos(S.helix.a.v[1]) - (alpha / S.helix.a.v[2]) * FastMath.cos(S.helix.a.v[1] + phi);
648         dxda[0][2] = -(alpha / (S.helix.a.v[2] * S.helix.a.v[2])) * (FastMath.cos(S.helix.a.v[1]) - FastMath.cos(S.helix.a.v[1] + phi));
649         dxda[1][2] = -(alpha / (S.helix.a.v[2] * S.helix.a.v[2])) * (FastMath.sin(S.helix.a.v[1]) - FastMath.sin(S.helix.a.v[1] + phi));
650         dxda[2][2] = (alpha / (S.helix.a.v[2] * S.helix.a.v[2])) * S.helix.a.v[4] * phi;
651         dxda[2][3] = 1.0;
652         dxda[2][4] = -(alpha / S.helix.a.v[2]) * phi;
653 
654         // System.out.format(" Matrix dxda:\n");
655         double[] dphida = new double[5];
656         double denom = m.p.T().dot(dxdphi);
657         for (int i = 0; i < 5; i++) {
658             // System.out.format(" %10.6f, %10.6f, %10.6f\n", dxda[0][i], dxda[1][i],
659             // dxda[2][i]);
660             for (int j = 0; j < 3; j++) { dphida[i] -= m.p.T().v[j] * dxda[j][i]; }
661             dphida[i] = dphida[i] / denom;
662         }
663         // System.out.format(" dphida=%10.7f, %10.7f, %10.7f, %10.7f, %10.7f\n",dphida[0], dphida[1], dphida[2], dphida[3], dphida[4]);
664         double[][] DxDa = new double[3][5];
665         for (int i = 0; i < 5; i++) {
666             for (int j = 0; j < 3; j++) {
667                 DxDa[j][i] = dxdphi.v[j] * dphida[i] + dxda[j][i];
668                 // if (verbose) System.out.format(" %d %d DxDa=%10.7f\n", j,i,DxDa[j][i]);
669             }
670         }
671         RotMatrix Rt = m.Rinv.multiply(S.helix.Rot.invert());
672         for (int i = 0; i < 5; i++) { 
673             double HHi = 0;
674             for (int j = 0; j < 3; j++) { 
675                 HHi += Rt.M[1][j] * DxDa[j][i]; 
676             }
677             H.unsafe_set(i, 0, HHi);
678         }
679 
680         // Testing the derivatives
681         if (debug) {
682             StateVector sVp = S.copy();
683             double daRel[] = { 0.01, 0.03, -0.02, 0.05, -0.01 };
684             for (int i = 0; i < 5; i++) { sVp.helix.a.v[i] = sVp.helix.a.v[i] * (1.0 + daRel[i]); }
685             Vec da = new Vec(S.helix.a.v[0] * daRel[0], S.helix.a.v[1] * daRel[1], S.helix.a.v[2] * daRel[2], S.helix.a.v[3] * daRel[3], S.helix.a.v[4] * daRel[4]);
686             double phi1 = phi;
687             Vec x1Global = S.helix.toGlobal(S.helix.atPhi(phi1));
688             Vec x1Local = m.toLocal(x1Global);
689             double phi2 = sVp.helix.planeIntersect(m.p);
690             Vec x2Global = sVp.helix.toGlobal(sVp.helix.atPhi(phi2));
691             double dot2 = x2Global.dif(m.p.X()).dot(m.p.T());
692             double dot1 = x1Global.dif(m.p.X()).dot(m.p.T());
693             System.out.format("h derivative testing: dot1=%10.8f, dot2=%10.8f, both should be zero\n", dot1, dot2);
694             Vec x2Local = m.toLocal(x2Global);
695 
696             S.helix.a.print("Test helix parameters");
697             // m.p.print("of the measurement");
698             sVp.helix.a.print("Modified helix parameters");
699             System.out.format("Phi1=%10.7f,  Phi2=%10.7f\n", phi1, phi2);
700             x1Global.print("x1");
701             x2Global.print("x2");
702             for (int j = 0; j < 3; j++) {
703                 double dx = 0.;
704                 for (int i = 0; i < 5; i++) { dx += DxDa[j][i] * da.v[i]; }
705                 System.out.format("j=%d dx=%10.7f,   dxExact=%10.7f\n", j, dx, x2Global.v[j] - x1Global.v[j]);
706             }
707 
708             double dmExact = x2Local.v[1] - x1Local.v[1];
709             double mP1 = h(S, m);
710             double mP2 = h(sVp, m);
711             x1Local.print("x1Local");
712             x2Local.print("x2Local");
713             System.out.format("mP1=%10.5f,   mP2=%10.5f\n", mP1, mP2);
714             double dm = 0.;
715             for (int i = 0; i < 5; i++) { dm += H.get(i,0) * da.v[i]; }
716             System.out.format("Test of H matrix: dm=%10.8f,  dmExact=%10.8f\n\n", dm, dmExact);
717         }
718     }
