1   package org.hps.recon.tracking.kalman;
2   
3   import java.util.Random;
4   
5   import org.apache.commons.math.util.FastMath;
6   
7   /**
8    * Runge-Kutta propagation through the detector, including Gaussian MCS at silicon planes.
9    * This code is only to help with internal testing of the Kalman package and is not part of the fitting or pattern recognition.
10   */
11  class RKhelix {
12  
13      Vec x;    // point on the track
14      Vec p;    // momentum of the track at point x
15      double Q; // charge of the particle
16      
17      private org.lcsim.geometry.FieldMap fM;
18      private Random rndm;
19      private HelixPlaneIntersect hpi;
20      private double rho;
21      private double radLen;
22      
23      RKhelix(Vec x, Vec p, double Q, org.lcsim.geometry.FieldMap fM, Random rndm) {
24          this.rndm = rndm;
25          this.fM = fM;
26          this.x = x.copy();
27          this.p = p.copy();
28          this.Q = Q;
29          hpi = new HelixPlaneIntersect();
30          rho = 2.329; // Density of silicon in g/cm^2
31          radLen = (21.82 / rho) * 10.0; // Radiation length of silicon in millimeters
32      }
33      
34      RKhelix propagateRK(Plane pln) {
35          Vec newP = new Vec(3);
36          Vec newX = planeIntersect(pln, newP);
37          return new RKhelix(newX, newP, Q, fM, rndm);
38      }
39      
40      Vec planeIntersect(Plane pln, Vec pInt) { // phi value where the helix intersects the plane P (given in global coordinates)        
41          return hpi.rkIntersect(pln, x, p, Q, fM, pInt);
42      }
43  
44      void print(String s) {
45          System.out.format("RKhelix %s: x=%s, p=%s, Q=%3.1f\n", s, x.toString(), p.toString(), Q);
46      }
47      
48      // Get parameters for the helix passing through the point x.
49      // pivotF is the pivot point in the helix field reference system.
50      // Input "pivot", the desired pivot point in global coordinates. This will be the origin of the field reference system.
51      Vec helixParameters(Vec pivot, Vec pivotF) {
52          Vec B = KalmanInterface.getField(pivot, fM);
53          double Bmag = B.mag();
54          double alpha = 1.0e12 / (2.99793e8 * Bmag);
55          RotMatrix Rot = R(pivot);
56          // Transform the momentum into the field frame
57          Vec pF = Rot.rotate(p);
58          Vec helixAtX = HelixState.pTOa(pF, 0., 0., Q);   // Helix with pivot at x in field frame
59          // Transform the desired pivot point into the field frame
60          Vec pivotTrans = Rot.rotate(pivot.dif(x));
61          for (int i=0; i<3; ++i) {
62              pivotF.v[i] = pivotTrans.v[i];
63          }
64          return HelixState.pivotTransform(pivotF, helixAtX, new Vec(0., 0., 0.), alpha, 0.);
65      }
66      
67      RKhelix copy() {        
68          return new RKhelix(x.copy(),p.copy(),Q,fM,rndm);
69      }
70      
71      RotMatrix R(Vec position) {
72          Vec B = KalmanInterface.getField(position, fM);
73          double Bmag = B.mag();
74          Vec t = B.unitVec(Bmag);
75          Vec yhat = new Vec(0., 1.0, 0.);
76          Vec u = yhat.cross(t).unitVec();
77          Vec v = t.cross(u);
78          return new RotMatrix(u, v, t);
79      }
80      
81      RKhelix randomScat(Plane P, double X) { // Produce a new helix scattered randomly in a given plane P
82  
83          Vec t = p.unitVec();
84          Vec zhat = new Vec(0., 0., 1.);
85          Vec uhat = t.cross(zhat).unitVec(); // A unit vector u perpendicular to the momentum
86          Vec vhat = t.cross(uhat);
87          RotMatrix Rp = new RotMatrix(uhat, vhat, t);
88          double ct = Math.abs(P.T().dot(t));
89          double theta0;
90          
91          if (X == 0.) theta0 = 0.;  // Get the scattering angle
92          else theta0 = FastMath.sqrt((X / radLen) / ct) * (0.0136 / p.mag()) * (1.0 + 0.038 * FastMath.log((X / radLen) / ct));
93          double thetaX = rndm.nextGaussian() * theta0;
94          double thetaY = rndm.nextGaussian() * theta0;
95          double tx = FastMath.sin(thetaX);
96          double ty = FastMath.sin(thetaY);
97          Vec tLoc = new Vec(tx, ty, FastMath.sqrt(1.0 - tx * tx - ty * ty));
98          Vec tnew = Rp.inverseRotate(tLoc);
99  
100         double E = p.mag(); // Everything is assumed electron
101         double sp = 0.0;    // 0.002; // Estar collision stopping power for electrons in silicon at about a GeV, in GeV cm2/g
102         double dEdx = 0.1 * sp * rho; // in GeV/mm
103         double eLoss = dEdx * X / ct;
104         E = E - eLoss;
105         Vec pNew = tnew.scale(E);
106 
107         return new RKhelix(x, pNew, Q, fM, rndm); // Create the new helix with new origin and pivot point
108     }
