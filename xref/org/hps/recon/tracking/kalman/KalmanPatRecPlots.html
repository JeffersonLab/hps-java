198     void process(EventHeader event, List<HpsSiSensor> sensors, ArrayList<KalTrack>[] kPatList, 
199             List<Track> outputFullTracks) {
200         
201         nEvents++;
202         
203         setupSensors(event);
204         
205         // Set up relatations between raw hits and MC hits
206         RelationalTable rawtomc = new BaseRelationalTable(RelationalTable.Mode.MANY_TO_MANY, RelationalTable.Weighting.UNWEIGHTED);
207         if (event.hasCollection(LCRelation.class, "SVTTrueHitRelations")) {
208             List<LCRelation> trueHitRelations = event.get(LCRelation.class, "SVTTrueHitRelations");
209             for (LCRelation relation : trueHitRelations)
210                 if (relation != null && relation.getFrom() != null && relation.getTo() != null) {
211                     rawtomc.add(relation.getFrom(), relation.getTo());
212                 }
213         }
214         
215         // Analysis of tracker hits (noise hits versus real hits)
216         // Get the list of fitted hits from the event
217         List<LCRelation> fittedHits = event.get(LCRelation.class, fittedHitsCollectionName);
218 
219         // Map the fitted hits to their corresponding raw hits
220         Map<RawTrackerHit, LCRelation> fittedRawTrackerHitMap = new HashMap<RawTrackerHit, LCRelation>();
221 
222         List<RawTrackerHit> rawHitsList = event.get(RawTrackerHit.class, rawTrackerHitCollectionName);
223         for (LCRelation fittedHit : fittedHits) {
224             fittedRawTrackerHitMap.put(FittedRawTrackerHit.getRawTrackerHit(fittedHit), fittedHit);
225         }
226         Map<String, Integer> hits = new HashMap<String, Integer>();
227         for (HpsSiSensor sensor : sensors) {
228             hits.put(sensor.getName(), 0);
229         }
230         
231         // Map the raw hits to the Kalman tracks and to TrackerHit
232         Map<TrackerHit, KalTrack> hitToTk = new HashMap<TrackerHit, KalTrack>();
233         for (int topBottom=0; topBottom<2; ++topBottom) {
234             for (KalTrack kTk : kPatList[topBottom]) {
235                 for (MeasurementSite site : kTk.SiteList) {
236                     if (site.hitID < 0) continue;
237                     StateVector aS = site.aS;
238                     SiModule mod = site.m;
239                     if (aS != null && mod != null) {
240                         TrackerHit hpsHit = KI.getHpsHit(mod.hits.get(site.hitID));
241                         hitToTk.put(hpsHit, kTk);
242                         List<RawTrackerHit> rawHits = hpsHit.getRawHits();
243                         double totalPh = 0.;
244                         HpsSiSensor sensor = null;
245                         int Layer = -1;
246                         int nstrip = 0;
247                         double chan = 0.;
248                         for (RawTrackerHit rawHit : rawHits) { 
249                             double amplitude = FittedRawTrackerHit.getAmp(fittedRawTrackerHitMap.get(rawHit));
250                             totalPh += amplitude;
251                             nstrip++;
252                             chan += rawHit.getIdentifierFieldValue("strip");
253                             if (sensor == null) {
254                                 sensor = (HpsSiSensor) rawHit.getDetectorElement();
255                                 Layer = sensor.getLayerNumber();
256                             }
257                         }
258                         chan = chan/(double)nstrip;
259                         aida.histogram1D("hits/" + sensor.getName() + " average channel with track", 640, 0, 640).fill(chan);
260                         aida.histogram1D("hits/dEdx of hits on tracks",250,0.,20.).fill(hpsHit.getdEdx()*1000000.);
261                         aida.histogram1D("hits/Layer " + Integer.toString(Layer) + " cluster amplitude of hit on track", 250, 0., 5000.).fill(totalPh);
262                         aida.histogram1D("hits/cluster amplitude of hit on track", 250, 0., 5000.).fill(totalPh);
263                     }
264                 }
265             }
266         }
267         Map<RawTrackerHit, TrackerHit> hitToTkHit = new HashMap<RawTrackerHit, TrackerHit>();
268         List<TrackerHit> stripHits = event.get(TrackerHit.class, stripHitInputCollectionName);
269         if (stripHits != null) {
270             for (TrackerHit tkHit : stripHits) {
271                 KalTrack kTk = hitToTk.get(tkHit);
272                 List<RawTrackerHit> rawHits = tkHit.getRawHits();
273                 double totalPh = 0.;
274                 HpsSiSensor sensor = null;
275                 int Layer = -1;
276                 int nstrip = 0;
277                 double chan = 0.;
278                 for (RawTrackerHit rawHit : rawHits) { 
279                     hitToTkHit.put(rawHit, tkHit);
280                     double amplitude = FittedRawTrackerHit.getAmp(fittedRawTrackerHitMap.get(rawHit));
281                     totalPh += amplitude;
282                     nstrip++;
283                     chan += rawHit.getIdentifierFieldValue("strip");
284                     if (sensor == null) {
285                         sensor = (HpsSiSensor) rawHit.getDetectorElement();
286                         Layer = sensor.getLayerNumber();
287                     }
288                 }
289                 if (kTk == null) {
290                     chan = chan/(double)nstrip;
291                     aida.histogram1D("hits/" + sensor.getName() + " average channel with no track", 640, 0, 640).fill(chan);
292                     aida.histogram1D("hits/Layer " + Integer.toString(Layer) + " cluster amplitude with no track", 250, 0., 5000.).fill(totalPh);
293                     aida.histogram1D("hits/" + sensor.getName() + " cluster amplitude with no track", 250, 0., 5000.).fill(totalPh);
294                     aida.histogram1D("hits/cluster amplitude with no track", 250, 0., 5000.).fill(totalPh);
295                 }
296             }
297         }
298         // Look for raw strip hits that were not associated with a tracker hit
299         for (RawTrackerHit rawHit : rawHitsList) {
300             // Access the sensor associated with the raw hit
301             HpsSiSensor sensor = (HpsSiSensor) rawHit.getDetectorElement();
302             int Layer = sensor.getLayerNumber();
303             aida.histogram1D("hits/layer number", 20,0.,20.).fill(Layer);
304             Integer nHits = hits.get(sensor.getName());
305             if (nHits == null) {
306                 nHits = 0;
307             }
308             nHits++;
309             hits.put(rawHit.getDetectorElement().getName(), nHits);
310             TrackerHit hpsHit = hitToTkHit.get(rawHit);
311             if (hpsHit != null) continue;
312             
313             double amplitude = FittedRawTrackerHit.getAmp(fittedRawTrackerHitMap.get(rawHit));
314             aida.histogram1D("hits/" + sensor.getName() + " strip amplitude with no tracker hit", 250, 0., 5000.).fill(amplitude);
315             aida.histogram1D("hits/strip amplitude with no tracker hit", 250, 0., 5000.).fill(amplitude);
316             int channel = rawHit.getIdentifierFieldValue("strip");
317             aida.histogram1D("hits/" + sensor.getName() + " channel with no tracker hit", 640, 0, 640).fill(channel);
318             double t0 = FittedRawTrackerHit.getT0(fittedRawTrackerHitMap.get(rawHit));
319             aida.histogram1D("hits/" + sensor.getName() + " t0 with no tracker hit", 250, -400., 100.).fill(t0);
320         }
321 
322         
323         // Tracker ECAL matching analysis
324         if (event.hasCollection(Cluster.class, ecalClusterCollectionName)) {
325             List<Cluster> clusters = event.get(Cluster.class, ecalClusterCollectionName);
326             for (Track tkr : outputFullTracks) {            
327                 if (tkr.getTrackerHits().size() < 12) continue;
328                 if (tkr.getChi2()/tkr.getNDF() > 2.) continue;
329                 List<TrackState> tkStates = tkr.getTrackStates();
330                 TrackState lastState = null;
331                 for (TrackState tkState : tkStates) {
332                     if (tkState.getLocation() == TrackState.AtLastHit) {
333                         lastState = tkState;
334                         break;
335                     }
336                 }
337                 if (lastState != null) {
338                     // test propagation of a track state from one end of the track to the ECAL
339                     for (Cluster cluster : clusters) {
340                         double [] location = cluster.getPosition();
341                         double [] direction = new double[3];
342                         direction[0] = 0.; direction[1] = 0.; direction[2] = 1.;
343                         PropagatedTrackState pts = KI.propagateTrackState(lastState, location, direction);
344                         double [] intPnt = pts.getIntersection();
345                         double [][] intPntCov = pts.getIntersectionCov();
346                         //new Vec(3,location).print("ECAL cluster position");
347                         //pts.print("to ECAL");
348                         aida.histogram1D("projected track-state x error").fill(intPnt[0] - location[0]);
349                         aida.histogram1D("projected track-state x uncertainty").fill(Math.sqrt(intPntCov[0][0]));
350                         aida.histogram1D("projected track-state x error, sigmas").fill((intPnt[0] - location[0])/Math.sqrt(intPntCov[0][0]));
351                         aida.histogram1D("projected track-state y error").fill(intPnt[1] - location[1]);
352                         aida.histogram1D("projected track-state y uncertainty").fill(Math.sqrt(intPntCov[1][1]));
353                         aida.histogram1D("projected track-state y error, sigmas").fill((intPnt[1] - location[1])/Math.sqrt(intPntCov[1][1]));                        
354                     }
355                 }
356             }
357             for (int topBottom=0; topBottom<2; ++topBottom) {
358                 for (KalTrack tkr : kPatList[topBottom]) {
359                     if (tkr.nHits < 12) continue;
360                     if (tkr.chi2/tkr.nHits > 2.) continue;
361                     MeasurementSite lastSite = tkr.SiteList.get(tkr.SiteList.size()-1);
362                     for (Cluster cluster : clusters) {
363                         Vec eCalPos = KalmanInterface.vectorGlbToKalman(cluster.getPosition());
364                         Plane plnAtEcal = new Plane(eCalPos, new Vec(0.,1.,0.));
365                         ArrayList<Double> yScat = new ArrayList<Double>();
366                         ArrayList<Double> XLscat = new ArrayList<Double>();
367                         //eCalPos.print("ECAL cluster position");
368                         //lastSite.aS.helix.print("helix at last layer");
369                         double [] arcLength = new double[1];
370                         HelixState helixAtEcal = lastSite.aS.helix.propagateRungeKutta(plnAtEcal, yScat, XLscat, fm, arcLength);
371                         if (MatrixFeatures_DDRM.hasNaN(helixAtEcal.C)) continue;
372                         Vec intPnt = helixAtEcal.getRKintersection();
373                         //helixAtEcal.print("helix at ECAL cluster");
374                         //intPnt.print("RK intersection point");
375                         aida.histogram1D("Kalman projected track-state x error").fill(intPnt.v[0] - eCalPos.v[0]);
376                         aida.histogram1D("Kalman projected track-state z error").fill(intPnt.v[2] - eCalPos.v[2]);
377                     }
378                 }
379             }
380         }
381         
382         boolean doGBL = true;
383         if (doGBL) {
384             hitToStrips = TrackUtils.getHitToStripsTable(event);
385             hitToRotated = TrackUtils.getHitToRotatedTable(event);
386         }
387         
388         int minHits = 999;
389         int nKalTracks = 0;
390         boolean sharedHitTrack = false;
391         for (int topBottom=0; topBottom<2; ++topBottom) {
392             for (KalTrack kTk : kPatList[topBottom]) {
393                 nKalTracks++;
394                 aida.histogram1D("Kalman Track Number Hits").fill(kTk.nHits);
395                 if (kTk.nHits < minHits) minHits = kTk.nHits;
396                 
397                 // Vertex constraint
398                 double [] vtx = {0.1735, -3.168, 0.1687};
399                 double [][] vtxCov = {{0.686, 0., 0.}, {0., 10.09, 0.}, {0., 0., 0.017}};
400                 HelixState constrained = kTk.originConstraint(vtx, vtxCov);
401                 if (constrained == null) continue;
402                 double pConstrained = constrained.getMom(0.).mag(); 
403                 if (kTk.nHits >= 10) {
404                     aida.histogram1D("Kalman Track Chi2, >=10 hits").fill(kTk.chi2);
405                     aida.histogram1D("Kalman Track simple Chi2, >=10 hits").fill(kTk.chi2prime());
406                 }
407                 if (kTk.nHits >= 12) aida.histogram1D("Kalman Track Chi2, >=12 hits").fill(kTk.chi2);
408                 aida.histogram1D("Kalman Track Chi2").fill(kTk.chi2);
409                 aida.histogram1D("Kalman arc length to first measurement").fill(kTk.originArcLength());
410                 double[] momentum = kTk.originP();
411                 double pMag = Math.sqrt(momentum[0]*momentum[0]+momentum[1]*momentum[1]+momentum[2]*momentum[2]);
412                 switch (kTk.nHits) {
413                     case 11:
414                         aida.histogram1D("Kalman track Momentum 11-hit").fill(pMag);
415                         break;
416                     case 12:
417                         aida.histogram1D("Kalman track Momentum 12-hit").fill(pMag);
418                         break;
419                     case 13:
420                         aida.histogram1D("Kalman track Momentum 13-hit").fill(pMag);

910     // Make histograms of the MC hit resolution
911     private double simHitRes(EventHeader event) {
912         // Get the collection of 1D hits
913         String stripHitInputCollectionName = "StripClusterer_SiTrackerHitStrip1D";
914         if (!event.hasCollection(TrackerHit.class, stripHitInputCollectionName)) return 999.;
915         List<TrackerHit> stripHits = event.get(TrackerHit.class, stripHitInputCollectionName);
916         
917         if (stripHits == null) return 999.;
918 
919         // Make a mapping from sensor to hits
920         Map<HpsSiSensor, ArrayList<TrackerHit>> hitSensorMap = new HashMap<HpsSiSensor, ArrayList<TrackerHit>>();
921         for (TrackerHit hit1D : stripHits) {
922             HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) hit1D.getRawHits().get(0)).getDetectorElement();
923 
924             ArrayList<TrackerHit> hitsInSensor = null;
925             if (hitSensorMap.containsKey(sensor)) {
926                 hitsInSensor = hitSensorMap.get(sensor);
927             } else {
928                 hitsInSensor = new ArrayList<TrackerHit>();
929             }
930             hitsInSensor.add(hit1D);
931             hitSensorMap.put(sensor, hitsInSensor);
932         }
933         
934         String MCHitInputCollectionName = "TrackerHits";
935         if (!event.hasCollection(SimTrackerHit.class, MCHitInputCollectionName)) return 999.;
936         List<SimTrackerHit> MChits = event.get(SimTrackerHit.class, MCHitInputCollectionName);
937         
938         if (MChits == null) return 999.;
939 
940         // Make a map from MC particle to the hit in the layer
941         Map<Integer, ArrayList<SimTrackerHit>> hitMCparticleMap = new HashMap<Integer, ArrayList<SimTrackerHit>>();
942         for (SimTrackerHit hit1D : MChits) {
943             decoder.setID(hit1D.getCellID());
944             int Layer = decoder.getValue("layer") + 1;  // Kalman layers go from 0 to 13, with 0 and 1 for new 2019 modules
945             //int Module = decoder.getValue("module");
946             //MCParticle MCpart = hit1D.getMCParticle();
947             ArrayList<SimTrackerHit> partInLayer = null;
948             if (hitMCparticleMap.containsKey(Layer)) {
949                 partInLayer = hitMCparticleMap.get(Layer);
950             } else {
951                 partInLayer = new ArrayList<SimTrackerHit>();
952             }
953             partInLayer.add(hit1D);
954             hitMCparticleMap.put(Layer,partInLayer);
955         }
956         //System.out.format("KalmanPatRecDriver.simHitRes: found both hit collections in event %d\n", event.getEventNumber());
957         
958         double maxErr = 0.;
959         ArrayList<SiModule> SiMlist = KI.getSiModuleList();
960         Map<SiModule, SiStripPlane> moduleMap = KI.getModuleMap();
961         for (SiModule siMod : SiMlist) {
962             int layer = siMod.Layer;
963             if (hitMCparticleMap.get(layer) == null) continue;
964             SiStripPlane plane = moduleMap.get(siMod);
965             if (!hitSensorMap.containsKey(plane.getSensor())) continue;
966             ArrayList<TrackerHit> hitsInSensor = hitSensorMap.get(plane.getSensor());
967             if (hitsInSensor == null) continue;
968             for (TrackerHit hit : hitsInSensor) {
969                 //System.out.format("simHitRes: Hit in sensor of layer %d, detector %d\n",layer,siMod.detector);
970                 //Vec tkrHit = new Vec(3,hit.getPosition());
971                 //tkrHit.print("Tracker hit position");
972                 //SiTrackerHitStrip1D global = (new SiTrackerHitStrip1D(hit)).getTransformedHit(TrackerHitType.CoordinateSystem.GLOBAL);
973                 SiTrackerHitStrip1D localHit = (new SiTrackerHitStrip1D(hit)).getTransformedHit(TrackerHitType.CoordinateSystem.SENSOR);
974                 double du = Math.sqrt(localHit.getCovarianceAsMatrix().diagonal(0));
975                 //Vec globalHPS = new Vec(3,global.getPosition());
976                 //Vec localHPS = new Vec(3,localHit.getPosition());
977                 //globalHPS.print("simulated hit in HPS global system");
978                 //localHPS.print("simulated hit in HPS local system");
979                 
980                 //Vec hitGlobal = KalmanInterface.vectorGlbToKalman(global.getPosition());
981                 Vec hitLocal = new Vec(-localHit.getPosition()[1], localHit.getPosition()[0], localHit.getPosition()[2]);
982                 
983                 //hitGlobal.print("simulated hit Kal global position");
984                 //hitLocal.print("simulated hit Kal local position");
985                 //Vec kalGlobal = siMod.toGlobal(hitLocal);
986                 //kalGlobal.print("simulated hit global position from Kalman transformation");
987                 for (SimTrackerHit hitMC : hitMCparticleMap.get(layer)) {
988                     Vec hitMCglobal = KalmanInterface.vectorGlbToKalman(hitMC.getPosition());
989                     //hitMCglobal.print("true hit Kal global position");
990                     Vec kalMClocal = siMod.toLocal(hitMCglobal);
991                     //kalMClocal.print("true hit Kal local position");
992                     //hitLocal.print("sim hit Kal local position");
993                     double hitError = hitLocal.v[1] - kalMClocal.v[1];
994                     if (Math.abs(hitError) > maxErr) maxErr = Math.abs(hitError);
995                     aida.histogram1D("Kalman hit true error").fill(hitError);
996                     aida.histogram1D("Kalman hit true error over uncertainty").fill(hitError/du);
997                     aida.histogram1D(String.format("Layers/Kalman true error in layer %d",layer)).fill(hitError);
998                     aida.histogram1D("MC hit z in local system (should be zero)").fill(kalMClocal.v[2]);
999                 }
1000             }
1001         }
1002         return maxErr;
1003     }

1022     private void setupSensors(EventHeader event) {
1023         List<RawTrackerHit> rawTrackerHits = event.get(RawTrackerHit.class, "SVTRawTrackerHits");
1024         EventHeader.LCMetaData meta = event.getMetaData(rawTrackerHits);
1025         // Get the ID dictionary and field information.
1026         IIdentifierDictionary dict = meta.getIDDecoder().getSubdetector().getDetectorElement().getIdentifierHelper().getIdentifierDictionary();
1027         int fieldIdx = dict.getFieldIndex("side");
1028         int sideIdx = dict.getFieldIndex("strip");
1029         for (RawTrackerHit hit : rawTrackerHits) {
1030             // The "side" and "strip" fields needs to be stripped from the ID for sensor lookup.
1031             IExpandedIdentifier expId = dict.unpack(hit.getIdentifier());
1032             expId.setValue(fieldIdx, 0);
1033             expId.setValue(sideIdx, 0);
1034             IIdentifier strippedId = dict.pack(expId);
1035             // Find the sensor DetectorElement.
1036             List<IDetectorElement> des = DetectorElementStore.getInstance().find(strippedId);
1037             if (des == null || des.size() == 0) {
1038                 throw new RuntimeException("Failed to find any DetectorElements with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
1039             } else if (des.size() == 1) {
1040                 hit.setDetectorElement((SiSensor) des.get(0));
1041             } else {
1042                 // Use first sensor found, which should work unless there are sensors with duplicate IDs.
1043                 for (IDetectorElement de : des) {
1044                     if (de instanceof SiSensor) {
1045                         hit.setDetectorElement((SiSensor) de);
1046                         break;
1047                     }
1048                 }
1049             }
1050             // No sensor was found.
1051             if (hit.getDetectorElement() == null) {
1052                 throw new RuntimeException("No sensor was found for hit with stripped ID <0x" + Long.toHexString(strippedId.getValue()) + ">.");
1053             }
1054         }
1055     }
