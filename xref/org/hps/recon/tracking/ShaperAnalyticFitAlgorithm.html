13   * fits values and errors for T0 and amplitude.

29      @Override
30      public Collection<ShapeFitParameters> fitShape(RawTrackerHit rth, PulseShape shape) {
31          short[] samples = rth.getADCValues();
32          HpsSiSensor sensor = (HpsSiSensor) rth.getDetectorElement();
33          int channel = rth.getIdentifierFieldValue("strip");
34          return this.fitShape(channel, samples, sensor);
35          //===> return this.fitShape(rth.getADCValues(), constants);
36      }

38      public Collection<ShapeFitParameters> fitShape(int channel, short[] samples, HpsSiSensor sensor) {
39          double minChisq = Double.POSITIVE_INFINITY;
40          int bestStart = 0;
41          ShapeFitParameters fit = new ShapeFitParameters();
42          for (int i = 0; i < samples.length - 2; i++) {
43              double chisq = fitSection(channel, samples, sensor, fit, i);
44              // System.out.println("i = " + i + ", " + fit);
45              if (chisq < minChisq) {
46                  minChisq = chisq;
47                  bestStart = i;
48              }
49          }
50          fitSection(channel, samples, sensor, fit, bestStart);
51          // System.out.format("%f\t%f\t%f\t%f\t%f\t%f\n", samples[0] - constants.getPedestal(),
52          // samples[1] - constants.getPedestal(), samples[2] - constants.getPedestal(), samples[3] -
53          // constants.getPedestal(), samples[4] - constants.getPedestal(), samples[5] -
54          // constants.getPedestal());
55          // System.out.println("start = " + bestStart + ", " + fit);
56          ArrayList<ShapeFitParameters> fits = new ArrayList<ShapeFitParameters>();
57          fits.add(fit);
58          return fits;
59      }

61      private double fitSection(int channel, short[] samples, HpsSiSensor sensor, ShapeFitParameters fit, int start) {
62          int length = samples.length - start;
63          double[] y = new double[length];
64          double[] t = new double[length];
65  
66          double tp = 2.5*Math.pow(sensor.getShapeFitParameters(channel)[HpsSiSensor.TP_INDEX], 0.25) * Math.pow(sensor.getShapeFitParameters(channel)[HpsSiSensor.TP_INDEX + 1], 0.75);
67  
68          for (int i = 0; i < length; i++) {
69              //===> y[i] = samples[start + i] - constants.getPedestal();
70              y[i] = samples[start + i] - sensor.getPedestal(channel, i);
71              t[i] = HPSSVTConstants.SAMPLING_INTERVAL * i;
72          }
73  
74          double[] p = new double[length];
75          double[] a = new double[length];
76          for (int i = 0; i < length; i++) {
77              //===> p[i] = y[i] / constants.getNoise();
78              p[i] = y[i] / sensor.getNoise(channel, i);
79              //===> a[i] = Math.exp(1 - t[i] / constants.getTp()) / (constants.getTp() * constants.getNoise());
80  
81              a[i] = Math.exp(1 - t[i] / tp) / (tp * sensor.getNoise(channel, i));
82          }
83  
84          double pa, aatt, pat, aat, aa;
85          pa = 0;
86          aatt = 0;
87          pat = 0;
88          aat = 0;
89          aa = 0;
90          for (int i = 0; i < length; i++) {
91              pa += p[i] * a[i];
92              aatt += a[i] * a[i] * t[i] * t[i];
93              pat += p[i] * a[i] * t[i];
94              aat += a[i] * a[i] * t[i];
95              aa += a[i] * a[i];
96          }
97  
98          double t0 = (pa * aatt - pat * aat) / (pa * aat - aa * pat);
99          //===> double A = pa / ((Math.exp(t0 / constants.getTp()) * (aat - t0 * aa)));
100         double A = pa / ((Math.exp(t0 / tp) * (aat - t0 * aa)));
101 
102         double time_var = 0;
103         double height_var = 0;
104         for (int i = 0; i < length; i++) {
105             double dt_dp = a[i] * (aatt - t[i] * aat - t0 * (aat - t[i] * aa)) / (pa * aat - aa * pat);
106             //===> double dh_dp = (a[i] * Math.exp(-1.0 * t0 / constants.getTp()) + A * dt_dp * aa) / (aat - t0 * aa) - A * dt_dp / constants.getTp();
107             double dh_dp = (a[i] * Math.exp(-1.0 * t0 / tp) + A * dt_dp * aa) / (aat - t0 * aa) - A * dt_dp / tp;
108             time_var += dt_dp * dt_dp;
109             height_var += dh_dp * dh_dp;
110         }
111         t0 += HPSSVTConstants.SAMPLING_INTERVAL * start;
112         fit.setAmp(A);
113         fit.setAmpErr(Math.sqrt(height_var));
114         fit.setT0(t0);
115         fit.setT0Err(Math.sqrt(time_var));
116 
117         double chisq = 0;
118         for (int i = 0; i < samples.length; i++) {
119             double ti = HPSSVTConstants.SAMPLING_INTERVAL * i;
120             //===> double fit_y = A * (Math.max(0, (ti - t0)) / constants.getTp()) * Math.exp(1 - (ti - t0) / constants.getTp()) + constants.getPedestal();
121             double fit_y = A * (Math.max(0, (ti - t0)) / tp) * Math.exp(1 - (ti - t0) / tp) + sensor.getPedestal(channel, i);
122             //===> chisq += Math.pow((fit_y - samples[i]) / constants.getNoise(), 2);
123             chisq += Math.pow((fit_y - samples[i]) / sensor.getNoise(channel, i), 2);
124         }
125 
126         if (A > 0 && chisq < Double.POSITIVE_INFINITY) {
127             // return ChisqProb.gammp(samples.length - 2, chisq);
128             fit.setChiProb(Gamma.regularizedGammaQ(samples.length - 2, chisq));
129             return chisq / (samples.length - 2);
130         } else {
131             return Double.POSITIVE_INFINITY;
132         }
133     }
