3   import java.util.ArrayList;
4   import java.util.List;
5   import org.lcsim.fit.helicaltrack.HelicalTrackHit;
6   
7   /**
8    * Some methods which manipulate hit collection in various ways --right now, uses HelicalTrackHits everywhere since I
9    * plan to use it for tracking...probably can use TrackerHits to make it more general
10   */
11  public class HitCollectionUtilites {
12  
13      /* return a list of hits sorted by there distance from the beam (in y) */
14  
15      public static List<HelicalTrackHit> SortHitsByDistanceFromBeam(List<HelicalTrackHit> hits) {
16          List<HelicalTrackHit> sortedHits = new ArrayList<>();
17          while (!hits.isEmpty()) {
18              double minPosition = 999;
19              HelicalTrackHit minHit = null;
20              for (HelicalTrackHit hit : hits)
21                  if (Math.abs(hit.getCorrectedPosition().y()) < minPosition) {
22                      minHit = hit;
23                      minPosition = hit.getCorrectedPosition().y();
24                  }
25              if (minHit != null) {
26                  sortedHits.add(minHit);
27                  hits.remove(minHit);
28              }
29          }
30          return sortedHits;
31      }
32  
33      /*
34       * split the hit collection into top and bottom hit collections
35       */
36      public static List<List<HelicalTrackHit>> SplitTopBottomHits(List<HelicalTrackHit> hits) {
37          List<List<HelicalTrackHit>> splitLists = new ArrayList<>();
38          List<HelicalTrackHit> topHits = new ArrayList<>();
39          List<HelicalTrackHit> bottomHits = new ArrayList<>();
40          for (HelicalTrackHit hit : hits)
41              if (hit.getPosition()[1] > 0)
42                  topHits.add(hit);
43              else
44                  bottomHits.add(hit);
45          splitLists.add(topHits);
46          splitLists.add(bottomHits);
47  
48          return splitLists;
49      }
50  
51      /*
52       * get the hits from a certain layer
53       */
54      public static List<HelicalTrackHit> GetHitsFromLayer(List<HelicalTrackHit> hits, int nlayer) {
55          List<HelicalTrackHit> hitList = new ArrayList<>();
56          for (HelicalTrackHit hit : hits)
57              if (hit.Layer() == nlayer)
58                  hitList.add(hit);
59          return hitList;
60      }
61  
62      /*
63       * get the hits from a certain layer sorted by distance from beamline
64       */
65      public static List<HelicalTrackHit> GetSortedHits(List<HelicalTrackHit> hits, int nlayer) {
66          return SortHitsByDistanceFromBeam(GetHitsFromLayer(hits, nlayer));
67      }
68  
69  }
