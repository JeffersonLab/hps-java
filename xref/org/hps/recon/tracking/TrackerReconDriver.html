148     /**
149      * This is used to setup the Drivers after XML config.
150      */
151     @Override
152     public void detectorChanged(Detector detector) {
153         // Cache Detector object.
154         this.detector = detector;
155 
156         // Get B-field Y with no sign. Seed Tracker doesn't like signed B-field components.
157         // FIXME Is this always right?
158         Hep3Vector fieldInTracker = TrackUtils.getBField(detector);
159         LOGGER.config("fieldInTracker: Bx = " + fieldInTracker.x() + "; By = " + fieldInTracker.y() + "; Bz = " + fieldInTracker.z());
160         this.bfield = Math.abs(fieldInTracker.y());
161         LOGGER.config(String.format("%s: Set B-field to %.6f\n", this.getClass().getSimpleName(), this.bfield));
162 
163         initialize();
164 
165         super.detectorChanged(detector);
166     }

168     /**
169      * Setup all the child Drivers necessary for track reconstruction.
170      */
171     private void initialize() {
172 
173         if (!strategyResource.startsWith("/")) {
174             strategyResource = "/org/hps/recon/tracking/strategies/" + strategyResource;
175         }
176         List<SeedStrategy> sFinallist = StrategyXMLUtils.getStrategyListFromInputStream(this.getClass().getResourceAsStream(strategyResource));
177         SeedTracker stFinal = new SeedTracker(sFinallist, this._useHPSMaterialManager, this.includeMS);
178         stFinal.setSubdetectorName(subdetectorName);
179         stFinal.setApplySectorBinning(_applySectorBinning);
180         stFinal.setUseDefaultXPlane(false);
181         stFinal.setDebug(this.debug);
182         stFinal.setIterativeConfirmed(_iterativeConfirmed);
183         stFinal.setMaterialManagerTransform(CoordinateTransformations.getTransform());
184         stFinal.setInputCollectionName(stInputCollectionName);
185         stFinal.setTrkCollectionName(trackCollectionName);
186         stFinal.setBField(bfield);
187         stFinal.setMaxHelicalTrackHits(maxTrackerHits);
188         if (debug) {
189             stFinal.setDiagnostics(new SeedTrackerDiagnostics());
190         }
191         // stFinal.setSectorParams(false); //this doesn't actually seem to do anything
192         stFinal.setSectorParams(1, 10000);
193         add(stFinal);
194 
195         if (rmsTimeCut > 0) {
196             HitTimeTrackCheck timeCheck = new HitTimeTrackCheck(rmsTimeCut);
197             timeCheck.setDebug(debug);
198             stFinal.setTrackCheck(timeCheck);
199         }        
200     }

202     /**
203      * This method is used to run the reconstruction and print debug
204      * information.
205      */
206     @Override
207     public void process(EventHeader event) {
208         // This call runs the track reconstruction using the sub-Drivers.
209         super.process(event);
210 
211         // Debug printouts.
212         if (debug) {
213             if (event.hasCollection(HelicalTrackHit.class, stInputCollectionName)) {
214                 System.out.println(this.getClass().getSimpleName() + ": The HelicalTrackHit collection " + stInputCollectionName + " has " + event.get(HelicalTrackHit.class, stInputCollectionName).size() + " hits.");
215             } else {
216                 System.out.println(this.getClass().getSimpleName() + ": No HelicalTrackHit collection for this event");
217             }
218             // Check for Tracks.
219             List<Track> tracks = event.get(Track.class, trackCollectionName);
220             System.out.println(this.getClass().getSimpleName() + ": The Track collection " + trackCollectionName + " has " + tracks.size() + " tracks.");
221 
222             // Print out track info.
223             for (Track track : tracks) {
224                 System.out.println(this.getClass().getSimpleName() + ": " + track.toString());
225                 System.out.println(this.getClass().getSimpleName() + ": number of layers = " + track.getTrackerHits().size());
226                 System.out.println(this.getClass().getSimpleName() + ": chi2 = " + track.getChi2());
227             }
228         }
229 
230         // Set the type of track to indicate B-field in Y e.g. for swimming in Wired.
231         List<Track> tracks = event.get(Track.class, trackCollectionName);
232 
233         if (rejectUncorrectedHits) {
234             Iterator<Track> iter = tracks.iterator();
235             trackLoop:
236             while (iter.hasNext()) {
237                 Track track = iter.next();
238                 for (TrackerHit hit : track.getTrackerHits()) {
239                     HelicalTrackHit hth = (HelicalTrackHit) hit;
240                     double correction = VecOp.sub(hth.getCorrectedPosition(), new BasicHep3Vector(hth.getPosition())).magnitude();
241                     double chisq = hth.chisq();
242                     if (correction < 1e-6) {
243                         this.getLogger().fine(String.format(event.getRunNumber()+ " " + event.getEventNumber()+
244                                 " Discarding track with bad HelicalTrackHit (correction distance %f, chisq penalty %f) Layer = %d"
245                                 , correction, chisq,hth.Layer(),hth.Layer()));
246 //                        System.out.println("Y = "+hth.getPosition()[2]);
247                         iter.remove();
248                         continue trackLoop;
249                     }
250                 }
251             }
252         }
253 
254         setTrackType(tracks);
255 
256         // Increment number of events.
257         ++nevents;
258 
259         // Add to tracks found.
260         ntracks += event.get(Track.class, trackCollectionName).size();
261     }

263     /**
264      * Set the track type to Y_FIELD so swimming is done correctly in Wired.
265      *
266      * @param tracks The list of <code>Track</code> objects.
267      */
268     private void setTrackType(List<Track> tracks) {
269         for (Track track : tracks) {
270             ((BaseTrack) track).setTrackType(BaseTrack.TrackType.Y_FIELD.ordinal());
271         }
272     }
