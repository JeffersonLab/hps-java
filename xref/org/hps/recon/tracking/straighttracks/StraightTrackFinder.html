37      public StraightTrackFinder(HitManager hitmanager, StraightTrackFitter helixfitter) {
38  
39          //  Save the pointers to the hit manager and helix fitter classes
40          _hitmanager = hitmanager;
41          _fitter = helixfitter;
42  
43          //  Instantiate the Confirmer/Extender and Seed Candidate merging classes
44          _confirmer = new StraightTrackConfirmerExtender(_hitmanager, _fitter);
45  
46          //  Create a list of track seeds that have been found
47          _trackseeds = new ArrayList<SeedCandidate>();
48  
49          //  Create a set of MC Particles that have been seeded, confirmed
50          _seededmcp = new HashSet<MCParticle>();
51          _confirmedmcp = new HashSet<MCParticle>();
52  
53      }

172 
173                         //  See if we can extend this seed candidate
174                         _confirmer.Extend(confirmedseed, strategy, bfield, _trackseeds);
175                 }
176             }

179 //        if (_diag != null)
180 //            _diag.fireFinderDone(_trackseeds, _seededmcp);
