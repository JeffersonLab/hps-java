23  /**
24   * The StraightTrackConfirmerExtender class attempts to add hits to an input seed. While the algorithms used for the
25   * confirm and extend phases are identical, there are small differences in procedures when there are no more tracker
26   * layers to check. The confirm phase simply outputs a list of SeedCandidates that have at least the minimum number of
27   * confirm hits added to the track. The extend phases completes track finding and imposes the merge criteria to
28   * eliminate inferior track candidates when a pair of candidates shares more than one hit.
29   * 
30   * Modified for HPS straight track fitting by Matt Graham 10/29/2014
31   */
32  public class StraightTrackConfirmerExtender {
33  
34      public enum Task {
35          CONFIRM, EXTEND;
36      }

39      private int _maxfit = 1000000000; // initialize maximum number of fits to 10^9

43      private List<SeedCandidate> _result;
44      private ISeedTrackerDiagnostics _diag = null;

97      /**
98       * Try to confirm a seed using a specified strategy. The strategy specifies the layers to use in trying to confirm
99       * the seed as well as the minimum number of confirm layers that were added to the seed. Typically, there will be a
100      * single confirm layer that is required for a seed to be confirmed. Confirmed seeds must also pass the chisq cut
101      * and other constraints specified in the strategy.
102      *
103      * @param seed seed to be confirmed
104      * @param strategy strategy to use for confirming this seed
105      * @param bfield magnetic field
106      * @return true if seed was confirmed
107      */
108     public boolean Confirm(SeedCandidate seed, SeedStrategy strategy, double bfield) {
109 
110         // Create a list to hold the confirmed / extended seeds
111         _result = new ArrayList<SeedCandidate>();
112 
113         // Establish which layers are to be checked
114         seed.setUncheckedLayers(strategy.getLayers(SeedLayer.SeedType.Confirm));
115 
116         // Process the seed
117         doTask(seed, Task.CONFIRM, strategy, bfield);
118 
119         // Return true if we found at least one confirming seed candidate
120         return _result.size() > 0;
121     }
122 
123     /**
124      * Try to extend a seed using a specified strategy. The strategy specifies the layers to use in extending the seed
125      * as well as the minimum number of layers required for a seed to be a track candidate. Any track candidates found
126      * as a result of the extend operation are merged with the list of track candidates found so far to eliminate
127      * inferior fits when a pair of track candidates shares more than one hit.
128      *
129      * @param seed seed to be extended
130      * @param strategy strategy to use
131      * @param bfield magnetic field
132      * @param foundseeds list of track candidates found so far
133      */
134     public void Extend(SeedCandidate seed, SeedStrategy strategy, double bfield, List<SeedCandidate> foundseeds) {
135 
136         // Initialize the list of extended seed candidates to those already found
137         _result = foundseeds;
138 
139         // Establish which layers are to be checked
140         seed.setUncheckedLayers(strategy.getLayers(SeedLayer.SeedType.Extend));
141 
142         // Extend the seed and return
143         doTask(seed, Task.EXTEND, strategy, bfield);
144         return;
145     }
146 
147     /**
148      * Perform the confirm or extend step.
149      *
150      * @param inputseed seed to be confirmed/extended
151      * @param task confirm or extend (enum)
152      * @param strategy strategy to use
153      * @param bfield magnetic field
154      */
155     private void doTask(SeedCandidate inputseed, Task task, SeedStrategy strategy, double bfield) {
156 
157         // Initialize the counter for the number of fits performed on this seed
158         _nfit = 0;
159 
160         // // Instantiate the fast hit checker
161         // FastCheck checker = new FastCheck(strategy, bfield, _diag);
162         // if(this._applySectorBinning) checker.setDoSectorBinCheck(this._hmanager.getSectorManager());
163 
164         // Calculate the minimum number of hits to succeed, retrieve the chisq cuts
165         int minhits = strategy.getMinHits();
166         if (task == Task.CONFIRM)
167             minhits = strategy.getMinConfirm() + 3;
168         double badhitchisq = strategy.getBadHitChisq();
169         double maxchisq = strategy.getMaxChisq();
170 
171         // Create a LIFO queue of seeds to be searched for a confirmation/extension
172         // hit (note that a LIFO queue is used to minimize memory usage)
173         LinkedList<SeedCandidate> seedlist = new LinkedList<SeedCandidate>();
174 
175         // The bestseed is a SeedCandidate the meets the requirements for becoming
176         // a track, shares at least one hit with the inputseed, and has been deemed
177         // the best such seed by the track merging criteria
178         //
179         // Initialize the best seed to null
180         SeedCandidate bestseed = null;
181 
182         // If we have already found track candidates, check for duplicates
183         // that share hits with the seed, finding the best such duplicate candidate.
184         for (SeedCandidate trkcand : _result) {
185             if (_merger.isDuplicate(inputseed, trkcand))
186                 bestseed = findBestCandidate(trkcand, bestseed, strategy);
187         }
188 
189         // Create a map between the SeedLayers to be checked and a list of hits on the layer to check
190         Map<SeedLayer, List<HelicalTrackHit>> hitmap = new HashMap<SeedLayer, List<HelicalTrackHit>>();
191 
192         // Loop over the layers to be checked
193         for (SeedLayer lyr : inputseed.getUncheckedLayers()) {
194 
195             hitmap.put(lyr, _hmanager.getTrackerHits(lyr));
196         }
197 
198         // Create a list of layers that have hits to check
199         List<SeedLayer> lyrlist = new ArrayList<SeedLayer>();
200         lyrlist.addAll(hitmap.keySet());
201 
202         // Sort the layers in order of increasing number of hits
203         SortLayers lyrsort = new SortLayers(hitmap);
204         Collections.sort(lyrlist, lyrsort);
205 
206         // Store the layers to check in the seed
207         inputseed.setUncheckedLayers(lyrlist);
208 
209         // Start with the input seed
210         seedlist.add(inputseed);
211 
212         // Keep looping until we have fully processed all seed candidates
213         while (seedlist.size() > 0) {
214 
215             // If we have exceeded the maximum number of fits, print warning and stop processing seed candidates
216             if (_nfit > _maxfit) {
217                 System.out.println("Maximum number of fits exceeded in " + task.toString() + " step");
218                 if (bestseed == null) {
219                     System.out.println("No track candidates are associated with the seed hits");
220                 } else {
221                     System.out.println("Track candidate with " + bestseed.getHits().size() + " hits and chisq of "
222                             + bestseed.getHelix().chisqtot() + " associated with the seed hits");
223                 }
224                 break;
225             }
226 
227             // Pull the last seed off the queue (use a LIFO queue to minimize queue length)
228             SeedCandidate seed = seedlist.poll();
229 
230             // Check if there are enough unchecked layers to meet the minimum number of hits
231             int lyrsleft = seed.getUncheckedLayers().size();
232             int possiblehits = lyrsleft + seed.getHits().size();
233             if (possiblehits < minhits)
234                 continue;
235 
236             // If there is a best fit candidate, see if there is still a chance of beating it
237             if (bestseed != null) {
238 
239                 // If the maximimum hits we can achieve is >1 fewer than the best fit, skip this candidate
240                 int besthits = bestseed.getHits().size();
241                 if (possiblehits < besthits - 1)
242                     continue;
243 
244                 // If the maximum hits we can achieve equals the best fit, skip if we have a worse chi2
245                 double chisq = seed.getHelix().chisqtot();
246                 double bestchisq = seed.getHelix().chisqtot();
247                 if ((possiblehits == besthits) && chisq > bestchisq)
248                     continue;
249 
250                 // If the maximum hits we can achieve is 1 fewer than the best fit, skip if the bad hit criteria can't
251                 // be met
252                 if ((possiblehits == besthits - 1) && (chisq > bestchisq - badhitchisq))
253                     continue;
254             }
255 
256             // See if there are any layers left for confirm/extend
257             if (lyrsleft == 0) {
258 
259                 // Take final action on this seed
260                 if (task == Task.CONFIRM) {
261 
262                     // No more layers and min hit requirement is met, seed is confirmed
263                     _result.add(seed);
264 
265                 } else if (task == Task.EXTEND) {
266 
267                     // Merge the seed into the list of extended seeds
268                     boolean merged = _merger.Merge(_result, seed, strategy);
269 
270                     // If the seed survived the merge, make it our new best candidate
271                     if (merged)
272                         bestseed = findBestCandidate(seed, bestseed, strategy);
273                 }
274 
275                 // Done with this seed
276                 continue;
277             }
278 
279             // Pull the next layer off the queue
280             SeedLayer lyr = seed.getNextLayer();
281             HelicalTrackFit helix = seed.getHelix();
282 
283             // Retrieve the chisq for the last fit and initialize the best fit chisq for this layer
284             double oldchisq = helix.chisqtot();
285             double oldcirclechisq = helix.chisq()[0];
286             double chisqbest = 1.e99;
287 
288             // Get the list of hits to check for this layer and sort them by x-y distance from current helix
289             List<HelicalTrackHit> hitlist = hitmap.get(lyr);
290             SortHits comp = new SortHits(helix);
291             Collections.sort(hitlist, comp);
292 
293             // Loop over the sorted hits in this layer
294             for (HelicalTrackHit hit : hitlist) {
295 
296                 // Make a test seed including the new hit
297                 SeedCandidate test = new SeedCandidate(seed);
298                 test.addHit(hit);
299 
300                 // // Check that this hit is potentially viable
301                 // if (!checker.CheckHitSeed(hit, seed)) {
302                 // if (_diag != null) _diag.fireCheckHitFailed(hit, test);
303                 // continue;
304                 // }
305 
306                 // Fit the test seed
307                 boolean success = _fitter.FitCandidate(test, strategy);
308                 _nfit++;
309 
310                 // Check if the fit was successful
311                 if (success) {
312 
313                     // Success - attach the fit to the test seed
314                     HelicalTrackFit newhelix = _fitter.getHelix();
315                     test.setHelix(newhelix);
316 
317                     // Add the seed to the LIFO queue of seed candidates and update the best chisq
318                     seedlist.addFirst(test);
319                     chisqbest = Math.min(chisqbest, newhelix.chisqtot());
320 
321                 }
322             }
323 
324             // Finished checking hits in the current layer. If all the fit trials for
325             // this layer are potentially bad hits, include the starting seed (less the
326             // current layer, which was popped off the layer queue) in the seed list.
327             if (chisqbest - oldchisq > strategy.getBadHitChisq())
328                 seedlist.addFirst(seed);
329         }
330 
331         // Finished looping over the seeds in the LIFO candidate queue - we are done!
332         return;
333     }

335     /**
336      * Check two track candidates and return the best one subject to the merging criteria.
337      *
338      * @param trial new candidate to try
339      * @param oldbest previous best candidate (or null if no best candidate has been found)
340      * @param strategy strategy in use
341      * @return best track candidate
342      */
343     private SeedCandidate findBestCandidate(SeedCandidate trial, SeedCandidate oldbest, SeedStrategy strategy) {
344 
345         // If the old best candidate is null, return the trial candidate
346         if (oldbest == null)
347             return trial;
348 
349         // If the trial candidate is better, return it
350         if (_merger.isBetter(trial, oldbest, strategy))
351             return trial;
352 
353         // If no improvement, return the old candidate
354         return oldbest;
355     }
