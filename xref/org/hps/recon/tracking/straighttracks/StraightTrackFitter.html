43      public boolean FitCandidate(SeedCandidate seed, SeedStrategy strategy) {// change this to something reasonable
44          if (_debug)
45              System.out.printf("%s: FitCandidate() with strategy: \"%s\"\n", this.getClass().getSimpleName(),
46                      strategy.getName());
47  
48          // Initialize fit results to null objects
49          _helix = null;
50  
51          // // Check that we have set the magnetic field
52          // if (_bfield != 0.)
53          // throw new RuntimeException("B Field should be zero for the straight track fitter");
54  
55          // Retrieve list of hits to be fit
56          List<HelicalTrackHit> hitlist = seed.getHits();
57          if (_debug) {
58              System.out.println(this.getClass().getSimpleName() + ": hitlist size " + hitlist.size() + ":");
59              double z_prev = -99999999.9;
60              for (HelicalTrackHit hit : hitlist) {
61                  System.out.printf("%s: (%.2f,%.2f,%.2f) corrected  %s\n", this.getClass().getSimpleName(), hit
62                          .getPosition()[0], hit.getPosition()[1], hit.getPosition()[2], hit.getCorrectedPosition()
63                          .toString());
64                  if (z_prev < -99999999.0)
65                      z_prev = hit.getPosition()[2];
66                  else {
67                      if (Math.signum(z_prev) != Math.signum(hit.getPosition()[2]))
68                          System.out.printf("%s: topBotHits in event\n", this.getClass().getSimpleName());
69                      z_prev = hit.getPosition()[2];
70                  }
71              }
72          }
73          // Retrieve the old helix
74          HelicalTrackFit oldhelix = seed.getHelix();
75  
76          // If this is the candidate's first helix fit, first do a fit without MS errors
77          if (oldhelix == null) {
78  
79              if (_debug)
80                  System.out.println(this.getClass().getSimpleName()
81                          + ": no old helix do the fit without MS scattering map");
82  
83              // Reset the stereo hit positions to their nominal value
84              for (HelicalTrackHit hit : hitlist)
85                  if (hit instanceof HelicalTrackCross)
86                      ((HelicalTrackCross) hit).resetTrackDirection();
87              // Do the fit
88              _status = _fitter.fit(hitlist);
89              SaveFit();
90  
91              // Retrieve the helix parameters from this fit and save them in the seed
92              oldhelix = _fitter.getFit();
93              seed.setHelix(oldhelix);
94  
95              if (_debug)
96                  System.out.printf("%s: fit succeeded, will be used as seed, with chi2=%.3f and helix:\n%s \n", this
97                          .getClass().getSimpleName(), oldhelix.chisqtot(), oldhelix.toString());
98  
99              // Calculate the multiple scattering angles for this helix
100             try {
101                 seed.setScatterAngles(_scattering.FindScatters(oldhelix));
102             } catch (Exception e) {
103                 System.err.println(e);
104                 if (_debug) {
105                     e.printStackTrace();
106                 }
107                 return false;
108             }
109 
110             if (_debug) {
111                 System.out.printf("%s: after calculating the MS map it has %d size:\n",
112                         this.getClass().getSimpleName(), seed.getMSMap().size());
113                 for (Map.Entry<HelicalTrackHit, MultipleScatter> ms : seed.getMSMap().entrySet()) {
114                     System.out.printf("%s: Hit at layer %d and position %s has MS drpdhi=%f and dz=%f\n", this
115                             .getClass().getSimpleName(), ms.getKey().Layer(), ms.getKey().getCorrectedPosition()
116                             .toString(), ms.getValue().drphi(), ms.getValue().dz());
117                 }
118             }
119 
120         }
121 
122         if (_debug)
123             System.out.printf("%s: update the stereo hit positions with the old helix:\n%s \n", this.getClass()
124                     .getSimpleName(), oldhelix.toString());
125 
126         // Update the stereo hit positions and covariance matrices
127         CorrectStereoHits(hitlist, oldhelix);
128 
129         // Do a helix fit including MS errors
130         if (_debug) {
131             System.out.printf("%s: do the helix fit including MS map this time: \n", this.getClass().getSimpleName());
132             for (Map.Entry<HelicalTrackHit, MultipleScatter> ms : seed.getMSMap().entrySet()) {
133                 System.out.printf("%s: Hit at layer %d and position %s has MS drpdhi=%f and dz=%f\n", this.getClass()
134                         .getSimpleName(), ms.getKey().Layer(), ms.getKey().getCorrectedPosition().toString(), ms
135                         .getValue().drphi(), ms.getValue().dz());
136             }
137         }
138         _status = _fitter.fit(hitlist, seed.getMSMap(), oldhelix);
139         SaveFit();
140 
141         // Retrieve and save the new helix fit
142         _helix = _fitter.getFit();
143         seed.setHelix(_helix);
144         if ((_trackCheck != null) && (!_trackCheck.checkSeed(seed)))
145             return false;
146 
147         // Set the non-holonomic constraint chi square...don't worry about this for now-- MG 10/29/2014
148         // _constrain.setConstraintChisq(strategy, _helix, hitlist);
149         // If the total chi square is below the cut, we have a successful fit
150         boolean success = _helix.chisqtot() <= strategy.getMaxChisq();
151         if (!success)
152             if (_diag != null)
153                 _diag.fireFailedChisqCut(seed);
154 
155         // // If fit was successful, set the new multiple scattering angles
156         if (success) {
157             seed.setScatterAngles(_scattering.FindScatters(_helix));
158             if (_debug)
159                 System.out.printf("%s: this fit was successful, chi2=%f with resulting helix paramaters:\n%s\n", this
160                         .getClass().getSimpleName(), _helix.chisqtot(), _helix.toString()); // System.out.printf("%s: updated MS map before returning from fitCandidate():\n",this.getClass().getSimpleName());
161             // for(Map.Entry<HelicalTrackHit, MultipleScatter> ms : seed.getMSMap().entrySet()) {
162             // System.out.printf("%s: Hit at layer %d and position %s has MS drpdhi=%f and dz=%f\n",
163             // this.getClass().getSimpleName(),ms.getKey().Layer(),ms.getKey().getCorrectedPosition().toString(),ms.getValue().drphi(),ms.getValue().dz());
164 
165         } else if (_debug)
166             System.out
167                     .printf("%s: this fit with chi2=%f failed!\n", this.getClass().getSimpleName(), _helix.chisqtot());
168 
169         return success;
170     }

202

204 
205         _linefit = null;
206 
207         // If we have a circle fit, try to save the line fit / zsegment fit results
208         if (_status == FitStatus.InconsistentSeed)
209             return;
210         if (_status == FitStatus.XYLineFitFailed)
211             return;

219     private void CorrectStereoHits(List<HelicalTrackHit> hitlist, HelicalTrackFit helix) {
220 
221         // Get the PathMap - used to find the track direction at the hit location
222         Map<HelicalTrackHit, Double> pathmap = helix.PathMap();
223 
224         // Loop over the hits and look for stereo hits
225         for (HelicalTrackHit hit : hitlist)
226             if (hit instanceof HelicalTrackCross)
227                 if (pathmap.containsKey(hit)) {
228                     // Found a stereo hit - calculate the track direction and pass it to the hit
229                     ((HelicalTrackCross) hit).setTrackDirection(helix);
230                 }
231         return;
232     }
