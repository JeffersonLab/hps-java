24  /**
25   * Tracking algorithm based on forming straight track seeds from all 3-hit combinations,
26   * confirming this tenantive track by requiring additional hits, and extending
27   * the track to additional layers. The operation of the algorithm is controlled
28   * by a list of SeedStrategy that define the tracker layers to be used and the
29   * cuts on the tracking algorithm.
30   *
31   * Copied/Modified from org.lcsim.recon.tracking.SeedTracker
32   */
33  public class StraightTracker extends Driver {

68      private void initialize(List<SeedStrategy> strategylist, boolean useHPSMaterialManager, boolean includeMS) {
69          _strategylist = strategylist;
70          //  Instantiate the hit manager
71          _hitmanager = new HitManager();
72          //  Instantiate the material manager for HPS,  the  fitter and seed track finder as tey depends on the material manager
73          if (useHPSMaterialManager) {
74              MaterialSupervisor materialSupervisor = new MaterialSupervisor(includeMS);
75              _materialmanager = materialSupervisor;
76              _fitter = new StraightTrackFitter(materialSupervisor);
77          } else {
78              MaterialManager materialmanager = new MaterialManager(includeMS);
79              _materialmanager = materialmanager; //mess around with types here...
80              _fitter = new StraightTrackFitter(materialmanager);
81          }
82          //  Instantiate the Seed Finder
83          _finder = new StraightTrackFinder(_hitmanager, _fitter);
84          //  Instantiate the Track Maker
85          _maketracks = new MakeTracks();
86      }

100     @Override
101     protected void process(EventHeader event) {
102 
103 //        System.out.println("New event");
104         //  Pass the event to the diagnostics package
105         if (_diag != null)
106             _diag.setEvent(event);
107 
108         //  Initialize timing
109         long last_time = System.currentTimeMillis();
110 
111         //  Get the hit collection from the event
112         List<HelicalTrackHit> hitcol = event.get(HelicalTrackHit.class, _inputCol);
113         if (_debug)
114             System.out.println("In " + this.getClass().getSimpleName() + ":  Number of HelicalTrackHits=" + hitcol.size());
115         //  Sort the hits for this event
116         _hitmanager.OrganizeHits(hitcol);
117 
118         //  Make the timing plots if requested
119         long start_time = System.currentTimeMillis();
120         double dtime = ((double) (start_time - last_time)) / 1000.;
121         last_time = start_time;
122         if (_timing)
123             aida.cloud1D("Organize Hits").fill(dtime);
124 
125         //  Make sure that we have cleared the list of track seeds in the finder
126         _finder.clearTrackSeedList();
127 
128         //  Loop over strategies and perform track finding
129         for (SeedStrategy strategy : _strategylist) {
130             if(hitcol.size()>250)
131                 continue;
132             //  Set the strategy for the diagnostics
133             if (_diag != null)
134                 _diag.fireStrategyChanged(strategy);
135 
136             //  Perform track finding under this strategy
137             _finder.FindTracks(strategy, _bfield);
138             if (_debug)
139                 System.out.println("In " + this.getClass().getSimpleName() + ":  Number of Tracks Found=" + _finder.getTrackSeeds().size()+" for Strategy = "+strategy.getName());
140             //  Make the timing plots if requested
141             long time = System.currentTimeMillis();
142             dtime = ((double) (time - last_time)) / 1000.;
143             last_time = time;
144             if (_timing)
145                 aida.cloud1D("Tracking time for strategy " + strategy.getName()).fill(dtime);
146         }
147 
148         //  Get the list of final list of SeedCandidates
149         List<SeedCandidate> trackseeds = _finder.getTrackSeeds();
150 
151         if (_iterativeConfirmedFits > 0) {
152             // Iteratively re-fit tracks to take into account  and hit position correlations
153             if (_debug)
154                 System.out.printf("%s: Iteratively improve %d seeds\n", this.getClass().getSimpleName(), trackseeds.size());
155             List<SeedCandidate> seedsToRemove = new ArrayList<SeedCandidate>();
156             for (SeedCandidate seed : trackseeds) {
157                 SeedStrategy strategy = seed.getSeedStrategy();
158                 boolean success = false;
159                 for (int iterFit = 0; iterFit < _iterativeConfirmedFits; ++iterFit)
160                     success = _fitter.FitCandidate(seed, strategy);
161                 if (!success)
162                     seedsToRemove.add(seed);
163 //                else if (_debug)
164 //                    System.out.printf("%s: done iterating, this seed will be added to event:\n%s\n", this.getClass().getSimpleName(), seed.toString());
165             }
166             for (SeedCandidate badseed : seedsToRemove)
167                 trackseeds.remove(badseed);
168         }
169 
170         //  Make tracks from the final list of track seeds
171         _maketracks.Process(event, trackseeds, _bfield);
172 
173         //  Save the MC Particles that have been seeded / confirmed if diagnostics are enabled
174         if (_diag != null) {
175             Set<MCParticle> seededmcpset = _finder.getSeededMCParticles();
176             List<MCParticle> seededmcp = new ArrayList<MCParticle>(seededmcpset);
177             event.put("SeededMCParticles", seededmcp, MCParticle.class, 0);
178             Set<MCParticle> confirmedmcpset = _finder.getConfirmedMCParticles();
179             List<MCParticle> confirmedmcp = new ArrayList<MCParticle>(confirmedmcpset);
180             event.put("ConfirmedMCParticles", confirmedmcp, MCParticle.class, 0);
181         }
182 
183         //  Clear the list of track seeds accumulated in the track finder
184         _finder.clearTrackSeedList();
185 
186         //  Make the total time plot if requested
187         long end_time = System.currentTimeMillis();
188         dtime = ((double) (end_time - start_time)) / 1000.;
189         if (_timing)
190             aida.cloud1D("Total tracking time").fill(dtime);
191 
192         return;
193     }

195     @Override
196     protected void detectorChanged(Detector detector) {
197           //  Only build the model when the detector is changed
198         _materialmanager.buildModel(detector);
199 
200         //  Find the bfield and pass it to the  fitter and diagnostic package
201         if (!_forceBField)
202             _bfield = detector.getFieldMap().getField(_IP).z();
203 
204         if (_diag != null)
205             _diag.setBField(_bfield);
206         _fitter.setBField(_bfield);
207 
208         //  Get the tracking radius
209         _rtrk = _materialmanager.getRMax();
210 
211         //  Set the sectoring parameters
212         if (_autosectoring)
213             _hitmanager.setSectorParams(_strategylist, _bfield, _rtrk);
214     }
