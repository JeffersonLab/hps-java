73      // Make hits for all sensors within a DetectorElement
74      public List<SiTrackerHit> makeHits(IDetectorElement detector) {
75          List<SiTrackerHit> hits = new ArrayList<SiTrackerHit>();
76          List<SiSensor> sensors = detector.findDescendants(SiSensor.class);
77  
78          // Loop over all sensors
79          for (SiSensor sensor : sensors)
80              if (sensor.hasStrips())
81                  hits.addAll(makeHits(sensor));
82  
83          // Return hit list
84          return hits;
85      }

87      // Make hits for a sensor
88      public List<SiTrackerHit> makeHits(SiSensor sensor) {
89  
90          List<SiTrackerHit> hits = new ArrayList<SiTrackerHit>();
91  
92          // Get SiTrackerIdentifierHelper for this sensor and refresh the 
93          // strip map used to increase speed
94          _sid_helper = (SiTrackerIdentifierHelper) sensor.getIdentifierHelper();
95          stripMap_.clear();
96  
97          // Get the collection of LCRelations (RawTrackerHit to fit parameters)
98          // from the sensor readout. 
99          List< LCRelation > fittedHits = sensor.getReadout().getHits(LCRelation.class);
100 
101         // Map out the electrodes to the corresponding hit
102         // TODO: Do we need a map for each sensor type? 
103         // TODO: Are we still using the thin sensors class? 
104         Map<SiSensorElectrodes, List<LCRelation>> electrode_hits = new LinkedHashMap<SiSensorElectrodes, List<LCRelation>>();
105         Map<SiSensorElectrodes, List<LCRelation>> thin_hits = new LinkedHashMap<SiSensorElectrodes, List<LCRelation>>();
106         Map<SiSensorElectrodes, List<LCRelation>> stripletHits = new LinkedHashMap<SiSensorElectrodes, List<LCRelation>>(); 
107 
108         for (LCRelation fittedHit : fittedHits) {
109 
110             // Get the RawTrackerHit associated with the fitted hit.
111             RawTrackerHit hit = FittedRawTrackerHit.getRawTrackerHit(fittedHit);
112 
113             // Get the ID and create strip map, get electrodes.
114             IIdentifier id = hit.getIdentifier();
115             int strip = _sid_helper.getElectrodeValue(id); 
116             
117             
118 
119             stripMap_.put(fittedHit, _sid_helper.getElectrodeValue(id));
120 
121             // Get electrodes and check that they are strips
122             ChargeCarrier carrier = ChargeCarrier.getCarrier(_sid_helper.getSideValue(id));
123             SiSensorElectrodes electrodes = ((SiSensor) hit.getDetectorElement()).getReadoutElectrodes(carrier);
124 
125             //TODO:: This is deprecated and should be removed
126             if (electrodes instanceof ThinSiStrips) {
127 
128                 // Drop the unbonded channel in thin sensors: 511 and 0
129                 if ((strip == (HPSSVTConstants.TOTAL_STRIPS_PER_THIN_SENSOR - 1)) || strip == 0) continue;
130 
131                 if (thin_hits.get(electrodes) == null)
132                     thin_hits.put(electrodes, new ArrayList<LCRelation>());                    
133                 thin_hits.get(electrodes).add(fittedHit);
134             } else if (electrodes instanceof SiStriplets) {
135                 
136                 // Drop the unbonded channel in thin sensors: 511 and 0
137                 if ( (strip == (HPSSVTConstants.TOTAL_STRIPS_PER_THIN_SENSOR - 1)) || strip == 0) continue;
138                 
139                 if (_debug) System.out.println("StripMaker::makeHits : Electrodes are of SiStriplets"); 
140                 
141                 if (stripletHits.get(electrodes) == null) stripletHits.put(electrodes, new ArrayList<LCRelation>()); 
142                 
143                 stripletHits.get(electrodes).add(fittedHit);
144                 
145             } else {
146 
147                 // Drop the unbonded channel in sensors
148                 if (strip == HPSSVTConstants.TOTAL_STRIPS_PER_SENSOR) continue;
149                 
150                 if (electrode_hits.get(electrodes) == null)
151                     electrode_hits.put(electrodes, new ArrayList<LCRelation>());
152                 electrode_hits.get(electrodes).add(fittedHit);
153             }
154         }
155 
156         for (Map.Entry entry : electrode_hits.entrySet())
157             hits.addAll(makeHits(sensor, (SiStrips) entry.getKey(), (List<LCRelation>) entry.getValue()));
158         for (Map.Entry entry : thin_hits.entrySet())
159             hits.addAll(makeHits(sensor, (ThinSiStrips) entry.getKey(), (List<LCRelation>) entry.getValue()));
160         for (Map.Entry entry : stripletHits.entrySet())
161             hits.addAll(makeHits(sensor, (SiStriplets) entry.getKey(), (List<LCRelation>) entry.getValue()));
162 
163 
164         if (_debug)
165             System.out.println(this.getClass().getSimpleName() + "::makeHits returning " + hits.size() + " clusters from sensor "+sensor.getName());
166         return hits;
167     }

169     public List<SiTrackerHit> makeHits(SiSensor sensor, SiSensorElectrodes electrodes, List<LCRelation> fittedHits) {
170 
171         // Call the clustering algorithm to make clusters
172         List<List<LCRelation>> cluster_list = _clustering.findClusters(fittedHits);
173         if (_debug)
174             System.out.println(this.getClass().getSimpleName() + "::makeHits : Found clusters = " + cluster_list.size());
175         // Create an empty list for the pixel hits to be formed from clusters
176         List<SiTrackerHit> hits = new ArrayList<SiTrackerHit>();
177 
178         // Make a pixel hit from this cluster
179         for (List<LCRelation> cluster : cluster_list)
180 
181             // Make a TrackerHit from the cluster if it meets max cluster size requirement
182             if (cluster.size() <= _max_cluster_nstrips) {
183                 SiTrackerHitStrip1D hit = makeTrackerHit(cluster, electrodes);
184                 // Add to readout and to list of hits
185                 // ((SiSensor) electrodes.getDetectorElement()).getReadout().addHit(hit);
186                 hits.add(hit);
187                 sensor.getReadout().addHit(hit);
188             }
189         if (_debug)
190             System.out.println(this.getClass().getSimpleName() + "::makeHits : Returning " + hits.size() + " hits ");
191         return hits;
192     }

202     private SiTrackerHitStrip1D makeTrackerHit(List<LCRelation> cluster, SiSensorElectrodes electrodes) {
203         
204         Hep3Vector position = getPosition(cluster, electrodes);
205         if (_debug) System.out.println("StripMaker::makeTrackerHit : Cluster position: " + position.toString()); 
206         
207         SymmetricMatrix covariance = getCovariance(cluster, electrodes);
208         if (_debug) System.out.println("StripMaker::makeTrackerHit : Cluster covariance: " + covariance.toString()); 
209         
210         double time = getTime(cluster);
211         double energy = getEnergy(cluster);
212         TrackerHitType type = new TrackerHitType(TrackerHitType.CoordinateSystem.GLOBAL, TrackerHitType.MeasurementType.STRIP_1D);
213         List<RawTrackerHit> rth_cluster = new ArrayList<RawTrackerHit>();
214         for (LCRelation bth : cluster)
215             rth_cluster.add(FittedRawTrackerHit.getRawTrackerHit(bth)); 
216         SiTrackerHitStrip1D hit = new SiTrackerHitStrip1D(position, covariance, energy, time, rth_cluster, type);
217         if (_debug)
218             System.out.println(this.getClass().getSimpleName() + " SiTrackerHitStrip1D created at " + position + "(" + hit.getPositionAsVector().toString() + ")" + " E " + energy + " time " + time);
219         return hit;
220     }

222     private Hep3Vector getPosition(List<LCRelation> cluster, SiSensorElectrodes electrodes) {
223         if (_debug) System.out.println("StripMaker::getPosition : Calculating position for cluster size " + cluster.size());
224         List<Double> signals = new ArrayList<Double>();
225         List<Hep3Vector> positions = new ArrayList<Hep3Vector>();
226 
227         if (_debug)
228             System.out.println(this.getClass().getSimpleName() + " Loop of " + cluster.size() + " and add signals and positions to vectors");
229 
230         for (LCRelation hit : cluster) {
231             signals.add(FittedRawTrackerHit.getAmp(hit));
232             if (electrodes instanceof ThinSiStrips) {
233                 positions.add(((ThinSiStrips) electrodes).getStripCenter(stripMap_.get(hit)));
234 //                if (hit.getRawTrackerHit().getLayerNumber() < 4)
235 //                    System.out.println("thinStripCenter is at " + ((ThinSiStrips) electrodes).getStripCenter(stripMap_.get(hit)).toString());
236             } else if (electrodes instanceof SiStriplets) {
237                 Hep3Vector position = ((SiStriplets) electrodes).getStripCenter(stripMap_.get(hit));
238                 if (_debug) System.out.println("StripMaker::getPosition : Position " + position.toString());
239                 positions.add(position);
240             } else {
241                 positions.add(((SiStrips) electrodes).getStripCenter(stripMap_.get(hit)));
242 //                if (hit.getRawTrackerHit().getLayerNumber() < 4)
243 //                    System.out.println("stripCenter is at " + ((SiStrips) electrodes).getStripCenter(stripMap_.get(hit)).toString());
244             }
245         }
246 
247         // Average charge on central strips of longer clusters
248         if (signals.size() > _max_noaverage_nstrips) {
249             int nstrips_center = signals.size() - 2;
250 
251             // collect sum of charges on center strips
252             double center_charge_sum = 0.0;
253             for (int istrip = 1; istrip < signals.size() - 1; istrip++)
254                 center_charge_sum += signals.get(istrip);
255 
256             // distribute evenly on center strips
257             double center_charge_strip = center_charge_sum / nstrips_center;
258             for (int istrip = 1; istrip < signals.size() - 1; istrip++)
259                 signals.set(istrip, center_charge_strip);
260         }
261 
262         if (_debug)
263             System.out.println(this.getClass().getSimpleName() + " Calculate charge weighted mean for " + signals.size() + " signals");
264 
265         Hep3Vector position = _res_model.weightedAveragePosition(signals, positions);
266 
267         if (_debug)
268             System.out.println(this.getClass().getSimpleName() + " charge weighted position " + position.toString() + " (before trans)");
269         electrodes.getParentToLocal().inverse().transform(position);
270         if (_debug)
271             System.out.println(this.getClass().getSimpleName() + " charge weighted position " + position.toString() + " (after trans)");
272 
273         // Swim position back through lorentz drift direction to midpoint between bias surfaces
274         if (_simulation != null) {
275             _simulation.setSensor((SiSensor) electrodes.getDetectorElement());
276             _simulation.lorentzCorrect(position, electrodes.getChargeCarrier());
277             if (_debug)
278                 System.out.println(this.getClass().getSimpleName() + ": Position " + position.toString() + " ( after Lorentz)");
279         }
280 
281         // return position in global coordinates
282         Hep3Vector newpos = ((SiSensor) electrodes.getDetectorElement()).getGeometry().getLocalToGlobal().transformed(position);
283         if (_debug)
284             System.out.println(this.getClass().getSimpleName() + " final cluster position " + newpos.toString());
285 
286         return ((SiSensor) electrodes.getDetectorElement()).getGeometry().getLocalToGlobal().transformed(position);
287     }

289     private double getTime(List<LCRelation> cluster) {
290         double time_sum = 0;
291         double signal_sum = 0;
292 
293         for (LCRelation hit : cluster) {
294 
295             double signal = FittedRawTrackerHit.getAmp(hit);
296             double time = FittedRawTrackerHit.getT0(hit);
297 
298             time_sum += time * signal * signal;
299             signal_sum += signal * signal;
300 
301         }
302         return time_sum / signal_sum;
303     }

333     private double getEnergy(List<LCRelation> cluster) {
334         double total_charge = 0.0;
335         for (LCRelation hit : cluster) {
336             double signal = FittedRawTrackerHit.getAmp(hit);
337             total_charge += signal;
338         }
339         return total_charge * DopedSilicon.ENERGY_EHPAIR;
340     }
