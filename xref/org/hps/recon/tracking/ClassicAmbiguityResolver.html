12   *Performs: removal of duplicates, removal of partials, removal of shared tracks based on

145      *
146      */
147     protected boolean areShared(Track trk1, Track trk2) {
148         if (TrackUtils.numberOfSharedHits(trk1, trk2) > this.shareThreshold) {
149             if ((!doChargeCheck) || (areSameCharge(trk1, trk2))) {
150                 return true;
151             }
152         }
153         return false;
154     }

174     /**
175      * Remove tracks with poor scores from operable tracks list, put them in
176      * poorScores list
177      */
178     protected void RemovePoorScores() {
179 
180         for (Iterator<Track> iterator = this.tracks.iterator(); iterator.hasNext();) {
181             Track trk = iterator.next();
182             if (trackScoreMap.get(trk)[0] < this.scoreThreshold) {
183                 iterator.remove();
184                 this.poorScore.add(trk);
185             }
186         }
187 
188     }

190     /**
191      * @override
192      *
193      */
194     public double scoreTrack(Track track) {
195         /*
196          * bonuses for hits (unshared and shared), penalties for holes and
197          * missing acceptances of layers
198          * 
199          * define p = 1.0 - CumulativeChiSquare(chisquared/dof) if (p>0): add
200          * log10(p) to score. else: apply penalty to score
201          */
202         double score = 0;
203         List<TrackerHit> hitsOnTrack = track.getTrackerHits();
204 
205         int[] holes = holesOnTrack(track);
206 
207         for (TrackerHit hit : hitsOnTrack) {
208             int layer = ((RawTrackerHit) hit.getRawHits().get(0)).getLayerNumber();
209             layer = (layer + 1) / 2;
210             if (TrackUtils.isSharedHit(hit, this.tracks))
211                 score += sharedHitScore[layer - 1];
212             else
213                 score += unsharedHitScore[layer - 1];
214         }
215 
216         for (int i = 0; i < holes.length; i++) {
217             if (holes[i] == 1)
218                 score -= holePenalty[i];
219             else if (holes[i] == 2)
220                 score -= outsideAcceptancePenalty[i];
221         }
222 
223         double cumProb = ChisqProb.gammp(track.getNDF(), track.getChi2());
224         // System.out.printf("cumProb %f log %f \n", cumProb,
225         // Math.log10(cumProb));
226         if (cumProb < cumProbThreshold)
227             score -= ChisqScoring * Math.log10(cumProb);
228         else
229             score -= badChisqPenalty;
230 
231         return score;
232     }

234     /**
235      * @override
236      */
237     public int[] holesOnTrack(Track trk) {
238         int[] holes = { 1, 1, 1, 1, 1, 1 };
239         // 0=hit, 1=hole, 2=outside acceptance
240 
241         List<TrackerHit> stereoHits = trk.getTrackerHits();
242 
243         for (TrackerHit stereoHit : stereoHits) {
244             int layer = ((RawTrackerHit) stereoHit.getRawHits().get(0)).getLayerNumber();
245             layer = (layer + 1) / 2;
246             holes[layer - 1] = 0;
247         }
248 
249         // for layers with no hits: holes or outside-acceptance?
250         for (int i = 0; i < holes.length; i++) {
251             if (holes[i] == 0)
252                 continue;
253 
254             if (!acc.isWithinAcceptance(trk, i + 1))
255                 holes[i] = 2;
256         }
257 
258         return holes;
259     }
