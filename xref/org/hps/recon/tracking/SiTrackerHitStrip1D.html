6   import hep.physics.vec.Hep3Vector;

8   import org.lcsim.detector.tracker.silicon.SiSensorElectrodes;
9   import org.lcsim.detector.tracker.silicon.SiStrips;
10  import org.lcsim.detector.tracker.silicon.SiStriplets;

12  import org.lcsim.event.TrackerHit;
13  import org.lcsim.recon.tracking.digitization.sisim.TrackerHitType;
14  
15  /**
16   * Class describing a 1D cluster.  This class inherits 
17   * {@link SiTrackerHitStrip1D} in lcsim but overrides methods that require the 
18   * use of both SiStrips and SiStriplets. 
19   */
20  public class SiTrackerHitStrip1D extends org.lcsim.recon.tracking.digitization.sisim.SiTrackerHitStrip1D {
21      
22      /**
23       * Creates a new instance of SiTrackerHitStrip1D
24       */
25      public SiTrackerHitStrip1D(Hep3Vector position, 
26              SymmetricMatrix covarianceMatrix, double energy, double time, 
27              List<RawTrackerHit> rawHits, TrackerHitType decodedType) {
28          super(position, covarianceMatrix, energy, time, rawHits, decodedType);
29      }
30  
31      public SiTrackerHitStrip1D(TrackerHit hit) {
32          super(hit);
33      }
34      
35      public SiTrackerHitStrip1D(TrackerHit hit, TrackerHitType.CoordinateSystem coordinateSystem) {
36          super(hit,coordinateSystem);
37      }
38  
39  
40      @Override 
41      public double getHitLength() {
42          
43          double maxHitLength = 0;
44          for (RawTrackerHit rawHit : getRawHits()) {
45              
46              int cellID = getIdentifierHelper().getElectrodeValue(rawHit.getIdentifier()); 
47              
48              SiSensorElectrodes electrodes = getReadoutElectrodes(); 
49              
50              double hitLength = 0;
51              if (electrodes instanceof SiStrips) hitLength = ((SiStrips) electrodes).getStripLength(cellID); 
52              else if (electrodes instanceof SiStriplets) hitLength = ((SiStriplets) electrodes).getStripLength(cellID); 
53              
54              maxHitLength = Math.max( maxHitLength, hitLength );
55              
56          }
57          
58          //System.out.println("SiTrackerHitStrip1D::getHitLength : Hit length: " + maxHitLength);
59          
60          return maxHitLength;
61  
62      }
63  
64  }
