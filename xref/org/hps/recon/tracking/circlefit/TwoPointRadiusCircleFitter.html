1   package org.hps.recon.tracking.circlefit;
2   
3   import static java.lang.Math.hypot;
4   import static java.lang.Math.pow;
5   import static java.lang.Math.sqrt;
6   
7   /**
8    * Fits a circle to two points and a radius
9    */
10  public class TwoPointRadiusCircleFitter
11  {
12  
13      private static boolean _debug = false;
14  
15      static double distance(double[] p1, double[] p2)
16      {
17          return hypot((p1[0] - p2[0]), p1[1] - p2[1]);
18      }
19  
20      /**
21       * Returns two circles which can be fit to two points and a radius
22       * @param p1 A point on the cricle 
23       * @param p2 Another point on the circle
24       * @param radius The radius of the circle
25       * @return The two circles which can be fit, or null 
26       */
27      public static CircleFit[] findCircles(double[] p1, double[] p2, double radius)
28      {
29          CircleFit[] results = null;
30          double separation = distance(p1, p2);
31          double mirrorDistance;
32  
33          if (separation == 0.0) {
34              if (radius == 0.0) {
35                  if (_debug) {
36                      System.out.printf("\nNo circles can be drawn through (%.4f,%.4f)", p1[0], p1[1]);
37                  }
38              } else {
39                  if (_debug) {
40                      System.out.printf("\nInfinitely many circles can be drawn through (%.4f,%.4f)", p1[0], p1[1]);
41                  }
42              }
43          } else if (separation == 2 * radius) {
44              results = new CircleFit[1];
45              results[0] = new CircleFit((p1[0] + p2[0]) / 2, (p1[1] + p2[1]) / 2, radius);
46              if (_debug) {
47                  System.out.printf("\nGiven points are opposite ends of a diameter of the circle with center (%.4f,%.4f) and radius %.4f", (p1[0] + p2[0]) / 2, (p1[1] + p2[1]) / 2, radius);
48              }
49          } else if (separation > 2 * radius) {
50              if (_debug) {
51                  System.out.printf("\nGiven points are farther away from each other than a diameter of a circle with radius %.4f", radius);
52              }
53          } else {
54              mirrorDistance = sqrt(pow(radius, 2) - pow(separation / 2, 2));
55              results = new CircleFit[2];
56              results[0] = new CircleFit((p1[0] + p2[0]) / 2 + mirrorDistance * (p1[1] - p2[1]) / separation, (p1[1] + p2[1]) / 2 + mirrorDistance * (p2[0] - p1[0]) / separation, radius);
57              results[1] = new CircleFit((p1[0] + p2[0]) / 2 - mirrorDistance * (p1[1] - p2[1]) / separation, (p1[1] + p2[1]) / 2 - mirrorDistance * (p2[0] - p1[0]) / separation, radius);
58              if (_debug) {
59                  System.out.printf("\nTwo circles are possible.");
60              }
61              if (_debug) {
62                  System.out.printf("\nCircle C1 with center (%.4f,%.4f), radius %.4f and Circle C2 with center (%.4f,%.4f), radius %.4f", (p1[0] + p2[0]) / 2 + mirrorDistance * (p1[1] - p2[1]) / separation, (p1[1] + p2[1]) / 2 + mirrorDistance * (p2[0] - p1[0]) / separation, radius, (p1[0] + p2[0]) / 2 - mirrorDistance * (p1[1] - p2[1]) / separation, (p1[1] + p2[1]) / 2 - mirrorDistance * (p2[0] - p1[0]) / separation, radius);
63              }
64          }
65          return results;
66      }
