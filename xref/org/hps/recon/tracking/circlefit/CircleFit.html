1   package org.hps.recon.tracking.circlefit;
2   
3   /**
4    * A class which encapsulates the result of a circle fit.
5    */
6   public class CircleFit
7   {
8   
9       private double _x0;
10      private double _y0;
11      private double _r;
12  
13      /**
14       * Creates a new instance of CircleFit
15       *
16       * @param x0 the x coordinate of the center of the circle
17       * @param y0 the y coordinate of the center of the circle
18       * @param r the radius of the circle
19       */
20      public CircleFit(double x0, double y0, double r)
21      {
22          _x0 = x0;
23          _y0 = y0;
24          _r = r;
25      }
26  
27      /**
28       * The x coordinate of the center of the circle.
29       *
30       * @return the x coordinate of the center of the circle
31       */
32      public double x0()
33      {
34          return _x0;
35      }
36  
37      /**
38       * The y coordinate of the center of the circle
39       *
40       * @return
41       */
42      public double y0()
43      {
44          return _y0;
45      }
46  
47      /**
48       * The radius of the circle
49       *
50       * @return the radius of the circle
51       */
52      public double radius()
53      {
54          return _r;
55      }
56  
57      /**
58       * Return the slope of the tangent (dy/dx) at a point on this circle (does
59       * not check if point is on circle)
60       *
61       * @param p double array assumed to be (x, y)
62       * @return the slope of the tangent to this circle at the point p
63       */
64      public double tangentAtPoint(double[] p)
65      {
66          double tangent;
67          tangent = -(p[0] - _x0) / (p[1] - _y0);
68          return tangent;
69      }
70  
71      /**
72       * String representation of this object
73       *
74       * @return String representation of this object
75       */
76      public String toString()
77      {
78          return "CircleFit: x0= " + _x0 + " y0= " + _y0 + " r= " + _r;
79      }
80  
81  }
