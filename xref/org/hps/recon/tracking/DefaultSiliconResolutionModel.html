18      @Override
19      public double getMeasuredResolution(List< LCRelation > cluster, SiSensorElectrodes electrodes) 
20      
21      {
22          double measured_resolution;
23  
24          double sense_pitch = ((SiSensor) electrodes.getDetectorElement()).getSenseElectrodes(electrodes.getChargeCarrier()).getPitch(0);
25  
26          // double readout_pitch = electrodes.getPitch(0);
27          // double noise =
28          // _readout_chip.getChannel(strip_number).computeNoise(electrodes.getCapacitance(strip_number));
29          // double signal_expected = (0.000280/DopedSilicon.ENERGY_EHPAIR) *
30          // ((SiSensor)electrodes.getDetectorElement()).getThickness(); // ~280 KeV/mm for thick Si
31          // sensors
32          if (cluster.size() == 1) {
33              measured_resolution = sense_pitch * _oneClusterErr;
34          } else if (cluster.size() == 2) {
35              measured_resolution = sense_pitch * _twoClusterErr;
36          } else if (cluster.size() == 3) {
37              measured_resolution = sense_pitch * _threeClusterErr;
38          } else if (cluster.size() == 4) {
39              measured_resolution = sense_pitch * _fourClusterErr;
40          } else {
41              measured_resolution = sense_pitch * _fiveClusterErr;
42          }
43  
44          //System.out.println("DefaultSiliconResolutionModel::getMeasuredResolution : Measured resolution: " + measured_resolution); 
45          return measured_resolution;
46      }

48      // TODO: Given that all of the hits associated with a cluster will 
49      //       have the same strip lenght, looping over all hits and finding the
50      //       one with the longest length is uncessary.  Also, for now, the 
51      //       instance where we have hits from an SiStriplets object needs to 
52      //       be handled differently because it does not inherit from SiStrips.  
53      //       To clean this up, the getStripLength method should be added to 
54      //       SiSensorElectrodes.  This would elimininate the need for casting. 
55      public double getUnmeasuredResolution(List< LCRelation > cluster, SiSensorElectrodes electrodes, Map<LCRelation, Integer> strip_map) {
56          // Get length of longest strip in hit
57  
58          if (electrodes instanceof SiStriplets) return ((SiStriplets) electrodes).getStripLength(strip_map.get(cluster.get(0))); 
59  
60          double hit_length = 0;
61          for (LCRelation hit : cluster) {
62              hit_length = Math.max(hit_length, ((SiStrips) electrodes).getStripLength(strip_map.get(hit)));
63          }
64          return hit_length / Math.sqrt(12);
65      }

101     @Override
102     public Hep3Vector weightedAveragePosition(List<Double> signals, List<Hep3Vector> positions) {
103         double total_weight = 0;
104         Hep3Vector position = new BasicHep3Vector(0, 0, 0);
105 
106         for (int istrip = 0; istrip < signals.size(); istrip++) {
107             double signal = signals.get(istrip);
108             
109             double weight = _useWeights ? signal : 1;
110             total_weight += weight;
111             position = VecOp.add(position, VecOp.mult(weight, positions.get(istrip)));
112             /*if (_debug) {
113                 System.out.println(this.getClass().getSimpleName() + "strip " + istrip + ": signal " + signal + " position " + positions.get(istrip) + " -> total_position " + position.toString() + " ( total charge " + total_charge + ")");
114             }*/
115 
116         }
117 
118         return VecOp.mult(1 / total_weight, position);
119     }
