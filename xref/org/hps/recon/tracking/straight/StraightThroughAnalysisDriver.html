467             int layer = HPSTrackerBuilder.getLayerFromVolumeName(sensor.getName());

475             } else {
476                 botHits.add(cluster);
477             }
478             // hit time selection
479             if (!selectTime || (cluster.getTime() < timeMax && cluster.getTime() > timeMin)) {
480 
481                 // look at axials and stereo separately

483 
484                     Map<Integer, List<SiTrackerHitStrip1D>> hitsPerLayer;

493                     if (hits == null) {
494                         hits = new ArrayList<SiTrackerHitStrip1D>();
495                         hitsPerLayer.put(layer, hits);
496                     }
497                     hits.add(cluster);
498 
499                 } else {
500 
501                     // look at stereos
502 
503                     Map<Integer, List<SiTrackerHitStrip1D>> hitsPerLayer;
504                     if (sensor.isTopLayer()) {
505                         hitsPerLayer = stereoHitsPerLayer.get(0);
506                     } else {
507                         hitsPerLayer = stereoHitsPerLayer.get(1);
508                     }
509 
510                     List<SiTrackerHitStrip1D> hits;
511                     hits = hitsPerLayer.get(layer);
512                     if (hits == null) {
513                         hits = new ArrayList<SiTrackerHitStrip1D>();
514                         hitsPerLayer.put(layer, hits);
515                     }
516                     hits.add(cluster);
517 
518                 }
519 
520             }
521 
522         }

525         hitCount.fill(stripClusters.size());
526         topHitCount.fill(topHits.size());
527         botHitCount.fill(botHits.size());
528 
529         // fill hit positions
530         for (HpsSiSensor sensor : sensors)
531             sensorHitCounts.get(sensor.getName()).fill(sensorHitCountMap.get(sensor.getName())[0]);
532 
533         // Pattern recognition for axial hits - yeah!
534         List<List<SiTrackerHitStrip1D>> axialSeedHits = new ArrayList<List<SiTrackerHitStrip1D>>();
535         List<List<SiTrackerHitStrip1D>> axialSeedL13Hits = new ArrayList<List<SiTrackerHitStrip1D>>();
536         List<List<SiTrackerHitStrip1D>> axialSeedL46Hits = new ArrayList<List<SiTrackerHitStrip1D>>();
537         for (int ihalf = 0; ihalf < 2; ++ihalf) {
538             List<SiTrackerHitStrip1D> seedHits = new ArrayList<SiTrackerHitStrip1D>();
539             List<SiTrackerHitStrip1D> seedL13Hits = new ArrayList<SiTrackerHitStrip1D>();
540             List<SiTrackerHitStrip1D> seedL46Hits = new ArrayList<SiTrackerHitStrip1D>();
541             for (int layer : axialHitsPerLayer.get(ihalf).keySet()) {
542                 // single hit on the sensor
543                 if (axialHitsPerLayer.get(ihalf).get(layer).size() == 1) {
544                     seedHits.add(axialHitsPerLayer.get(ihalf).get(layer).get(0));
545                     if (layer < 4)
546                         seedL13Hits.add(axialHitsPerLayer.get(ihalf).get(layer).get(0));
547                     else
548                         seedL46Hits.add(axialHitsPerLayer.get(ihalf).get(layer).get(0));
549                 }
550             }

552             axialSeedL13Hits.add(seedL13Hits);
553             axialSeedL46Hits.add(seedL46Hits);

556         // Pattern recognition for stereo hits - yeah!
557         List<List<SiTrackerHitStrip1D>> stereoSeedHits = new ArrayList<List<SiTrackerHitStrip1D>>();
558         List<List<SiTrackerHitStrip1D>> stereoSeedL13Hits = new ArrayList<List<SiTrackerHitStrip1D>>();
559         List<List<SiTrackerHitStrip1D>> stereoSeedL46Hits = new ArrayList<List<SiTrackerHitStrip1D>>();
560         for (int ihalf = 0; ihalf < 2; ++ihalf) {
561             List<SiTrackerHitStrip1D> seedHits = new ArrayList<SiTrackerHitStrip1D>();
562             List<SiTrackerHitStrip1D> seedL13Hits = new ArrayList<SiTrackerHitStrip1D>();
563             List<SiTrackerHitStrip1D> seedL46Hits = new ArrayList<SiTrackerHitStrip1D>();
564             for (int layer : stereoHitsPerLayer.get(ihalf).keySet()) {
565                 // single hit on the sensor
566                 if (stereoHitsPerLayer.get(ihalf).get(layer).size() == 1) {
567                     seedHits.add(stereoHitsPerLayer.get(ihalf).get(layer).get(0));
568                     if (layer < 4)
569                         seedL13Hits.add(stereoHitsPerLayer.get(ihalf).get(layer).get(0));
570                     else
571                         seedL46Hits.add(stereoHitsPerLayer.get(ihalf).get(layer).get(0));
572                 }
573             }

575             stereoSeedL13Hits.add(seedL13Hits);
576             stereoSeedL46Hits.add(seedL46Hits);

579         // try to make stereo hits
580 
581         List<List<STUtils.StereoPair>> stereoPairs = new ArrayList<List<STUtils.StereoPair>>();
582 
583         for (int ihalf = 0; ihalf < 2; ++ihalf) {
584             List<STUtils.StereoPair> stereoPairCandidates = new ArrayList<STUtils.StereoPair>();
585             List<SiTrackerHitStrip1D> aSeedHits = axialSeedHits.get(ihalf);
586             for (SiTrackerHitStrip1D axialSeedHit : aSeedHits) {
587 
588                 // find the stereo sensor and its hit from the pre-compiled stereo pair list
589                 HpsSiSensor stereoSensor = null;
590                 SiTrackerHitStrip1D stereoSeedHit = null;
591 
592                 HpsSiSensor axialSensor = (HpsSiSensor) axialSeedHit.getRawHits().get(0).getDetectorElement();
593                 logger.fine("Look for stereo sensor to \"" + axialSensor.getName() + "\"");
594 
595                 for (SvtStereoLayer stereoLayer : stereoLayers) {
596                     if (stereoLayer.getAxialSensor().getName().equals(axialSensor.getName())) {
597                         stereoSensor = stereoLayer.getStereoSensor();
598                         break;
599                     }
600                 }
601 
602                 // make sure it was found
603                 if (stereoSensor == null)
604                     throw new RuntimeException("Couldn't find stereo sensor to \"" + axialSensor.getName() + "\"");
605 
606                 logger.fine("Found stereo sensor \"" + stereoSensor.getName() + "\"");
607 
608                 // find the hit
609                 // this only works for single hits per sensor
610                 for (List<SiTrackerHitStrip1D> sSeedHits : stereoSeedHits) {
611                     for (SiTrackerHitStrip1D stereoSeedHitCandidate : sSeedHits) {
612                         HpsSiSensor sensor = (HpsSiSensor) stereoSeedHitCandidate.getRawHits().get(0)
613                                 .getDetectorElement();
614                         if (sensor.getName().equals(stereoSensor.getName())) {
615                             logger.fine("Found stereo hit at "
616                                     + stereoSeedHitCandidate.getPositionAsVector().toString() + " at sensor \""
617                                     + stereoSensor.getName() + "\"");
618                             stereoSeedHit = stereoSeedHitCandidate;
619                             break; // ok, this only works for single hits per sensor
620                         }
621                     }
622                 }
623 
624                 // Check if we found a candidate pair
625                 if (stereoSeedHit != null) {
626                     STUtils.StereoPair pair = new STUtils.StereoPair(axialSeedHit, stereoSeedHit, origoStraightThroughs);
627                     if (STUtils.StereoPair.passCuts(pair))
628                         stereoPairCandidates.add(pair);
629                 }
630             }
631             stereoPairs.add(stereoPairCandidates);
632         }
633 
634         // loop over the stereo pairs
635         for (int ihalf = 0; ihalf < 2; ++ihalf) {
636             List<STUtils.StereoPair> pairs = stereoPairs.get(ihalf);
637 
638             logger.fine("Found " + pairs.size() + " stereo candidates for " + (ihalf == 0 ? "top" : "bottom"));
639 
640             // Fill count
641             stereoHitCount[ihalf].fill(pairs.size());
642 
643             // Plot the position of the hits
644             for (STUtils.StereoPair pair : pairs) {
645                 Hep3Vector p = pair.getPosition();
646                 stereoHitPositionsXY.get(pair.getAxial().getRawHits().get(0).getDetectorElement().getName()).fill(
647                         p.x(), p.y());
648                 logger.fine(p.toString() + " from " + pair.getAxial().getPositionAsVector().toString() + "  "
649                         + pair.getStereo().getPositionAsVector().toString() + " ("
650                         + pair.getAxial().getRawHits().get(0).getDetectorElement().getName() + " and "
651                         + pair.getStereo().getRawHits().get(0).getDetectorElement().getName() + ")");
652             }
653         }

655         // add hits to axial tracks and fit them
656         List<STUtils.STTrack> axialTracks = new ArrayList<STUtils.STTrack>();

668

676                 continue;
677             STUtils.STTrack track = new STUtils.STTrack();
678             track.setHits(seedHits);
679             logger.fine("Fit axial L13 track");
680             regressionFitter.fit(track);
681             track.addFit(regressionFitter.getFit());

687         for (List<SiTrackerHitStrip1D> seedHits : axialSeedL46Hits) {

689                 continue;
690             STUtils.STTrack track = new STUtils.STTrack();
691             track.setHits(seedHits);
692             logger.fine("Fit axial L46 track");
693             regressionFitter.fit(track);

701 
702             // require min number of hits

704                 continue;
705 
706             // Create the track and set the hits
707             STUtils.STStereoTrack track = new STUtils.STStereoTrack();
708             track.setHits(seedHits);

713             // add fitted track to list of tracks
714             stereoTracks.add(track);

723                 double zHit = hit.getPositionAsVector().z();

726                 sensorHitResGlobal.get(hit.getRawHits().get(0).getDetectorElement().getName()).fill(resGlobal);
727             }
728

735                 trackAxialIntercept[0].fill(track.getIntercept());
736             } else {
737                 nTracksAxial[1]++;
738                 trackAxialHitCount[1].fill(track.getHits().size());
739                 trackAxialSlope[1].fill(track.getSlope());
740                 trackAxialIntercept[1].fill(track.getIntercept());
741             }
742         }

744         trackAxialCount[1].fill(nTracksAxial[1]);
745 
746         // Fill histograms for axial L13 tracks
747 
748         for (STUtils.STTrack track : axialL13Tracks) {
749 
750             for (SiTrackerHitStrip1D hit : track.getHits()) {
751                 double yhit = hit.getPositionAsVector().y();
752                 double zHit = hit.getPositionAsVector().z();
753                 double yPred = track.predict(zHit);

819                     Hep3Vector p = new BasicHep3Vector(pair.getPosition().v());
820                     Hep3Vector trackDirection = track.getDirection();
821                     logger.finest("updatePosition " + p.toString() + " with track dir " + trackDirection.toString());
822                     pair.updatePosition(trackDirection);

824                     logger.finest("new position " + pnew.toString());

866                 sensorURes.get(hit.getStereo().getRawHits().get(0).getDetectorElement().getName()).fill(uResStereo);
867 
868             }

871             trackHitCount[half].fill(track.getHits().size());
872             for (int v = 0; v < 2; ++v) {
873                 STUtils.STStereoTrack.VIEW view = STUtils.STStereoTrack.VIEW.values()[v];
874                 trackSlope[half][v].fill(track.getSlope()[view.ordinal()]);
875                 trackIntercept[half][v].fill(track.getIntercept()[view.ordinal()]);

881         trackCount[1].fill(nTracks[1]);

883         // predict where these axial tracks go upstream in Y

885         for (STUtils.STTrack track : axialTracks) {
886             int half = track.isTop() ? 0 : 1;
887             for (int i = 0; i < (nPointsZ + 1); ++i) {
888                 zIter = startPointZ - i * (startPointZ - endPointZ) / (double) nPointsZ;
889                 yIter = track.predict(zIter);
890                 trackAxialExtraPolation[half].fill(zIter, yIter);
891                 trackAxialExtraPolation[2].fill(zIter, yIter);
892             }
893             start = wirePosition[2] + 350.0;
894             end = wirePosition[2] - 350.0;
895             for (int i = 0; i < (nPointsZ + 1); ++i) {
896                 zIter = start - i * (start - end) / (double) nPointsZ;
897                 yIter = track.predict(zIter);
898                 trackAxialExtraPolation[3].fill(zIter, yIter);
899             }
900             start = 200.0;
901             end = -200.0;
902             for (int i = 0; i < (nPointsZ + 1); ++i) {
903                 zIter = start - i * (start - end) / (double) nPointsZ;
904                 yIter = track.predict(zIter);
905                 trackAxialExtraPolation[4].fill(zIter, yIter);
906             }
907         }
908 
909         // predict where the stereo tracks go upstream in X-Y
910         double xyIter[];
911         for (STStereoTrack track : stereoTracks) {
912             int half = track.isTop() ? 0 : 1;
913             for (int i = 0; i < (nPointsZ + 1); ++i) {
914                 zIter = startPointZ - i * (startPointZ - endPointZ) / (double) nPointsZ;
915                 xyIter = track.predict(zIter);
916                 trackExtraPolationY[half].fill(zIter, xyIter[STStereoTrack.VIEW.YZ.ordinal()]);
917                 trackExtraPolationY[2].fill(zIter, xyIter[STStereoTrack.VIEW.YZ.ordinal()]);
918                 trackExtraPolationX[half].fill(zIter, xyIter[STStereoTrack.VIEW.XZ.ordinal()]);
919                 trackExtraPolationX[2].fill(zIter, xyIter[STStereoTrack.VIEW.XZ.ordinal()]);
920             }
921             start = wirePosition[2] + 350.0;
922             end = wirePosition[2] - 350.0;
923             for (int i = 0; i < (nPointsZ + 1); ++i) {
924                 zIter = start - i * (start - end) / (double) nPointsZ;
925                 xyIter = track.predict(zIter);
926                 trackExtraPolationY[3].fill(zIter, xyIter[STStereoTrack.VIEW.YZ.ordinal()]);
927                 trackExtraPolationX[3].fill(zIter, xyIter[STStereoTrack.VIEW.XZ.ordinal()]);
928             }
929             start = 200.0;
930             end = -200.0;
931             for (int i = 0; i < (nPointsZ + 1); ++i) {
932                 zIter = start - i * (start - end) / (double) nPointsZ;
933                 xyIter = track.predict(zIter);
934                 trackExtraPolationY[4].fill(zIter, xyIter[STStereoTrack.VIEW.YZ.ordinal()]);
935                 trackExtraPolationX[4].fill(zIter, xyIter[STStereoTrack.VIEW.XZ.ordinal()]);
936             }
937         }
938 
939         // GBL interface
940         if (writeGbl) {
941             if (gblPrintWriter == null)
942                 throw new RuntimeException("No file was opened!");
943             STUtils.printGBL(gblPrintWriter, event, stereoTracks);
944         }
945 
946     }
