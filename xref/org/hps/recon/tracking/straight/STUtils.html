101             return hits;
102         }

104         public double getIntercept() {
105             checkFit();
106             return getFit().intercept();
107         }
108 
109         public double getSlope() {
110             checkFit();
111             return getFit().slope();
112         }
113 
114         protected double getPath(double dz) {
115             double slope = getSlope();
116             double dy = slope * dz;
117             return Math.sqrt(dz * dz + dy * dy);

121             if (hits.size() == 0)
122                 throw new RuntimeException("need hits to determine half.");
123             return ((HpsSiSensor) hits.get(0).getRawHits().get(0).getDetectorElement()).isTopLayer();
124         }

127             List<double[]> l = new ArrayList<double[]>();
128             for (SiTrackerHitStrip1D hit : hits) {
129                 l.add(new double[] {hit.getPositionAsVector().z(), hit.getPositionAsVector().y()});
130             }
131             return l;
132         }

135             String s = "STTrack: \n";

138                 s += hit.getPositionAsVector().toString() + "("

250         /**
251          * Get list of points that would be used in a simple univariate regression. Note that one {@link VIEW} uses the
252          * path length.
253          * 
254          * @param view - {@link VIEW} to be obtained
255          * @return {@link List} of points for this track.
256          */
257         public List<double[]> getPointList(VIEW view) {
258             checkView(view);
259             List<double[]> l = new ArrayList<double[]>();
260             for (StereoPair hit : hits) {
261                 Hep3Vector p = hit.getPosition();
262                 // get the path length
263                 // in one view this is simple the z-position.
264                 double s = view.compareTo(VIEW.YZ) == 0 ? p.z() : getSignedPathLength(this, p.z(), VIEW.YZ);
265                 double y = view.equals(VIEW.YZ) ? p.y() : p.x();
266                 l.add(new double[] {s, y});
267             }
268             return l;
269         }

273             sb.append(hits.size() + " stereo hits.\n");
274             for (StereoPair pair : hits) {
275                 sb.append("Pair: pos " + pair.getPosition() + " "
276                         + pair.getAxial().getRawHits().get(0).getDetectorElement().getName() + " org "
277                         + getOrigin(pair.getAxial()).toString() + " "
278                         + pair.getStereo().getRawHits().get(0).getDetectorElement().getName() + " org "
279                         + getOrigin(pair.getStereo()).toString() + "\n");
280             }
281             sb.append("\nPointlists:\n");
282             for (int i = 0; i < 2; ++i) {
283                 VIEW v = VIEW.values()[i];
284                 sb.append("View " + v.name() + ": \n");
285                 for (double[] xy : getPointList(v))
286                     sb.append(xy[0] + ", " + xy[1] + "\n");
287                 if (getFit(v) != null)
288                     sb.append(getFit(v).toString() + "\n");
289             }
290             return sb.toString();
291         }

419     private static class LineFit extends STTrackFitter {
420 
421         private static final String TYPE = "LineFit";
422         private double errY = 6e-3;
423 
424         String getType() {
425             return TYPE;
426         }
427 
428         public void setErrY(double err_y) {
429             this.errY = err_y;
430         }
431 
432         public void fit(STTrack track) {
433             fit(track.getPointList());
434         }
435 
436         @Override
437         public void fit(List<double[]> xyList) {
438             int n = xyList.size();
439             double[] x = new double[n];
440             double[] y = new double[n];
441             double[] sigma_y = new double[n];
442             for (int i = 0; i < n; ++i) {
443                 x[i] = xyList.get(i)[0];
444                 y[i] = xyList.get(i)[1];
445                 sigma_y[i] = errY;
446             }
447             SlopeInterceptLineFitter fitter = new SlopeInterceptLineFitter();
448             fitter.fit(x, y, sigma_y, n);
449             setFit(new STTrackFit(TYPE, fitter.getFit()));
450         }

466     /**
467      * Basically an adaptation of what's in {@link HelicalTrackCross}.
468      * 
469      * @param origin
470      * @param strip1
471      * @param strip2
472      * @return the stereo hit position
473      */
474     static Hep3Vector getStereoHitPositionFromReference(Hep3Vector origin, SiTrackerHitStrip1D strip1,
475             SiTrackerHitStrip1D strip2) {
476         SiTrackerHitStrip1D s1 = strip1;
477         SiTrackerHitStrip1D s2 = strip2;
478         // sort in direction of track
479         // TODO do I need to sort these?
480         if (s1.getPositionAsVector().z() > s2.getPositionAsVector().z()) {
481             s1 = strip2;
482             s2 = strip1;
483         }
484 
485         StraightThroughAnalysisDriver.logger.finest("Calculate stereo hit position assuming track from "
486                 + origin.toString());
487 
488         // Get origin of sensors
489         Hep3Vector o1 = getOrigin(s1);
490         Hep3Vector o2 = getOrigin(s2);
491 
492         // Get sensor orientation
493         Hep3Vector u1 = s1.getMeasuredCoordinate();
494         Hep3Vector v1 = s1.getUnmeasuredCoordinate();
495         Hep3Vector w1 = getNormal(s1);
496         Hep3Vector u2 = s2.getMeasuredCoordinate();
497         Hep3Vector v2 = s2.getUnmeasuredCoordinate();
498         Hep3Vector w2 = getNormal(s2);
499 
500         // update origin calculation with displaced origin
501         o1 = VecOp.sub(o1, origin);
502         o2 = VecOp.sub(o2, origin);
503 
504         // For this to work the normal needs to point along track i.e. along z for both sensors
505         // so rotate coordinate system pi around u axis in this case
506         if (VecOp.dot(w1, o1) < 0) {
507             v1 = VecOp.mult(-1.0, v1);
508             w1 = VecOp.cross(u1, v1);
509         }
510         if (VecOp.dot(w2, o2) < 0) {
511             v2 = VecOp.mult(-1.0, v2);
512             w2 = VecOp.cross(u2, v2);
513         }
514 
515         StraightThroughAnalysisDriver.logger.finest("o1 " + o1.toString());
516         StraightThroughAnalysisDriver.logger.finest("u1 " + u1.toString());
517         StraightThroughAnalysisDriver.logger.finest("v1 " + v1.toString());
518         StraightThroughAnalysisDriver.logger.finest("w1 " + w1.toString());
519         StraightThroughAnalysisDriver.logger.finest("o2 " + o2.toString());
520         StraightThroughAnalysisDriver.logger.finest("u2 " + u2.toString());
521         StraightThroughAnalysisDriver.logger.finest("v2 " + v2.toString());
522         StraightThroughAnalysisDriver.logger.finest("w2 " + w2.toString());
523 
524         // Get the measured strip position on each sensor (basically origin + mult(umeas,u))
525         Hep3Vector p1 = s1.getPositionAsVector();
526         Hep3Vector p2 = s2.getPositionAsVector();
527         StraightThroughAnalysisDriver.logger.finest("p1 " + p1.toString());
528         StraightThroughAnalysisDriver.logger.finest("p2 " + p2.toString());
529 
530         // sin(stereo angle)
531         double sinAlpha = VecOp.dot(v1, u2);
532         StraightThroughAnalysisDriver.logger.finest("sinAlpha " + sinAlpha);
533 
534         // check
535         if (Math.abs(VecOp.dot(o1, w1)) < 0.0001)
536             throw new RuntimeException("o1 and w1 are orthogonal: o1 " + o1.toString() + " w1 " + w1.toString());
537 
538         // calculate effective distance along z
539         double gamma = VecOp.dot(o2, w2) / VecOp.dot(o1, w1);
540         StraightThroughAnalysisDriver.logger.finest("gamma " + gamma);
541 
542         // Get the vector between the two strip (center) positions
543         Hep3Vector dp = VecOp.sub(p2, VecOp.mult(gamma, p1));
544         StraightThroughAnalysisDriver.logger.finest("dp " + dp.toString());
545 
546         // Get the projection onto u2 and scale with stereo angle to get un-measured coordinate
547         double p1_v = VecOp.dot(dp, u2) / (gamma * sinAlpha);
548         StraightThroughAnalysisDriver.logger.finest("p1_v " + p1_v);
549 
550         // calculate the position of the hit on the strip
551         Hep3Vector r1 = VecOp.add(p1, VecOp.mult(p1_v, v1));
552         StraightThroughAnalysisDriver.logger.finest("r1 " + r1.toString());
553 
554         // Take the final position as the half-way between the sensors
555         Hep3Vector p = VecOp.mult(0.5 * (1 + gamma), r1);
556 
557         StraightThroughAnalysisDriver.logger.finest("p " + p.toString());
558 
559         return p;
560 
561     }

563     /**
564      * Basically the same as what's in {@link HitUtils}.
565      * 
566      * @param origin
567      * @param strip1
568      * @param strip2
569      */
570     static Hep3Vector getPosition(Hep3Vector t, SiTrackerHitStrip1D strip1, SiTrackerHitStrip1D strip2) {
571         SiTrackerHitStrip1D s1 = strip1;
572         SiTrackerHitStrip1D s2 = strip2;
573         // sort in direction of track
574         // TODO do I need to sort these?
575         if (s1.getPositionAsVector().z() > s2.getPositionAsVector().z()) {
576             s1 = strip2;
577             s2 = strip1;
578         }
579 
580         StraightThroughAnalysisDriver.logger.finest("Calculate stereo hit position assuming track direction "
581                 + t.toString());
582 
583         // Get origin of sensors
584         Hep3Vector o1 = getOrigin(s1);
585         Hep3Vector o2 = getOrigin(s2);
586 
587         // Get sensor orientation
588         Hep3Vector u1 = s1.getMeasuredCoordinate();
589         Hep3Vector v1 = s1.getUnmeasuredCoordinate();
590         Hep3Vector w1 = getNormal(s1);
591         Hep3Vector u2 = s2.getMeasuredCoordinate();
592         Hep3Vector v2 = s2.getUnmeasuredCoordinate();
593         Hep3Vector w2 = getNormal(s2);
594 
595         // For this to work the normal needs to point along track i.e. along z for both sensors
596         // so rotate coordinate system pi around u axis in this case
597         if (VecOp.dot(w1, o1) < 0) {
598             v1 = VecOp.mult(-1.0, v1);
599             w1 = VecOp.cross(u1, v1);
600         }
601         if (VecOp.dot(w2, o2) < 0) {
602             v2 = VecOp.mult(-1.0, v2);
603             w2 = VecOp.cross(u2, v2);
604         }
605 
606         StraightThroughAnalysisDriver.logger.finest("o1 " + o1.toString());
607         StraightThroughAnalysisDriver.logger.finest("u1 " + u1.toString());
608         StraightThroughAnalysisDriver.logger.finest("v1 " + v1.toString());
609         StraightThroughAnalysisDriver.logger.finest("w1 " + w1.toString());
610         StraightThroughAnalysisDriver.logger.finest("o2 " + o2.toString());
611         StraightThroughAnalysisDriver.logger.finest("u2 " + u2.toString());
612         StraightThroughAnalysisDriver.logger.finest("v2 " + v2.toString());
613         StraightThroughAnalysisDriver.logger.finest("w2 " + w2.toString());
614 
615         // Get the measured strip position on each sensor (basically origin + mult(umeas,u))
616         Hep3Vector p1 = s1.getPositionAsVector();
617         Hep3Vector p2 = s2.getPositionAsVector();
618         StraightThroughAnalysisDriver.logger.finest("p1 " + p1.toString());
619         StraightThroughAnalysisDriver.logger.finest("p2 " + p2.toString());
620 
621         // sin(stereo angle)
622         double sinAlpha = VecOp.dot(v1, u2);
623         StraightThroughAnalysisDriver.logger.finest("sinAlpha " + sinAlpha);
624 
625         // check
626         if (Math.abs(VecOp.dot(o1, w1)) < 0.0001)
627             throw new RuntimeException("o1 and w1 are orthogonal: o1 " + o1.toString() + " w1 " + w1.toString());
628 
629         // calculate the distance between the sensors
630         double gamma = VecOp.dot(w1, VecOp.sub(o2, o1)) / VecOp.dot(w1, t);
631         StraightThroughAnalysisDriver.logger.finest("gamma " + gamma);
632 
633         // Get the vector along the track direction
634         Hep3Vector tgamma = VecOp.mult(gamma, t);
635         StraightThroughAnalysisDriver.logger.finest("tgamma " + tgamma.toString());
636 
637         // Get the vector taking you from the midpoint of strip1 in the direction of the track
638         Hep3Vector p1Prime = VecOp.add(p1, tgamma);
639         StraightThroughAnalysisDriver.logger.finest("p1Prime " + p1Prime.toString());
640 
641         // Get the vector between the predicted and measured center values on strip2
642         Hep3Vector dp = VecOp.sub(p2, p1Prime);
643         StraightThroughAnalysisDriver.logger.finest("dp " + dp.toString());
644 
645         // Get the projection onto u2 and scale with stereo angle to get un-measured coordinate
646         double p1_v = VecOp.dot(dp, u2) / (sinAlpha);
647         StraightThroughAnalysisDriver.logger.finest("p1_v " + p1_v + " (u2=" + u2.toString() + ")");
648 
649         // calculate the position of the hit on the strip
650         Hep3Vector r1 = VecOp.add(p1, VecOp.mult(p1_v, v1));
651         StraightThroughAnalysisDriver.logger.finest("r1 " + r1.toString() + "(v1=" + v1.toString() + ")");
652 
653         // Take the final position as the half-way between the sensors
654         Hep3Vector p = VecOp.add(r1, VecOp.mult(0.5 * gamma, t));
655 
656         StraightThroughAnalysisDriver.logger.finest("p " + p.toString());
657 
658         return p;
659

815      * Calculate the multiple scattering angle for a given momentum and thickness
816      * 
817      * @param p
818      * @param radlength
819      * @return the multiple scattering angle for a given momentum and thickness
820      */
821     public static double msangle(double p, double radlength) {
822         double angle = (0.0136 / p) * Math.sqrt(radlength) * (1.0 + 0.038 * Math.log(radlength));
823         return angle;

839     /**
840      * Finds point of intercept between a generic straight line and a plane.
841      * 
842      * @param l - vector pointing along the line
843      * @param l0 - point on the line
844      * @param p0 - point on the plane
845      * @param n - normal vector of the plane.
846      * @return point of intercept.
847      */
848     private static Hep3Vector getLinePlaneIntercept(Hep3Vector l, Hep3Vector l0, Hep3Vector p0, Hep3Vector n) {
849         if (VecOp.dot(l, n) == 0)
850             throw new RuntimeException("This line and plane are parallel!");
851         final double d = VecOp.dot(VecOp.sub(p0, l0), n) / VecOp.dot(l, n);
852         Hep3Vector p = VecOp.add(VecOp.mult(d, l), l0);
853         return p;
854 
855     }

862      * @param track

865     private static Hep3Vector getLinePlaneIntercept(SiTrackerHitStrip1D strip, STStereoTrack track) {

881      * Calculate the residual (measured - predicted) for this hit in the measurement frame.
882      * 
883      * @param strip
884      * @param track
885      * @return the residual (measured - predicted) for the hit
886      */
887     protected static double getUResidual(SiTrackerHitStrip1D strip, STStereoTrack track) {
888 
889         // Predict track position on sensor
890         Hep3Vector trkpos = getLinePlaneIntercept(strip, track);
891 
892         logger.finest("trkpos " + trkpos.toString());
893         logger.finest("origin " + getOrigin(strip).toString());
894         logger.finest("u " + strip.getMeasuredCoordinate().toString() + " v "
895                 + strip.getUnmeasuredCoordinate().toString() + " w "
896                 + VecOp.cross(strip.getMeasuredCoordinate(), strip.getUnmeasuredCoordinate()).toString());
897 
898         // vector from origin to track position
899         Hep3Vector trkposOrigin = VecOp.sub(trkpos, getOrigin(strip));
900 
901         logger.finest("trkposOrigin " + trkposOrigin.toString());
902 
903         // trkposOriginLocal = globalToLocal.rotated(trkposOrigin);
904 
905         // Transform from JLab frame to sensor frame (done through the RawTrackerHit)
906         SiTrackerHitStrip1D local = strip.getTransformedHit(CoordinateSystem.SENSOR);
907         ITransform3D globalToLocal = local.getLocalToGlobal().inverse();
908 
909         logger.finest("local pos " + local.getPositionAsVector());
910 
911         logger.finest("localToGlobal:\n" + local.getLocalToGlobal().toString());
912         logger.finest("localToGlobal:\n" + local.getLocalToGlobal().inverse().toString());
913 
914         // Hep3Vector trkposOriginLocal = strip.getLocalToGlobal().getRotation().inverse().rotated(trkposOrigin);
915         Hep3Vector trkposOriginLocal = globalToLocal.getRotation().rotated(trkposOrigin);
916 
917         logger.finest("trkposOriginLocal " + trkposOriginLocal.toString());
918 
919         // get the measured position
920         Hep3Vector measpos = VecOp.sub(strip.getPositionAsVector(), getOrigin(strip));
921 
922         logger.finest("strip pos " + strip.getPositionAsVector().toString());
923         logger.finest("strip origin " + getOrigin(strip).toString());
924         logger.finest("measpos " + measpos.toString());
925 
926         // Hep3Vector measposLocal = strip.getLocalToGlobal().getRotation().inverse().rotated(measpos);
927         Hep3Vector measposLocal = globalToLocal.rotated(measpos);
928 
929         logger.finest("measposLocal " + measposLocal.toString());
930 
931         final double ures = measposLocal.x() - trkposOriginLocal.x();
932 
933         logger.finest("res " + VecOp.sub(measposLocal, trkposOriginLocal).toString());
934 
935         logger.finest("\nures " + ures + " \ntrkpos " + trkpos.toString() + " \ntrkposOriginLocal "
936                 + trkposOriginLocal.toString() + " \nmeaspos " + measpos.toString() + " \nmeasposLocal "
937                 + measposLocal.toString());
938 
939         return ures;

977     /**
978      * Compute the Jacobian for transporting uncertainties along the track a certain distance.
979      * 
980      * @param ds - distance along the track
981      * @return Jacobian matrix
982      */
983     private static BasicMatrix getSimpleJacobian(double ds) {
984 
985         // Spell out variables for readibility
986         double d_xT_d_xT = 1;
987         double d_xT_d_yT = 0;
988         double d_xT_d_xTprime = ds;
989         double d_xT_d_yTprime = 0;
990 
991         double d_yT_d_xT = 0;
992         double d_yT_d_yT = 1;
993         double d_yT_d_xTprime = 0;
994         double d_yT_d_yTprime = ds;
995 
996         double d_xTprime_d_xT = 0;
997         double d_xTprime_d_yT = 0;
998         double d_xTprime_d_xTprime = 1;
999         double d_xTprime_d_yTprime = 0;
1000 
1001         double d_yTprime_d_xT = 0;
1002         double d_yTprime_d_yT = 0;
1003         double d_yTprime_d_xTprime = 0;
1004         double d_yTprime_d_yTprime = 1;
1005 
1006         BasicMatrix jacobian = new BasicMatrix(4, 4);
1007         jacobian.setElement(0, 0, d_xT_d_xT);
1008         jacobian.setElement(0, 1, d_xT_d_yT);
1009         jacobian.setElement(0, 2, d_xT_d_xTprime);
1010         jacobian.setElement(0, 3, d_xT_d_yTprime);
1011 
1012         jacobian.setElement(1, 0, d_yT_d_xT);
1013         jacobian.setElement(1, 1, d_yT_d_yT);
1014         jacobian.setElement(1, 2, d_yT_d_xTprime);
1015         jacobian.setElement(1, 3, d_yT_d_yTprime);
1016 
1017         jacobian.setElement(2, 0, d_xTprime_d_xT);
1018         jacobian.setElement(2, 1, d_xTprime_d_yT);
1019         jacobian.setElement(2, 2, d_xTprime_d_xTprime);
1020         jacobian.setElement(2, 3, d_xTprime_d_yTprime);
1021 
1022         jacobian.setElement(3, 0, d_yTprime_d_xT);
1023         jacobian.setElement(3, 1, d_yTprime_d_yT);
1024         jacobian.setElement(3, 2, d_yTprime_d_xTprime);
1025         jacobian.setElement(3, 3, d_yTprime_d_yTprime);
1026 
1027         // debug

1029         sb.append("\nJacobian d(xT,yT,xT',yT')/d(xT,yT,xT',yT'):\n");
1030         for (int i = 0; i != 4; ++i) {

1041     /**
1042      * Computes projection matrix from the intercept variables x0,y0,z0 to the curvilinear xT,yT,zT variables.
1043      *
1044      * @param track - {@link STStereoTrack}
1045      * @return 3x3 projection matrix
1046      */
1047     private static Hep3Matrix curvilinearProjectionMatrix(Hep3Vector dir) {
1048 
1049         Hep3Vector X = new BasicHep3Vector(1, 0, 0);
1050         Hep3Vector Y = new BasicHep3Vector(0, 1, 0);
1051         Hep3Vector Z = new BasicHep3Vector(0, 0, 1);
1052 
1053         // build the curvilinear unit vectors: UVT
1054         Hep3Vector T = dir;
1055         Hep3Vector J = VecOp.mult(1. / VecOp.cross(T, Z).magnitude(), VecOp.cross(T, Z));
1056         Hep3Vector U = VecOp.mult(-1, J);
1057         Hep3Vector V = VecOp.cross(T, U);
1058         // Hep3Vector K = Z;
1059         // Hep3Vector I = VecOp.cross(J, K);
1060 
1061         BasicHep3Matrix trans = new BasicHep3Matrix();
1062         trans.setElement(0, 0, VecOp.dot(X, U));
1063         trans.setElement(0, 1, VecOp.dot(Y, U));
1064         trans.setElement(0, 2, VecOp.dot(Z, U));
1065         trans.setElement(1, 0, VecOp.dot(X, V));
1066         trans.setElement(1, 1, VecOp.dot(Y, V));
1067         trans.setElement(1, 2, VecOp.dot(Z, V));
1068         trans.setElement(2, 0, VecOp.dot(X, T));
1069         trans.setElement(2, 1, VecOp.dot(Y, T));
1070         trans.setElement(2, 2, VecOp.dot(Z, T));
1071 
1072         // debug
1073         StringBuffer sb = new StringBuffer();
1074         sb.append("\nU " + U.toString());
1075         sb.append("\nV " + V.toString());
1076         sb.append("\nT " + T.toString());
1077         sb.append("\nX " + X.toString());
1078         sb.append("\nY " + Y.toString());
1079         sb.append("\nZ " + Z.toString());
1080         for (int i = 0; i != 3; ++i) {
1081             sb.append("\n");
1082             for (int j = 0; j != 3; ++j) {
1083                 sb.append(trans.e(i, j) + "  ");
1084             }
1085         }
1086         logger.finest(sb.toString());
1087 
1088         return trans;
1089     }

1093      *
1094      * @param track - {@link STStereoTrack}
1095      * @return 3x3 projection matrix

1097     static Hep3Vector curvilinearProjection(double x0, double y0, Hep3Vector dir) {

1106

1108
