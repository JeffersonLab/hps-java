1   package org.hps.recon.tracking;

3   import java.util.Iterator;
4   import java.util.List;
5   
6   import org.lcsim.detector.tracker.silicon.HpsSiSensor;
7   import org.lcsim.event.EventHeader;
8   import org.lcsim.event.EventHeader.LCMetaData;
9   import org.lcsim.event.RawTrackerHit;
10  import org.lcsim.util.Driver;
11  
12  /**
13   * Driver used to filter out RawTrackerHits that have been identified to come
14   * from noisy/dead channels.  
15   */
16  public class SVTBadChannelFilterDriver extends Driver {
17  
18      // RawTrackerHit collection name 
19      private String rawTrackerHitCollection = "SVTRawTrackerHits";
20  
21      @Override
22      public void process(EventHeader event) {
23          
24          if (event.hasCollection(RawTrackerHit.class, rawTrackerHitCollection)) {
25              
26              // Get the list of raw hits from the event
27              List<RawTrackerHit> hits = event.get(RawTrackerHit.class, rawTrackerHitCollection);
28              
29              // Get the hits meta data from the event
30              LCMetaData meta = event.getMetaData(hits);
31              
32              // Iterate over all raw hits in the event.  If the raw hit is 
33              // identified to come from a noisy/bad channel, remove it from
34              // the list of raw hits.
35              Iterator<RawTrackerHit> hitsIterator = hits.iterator();
36              while (hitsIterator.hasNext()) {
37                  
38                  RawTrackerHit hit = hitsIterator.next();
39                  //hit.setMetaData(meta);
40                  int strip = hit.getIdentifierFieldValue("strip");
41                  HpsSiSensor sensor = (HpsSiSensor) hit.getDetectorElement();
42  
43                  if(sensor.isBadChannel(strip)){
44                      hitsIterator.remove();
45                  }
46  
47                  if (!sensor.getReadout().getHits(RawTrackerHit.class).isEmpty()) {
48                      throw new RuntimeException(this.getClass().getSimpleName() + " must be run before any SVT readout drivers.");
49                  }
50              }
51          }
52      }
53  }
