36      private void initialize(List<SeedStrategy> strategylist, boolean useHPSMaterialManager, boolean includeMS) {
37  
38          // Explicitly only replace the objects that might change to avoid getting the lcsim versions
39  
40          //  Instantiate the material manager for HPS,  the helix fitter and seed track finder as tey depends on the material manager
41          if (useHPSMaterialManager) {
42              MaterialSupervisor materialSupervisor = new MaterialSupervisor(includeMS);
43              materialSupervisor.setDebug(true);
44              _materialmanager = materialSupervisor;
45              _helixfitter = new HelixFitter(materialSupervisor, doIterativeHelix);
46          } else {
47              MaterialManager materialmanager = new MaterialManager(includeMS);
48              _materialmanager = materialmanager; //mess around with types here...
49              _helixfitter = new HelixFitter(materialmanager, doIterativeHelix);
50          }
51  
52          //  Instantiate the helix finder since it depends on the material manager
53          _finder = new SeedTrackFinder(_hitmanager, _helixfitter);
54  
55      }

105     @Override
106     protected void process(EventHeader event) {
107 
108         // System.out.println("New event");
109         // Pass the event to the diagnostics package
110         if (_diag != null)
111             _diag.setEvent(event);
112 
113         // Initialize timing
114         long last_time = System.currentTimeMillis();
115         long start_time = 0;
116         double dtime = 0;
117 
118         // Get the hit collection from the event
119         List<HelicalTrackHit> hitcol = event.get(HelicalTrackHit.class, _inputCol);
120 
121         // Sort the hits for this event
122         _hitmanager.OrganizeHits(hitcol);
123 
124         // Make the timing plots if requested
125         if (_timing) {
126             start_time = System.currentTimeMillis();
127             dtime = ((double) (start_time - last_time)) / 1000.;
128             last_time = start_time;
129             aida.cloud1D("Organize Hits").fill(dtime);
130         }
131 
132         // Make sure that we have cleared the list of track seeds in the finder
133         _finder.clearTrackSeedList();
134 
135         // Loop over strategies and perform track finding
136         for (SeedStrategy strategy : _strategylist) {
137             if(hitcol.size()>maxHelicalTrackHits)
138                 continue;
139             // Set the strategy for the diagnostics
140             if (_diag != null)
141                 _diag.fireStrategyChanged(strategy);
142 
143             FastCheck checker = new FastCheck(strategy, _bfield, _diag);
144             // Perform track finding under this strategy
145             _finder.FindTracks(strategy, _bfield, checker);
146 
147             // Make the timing plots if requested
148             if (_timing) {
149                 long time = System.currentTimeMillis();
150                 dtime = ((double) (time - last_time)) / 1000.;
151                 last_time = time;
152                 aida.cloud1D("Tracking time for strategy " + strategy.getName()).fill(dtime);
153             }
154         }
155 
156         // Get the list of final list of SeedCandidates
157         List<SeedCandidate> trackseeds = _finder.getTrackSeeds();
158         ((HelixFitter) _helixfitter).setIterative(true);
159 
160         if (_iterativeConfirmedFits > 0) {
161             // Iteratively re-fit tracks to take into account helix and hit position correlations
162             if (this.debug)
163                 System.out.printf("%s: Iteratively improve %d seeds\n", this.getClass().getSimpleName(), trackseeds.size());
164             List<SeedCandidate> seedsToRemove = new ArrayList<SeedCandidate>();
165             for (SeedCandidate seed : trackseeds) {
166                 SeedStrategy strategy = seed.getSeedStrategy();
167                 boolean success = false;
168                 for (int iterFit = 0; iterFit < _iterativeConfirmedFits; ++iterFit) {
169                     try {
170                         success = _helixfitter.FitCandidate(seed, strategy);
171                     } catch (Exception e) {
172                         System.out.printf("ERROR -- Exception in SeedTracker:: \n");
173                         if (e.getMessage().equals("NaN in track direction")) {
174                             System.out.printf("Known error - Nan in tack direction ... carry on, nothing to see.\n");
175                             e.printStackTrace();
176                         } else {
177                             throw (e); // Re-throw.
178                         }
179                     }
180                 }
181                 if (!success) {
182                     seedsToRemove.add(seed);
183                 } else {
184                     if (this.debug)
185                         System.out.printf("%s: done iterating, this seed will be added to event:\n%s\n", this.getClass().getSimpleName(), seed.toString());
186                 }
187             }
188             for (SeedCandidate badseed : seedsToRemove) {
189                 trackseeds.remove(badseed);
190             }
191         }
192 
193         // Make tracks from the final list of track seeds
194         _maketracks.Process(event, trackseeds, _bfield);
195 
196         // Save the MC Particles that have been seeded / confirmed if diagnostics are enabled
197         if (_diag != null) {
198             Set<MCParticle> seededmcpset = _finder.getSeededMCParticles();
199             List<MCParticle> seededmcp = new ArrayList<MCParticle>(seededmcpset);
200             event.put("SeededMCParticles", seededmcp, MCParticle.class, 0);
201             Set<MCParticle> confirmedmcpset = _finder.getConfirmedMCParticles();
202             List<MCParticle> confirmedmcp = new ArrayList<MCParticle>(confirmedmcpset);
203             event.put("ConfirmedMCParticles", confirmedmcp, MCParticle.class, 0);
204         }
205 
206         // Clear the list of track seeds accumulated in the track finder
207         _finder.clearTrackSeedList();
208 
209         // Make the total time plot if requested
210         if (_timing) {
211             long end_time = System.currentTimeMillis();
212             dtime = ((double) (end_time - start_time)) / 1000.;
213             aida.cloud1D("Total tracking time").fill(dtime);
214         }
215 
216         return;
217     }
