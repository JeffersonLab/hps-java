176     protected void detectorChanged(Detector detector) {
177 
178         // Make sure super-class setup is activated.
179         super.detectorChanged(detector);
180 
181         // Setup optional usage of beam positions from database.
182         final DatabaseConditionsManager mgr = DatabaseConditionsManager.getInstance();
183         if (this.useBeamPositionConditions && mgr.hasConditionsRecord("beam_positions")) {
184             LOGGER.config("Using beam position from conditions database");
185             BeamPositionCollection beamPositions
186                     = mgr.getCachedConditions(BeamPositionCollection.class, "beam_positions").getCachedData();
187             BeamPosition beamPositionCond = beamPositions.get(0);
188             beamPositionToUse = new double[]{
189                 beamPositionCond.getPositionZ(),
190                 beamPositionCond.getPositionX(),
191                 beamPositionCond.getPositionY()
192             };
193             if (this.useFixedVertexZPosition) {
194                 LOGGER.config("Using fixed Z position: " + this.beamPosition[0]);
195                 beamPositionToUse[0] = this.beamPosition[0];
196             }
197         } else {
198             LOGGER.config("Using beam position from steering file or default");
199             beamPositionToUse = beamPosition;
200         }
201         LOGGER.config("Using beam position [ Z, X, Y ]: " + String.format("[ %f, %f, %f ]",
202                 beamPositionToUse[0], beamPositionToUse[1], beamPositionToUse[2]));
203     }

377     @Override
378     protected List<ReconstructedParticle> particleCuts(List<ReconstructedParticle> finalStateParticles) {
379         List<ReconstructedParticle> goodFinalStateParticles = new ArrayList<ReconstructedParticle>();
380         for (ReconstructedParticle part : finalStateParticles) {
381             // good electrons
382             if (part.getCharge() == -1) {
383                 if (part.getMomentum().magnitude() < cuts.getMaxElectronP()) {
384                     if (includeUnmatchedTracksInFSP || part.getGoodnessOfPID() < cuts.getMaxMatchChisq()) {
385                         goodFinalStateParticles.add(part);
386                     }
387                 }
388             } // good positrons
389             else if (part.getCharge() == 1) {
390                 if (includeUnmatchedTracksInFSP || part.getGoodnessOfPID() < cuts.getMaxMatchChisq()) {
391                     goodFinalStateParticles.add(part);
392                 }
393             } // photons
394             else {
395                 goodFinalStateParticles.add(part);
396             }
397         }
398         return goodFinalStateParticles;
399     }

401     public void findV0s(List<ReconstructedParticle> electrons, List<ReconstructedParticle> positrons) {
402         List<ReconstructedParticle> goodElectrons = particleCuts(electrons);
403         List<ReconstructedParticle> goodPositrons = particleCuts(positrons);
404         for (ReconstructedParticle positron : goodPositrons) {
405             for (ReconstructedParticle electron : goodElectrons) {
406                 // Don't vertex a GBL track with a SeedTrack.
407                 if (TrackType.isGBL(positron.getType()) != TrackType.isGBL(electron.getType())) {
408                     continue;
409                 }
410 
411                 // Make V0 candidates
412                 try {
413                     this.makeV0Candidates(electron, positron);
414                 } catch (RuntimeException e) {
415                     e.printStackTrace();
416                     System.out.println("HpsReconParticleDriver::makeV0Candidates fails:: skipping ele/pos pair.");
417                     continue;
418                 }
419             }
420         }
421     }

423     public void findMollers(List<ReconstructedParticle> electrons) {
424         List<ReconstructedParticle> topElectrons = new ArrayList<ReconstructedParticle>();
425         List<ReconstructedParticle> botElectrons = new ArrayList<ReconstructedParticle>();
426 
427         for (ReconstructedParticle electron : electrons) {
428             if (electron.getTracks().get(0).getTrackStates().get(0).getTanLambda() > 0) {
429                 topElectrons.add(electron);
430             } else {
431                 botElectrons.add(electron);
432             }
433         }
434 
435         if (topElectrons.size() > 1 || botElectrons.size() > 1) {
436             return;
437         }
438 
439         // Iterate over the collection of electrons and create e-e- pairs 
440         for (ReconstructedParticle topElectron : topElectrons) {
441             for (ReconstructedParticle botElectron : botElectrons) {
442                 // Don't vertex a GBL track with a SeedTrack.
443                 if (TrackType.isGBL(topElectron.getType()) != TrackType.isGBL(botElectron.getType())) {
444                     continue;
445                 }
446 
447                 // Only vertex two particles if at least one strategy found both tracks. Take out this check once we reduce the number of tracks.
448                 //if ((topElectron.getType() & botElectron.getType() & 0x1f) == 0)
449                 //    continue;
450                 // Make Moller candidates
451                 this.makeMollerCandidates(topElectron, botElectron);
452             }
453         }
454     }

493     /**
494      * Fits a vertex from an electron/positron track pair using the indicated
495      * constraint.
496      *
497      * @param constraint - The constraint type to use.
498      * @param electron - The electron track.
499      * @param positron - The positron track.
500      * @return Returns the reconstructed vertex as a <code>BilliorVertex
501      * </code> object. mg--8/14/17--add the displaced vertex refit for the
502      * UNCONSTRAINED and BS_CONSTRAINED fits
503      */
504     private BilliorVertex fitVertex(Constraint constraint, ReconstructedParticle electron, ReconstructedParticle positron) {
505 
506         // Covert the tracks to BilliorTracks.
507         BilliorTrack electronBTrack = toBilliorTrack(electron.getTracks().get(0));
508         BilliorTrack positronBTrack = toBilliorTrack(positron.getTracks().get(0));
509 
510         // Create a vertex fitter from the magnetic field.
511         // Note that the vertexing code uses the tracking frame coordinates
512         // HPS X => TRACK Y
513         // HPS Y => TRACK Z
514         // HPS Z => TRACK X
515         BilliorVertexer vtxFitter = new BilliorVertexer(bField);
516         // TODO: The beam size should come from the conditions database.
517         vtxFitter.setBeamSize(beamSize);
518         vtxFitter.setBeamPosition(beamPositionToUse);
519         vtxFitter.setStoreCovTrkMomList(_storeCovTrkMomList);
520         vtxFitter.setDebug(debug);
521 
522         // Perform the vertexing based on the specified constraint.
523         switch (constraint) {
524             case UNCONSTRAINED:
525                 vtxFitter.doBeamSpotConstraint(false);
526                 break;
527             case BS_CONSTRAINED:
528                 vtxFitter.doBeamSpotConstraint(true);
529                 break;
530             case TARGET_CONSTRAINED:
531                 vtxFitter.doTargetConstraint(true);
532                 break;
533         }
534 
535         // Add the electron and positron tracks to a track list for
536         // the vertex fitter.
537         List<BilliorTrack> billiorTracks = new ArrayList<BilliorTrack>();
538         billiorTracks.add(electronBTrack);
539         billiorTracks.add(positronBTrack);
540 
541         // Find a vertex based on the tracks.
542         BilliorVertex vtx = vtxFitter.fitVertex(billiorTracks);
543 
544         int minLayEle = 6;
545         int minLayPos = 6;
546         List<TrackerHit> allTrackHits = electron.getTracks().get(0).getTrackerHits();
547         for (TrackerHit temp : allTrackHits) {
548             // Retrieve the sensor associated with one of the hits. This will
549             // be used to retrieve the layer number
550             HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) temp.getRawHits().get(0)).getDetectorElement();
551 
552             // Retrieve the layer number by using the sensor
553             int layer = (sensor.getLayerNumber() + 1) / 2;
554             if (layer < minLayEle) {
555                 minLayEle = layer;
556             }
557         }
558         allTrackHits = positron.getTracks().get(0).getTrackerHits();
559         for (TrackerHit temp : allTrackHits) {
560             // Retrieve the sensor associated with one of the hits. This will
561             // be used to retrieve the layer number
562             HpsSiSensor sensor = (HpsSiSensor) ((RawTrackerHit) temp.getRawHits().get(0)).getDetectorElement();
563 
564             // Retrieve the layer number by using the sensor
565             int layer = (sensor.getLayerNumber() + 1) / 2;
566             if (layer < minLayPos) {
567                 minLayPos = layer;
568             }
569         }
570         vtx.setLayerCode(minLayPos + minLayEle);
571         vtx.setProbability(DOF[constraint.ordinal()]);
572 
573         // mg 8/14/17 
574         // if this is an unconstrained or BS constrained vertex, propogate the 
575         // tracks to the vertex found in previous fit and do fit again
576         //  ...  this is required because the vertex fit assumes trajectories 
577         // change linearly about the reference point (which we initially guess to be 
578         // (0,0,0) while for long-lived decays there is significant curvature
579         if (constraint == Constraint.BS_CONSTRAINED || constraint == Constraint.UNCONSTRAINED) {
580             List<ReconstructedParticle> recoList = new ArrayList<ReconstructedParticle>();
581             recoList.add(electron);
582             recoList.add(positron);
583             List<BilliorTrack> shiftedTracks = shiftTracksToVertex(recoList, vtx.getPosition());
584 //            if (constraint == Constraint.BS_CONSTRAINED) {
585 //                Hep3Vector beamRelToNewRef = new BasicHep3Vector(-vtx.getPosition().z() + beamPosition[0], -vtx.getPosition().x() + beamPosition[1], 0);
586 ////                vtxFitter.setBeamPosition(beamRelToNewRef.v());
587 //                //mg 5/11/2018:  use referencePostion, separate from beam position  
588 //                vtxFitter.setReferencePosition(beamRelToNewRef.v());
589 //            }
590             //mg 5/11/2018:  use referencePosition, separate from beam position  
591             Hep3Vector newRefPoint = new BasicHep3Vector(vtx.getPosition().z(), vtx.getPosition().x(), 0);
592             vtxFitter.setReferencePosition(newRefPoint.v());
593 
594             BilliorVertex vtxNew = vtxFitter.fitVertex(shiftedTracks);
595             vtxNew.setLayerCode(vtx.getLayerCode());
596             vtxNew.setProbability(DOF[constraint.ordinal()]);
597             return vtxNew;
598         } else {
599             return vtx;
600         }
601     }

603     /**
604      *
605      */
606     private void makeV0Candidates(ReconstructedParticle electron, ReconstructedParticle positron) {
607 
608         //boolean eleIsTop = (electron.getTracks().get(0).getTrackerHits().get(0).getPosition()[2] > 0);
609         //boolean posIsTop = (positron.getTracks().get(0).getTrackerHits().get(0).getPosition()[2] > 0);
610         //This eleIsTop/posIsTop logic is valid for all track types [both Helix+GBL and Kalman]
611         boolean eleIsTop = (electron.getTracks().get(0).getTrackStates().get(0).getTanLambda() > 0);
612         boolean posIsTop = (positron.getTracks().get(0).getTrackStates().get(0).getTanLambda() > 0);
613 
614         if ((eleIsTop == posIsTop) && (!makeConversionCols)) {
615             return;
616         }
617 
618         if (electron.getClusters() == null || positron.getClusters() == null) {
619             return;
620         }
621         if (requireClustersForV0 && (electron.getClusters().isEmpty() || positron.getClusters().isEmpty())) {
622             return;
623         }
624         if (requireClustersForV0) {
625             double eleClusTime = ClusterUtilities.getSeedHitTime(electron.getClusters().get(0));
626             double posClusTime = ClusterUtilities.getSeedHitTime(positron.getClusters().get(0));
627 
628             if (Math.abs(eleClusTime - posClusTime) > cuts.getMaxVertexClusterDt()) {
629                 return;
630             }
631         }
632         // Handle UNCONSTRAINED case, to make decisions whether we store the vertexes.
633         // This is done here so that we either store all types, or none, but never a mix.
634         BilliorVertex vtxFit = fitVertex(Constraint.UNCONSTRAINED, electron, positron);
635 
636         ReconstructedParticle candidate = makeReconstructedParticle(electron, positron, vtxFit);
637 
638         if (candidate.getMomentum().magnitude() > cuts.getMaxVertexP()) {
639             return;
640         }
641 
642         if (candidate.getStartVertex().getProbability() < cuts.getMinVertexChisqProb()) {
643             return;
644         }
645 
646         // patch the track parameters at the found vertex
647         if (_patchVertexTrackParameters) {
648             patchVertex(vtxFit);
649         }
650         if (eleIsTop != posIsTop) {
651             unconstrainedV0Vertices.add(vtxFit);
652             unconstrainedV0Candidates.add(candidate);
653         } else {
654             unconstrainedVcVertices.add(vtxFit);
655             unconstrainedVcCandidates.add(candidate);
656         }
657 
658         // Create candidate particles for the other two constraints.
659         for (Constraint constraint : Constraint.values()) {
660             if (constraint == Constraint.UNCONSTRAINED) {
661                 continue;           // Skip the UNCONSTRAINED case, done already
662             }
663             // Generate a candidate vertex and particle.
664             vtxFit = fitVertex(constraint, electron, positron);
665 
666             candidate = makeReconstructedParticle(electron, positron, vtxFit);
667 
668             // Add the other candidate vertex and particle to the
669             // appropriate LCIO collection.
670             switch (constraint) {
671 
672                 case BS_CONSTRAINED:
673                     if (eleIsTop != posIsTop) {
674                         beamConV0Vertices.add(vtxFit);
675                         beamConV0Candidates.add(candidate);
676                     }
677                     break;
678 
679                 case TARGET_CONSTRAINED:
680                     if (eleIsTop != posIsTop) {
681                         targetConV0Vertices.add(vtxFit);
682                         targetConV0Candidates.add(candidate);
683                     }
684                     break;
685 
686             }
687         }
688     }

734     /**
735      * Creates a reconstructed V0 candidate particle from an electron, positron,
736      * and billior vertex.
737      *
738      * @param electron - The electron.
739      * @param positron - The positron.
740      * @param vtxFit - The billior vertex.
741      * @return Returns a reconstructed particle with properties generated from
742      * the child particles and vertex given as an argument.
743      */
744     public static ReconstructedParticle makeReconstructedParticle(ReconstructedParticle electron, ReconstructedParticle positron, BilliorVertex vtxFit) {
745 
746         // Create a new reconstructed particle to represent the V0
747         // candidate and populate it with the electron and positron.
748         ReconstructedParticle candidate = new BaseReconstructedParticle();
749         ((BaseReconstructedParticle) candidate).setStartVertex(vtxFit);
750         candidate.addParticle(electron);
751         candidate.addParticle(positron);
752 
753         // Set the type of the V0 particle.  This will only be needed for pass 2.
754         ((BaseReconstructedParticle) candidate).setType(electron.getType());
755 
756         // TODO: The calculation of the total fitted momentum should be
757         //       done within BilloirVertex.
758         // Calculate the candidate particle momentum and associate it
759         // with the reconstructed candidate particle.
760         ((BaseReconstructedParticle) candidate).setMass(vtxFit.getParameters().get("invMass"));
761         Hep3Vector fittedMomentum = new BasicHep3Vector(vtxFit.getParameters().get("p1X"),
762                 vtxFit.getParameters().get("p1Y"),
763                 vtxFit.getParameters().get("p1Z"));
764         fittedMomentum = VecOp.add(fittedMomentum, new BasicHep3Vector(vtxFit.getParameters().get("p2X"),
765                 vtxFit.getParameters().get("p2Y"),
766                 vtxFit.getParameters().get("p2Z")));
767         //mg 10/24/2017...billiorvertex now returns momentum in JLAB frame
768         //         fittedMomentum = CoordinateTransformations.transformVectorToDetector(fittedMomentum);
769 
770         // If both the electron and positron have an associated Ecal cluster,
771         // calculate the total energy and assign it to the V0 particle
772         double v0Energy = 0;
773         if (!electron.getClusters().isEmpty() && !positron.getClusters().isEmpty()) {
774             v0Energy += electron.getClusters().get(0).getEnergy();
775             v0Energy += positron.getClusters().get(0).getEnergy();
776         }
777 
778         HepLorentzVector fourVector = new BasicHepLorentzVector(v0Energy, fittedMomentum);
779         //((BasicHepLorentzVector) fourVector).setV3(fourVector.t(), fittedMomentum);
780         ((BaseReconstructedParticle) candidate).set4Vector(fourVector);
781 
782         // Set the charge of the particle
783         double particleCharge = electron.getCharge() + positron.getCharge();
784         ((BaseReconstructedParticle) candidate).setCharge(particleCharge);
785 
786         // VERBOSE :: Output the fitted momentum data.
787 //        printDebug("Fitted momentum in tracking frame: " + fittedMomentum.toString());
788 //        printDebug("Fitted momentum in detector frame: " + fittedMomentum.toString());
789         // Add the ReconstructedParticle to the vertex.
790         vtxFit.setAssociatedParticle(candidate);
791 
792         // Set the vertex position as the reference point of the V0 particle
793         ((BaseReconstructedParticle) candidate).setReferencePoint(vtxFit.getPosition());
794 
795         // Return the V0 candidate.
796         return candidate;
797     }

829     private void patchVertex(BilliorVertex v) {
830         ReconstructedParticle rp = v.getAssociatedParticle();
831         List<ReconstructedParticle> parts = rp.getParticles();
832         ReconstructedParticle rp1 = parts.get(0);
833         ReconstructedParticle rp2 = parts.get(1);
834         //p1
835         Track et = rp1.getTracks().get(0);
836         double etrackMom = rp1.getMomentum().magnitude();
837         HelicalTrackFit ehtf = TrackUtils.getHTF(et);
838         // propagate this to the vertex z position...
839         // Note that HPS y is lcsim z
840         double es = HelixUtils.PathToZPlane(ehtf, v.getPosition().y());
841         //Hep3Vector epointOnTrackAtVtx = HelixUtils.PointOnHelix(ehtf, es);
842         Hep3Vector edirOfTrackAtVtx = HelixUtils.Direction(ehtf, es);
843         Hep3Vector emomAtVtx = VecOp.mult(etrackMom, VecOp.unit(edirOfTrackAtVtx));
844         //p2
845         Track pt = rp2.getTracks().get(0);
846         double ptrackMom = rp2.getMomentum().magnitude();
847         HelicalTrackFit phtf = TrackUtils.getHTF(pt);
848         // propagate this to the vertex z position...
849         // Note that HPS y is lcsim z
850         double ps = HelixUtils.PathToZPlane(phtf, v.getPosition().y());
851         //Hep3Vector ppointOnTrackAtVtx = HelixUtils.PointOnHelix(phtf, ps);
852         Hep3Vector pdirOfTrackAtVtx = HelixUtils.Direction(phtf, ps);
853         Hep3Vector pmomAtVtx = VecOp.mult(ptrackMom, VecOp.unit(pdirOfTrackAtVtx));
854 
855         double mass = invMass(emomAtVtx, pmomAtVtx);
856         v.setVertexTrackParameters(emomAtVtx, pmomAtVtx, mass);
857     }

859     private double invMass(Hep3Vector p1, Hep3Vector p2) {
860         double me2 = 0.000511 * 0.000511;
861         double esum = sqrt(p1.magnitudeSquared() + me2) + sqrt(p2.magnitudeSquared() + me2);
862         double pxsum = p1.x() + p2.x();
863         double pysum = p1.y() + p2.y();
864         double pzsum = p1.z() + p2.z();
865 
866         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
867         double evtmass = esum * esum - psum * psum;
868 
869         if (evtmass > 0) {
870             return Math.sqrt(evtmass);
871         } else {
872             return -99;
873         }
874     }

876     private List<BilliorTrack> shiftTracksToVertex(List<ReconstructedParticle> particles, Hep3Vector vtxPos) {
877         ///     Ok...shift the reference point....        
878         double[] newRef = {vtxPos.z(), vtxPos.x(), 0.0};//the  TrackUtils.getParametersAtNewRefPoint method only shifts in xy tracking frame
879         List<BilliorTrack> newTrks = new ArrayList<BilliorTrack>();
880         for (ReconstructedParticle part : particles) {
881             BaseTrackState oldTS = (BaseTrackState) part.getTracks().get(0).getTrackStates().get(0);
882             double[] newParams = TrackUtils.getParametersAtNewRefPoint(newRef, oldTS);
883             SymmetricMatrix newCov = TrackUtils.getCovarianceAtNewRefPoint(newRef, oldTS.getReferencePoint(), oldTS.getParameters(), new SymmetricMatrix(5, oldTS.getCovMatrix(), true));
884             //mg...I don't like this re-casting, but toBilliorTrack only takes Track as input
885             BaseTrackState newTS = new BaseTrackState(newParams, newRef, newCov.asPackedArray(true), TrackState.AtIP, bField);
886             BilliorTrack electronBTrackShift = this.toBilliorTrack(newTS);
887             newTrks.add(electronBTrackShift);
888         }
889         return newTrks;
890     }
