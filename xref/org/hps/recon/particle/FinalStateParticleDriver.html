28  /**
29   * A Driver used to create unvertexed ReconstructedParticles from SVT tracks 
30   * and ECal clusters.  First, all track-cluster matches are found and used 
31   * to create ReconstructedParticles.  The remaining unmatched tracks and 
32   * clusters are also made into recon particles. The collection of 
33   * recon particles (FinalStateParticles) is then added to the event. 
34   */
35  public class FinalStateParticleDriver extends Driver {
36  
37      /// Collection of ECal clusters
38      private String ecalClustersCollectionName = "";
39  
40      /// Collection of SVT Tracks
41      private String trackCollectionName = "";
42  
43      /// Track cluster matcher.
44      private TrackClusterMatcher matcher;
45  
46      /// The name of the track-cluster matching algorithm to use. 
47      private String trackClusterMatcherAlgo = "TrackClusterMatcherMinDistance";
48     
49      /// Enable/disable track-cluster matching plots.
50      private boolean enableTrackClusterMatchPlots = false;
51  
52      /// Beam energy. This is retrieved from the conditions database. 
53      private double beamEnergy = 0;
54  
55      /// MOUSE cuts 
56      private StandardCuts cuts = null;

58      /**
59       * These are new for 2019 running and should be set to false in the steering
60       * file. Default values should replicate correct behavior for 2015 and 2016
61       * data.
62       */
63      private boolean useTrackPositionForClusterCorrection = true;

65      /**
66       * flipSign is a kludge...
67       * HelicalTrackFitter doesn't deal with B-fields in -ive Z correctly
68       * so we set the B-field in +iveZ and flip signs of fitted tracks
69       * Note: This should be -1 for test run configurations and +1 for
70       * prop-2014 configurations
71       */
72      private int flipSign = 1;

75      private boolean isMC = false;

141     /**
142      * Called by the framework before the process method when the detector 
143      * geometry changes. This method is gauranteed to be called once before 
144      * the first call to process.
145      *
146      * @param Detector The new detector
147      */
148     @Override
149     final protected void detectorChanged(Detector detector) { 
150   
151         // Get the beam energy from the conditions database
152         BeamEnergyCollection beamEnergyCollection = this.getConditionsManager()
153                 .getCachedConditions(
154                         BeamEnergyCollection.class, "beam_energies")
155                 .getCachedData();
156         beamEnergy = beamEnergyCollection.get(0).getBeamEnergy();
157 
158         // Instantiate the "MOUSE" cuts
159         cuts = new StandardCuts(beamEnergy);
160 
161         // By default, use the original track-cluster matching class
162         matcher = TrackClusterMatcherFactory.create(trackClusterMatcherAlgo);
163         matcher.setBFieldMap(detector.getFieldMap());
164         matcher.setTrackCollectionName(trackCollectionName);
165         matcher.enablePlots(enableTrackClusterMatchPlots);
166         
167         // Retrieve the ECal geometry
168         ecal = (HPSEcal3) detector.getSubdetector("Ecal");
169 
170     }

172     /**
173      * Called by the framework to process an event. 
174      *
175      * @param event The event to be processed
176      */
177     @Override
178     final protected void process(EventHeader event) {
179 
180         // Get the collection of ECal clusters from the event.  If there are no
181         // clusters in the event, create an empty collection to pass to the matcher.
182         List<Cluster> clusters 
183                 = event.hasCollection(Cluster.class, ecalClustersCollectionName)
184                 ? event.get(Cluster.class, ecalClustersCollectionName)
185                 : new ArrayList<Cluster>();
186 
187         // Get the collection of tracks from the event.  If a collection
188         // of tracks isn't found, create an empty collection to pass to the 
189         // matcher.
190         List<Track> tracks 
191                 = event.hasCollection(Track.class, trackCollectionName)
192                 ? event.get(Track.class, trackCollectionName)
193                 : new ArrayList<Track>();
194       
195         // Loop through all of the track collections present in theevent and 
196         // create final state particles. These particles include those which 
197         // are composed by tracks or clusters only as well as track-cluster pairs.
198         List<ReconstructedParticle> finalStateParticles 
199                 = makeFinalStateParticles(event, clusters, tracks);
200 
201         // Add the final state ReconstructedParticles to the event
202         event.put(finalStateParticlesCollName, 
203                   finalStateParticles, ReconstructedParticle.class, 0);
204         
205     }

207     /**
208      * Create a collection of ReconstructedParticles from a list of tracks
209      * and clusters.  The collection of ReconstructedParticles will include
210      * particles where a track-cluster have been matched and those that 
211      * are made of just tracks or clusters. 
212      *
213      * @param event The event to be processed.
214      * @param clusters The collection of ECal clusters. 
215      * @param tracks The collection of SVT tracks.
216      * @return A collection of ReconstructedParticles.
217      */
218     private List<ReconstructedParticle> makeFinalStateParticles(
219             EventHeader event, List<Cluster> clusters, List<Track> tracks) { 
220   
221         // Create the list of list needed by the track-cluster matcher
222         // TODO: OM: The track-cluster matchers need to be updated such that 
223         //  they just take a single list of tracks i.e. List<*> versus 
224         //  List<List<*>>.
225         List<List<Track>> trackColl = new ArrayList<List<Track>>();
226         trackColl.add(tracks);
227 
228         // Matcher returns a mapping of Tracks with matched Clusters.
229         HashMap<Track, Cluster> trackClusterPairs 
230                 = matcher.matchTracksToClusters(event, trackColl, clusters, cuts,
231                 flipSign, useTrackPositionForClusterCorrection, isMC, ecal, 
232                 beamEnergy);
233 
234         // Create a list in which to store reconstructed particles.
235         List<ReconstructedParticle> particles 
236                 = new ArrayList<ReconstructedParticle>();
237 
238         // Create a list of final state particles that are composed of a track
239         // or a track-cluster pair.
240         trackClusterPairs.entrySet().parallelStream()
241                 .filter(entry -> entry.getKey() != null)
242                 .forEach(entry -> particles.add(
243                             makeFinalStateParticle(entry.getKey(), 
244                                 entry.getValue())));
245 
246         // Create a list of final state particles out of unmatched clusters.
247         // These would be track-cluster pairs where the track == null.
248         trackClusterPairs.entrySet().parallelStream()
249                 .filter(entry -> entry.getKey() == null)
250                 .forEach(entry -> particles.add(
251                             makeReconstructedParticle(entry.getValue())));
252 
253         // If available, apply the corrections to the Ecal clusters using track
254         // information.
255         if (applyClusterCorrections) {
256             matcher.applyClusterCorrections(
257                     useTrackPositionForClusterCorrection, clusters, beamEnergy,
258                     ecal, isMC);
259         }
260 
261         return particles;
262     }

264     /**
265      * Create a ReconParticle out of a track and cluster or a track alone. 
266      * If  track-cluster match has been found, this method is used to create 
267      * a recon particle and to set all available information about the match.
268      * This method also handles the case where a track doesn't have a matching
269      * cluster.
270      *
271      * @param track The SVT track used to build this recon particle.
272      * @param cluster The ECal cluster used to build this recon particle.
273      * @return The ReconstructedParticle object.
274      */
275     private ReconstructedParticle makeFinalStateParticle(Track track, Cluster cluster) {
276 
277         // Create a reconstructed particle to represent the track or 
278         // track-cluster pair.
279         BaseReconstructedParticle particle = new BaseReconstructedParticle();
280 
281         // Store the track in the particle.
282         particle.addTrack(track);
283 
284         // Derive the charge of the particle from the track.
285         int charge = ((int) Math.signum(
286                     track.getTrackStates().get(0).getOmega())) * flipSign;
287         particle.setCharge(charge);
288 
289         // initialize PID quality to a junk value:
290         particle.setGoodnessOfPid(Integer.MIN_VALUE);
291 
292         // Extrapolate the particle ID from the track. Positively
293         // charged particles are assumed to be positrons and those
294         // with negative charges are assumed to be electrons.
295         int pdgID = (charge != 0) && charge > 0 ? -11 : 11;
296         particle.setParticleIdUsed(new SimpleParticleID(pdgID, 0, 0, 0));
297 
298         // If a cluster was found that matches the track...
299         double clusterEnergy = 0.;
300         if (cluster != null) {
301 
302             // Get the cluster energy
303             clusterEnergy = cluster.getEnergy();
304 
305             // Add cluster to the particle:
306             particle.addCluster(cluster);
307 
308             // Use PID quality to store track-cluster matching quality:
309             particle.setGoodnessOfPid(matcher.getMatchQC(cluster, particle));
310 
311             // Propogate PID to the cluster
312             if ((Math.abs(pdgID) == 11) && !disablePID)
313                 ((BaseCluster) cluster).setParticleId(pdgID);
314         }
315 
316         // Calculate and set the momentum of the track
317         Hep3Vector momentum 
318                 = CoordinateTransformations.transformVectorToDetector(
319                 new BasicHep3Vector(
320                     particle.getTracks()
321                             .get(0).getTrackStates().get(0).getMomentum()));
322         particle.set4Vector(new BasicHepLorentzVector(clusterEnergy, momentum));
323 
324         return particle;
325     }

327     /**
328      * Make a ReconstructedParticle out of a cluster.  
329      *
330      * @param cluster The ECal cluster used to make the recon particle.
331      * @return A ReconstructedParticle.
332      */
333     private ReconstructedParticle makeReconstructedParticle(Cluster cluster) {
334 
335         // Create a reconstructed particle to represent the unmatched cluster.
336         BaseReconstructedParticle particle = new BaseReconstructedParticle();
337 
338         // The particle is assumed to be a photon, since it did not leave a
339         // track.
340         int pdgID = 22;
341         particle.setParticleIdUsed(new SimpleParticleID(pdgID, 0, 0, 0));
342 
343         if (!disablePID)
344             ((BaseCluster) cluster).setParticleId(pdgID);
345 
346         // Add the cluster to the particle.
347         particle.addCluster(cluster);
348 
349         // Set the reconstructed particle properties based on the cluster 
350         // properties.
351         particle.setCharge(0);
352 
353         // Set the momentum of the cluster based on its energy
354         double clusterEnergy = cluster.getEnergy();
355         Hep3Vector momentum 
356                 = new BasicHep3Vector(particle.getClusters().get(0)
357                                               .getPosition());
358         momentum = VecOp.mult(clusterEnergy, VecOp.unit(momentum));
359         particle.set4Vector(
360                 new BasicHepLorentzVector(clusterEnergy, momentum));
361 
362         return particle;
363     }
