1   package org.hps.recon.utils;
2   
3   import org.lcsim.event.Cluster;
4   import org.lcsim.event.Track;
5   import org.lcsim.event.EventHeader;
6   import org.lcsim.event.ReconstructedParticle;
7   import org.lcsim.geometry.subdetector.HPSEcal3;
8   import org.lcsim.geometry.FieldMap;
9   import org.hps.record.StandardCuts;
10  import java.util.List;
11  import java.util.HashMap;
12  
13  /**
14   * This is an interface for creating TrackClusterMatcher algorithms used in
15   * reconstruction.
16   */
17  public interface TrackClusterMatcher {
18  
19      /**
20       * Return a map of Tracks with matched Clusters.
21       * Tracks may be returned with null Cluster match.
22       */
23      HashMap<Track,Cluster> matchTracksToClusters(EventHeader event, List<List<Track>> trackCollections, List<Cluster> clusters, StandardCuts cuts, int flipSign, boolean useCorrectedClusterPositionsForMatching, boolean isMC, HPSEcal3 ecal, double beamEnergy);
24  
25      /**
26       * If set to true in steering-file, apply cluster corrections before
27       * matching Tracs to Clusters.
28       */
29      void applyClusterCorrections(boolean useTrackPositionClusterCorrection, List<Cluster> clusters, double beamEnergy, HPSEcal3 ecal, boolean isMC);
30  
31      /**
32       * Return match quality of Track+Cluster pair
33       */
34      double getMatchQC(Cluster cluster, ReconstructedParticle particle);
35  
36      /**
37       * Set BField Map
38       */
39      void setBFieldMap(FieldMap bFieldMap);
40  
41      /**
42       * Set beam energy
43       */
44      void setBeamEnergy(double beamEnergy);
45      
46      void setSnapToEdge(boolean val);
47  
48      /**
49       * Pass generic parameterization file to matcher
50       */
51      void initializeParameterization(String fname);
52  
53      /**
54       * Set Track collection name to GBLTracks or KalmanFullTracks
55       */
56      void setTrackCollectionName(String trackCollectionName);
57  
58      void enablePlots(boolean enablePlots);
59  
60      void saveHistograms();
61  
62  
63  }
