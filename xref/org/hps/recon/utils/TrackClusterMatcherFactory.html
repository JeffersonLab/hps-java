3   /**
4    * This is a convenience class for creating specific Track-to-Cluster matching
5    * algorithms via their name in the package org.hps.recon.utils. They must
6    * implement the {@link TrackClusterMatcher} interface.
7    *
8    * @see TrackClusterMatcher
9    * @see AbstractTrackClusterMatcher
10   * @see TrackClusterMatcherNSigma
11   * @see TrackClusterMatcherMinDistance
12   */
13  public final class TrackClusterMatcherFactory {
14  
15      /**
16       * Dont instantiate this class.
17       */
18      private TrackClusterMatcherFactory() {
19  
20      }
21  
22      /**
23       * Create a Track to Cluster matching algorithm
24       * @param name The name of the matching algorithm
25       * @return The matching algorithm
26       * @throws IllegalArgumentException if there is no matcher found with name.
27       */
28      public static TrackClusterMatcher create(String name) {
29  
30          TrackClusterMatcher matcher = null;
31          try {
32              if (TrackClusterMatcherNSigma.class.getSimpleName().equals(name))
33                  matcher = new TrackClusterMatcherNSigma();
34              else if (TrackClusterMatcherMinDistance.class.getSimpleName().equals(name))
35                  matcher = new TrackClusterMatcherMinDistance();
36          }
37          catch (Exception e){
38              //No matcher by the given name has been found :(
39              throw new IllegalArgumentException("Unknown Mactcher algorithm " + name + " cannot be instantiated.", e);
40          }
41          return matcher;
42      }
43  }
