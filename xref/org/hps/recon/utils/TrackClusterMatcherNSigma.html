212      * Get distance between track and cluster.
213      * 
214      * @param cluster
215      * @param track
216      * @return distance between cluster and track
217      */
218     public double getDistance(Cluster cluster,Track track) {
219 
220         // Get the cluster position
221         Hep3Vector cPos = new BasicHep3Vector(cluster.getPosition());
222 
223         // Extrapolate the track to the Ecal cluster position
224         Hep3Vector tPos = null;
225         if (this.useAnalyticExtrapolator) {
226             tPos = TrackUtils.extrapolateTrack(track, cPos.z());
227         } else {
228             TrackState trackStateAtEcal = TrackUtils.getTrackStateAtECal(track);
229             tPos = new BasicHep3Vector(trackStateAtEcal.getReferencePoint());
230             tPos = CoordinateTransformations.transformVectorToDetector(tPos);
231         }
232 
233         return Math.sqrt(Math.pow(cPos.x()-tPos.x(),2)+Math.pow(cPos.y()-tPos.y(),2));
234     }

237      * Calculate #sigma between cluster-track x/y position at calorimeter.
238      *
239      * Based on Rafo's parameterizations.  Requires non-analytic extrapolation
240      * and uncorrected cluster positions.
241      * 
242      * @param cluster = position-uncorrected cluster
243      * @param particle recon particle with tracks
244      *
245      * @return #sigma between cluster and track positions
246      */
247     public double getNSigmaPosition(Cluster cluster, ReconstructedParticle particle) {
248         if (particle.getTracks().size()<1) return Double.MAX_VALUE;
249         Track track=particle.getTracks().get(0);
250         return getNSigmaPosition(cluster, track, new BasicHep3Vector(track.getTrackStates().get(0).getMomentum()).magnitude());

252     public double getNSigmaPosition(Cluster cluster, Track track, double p){
253 
254 
255         if (this.useAnalyticExtrapolator)
256             throw new RuntimeException("This is to be used with non-analytic extrapolator only.");
257 
258         // Get the cluster position:
259         Hep3Vector cPos = new BasicHep3Vector(cluster.getPosition());
260 
261         // whether track is in top half of detector:
262         final boolean isTopTrack = track.getTrackStates().get(0).getTanLambda() > 0;
263 
264         // ignore if track and cluster in different halves:
265         if (isTopTrack != cPos.y()>0) return Double.MAX_VALUE;
266 
267         // Get the extrapolated track position at the calorimeter:
268         TrackState trackStateAtEcal = TrackUtils.getTrackStateAtECal(track);
269         if(trackStateAtEcal == null){
270             // Track never made it to the ECAL, so it curled before doing this and probably extrapolateTrackUsingFieldMap aborted.
271             return Double.MAX_VALUE;
272         }
273         Hep3Vector tPos = new BasicHep3Vector(trackStateAtEcal.getReferencePoint());
274         tPos = CoordinateTransformations.transformVectorToDetector(tPos);
275 
276         boolean hasL6 = false;
277         for(TrackerHit hit : track.getTrackerHits()){
278             if(TrackUtils.getLayer(hit) == 11)
279                 hasL6 = true;
280         }
281 
282         // choose which parameterization of mean and sigma to use:
283         double dxMean[],dyMean[],dxSigm[],dySigm[];
284         int charge = TrackUtils.getCharge(track);
285 
286         if (charge>0) {
287             if (isTopTrack) {
288                 dxMean = !hasL6 ? paramMap.get("dxMeanTopPosiGBL_noL6") : paramMap.get("dxMeanTopPosiGBL_hasL6");
289                 dxSigm = !hasL6 ? paramMap.get("dxSigmTopPosiGBL_noL6") : paramMap.get("dxSigmTopPosiGBL_hasL6");
290                 dyMean = !hasL6 ? paramMap.get("dyMeanTopPosiGBL_noL6") : paramMap.get("dyMeanTopPosiGBL_hasL6");
291                 dySigm = !hasL6 ? paramMap.get("dySigmTopPosiGBL_noL6") : paramMap.get("dySigmTopPosiGBL_hasL6");
292             }
293             else {
294                 dxMean = !hasL6 ? paramMap.get("dxMeanBotPosiGBL_noL6") : paramMap.get("dxMeanBotPosiGBL_hasL6");
295                 dxSigm = !hasL6 ? paramMap.get("dxSigmBotPosiGBL_noL6") : paramMap.get("dxSigmBotPosiGBL_hasL6");
296                 dyMean = !hasL6 ? paramMap.get("dyMeanBotPosiGBL_noL6") : paramMap.get("dyMeanBotPosiGBL_hasL6");
297                 dySigm = !hasL6 ? paramMap.get("dySigmBotPosiGBL_noL6") : paramMap.get("dySigmBotPosiGBL_hasL6");
298             }
299         }
300         else if (charge<0) {
301             if (isTopTrack) {
302                 dxMean = !hasL6 ? paramMap.get("dxMeanTopElecGBL_noL6") : paramMap.get("dxMeanTopElecGBL_hasL6");
303                 dxSigm = !hasL6 ? paramMap.get("dxSigmTopElecGBL_noL6") : paramMap.get("dxSigmTopElecGBL_hasL6");
304                 dyMean = !hasL6 ? paramMap.get("dyMeanTopElecGBL_noL6") : paramMap.get("dyMeanTopElecGBL_hasL6");
305                 dySigm = !hasL6 ? paramMap.get("dySigmTopElecGBL_noL6") : paramMap.get("dySigmTopElecGBL_hasL6");
306             }
307             else {
308                 dxMean = !hasL6 ? paramMap.get("dxMeanBotElecGBL_noL6") : paramMap.get("dxMeanBotElecGBL_hasL6");
309                 dxSigm = !hasL6 ? paramMap.get("dxSigmBotElecGBL_noL6") : paramMap.get("dxSigmBotElecGBL_hasL6");
310                 dyMean = !hasL6 ? paramMap.get("dyMeanBotElecGBL_noL6") : paramMap.get("dyMeanBotElecGBL_hasL6");
311                 dySigm = !hasL6 ? paramMap.get("dySigmBotElecGBL_noL6") : paramMap.get("dySigmBotElecGBL_hasL6");
312             }
313         }
314         else return Double.MAX_VALUE;
315 
316 
317 
318         // Beyond the edges of the fits in momentum, assume that the parameters are constant:
319         if ((p > paramMap.get("pHigh_hasL6")[0]) && hasL6) 
320             p = paramMap.get("pHigh_hasL6")[0];
321         else if ((p > paramMap.get("pHigh_noL6")[0]) && !hasL6) 
322             p= paramMap.get("pHigh_noL6")[0];
323         else if ((p < paramMap.get("pLow_hasL6")[0]) && hasL6)
324             p = paramMap.get("pLow_hasL6")[0];
325         else if ((p < paramMap.get("pLow_noL6")[0]) && !hasL6)
326             p = paramMap.get("pLow_noL6")[0];
327 
328         // calculate measured mean and sigma of deltaX and deltaY for this energy:
329         double aDxMean=0,aDxSigm=0,aDyMean=0,aDySigm=0;
330         for (int ii=dxMean.length-1; ii>=0; ii--) aDxMean = dxMean[ii] + p*aDxMean;
331         for (int ii=dxSigm.length-1; ii>=0; ii--) aDxSigm = dxSigm[ii] + p*aDxSigm;
332         for (int ii=dyMean.length-1; ii>=0; ii--) aDyMean = dyMean[ii] + p*aDyMean;
333         for (int ii=dySigm.length-1; ii>=0; ii--) aDySigm = dySigm[ii] + p*aDySigm;
334 
335         //if the track's extrapolated position is within 1/2 a crystal width of the edge of 
336         // the ecal edge, then move it to be 1/2 a crystal from the edge in y.  
337 
338         Hep3Vector originalTPos = tPos;
339 
340         if(snapToEdge )
341             tPos= snapper.snapToEdge(tPos,cluster);
342 
343 
344         // calculate nSigma between track and cluster:
345         final double nSigmaX = (cPos.x() - tPos.x() - aDxMean) / aDxSigm;
346         final double nSigmaY = (cPos.y() - tPos.y() - aDyMean) / aDySigm;
347 
348         double nSigma = Math.sqrt(nSigmaX*nSigmaX + nSigmaY*nSigmaY);
349 
350         if(Math.abs(cPos.x()-tPos.x())<50 &&  Math.abs(cPos.y()-tPos.y())<50){
351             LOGGER.fine("TC MATCH RESULTS:");
352             LOGGER.fine("isTop:  " + isTopTrack);
353             LOGGER.fine("charge: " + charge);
354             LOGGER.fine("hasL6:  " + hasL6);
355             LOGGER.fine("p: " + p);
356             LOGGER.fine("cx: " + cPos.x());
357             LOGGER.fine("cy: " + cPos.y());
358             LOGGER.fine("tx: " + originalTPos.x());
359             LOGGER.fine("ty: " + originalTPos.y());
360 
361             LOGGER.fine("nSigmaX: " + nSigmaX);
362             LOGGER.fine("nSigmaY: " + nSigmaY);
363             LOGGER.fine("nSigma: " + nSigma);
364         }
365 
366         return nSigma;
367         //return Math.sqrt( 1 / ( 1/nSigmaX/nSigmaX + 1/nSigmaY/nSigmaY ) );
368     }

374     /**
375      * Determine if a track is matched to a cluster. Currently, this is
376      * determined by checking that the track and cluster are within the same
377      * detector volume of each other and that the extrapolated track position is
378      * within some defined distance of the cluster.
379      *
380      * @param cluster : The Ecal cluster to check
381      * @param track : The SVT track to check
382      * @return true if all cuts are pased, false otherwise.
383      */
384     public boolean isMatch(Cluster cluster, Track track) {
385 
386         // Check that the track and cluster are in the same detector volume.
387         // If not, there is no way they can be a match.
388         if ((track.getTrackStates().get(0).getTanLambda() > 0 && cluster.getPosition()[1] < 0)
389                 || (track.getTrackStates().get(0).getTanLambda() < 0 && cluster.getPosition()[1] > 0)) {
390             return false;
391         }
392 
393         // Get the cluster position
394         Hep3Vector clusterPosition = new BasicHep3Vector(cluster.getPosition());
395 
396         // Extrapolate the track to the Ecal cluster position
397         Hep3Vector trackPosAtEcal = null;
398         if (this.useAnalyticExtrapolator) {
399             trackPosAtEcal = TrackUtils.extrapolateTrack(track, clusterPosition.z());
400         } else {
401             TrackState trackStateAtEcal = TrackUtils.getTrackStateAtECal(track);
402             trackPosAtEcal = new BasicHep3Vector(trackStateAtEcal.getReferencePoint());
403             trackPosAtEcal = CoordinateTransformations.transformVectorToDetector(trackPosAtEcal);
404         }
405 
406         // Calculate the difference between the cluster position at the Ecal and
407         // the track in both the x and y directions
408         double deltaX = clusterPosition.x() - trackPosAtEcal.x();
409         double deltaY = clusterPosition.y() - trackPosAtEcal.y();
410 
411         if (enablePlots) {
412             if (track.getTrackStates().get(0).getTanLambda() > 0) {
413 
414                 plots1D.get("Ecal cluster x - track x @ Ecal - top - all").fill(deltaX);
415                 plots2D.get("Ecal cluster x v track x @ Ecal - top - all").fill(clusterPosition.x(),
416                         trackPosAtEcal.x());
417                 plots1D.get("Ecal cluster y - track y @ Ecal - top - all").fill(deltaY);
418                 plots2D.get("Ecal cluster y v track y @ Ecal - top - all").fill(clusterPosition.y(),
419                         trackPosAtEcal.y());
420 
421             } else if (track.getTrackStates().get(0).getTanLambda() < 0) {
422 
423                 plots1D.get("Ecal cluster x - track x @ Ecal - bottom - all").fill(deltaX);
424                 plots2D.get("Ecal cluster x v track x @ Ecal - bottom - all").fill(clusterPosition.x(),
425                         trackPosAtEcal.x());
426                 plots1D.get("Ecal cluster y - track y @ Ecal - bottom - all").fill(deltaY);
427                 plots2D.get("Ecal cluster y v track y @ Ecal - bottom - all").fill(clusterPosition.y(),
428                         trackPosAtEcal.y());
429             }
430         }
431 
432         // Check that dx and dy between the extrapolated track and cluster 
433         // positions is reasonable.  Different requirements are imposed on 
434         // top and bottom tracks in order to account for offsets.
435         if ((track.getTrackStates().get(0).getTanLambda() > 0 && (deltaX > paramMap.get("topClusterTrackMatchDeltaXHigh")[0]
436                         || deltaX < paramMap.get("topClusterTrackMatchDeltaXLow")[0]))
437                 || (track.getTrackStates().get(0).getTanLambda() < 0 && (deltaX > paramMap.get("bottomClusterTrackMatchDeltaXHigh")[0]
438                         || deltaX < paramMap.get("bottomClusterTrackMatchDeltaXLow")[0]))) {
439             return false;
440         }
441 
442         if ((track.getTrackStates().get(0).getTanLambda() > 0 && (deltaY > paramMap.get("topClusterTrackMatchDeltaYHigh")[0]
443                         || deltaY < paramMap.get("topClusterTrackMatchDeltaYLow")[0]))
444                 || (track.getTrackStates().get(0).getTanLambda() < 0 && (deltaY > paramMap.get("bottomClusterTrackMatchDeltaYHigh")[0]
445                         || deltaY < paramMap.get("bottomClusterTrackMatchDeltaYLow")[0]))) {
446             return false;
447         }
448 
449         if (enablePlots) {
450             if (track.getTrackStates().get(0).getTanLambda() > 0) {
451 
452                 plots1D.get("Ecal cluster x - track x @ Ecal - top - matched").fill(deltaX);
453                 plots2D.get("Ecal cluster x v track x @ Ecal - top - matched").fill(clusterPosition.x(),
454                         trackPosAtEcal.x());
455                 plots1D.get("Ecal cluster y - track y @ Ecal - top - matched").fill(deltaY);
456                 plots2D.get("Ecal cluster y v track y @ Ecal - top - matched").fill(clusterPosition.y(),
457                         trackPosAtEcal.y());
458 
459             } else if (track.getTrackStates().get(0).getTanLambda() < 0) {
460 
461                 plots1D.get("Ecal cluster x - track x @ Ecal - bottom - matched").fill(deltaX);
462                 plots2D.get("Ecal cluster x v track x @ Ecal - bottom - matched").fill(clusterPosition.x(),
463                         trackPosAtEcal.x());
464                 plots1D.get("Ecal cluster y - track y @ Ecal - bottom - matched").fill(deltaY);
465                 plots2D.get("Ecal cluster y v track y @ Ecal - bottom - matched").fill(clusterPosition.y(),
466                         trackPosAtEcal.y());
467             }
468         }
469 
470         // If all cuts are pased, return true.
471         return true;
472     }

569     /**
570      * Applies EcalCluster corrections, with option to use Track position for
571      * corrections.
572      */
573     @Override
574     public void applyClusterCorrections(boolean useTrackPositionForClusterCorrection, List<Cluster> clusters, double beamEnergy, HPSEcal3 ecal, boolean isMC){
575         // Apply the corrections to the Ecal clusters using track information, if available
576         for (Cluster cluster : clusters) {
577             if (cluster.getParticleId() != 0) {
578                 if (useTrackPositionForClusterCorrection && this.clusterToTrack.containsKey(cluster)) {
579                     Track matchedT = clusterToTrack.get(cluster);
580                     double ypos = TrackUtils.getTrackStateAtECal(matchedT).getReferencePoint()[2];
581                     ClusterCorrectionUtilities.applyCorrections(beamEnergy, ecal, cluster, ypos, isMC);
582                 } else {
583                     ClusterCorrectionUtilities.applyCorrections(beamEnergy, ecal, cluster, isMC);
584                 }
585             }
586         }
587

590     /**
591      * Return Map of Tracks with matched EcalCluster. 
592      * Tracks are not matched to unique Clusters, and the same Cluster may be
593      * matched to different Tracks.
594      * If Track is not matched to a Cluster, Map value set to null
595      */
596     @Override
597     public HashMap<Track,Cluster> matchTracksToClusters(EventHeader event, List<List<Track>> trackCollections, List<Cluster> clusters, StandardCuts cuts, int flipSign, boolean useCorrectedClusterPositionsForMatching, boolean isMC, HPSEcal3 ecal, double beamEnergy){
598 
599         //Relational Tables used to get Track data
600         RelationalTable hitToRotated = TrackUtils.getHitToRotatedTable(event);
601         RelationalTable hitToStrips = TrackUtils.getHitToStripsTable(event);
602 
603         // Create a list in which to store reconstructed particles.
604         List<ReconstructedParticle> particles = new ArrayList<ReconstructedParticle>();
605 
606         // Create a mapping of matched clusters to corresponding tracks.
607         this.clusterToTrack = new HashMap<Cluster, Track>();
608 
609         // Create a mapping of Tracks and their corresponding Clusters.
610         HashMap<Track, Cluster> trackClusterPairs = new HashMap<Track, Cluster>();
611 
612         // Loop through all of the track collections and try to match every
613         // track to a cluster. Allow a cluster to be matched to multiple
614         // tracks and use a probability (to be coded later) to determine what
615         // the best match is.
616         for (List<Track> tracks : trackCollections) {
617 
618             for (Track track : tracks){
619 
620                 //create a mapping of smallestNSigma Clusters and their sigma values
621                 HashMap<Cluster,Double> clusterNSigma = new HashMap<Cluster,Double>();
622 
623                 //Create a reconstructed particle to represent the track.
624                 //ReconstructedParticle particle = super.addTrackToParticle(track, flipSign);
625                 ReconstructedParticle particle = new BaseReconstructedParticle();
626                 // Store the track in the particle.
627                 particle.addTrack(track);
628 
629                 // Set the type of the particle. This is used to identify
630                 // the tracking strategy used in finding the track associated with
631                 // this particle.
632                 ((BaseReconstructedParticle) particle).setType(track.getType());
633 
634                 // Derive the charge of the particle from the track.
635                 int charge = (int) Math.signum(track.getTrackStates().get(0).getOmega());
636                 ((BaseReconstructedParticle) particle).setCharge(charge * flipSign);
637 
638                 // initialize PID quality to a junk value:
639                 ((BaseReconstructedParticle) particle).setGoodnessOfPid(9999);
640 
641                 // Extrapolate the particle ID from the track. Positively
642                 // charged particles are assumed to be positrons and those
643                 // with negative charges are assumed to be electrons.
644                 if (particle.getCharge() > 0) {
645                     ((BaseReconstructedParticle) particle).setParticleIdUsed(new SimpleParticleID(-11, 0, 0, 0));
646                 } else if (particle.getCharge() < 0) {
647                     ((BaseReconstructedParticle) particle).setParticleIdUsed(new SimpleParticleID(11, 0, 0, 0));
648                 }
649 
650                 double smallestNSigma = Double.MAX_VALUE;
651                 // try to find a matching cluster:
652                 Cluster matchedCluster = null;
653                 for(Cluster cluster : clusters) {
654                     double clusTime = ClusterUtilities.getSeedHitTime(cluster);
655                     double trkT = TrackUtils.getTrackTime(track, hitToStrips, hitToRotated);
656 
657                     if (Math.abs(clusTime - trkT - cuts.getTrackClusterTimeOffset()) > cuts.getMaxMatchDt()) {
658                         LOGGER.fine("Failed cluster-track deltaT!");
659                         LOGGER.fine(clusTime + "  " + trkT + "  " + cuts.getTrackClusterTimeOffset() + ">" + cuts.getMaxMatchDt());
660                         continue;
661                     }
662 
663                     //if the option to use corrected cluster positions is selected, then
664                     //create a copy of the current cluster, and apply corrections to it
665                     //before calculating nsigma.  Default is don't use corrections.  
666                     Cluster originalCluster = cluster;
667                     if (useCorrectedClusterPositionsForMatching) {
668                         BaseCluster clusterBase = new BaseCluster(cluster);
669                         clusterBase.setNeedsPropertyCalculation(false);
670                         cluster = clusterBase;
671                         double ypos = TrackUtils.getTrackStateAtECal(particle.getTracks().get(0)).getReferencePoint()[2];
672                         ClusterCorrectionUtilities.applyCorrections(beamEnergy, ecal, cluster, ypos, isMC);
673                     }                    
674 
675                     // normalized distance between this cluster and track:
676                     final double thisNSigma = this.getNSigmaPosition(cluster, particle);
677 
678                     if (enablePlots) {
679                         if (TrackUtils.getTrackStateAtECal(track) != null) {
680                             this.isMatch(cluster, track);
681                         }
682                     }
683 
684                     // ignore if matching quality doesn't make the cut:
685                     if (thisNSigma > MAXNSIGMAPOSITIONMATCH) {
686                         LOGGER.fine("Failed cluster-track NSigma Cut!");
687                         LOGGER.fine("match NSigma = " + thisNSigma + "; Max NSigma =  " + MAXNSIGMAPOSITIONMATCH);
688                         continue;
689                     }
690 
691                     // ignore if we already found a cluster that's a better match:
692                     if (thisNSigma > smallestNSigma) {
693                         LOGGER.fine("Already found a better match than this!");
694                         LOGGER.fine("match NSigma = " + thisNSigma + "; smallest NSigma =  " + smallestNSigma);
695                         continue;
696                     }
697                     // we found a new best cluster candidate for this track:
698                     smallestNSigma = thisNSigma;
699                     matchedCluster = originalCluster;
700 
701                     // prefer using GBL tracks to correct (later) the clusters, for some consistency:
702                     if (track.getType() >= 32 || !this.clusterToTrack.containsKey(matchedCluster)) {
703                         this.clusterToTrack.put(matchedCluster, track);
704                     }
705                 }
706                 trackClusterPairs.put(track, matchedCluster);
707                 if(enablePlots)
708                     plotMatchedPairs(track, matchedCluster, TrackUtils.getTrackTime(track, hitToStrips, hitToRotated), flipSign, cuts);
709             }
710         }
711         return trackClusterPairs;
712     }

714     private void plotMatchedPairs(Track track, Cluster cluster, double trkT, int flipSign, StandardCuts cuts){
715 
716         if(cluster == null)
717             return;
718 
719         int charge = (int) Math.signum(track.getTrackStates().get(0).getOmega())*flipSign;
720         double trackt = trkT;
721         double tanlambda = track.getTrackStates().get(0).getTanLambda();
722 
723         //Track momentum magnitude
724         double trackPmag = new BasicHep3Vector(track.getTrackStates().get(0).getMomentum()).magnitude();
725 
726         //Track position
727         Hep3Vector trackPosAtEcal;
728         if (this.useAnalyticExtrapolator) {
729             trackPosAtEcal = TrackUtils.extrapolateTrack(track, cluster.getPosition()[2]);
730         } else {
731             TrackState trackStateAtEcal = TrackUtils.getTrackStateAtECal(track);
732             trackPosAtEcal = new BasicHep3Vector(trackStateAtEcal.getReferencePoint());
733             trackPosAtEcal = CoordinateTransformations.transformVectorToDetector(trackPosAtEcal);
734         }
735         double trackx = trackPosAtEcal.x();
736         double tracky = trackPosAtEcal.y();
737         double trackz = trackPosAtEcal.z();
738 
739         //Cluster Position
740         double cluster_time = ClusterUtilities.getSeedHitTime(cluster);
741         double clusterx = cluster.getPosition()[0];
742         double clustery = cluster.getPosition()[1];
743         double clusterz = cluster.getPosition()[2];
744 
745         //Calculate position residuals
746         double dx = clusterx - trackx;
747         double dy = clustery - tracky;
748         double dz = clusterz - trackz;
749         double dr = Math.sqrt(Math.pow(clusterx-trackx,2) + Math.pow(clustery-tracky,2));
750         double dt = cluster_time - cuts.getTrackClusterTimeOffset() - trackt;
751 
752         if(charge < 0){
753             if(tanlambda > 0){
754                 plots2D.get(String.format("%s_ele_TOP_track_cluster_matched_pair_dx",this.trackCollectionName)).fill(trackPmag,trackx-clusterx);
755                 plots2D.get(String.format("%s_ele_TOP_track_cluster_matched_pair_dy",this.trackCollectionName)).fill(trackPmag,tracky-clustery);
756                 plots2D.get(String.format("%s_ele_TOP_track_cluster_matched_pair_dz",this.trackCollectionName)).fill(trackPmag,trackz-clusterz);
757             }
758             else{
759                 plots2D.get(String.format("%s_ele_BOTTOM_track_cluster_matched_pair_dx",this.trackCollectionName)).fill(trackPmag,trackx-clusterx);
760                 plots2D.get(String.format("%s_ele_BOTTOM_track_cluster_matched_pair_dy",this.trackCollectionName)).fill(trackPmag,tracky-clustery);
761                 plots2D.get(String.format("%s_ele_BOTTOM_track_cluster_matched_pair_dz",this.trackCollectionName)).fill(trackPmag,trackz-clusterz);
762             }
763         }
764         else{
765             if(tanlambda > 0){
766                 plots2D.get(String.format("%s_pos_TOP_track_cluster_matched_pair_dx",this.trackCollectionName)).fill(trackPmag,trackx-clusterx);
767                 plots2D.get(String.format("%s_pos_TOP_track_cluster_matched_pair_dy",this.trackCollectionName)).fill(trackPmag,tracky-clustery);
768                 plots2D.get(String.format("%s_pos_TOP_track_cluster_matched_pair_dz",this.trackCollectionName)).fill(trackPmag,trackz-clusterz);
769             }
770             else{
771                 plots2D.get(String.format("%s_pos_BOTTOM_track_cluster_matched_pair_dx",this.trackCollectionName)).fill(trackPmag,trackx-clusterx);
772                 plots2D.get(String.format("%s_pos_BOTTOM_track_cluster_matched_pair_dy",this.trackCollectionName)).fill(trackPmag,tracky-clustery);
773                 plots2D.get(String.format("%s_pos_BOTTOM_track_cluster_matched_pair_dz",this.trackCollectionName)).fill(trackPmag,trackz-clusterz);
774             }
775         }
