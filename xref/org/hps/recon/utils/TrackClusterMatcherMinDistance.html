106     @Override
107     public void applyClusterCorrections(boolean useTrackPositionForClusterCorrection, List<Cluster> clusters, double beamEnergy, HPSEcal3 ecal, boolean isMC){
108 
109         // Apply the corrections to the Ecal clusters using track information, if available
110         for (Cluster cluster : clusters) {
111             if (cluster.getParticleId() != 0) {
112                 if (useTrackPositionForClusterCorrection && this.clusterToTrack.containsKey(cluster)) {
113                     Track matchedT = clusterToTrack.get(cluster);
114                     double ypos = TrackUtils.getTrackStateAtECal(matchedT).getReferencePoint()[2];
115                     ClusterCorrectionUtilities.applyCorrections(beamEnergy, ecal, cluster, ypos, isMC);
116                 } else {
117                     ClusterCorrectionUtilities.applyCorrections(beamEnergy, ecal, cluster, isMC);
118                 }
119             }
120         }
121     }

317     private List<Double> getTrackPositionAtEcal(Track track){
318 
319         double trackx;
320         double tracky;
321         double trackz;
322 
323         if (this.trackCollectionName.contains("GBLTracks")){
324             trackx = TrackUtils.getTrackStateAtECal(track).getReferencePoint()[1];
325             tracky = TrackUtils.getTrackStateAtECal(track).getReferencePoint()[2];
326             trackz = TrackUtils.getTrackStateAtECal(track).getReferencePoint()[0];
327         }
328         else {
329             TrackData trackdata = (TrackData) trackToData.from(track);
330             TrackState ts_ecal = TrackUtils.getTrackStateAtECal(track);
331             //If trackstate is null, upstream extrapolation error. Skip this
332             //Track
333             if(ts_ecal == null){
334                 return null;
335             }
336             double[] ts_ecalPos = ts_ecal.getReferencePoint();
337             trackx = ts_ecalPos[1];
338             tracky = ts_ecalPos[2];
339             trackz = ts_ecalPos[0];
340         }
341         List<Double> trackxyz = new ArrayList<Double>();
342         trackxyz.add(trackx);
343         trackxyz.add(tracky);
344         trackxyz.add(trackz);
345         return trackxyz;
346     }

349 
350         //GBL
351         double trackt = 99999;
352         if (this.trackCollectionName.contains("GBLTracks")){
353             trackt = TrackUtils.getTrackTime(track, hitToStrips, hitToRotated);
354         }
355 
356         //KF
357         else{
358             TrackData trackdata = (TrackData) trackToData.from(track);
359             trackt = trackdata.getTrackTime();
360         }
361         return trackt;
362     }

364     private HashMap<Track,Cluster> trackClusterMatcher(List<List<Track>> trackCollections, EventHeader event, List<Cluster> clusters, StandardCuts cuts, int flipSign, boolean useCorrectedClusterPositionsForMatching, boolean isMC, HPSEcal3 ecal, double beamEnergy)  {
365 
366         /*
367          * Matching algorithm uses a closest distance match between Track and
368          * Cluster.
369          * This matcher works for both GBL and KF Tracks.
370          * Returns Map of Tracks matched do unique Clusters
371          */
372         trackToData = new BaseRelationalTable(RelationalTable.Mode.ONE_TO_ONE, RelationalTable.Weighting.UNWEIGHTED);
373 
374         //define cuts
375         this.maxMatchDt = cuts.getMaxMatchDt();
376         this.maxMatchDx = cuts.getMaxMatchDx();
377         this.maxMatchDy = cuts.getMaxMatchDy();
378 
379         //Initialize clusterToTrack map, used for applying cluster corrections
380         this.clusterToTrack = new HashMap<Cluster, Track>();
381 
382         // Map stores all Track Cluster pairs in Event.
383         HashMap<Track, Cluster> alltrackClusterPairs = new HashMap<Track, Cluster>();
384 
385         //GBL Tracks require rotation tables
386         if(this.trackCollectionName.contains("GBLTracks")){
387             hitToRotated = TrackUtils.getHitToRotatedTable(event);
388             hitToStrips = TrackUtils.getHitToStripsTable(event);
389         }
390 
391         //For KF
392         //Relational Table is used to get KF track time from
393         //KFTrackDataRelations
394         if(this.trackCollectionName.contains("KalmanFullTracks")){
395             List<TrackData> TrackData;
396             List<LCRelation> trackRelations;
397             TrackData trackdata;
398             if (this.trackCollectionName.contains("KalmanFullTracks")) {
399                 TrackData = event.get(TrackData.class, "KFTrackData");
400                 trackRelations = event.get(LCRelation.class, "KFTrackDataRelations");
401                 for (LCRelation relation : trackRelations) {
402                     if (relation != null && relation.getTo() != null){
403                         trackToData.add(relation.getFrom(), relation.getTo());
404                     }
405                 }
406             }
407         }
408 
409         if(enablePlots){
410             plotClusterValues(clusters,cuts.getTrackClusterTimeOffset());
411         }
412 
413         //For all Track lists in trackCollections, match Tracks to Clusters
414         for (List<Track> tracks : trackCollections) {
415 
416             if(tracks == null || tracks.isEmpty() || clusters == null || clusters.isEmpty()){
417                 continue;
418             }
419 
420             //Map relating Tracks to potential Cluster matches, and their
421             //position residuals
422             Map<Track, Map<Cluster, Double>> trackClusterResidualsMap = new HashMap<Track, Map<Cluster, Double>>(); 
423 
424             for(Track track : tracks) {
425 
426                  // Derive the charge of the particle from the track.
427                 int charge = (int) Math.signum(track.getTrackStates().get(0).getOmega())*flipSign;
428 
429                 double trackt = getTrackTime(track);
430 
431                 //Track positions depend on GBL or KF
432                 List<Double> trackPos = this.getTrackPositionAtEcal(track);
433                 //If trackPos null, Track extrapolation to Ecal failed. Skip.
434                 if(trackPos == null){
435                     //
436                     trackClusterResidualsMap.put(track, null);
437                     continue;
438                 }
439                 double trackx = trackPos.get(0);
440                 double tracky = trackPos.get(1);
441                 double trackz = trackPos.get(2);
442 
443 
444                 //Defines Track top or bottom
445                 double tanlambda = track.getTrackParameter(4);
446 
447                 //Track momentum magnitude
448                 double trackPmag = new BasicHep3Vector(track.getTrackStates().get(0).getMomentum()).magnitude();
449 
450                 //Plots
451                 if(enablePlots){
452                     if (charge > 0) {
453                         plots1D.get(String.format("%s_pos_track_time",this.trackCollectionName)).fill(trackt);
454                         plots1D.get(String.format("%s_pos_track_momentum",this.trackCollectionName)).fill(trackPmag);
455                         plots1D.get(String.format("%s_pos_track_x",this.trackCollectionName)).fill(trackx);
456                         plots1D.get(String.format("%s_pos_track_y",this.trackCollectionName)).fill(tracky);
457                         plots1D.get(String.format("%s_pos_track_z",this.trackCollectionName)).fill(trackz);
458                     }
459                     else {
460                         plots1D.get(String.format("%s_ele_track_time",this.trackCollectionName)).fill(trackt);
461                         plots1D.get(String.format("%s_ele_track_momentum",this.trackCollectionName)).fill(trackPmag);
462                         plots1D.get(String.format("%s_ele_track_x",this.trackCollectionName)).fill(trackx);
463                         plots1D.get(String.format("%s_ele_track_y",this.trackCollectionName)).fill(tracky);
464                         plots1D.get(String.format("%s_ele_track_z",this.trackCollectionName)).fill(trackz);
465                     }
466                 }
467 
468                 //Begin Cluster Matching Algorithm
469                 Map<Cluster, Double> cluster_dr_Map = new HashMap<Cluster, Double>();
470                 Cluster matchedCluster = null;
471 
472                 //Loop over all clusters, looking for best match to current track
473                 double smallestdt = Double.MAX_VALUE;
474                 double smallestdr = Double.MAX_VALUE;
475                 for(Cluster cluster : clusters) {
476                     //if the option to use corrected cluster positions is selected, then
477                     //create a copy of the current cluster, and apply corrections to it
478                     //before calculating position residual.  Default is don't use corrections.  
479                     Cluster originalcluster = cluster;
480                     if (useCorrectedClusterPositionsForMatching) {
481                         BaseCluster clusterBase = new BaseCluster(cluster);
482                         clusterBase.setNeedsPropertyCalculation(false);
483                         cluster = clusterBase;
484                         double ypos = tracky;
485                         ClusterCorrectionUtilities.applyCorrections(beamEnergy, ecal, cluster, ypos, isMC);
486                     }
487 
488                     double clusterEnergy = cluster.getEnergy();
489                     double cluster_time = ClusterUtilities.getSeedHitTime(cluster);
490                     double dt = cluster_time - cuts.getTrackClusterTimeOffset() - trackt;
491 
492                     double clusterx = cluster.getPosition()[0];
493                     double clustery = cluster.getPosition()[1];
494                     double clusterz = cluster.getPosition()[2];
495 
496                     //Calculate position residuals
497                     double dx = clusterx - trackx + this.maxMatchDxOffset;
498                     double dy = clustery - tracky + this.maxMatchDyOffset;
499                     double dz = clusterz - trackz;
500                     double dr = Math.sqrt(Math.pow(clusterx-trackx,2) + Math.pow(clustery-tracky,2));
501 
502                     //Check that Top/Bottom Track is only matched to
503                     //Top/Bottom Clusters
504                     if(clustery > 0 && tanlambda < 0)
505                         continue;
506                     if(clustery < 0 && tanlambda > 0)
507                         continue;
508 
509                     //If true, will fill track cluster position residual plots
510                     //to be used externally to generate track+cluster residual
511                     //param map
512                     if(buildParamHistos){
513                         trackClusterResidualParameterization(track, cluster);
514                     }
515 
516                     if(enablePlots) {
517                         if(charge > 0) {
518                             //Time residual plot
519                             plots1D.get(String.format("%s_pos_track_cluster_dt",this.trackCollectionName)).fill(dt);
520                             //Track@Ecal Cluster Residuals
521                             plots1D.get(String.format("%s_pos_track_cluster_dx",this.trackCollectionName)).fill(dx);
522                             plots1D.get(String.format("%s_pos_track_cluster_dy",this.trackCollectionName)).fill(dy);
523                             plots1D.get(String.format("%s_pos_track_cluster_dz",this.trackCollectionName)).fill(dz);
524                             plots1D.get(String.format("%s_pos_track_cluster_dr",this.trackCollectionName)).fill(dr);
525                         }
526                         else {
527                             //Time residual plot
528                             plots1D.get(String.format("%s_ele_track_cluster_dt",this.trackCollectionName)).fill(dt);
529                             //Track@Ecal Cluster Residuals
530                             plots1D.get(String.format("%s_ele_track_cluster_dx",this.trackCollectionName)).fill(dx);
531                             plots1D.get(String.format("%s_ele_track_cluster_dy",this.trackCollectionName)).fill(dy);
532                             plots1D.get(String.format("%s_ele_track_cluster_dz",this.trackCollectionName)).fill(dz);
533                             plots1D.get(String.format("%s_ele_track_cluster_dr",this.trackCollectionName)).fill(dr);
534                         }
535                     }
536 
537                     //If position and time residual cuts are passed, build map of
538                     //all cluster position residuals with this track
539                     if((Math.abs(dt) < this.maxMatchDt) && (Math.abs(dx) < this.maxMatchDx) && (Math.abs(dy) < this.maxMatchDy) ) {
540                         cluster_dr_Map.put(originalcluster, dr);
541                     }
542                 }
543 
544                 //Position residual between Track and each potential Cluster is
545                 //stored in a map so that best match can be chosen
546                 trackClusterResidualsMap.put(track, cluster_dr_Map);
547             }
548 
549             //Match Track to the closest Ecal Cluster
550             //Check for Clusters that are matched to multiple Tracks, and sort
551             //until only unique matches remain
552             //trackClusterPairs maps Track to the Cluster with the smallest dr
553             Map<Track,Cluster> trackClusterPairs = new HashMap<Track, Cluster>();
554 
555             for(int i=0; i < clusters.size(); i++){
556                 trackClusterPairs = getMinTrackClusterResidualMap(trackClusterResidualsMap);
557                 trackClusterResidualsMap = checkDuplicateClusterMatching(trackClusterResidualsMap,trackClusterPairs);
558             }
559             trackClusterPairs = getMinTrackClusterResidualMap(trackClusterResidualsMap);
560             //add Track Cluster pairs for this trackCollection to map for all trackCollections
561             for(Map.Entry<Track, Cluster> entry : trackClusterPairs.entrySet()){
562                 alltrackClusterPairs.put(entry.getKey(), entry.getValue());
563             }
564         }
565         if(enablePlots)
566             plotMatchedPairs(alltrackClusterPairs, flipSign, cuts);
567         return alltrackClusterPairs;
568     }

661     private Map<Track, Map<Cluster,Double>> checkDuplicateClusterMatching(Map<Track, Map<Cluster,Double>> trackClusterResidualsMap, Map<Track, Cluster> trackClusterPairs){
662         
663         boolean duplicateCluster = false;
664         List<Track> sharedClusterTracks = new ArrayList<Track>();
665         List<Track> skipTracks = new ArrayList<Track>();
666         
667         for(Track track : trackClusterPairs.keySet()){
668             Map<Track, Cluster> trackswDuplicateClusters = new HashMap<Track, Cluster>();
669             if(skipTracks.contains(track))
670                 continue;
671             Cluster smallestdrCluster = trackClusterPairs.get(track);
672             if(smallestdrCluster == null)
673                 continue;
674             for(Track otherTrack : trackClusterPairs.keySet()){
675                 if(skipTracks.contains(track))
676                     continue;
677                 if(otherTrack == track)
678                     continue;
679                 Cluster othersmallestdrCluster = trackClusterPairs.get(otherTrack);
680                 if(othersmallestdrCluster == smallestdrCluster)
681                 {
682                     duplicateCluster = true;
683                     trackswDuplicateClusters.put(track, smallestdrCluster);
684                     trackswDuplicateClusters.put(otherTrack, othersmallestdrCluster);
685                 }
686             }
687 
688             double smallestdr = 99999.0;
689             Track smallestdrTrack = null;
690             if(trackswDuplicateClusters == null)
691                 return trackClusterResidualsMap;
692             for(Track duptrack : trackswDuplicateClusters.keySet()){
693                 double dr = trackClusterResidualsMap.get(duptrack).get(trackswDuplicateClusters.get(duptrack));
694                 if(dr < smallestdr){
695                     smallestdr = dr;
696                     smallestdrTrack = duptrack;
697                 }
698             }
699             for(Track duptrack : trackswDuplicateClusters.keySet()){
700                 skipTracks.add(duptrack);
701                 if(duptrack != smallestdrTrack){
702                     trackClusterResidualsMap.get(duptrack).remove(trackswDuplicateClusters.get(duptrack));
703                 }
704             }
705         }
706         return trackClusterResidualsMap;
707     }

709     private Map<Track,Cluster>  getMinTrackClusterResidualMap(Map<Track, Map<Cluster, Double>> trackClusterResidualsMap){
710 
711         //inputs a mapping of tracks with residuals for each possible cluster
712         //from all clusters in the map, for each track, match the cluster with
713         //the smallest position residual to that track
714         //build output map of track -> closest cluster
715         Map<Track,Cluster> Map = new HashMap<Track, Cluster>();
716         for(Track track : trackClusterResidualsMap.keySet()){
717             double smallestdr = 99999.0;
718             Cluster smallestdrCluster = null;
719             Map<Cluster, Double> cluster_dr_Map = trackClusterResidualsMap.get(track);
720             if(cluster_dr_Map == null){
721                 Map.put(track, null);
722                 continue;
723             }
724             for(Cluster c : cluster_dr_Map.keySet()){
725                 double dr = cluster_dr_Map.get(c);
726                 if(dr < smallestdr){
727                     // prefer using GBL tracks to correct (later) the clusters, for some consistency:
728                     if (track.getType() >= 32 || !this.clusterToTrack.containsKey(smallestdrCluster)) {
729                         this.clusterToTrack.put(smallestdrCluster, track);
730                     }
731 
732                     smallestdr = dr;
733                     smallestdrCluster = c;
734                 }
735             }
736             Map.put(track, smallestdrCluster);
737         }
738         return Map;
739     }
