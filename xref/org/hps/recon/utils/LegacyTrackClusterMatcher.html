80      public void initializeParameterization(String fname) {
81  
82          java.io.InputStream fis = LegacyTrackClusterMatcher.class.getResourceAsStream(fname);
83          //fis = new java.io.FileInputStream(fin);
84          java.io.BufferedReader br = new java.io.BufferedReader(new java.io.InputStreamReader(fis));
85  
86          paramMap = new HashMap<String, double[]>();
87          String line = null;
88  
89          try {
90              while ((line = br.readLine()) != null) {
91                  // test line length
92                  String[] arrOfStr = line.split(" ");
93                  if (arrOfStr.length < 2) {
94                      System.out.printf("TrackClusterError reading paramaterization file");
95                      continue;
96                  }
97                  double[] paramVals = new double[arrOfStr.length - 1];
98  
99                  for (int i=1; i<arrOfStr.length; i++) {
100                     paramVals[i-1] = new Double(arrOfStr[i]);
101                 }
102                 paramMap.put(arrOfStr[0], paramVals);
103                 //                Scanner s = new Scanner(line);
104                 //                String paramName = s.next();
105                 //                
106                 //                while (s.hasNext()) {
107                 //                float f = s.nextFloat();
108 
109 
110             }
111         } catch (IOException x) {
112             System.err.format("LegacyTrackClusterMatcher error reading parameterization file: %s%n", x);
113         }
114 
115         try {
116             br.close();
117         } catch (IOException x) {
118             System.err.format("LegacyTrackClusterMatcher error closing parameterization file: %s%n", x);
119         }

168     /**
169      * Get distance between track and cluster.
170      * 
171      * @param cluster
172      * @param track
173      * @return distance between cluster and track
174      */
175     public double getDistance(Cluster cluster,Track track) {
176 
177         // Get the cluster position
178         Hep3Vector cPos = new BasicHep3Vector(cluster.getPosition());
179 
180         // Extrapolate the track to the Ecal cluster position
181         Hep3Vector tPos = null;
182         if (this.useAnalyticExtrapolator) {
183             tPos = TrackUtils.extrapolateTrack(track, cPos.z());
184         } else {
185             TrackState trackStateAtEcal = TrackUtils.getTrackStateAtECal(track);
186             tPos = new BasicHep3Vector(trackStateAtEcal.getReferencePoint());
187             tPos = CoordinateTransformations.transformVectorToDetector(tPos);
188         }
189 
190         return Math.sqrt(Math.pow(cPos.x()-tPos.x(),2)+Math.pow(cPos.y()-tPos.y(),2));
191     }

193     /**
194      * Calculate #sigma between cluster-track x/y position at calorimeter.
195      *
196      * Based on Rafo's parameterizations.  Requires non-analytic extrapolation
197      * and uncorrected cluster positions.
198      * 
199      * @param cluster = position-uncorrected cluster
200      * @param particle recon particle with tracks
201      *
202      * @return #sigma between cluster and track positions
203      */
204     public double getNSigmaPosition(Cluster cluster, ReconstructedParticle particle) {
205         if (particle.getTracks().size()<1) return Double.MAX_VALUE;
206         Track track=particle.getTracks().get(0);
207         return getNSigmaPosition(cluster, track, new BasicHep3Vector(track.getTrackStates().get(0).getMomentum()).magnitude());
208     }
209     public double getNSigmaPosition(Cluster cluster, Track track, double p){
210 
211 
212         if (this.useAnalyticExtrapolator)
213             throw new RuntimeException("This is to be used with non-analytic extrapolator only.");
214 
215         // Get the cluster position:
216         Hep3Vector cPos = new BasicHep3Vector(cluster.getPosition());
217 
218         // whether track is in top half of detector:
219         final boolean isTopTrack = track.getTrackStates().get(0).getTanLambda() > 0;
220 
221         // ignore if track and cluster in different halves:
222         if (isTopTrack != cPos.y()>0) return Double.MAX_VALUE;
223 
224         // Get the extrapolated track position at the calorimeter:
225         TrackState trackStateAtEcal = TrackUtils.getTrackStateAtECal(track);
226         if(trackStateAtEcal == null){
227             // Track never made it to the ECAL, so it curled before doing this and probably extrapolateTrackUsingFieldMap aborted.
228             return Double.MAX_VALUE;
229         }
230         Hep3Vector tPos = new BasicHep3Vector(trackStateAtEcal.getReferencePoint());
231         tPos = CoordinateTransformations.transformVectorToDetector(tPos);
232 
233         boolean hasL6 = false;
234         for(TrackerHit hit : track.getTrackerHits()){
235             if(TrackUtils.getLayer(hit) == 11)
236                 hasL6 = true;
237         }
238 
239         // choose which parameterization of mean and sigma to use:
240         double dxMean[],dyMean[],dxSigm[],dySigm[];
241         int charge = TrackUtils.getCharge(track);
242 
243         if (charge>0) {
244             if (isTopTrack) {
245                 dxMean = !hasL6 ? paramMap.get("dxMeanTopPosiGBL_noL6") : paramMap.get("dxMeanTopPosiGBL_hasL6");
246                 dxSigm = !hasL6 ? paramMap.get("dxSigmTopPosiGBL_noL6") : paramMap.get("dxSigmTopPosiGBL_hasL6");
247                 dyMean = !hasL6 ? paramMap.get("dyMeanTopPosiGBL_noL6") : paramMap.get("dyMeanTopPosiGBL_hasL6");
248                 dySigm = !hasL6 ? paramMap.get("dySigmTopPosiGBL_noL6") : paramMap.get("dySigmTopPosiGBL_hasL6");
249             }
250             else {
251                 dxMean = !hasL6 ? paramMap.get("dxMeanBotPosiGBL_noL6") : paramMap.get("dxMeanBotPosiGBL_hasL6");
252                 dxSigm = !hasL6 ? paramMap.get("dxSigmBotPosiGBL_noL6") : paramMap.get("dxSigmBotPosiGBL_hasL6");
253                 dyMean = !hasL6 ? paramMap.get("dyMeanBotPosiGBL_noL6") : paramMap.get("dyMeanBotPosiGBL_hasL6");
254                 dySigm = !hasL6 ? paramMap.get("dySigmBotPosiGBL_noL6") : paramMap.get("dySigmBotPosiGBL_hasL6");
255             }
256         }
257         else if (charge<0) {
258             if (isTopTrack) {
259                 dxMean = !hasL6 ? paramMap.get("dxMeanTopElecGBL_noL6") : paramMap.get("dxMeanTopElecGBL_hasL6");
260                 dxSigm = !hasL6 ? paramMap.get("dxSigmTopElecGBL_noL6") : paramMap.get("dxSigmTopElecGBL_hasL6");
261                 dyMean = !hasL6 ? paramMap.get("dyMeanTopElecGBL_noL6") : paramMap.get("dyMeanTopElecGBL_hasL6");
262                 dySigm = !hasL6 ? paramMap.get("dySigmTopElecGBL_noL6") : paramMap.get("dySigmTopElecGBL_hasL6");
263             }
264             else {
265                 dxMean = !hasL6 ? paramMap.get("dxMeanBotElecGBL_noL6") : paramMap.get("dxMeanBotElecGBL_hasL6");
266                 dxSigm = !hasL6 ? paramMap.get("dxSigmBotElecGBL_noL6") : paramMap.get("dxSigmBotElecGBL_hasL6");
267                 dyMean = !hasL6 ? paramMap.get("dyMeanBotElecGBL_noL6") : paramMap.get("dyMeanBotElecGBL_hasL6");
268                 dySigm = !hasL6 ? paramMap.get("dySigmBotElecGBL_noL6") : paramMap.get("dySigmBotElecGBL_hasL6");
269             }
270         }
271         else return Double.MAX_VALUE;
272 
273 
274 
275         // Beyond the edges of the fits in momentum, assume that the parameters are constant:
276         if ((p > paramMap.get("pHigh_hasL6")[0]) && hasL6) 
277             p = paramMap.get("pHigh_hasL6")[0];
278         else if ((p > paramMap.get("pHigh_noL6")[0]) && !hasL6) 
279             p= paramMap.get("pHigh_noL6")[0];
280         else if ((p < paramMap.get("pLow_hasL6")[0]) && hasL6)
281             p = paramMap.get("pLow_hasL6")[0];
282         else if ((p < paramMap.get("pLow_noL6")[0]) && !hasL6)
283             p = paramMap.get("pLow_noL6")[0];
284 
285         // calculate measured mean and sigma of deltaX and deltaY for this energy:
286         double aDxMean=0,aDxSigm=0,aDyMean=0,aDySigm=0;
287         for (int ii=dxMean.length-1; ii>=0; ii--) aDxMean = dxMean[ii] + p*aDxMean;
288         for (int ii=dxSigm.length-1; ii>=0; ii--) aDxSigm = dxSigm[ii] + p*aDxSigm;
289         for (int ii=dyMean.length-1; ii>=0; ii--) aDyMean = dyMean[ii] + p*aDyMean;
290         for (int ii=dySigm.length-1; ii>=0; ii--) aDySigm = dySigm[ii] + p*aDySigm;
291 
292         //if the track's extrapolated position is within 1/2 a crystal width of the edge of 
293         // the ecal edge, then move it to be 1/2 a crystal from the edge in y.  
294 
295         Hep3Vector originalTPos = tPos;
296 
297         if(snapToEdge )
298             tPos= snapper.snapToEdge(tPos,cluster);
299 
300 
301         // calculate nSigma between track and cluster:
302         final double nSigmaX = (cPos.x() - tPos.x() - aDxMean) / aDxSigm;
303         final double nSigmaY = (cPos.y() - tPos.y() - aDyMean) / aDySigm;
304 
305         double nSigma = Math.sqrt(nSigmaX*nSigmaX + nSigmaY*nSigmaY);
306 
307         if(debug && Math.abs(cPos.x()-tPos.x())<50 &&  Math.abs(cPos.y()-tPos.y())<50){
308             System.out.println("TC MATCH RESULTS:");
309             System.out.println("isTop:  " + isTopTrack);
310             System.out.println("charge: " + charge);
311             System.out.println("hasL6:  " + hasL6);
312             System.out.println("p: " + p);
313             System.out.println("cx: " + cPos.x());
314             System.out.println("cy: " + cPos.y());
315             System.out.println("tx: " + originalTPos.x());
316             System.out.println("ty: " + originalTPos.y());
317 
318             System.out.println("nSigmaX: " + nSigmaX);
319             System.out.println("nSigmaY: " + nSigmaY);
320             System.out.println("nSigma: " + nSigma);
321         }
322 
323         return nSigma;
324         //return Math.sqrt( 1 / ( 1/nSigmaX/nSigmaX + 1/nSigmaY/nSigmaY ) );
325     }

331     /**
332      * Determine if a track is matched to a cluster. Currently, this is
333      * determined by checking that the track and cluster are within the same
334      * detector volume of each other and that the extrapolated track position is
335      * within some defined distance of the cluster.
336      *
337      * @param cluster : The Ecal cluster to check
338      * @param track : The SVT track to check
339      * @return true if all cuts are pased, false otherwise.
340      */
341     public boolean isMatch(Cluster cluster, Track track) {
342 
343         // Check that the track and cluster are in the same detector volume.
344         // If not, there is no way they can be a match.
345         if ((track.getTrackStates().get(0).getTanLambda() > 0 && cluster.getPosition()[1] < 0)
346                 || (track.getTrackStates().get(0).getTanLambda() < 0 && cluster.getPosition()[1] > 0)) {
347             return false;
348         }
349 
350         // Get the cluster position
351         Hep3Vector clusterPosition = new BasicHep3Vector(cluster.getPosition());
352         //System.out.println("Cluster Position: " + clusterPosition.toString());
353 
354         // Extrapolate the track to the Ecal cluster position
355         Hep3Vector trackPosAtEcal = null;
356         if (this.useAnalyticExtrapolator) {
357             //System.out.println("Using analytic field extrapolator."); 
358             trackPosAtEcal = TrackUtils.extrapolateTrack(track, clusterPosition.z());
359         } else {
360             //System.out.println("Using field map extrapolator"); 
361             TrackState trackStateAtEcal = TrackUtils.getTrackStateAtECal(track);
362             trackPosAtEcal = new BasicHep3Vector(trackStateAtEcal.getReferencePoint());
363             trackPosAtEcal = CoordinateTransformations.transformVectorToDetector(trackPosAtEcal);
364         }
365         //System.out.println("Track position at Ecal: " + trackPosAtEcal.toString());
366 
367         // Calculate the difference between the cluster position at the Ecal and
368         // the track in both the x and y directions
369         double deltaX = clusterPosition.x() - trackPosAtEcal.x();
370         double deltaY = clusterPosition.y() - trackPosAtEcal.y();
371 
372         //System.out.println("delta X: " + deltaX);
373         //System.out.println("delta Y: " + deltaY);
374         if (enablePlots) {
375             if (track.getTrackStates().get(0).getTanLambda() > 0) {
376 
377                 plots1D.get("Ecal cluster x - track x @ Ecal - top - all").fill(deltaX);
378                 plots2D.get("Ecal cluster x v track x @ Ecal - top - all").fill(clusterPosition.x(),
379                         trackPosAtEcal.x());
380                 plots1D.get("Ecal cluster y - track y @ Ecal - top - all").fill(deltaY);
381                 plots2D.get("Ecal cluster y v track y @ Ecal - top - all").fill(clusterPosition.y(),
382                         trackPosAtEcal.y());
383 
384             } else if (track.getTrackStates().get(0).getTanLambda() < 0) {
385 
386                 plots1D.get("Ecal cluster x - track x @ Ecal - bottom - all").fill(deltaX);
387                 plots2D.get("Ecal cluster x v track x @ Ecal - bottom - all").fill(clusterPosition.x(),
388                         trackPosAtEcal.x());
389                 plots1D.get("Ecal cluster y - track y @ Ecal - bottom - all").fill(deltaY);
390                 plots2D.get("Ecal cluster y v track y @ Ecal - bottom - all").fill(clusterPosition.y(),
391                         trackPosAtEcal.y());
392             }
393         }
394 
395         // Check that dx and dy between the extrapolated track and cluster 
396         // positions is reasonable.  Different requirements are imposed on 
397         // top and bottom tracks in order to account for offsets.
398         if ((track.getTrackStates().get(0).getTanLambda() > 0 && (deltaX > paramMap.get("topClusterTrackMatchDeltaXHigh")[0]
399                 || deltaX < paramMap.get("topClusterTrackMatchDeltaXLow")[0]))
400                 || (track.getTrackStates().get(0).getTanLambda() < 0 && (deltaX > paramMap.get("bottomClusterTrackMatchDeltaXHigh")[0]
401                         || deltaX < paramMap.get("bottomClusterTrackMatchDeltaXLow")[0]))) {
402             return false;
403         }
404 
405         if ((track.getTrackStates().get(0).getTanLambda() > 0 && (deltaY > paramMap.get("topClusterTrackMatchDeltaYHigh")[0]
406                 || deltaY < paramMap.get("topClusterTrackMatchDeltaYLow")[0]))
407                 || (track.getTrackStates().get(0).getTanLambda() < 0 && (deltaY > paramMap.get("bottomClusterTrackMatchDeltaYHigh")[0]
408                         || deltaY < paramMap.get("bottomClusterTrackMatchDeltaYLow")[0]))) {
409             return false;
410         }
411 
412         if (enablePlots) {
413             if (track.getTrackStates().get(0).getTanLambda() > 0) {
414 
415                 plots1D.get("Ecal cluster x - track x @ Ecal - top - matched").fill(deltaX);
416                 plots2D.get("Ecal cluster x v track x @ Ecal - top - matched").fill(clusterPosition.x(),
417                         trackPosAtEcal.x());
418                 plots1D.get("Ecal cluster y - track y @ Ecal - top - matched").fill(deltaY);
419                 plots2D.get("Ecal cluster y v track y @ Ecal - top - matched").fill(clusterPosition.y(),
420                         trackPosAtEcal.y());
421 
422             } else if (track.getTrackStates().get(0).getTanLambda() < 0) {
423 
424                 plots1D.get("Ecal cluster x - track x @ Ecal - bottom - matched").fill(deltaX);
425                 plots2D.get("Ecal cluster x v track x @ Ecal - bottom - matched").fill(clusterPosition.x(),
426                         trackPosAtEcal.x());
427                 plots1D.get("Ecal cluster y - track y @ Ecal - bottom - matched").fill(deltaY);
428                 plots2D.get("Ecal cluster y v track y @ Ecal - bottom - matched").fill(clusterPosition.y(),
429                         trackPosAtEcal.y());
430             }
431         }
432 
433         // If all cuts are pased, return true.
434         return true;
435     }
