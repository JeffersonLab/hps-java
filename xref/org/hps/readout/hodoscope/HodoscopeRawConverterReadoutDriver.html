14  /**
15   * <code>HodoscopeRawConverterReadoutDriver</code> is an
16   * implementation of {@link org.hps.readout.RawConverterReadoutDriver
17   * RawConverterReadoutDriver} for the hodoscope subdetector.
18   * 
19   * @see org.hps.readout.RawConverterReadoutDriver
20   */
21  public class HodoscopeRawConverterReadoutDriver extends RawConverterReadoutDriver {    
22      /**
23       * The converter object responsible for processing raw hits into
24       * proper {@link org.lcsim.event.CalorimeterHit CalorimeterHit}
25       * objects.
26       */
27      private HodoscopeReadoutMode3RawConverter converter = new HodoscopeReadoutMode3RawConverter();
28      
29      /**
30       * Instantiates the driver with the correct default parameters.
31       */
32      public HodoscopeRawConverterReadoutDriver() {
33          super("HodoscopeRawHits", "HodoscopeCorrectedHits");
34      }
35      
36      /**
37       * Sets whether or not the DAQ configuration is applied into the driver
38       * the EvIO data stream or whether to read the configuration from data files.
39       * 
40       * @param state - <code>true</code> indicates that the DAQ configuration is
41       * applied into the readout system, and <code>false</code> that it
42       * is not applied into the readout system.
43       */
44      public void setDaqConfigurationAppliedintoReadout(boolean state) {
45          // Track changes in the DAQ configuration.
46          if (state) {
47              ConfigurationManager2019.addActionListener(new ActionListener() {
48                  @Override
49                  public void actionPerformed(ActionEvent e) {
50                      // Get the DAQ configuration.
51                      DAQConfig2019 daq = ConfigurationManager2019.getInstance();
52  
53                      // Load the DAQ settings from the configuration manager.
54                      getConverter().setNumberSamplesAfter(daq.getHodoFADCConfig().getNSA());
55                      getConverter().setNumberSamplesBefore(daq.getHodoFADCConfig().getNSB());
56                      
57                      // Get the FADC configuration.
58                      getConverter().setFADCConfigHodo2019(daq.getHodoFADCConfig());
59                  }
60              });
61          }         
62      }    
63      
64      @Override
65      protected AbstractMode3RawConverter getConverter() {
66          return converter;
67      }
68      
69      @Override
70      protected String getSubdetectorReadoutName(Detector detector) {
71          Hodoscope_v1 hodoscopeGeometry = (Hodoscope_v1) detector.getSubdetector("Hodoscope");
72          return hodoscopeGeometry.getReadout().getName();
73      }
74      
75      @Override
76      protected void updateDetectorDependentParameters(Detector detector) { }    
77  }
