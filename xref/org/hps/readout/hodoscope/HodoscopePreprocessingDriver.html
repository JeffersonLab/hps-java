19  /**
20   * <code>HodoscopeEnergySplitDriver</code> handles the preprocessing
21   * of hodoscope hits from SLIC. It is responsible for converting the
22   * hits from {@link org.lcsim.event.SimTrackerHit SimTrackerHit}
23   * objects into {@link org.lcsim.event.SimCalorimeterHit
24   * SimCalorimeterHit} objects. It is also responsible for converting
25   * hits from scintillator-based channels to those that match the FADC
26   * channels in the hardware. This entails splitting the energy of
27   * those hits where the physical scintillator feeds into multiple
28   * FADC channels.
29   */
30  public class HodoscopePreprocessingDriver extends ReadoutDriver {
31      /** Object for accessing the geometric data of the hodoscope detector model. */
32      private HodoscopeDetectorElement hodoscopeDetectorElement;
33      /** The name of the SLiC truth hit collection. */
34      private String truthHitCollectionName = "HodoscopeHits";
35      /** Name of the output preprocessed hit collection. */
36      private String outputHitCollectionName = "HodoscopePreprocessedHits";
37      
38      @Override
39      public void detectorChanged(Detector detector) {
40          // Update the hodoscope detector object.
41          hodoscopeDetectorElement = (HodoscopeDetectorElement) detector.getSubdetector("Hodoscope").getDetectorElement();
42      }
43      
44      @Override
45      public void process(EventHeader event) {
46          // Get the hodoscope hits from the data manager.
47          Collection<SimTrackerHit> hodoscopeHits = ReadoutDataManager.getData(ReadoutDataManager.getCurrentTime(), ReadoutDataManager.getCurrentTime() + 2.0,
48                  truthHitCollectionName, SimTrackerHit.class);
49          
50          // Create a list to store the output hits in.
51          List<SimCalorimeterHit> outputHits = new ArrayList<SimCalorimeterHit>();
52          
53          // Iterate over the hodoscope hits and create new calorimeter
54          // hit objects from them. For hodoscope scintillators with
55          // multiple FADC channels, split the energy between them.
56          for(SimTrackerHit hit : hodoscopeHits) {
57              // Get the number of FADC channels that are associated
58              // with this scintillator.
59              int fadcChannels = hodoscopeDetectorElement.getScintillatorChannelCount(hit);
60              
61              // Get the SLIC-to-hardware mapping key.
62              
63              // If there is only one hit, then create a calorimeter
64              // hit object that contains the full energy of the truth
65              // hit. There is no need to split it.
66              if(fadcChannels == 1) {
67                  // Get the hardware channel for this scintillator.
68                  int cellID = hodoscopeDetectorElement.getHardwareChannels(hit).get(0).getChannelId().intValue();
69                  
70                  // Convert the hit to a calorimeter hit with the new
71                  // hardware channel ID and store it.
72                  outputHits.add(makeHit(hit, cellID));
73              }
74              
75              // If there are two channels, the energy must be split
76              // between them.
77              else {
78                  // Get the positions of the two fiber bundle holes.
79                  double[] holePos = hodoscopeDetectorElement.getScintillatorHolePositions(hit);
80                  
81                  // Get the hardware channel IDs for the FADC channels
82                  // associated with each fiber bundle hole.
83                  int[] fadcChannelIDs = new int[2];
84                  List<HodoscopeChannel> fadcChannelIDList = hodoscopeDetectorElement.getHardwareChannels(hit);
85                  fadcChannelIDs[getHolePositionArrayIndex(fadcChannelIDList.get(0))] = fadcChannelIDList.get(0).getChannelId();
86                  fadcChannelIDs[getHolePositionArrayIndex(fadcChannelIDList.get(1))] = fadcChannelIDList.get(1).getChannelId();
87                  
88                  // Get the x-position of the hit.
89                  double hitXPos = hit.getPosition()[0];
90                  
91                  // If the hit exists entirely before the first fiber
92                  // bundle hole, all its energy goes to that channel.
93                  // Likewise, if the hit exists after the second hole,
94                  // all of its energy goes to the second channel.
95                  if(hitXPos < holePos[0]) {
96                      outputHits.add(makeHit(hit, fadcChannelIDs[0]));
97                  } else if(hitXPos > holePos[1]) {
98                      outputHits.add(makeHit(hit, fadcChannelIDs[1]));
99                  }
100                 
101                 // Otherwise, the energy must be split linearly based
102                 // on its distance from each hole.
103                 else {
104                     // Calculate the distance between each fiber
105                     // bundle hole and the hit. Only the horizontal
106                     // distance matters.
107                     double la = Math.abs(holePos[0] - hitXPos);
108                     double lb = Math.abs(holePos[1] - hitXPos);
109                     
110                     // Calculate the percentage of the hit's energy
111                     // that should go to each channel. This should be
112                     // linearly proportional to the x-displacement of
113                     // the hit from that channel's fiber bundle hole.
114                     double ra = lb / (la + lb);
115                     double rb = la / (la + lb);
116                     
117                     // Create two new hits.
118                     outputHits.add(makeHit(hit, fadcChannelIDs[0], ra));
119                     outputHits.add(makeHit(hit, fadcChannelIDs[1], rb));
120                 }
121             }
122         }
123         
124         // Output the preprocessed hits to the data manager.
125         ReadoutDataManager.addData(outputHitCollectionName, outputHits, SimCalorimeterHit.class);
126     }

168     /**
169      * Translates a hole index into a form usable in an array.
170      * @param channel - A hodoscope channel.
171      * @return Returns a value of <code>0</code> for the hole closest
172      * to the calorimeter center and a value of <code>1</code> for
173      * hole closest to the positron-side of the calorimeter.
174      * @throws IllegalArgumentException Occurs if the input channel
175      * either has an unrecognized hole number, or only has one hole.
176      */
177     private static final int getHolePositionArrayIndex(HodoscopeChannel channel) throws IllegalArgumentException {
178         if(channel.getHole().intValue() == HodoscopeChannel.HOLE_LOW_X) { return 0; }
179         else if(channel.getHole().intValue() == HodoscopeChannel.HOLE_HIGH_X) { return 1; }
180         else { throw new IllegalArgumentException("Unexpected hole number \"" + channel.getHole().intValue() + "\"!"); }
181     }

183     /**
184      * Creates a {@link org.lcsim.event.SimCalorimeterHit
185      * SimCalorimeterHit} object from an input {@link
186      * org.lcsim.event.SimTrackerHit SimTrackerHit} object. The new
187      * hit will have the same truth information, but will have the
188      * cell ID specified instead.
189      * @param hit - The hit to convert.
190      * @param cellID - The new cell ID for the hit.
191      * @return Returns a new <code>SimCalorimeterHit</code> object
192      * that represents the same data as the input
193      * <code>RawTrackerHit</code> object.
194      */
195     private static final SimCalorimeterHit makeHit(SimTrackerHit hit, int cellID) {
196         return makeHit(hit, cellID, 1.0);
197     }

199     /**
200      * Creates a {@link org.lcsim.event.SimCalorimeterHit
201      * SimCalorimeterHit} object from an input {@link
202      * org.lcsim.event.SimTrackerHit SimTrackerHit} object. The new
203      * hit will have the same truth information, but will have the
204      * cell ID specified and also its energy will be scaled by the
205      * amount indicated by <code>energyScale</code>.
206      * @param hit - The hit to convert.
207      * @param cellID - The new cell ID for the hit.
208      * @param energyScale - The amount by which the energy should be
209      * scaled.
210      * @return Returns a new <code>SimCalorimeterHit</code> object
211      * that represents the same data as the input
212      * <code>RawTrackerHit</code> object, but with the energy scaled
213      * as specified.
214      */
215     private static final SimCalorimeterHit makeHit(SimTrackerHit hit, int cellID, double energyScale) {
216         // Create the necessary data objects to clone the hit. Note
217         // that SimTrackerHit objects only ever have a single truth
218         // particle associated with them.
219         int[] pdgs = new int[] { hit.getMCParticle().getPDGID() };
220         float[] times = new float[] {  (float) hit.getTime() };
221         float[] energies = new float[] { (float) (energyScale * hit.getdEdx()) };
222         Object[] particles = new Object[] { hit.getMCParticle() };
223         
224         // Create the calorimeter hit from tracker hit data.
225         return new BaseSimCalorimeterHit(cellID, (energyScale * hit.getdEdx()), hit.getTime(),
226                 particles, energies, times, pdgs, hit.getMetaData());
227     }
