1   package org.hps.readout.rawconverter;
2   
3   import org.hps.conditions.database.DatabaseConditionsManager;
4   import org.hps.conditions.ecal.EcalChannelConstants;
5   import org.hps.conditions.ecal.EcalConditions;
6   import org.lcsim.geometry.Detector;
7   
8   /**
9    * <code>EcalReadoutMode3RawConverter</code> handles the
10   * implementation of calorimeter-specific functionality for {@link
11   * org.hps.readout.rawconverter.AbstractMode3RawConverter
12   * AbstractMode3RawConverter}. Only the behavior needed for readout
13   * is implemented. Calibrations specific to reconstruction are not
14   * supported.
15   * 
16   * @see org.hps.readout.rawconverter.AbstractMode3RawConverter
17   */
18  public class EcalReadoutMode3RawConverter extends AbstractMode3RawConverter {        
19  
20      /**
21       * Stores the calibrations and conditions for the calorimeter
22       * subdetector.
23       */
24      private EcalConditions ecalConditions = null;   
25          
26      @Override
27      public void updateDetector(Detector detector) {
28          ecalConditions = DatabaseConditionsManager.getInstance().getEcalConditions();
29      }    
30      
31      @Override
32      protected double getGain(long cellID) {        
33          if(configEcal != null) return configEcal.getGain(cellID);
34          else return findChannel(cellID).getGain().getGain();
35      }
36      
37      @Override
38      protected double getPedestal(long cellID) {        
39          if(configEcal != null) return configEcal.getPedestal(cellID);
40          else return findChannel(cellID).getCalibration().getPedestal();
41      }
42      
43      @Override
44      protected double getTimeShift(long cellID) {
45          return findChannel(cellID).getTimeShift().getTimeShift();
46      }
47      
48      /**
49       * Get the calorimeter conditions for the specified channel.
50       * @param cellID - The channel ID.
51       * @return Returns an object containing the conditions for the
52       * specified calorimeter channel.
53       */
54      private EcalChannelConstants findChannel(long cellID) {
55          return ecalConditions.getChannelConstants(ecalConditions.getChannelCollection().findGeometric(cellID));
56      }
57  }
