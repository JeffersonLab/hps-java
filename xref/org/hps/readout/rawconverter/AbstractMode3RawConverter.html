1   package org.hps.readout.rawconverter;
2   
3   import org.hps.recon.ecal.CalorimeterHitUtilities;
4   import org.lcsim.event.CalorimeterHit;
5   import org.lcsim.event.RawCalorimeterHit;
6   
7   /**
8    * <code>AbstractMode3RawConverter</code> provides functionality for
9    * the {@link org.hps.readout.rawconverter.AbstractBaseRawConverter
10   * AbstractBaseRawConverter} for the handling of Mode-3 hits.
11   * <br/><br/>
12   * <code>AbstractMode3RawConverter</code> does not implement any
13   * behavior that is specific to a subdetector. That is left for its
14   * implementing drivers.
15   * 
16   * @see org.hps.readout.rawconverter.AbstractBaseRawConverter
17   */
18  public abstract class AbstractMode3RawConverter extends AbstractBaseRawConverter {
19      /**
20       * Stores the number of samples in the readout window. This is
21       * needed to properly account for pulse-clipping in Mode-3 and
22       * Mode-7 data, where this value is not retained. This value is
23       * not used for Mode-1 input, as the number of samples is known
24       * for this data.
25       */
26      protected int windowSamples = -1;
27      
28      /**
29       * Converts a Mode-3 digitized hit to a {@link
30       * org.lcsim.event.CalorimeterHit CalorimeterHit} object with
31       * calibrated energy and time values.
32       * @param hit - The hit to convert.
33       * @param timeOffset - The time offset of the hit.
34       * @return Returns the converted hit.
35       */
36      public CalorimeterHit convertHit(RawCalorimeterHit hit, double timeOffset) {
37          // Get the hit time and channel ID.
38          long channelID = hit.getCellID();
39          double time = hit.getTimeStamp() / 16.0;
40          
41          // Get the pedestal for the entire hit integration window.
42          double pedestal = getTotalPedestal(channelID, windowSamples, (int) time / NS_PER_SAMPLE);
43          
44          // Get the pedestal-corrected ADC sum and convert it to
45          // units of energy.
46          double adcSum = hit.getAmplitude() - pedestal;
47          double rawEnergy = adcToEnergy(adcSum, channelID);
48          
49          // Calculate the time for the hit.
50          time -= getTimeShift(channelID);
51          
52          // Create a new calorimeter hit and return it.
53          return CalorimeterHitUtilities.create(rawEnergy, time + timeOffset, channelID);
54      }
55      
56      /**
57       * Gets the time shift for the specified subdetector channel.
58       * @param cellID - The detector element ID.
59       * @return Returns the value of the time shift as a
60       * <code>double</code>.
61       */
62      protected abstract double getTimeShift(long channelID);
63      
64      /**
65       * Sets the number of samples in the readout window. Note that
66       * this value is ignored for Mode-1 data, since Mode-1 data
67       * retains the number of ADC samples already.
68       * @param windowSamples - The number of 4 ns clock-cycle samples
69       * present in the readout window.
70       */
71      public void setWindowSamples(int windowSamples) {
72          this.windowSamples = windowSamples;
73      }
74  }
