1   package org.hps.readout.rawconverter;
2   
3   import java.util.HashMap;
4   import java.util.Map;
5   
6   import org.hps.conditions.database.DatabaseConditionsManager;
7   import org.hps.conditions.hodoscope.HodoscopeCalibration;
8   import org.hps.conditions.hodoscope.HodoscopeGain;
9   import org.hps.conditions.hodoscope.HodoscopeTimeShift;
10  import org.hps.conditions.hodoscope.HodoscopeCalibration.HodoscopeCalibrationCollection;
11  import org.hps.conditions.hodoscope.HodoscopeGain.HodoscopeGainCollection;
12  import org.hps.conditions.hodoscope.HodoscopeTimeShift.HodoscopeTimeShiftCollection;
13  import org.lcsim.geometry.Detector;
14  
15  public class HodoscopeMode1RawConverter extends AbstractMode1RawConverter {
16      /**
17       * Maps hodoscope channels to the gain for that channel.
18       */
19      private Map<Long, HodoscopeGain> channelToGainsMap = new HashMap<Long, HodoscopeGain>();
20      
21      /**
22       * Maps hodoscope channels to the time shifts for that channel.
23       */
24      private Map<Long, HodoscopeTimeShift> channelToTimeShiftsMap = new HashMap<Long, HodoscopeTimeShift>();
25      
26      /**
27       * Maps hodoscope channels to the noise sigma and pedestals for that channel.
28       */
29      private Map<Long, HodoscopeCalibration> channelToCalibrationsMap = new HashMap<Long, HodoscopeCalibration>();
30      
31      @Override
32      public void updateDetector(Detector detector) {
33          // Load the conditions database and get the hodoscope channel
34          // collection data.
35          final DatabaseConditionsManager conditions = DatabaseConditionsManager.getInstance();
36          final HodoscopeGainCollection gains = conditions.getCachedConditions(HodoscopeGainCollection.class, "hodo_gains").getCachedData();
37          final HodoscopeTimeShiftCollection timeShifts = conditions.getCachedConditions(HodoscopeTimeShiftCollection.class, "hodo_time_shifts").getCachedData();
38          final HodoscopeCalibrationCollection calibrations = conditions.getCachedConditions(HodoscopeCalibrationCollection.class, "hodo_calibrations").getCachedData();
39          
40          // Map the gains to channel IDs.
41          for(HodoscopeGain gain : gains) {
42              channelToGainsMap.put(Long.valueOf(gain.getChannelId().intValue()), gain);
43          }
44          
45          // Map the pedestals and noise to channel IDs.
46          for(HodoscopeCalibration calibration : calibrations) {
47              channelToCalibrationsMap.put(Long.valueOf(calibration.getChannelId().intValue()), calibration);
48          }
49          
50          // Map time shifts to channel IDs.
51          for(HodoscopeTimeShift timeShift :  timeShifts) {
52              channelToTimeShiftsMap.put(Long.valueOf(timeShift.getChannelId().intValue()), timeShift);
53          }
54      }
55      
56      @Override
57      protected double getGain(long channelID) {
58          if(channelToGainsMap.containsKey(Long.valueOf(channelID))) {
59              return channelToGainsMap.get(Long.valueOf(channelID)).getGain();
60          } else {
61              throw new IllegalArgumentException("No gain conditions exist for hodoscope channel ID \"" + channelID + "\".");
62          }
63      }
64      
65      @Override
66      protected double getPedestal(long channelID) {
67          if(channelToCalibrationsMap.containsKey(Long.valueOf(channelID))) {
68              return channelToCalibrationsMap.get(Long.valueOf(channelID)).getPedestal();
69          } else {
70              throw new IllegalArgumentException("No pedestal conditions exist for hodoscope channel ID \"" + channelID + "\".");
71          }
72      }
73  }
