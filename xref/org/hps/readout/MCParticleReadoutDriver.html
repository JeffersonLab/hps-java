1   package org.hps.readout;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.lcsim.event.EventHeader;
7   import org.lcsim.event.MCParticle;
8   import org.lcsim.lcio.SIOMCParticle;
9   
10  /**
11   * <code>MCParticleReadoutDriver</code> handles SLIC objects in input
12   * Monte Carlo files of type {@link org.lcsim.event.MCParticle
13   * MCParticle}.
14   * 
15   * @see org.hps.readout.SLICDataReadoutDriver
16   */
17  public class MCParticleReadoutDriver extends SLICDataReadoutDriver<MCParticle> {
18  
19      /**
20       * Instantiate an instance of {@link
21       * org.hps.readout.SLICDataReadoutDriver SLICDataReadoutDriver}
22       * for objects of type {@link org.lcsim.event.MCParticle
23       * MCParticle}. These do not require any special LCIO flags.
24       */
25      public MCParticleReadoutDriver() {
26          super(MCParticle.class);
27          setCollectionName("MCParticle");
28      }
29      
30      @Override
31      public void process(EventHeader event) {
32          // Get the MCParticles.
33          List<MCParticle> particles = null;
34          if(event.hasCollection(MCParticle.class, "MCParticle")) {
35              particles = event.get(MCParticle.class, "MCParticle");
36          } else {
37              particles = new ArrayList<MCParticle>(0);
38          }
39          
40          // Modify the particle times.
41          if(!particles.isEmpty()) {
42              double currentTime = ReadoutDataManager.getCurrentTime();
43              for(MCParticle particle : particles) {
44                  if(SIOMCParticle.class.isAssignableFrom(particle.getClass())) {
45                      SIOMCParticle baseParticle = SIOMCParticle.class.cast(particle);
46                      baseParticle.setTime(particle.getProductionTime() + currentTime);
47                  } else {
48                      throw new RuntimeException("Error: Saw unexpected MC particle class \"" + particle.getClass().getSimpleName() + "\".");
49                  }
50              }
51          }
52          
53          // Run the superclass method to store the particle collection
54          // in the data manager.
55          super.process(event);
56      }
57  }
