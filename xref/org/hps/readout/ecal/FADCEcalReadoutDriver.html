308     /**
309      * Digitize values in the analog pipelines and append them to the digital pipelines. Integrate trigger-path hits and
310      * add them to the trigger path queues. Read out trigger-path hits to the list sent to the clusterer.
311      *
312      * @param hits List to be filled by this method.
313      */
314     @Override
315     protected void readHits(List<RawCalorimeterHit> hits) {
316 
317         for (Long cellID : analogPipelines.keySet()) {
318             RingBuffer signalBuffer = analogPipelines.get(cellID);
319 
320             FADCPipeline pipeline = digitalPipelines.get(cellID);
321             pipeline.step();
322 
323             // Get the channel data.
324             EcalChannelConstants channelData = findChannel(cellID);
325 
326             double currentValue = signalBuffer.currentValue() * ((Math.pow(2, nBit) - 1) / maxVolt); // 12-bit ADC with
327                                                                                                      // maxVolt V range
328             int pedestal = (int) Math.round(channelData.getCalibration().getPedestal());
329             
330             // ADC can't return a value larger than 4095; 4096 (overflow) is returned for any input >2V
331             int digitizedValue = Math.min((int) Math.round(pedestal + currentValue), (int) Math.pow(2, nBit));
332             
333             pipeline.writeValue(digitizedValue);
334             int pedestalSubtractedValue = digitizedValue - pedestal;
335             // System.out.println(signalBuffer.currentValue() + "   " + currentValue + "   " + pipeline.currentValue());
336 
337             Integer sum = triggerPathHitSums.get(cellID);
338             if (sum == null && pedestalSubtractedValue > triggerThreshold) {
339                 triggerPathHitTimes.put(cellID, readoutCounter);
340                 if (constantTriggerWindow) {
341                     int sumBefore = 0;
342                     for (int i = 0; i < numSamplesBefore; i++) {
343                         if (debug) {
344                             System.out.format("trigger %d, %d: %d\n", cellID, i,
345                                     pipeline.getValue(numSamplesBefore - i - 1));
346                         }
347                         sumBefore += pipeline.getValue(numSamplesBefore - i - 1);
348                     }
349                     triggerPathHitSums.put(cellID, sumBefore);
350                 } else {
351                     triggerPathHitSums.put(cellID, pedestalSubtractedValue);
352                 }
353             }
354             if (sum != null) {
355                 if (constantTriggerWindow) {
356                     if (triggerPathHitTimes.get(cellID) + numSamplesAfter >= readoutCounter) {
357                         if (debug) {
358                             System.out.format("trigger %d, %d: %d\n", cellID,
359                                     readoutCounter - triggerPathHitTimes.get(cellID) + numSamplesBefore - 1,
360                                     pipeline.getValue(0));
361                         }
362                         triggerPathHitSums.put(cellID, sum + pipeline.getValue(0));
363                     } else if (triggerPathHitTimes.get(cellID) + delay0 <= readoutCounter) {
364                         // System.out.printf("sum = %f\n", sum);
365                         triggerPathDelayQueue.add(new BaseRawCalorimeterHit(cellID,
366                                 (int) Math.round(sum / scaleFactor), 64 * triggerPathHitTimes.get(cellID)));
367                         triggerPathHitSums.remove(cellID);
368                     }
369                 } else {
370                     if (pedestalSubtractedValue < triggerThreshold
371                             || triggerPathHitTimes.get(cellID) + delay0 == readoutCounter) {
372                         // System.out.printf("sum = %f\n",sum);
373                         triggerPathDelayQueue.add(new BaseRawCalorimeterHit(cellID, (int) Math
374                                 .round((sum + pedestalSubtractedValue) / scaleFactor), 64 * triggerPathHitTimes
375                                 .get(cellID)));
376                         triggerPathHitSums.remove(cellID);
377                     } else {
378                         triggerPathHitSums.put(cellID, sum + pedestalSubtractedValue);
379                     }
380                 }
381             }
382             signalBuffer.step();
383         }
384         while (triggerPathDelayQueue.peek() != null
385                 && triggerPathDelayQueue.peek().getTimeStamp() / 64 <= readoutCounter - delay0) {
386             if (triggerPathDelayQueue.peek().getTimeStamp() / 64 < readoutCounter - delay0) {
387                 System.out.println(this.getName() + ": Stale hit in output queue");
388                 triggerPathDelayQueue.poll();
389             } else {
390                 triggerPathCoincidenceQueue.add(triggerPathDelayQueue.poll());
391             }
392         }
393         while (!triggerPathCoincidenceQueue.isEmpty()
394                 && triggerPathCoincidenceQueue.peek().getTimeStamp() / 64 <= readoutCounter - delay0
395                         - coincidenceWindow) {
396             triggerPathCoincidenceQueue.remove();
397         }
398         if (debug) {
399             for (RawCalorimeterHit hit : triggerPathCoincidenceQueue) {
400                 System.out.format("new hit: energy %d\n", hit.getAmplitude());
401             }
402         }
403 
404         hits.addAll(triggerPathCoincidenceQueue);
405     }

415     @Override
416     protected void processTrigger(EventHeader event) {
417         switch (mode) {
418             case ECAL_RAW_MODE:
419                 if (debug) {
420                     System.out.println("Reading out ECal in raw mode");
421                 }
422                 event.put(ecalReadoutCollectionName, readWindow(), RawTrackerHit.class, 0, ecalReadoutName);
423                 break;
424             case ECAL_PULSE_MODE:
425                 if (debug) {
426                     System.out.println("Reading out ECal in pulse mode");
427                 }
428                 event.put(ecalReadoutCollectionName, readPulses(), RawTrackerHit.class, 0, ecalReadoutName);
429                 break;
430             case ECAL_PULSE_INTEGRAL_MODE:
431                 if (debug) {
432                     System.out.println("Reading out ECal in integral mode");
433                 }
434                 event.put(ecalReadoutCollectionName, readIntegrals(), RawCalorimeterHit.class, flags, ecalReadoutName);
435                 break;
436         }
437     }

439     @Override
440     public double readoutDeltaT() {
441         double triggerTime = ClockSingleton.getTime() + triggerDelay;
442         int cycle = (int) Math.floor((triggerTime - readoutOffset + ClockSingleton.getDt()) / readoutPeriod);
443         double readoutTime = (cycle - readoutLatency) * readoutPeriod + readoutOffset - ClockSingleton.getDt();
444         return readoutTime;
445     }

447     protected short[] getWindow(long cellID) {
448         FADCPipeline pipeline = digitalPipelines.get(cellID);
449         short[] adcValues = new short[readoutWindow];
450         for (int i = 0; i < readoutWindow; i++) {
451             adcValues[i] = (short) pipeline.getValue(readoutLatency - i - 1);
452             // if (adcValues[i] != 0) {
453             // System.out.println("getWindow: " + adcValues[i] + " at i = " + i);
454             // }
455         }
456         return adcValues;
457     }

459     protected List<RawTrackerHit> readWindow() {
460         // System.out.println("Reading FADC data");
461         List<RawTrackerHit> hits = new ArrayList<RawTrackerHit>();
462         for (Long cellID : digitalPipelines.keySet()) {
463             short[] adcValues = getWindow(cellID);
464             EcalChannelConstants channelData = findChannel(cellID);
465             boolean isAboveThreshold = false;
466             for (int i = 0; i < adcValues.length; i++) {
467                 if (adcValues[i] > channelData.getCalibration().getPedestal() + readoutThreshold) {
468                     isAboveThreshold = true;
469                     break;
470                 }
471             }
472             if (isAboveThreshold) {
473                 hits.add(new BaseRawTrackerHit(cellID, 0, adcValues));
474             }
475         }
476         return hits;
477     }

479     protected List<RawTrackerHit> readPulses() {
480         // System.out.println("Reading FADC data");
481         List<RawTrackerHit> hits = new ArrayList<RawTrackerHit>();
482         for (Long cellID : digitalPipelines.keySet()) {
483             short[] window = getWindow(cellID);
484             short[] adcValues = null;
485             int pointerOffset = 0;
486             int numSamplesToRead = 0;
487             int thresholdCrossing = 0;
488 
489             // Get the channel data.
490             EcalChannelConstants channelData = findChannel(cellID);
491 
492             for (int i = 0; i < readoutWindow; i++) {
493                 if (numSamplesToRead != 0) {
494                     if (adcValues == null) {
495                         throw new RuntimeException("expected a pulse buffer, none found (this should never happen)");
496                     }
497                     adcValues[adcValues.length - numSamplesToRead] = window[i - pointerOffset];
498                     numSamplesToRead--;
499                     if (numSamplesToRead == 0) {
500                         hits.add(new BaseRawTrackerHit(cellID, thresholdCrossing, adcValues));
501                     }
502                 } else if ((i == 0 || window[i - 1] <= channelData.getCalibration().getPedestal() + readoutThreshold)
503                         && window[i] > channelData.getCalibration().getPedestal() + readoutThreshold) {
504                     thresholdCrossing = i;
505                     pointerOffset = Math.min(numSamplesBefore, i);
506                     numSamplesToRead = pointerOffset + Math.min(numSamplesAfter, readoutWindow - i - pointerOffset - 1);
507                     adcValues = new short[numSamplesToRead];
508                 }
509             }
510         }
511         return hits;
512     }

514     protected List<RawCalorimeterHit> readIntegrals() {
515         // System.out.println("Reading FADC data");
516         List<RawCalorimeterHit> hits = new ArrayList<RawCalorimeterHit>();
517         for (Long cellID : digitalPipelines.keySet()) {
518             short[] window = getWindow(cellID);
519             int adcSum = 0;
520             int pointerOffset = 0;
521             int numSamplesToRead = 0;
522             int thresholdCrossing = 0;
523 
524             // Get the channel data.
525             EcalChannelConstants channelData = findChannel(cellID);
526 
527             if (window != null) {
528                 for (int i = 0; i < readoutWindow; i++) {
529                     if (numSamplesToRead != 0) {
530                         if (debug) {
531                             System.out.format("readout %d, %d: %d\n", cellID, numSamplesBefore + numSamplesAfter
532                                     - numSamplesToRead, window[i - pointerOffset]);
533                         }
534                         adcSum += window[i - pointerOffset];
535                         numSamplesToRead--;
536                         if (numSamplesToRead == 0) {
537                             hits.add(new BaseRawCalorimeterHit(cellID, adcSum, 64 * thresholdCrossing));
538                         }
539                     } else if ((i == 0 || window[i - 1] <= channelData.getCalibration().getPedestal()
540                             + readoutThreshold)
541                             && window[i] > channelData.getCalibration().getPedestal() + readoutThreshold) {
542                         thresholdCrossing = i;
543                         pointerOffset = Math.min(numSamplesBefore, i);
544                         numSamplesToRead = pointerOffset
545                                 + Math.min(numSamplesAfter, readoutWindow - i - pointerOffset - 1);
546                         adcSum = 0;
547                     }
548                 }
549             }
550         }
551         return hits;
552     }

554     /**
555      * Fill the analog pipelines with the preamp pulses generated by hits in the ECal.
556      *
557      * @param hits ECal hits from SLIC/Geant4.
558      */
559     @Override
560     protected void putHits(List<CalorimeterHit> hits) {
561         // fill the readout buffers
562         for (CalorimeterHit hit : hits) {
563             RingBuffer eDepBuffer = analogPipelines.get(hit.getCellID());
564             double energyAmplitude = hit.getRawEnergy();
565             // Get the channel data.
566             EcalChannelConstants channelData = findChannel(hit.getCellID());
567             if (addNoise) {
568                 // add preamp noise and photoelectron Poisson noise in quadrature
569                 double noise;
570                 if (use2014Gain) {
571                     noise = Math.sqrt(Math.pow(channelData.getCalibration().getNoise()
572                             * channelData.getGain().getGain() * EcalUtils.gainFactor * EcalUtils.ecalReadoutPeriod, 2)
573                             + hit.getRawEnergy() / (EcalUtils.lightYield * EcalUtils.quantumEff * EcalUtils.surfRatio));
574                 } else {
575                     noise = Math.sqrt(Math.pow(channelData.getCalibration().getNoise()
576                             * channelData.getGain().getGain() * EcalUtils.MeV, 2)
577                             + hit.getRawEnergy() * EcalUtils.MeV / pePerMeV);
578                 }
579                 energyAmplitude += RandomGaussian.getGaussian(0, noise);
580             }
581             if ((1) * readoutPeriod + readoutTime() - (ClockSingleton.getTime() + hit.getTime()) >= readoutPeriod) {
582                 throw new RuntimeException("trying to add a hit to the analog pipeline, but time seems incorrect");
583             }
584             for (int i = 0; i < bufferLength; i++) {
585                 // eDepBuffer.addToCell(i, energyAmplitude * pulseAmplitude((i + 1) * readoutPeriod + readoutTime() -
586                 // (ClockSingleton.getTime() + hit.getTime()), hit.getCellID()));
587                 eDepBuffer.addToCell(
588                         i,
589                         energyAmplitude
590                                 * pulseAmplitude((i + 1) * readoutPeriod + readoutTime()
591                                         - (ClockSingleton.getTime() + hit.getTime())
592                                         - findChannel(hit.getCellID()).getTimeShift().getTimeShift(), hit.getCellID()));
593 
594             }
595         }
596     }

618     private boolean resetFADCBuffers() {
619         if (ecal == null) {
620             return false;
621         }
622         analogPipelines = new HashMap<Long, RingBuffer>();
623         digitalPipelines = new HashMap<Long, FADCPipeline>();
624         Set<Long> cells = ((HPSEcal3) ecal).getNeighborMap().keySet();
625         for (Long cellID : cells) {
626             EcalChannelConstants channelData = findChannel(cellID);
627             analogPipelines.put(cellID, new RingBuffer(bufferLength));
628             digitalPipelines.put(cellID,
629                     new FADCPipeline(pipelineLength, (int) Math.round(channelData.getCalibration().getPedestal())));
630         }
631         return true;
632     }
633 
634     /**
635      * Returns pulse amplitude at the given time (relative to hit time). Gain is applied.
636      *
637      * @param time Units of ns. Relative to hit time (negative=before the start of the pulse).
638      * @param cellID Crystal ID as returned by hit.getCellID().
639      * @return Amplitude, units of volts/GeV.
640      */
641     private double pulseAmplitude(double time, long cellID) {
642         // Get the channel data.
643         EcalChannelConstants channelData = findChannel(cellID);
644 
645         if (use2014Gain) {
646             // if fixedGain is set, multiply the default gain by this factor
647             double corrGain;
648             if (fixedGain > 0) {
649                 corrGain = fixedGain;
650             } else {
651                 corrGain = 1.0 / channelData.getGain().getGain();
652             }
653 
654             return corrGain * readoutGain * pulseAmplitude(time, pulseShape, tp);
655         } else {
656             // normalization constant from cal gain (MeV/integral bit) to amplitude gain (amplitude bit/GeV)
657             double gain;
658             if (fixedGain > 0) {
659                 gain = readoutPeriod / (fixedGain * EcalUtils.MeV * ((Math.pow(2, nBit) - 1) / maxVolt));
660             } else {
661                 gain = readoutPeriod
662                         / (channelData.getGain().getGain() * EcalUtils.MeV * ((Math.pow(2, nBit) - 1) / maxVolt));
663             }
664 
665             return gain * pulseAmplitude(time, pulseShape, tp);
666         }
667     }
668 
669     /**
670      * Returns pulse amplitude at the given time (relative to hit time).
671      *
672      * @param time Units of ns. Relative to hit time (negative=before the start of the pulse).
673      * @return Amplitude, units of inverse ns. Normalized so the pulse integral is 1.
674      */
675     public static double pulseAmplitude(double time, PulseShape shape, double shapingTime) {
676         if (time <= 0.0) {
677             return 0.0;
678         }
679         switch (shape) {
680             case CRRC:
681                 // peak at tp
682                 // peak value 1/(tp*e)
683                 return ((time / (shapingTime * shapingTime)) * Math.exp(-time / shapingTime));
684             case DoubleGaussian:
685                 // According to measurements the output signal can be fitted by two gaussians, one for the rise of the
686                 // signal, one for the fall
687                 // peak at 3*riseTime
688                 // peak value 1/norm
689 
690                 double norm = ((riseTime + fallTime) / 2) * Math.sqrt(2 * Math.PI); // to ensure the total integral is
691                                                                                     // equal to 1: = 33.8
692                 return funcGaus(time - 3 * riseTime, (time < 3 * riseTime) ? riseTime : fallTime) / norm;
693             case ThreePole:
694                 // peak at 2*tp
695                 // peak value 2/(tp*e^2)
696                 return ((time * time / (2 * shapingTime * shapingTime * shapingTime)) * Math.exp(-time / shapingTime));
697             default:
698                 return 0.0;
699         }
700     }
701 
702     // Gaussian function needed for the calculation of the pulse shape amplitude
703     public static double funcGaus(double t, double sig) {
704         return Math.exp(-t * t / (2 * sig * sig));
705     }

707     public class FADCPipeline {
708 
709         private final int[] array;
710         private final int size;
711         private int ptr;
712 
713         public FADCPipeline(int size) {
714             this.size = size;
715             array = new int[size]; // initialized to 0
716             ptr = 0;
717         }
718 
719         // construct pipeline with a nonzero initial value
720         public FADCPipeline(int size, int init) {
721             this.size = size;
722             array = new int[size];
723             for (int i = 0; i < size; i++) {
724                 array[i] = init;
725             }
726             ptr = 0;
727         }
728 
729         /**
730          * Write value to current cell
731          */
732         public void writeValue(int val) {
733             array[ptr] = val;
734         }
735 
736         /**
737          * Write value to current cell
738          */
739         public void step() {
740             ptr++;
741             if (ptr == size) {
742                 ptr = 0;
743             }
744         }
745 
746         // return content of specified cell (pos=0 for current cell)
747         public int getValue(int pos) {
748             if (pos >= size || pos < 0) {
749                 throw new ArrayIndexOutOfBoundsException();
750             }
751             return array[((ptr - pos) % size + size) % size];
752         }
753     }
