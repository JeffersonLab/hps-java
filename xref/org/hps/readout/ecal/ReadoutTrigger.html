124     /**
125      * Produces both uncut and cut distributions from clusters.
126      */
127     @Override
128     public void process(EventHeader event) {
129         // Check for a collection of clusters.
130         if (event.hasCollection(Cluster.class, clusterCollectionName)) {
131             // Get the list of clusters.
132             List<Cluster> clusters = event.get(Cluster.class, clusterCollectionName);
133 
134             // Track which clusters have already been plotted.
135             Set<Cluster> plottedClustersUncut = new HashSet<Cluster>(clusters.size());
136             Set<Cluster> plottedClustersCut = new HashSet<Cluster>(clusters.size());
137 
138             // Populate a list of cluster pairs.
139             List<Cluster[]> pairs = getClusterPairs(clusters);
140 
141             // Process all cluster pairs.
142             pairLoop: for (Cluster[] pair : pairs) {
143                 // Get the x and y indices for each cluster in the pair.
144                 int[] ix = {pair[0].getCalorimeterHits().get(0).getIdentifierFieldValue("ix"),
145                         pair[1].getCalorimeterHits().get(0).getIdentifierFieldValue("ix")};
146                 int[] iy = {pair[0].getCalorimeterHits().get(0).getIdentifierFieldValue("iy"),
147                         pair[1].getCalorimeterHits().get(0).getIdentifierFieldValue("iy")};
148 
149                 // Iterate over the clusters in the pair and plot the
150                 // cluster singles distributions.
151                 for (int clusterIndex = 0; clusterIndex < 2; clusterIndex++) {
152                     // Only plot cluster singles distributions for
153                     // clusters if they have not already been plotted.
154                     // Note that this is needed because the same cluster
155                     // can appear across multiple pairs.
156                     if (!plottedClustersUncut.contains(pair[clusterIndex])) {
157                         clusterSeedEnergy[NO_CUTS].fill(TriggerModule.getValueClusterSeedEnergy(pair[clusterIndex]));
158                         clusterTotalEnergy[NO_CUTS].fill(TriggerModule.getValueClusterTotalEnergy(pair[clusterIndex]));
159                         clusterHitCount[NO_CUTS].fill(TriggerModule.getValueClusterHitCount(pair[clusterIndex]));
160                         clusterDistribution[NO_CUTS].fill(ix[clusterIndex], iy[clusterIndex]);
161                         clusterTime[NO_CUTS].fill(pair[clusterIndex].getCalorimeterHits().get(0).getTime());
162                         plottedClustersUncut.add(pair[clusterIndex]);
163                     }
164                 }
165 
166                 // Plot the cluster pair distributions.
167                 pairEnergySum[NO_CUTS].fill(TriggerModule.getValueEnergySum(pair));
168                 pairEnergyDifference[NO_CUTS].fill(TriggerModule.getValueEnergyDifference(pair));
169                 pairEnergySlope[NO_CUTS].fill(TriggerModule.getValueEnergySlope(pair, energySlopeParamF));
170                 pairCoplanarity[NO_CUTS].fill(TriggerModule.getValueCoplanarity(pair));
171                 pairTime[NO_CUTS].fill(pair[1].getCalorimeterHits().get(0).getTime());
172                 pairCoincidence[NO_CUTS].fill(TriggerModule.getValueTimeCoincidence(pair));
173                 pairEnergySum2D[NO_CUTS].fill(pair[0].getEnergy(), pair[1].getEnergy());
174                 if (pair[0].getEnergy() < pair[1].getEnergy()) {
175                     pairEnergySlope2D[NO_CUTS].fill(pair[0].getEnergy(), TriggerModule.getClusterDistance(pair[0]));
176                 } else {
177                     pairEnergySlope2D[NO_CUTS].fill(pair[1].getEnergy(), TriggerModule.getClusterDistance(pair[1]));
178                 }
179 
180                 // Perform the cluster singles cuts.
181                 if (!(trigger.clusterHitCountCut(pair[0]) && trigger.clusterHitCountCut(pair[1]))) {
182                     continue pairLoop;
183                 }
184                 if (!(trigger.clusterTotalEnergyCut(pair[0]) && trigger.clusterTotalEnergyCut(pair[1]))) {
185                     continue pairLoop;
186                 }
187                 if (!(trigger.clusterSeedEnergyCut(pair[0]) && trigger.clusterSeedEnergyCut(pair[1]))) {
188                     continue pairLoop;
189                 }
190 
191                 // Perform the cluster pair cuts.
192                 if (!trigger.pairCoplanarityCut(pair)) {
193                     continue pairLoop;
194                 }
195                 if (!trigger.pairEnergyDifferenceCut(pair)) {
196                     continue pairLoop;
197                 }
198                 if (!trigger.pairEnergySlopeCut(pair)) {
199                     continue pairLoop;
200                 }
201                 if (!trigger.pairEnergySumCut(pair)) {
202                     continue pairLoop;
203                 }
204 
205                 // Iterate over the clusters in the pair and plot the
206                 // cluster singles distributions.
207                 for (int clusterIndex = 0; clusterIndex < 2; clusterIndex++) {
208                     // Only plot cluster singles distributions for
209                     // clusters if they have not already been plotted.
210                     // Note that this is needed because the same cluster
211                     // can appear across multiple pairs.
212                     if (!plottedClustersCut.contains(pair[clusterIndex])) {
213                         clusterSeedEnergy[ALL_CUTS].fill(TriggerModule.getValueClusterSeedEnergy(pair[clusterIndex]));
214                         clusterTotalEnergy[ALL_CUTS].fill(TriggerModule.getValueClusterTotalEnergy(pair[clusterIndex]));
215                         clusterHitCount[ALL_CUTS].fill(TriggerModule.getValueClusterHitCount(pair[clusterIndex]));
216                         clusterDistribution[ALL_CUTS].fill(ix[clusterIndex], iy[clusterIndex]);
217                         clusterTime[ALL_CUTS].fill(pair[clusterIndex].getCalorimeterHits().get(0).getTime());
218                         plottedClustersCut.add(pair[clusterIndex]);
219                     }
220                 }
221 
222                 // Plot the cluster pair distributions.
223                 pairEnergySum[ALL_CUTS].fill(TriggerModule.getValueEnergySum(pair));
224                 pairEnergyDifference[ALL_CUTS].fill(TriggerModule.getValueEnergyDifference(pair));
225                 pairEnergySlope[ALL_CUTS].fill(TriggerModule.getValueEnergySlope(pair, energySlopeParamF));
226                 pairCoplanarity[ALL_CUTS].fill(TriggerModule.getValueCoplanarity(pair));
227                 pairTime[ALL_CUTS].fill(pair[1].getCalorimeterHits().get(0).getTime());
228                 pairCoincidence[ALL_CUTS].fill(TriggerModule.getValueTimeCoincidence(pair));
229                 pairEnergySum2D[ALL_CUTS].fill(pair[0].getEnergy(), pair[1].getEnergy());
230                 if (pair[0].getEnergy() < pair[1].getEnergy()) {
231                     pairEnergySlope2D[ALL_CUTS].fill(pair[0].getEnergy(), TriggerModule.getClusterDistance(pair[0]));
232                 } else {
233                     pairEnergySlope2D[ALL_CUTS].fill(pair[1].getEnergy(), TriggerModule.getClusterDistance(pair[1]));
234                 }
235 
236             }
237         }
238     }

289     /**
290      * Creates all top/bottom pairs from the event data.
291      * 
292      * @param clusters - A list of clusters from which to form pairs.
293      * @return Returns a <code>List</code> collection that contains <code>Cluster</code> arrays of size two.
294      */
295     private List<Cluster[]> getClusterPairs(List<Cluster> clusters) {
296         // Separate the clusters into top and bottom clusters.
297         List<Cluster> topList = new ArrayList<Cluster>();
298         List<Cluster> botList = new ArrayList<Cluster>();
299         for (Cluster cluster : clusters) {
300             if (cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("iy") > 0) {
301                 topList.add(cluster);
302             } else {
303                 botList.add(cluster);
304             }
305         }
306 
307         // Create all possible top/bottom cluster pairs.
308         List<Cluster[]> pairList = new ArrayList<Cluster[]>();
309         for (Cluster topCluster : topList) {
310             for (Cluster botCluster : botList) {
311                 pairList.add(new Cluster[] {topCluster, botCluster});
312             }
313         }
314 
315         // Return the pairs.
316         return pairList;
317     }
