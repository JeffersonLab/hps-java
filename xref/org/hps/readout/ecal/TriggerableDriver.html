1   package org.hps.readout.ecal;
2   
3   import java.util.LinkedList;
4   import java.util.Queue;
5   import org.lcsim.event.EventHeader;
6   import org.lcsim.util.Driver;
7   
8   /**
9    * A driver that accepts triggers from TriggerDriver.
10   * To implement, write your own processTrigger(), and call checkTrigger() somewhere in process().
11   * You might want to set your own default latency in your constructor.
12   * readoutDeltaT() and isLive() are meant to be overridden if you're doing something unusual.
13   */
14  public abstract class TriggerableDriver extends Driver {
15  
16      private Queue<Double> triggerTimestamps = new LinkedList<Double>();
17      protected double triggerDelay = 0.0; // [ns]
18  
19      public void setTriggerDelay(double triggerDelay) {
20          this.triggerDelay = triggerDelay;
21      }
22  
23      /**
24       * 
25       * @return time reference for hits written by this driver in response to a trigger
26       */
27      public double readoutDeltaT() {
28          return ClockSingleton.getTime() + triggerDelay;
29      }
30  
31      @Override
32      protected void startOfData() {
33          TriggerDriver.addTriggerable(this);
34      }
35  
36      protected abstract void processTrigger(EventHeader event);
37  
38      protected void checkTrigger(EventHeader event) {
39          while (triggerTimestamps.peek() != null && ClockSingleton.getTime() >= triggerTimestamps.peek()) {
40              processTrigger(event);
41              triggerTimestamps.remove();
42          }
43      }
44  
45      public void addTrigger() {
46          triggerTimestamps.add(ClockSingleton.getTime() + triggerDelay);
47      }
48  
49      public boolean isLive() {
50          return true;
51      }
52      
53      public abstract int getTimestampType();
54  }
