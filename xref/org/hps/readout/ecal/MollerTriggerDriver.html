15  /**
16   * Class <code>MollerTriggerDriver</code> simulates a Møller trigger. It
17   * executes four cuts, three of which are single cluster cuts and the
18   * last of which is a positional cut. The single cluster cuts are on
19   * the seed and total energies of the cluster and the energy of the seed
20   * hit of the cluster. The positional cut ignores clusters that occur
21   * outside of a high-activity region for Møller events and excludes the
22   * primary background high-activity region.
23   * <br/><br/>
24   * At each event, any extant clusters are examined one at a time. If
25   * a cluster passes all four of the cuts, a trigger is produced and
26   * the remaining clusters are ignored. If either no clusters are present
27   * or no clusters pass the trigger cuts, no trigger is produced.
28   * <br/><br/>
29   * All thresholds can be set through a steering file. The driver also
30   * supports a verbose mode where it will output more details with every
31   * event to help with diagnostics.
32   */
33  public class MollerTriggerDriver extends TriggerDriver {

61      @Override
62      public void startOfData() {
63          // Initialize the cluster hit count diagnostic plots.
64          clusterHitCount = aida.histogram1D("Trigger Plots :: Cluster Hit Count Distribution", 9, 1, 10);
65          aClusterHitCount = aida.histogram1D("Trigger Plots :: Cluster Hit Count Distribution (Passed All Cuts)", 9, 1, 10);
66          
67          // Initialize the cluster total energy diagnostic plots.
68          clusterTotalEnergy = aida.histogram1D("Trigger Plots :: Cluster Total Energy Distribution", 176, 0.0, 2.2);
69          aClusterTotalEnergy = aida.histogram1D("Trigger Plots :: Cluster Total Energy Distribution (Passed All Cuts)", 176, 0.0, 2.2);
70          
71          // Initialize the cluster seed energy diagnostic plots.
72          clusterSeedEnergy = aida.histogram1D("Trigger Plots :: Cluster Seed Energy Distribution", 176, 0.0, 2.2);
73          aClusterSeedEnergy = aida.histogram1D("Trigger Plots :: Cluster Seed Energy Distribution (Passed All Cuts)", 176, 0.0, 2.2);
74          
75          // Initialize the seed distribution diagnostic plots.
76          clusterDistribution = aida.histogram2D("Trigger Plots :: Cluster Seed Distribution", 46, -23, 23, 11, -5.5, 5.5);
77          aClusterDistribution = aida.histogram2D("Trigger Plots :: Cluster Seed Distribution (Passed All Cuts)", 46, -23, 23, 11, -5.5, 5.5);
78          clusterDistribution100 = aida.histogram2D("Trigger Plots :: Cluster Seed Distribution (Over 100 MeV)", 46, -23, 23, 11, -5.5, 5.5);
79          
80          // Initialize the seed percentage of cluster energy.
81          seedPercent = aida.histogram1D("Analysis Plots :: Seed Percentage of Total Energy", 400, 0.0, 1.0);
82          
83          // Initialize the momentum angle plot.
84          momentumAngle = aida.histogram2D("Trigger Plots :: Particle Momentum Distribution (t = 0)", 500, -0.01, 0.06, 500, -0.04, 0.04);
85          aMomentumAngle = aida.histogram2D("Trigger Plots :: Particle Momentum Distribution (t = 0, Passed All Cuts)", 500, -0.01, 0.06, 500, -0.04, 0.04);
86          
87          // Add the allowed seed crystal positions to the seed set.
88          if(useVersionOne) {
89              // Add the allowed seed crystal positions to the seed set.
90              // y = +/- 1, x = -11 -> -15
91              for(int ix = -15; ix <= -11; ix++) {
92                  allowedSeedSet.add(new Point(ix, 1));
93                  allowedSeedSet.add(new Point(ix, -1));
94              } // y = +/- 2, x = -9 -> -15
95              for(int ix = -15; ix <= -9; ix++) {
96                  allowedSeedSet.add(new Point(ix, 2));
97                  allowedSeedSet.add(new Point(ix, -2));
98              }
99          }
100         else {
101             // y = +/- 1, x = -11 -> -13
102             for(int ix = -13; ix <= -11; ix++) {
103                 allowedSeedSet.add(new Point(ix, 1));
104                 allowedSeedSet.add(new Point(ix, -1));
105             } // y = +/- 2, x = -10 -> -14
106             for(int ix = -14; ix <= -10; ix++) {
107                 allowedSeedSet.add(new Point(ix, 2));
108                 allowedSeedSet.add(new Point(ix, -2));
109             }
110         }
111     }

113     @Override
114     protected boolean triggerDecision(EventHeader event) {
115         // Check if there is a cluster collection. If not, there is no
116         // reason to continue; a trigger can not be produced if there
117         // are no clusters.
118         if(!event.hasCollection(Cluster.class, clusterCollectionName)) {
119             // VERBOSE :: Note that no cluster collection exists for
120             //            this event.
121             if(verbose) { System.out.println("No cluster collection is present for event."); }
122             
123             // Indicate that there is no trigger.
124             return false;
125         }
126         
127         // VERBOSE :: Note that a cluster collection exists for
128         //            this event.
129         if(verbose) { System.out.println("Cluster collection is present for event."); }
130         
131         // Get the cluster list from the event.
132         List<Cluster> eventList = event.get(Cluster.class, clusterCollectionName);
133         
134         // VERBOSE :: Output the number of extant clusters.
135         if(verbose) { System.out.printf("%d clusters in event.%n", eventList.size()); }
136         
137         // Add the clusters from the event into the cluster list
138         // if they pass the minimum total cluster energy and seed
139         // energy thresholds.
140         for(Cluster cluster : eventList) {
141             // Increment the clusters processed count.
142             allClusters++;
143             
144             // Plot the seed energy / cluster energy histogram.
145             seedPercent.fill(cluster.getCalorimeterHits().get(0).getCorrectedEnergy() / cluster.getEnergy(), 1);
146             
147             // Get the cluster position indices.
148             int ix = cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
149             int iy = cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("iy");
150             
151             // VERBOSE :: Output the current cluster's properties.
152             if(verbose) {
153                 System.out.printf("\tTesting cluster at (%d, %d) with total energy %f and seed energy %f.%n",
154                         ix, iy, cluster.getCalorimeterHits().get(0).getCorrectedEnergy(), cluster.getEnergy());
155             }
156             
157             // Add the clusters to the uncut histograms.
158             clusterHitCount.fill(cluster.getCalorimeterHits().size());
159             clusterTotalEnergy.fill(cluster.getEnergy());
160             clusterSeedEnergy.fill(cluster.getCalorimeterHits().get(0).getCorrectedEnergy());
161             clusterDistribution.fill(ix > 0 ? ix - 1 : ix, iy, 1);
162             if(cluster.getCalorimeterHits().get(0).getCorrectedEnergy() > 0.100) { clusterDistribution100.fill(ix > 0 ? ix - 1 : ix, iy, 1); }
163             
164             // VERBOSE :: Output the single cluster trigger thresholds.
165             if(verbose) {
166                 System.out.printf("\tCluster seed energy threshold  :: [%f, %f]%n", clusterSeedEnergyThresholdLow, clusterSeedEnergyThresholdHigh);
167                 System.out.printf("\tCluster total energy threshold :: %f%n%n", clusterTotalEnergyThresholdLow);
168             }
169             
170             // Perform the single cluster cuts.
171             boolean totalEnergyCut = clusterTotalEnergyCut(cluster);
172             boolean seedEnergyCut = clusterSeedEnergyCut(cluster);
173             boolean hitCountCut = clusterHitCountCut(cluster);
174             boolean positionCut = clusterPositionCut(cluster);
175             
176             // Increment the single cut counts.
177             if(positionCut) {
178                 clusterPositionCount++;
179                 if(seedEnergyCut) {
180                     clusterSeedEnergyCount++;
181                     if(totalEnergyCut) {
182                         clusterTotalEnergyCount++;
183                         if(hitCountCut) {
184                             clusterHitCountCount++;
185                         }
186                     }
187                 }
188             }
189             
190             // VERBOSE :: Note whether the cluster passed the single
191             //            cluster cuts.
192             if(verbose) {
193                 System.out.printf("\tPassed seed energy cut    :: %b%n", seedEnergyCut);
194                 System.out.printf("\tPassed cluster energy cut :: %b%n", totalEnergyCut);
195                 System.out.printf("\tPassed hit count cut      :: %b%n", hitCountCut);
196                 System.out.printf("\tWithin allowed region     :: %b%n%n", positionCut);
197             }
198             
199             // Require that the cluster pass each of the cuts in
200             // order to qualify for a trigger.
201             if(totalEnergyCut && seedEnergyCut && hitCountCut && positionCut) {
202                 // Increment the number of events that have passed
203                 // the cuts.
204                 passedEvents++;
205                 
206                 // If the number of passed events exceeds the prescaling
207                 // threshold, throw a trigger.
208                 if(passedEvents >= prescale) {
209                     // Reset the number of passed events.
210                     passedEvents = 0;
211                     
212                     // Add the clusters to the cut histograms.
213                     aClusterHitCount.fill(cluster.getCalorimeterHits().size());
214                     aClusterTotalEnergy.fill(cluster.getEnergy());
215                     aClusterSeedEnergy.fill(cluster.getCalorimeterHits().get(0).getCorrectedEnergy());
216                     aClusterDistribution.fill(ix > 0 ? ix - 1 : ix, iy, 1);
217                     
218                     // Increment the trigger count.
219                     triggers++;
220                     
221                     // VERBOSE :: Indicate that a trigger occurred.
222                     if(verbose) { System.out.printf("\tTriggered!%n%n"); }
223                     
224                     // Return a trigger.
225                     return true;
226                 }
227             }
228         }
229         
230         // VERBOSE :: Note that the event has failed to trigger.
231         if(verbose) { System.out.println("No trigger.\n\n"); }
232         
233         // If one or more of the pair cuts failed, the we do not trigger.
234         return false;
235     }

238     // ==== Trigger Cut Methods =========================================
239     // ==================================================================
240     
241     /**
242      * Checks whether the cluster passes the threshold for minimum
243      * number of component hits.
244      * @param cluster - The cluster to check.
245      * @return Returns <code>true</code> if the cluster passes and <code>
246      * false</code> if it does not.
247      */
248     private boolean clusterHitCountCut(Cluster cluster) {
249         return cluster.getCalorimeterHits().size() >= clusterHitCountThreshold;
250     }
251     
252     /**
253      * Checks whether the cluster falls within the allowed range for
254      * the seed hit energy cut.
255      * @param cluster - The cluster to check.
256      * @return Returns <code>true</code> if the cluster passes and <code>
257      * false</code> if it does not.
258      */
259     private boolean clusterSeedEnergyCut(Cluster cluster) {
260         // Get the seed energy value.
261         double seedEnergy = cluster.getCalorimeterHits().get(0).getCorrectedEnergy();
262         
263         // Perform the seed energy cut.
264         return seedEnergy >= clusterSeedEnergyThresholdLow && seedEnergy <= clusterSeedEnergyThresholdHigh;
265     }
266     
267     /**
268      * Checks whether the cluster passes the threshold for minimum
269      * total cluster energy.
270      * @param cluster - The cluster to check.
271      * @return Returns <code>true</code> if the cluster passes and <code>
272      * false</code> if it does not.
273      */
274     private boolean clusterTotalEnergyCut(Cluster cluster) {
275         // Get the cluster energy.
276         double clusterEnergy = cluster.getEnergy();
277         
278         // Perform the cut.
279         return clusterEnergy >= clusterTotalEnergyThresholdLow && clusterEnergy <= clusterTotalEnergyThresholdHigh;
280     }
281     
282     /**
283      * Checks whether the cluster falls into the accepted region on
284      * the calorimeter face.
285      * @param cluster - The cluster to check.
286      * @return Returns <code>true</code> if the cluster passes and <code>
287      * false</code> if it does not.
288      */
289     private boolean clusterPositionCut(Cluster cluster) {
290         // Get the cluster position.
291         Point seedLoc =  new Point(cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("ix"),
292                 cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("iy"));
293         
294         // Check if it is one of the allowed seed crystals.
295         return allowedSeedSet.contains(seedLoc);
296     }
