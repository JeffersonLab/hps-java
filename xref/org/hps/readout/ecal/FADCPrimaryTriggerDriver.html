17  /**
18   * Class <code>FADCPrimaryTriggerDriver</code> reads reconstructed
19   * clusters and makes trigger decisions on them. It is designed to
20   * trigger off 2.2 GeV beam A' events. Cuts can either be set manually
21   * in a steering file or automatically by specifying a background level.
22   * The code for generating trigger pairs and handling the coincidence
23   * window comes from <code>FADCTriggerDriver</code>.
24   * 
25   * @see FADCTriggerDriver
26   */
27  public class FADCPrimaryTriggerDriver extends TriggerDriver {
28      // ==================================================================
29      // ==== Trigger General Default Parameters ==========================
30      // ==================================================================
31      private String clusterCollectionName = "EcalClusters";        // Name for the LCIO cluster collection.
32      private int pairCoincidence = 2;                              // Maximum allowed time difference between clusters. (4 ns clock-cycles)
33      private int backgroundLevel = -1;                             // Automatically sets the cuts to achieve a predetermined background rate.
34      private TriggerModule triggerModule = new TriggerModule(1.0, 0.050,
35              6.600, 0.010, 6.600, 0.000, 13.200, 6.600, 0.0, 360, 0.0055);

74       * Initializes the cluster pair queues and other variables.

76      @Override
77      public void startOfData() {
78          // Define plot type names.
79          String[] plotType = new String[PLOT_COUNT];
80          plotType[NO_CUTS] = "";
81          plotType[ALL_CUTS] = " (Passed All Cuts)";
82          plotType[OVER_1HIT] = " (More than 1 Hit)";
83          plotType[OVER_2HIT] = " (More than 2 Hits)";
84          plotType[SINGLES_CUTS] = " (Passed Single Cuts)";
85          
86          // Define plot type directories.
87          String[] plotDir = new String[PLOT_COUNT];
88          plotDir[NO_CUTS] = "NoCuts/";
89          plotDir[ALL_CUTS] = "PassedAll/";
90          plotDir[OVER_1HIT] = "2PlusHits/";
91          plotDir[OVER_2HIT] = "3PlusHits/";
92          plotDir[SINGLES_CUTS] = "PassedSingles/";
93          
94          // Instantiate the singles plot arrays.
95          clusterSeedEnergy = new IHistogram1D[PLOT_COUNT];
96          clusterHitCount = new IHistogram1D[PLOT_COUNT];
97          clusterTotalEnergy = new IHistogram1D[PLOT_COUNT];
98          clusterDistribution = new IHistogram2D[PLOT_COUNT];
99          
100         // Instantiate the pair plot arrays. Note that the pair cuts
101         // only ever see clusters that pass the singles cuts, so the
102         // "passed singles cuts" plots are meaningless. Thusly, the
103         // pair plots have one fewer plot than the singles.
104         pairEnergySum = new IHistogram1D[PLOT_COUNT - 1];
105         pairEnergyDifference = new IHistogram1D[PLOT_COUNT - 1];
106         pairCoplanarity = new IHistogram1D[PLOT_COUNT - 1];
107         pairEnergySlope = new IHistogram1D[PLOT_COUNT - 1];
108         pairEnergySum2DDistribution = new IHistogram2D[PLOT_COUNT - 1];
109         
110         // Instantiate the plots.
111         for(int i = 0; i < PLOT_COUNT; i++) {
112             System.out.println(plotDir[i] + "Cluster Seed Energy" + plotType[i]);
113             clusterSeedEnergy[i] = aida.histogram1D(plotDir[i] + "Cluster Seed Energy" + plotType[i], 176, 0.0, 2.2);
114             clusterHitCount[i] = aida.histogram1D(plotDir[i] + "Cluster Hit Count" + plotType[i], 9, 0.5, 9.5);
115             clusterTotalEnergy[i] = aida.histogram1D(plotDir[i] + "Cluster Total Energy" + plotType[i], 176, 0.0, 2.2);
116             clusterDistribution[i] = aida.histogram2D(plotDir[i] + "Cluster Seed" + plotType[i], 46, -23, 23, 11, -5.5, 5.5);
117             
118             if(i != PLOT_COUNT - 1) {
119                 pairEnergySum[i] = aida.histogram1D(plotDir[i] + "Pair Energy Sum" + plotType[i], 176, 0.0, 4.4);
120                 pairEnergyDifference[i] = aida.histogram1D(plotDir[i] + "Pair Energy Difference" + plotType[i], 176, 0.0, 2.2);
121                 pairCoplanarity[i] = aida.histogram1D(plotDir[i] + "Pair Coplanarity" + plotType[i], 180, 0.0, 180.0);
122                 pairEnergySlope[i] = aida.histogram1D(plotDir[i] + "Pair Energy Slope" + plotType[i], 200, 0.0, 4.0);
123                 pairEnergySum2DDistribution[i] = aida.histogram2D(plotDir[i] + "Pair Energy Sum 2D" + plotType[i], 176, 0.0, 4.4, 176, 0.0, 4.4);
124             }
125         }
126         
127         // Make sure that a valid cluster collection name has been
128         // defined. If it has not, throw an exception.
129         if (clusterCollectionName == null) {
130             throw new RuntimeException("The parameter clusterCollectionName was not set!");
131         }
132         
133         // Initialize the top and bottom cluster queues.
134         topClusterQueue = new LinkedList<List<Cluster>>();
135         botClusterQueue = new LinkedList<List<Cluster>>();
136         
137         // Populate the top cluster queue. It should be populated with
138         // a number of empty lists equal to (2 * pairCoincidence + 1).
139         for (int i = 0; i < 2 * pairCoincidence + 1; i++) {
140             topClusterQueue.add(new ArrayList<Cluster>());
141         }
142         
143         // Populate the bottom cluster queue. It should be populated with
144         // a number of empty lists equal to (2 * pairCoincidence + 1).
145         for (int i = 0; i < pairCoincidence + 1; i++) {
146             botClusterQueue.add(new ArrayList<Cluster>());
147         }
148         
149         // If a background level has been set, pick the correct cuts.
150         if(backgroundLevel != -1) { setBackgroundCuts(backgroundLevel); }
151         
152         // Run the superclass method.
153         super.startOfData();
154     }

195     /**
196      * Performs single cluster cuts for the event and passes any clusters
197      * which survive to be formed into cluster pairs for the trigger.
198      */
199     @Override
200     public void process(EventHeader event) {
201         // Process the list of clusters for the event, if it exists.
202         if (event.hasCollection(Cluster.class, clusterCollectionName)) {
203             // Get the collection of clusters.
204             List<Cluster> clusterList = event.get(Cluster.class, clusterCollectionName);
205             
206             // Create a list to hold clusters which pass the single
207             // cluster cuts.
208             List<Cluster> goodClusterList = new ArrayList<Cluster>(clusterList.size());
209             
210             // Sort through the cluster list and add clusters that pass
211             // the single cluster cuts to the good list.
212             clusterLoop:
213             for(Cluster cluster : clusterList) {
214                 // Increment the number of processed clusters.
215                 allClusters++;
216                 
217                 // Get the cluster plot values.
218                 int hitCount = cluster.getCalorimeterHits().size();
219                 double seedEnergy = cluster.getCalorimeterHits().get(0).getCorrectedEnergy();
220                 double clusterEnergy = cluster.getEnergy();
221                 int ix = cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
222                 int iy = cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("iy");
223                 
224                 // VERBOSE :: Note that a cluster is being processed.
225                 if(verbose) {
226                     System.out.printf("%nProcessing cluster at (% 2d, % 2d)%n", ix, iy);
227                 }
228                 
229                 // Correct for "hole" on the x-axis for plotting.
230                 if(ix > 0) { ix = ix - 1; }
231                 
232                 // Fill the general plots.
233                 clusterSeedEnergy[NO_CUTS].fill(seedEnergy);
234                 clusterTotalEnergy[NO_CUTS].fill(clusterEnergy);
235                 clusterHitCount[NO_CUTS].fill(hitCount);
236                 clusterDistribution[NO_CUTS].fill(ix, iy);
237                 
238                 // Fill the hit count plots for N > 1.
239                 if(hitCount > 1) {
240                     // Populate the plots.
241                     clusterSeedEnergy[OVER_1HIT].fill(seedEnergy);
242                     clusterTotalEnergy[OVER_1HIT].fill(clusterEnergy);
243                     clusterHitCount[OVER_1HIT].fill(hitCount);
244                     clusterDistribution[OVER_1HIT].fill(ix, iy);
245                     
246                     // Fill the hit count plots for N > 2.
247                     if(hitCount > 2) {
248                         // Populate the plots.
249                         clusterSeedEnergy[OVER_2HIT].fill(seedEnergy);
250                         clusterTotalEnergy[OVER_2HIT].fill(clusterEnergy);
251                         clusterHitCount[OVER_2HIT].fill(hitCount);
252                         clusterDistribution[OVER_2HIT].fill(ix, iy);
253                     }
254                 }
255                 
256                 // ==== Seed Hit Energy Cut ====================================
257                 // =============================================================
258                 // VERBOSE :: Print the seed energy comparison check.
259                 if(verbose) {
260                     System.out.printf("\tSeed Energy Cut    :: %.3f < %.3f < %.3f --> %b%n",
261                             triggerModule.getCutValue(TriggerModule.CLUSTER_SEED_ENERGY_LOW), seedEnergy,
262                             triggerModule.getCutValue(TriggerModule.CLUSTER_SEED_ENERGY_HIGH),
263                             triggerModule.clusterSeedEnergyCut(cluster));
264                 }
265                 
266                 // If the cluster fails the cut, skip to the next cluster.
267                 if(!triggerModule.clusterSeedEnergyCut(cluster)) {
268                     continue clusterLoop;
269                 }
270                 
271                 // Otherwise, note that it passed the cut.
272                 clusterSeedEnergyCount++;
273                 
274                 // ==== Cluster Hit Count Cut ==================================
275                 // =============================================================
276                 // VERBOSE :: Print the hit count comparison check.
277                 if(verbose) {
278                     System.out.printf("\tHit Count Cut      :: %d >= %.0f --> %b%n",
279                             hitCount, triggerModule.getCutValue(TriggerModule.CLUSTER_HIT_COUNT_LOW),
280                             triggerModule.clusterHitCountCut(cluster));
281                 }
282                 
283                 // If the cluster fails the cut, skip to the next cluster.
284                 if(!triggerModule.clusterHitCountCut(cluster)) {
285                     continue clusterLoop;
286                 }
287                 
288                 // Otherwise, note that it passed the cut.
289                 clusterHitCountCount++;
290                 
291                 // ==== Cluster Total Energy Cut ===============================
292                 // =============================================================
293                 // VERBOSE :: Print the cluster energy comparison check.
294                 if(verbose) {
295                     System.out.printf("\tCluster Energy Cut :: %.3f < %.3f < %.3f --> %b%n",
296                             triggerModule.getCutValue(TriggerModule.CLUSTER_TOTAL_ENERGY_LOW), clusterEnergy,
297                             triggerModule.getCutValue(TriggerModule.CLUSTER_TOTAL_ENERGY_HIGH),
298                             triggerModule.clusterTotalEnergyCut(cluster));
299                 }
300                 
301                 // If the cluster fails the cut, skip to the next cluster.
302                 if(!triggerModule.clusterTotalEnergyCut(cluster)) {
303                     continue clusterLoop;
304                 }
305                 
306                 // Otherwise, note that it passed the cut.
307                 clusterTotalEnergyCount++;
308                 
309                 // Fill the "passed single cuts" plots.
310                 clusterSeedEnergy[SINGLES_CUTS].fill(seedEnergy);
311                 clusterTotalEnergy[SINGLES_CUTS].fill(clusterEnergy);
312                 clusterHitCount[SINGLES_CUTS].fill(hitCount);
313                 clusterDistribution[SINGLES_CUTS].fill(ix, iy, 1);
314                 
315                 // A cluster that passes all of the single-cluster cuts
316                 // can be used in cluster pairs.
317                 goodClusterList.add(cluster);
318             }
319             
320             // Put the good clusters into the cluster queue.
321             updateClusterQueues(goodClusterList);
322         }
323         
324         // Perform the superclass event processing.
325         super.process(event);
326     }

470     /**
471      * Get a list of all unique cluster pairs in the event.
472      * @return A <code>List</code> collection of <code>Cluster</code> 
473      * objects containing all cluster pairs.
474      */
475     protected List<Cluster[]> getClusterPairsTopBot() {
476         // Create a list to store cluster pairs. 
477         List<Cluster[]> clusterPairs = new ArrayList<Cluster[]>();
478         
479         // Loop over all top-bottom pairs of clusters; higher-energy cluster goes first in the pair
480         // To apply pair coincidence time, use only bottom clusters from the 
481         // readout cycle pairCoincidence readout cycles ago, and top clusters 
482         // from all 2*pairCoincidence+1 previous readout cycles
483         for (Cluster botCluster : botClusterQueue.element()) {
484             for (List<Cluster> topClusters : topClusterQueue) {
485                 for (Cluster topCluster : topClusters) {
486                     // The first cluster in a pair should always be
487                     // the higher energy cluster. If the top cluster
488                     // is higher energy, it goes first.
489                     if (topCluster.getEnergy() > botCluster.getEnergy()) {
490                         Cluster[] clusterPair = {topCluster, botCluster};
491                         clusterPairs.add(clusterPair);
492                     }
493                     
494                     // Otherwise, the bottom cluster goes first.
495                     else {
496                         Cluster[] clusterPair = {botCluster, topCluster};
497                         clusterPairs.add(clusterPair);
498                     }
499                 }
500             }
501         }
502         
503         // Return the cluster pair lists.
504         return clusterPairs;
505     }

507     /**
508      * Determines if the event produces a trigger.
509      * @return Returns <code>true</code> if the event produces a trigger
510      * and <code>false</code> if it does not.
511      */
512     @Override
513     protected boolean triggerDecision(EventHeader event) {
514         // If there is a list of clusters present for this event,
515         // check whether it passes the trigger conditions.
516         if (event.hasCollection(Cluster.class, clusterCollectionName)) {
517             return testTrigger();
518         }
519         
520         // Otherwise, this event can not produce a trigger and should
521         // return false automatically.
522         else { return false; }
523     }

525     /**
526      * Sets the cuts to accept a certain background level. Levels are
527      * defined with an integer between 1 and 10, where each step is
528      * indicative of a hypothetical background rate of approximately
529      * 5 kHz. Thusly, <code>backgroundLevel = 1</code> represents a
530      * hypothetical background rate of approximately 5 kHz, <code>
531      * backgroundLevel = 2</code> represents 10 kHz, and so on. Using
532      * a value of 0 will set the cuts to the original test run values.
533      * @param backgroundLevel - The hypothetical background rate value.
534      * This must be between 1 and 10, or zero for the original test
535      * run cut values.
536      */
537     private void setBackgroundCuts(int backgroundLevel) {
538         // Make sure that the background level is valid.
539         if(backgroundLevel < -1 || backgroundLevel > 10) {
540             throw new RuntimeException(String.format("Trigger cuts are undefined for background level %d.", backgroundLevel));
541         }
542         
543         // Some cut values are almost always the same thing. Set those
544         // here and only overwrite if necessary.
545         triggerModule.setCutValue(TriggerModule.CLUSTER_SEED_ENERGY_LOW,     0.125);
546         triggerModule.setCutValue(TriggerModule.CLUSTER_SEED_ENERGY_HIGH,    1.300);
547         triggerModule.setCutValue(TriggerModule.CLUSTER_TOTAL_ENERGY_LOW,    0.200);
548         triggerModule.setCutValue(TriggerModule.CLUSTER_TOTAL_ENERGY_HIGH,   1.700);

605     /**
606      * Tests all of the current cluster pairs for triggers.
607      * @return Returns <code>true</code> if one of the cluster pairs
608      * passes all of the cluster cuts and <code>false</code> otherwise.
609      */
610     private boolean testTrigger() {
611         // Get the list of cluster pairs.
612         List<Cluster[]> clusterPairs = getClusterPairsTopBot();
613         
614         // Iterate over the cluster pairs and perform each of the cluster
615         // pair cuts on them. A cluster pair that passes all of the
616         // cuts registers as a trigger.
617         pairLoop:
618         for (Cluster[] clusterPair : clusterPairs) {
619             // Increment the number of processed cluster pairs.
620             allPairs++;
621             
622             // Get the plot values for the pair cuts.
623             double energySum = TriggerModule.getValueEnergySum(clusterPair);
624             double energyDifference = TriggerModule.getValueEnergyDifference(clusterPair);
625             double energySlope = TriggerModule.getValueEnergySlope(clusterPair, triggerModule.getCutValue(TriggerModule.PAIR_ENERGY_SLOPE_F));
626             double coplanarity = TriggerModule.getValueCoplanarity(clusterPair);
627             
628             // Fill the general plots.
629             pairEnergySum[NO_CUTS].fill(energySum);
630             pairEnergyDifference[NO_CUTS].fill(energyDifference);
631             pairEnergySlope[NO_CUTS].fill(energySlope);
632             pairCoplanarity[NO_CUTS].fill(coplanarity);
633             pairEnergySum2DDistribution[NO_CUTS].fill(clusterPair[0].getEnergy(), clusterPair[1].getEnergy());
634             
635             // Fill the hit count plots for N > 1.
636             if(clusterPair[0].getCalorimeterHits().size() > 1 && clusterPair[1].getCalorimeterHits().size() > 1) {
637                 // Populate the plots.
638                 pairEnergySum[OVER_1HIT].fill(energySum);
639                 pairEnergyDifference[OVER_1HIT].fill(energyDifference);
640                 pairEnergySlope[OVER_1HIT].fill(energySlope);
641                 pairCoplanarity[OVER_1HIT].fill(coplanarity);
642                 pairEnergySum2DDistribution[OVER_1HIT].fill(clusterPair[0].getEnergy(), clusterPair[1].getEnergy());
643                 
644                 // Fill the hit count plots for N > 2.
645                 if(clusterPair[0].getCalorimeterHits().size() > 2 && clusterPair[1].getCalorimeterHits().size() > 2) {
646                     // Populate the plots.
647                     pairEnergySum[OVER_2HIT].fill(energySum);
648                     pairEnergyDifference[OVER_2HIT].fill(energyDifference);
649                     pairEnergySlope[OVER_2HIT].fill(energySlope);
650                     pairCoplanarity[OVER_2HIT].fill(coplanarity);
651                     pairEnergySum2DDistribution[OVER_2HIT].fill(clusterPair[0].getEnergy(), clusterPair[1].getEnergy());
652                 }
653             }
654             
655             // ==== Pair Energy Sum Cut ====================================
656             // =============================================================
657             // If the cluster fails the cut, skip to the next pair.
658             if(!triggerModule.pairEnergySumCut(clusterPair)) {
659                 continue pairLoop;
660             }
661             
662             // Otherwise, note that it passed the cut.
663             pairEnergySumCount++;
664             
665             // ==== Pair Energy Difference Cut =============================
666             // =============================================================
667             // If the cluster fails the cut, skip to the next pair.
668             if(!triggerModule.pairEnergyDifferenceCut(clusterPair)) {
669                 continue pairLoop;
670             }
671             
672             // Otherwise, note that it passed the cut.
673             pairEnergyDifferenceCount++;
674             
675             // ==== Pair Energy Slope Cut ==================================
676             // =============================================================
677             // If the cluster fails the cut, skip to the next pair.
678             if(!triggerModule.pairEnergySlopeCut(clusterPair)) {
679                 continue pairLoop;
680             }
681             
682             // Otherwise, note that it passed the cut.
683             pairEnergySlopeCount++;
684             
685             // ==== Pair Coplanarity Cut ===================================
686             // =============================================================
687             // If the cluster fails the cut, skip to the next pair.
688             if(!triggerModule.pairCoplanarityCut(clusterPair)) {
689                 continue pairLoop;
690             }
691             
692             // Otherwise, note that it passed the cut.
693             pairCoplanarityCount++;
694             
695             // Get the cluster plot values.
696             int[] hitCount = new int[2];
697             double[] seedEnergy = new double[2];
698             double[] clusterEnergy = new double[2];
699             int[] ix = new int[2];
700             int[] iy = new int[2];
701             for(int i = 0; i < 2; i++) {
702                 hitCount[i] = clusterPair[i].getCalorimeterHits().size();
703                 seedEnergy[i] = clusterPair[i].getCalorimeterHits().get(0).getCorrectedEnergy();
704                 clusterEnergy[i] = clusterPair[i].getEnergy();
705                 ix[i] = clusterPair[i].getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
706                 iy[i] = clusterPair[i].getCalorimeterHits().get(0).getIdentifierFieldValue("iy");
707                 if(ix[i] > 0) { ix[i] = ix[i] - 1; }
708             }
709             
710             // Fill the general plots.
711             for(int i = 0; i < 2; i++) {
712                 clusterSeedEnergy[ALL_CUTS].fill(seedEnergy[i]);
713                 clusterTotalEnergy[ALL_CUTS].fill(clusterEnergy[i]);
714                 clusterHitCount[ALL_CUTS].fill(hitCount[i]);
715                 clusterDistribution[ALL_CUTS].fill(ix[i], iy[i]);
716             }
717             
718             // Fill the "passed all cuts" plots.
719             pairEnergySum[ALL_CUTS].fill(energySum, 1);
720             pairEnergyDifference[ALL_CUTS].fill(energyDifference, 1);
721             pairEnergySlope[ALL_CUTS].fill(energySlope, 1);
722             pairCoplanarity[ALL_CUTS].fill(coplanarity, 1);
723             pairEnergySum2DDistribution[ALL_CUTS].fill(clusterPair[0].getEnergy(), clusterPair[1].getEnergy());
724             
725             // Clusters that pass all of the pair cuts produce a trigger.
726             if (verbose) {
727                 System.out.format("Passed trigger cuts: cluster 0 (energy %f, ix %d, iy %d, size %d) and cluster 1 (energy %f, ix %d, iy %d, size %d)\n", clusterPair[0].getEnergy(), clusterPair[0].getCalorimeterHits().get(0).getIdentifierFieldValue("ix"), clusterPair[0].getCalorimeterHits().get(0).getIdentifierFieldValue("iy"), clusterPair[0].getSize(), clusterPair[1].getEnergy(), clusterPair[1].getCalorimeterHits().get(0).getIdentifierFieldValue("ix"), clusterPair[1].getCalorimeterHits().get(0).getIdentifierFieldValue("iy"), clusterPair[1].getSize());
728             }
729             return true;
730         }
731         
732         // If the loop terminates without producing a trigger, there
733         // are no cluster pairs which meet the trigger conditions.
734         return false;
735     }

737     /**
738      * Adds clusters from a new event into the top and bottom cluster
739      * queues so that they may be formed into pairs.
740      * @param clusterList - The clusters to add to the queues.
741      */
742     private void updateClusterQueues(List<Cluster> clusterList) {
743         // Create lists to store the top and bottom clusters.
744         ArrayList<Cluster> topClusterList = new ArrayList<Cluster>();
745         ArrayList<Cluster> botClusterList = new ArrayList<Cluster>();
746         
747         // Loop over the clusters in the cluster list.
748         for (Cluster cluster : clusterList) {
749             // If the cluster is on the top of the calorimeter, it
750             // goes into the top cluster list.
751             if (cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("iy") > 0) {
752                 topClusterList.add(cluster);
753             }
754             
755             // Otherwise, it goes into the bottom cluster list.
756             else { botClusterList.add(cluster); }
757         }
758         
759         // Add the new cluster lists to the cluster queues.
760         topClusterQueue.add(topClusterList);
761         botClusterQueue.add(botClusterList);
762         
763         // Remove the oldest cluster lists from the queues.
764         topClusterQueue.remove();
765         botClusterQueue.remove();
766     }
