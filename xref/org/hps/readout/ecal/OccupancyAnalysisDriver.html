14  public class OccupancyAnalysisDriver extends Driver {
15      // Internal variables.
16      private double scalingFactor = 0.05;
17      private double seedThreshold = 0.050;
18      private double beamRatio = 1.92 / 2.2;
19      private double clusterThreshold = 0.200;
20      private AIDA aida = AIDA.defaultInstance();
21      private boolean ignoreBeamGapRows = false;
22      
23      // LCIO Collection Names
24      private String clusterCollectionName = "EcalClusters";
25      private String hitCollectionName = "EcalCorrectedHits";
26      
27      // Trigger plots.
28      IHistogram2D occupancyDistribution;
29      IHistogram2D[] clusterDistribution = new IHistogram2D[2];
30      IHistogram1D[] clusterHitDistribution = new IHistogram1D[2];
31      IHistogram1D[] totalEnergyDistribution = new IHistogram1D[2];
32      IHistogram1D[] clusterEnergyDistribution = new IHistogram1D[2];
33

35          this.ignoreBeamGapRows = ignoreBeamGapRows;
36      }
37      
38      public void setBeamRatio(double beamRatio) {
39          this.beamRatio = beamRatio;
40      }
41

43          this.scalingFactor = scalingFactor;
44      }
45      
46      public void setSeedThreshold(double seedThreshold) {
47          this.seedThreshold = seedThreshold;
48      }
49      
50      public void setClusterThreshold(double clusterThreshold) {
51          this.clusterThreshold = clusterThreshold;
52      }
53      
54      public void setClusterCollectionName(String clusterCollectionName) {
55          this.clusterCollectionName = clusterCollectionName;
56      }
57      
58      public void setHitCollectionName(String hitCollectionName) {
59          this.hitCollectionName = hitCollectionName;
60      }

62      @Override
63      public void process(EventHeader event) {
64          // If clusters are present, process them.
65          if(event.hasCollection(Cluster.class, clusterCollectionName)) {
66              // Get the list of clusters.
67              List<Cluster> clusterList = event.get(Cluster.class, clusterCollectionName);
68              
69              // Use the clusters to populate the cluster plots.
70              for(Cluster cluster : clusterList) {
71                  // Get the ix and iy values for the cluster.
72                  int ix = cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
73                  int iy = cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("iy");
74                  
75                  // If we want to ignore the beam gap rows, make sure
76                  // that iy exceeds two.
77                  if(!ignoreBeamGapRows || (Math.abs(iy) > 2)) {
78                      // If the cluster passes the seed threshold, place it in
79                      // the level 1 plots.
80                      if(cluster.getCalorimeterHits().get(0).getCorrectedEnergy() >= seedThreshold) {
81                          clusterDistribution[0].fill(ix, iy, scalingFactor);
82                          clusterHitDistribution[0].fill(cluster.getCalorimeterHits().size(), scalingFactor);
83                          clusterEnergyDistribution[0].fill(cluster.getEnergy() * beamRatio, scalingFactor);
84                      }
85                      
86                      // If the cluster energy passes the cluster threshold,
87                      // populate the level 2 plots.
88                      if(cluster.getEnergy() >= clusterThreshold) {
89                          clusterDistribution[1].fill(ix, iy, scalingFactor);
90                          clusterHitDistribution[1].fill(cluster.getCalorimeterHits().size(), scalingFactor);
91                          clusterEnergyDistribution[1].fill(cluster.getEnergy() * beamRatio, scalingFactor);
92                      }
93                  }
94              }
95          }
96          
97          // If the event has hits, process them.
98          if(event.hasCollection(CalorimeterHit.class, hitCollectionName)) {
99              // Get the list of hits.
100             List<CalorimeterHit> hitList = event.get(CalorimeterHit.class, hitCollectionName);
101             
102             // Track the energy in the top and bottom of the calorimeter.
103             double[] energy = { 0.0, 0.0 };
104             
105             // Iterate over the hits.
106             for(CalorimeterHit hit : hitList) {
107                 // Get the ix and iy values.
108                 int ix = hit.getIdentifierFieldValue("ix");
109                 int iy = hit.getIdentifierFieldValue("iy");
110                 
111                 // If we want to ignore beam gap rows, ensure that iy
112                 // is greater than 2.
113                 if(!ignoreBeamGapRows || Math.abs(iy) > 2) {
114                     // Add the energy to the appropriate energy tracking
115                     // variable for the calorimeter halves.
116                     if(iy > 0) { energy[0] += hit.getCorrectedEnergy() * beamRatio; }
117                     else { energy[1] += hit.getCorrectedEnergy() * beamRatio; }
118                     
119                     // Populate the occupancy distribution.
120                     occupancyDistribution.fill(ix, iy, scalingFactor);
121                 }
122             }
123             
124             // Populate the total calorimeter energy plot.
125             totalEnergyDistribution[0].fill(energy[0], scalingFactor);
126             totalEnergyDistribution[1].fill(energy[1], scalingFactor);
127         }
128     }

130     @Override
131     public void startOfData() {
132         // Define the cluster distribution plots.
133         String[] clusterDistName = { String.format("Comp Plots :: Cluster Seed Distribution [Seed Threshold %.3f GeV]", seedThreshold),
134                 String.format("Comp Plots :: Cluster Seed Distribution [Cluster Threshold %.3f GeV]", clusterThreshold) };
135         clusterDistribution[0] = aida.histogram2D(clusterDistName[0], 46, -23, 23, 11, -5.5, 5.5);
136         clusterDistribution[1] = aida.histogram2D(clusterDistName[1], 46, -23, 23, 11, -5.5, 5.5);
137         
138         // Define the occupancy distribution plots.
139         String occupancyDistName = String.format("Comp Plots :: Crystal Occupancy");
140         occupancyDistribution = aida.histogram2D(occupancyDistName, 46, -23, 23, 11, -5.5, 5.5);
141         
142         // Define the cluster hit count distribution.
143         String[] clusterHitDistName = { String.format("Comp Plots :: Cluster Hit Count Distribution [Seed Threshold %.3f GeV]", seedThreshold),
144                 String.format("Comp Plots :: Cluster Hit Count Distribution [Cluster Threshold %.3f GeV]", clusterThreshold) };
145         clusterHitDistribution[0] = aida.histogram1D(clusterHitDistName[0], 9, 1, 10);
146         clusterHitDistribution[1] = aida.histogram1D(clusterHitDistName[1], 9, 1, 10);
147         
148         // Define the cluster total energy distribution.
149         String[] clusterEnergyDistName = { String.format("Comp Plots :: Cluster Total Energy Distribution [Seed Threshold %.3f GeV]", seedThreshold),
150                 String.format("Comp Plots :: Cluster Total Energy Distribution [Cluster Threshold %.3f GeV]", clusterThreshold) };
151         clusterEnergyDistribution[0] = aida.histogram1D(clusterEnergyDistName[0], 176, 0.0, 2.2);
152         clusterEnergyDistribution[1] = aida.histogram1D(clusterEnergyDistName[1], 176, 0.0, 2.2);
153         
154         // Define the calorimeter total energy distribution.
155         String[] totalEnergyDistName = { String.format("Comp Plots :: Calorimeter Event Energy Distribution [Top]"),
156                 String.format("Comp Plots :: Calorimeter Event Energy Distribution [Bottom]") };
157         totalEnergyDistribution[0] = aida.histogram1D(totalEnergyDistName[0], 500, 0.0, 10.0);
158         totalEnergyDistribution[1] = aida.histogram1D(totalEnergyDistName[1], 500, 0.0, 10.0);
159     }
