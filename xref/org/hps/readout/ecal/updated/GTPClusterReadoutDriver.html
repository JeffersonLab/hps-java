28  /**
29   * Class <code>GTPClusterReadoutDriver</code> produces GTP cluster
30   * objects for use in the readout trigger simulation. It takes in
31   * {@link org.lcsim.event.CalorimeterHit CalorimeterHit} objects as
32   * input and generates clusters from these using the GTP algorithm.
33   * This algorithm works by selected all hits in the current
34   * clock-cycle (4 ns period) and comparing them to adjacent hits. If
35   * a given hit is an energy maximum compared to all adjacent hits in
36   * both the current clock-cycle, and a number of clock-cycles before
37   * and after the current cycle (defined through the variable {@link
38   * org.hps.readout.ecal.updated.GTPClusterReadoutDriver#temporalWindow
39   * temporalWindow} and set through the method {@link
40   * org.hps.readout.ecal.updated.GTPClusterReadoutDriver#setClusterWindow(int)
41   * setClusterWindow(int)}), then it is a seed hit so long as it also
42   * exceeds a certain minimum energy (defined through the variable
43   * {@link
44   * org.hps.readout.ecal.updated.GTPClusterReadoutDriver#seedEnergyThreshold
45   * seedEnergyThreshold} and set through the method {@link
46   * org.hps.readout.ecal.updated.GTPClusterReadoutDriver#setSeedEnergyThreshold(double)
47   * setSeedEnergyThreshold(double)}).<br/><br/>
48   * Clusters are then output as objects of type {@link
49   * org.lcsim.event.Cluster Cluster} to the specified output
50   * collection. If the {@link
51   * org.hps.readout.ecal.updated.GTPClusterReadoutDriver#setWriteClusterCollection(boolean)
52   * setWriteClusterCollection(boolean)} is set to true, the clusters
53   * will also be persisted into the output LCIO file.
54   */
55  public class GTPClusterReadoutDriver extends ReadoutDriver {

60      /**
61       * The name of the collection that contains the calorimeter hits
62       * from which clusters should be generated.
63       */
64      private String inputCollectionName = "EcalCorrectedHits";
65      /**
66       * The name of the collection into which generated clusters should
67       * be output.
68       */
69      private String outputCollectionName = "EcalClustersGTP";
70      
71      // ==============================================================
72      // ==== Driver Options ==========================================
73      // ==============================================================
74      
75      /**
76       * The time window used for cluster verification. A seed hit must
77       * be the highest energy hit within plus or minus this range in
78       * order to be considered a valid cluster.
79       */
80      private int temporalWindow = 16;
81      /**
82       * The minimum energy needed for a hit to be considered as a seed
83       * hit candidate.
84       */
85      private double seedEnergyThreshold = 0.050;
86      /**
87       * The local time for the driver. This starts at 2 ns due to a
88       * quirk in the timing of the {@link
89       * org.hps.readout.ecal.updated.EcalReadoutDriver
90       * EcalReadoutDriver}.
91       */
92      private double localTime = 0.0;
93      /**
94       * The length of time by which objects produced by this driver
95       * are shifted due to the need to buffer data from later events.
96       * This is calculated automatically.
97       */
98      private double localTimeDisplacement = 0;

105      * An object which can provide, given an argument cell ID, a map
106      * of cell IDs that are physically adjacent to the argument ID.
107      * This is used to determine adjacency for energy comparisons in

110     private NeighborMap neighborMap;

114     /**
115      * Sets whether or not the DAQ configuration is applied into the driver
116      * the EvIO data stream or whether to read the configuration from data files.
117      * 
118      * @param state - <code>true</code> indicates that the DAQ configuration is
119      * applied into the readout system, and <code>false</code> that it
120      * is not applied into the readout system.
121      */
122     public void setDaqConfigurationAppliedintoReadout(boolean state) {
123         // If the DAQ configuration should be read, attach a listener
124         // to track when it updates.               
125         if (state) {
126             ConfigurationManager2019.addActionListener(new ActionListener() {
127                 @Override
128                 public void actionPerformed(ActionEvent e) {
129                     // Get the DAQ configuration.
130                     DAQConfig2019 daq = ConfigurationManager2019.getInstance();                    
131                     VTPConfig2019 config = daq.getVTPConfig();
132 
133                     // Load the DAQ settings from the configuration manager.
134                     seedEnergyThreshold = config.getEcalClusterSeedThr();
135                     temporalWindow = config.getEcalClusterHitDT();                  
136                 }
137             });
138         }

141     @Override
142     public void detectorChanged(Detector etector) {
143         // Get the calorimeter data object.
144         //HPSEcal3 ecal = (HPSEcal3) DatabaseConditionsManager.getInstance().getDetectorObject().getSubdetector("Ecal");
145         calorimeterGeometry = (HPSEcal3) DatabaseConditionsManager.getInstance().getDetectorObject().getSubdetector("Ecal");
146         if(calorimeterGeometry == null) {
147             throw new IllegalStateException("Error: Calorimeter geometry data object not defined.");
148         }
149         
150         // Get the calorimeter hit neighbor map.
151         neighborMap = calorimeterGeometry.getNeighborMap();
152         if(neighborMap == null) {
153             throw new IllegalStateException("Error: Calorimeter hit neighbor map is not defined.");
154         }
155     }

157     @Override
158     public void process(EventHeader event) {        
159         // Check the data management driver to determine whether the
160         // input collection is available or not.
161         if(!ReadoutDataManager.checkCollectionStatus(inputCollectionName, localTime + temporalWindow + 4.0)) {
162             return;
163         }
164         
165         // Get the hits that occur during the present clock-cycle, as
166         // well as the hits that occur in the verification window
167         // both before and after the current clock-cycle.
168         // TODO: Simplify this?
169         Collection<CalorimeterHit> seedCandidates = ReadoutDataManager.getData(localTime, localTime + 4.0, inputCollectionName, CalorimeterHit.class);
170         Collection<CalorimeterHit> foreHits = ReadoutDataManager.getData(localTime - temporalWindow, localTime, inputCollectionName, CalorimeterHit.class);
171         Collection<CalorimeterHit> postHits = ReadoutDataManager.getData(localTime + 4.0, localTime + temporalWindow + 4.0, inputCollectionName, CalorimeterHit.class);
172         
173         // Increment the local time.
174         localTime += 4.0;
175         
176         // DEBUG :: Print out the input hits.
177         // TODO: Simplify this?
178         List<CalorimeterHit> allHits = new ArrayList<CalorimeterHit>(seedCandidates.size() + foreHits.size() + postHits.size());
179         allHits.addAll(foreHits);
180         allHits.addAll(seedCandidates);
181         allHits.addAll(postHits);
182         
183         // Store newly created clusters.
184         List<Cluster> gtpClusters = new ArrayList<Cluster>();
185         
186         // Iterate over all seed hit candidates.
187         seedLoop:
188         for(CalorimeterHit seedCandidate : seedCandidates) {
189             // A seed candidate must meet a minimum energy cut to be
190             // considered for clustering.
191             if(seedCandidate.getRawEnergy() < seedEnergyThreshold) {
192                 continue seedLoop;
193             }
194             
195             // Collect other hits that are adjacent to the seed hit
196             // and may be a part of the cluster.
197             List<CalorimeterHit> clusterHits = new ArrayList<CalorimeterHit>();
198             
199             // Iterate over all other hits in the clustering window
200             // and check that the seed conditions are met for the
201             // seed candidate. Note that all hits are properly within
202             // the clustering time window by definition, so the time
203             // condition is not checked explicitly.
204             hitLoop:
205             for(CalorimeterHit hit : allHits) {
206                 // If the hit is not adjacent to the seed hit, it can
207                 // be ignored.
208                 if(!neighborMap.get(seedCandidate.getCellID()).contains(hit.getCellID())) {
209                     continue hitLoop;
210                 }
211                 
212                 // A seed hit must have the highest energy in its
213                 // spatiotemporal window. If it is not, this is not a
214                 // valid seed hit.
215                 if(seedCandidate.getRawEnergy() < hit.getRawEnergy()) {
216                     continue seedLoop;
217                 }
218                 
219                 // Add the hit to the list of cluster hits.
220                 clusterHits.add(hit);
221             }
222             
223             // If no adjacent hit was found that invalidates the seed
224             // condition, then the seed candidate is valid and a
225             // cluster should be formed.
226             gtpClusters.add(createBasicCluster(seedCandidate, clusterHits));
227         }
228         
229         // Pass the clusters to the data management driver.
230         ReadoutDataManager.addData(outputCollectionName, gtpClusters, Cluster.class);
231     }

233     @Override
234     public void startOfData() { 
235         // Define the output LCSim collection parameters.
236         LCIOCollectionFactory.setCollectionName(outputCollectionName);
237         LCIOCollectionFactory.setProductionDriver(this);
238         LCIOCollectionFactory.setFlags(1 << LCIOConstants.CLBIT_HITS);
239         LCIOCollection<Cluster> clusterCollectionParams = LCIOCollectionFactory.produceLCIOCollection(Cluster.class);
240         
241         // Instantiate the GTP cluster collection with the readout
242         // data manager.
243         localTimeDisplacement = temporalWindow + 4.0;
244         addDependency(inputCollectionName);
245         ReadoutDataManager.registerCollection(clusterCollectionParams, false);
246     }

248     @Override
249     protected Collection<TriggeredLCIOData<?>> getOnTriggerData(double triggerTime) {
250         // If clusters are not to be output, return null.
251         if(!isPersistent()) { return null; }
252         
253         // Create a list to store the on-trigger collections. There
254         // are two collections outputs for this driver - the clusters
255         // and the cluster hits. Unlike other drivers, the clusterer
256         // must handle its own output because the manager does not
257         // know that it must also specifically output the hits from
258         // each cluster as well.
259         List<TriggeredLCIOData<?>> collectionsList = new ArrayList<TriggeredLCIOData<?>>(2);
260         
261         // Define the LCIO collection settings for the clusters.
262         LCIOCollectionFactory.setCollectionName(outputCollectionName);
263         LCIOCollectionFactory.setProductionDriver(this);
264         LCIOCollectionFactory.setFlags(1 << LCIOConstants.CLBIT_HITS);
265         LCIOCollection<Cluster> clusterCollectionParams = LCIOCollectionFactory.produceLCIOCollection(Cluster.class);
266         
267         // Define the LCIO collection settings for the cluster hits.
268         int hitFlags = 0;
269         hitFlags += 1 << LCIOConstants.RCHBIT_TIME;
270         hitFlags += 1 << LCIOConstants.RCHBIT_LONG;
271         LCIOCollectionFactory.setCollectionName("EcalClustersGTPSimHits");
272         LCIOCollectionFactory.setProductionDriver(this);
273         LCIOCollectionFactory.setFlags(hitFlags);
274         LCIOCollectionFactory.setReadoutName(calorimeterGeometry.getReadout().getName());
275         LCIOCollection<CalorimeterHit> clusterHitsCollectionParams = LCIOCollectionFactory.produceLCIOCollection(CalorimeterHit.class);
276         
277         // Get the output time range for clusters. This is either the
278         // user defined output range, or the default readout window
279         // that is defined by the readout data manager.
280         double startTime;
281         if(Double.isNaN(getReadoutWindowBefore())) { startTime = triggerTime - ReadoutDataManager.getTriggerOffset(); }
282         else { startTime = triggerTime - getReadoutWindowBefore(); }
283         
284         double endTime;
285         if(Double.isNaN(getReadoutWindowAfter())) { endTime = startTime + ReadoutDataManager.getReadoutWindow(); }
286         else { endTime = triggerTime + getReadoutWindowAfter(); }
287         
288         // Get the cluster data and populate a list of cluster hits.
289         Collection<Cluster> clusters = ReadoutDataManager.getData(startTime, endTime, outputCollectionName, Cluster.class);
290         List<CalorimeterHit> clusterHits = new ArrayList<CalorimeterHit>();
291         for(Cluster cluster : clusters) {
292             clusterHits.addAll(cluster.getCalorimeterHits());
293         }
294         
295         // Create the LCIO on-trigger data lists.
296         TriggeredLCIOData<CalorimeterHit> clusterHitData = new TriggeredLCIOData<CalorimeterHit>(clusterHitsCollectionParams);
297         clusterHitData.getData().addAll(clusterHits);
298         collectionsList.add(clusterHitData);
299         
300         TriggeredLCIOData<Cluster> clusterData = new TriggeredLCIOData<Cluster>(clusterCollectionParams);
301         clusterData.getData().addAll(clusters);
302         collectionsList.add(clusterData);
303         
304         // Return the on-trigger data.
305         return collectionsList;
306     }

318     /**
319      * Creates a new cluster object from a seed hit and list of hits.
320      * @param seedHit - The seed hit of the new cluster.
321      * @param hits - The hits for the new cluster.
322      * @return Returns a {@link org.lcsim.event.Cluster Cluster}
323      * object with the specified properties.
324      */
325     private static final Cluster createBasicCluster(CalorimeterHit seedHit, List<CalorimeterHit> hits) {
326         BaseCluster cluster = new BaseCluster();
327         cluster.setType(ClusterType.GTP.getType());
328         cluster.addHit(seedHit);
329         cluster.setPosition(seedHit.getDetectorElement().getGeometry().getPosition().v());
330         cluster.setNeedsPropertyCalculation(false);
331         cluster.addHits(hits);
332         return cluster;
333     }

335     /**
336      * Sets the size of the hit verification temporal window. Note
337      * that this defines the size of the window in one direction, so
338      * the full time window will be <code>(2 * clusterWindow)+
339      * 1</code> clock-cycles in length. (i.e., it will be a length of
340      * <code>clusterWindow</code> before the seed hit, a length of
341      * <code>clusterWindow</code> after the seed hit, plus the cycle
342      * that includes the seed hit.) Time length is in clock-cycles.
343      * @param value - The number of clock-cycles around the hit in
344      * one direction.
345      */
346     public void setClusterWindow(int value) {
347         temporalWindow = value * 4;
348     }
349     
350     /**
351      * Sets the minimum seed energy needed for a hit to be considered
352      * for forming a cluster. This is the seed energy lower bound
353      * trigger cut and is in units of GeV.
354      * @param value - The minimum cluster seed energy in GeV.
355      */
356     public void setSeedEnergyThreshold(double value) {
357         seedEnergyThreshold = value;
358     }
