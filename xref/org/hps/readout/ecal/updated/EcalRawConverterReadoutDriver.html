17  /**
18   * <code>EcalRawConverterReadoutDriver</code> is an implementation of
19   * {@link org.hps.readout.RawConverterReadoutDriver
20   * RawConverterReadoutDriver} for the calorimeter subdetector.
21   * 
22   * @see org.hps.readout.RawConverterReadoutDriver
23   */
24  public class EcalRawConverterReadoutDriver extends RawConverterReadoutDriver {    
25      /**
26       * The converter object responsible for processing raw hits into
27       * proper {@link org.lcsim.event.CalorimeterHit CalorimeterHit}
28       * objects.
29       */
30      private EcalReadoutMode3RawConverter converter = new EcalReadoutMode3RawConverter();
31      
32      /**
33       * Cached copy of the calorimeter conditions. All calorimeter
34       * conditions should be called from here, rather than by directly
35       * accessing the database manager.
36       */
37      private EcalConditions ecalConditions = null;
38      
39      /**
40       * Instantiates the driver with the correct default parameters.
41       */
42      public EcalRawConverterReadoutDriver() {
43          super("EcalRawHits", "EcalCorrectedHits");
44          setSkipBadChannels(true);
45      }
46      
47      /**
48       * Sets whether or not the DAQ configuration is applied into the driver
49       * the EvIO data stream or whether to read the configuration from data files.
50       * 
51       * @param state - <code>true</code> indicates that the DAQ configuration is
52       * applied into the readout system, and <code>false</code> that it
53       * is not applied into the readout system.
54       */
55      public void setDaqConfigurationAppliedintoReadout(boolean state) {
56          // Track changes in the DAQ configuration.
57          if (state) {
58              ConfigurationManager2019.addActionListener(new ActionListener() {
59                  @Override
60                  public void actionPerformed(ActionEvent e) {
61                      // Get the DAQ configuration.
62                      DAQConfig2019 daq = ConfigurationManager2019.getInstance();
63  
64                      // Load the DAQ settings from the configuration manager.
65                      getConverter().setNumberSamplesAfter(daq.getEcalFADCConfig().getNSA());
66                      getConverter().setNumberSamplesBefore(daq.getEcalFADCConfig().getNSB());
67                      
68                      // Get the FADC configuration.
69                      getConverter().setFADCConfigEcal2019(daq.getEcalFADCConfig());
70                  }
71              });
72          }  
73      }    
74      
75      /**
76       * Indicates whether or not data from channels flagged as "bad"
77       * in the conditions system should be ignored. <code>true</code>
78       * indicates that they should be ignored, and <code>false</code>
79       * that they should not.
80       * @param apply - <code>true</code> indicates that "bad" channels
81       * will be ignored and <code>false</code> that they will not.
82       */
83      @Override
84      public void setSkipBadChannels(boolean state) {
85          super.skipBadChannels = state;
86      }
87      
88      @Override
89      protected AbstractMode3RawConverter getConverter() {
90          return converter;
91      }
92      
93      @Override
94      protected String getSubdetectorReadoutName(Detector detector) {
95          HPSEcal3 calorimeterGeometry = (HPSEcal3) detector.getSubdetector("Ecal");
96          return calorimeterGeometry.getReadout().getName();
97      }
98      
99      @Override
100     protected boolean isBadChannel(long channelID) {
101         return findChannel(channelID).isBadChannel();
102     }
103     
104     @Override
105     protected void updateDetectorDependentParameters(Detector detector) {
106         ecalConditions = DatabaseConditionsManager.getInstance().getEcalConditions();
107     }
108     
109     /**
110      * Gets the channel parameters for a given channel ID.
111      * @param cellID - The <code>long</code> ID value that represents
112      * the channel. This is typically acquired from the method {@link
113      * org.lcsim.event.CalorimeterHit#getCellID() getCellID()} in a
114      * {@link org.lcsim.event.CalorimeterHit CalorimeterHit} object.
115      * @return Returns the channel parameters for the channel as an
116      * {@link org.hps.conditions.ecal.EcalChannelConstants
117      * EcalChannelConstants} object.
118      */
119     private EcalChannelConstants findChannel(long cellID) {
120         return ecalConditions.getChannelConstants(ecalConditions.getChannelCollection().findGeometric(cellID));
121     }    
122 }
