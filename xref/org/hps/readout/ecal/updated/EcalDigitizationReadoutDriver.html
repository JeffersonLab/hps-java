19  /**
20   * Class <code>EcalReadoutDriver</code> is an implementation of the
21   * {@link org.hps.readout.ecal.updated.DigitizationReadoutDriver
22   * DigitizationReadoutDriver} for a subdetector of type {@link
23   * org.lcsim.geometry.subdetector.HPSEcal3 HPSEcal3}. It handles all
24   * of the calorimeter-specific functions needed by the superclass.
25   */
26  public class EcalDigitizationReadoutDriver extends DigitizationReadoutDriver<HPSEcal3> {    
27      // The DAQ configuration manager for FADC parameters.
28      private FADCConfigEcal2019 config = new FADCConfigEcal2019();
29      private boolean configStat = false; // Indicates if DAQ configuration is loaded
30      
31      // The number of nanoseconds in a clock-cycle (sample).
32      private static final int nsPerSample = 4;
33      
34      /** Stores the conditions for this subdetector. */
35      private EcalConditions ecalConditions = null;
36      
37      /** Stores the channel collection for this subdetector. */
38      private EcalChannelCollection geoMap = new EcalChannelCollection();
39      
40      public EcalDigitizationReadoutDriver() {
41          // Set the default values for each subdetector-dependent
42          // parameter.
43          setGeometryName("Ecal");
44          
45          setInputHitCollectionName("EcalHits");
46          setOutputHitCollectionName("EcalRawHits");
47          setTruthRelationsCollectionName("EcalTruthRelations");
48          setTriggerPathTruthRelationsCollectionName("TriggerPathTruthRelations");
49          setReadoutHitCollectionName("EcalReadoutHits");
50          
51          setPhotoelectronsPerMeV(32.8);
52          setPulseTimeParameter(9.6);
53      }    
54      
55      /**
56       * Sets whether or not the DAQ configuration is applied into the driver
57       * the EvIO data stream or whether to read the configuration from data files.
58       * 
59       * @param state - <code>true</code> indicates that the DAQ configuration is
60       * applied into the readout system, and <code>false</code> that it
61       * is not applied into the readout system.
62       */
63      public void setDaqConfigurationAppliedintoReadout(boolean state) {
64          // If the DAQ configuration should be read, attach a listener
65          // to track when it updates.               
66          if (state) {
67              ConfigurationManager2019.addActionListener(new ActionListener() {
68                  @Override
69                  public void actionPerformed(ActionEvent e) {
70                      // Get the DAQ configuration.
71                      DAQConfig2019 daq = ConfigurationManager2019.getInstance();
72  
73                      // Load the DAQ settings from the configuration manager.
74                      numSamplesAfter = daq.getEcalFADCConfig().getNSA() / nsPerSample;
75                      numSamplesBefore = daq.getEcalFADCConfig().getNSB() / nsPerSample;
76                      readoutWindow = daq.getEcalFADCConfig().getWindowWidth() / nsPerSample;
77                      
78                      // Get the FADC configuration.
79                      config = daq.getEcalFADCConfig();
80                      configStat = true;
81                      integrationThreshold = config.getThreshold((int)10);
82                  }
83              });
84          }
85      }
86      
87      @Override
88      public void detectorChanged(Detector detector) {        
89          // Get a copy of the calorimeter conditions for the detector.
90          ecalConditions = DatabaseConditionsManager.getInstance().getEcalConditions();
91          
92          // Store the calorimeter conditions table for converting between
93          // geometric IDs and channel objects.
94          geoMap = DatabaseConditionsManager.getInstance().getCachedConditions(EcalChannelCollection.class, "ecal_channels").getCachedData();
95          
96          // Run the superclass method.
97          super.detectorChanged(detector);
98      }    
99      
100     @Override
101     protected Set<Long> getChannelIDs() {
102         return getSubdetector().getNeighborMap().keySet();
103     }
104     
105     @Override
106     protected double getGainConditions(long cellID) {        
107         return findChannel(cellID).getGain().getGain();
108     }
109     
110     @Override
111     protected double getNoiseConditions(long cellID) {
112         return findChannel(cellID).getCalibration().getNoise();
113     }
114     
115     @Override
116     protected double getPedestalConditions(long cellID) { 
117         return findChannel(cellID).getCalibration().getPedestal();
118     }   
119     
120     @Override
121     protected double getTimeShiftConditions(long cellID) {
122         return findChannel(cellID).getTimeShift().getTimeShift();
123     }    
124     
125     @Override
126     protected int getTimestampFlag() {
127         return ReadoutTimestamp.SYSTEM_ECAL;
128     }
129     
130     /**
131      * Gets the channel parameters for a given channel ID.
132      * @param cellID - The <code>long</code> ID value that represents
133      * the channel. This is typically acquired from the method {@link
134      * org.lcsim.event.CalorimeterHit#getCellID() getCellID()} in a
135      * {@link org.lcsim.event.CalorimeterHit CalorimeterHit} object.
136      * @return Returns the channel parameters for the channel as an
137      * {@link org.hps.conditions.ecal.EcalChannelConstants
138      * EcalChannelConstants} object.
139      */
140     private EcalChannelConstants findChannel(long cellID) {
141         return ecalConditions.getChannelConstants(ecalConditions.getChannelCollection().findGeometric(cellID));
142     }
143 }
