22  /**
23   * Class <code>NeutralPionTriggerDriver</code> simulates a pi-0 trigger.
24   * It executes four cuts, two of which are single cluster cuts and two
25   * of which are cluster pair cuts. The single cluster cuts are on the
26   * total energy of the cluster and the energy of the seed hit of the
27   * cluster. The first cluster pair cut is on the sum of the energies of
28   * both clusters. The second calculates the invariant mass of the
29   * particle that produced the clusters, assuming that clusters were
30   * created by an electron/positron pair. The pair is then cut if the
31   * invariant mass is outside the expected range for a neutral pion decay.
32   * <br/><br/>
33   * All incoming clusters are passed through the single cluster cuts and
34   * those which survive are added to a list of clusters for their event
35   * and stored in a buffer. The buffer stores a number of event lists
36   * equal to coincidence window parameter. This limits the time frame
37   * in which clusters can be used for a trigger. Of the clusters stored
38   * in the cluster buffer, the two with the highest energies are chosen
39   * and the cluster pair cuts are applied to them. If the highest energy
40   * pair survives this process, the event triggers. If it does not,
41   * there is no trigger for the event.
42   * <br/><br/>
43   * All thresholds can be set through a steering file, along with the
44   * coincidence window. The driver also supports a verbose mode where
45   * it will output more details with every event to help with diagnostics.
46   */
47  public class NeutralPionTriggerDriver extends TriggerDriver {

76      public void process(EventHeader event) {
77          // Generate a temporary list to store the good clusters
78          // in before they are added to the buffer.
79          List<Cluster> tempList = new ArrayList<Cluster>();
80          
81          // If the current event has a cluster collection, get it.
82          if(event.hasCollection(Cluster.class, clusterCollectionName)) {
83              // VERBOSE :: Note that a cluster collection exists for
84              //            this event.
85              if(verbose) { System.out.println("Cluster collection is present for event."); }
86              
87              // Get the cluster list from the event.
88              List<Cluster> eventList = event.get(Cluster.class, clusterCollectionName);
89              
90              // VERBOSE :: Output the number of extant clusters.
91              if(verbose) { System.out.printf("%d clusters in event.%n", eventList.size()); }
92              
93              // Add the clusters from the event into the cluster list
94              // if they pass the minimum total cluster energy and seed
95              // energy thresholds.
96              for(Cluster cluster : eventList) {
97                  // Increment the clusters processed count.
98                  allClusters++;
99                  
100                 // Plot the seed energy / cluster energy histogram.
101                 seedPercent.fill(cluster.getCalorimeterHits().get(0).getCorrectedEnergy() / cluster.getEnergy(), 1);
102                 
103                 // Get the cluster position indices.
104                 int ix = cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
105                 int iy = cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("iy");
106                 
107                 // VERBOSE :: Output the current cluster's properties.
108                 if(verbose) {
109                     System.out.printf("\tTesting cluster at (%d, %d) with total energy %f and seed energy %f.%n",
110                             ix, iy, cluster.getCalorimeterHits().get(0).getCorrectedEnergy(), cluster.getEnergy());
111                 }
112                 
113                 // Add the clusters to the uncut histograms.
114                 clusterHitCount.fill(cluster.getCalorimeterHits().size());
115                 clusterTotalEnergy.fill(cluster.getEnergy());
116                 clusterSeedEnergy.fill(cluster.getCalorimeterHits().get(0).getCorrectedEnergy());
117                 clusterDistribution.fill(ix, iy, 1);
118                 
119                 // VERBOSE :: Output the single cluster trigger thresholds.
120                 if(verbose) {
121                     System.out.printf("\tCluster seed energy threshold  :: [%f, %f]%n", clusterSeedEnergyThresholdLow, clusterSeedEnergyThresholdHigh);
122                     System.out.printf("\tCluster total energy threshold :: %f%n%n", clusterTotalEnergyThresholdLow);
123                 }
124                 
125                 // Perform the single cluster cuts.
126                 boolean totalEnergyCut = clusterTotalEnergyCut(cluster);
127                 boolean seedEnergyCut = clusterSeedEnergyCut(cluster);
128                 boolean hitCountCut = clusterHitCountCut(cluster);
129                 boolean edgeCrystalCut = isEdgeCluster(cluster);
130                 
131                 // Increment the single cut counts.
132                 if(seedEnergyCut) {
133                     clusterSeedEnergyCount++;
134                     if(hitCountCut) {
135                         clusterHitCountCount++;
136                         if(rejectEdgeCrystals && edgeCrystalCut) {
137                             clusterEdgeCount++;
138                         }
139                     }
140                 }
141                 
142                 // VERBOSE :: Note whether the cluster passed the single
143                 //            cluster cuts.
144                 if(verbose) {
145                     System.out.printf("\tPassed seed energy cut    :: %b%n", seedEnergyCut);
146                     System.out.printf("\tPassed cluster energy cut :: %b%n%n", totalEnergyCut);
147                     System.out.printf("\tPassed hit count cut :: %b%n%n", hitCountCut);
148                     System.out.printf("\tIs an edge cluster :: %b%n%n", edgeCrystalCut);
149                 }
150                 
151                 // Determine whether the cluster passes all the single
152                 // cluster cuts.
153                 boolean passedCuts = false;
154                 
155                 // If edge crystals should be not be used for triggering,
156                 // require that the cluster not be centered in an edge
157                 // crystal.
158                 if(rejectEdgeCrystals) {
159                     if(totalEnergyCut && seedEnergyCut && hitCountCut && !edgeCrystalCut) {
160                         passedCuts = true;
161                     }
162                 }
163                 
164                 // Otherwise, it just needs to pass the standard trigger
165                 // cuts regardless of where it is located.
166                 else {
167                     if(totalEnergyCut && seedEnergyCut && hitCountCut) {
168                         passedCuts = true;
169                     }
170                 }
171                 
172                 // If both pass, add the cluster to the list.
173                 if(passedCuts) {
174                     // Add the cluster to the cluster list.
175                     tempList.add(cluster);
176                     
177                     // Add the cluster information to the single cut histograms.
178                     pClusterHitCount.fill(cluster.getCalorimeterHits().size());
179                     pClusterTotalEnergy.fill(cluster.getEnergy());
180                     pClusterSeedEnergy.fill(cluster.getCalorimeterHits().get(0).getCorrectedEnergy());
181                     pClusterDistribution.fill(ix, iy, 1);
182                 }
183             }
184             
185             // Remove the oldest cluster buffer element and add the new
186             // cluster list to the buffer.
187             clusterBuffer.removeFirst();
188             clusterBuffer.addLast(tempList);
189         }
190         
191         // Otherwise, clear the cluster list.
192         else {
193             // VERBOSE :: Note that the event has no clusters.
194             if(verbose) { System.out.println("No cluster collection is present for event.\n"); }
195         }
196         
197         // Reset the highest energy pair to null.
198         clusterTriplet[0] = null;
199         clusterTriplet[1] = null;
200         clusterTriplet[2] = null;
201         
202         // Loop over all of the cluster lists in the cluster buffer.
203         double[] energy = { 0.0, 0.0, 0.0 };
204         for(List<Cluster> bufferList : clusterBuffer) {
205             // Loop over all of the clusters in each buffer list.
206             for(Cluster cluster : bufferList) {
207                 // If the new cluster is higher energy than the first
208                 // slot cluster, move the subsequent clusters down and
209                 // insert the new one.
210                 if(cluster.getEnergy() > energy[0]) {
211                     clusterTriplet[2] = clusterTriplet[1];
212                     clusterTriplet[1] = clusterTriplet[0];
213                     clusterTriplet[0] = cluster;
214                     energy[2] = energy[1];
215                     energy[1] = energy[0];
216                     energy[0] = cluster.getEnergy();
217                 }
218                 
219                 // Otherwise, if the new cluster has more energy than
220                 // the second slot, it goes there and the second does
221                 // to the third.
222                 else if(cluster.getEnergy() > energy[1]) {
223                     clusterTriplet[2] = clusterTriplet[1];
224                     clusterTriplet[1] = cluster;
225                     energy[2] = energy[1];
226                     energy[1] = cluster.getEnergy();
227                 }
228                 
229                 // If the new cluster has more energy than the third
230                 // cluster, it just replaces it.
231                 else if(cluster.getEnergy() > energy[2]) {
232                     clusterTriplet[2] = cluster;
233                     energy[2] = cluster.getEnergy();
234                 }
235             }
236         }
237         
238         // The highest energy pair is the same as the first two slots
239         // of the highest energy triplet.
240         clusterPair[0] = clusterTriplet[0];
241         clusterPair[1] = clusterTriplet[1];
242         
243         // Run the superclass event process.
244         super.process(event);
245     }

288     protected boolean triggerDecision(EventHeader event) {
289         // If the active cluster pair has a null value, then there were
290         // fewer than two clusters in the buffer and we can not trigger.
291         if(!useClusterTriplet && (clusterPair[0] == null || clusterPair[1] == null)) {
292             // VERBOSE :: Note that triggering failed due to insufficient
293             // clusters. in the cluster buffer.
294             if(verbose) { System.out.println("Inufficient clusters in buffer -- no trigger."); }
295             
296             // Return false; we can not trigger without two clusters.
297             return false;
298         }
299         
300         // If the active cluster triplet has a null value, then there
301         // were fewer than three clusters in the buffer and we can not
302         // trigger.
303         if(useClusterTriplet && (clusterTriplet[0] == null || clusterTriplet[1] == null || clusterTriplet[2] == null)) {
304             // VERBOSE :: Note that triggering failed due to insufficient
305             // clusters. in the cluster buffer.
306             if(verbose) { System.out.println("Inufficient clusters in buffer -- no trigger."); }
307             
308             // Return false; we can not trigger without three clusters.
309             return false;
310         }
311         
312         // Increment the number of pairs considered.
313         allPairs++;
314         
315         // Get the cluster position indices.
316         int[] ix = { clusterPair[0].getCalorimeterHits().get(0).getIdentifierFieldValue("ix"), clusterPair[1].getCalorimeterHits().get(0).getIdentifierFieldValue("ix") };
317         int[] iy = { clusterPair[0].getCalorimeterHits().get(0).getIdentifierFieldValue("iy"), clusterPair[1].getCalorimeterHits().get(0).getIdentifierFieldValue("iy") };
318         
319         // VERBOSE :: Output the clusters selected for triggering.
320         if(verbose) {
321             System.out.printf("\tTesting first cluster at (%d, %d) with total energy %f and seed energy %f.%n",
322                     ix[0], iy[0], clusterPair[0].getCalorimeterHits().get(0).getCorrectedEnergy(), clusterPair[0].getEnergy());
323             System.out.printf("\tTesting second cluster at (%d, %d) with total energy %f and seed energy %f.%n",
324                     ix[1], iy[1], clusterPair[1].getCalorimeterHits().get(0).getCorrectedEnergy(), clusterPair[1].getEnergy());
325             if(useClusterTriplet) {
326                 System.out.printf("\tTesting third cluster at (%d, %d) with total energy %f and seed energy %f.%n",
327                         ix[1], iy[1], clusterTriplet[2].getCalorimeterHits().get(0).getCorrectedEnergy(), clusterTriplet[2].getEnergy());
328             }
329         }
330         
331         if(!useClusterTriplet) {
332             // Fill the uncut histograms.
333             pairEnergySum.fill(getEnergySumValue(clusterPair));
334             invariantMass.fill(getInvariantMassValue(clusterPair));
335             
336             // VERBOSE :: Output the cluster pair trigger thresholds.
337             if(verbose) {
338                 System.out.printf("\tCluster pair energy sum threshold     :: %f%n", pairEnergySumThresholdLow);
339                 System.out.printf("\tHypothetical invariant mass threshold :: [%f, %f]%n%n", invariantMassThresholdLow, invariantMassThresholdHigh);
340             }
341             
342             // Perform the cluster pair checks.
343             boolean energySumCut = pairEnergySumCut(clusterPair);
344             boolean invariantMassCut = pairInvariantMassCut(clusterPair);
345             
346             // Increment the pair cut counts.
347             if(energySumCut) {
348                 pairEnergySumCount++;
349                 if(invariantMassCut) {
350                     pairInvariantMassCount++;
351                 }
352             }
353             
354             // VERBOSE :: Note the outcome of the trigger cuts.
355             if(verbose) {
356                 System.out.printf("\tPassed energy sum cut     :: %b%n", energySumCut);
357                 System.out.printf("\tPassed invariant mass cut :: %b%n%n", invariantMassCut);
358             }
359             
360             // If the pair passes both cuts, we have a trigger.
361             if(energySumCut && invariantMassCut) {
362                 // Fill the cut histograms.
363                 pPairEnergySum.fill(getEnergySumValue(clusterPair));
364                 pInvariantMass.fill(getInvariantMassValue(clusterPair));
365                 
366                 // Fill the all cuts histograms.
367                 aClusterHitCount.fill(clusterPair[0].getCalorimeterHits().size());
368                 aClusterHitCount.fill(clusterPair[1].getCalorimeterHits().size());
369                 aClusterTotalEnergy.fill(clusterPair[0].getEnergy());
370                 aClusterTotalEnergy.fill(clusterPair[1].getEnergy());
371                 aClusterSeedEnergy.fill(clusterPair[0].getCalorimeterHits().get(0).getCorrectedEnergy());
372                 aClusterSeedEnergy.fill(clusterPair[1].getCalorimeterHits().get(0).getCorrectedEnergy());
373                 aClusterDistribution.fill(ix[0], iy[0], 1);
374                 aClusterDistribution.fill(ix[1], iy[1], 1);
375                 
376                 // VERBOSE :: Note that the event has triggered.
377                 if(verbose) { System.out.println("Event triggers!\n\n"); }
378                 
379                 // Increment the number of triggers.
380                 triggers++;
381                 
382                 // Return the trigger.
383                 return true;
384             }
385         }
386         
387         // If we are using a cluster triplet, apply the cluster triplet
388         // cuts.
389         else {
390             // Perform the cluster triplet checks.
391             boolean energySumCut = tripletEnergySumCut(clusterTriplet);
392             boolean horizontalCut = tripletHorizontalCut(clusterTriplet);
393             boolean energySpatialCut = tripletTotalEnergyCut(clusterTriplet);
394             
395             // Fill the all cuts histograms.
396             aClusterHitCount.fill(clusterPair[0].getCalorimeterHits().size());
397             aClusterHitCount.fill(clusterPair[1].getCalorimeterHits().size());
398             aClusterTotalEnergy.fill(clusterPair[0].getEnergy());
399             aClusterTotalEnergy.fill(clusterPair[1].getEnergy());
400             aClusterSeedEnergy.fill(clusterPair[0].getCalorimeterHits().get(0).getCorrectedEnergy());
401             aClusterSeedEnergy.fill(clusterPair[1].getCalorimeterHits().get(0).getCorrectedEnergy());
402             aClusterDistribution.fill(ix[0], iy[0], 1);
403             aClusterDistribution.fill(ix[1], iy[1], 1);
404             
405             if(energySumCut && horizontalCut && energySpatialCut) {
406                 return true;
407             }
408         }
409         
410         // VERBOSE :: Note that the event has failed to trigger.
411         if(verbose) { System.out.println("No trigger.\n\n"); }
412         
413         // If one or more of the pair cuts failed, the we do not trigger.
414         return false;
415     }

421     /**
422      * Checks whether the cluster passes the threshold for minimum
423      * number of component hits.
424      * @param cluster - The cluster to check.
425      * @return Returns <code>true</code> if the cluster passes and <code>
426      * false</code> if it does not.
427      */
428     private boolean clusterHitCountCut(Cluster cluster) {
429         return cluster.getCalorimeterHits().size() >= clusterHitCountThreshold;
430     }
431     
432     /**
433      * Checks whether the cluster falls within the allowed range for
434      * the seed hit energy cut.
435      * @param cluster - The cluster to check.
436      * @return Returns <code>true</code> if the cluster passes and <code>
437      * false</code> if it does not.
438      */
439     private boolean clusterSeedEnergyCut(Cluster cluster) {
440         // Get the seed energy value.
441         double seedEnergy = cluster.getCalorimeterHits().get(0).getCorrectedEnergy();
442         
443         // Perform the seed energy cut.
444         return seedEnergy >= clusterSeedEnergyThresholdLow && seedEnergy <= clusterSeedEnergyThresholdHigh;
445     }
446     
447     /**
448      * Checks whether the cluster passes the threshold for minimum
449      * total cluster energy.
450      * @param cluster - The cluster to check.
451      * @return Returns <code>true</code> if the cluster passes and <code>
452      * false</code> if it does not.
453      */
454     private boolean clusterTotalEnergyCut(Cluster cluster) {
455         // Get the cluster energy.
456         double clusterEnergy = cluster.getEnergy();
457         
458         // Perform the cut.
459         return clusterEnergy >= clusterTotalEnergyThresholdLow && clusterEnergy <= clusterTotalEnergyThresholdHigh;
460     }

462     /**
463      * Calculates the value used in the pair energy sum cut from a pair
464      * of two clusters.
465      * @param clusterPair - The cluster pair from which to derive the
466      * cut value.
467      * @return Returns the cut value as a <code>double</code>.
468      */
469     private static double getEnergySumValue(Cluster[] clusterGroup) {
470         // Track the sum.
471         double energySum = 0.0;
472         
473         // Add the energies of all clusters in the array.
474         for(Cluster cluster : clusterGroup) { energySum += cluster.getEnergy(); }
475         
476         // Return the sum.
477         return energySum;
478     }
479     
480     /**
481      * Calculates the value used in the invariant mass cut from a pair
482      * of two clusters.
483      * @param clusterPair - The cluster pair from which to derive the
484      * cut value.
485      * @return Returns the cut value as a <code>double</code>.
486      */
487     private double getInvariantMassValue(Cluster[] clusterPair) {
488         // Store the x/y positions for the seeds.
489         double x[] = new double[2];
490         double y[] = new double[2];
491         
492         // Get the seed hits.
493         CalorimeterHit[] seed = { clusterPair[0].getCalorimeterHits().get(0), clusterPair[1].getCalorimeterHits().get(0) };
494         
495         // Set the positions for each seed.
496         for(int index = 0; index < seed.length; index++) {
497             // Get the seed position array stored in the position map.
498             Double[] seedPos = seedPosMap.get(clusterPair[index].getCalorimeterHits().get(0));
499             
500             // If there is a position array for the seed, use it.
501             if(seedPos != null) {
502                 x[index] = seedPos[0];
503                 y[index] = seedPos[1];
504             }
505             
506             // Otherwise, calculate the position at the crystal face.
507             else {
508                 // Get the position and store it in a double array.
509                 IGeometryInfo geom = clusterPair[index].getCalorimeterHits().get(0).getDetectorElement().getGeometry();
510                 double[] pos = geom.transformLocalToGlobal(VecOp.add(geom.transformGlobalToLocal(geom.getPosition()),
511                         (Hep3Vector) new BasicHep3Vector(0, 0, -1 * ((Trd) geom.getLogicalVolume().getSolid()).getZHalfLength()))).v();
512                 
513                 // Set the seed location.
514                 x[index] = pos[0];
515                 y[index] = pos[1];
516                 
517                 // Store the seed location for future use.
518                 Double[] positionVec = { pos[0], pos[1], pos[2] };
519                 seedPosMap.put(clusterPair[index].getCalorimeterHits().get(0), positionVec);
520             }
521         }
522         
523         // Get the cluster energy for each seed.
524         double[] e = { clusterPair[0].getEnergy(), clusterPair[1].getEnergy() };
525         
526         //Return the invariant mass.
527         return (e[0] * e[1] * (Math.pow(x[0] - x[1], 2) + Math.pow(y[0] - y[1], 2)) / D2);
528     }

530     /**
531      * Indicates whether a cluster has a seed hit located on the edge
532      * of the calorimeter or not.
533      * 
534      * @param cluster - The cluster to check.
535      * @return Returns <code>true</code> if the cluster seed is on the
536      * edge of the calorimeter and <code>false</code> otherwise.
537      */
538     private static boolean isEdgeCluster(Cluster cluster) {
539         // Get the x- and y-indices of the cluster seed hit.
540         int ix = cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
541         int iy = cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("iy");
542         
543         // Track whether the cluster is an edge cluster or not.
544         boolean edge = false;
545         
546         // Get the absolute values of the coordinates.
547         int aix = Math.abs(ix);
548         int aiy = Math.abs(iy);
549         
550         // Check if this an outer edge crystal.
551         if(aix == 23 || aiy == 5) { edge = true; }
552         
553         // Check if this along the central beam gap.
554         if(aiy == 1) { edge = true; }
555         
556         // Check if this is around the beam gap.
557         if(aiy == 2 && (ix >= -11 && ix <= -1)) { edge = true; }
558         
559         // Otherwise, this is not an edge crystal.
560         return edge;
561     }

563     /**
564      * Checks whether the cluster pair passes the falls within the
565      * allowed range for the piar energy sum cut.
566      * @param clusterPair - An array of size two containing the cluster
567      * pair to check.
568      * @return Returns <code>true</code> if the clusters pass and <code>
569      * false</code> if they does not.
570      */
571     private boolean pairEnergySumCut(Cluster[] clusterPair) {
572         // Get the energy sum value.
573         double energySum = getEnergySumValue(clusterPair);
574         
575         // Otherwise, get the energy sum and compare it to the threshold.
576         return energySum >= pairEnergySumThresholdLow && energySum <= pairEnergySumThresholdHigh;
577     }

579     /**
580      * Checks whether the cluster pair passes the threshold for the
581      * invariant mass check.
582      * @param clusterPair - An array of size two containing the cluster
583      * pair to check.
584      * @return Returns <code>true</code> if the clusters pass and <code>
585      * false</code> if they does not.
586      */
587     private boolean pairInvariantMassCut(Cluster[] clusterPair) {
588         // Calculate the invariant mass.
589         double myy2 = getInvariantMassValue(clusterPair);
590         
591         // Perform the cut.
592         return ( (myy2 >= invariantMassThresholdLow) && (myy2 <= invariantMassThresholdHigh));
593     }
594     
595     /**
596      * Checks whether the cluster pair passes the threshold for the
597      * minimum pair energy sum check.
598      * @param clusterTriplet - An array of size three containing the
599      * cluster triplet to check.
600      * @return Returns <code>true</code> if the clusters pass and <code>
601      * false</code> if they does not.
602      */
603     private boolean tripletEnergySumCut(Cluster[] clusterTriplet) {
604         return (getEnergySumValue(clusterTriplet) >= tripletEnergySumThreshold);
605     }
606     
607     /**
608      * Checks that there is at least one cluster is located on the right
609      * side and at least one cluster on the left side of the calorimeter.
610      * @param clusterTriplet - An array of size three containing the
611      * cluster triplet to check.
612      * @return Returns <code>true</code> if the clusters pass and <code>
613      * false</code> if they does not.
614      */
615     private static boolean tripletHorizontalCut(Cluster[] clusterTriplet) {
616         // Track whether a cluster has occurred on each horizontal side
617         // of the calorimeter.
618         boolean leftCluster = false;
619         boolean rightCluster = false;
620         
621         // Sort through the cluster triplet and check where they occur.
622         for(Cluster cluster : clusterTriplet) {
623             int ix = cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
624             if(ix < 0) { leftCluster = true; }
625             if(ix > 0) { rightCluster = true; }
626         }
627         
628         // If a cluster fell on both sides, it passes.
629         if(leftCluster && rightCluster) { return true; }
630         else { return false; }
631     }
632     
633     private boolean tripletTotalEnergyCut(Cluster[] clusterTriplet) {
634         // Check to see if each cluster passes the check.
635         for(Cluster cluster1 : clusterTriplet) {
636             for(Cluster cluster2 : clusterTriplet) {
637                 // The cluster pair must be two different clusters.
638                 if(cluster1 == cluster2) { continue; }
639                 
640                 // Check to see if the clusters are over threshold.
641                 boolean over1 = cluster1.getEnergy() >= tripletTotalEnergyThreshold;
642                 boolean over2 = cluster1.getEnergy() >= tripletTotalEnergyThreshold;
643                 
644                 // If both the clusters are over threshold, check that
645                 // they are sufficiently far apart.
646                 if(over1 && over2) {
647                     // Get the x and y coordinates of the clusters.
648                     double x[] = { cluster1.getPosition()[0], cluster2.getPosition()[0] };
649                     double y[] = { cluster1.getPosition()[1], cluster2.getPosition()[1] };
650                     
651                     // Calculate the distance between the clusters.
652                     double dr = Math.sqrt(x[0] * x[0] + y[0] * y[0]);
653                     
654                     // Run the check.
655                     if(dr >= tripletPairSeparationThreshold) { return true; }
656                 }
657             }
658         }
659         
660         // If none of the cluster pairs pass all the checks, the
661         // triplet fails.
662         return false;
663     }
