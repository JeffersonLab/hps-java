221     @Override
222     public void detectorChanged(Detector detector) {
223         // TODO: What does this "SimTrackerHitReadoutDriver" do?
224         String[] readouts = { readout };
225         readoutDriver.setCollections(readouts);
226 
227         // Get the collection of all silicon sensors from the SVT.
228         sensors = detector.getSubdetector(SVT_SUBDETECTOR_NAME).getDetectorElement().findDescendants(HpsSiSensor.class);
229         
230         // If pile-up simulation is disabled, instantiate all
231         // possible processing queues. For the pile-up simulation,
232         // these are generated as needed.
233         if(!noPileup) {
234             for(HpsSiSensor sensor : sensors) {
235                 @SuppressWarnings("unchecked")
236                 PriorityQueue<StripHit>[] hitQueues = new PriorityQueue[sensor.getNumberOfChannels()];
237                 hitMap.put(sensor, hitQueues);
238             }
239         }
240         
241         // Load timing conditions from the conditions database, if
242         // this is requested.
243         if(useTimingConditions) {
244             SvtTimingConstants timingConstants = DatabaseConditionsManager.getInstance().getCachedConditions(SvtTimingConstants.SvtTimingConstantsCollection.class, "svt_timing_constants").getCachedData().get(0);
245             readoutOffset = 4 * (timingConstants.getOffsetPhase() + 3);
246             readoutLatency = 248.0 + timingConstants.getOffsetTime();
247         }
248     }

250     @Override
251     public void process(EventHeader event) {
252 
253         super.process(event);
254         
255         // Generate the truth hits.
256         List<StripHit> stripHits = doSiSimulation();
257         
258         // If pile-up is to be simulated, process the hits into hit
259         // queues. These hit queues are not integrated at this stage,
260         // and are instead only handled at readout, as they are not
261         // used downstream in the simulation.
262         if(!noPileup) {
263             //verboseWriter.write("Starting pile-up simulation...");
264             
265             // Process each of the truth hits.
266             for(StripHit stripHit : stripHits) {
267                 // Get the sensor and channel for the truth hit.
268                 SiSensor sensor = stripHit.sensor;
269                 int channel = stripHit.channel;
270                
271                 // This is required to account for the fact that the channel 
272                 // number of SiStriplets is shifted by 1. If left uncorrected, 
273                 // this leads to an 'out of bounds' error when trying to access
274                 // the queue associated with channel 510.
275                 // TODO: A cleaner solution is to add a CHANNEL_OFFSET parameter 
276                 //       to all instances of readout electrodes and use that to
277                 //       shift the channel. For SiStrips, the offset would be
278                 //       0.
279                 if (sensor.getReadoutElectrodes(ChargeCarrier.HOLE) instanceof SiStriplets)
280                     channel -= 1; 
281                 
282                 // Queue the hit in the processing queue appropriate
283                 // to its sensor and channel.
284                 PriorityQueue<StripHit>[] hitQueues = hitMap.get(sensor);
285                 if(hitQueues[channel] == null) {
286                     hitQueues[channel] = new PriorityQueue<StripHit>();
287                 }
288                 hitQueues[channel].add(stripHit);
289             }
290             
291             // Hits older than a certain time frame should no longer
292             // be used for pile-up simulation and should be removed
293             // from the processing queues.
294             for(SiSensor sensor : sensors) {
295                 // Get the processing queue for the current sensor.
296                 PriorityQueue<StripHit>[] hitQueues = hitMap.get(sensor);
297                 
298                 // Check each hit to see if it is still in-time.
299                 for(int i = 0; i < hitQueues.length; i++) {
300                     if(hitQueues[i] != null) {
301                         // Remove old hits.
302                         while(!hitQueues[i].isEmpty() && hitQueues[i].peek().time < ReadoutDataManager.getCurrentTime() - (readoutLatency + pileupCutoff)) {
303                             hitQueues[i].poll();
304                         }
305                         
306                         // If the queue is empty, remove it.
307                         if(hitQueues[i].isEmpty()) { hitQueues[i] = null; }
308                     }
309                 }
310             }
311         }
312         
313         // Otherwise, process the hits for a no pile-up simulation.
314         // When no pile-up is simulated, hits are fully processed and
315         // output on an event-by-event basis.
316         else {
317             // Create a list to hold the analog data.
318             List<RawTrackerHit> hits = new ArrayList<RawTrackerHit>();
319             
320             // Process each of the truth hits.
321             for(StripHit stripHit : stripHits) {
322                 // Get the hit parameters.
323                 HpsSiSensor sensor = (HpsSiSensor) stripHit.sensor;
324                 short[] samples = new short[6];
325                 
326                 // Create a signal buffer and populate it with the
327                 // appropriate pedestal values.
328                 double[] signal = new double[6];
329                 for(int sampleN = 0; sampleN < 6; sampleN++) {
330                     signal[sampleN] = sensor.getPedestal(stripHit.channel, sampleN);
331                 }
332                 
333                 // If noise should be added, do so.
334                 if(addNoise) {
335                     addNoise(sensor, stripHit.channel, signal);
336                 }
337                 
338                 // Emulate the pulse response and add it to the
339                 // sample array.
340                 for(int sampleN = 0; sampleN < 6; sampleN++) {
341                     double time = sampleN * HPSSVTConstants.SAMPLING_INTERVAL - timeOffset;
342                     shape.setParameters(stripHit.channel, (HpsSiSensor) sensor);
343                     signal[sampleN] += stripHit.amplitude * shape.getAmplitudePeakNorm(time);
344                     samples[sampleN] = (short) Math.round(signal[sampleN]);
345                 }
346                 
347                 // Create raw tracker hits from the sample data.
348                 long channel_id = sensor.makeChannelID(stripHit.channel);
349                 RawTrackerHit hit = new BaseRawTrackerHit(0, channel_id, samples, new ArrayList<SimTrackerHit>(stripHit.simHits), sensor);
350                 
351                 // If the analog hit passes the readout cuts, it may
352                 // be added to the data stream.
353                 if(readoutCuts(hit)) { hits.add(hit); }
354             }
355             
356             // Output the processed hits to the LCIO stream.
357             ReadoutDataManager.addData(outputCollection, hits, RawTrackerHit.class);
358         }
359     }

361     @Override
362     public void startOfData() {
363         // The output collection is only handled by the readout data
364         // manager if no pile-up simulation is included. Otherwise,
365         // the driver outputs its own collection at readout.
366         if(noPileup) {
367             LCIOCollectionFactory.setCollectionName(outputCollection);
368             LCIOCollectionFactory.setProductionDriver(this);
369             LCIOCollectionFactory.setFlags(1 << LCIOConstants.TRAWBIT_ID1);
370             LCIOCollectionFactory.setReadoutName(readout);
371             LCIOCollection<RawTrackerHit> noPileUpCollectionParams = LCIOCollectionFactory.produceLCIOCollection(RawTrackerHit.class);
372             ReadoutDataManager.registerCollection(noPileUpCollectionParams, true, 8.0, 32.0);
373         }
374         
375         // Define the LCSim on-trigger collection parameters.
376         LCIOCollectionFactory.setCollectionName(outputCollection);
377         LCIOCollectionFactory.setProductionDriver(this);
378         LCIOCollectionFactory.setFlags(1 << LCIOConstants.TRAWBIT_ID1);
379         LCIOCollectionFactory.setReadoutName(readout);
380         trackerHitCollectionParams = LCIOCollectionFactory.produceLCIOCollection(RawTrackerHit.class);
381         
382         LCIOCollectionFactory.setCollectionName(relationCollection);
383         LCIOCollectionFactory.setProductionDriver(this);
384         truthRelationsCollectionParams = LCIOCollectionFactory.produceLCIOCollection(LCRelation.class);
385         
386         LCIOCollectionFactory.setCollectionName("TrackerHits");
387         LCIOCollectionFactory.setFlags(0xc0000000);
388         LCIOCollectionFactory.setProductionDriver(this);
389         LCIOCollectionFactory.setReadoutName("TrackerHits");
390         truthHitsCollectionParams = LCIOCollectionFactory.produceLCIOCollection(SimTrackerHit.class);
391         
392         // Run the superclass method.
393         super.startOfData();
394     }

396     /**
397      * Performs a simulation of silicon sensor response and generates
398      * a collection of {@link
399      * org.hps.readout.svt.SVTReadoutDriver.StripHit StripHit}
400      * objects representing the detector response.
401      * @return Returns a collection of StripHit objects describing
402      * the detector response for the current event.
403      */
404     private List<StripHit> doSiSimulation() {
405         // Create a list to store the simulated hit objects.
406         List<StripHit> stripHits = new ArrayList<StripHit>();
407         
408         // Process each of the SVT sensors.
409         for(SiSensor sensor : sensors) {
410             // Set the sensor to be used in the charge deposition
411             // simulation.
412             siSimulation.setSensor(sensor);
413             
414             // Perform the charge deposition simulation.
415             Map<ChargeCarrier, SiElectrodeDataCollection> electrodeDataMap = siSimulation.computeElectrodeData();
416             
417             // Iterate over all possible charge carriers.
418             for(ChargeCarrier carrier : ChargeCarrier.values()) {
419                 // If the sensor is capable of collecting the given
420                 // charge carrier, then obtain the electrode data for
421                 // the sensor.
422                 if(sensor.hasElectrodesOnSide(carrier)) {
423                     // Attempt to obtain electrode data.
424                     SiElectrodeDataCollection electrodeDataCol = electrodeDataMap.get(carrier);
425 
426                     // If there is no electrode data available create
427                     // a new instance of electrode data.
428                     if(electrodeDataCol == null) {
429                         electrodeDataCol = new SiElectrodeDataCollection();
430                     }
431                     
432                     // Loop over all sensor channels.
433                     for(Integer channel : electrodeDataCol.keySet()) {
434 
435                         // Get the electrode data for this channel.
436                         SiElectrodeData electrodeData = electrodeDataCol.get(channel);
437                         Set<SimTrackerHit> simHits = electrodeData.getSimulatedHits();
438 
439                         // Compute hit time as the unweighted average
440                         // of SimTrackerHit times; this is dumb but
441                         // okay since there's generally only one
442                         // SimTrackerHit.
443                         double time = 0.0;
444                         for(SimTrackerHit hit : simHits) {
445                             time += hit.getTime();
446                         }
447                         time /= simHits.size();
448                         time += ReadoutDataManager.getCurrentTime();
449 
450                         // Get the charge in units of electrons.
451                         double charge = electrodeData.getCharge();
452                         
453                         // Calculate the amplitude.
454                         double resistorValue = 100; // Ohms
455                         double inputStageGain = 1.5;
456                         // FIXME: This should use the gains instead
457                         double amplitude = (charge / HPSSVTConstants.MIP) * resistorValue * inputStageGain * Math.pow(2, 14) / 2000;
458                         
459                         // Generate a StripHit object containing the
460                         // simulation data and add it to the list.
461                         stripHits.add(new StripHit(sensor, channel, amplitude, time, simHits));
462                     }
463                 }
464             }
465             
466             // Clear the sensors of all deposited charge
467             siSimulation.clearReadout();
468         }
469         
470         // Return the collection of StripHit objects.
471         return stripHits;
472     }

474     /**
475      * Adds a random Gaussian noise signature to the specified signal
476      * buffer based on the sensor and channel parameters.
477      * @param sensor - The sensor on which the signal buffer occurs.
478      * @param channel - The channel on which the signal buffer
479      * occurs.
480      * @param signal - The signal buffer. This must be an array of
481      * size six.
482      */
483     private void addNoise(SiSensor sensor, int channel, double[] signal) {
484         for(int sampleN = 0; sampleN < 6; sampleN++) {
485             signal[sampleN] += RandomGaussian.getGaussian(0, ((HpsSiSensor) sensor).getNoise(channel, sampleN));
486         }
487     }

489     /**
490      * Performs each of the three readout cuts, if they are enabled.
491      * This is the equivalent of calling, as appropriate, the methods
492      * {@link
493      * org.hps.readout.svt.SVTReadoutDriver#samplesAboveThreshold(RawTrackerHit)
494      * samplesAboveThreshold(RawTrackerHit)}, {@link
495      * org.hps.readout.svt.SVTReadoutDriver#pileupCut(RawTrackerHit)
496      * pileupCut(RawTrackerHit)}, and {@link
497      * org.hps.readout.svt.SVTReadoutDriver#badChannelCut(RawTrackerHit)
498      * badChannelCut(RawTrackerHit)}.
499      * @param hit - The analog hit to test.
500      * @return Returns <code>true</code> if all enabled cuts are
501      * passed, and <code>false</code> otherwise.
502      */
503     private boolean readoutCuts(RawTrackerHit hit) {
504         // Perform each enabled cut.
505         if(enableThresholdCut && !samplesAboveThreshold(hit)) {
506             return false;
507         }
508         if(enablePileupCut && !pileupCut(hit)) {
509             return false;
510         }
511         if(dropBadChannels && !badChannelCut(hit)) {
512             return false;
513         }
514         
515         // If all enabled cuts are passed, return true.
516         return true;
517     }

519     /**
520      * Checks whether an analog hit occurred on a "bad" channel.
521      * @param hit - The hit to be checked.
522      * @return Returns <code>true</code> if the hit <i>did not</i>
523      * occur on a bad channel, and <code>false</code> if it did.
524      */
525     private boolean badChannelCut(RawTrackerHit hit) {
526         HpsSiSensor sensor = (HpsSiSensor) hit.getDetectorElement();
527         int channel = hit.getIdentifierFieldValue("strip");
528         return !sensor.isBadChannel(channel);
529     }

531     /**
532      * Attempts to eliminate samples where the pulse starts before
533      * the sample array. This is done by requiring the second, third,
534      * and fourth samples of the array to be increasing in value with
535      * index.
536      * @param hit - The hit to check.
537      * @return Returns <code>true</code> if the no pile-up condition
538      * is met and <code>false</code> if it is not.
539      */
540     private boolean pileupCut(RawTrackerHit hit) {
541         short[] samples = hit.getADCValues();
542         return (samples[2] > samples[1] || samples[3] > samples[2]);
543     }

545     /**
546      * Attempts to eliminate false hits generated due to noise by
547      * requiring that a programmable number of samples exceed a
548      * similarly programmable noise threshold.
549      * @param hit - The hit to be checked.
550      * @return Returns <code>true</code> if the noise threshold count
551      * cut is met and <code>false</code> if it is not.
552      */
553     private boolean samplesAboveThreshold(RawTrackerHit hit) {
554         // Get the channel and sensor information for the hit.
555         int channel = hit.getIdentifierFieldValue("strip");
556         HpsSiSensor sensor = (HpsSiSensor) hit.getDetectorElement();
557         
558         // Track the noise and pedestal for each sample.
559         double noise;
560         double pedestal;
561         
562         // Iterate over the samples and count how many are above the
563         // noise threshold.
564         int count = 0;
565         short[] samples = hit.getADCValues();
566         for(int sampleN = 0; sampleN < samples.length; sampleN++) {
567             pedestal = sensor.getPedestal(channel, sampleN);
568             noise = sensor.getNoise(channel, sampleN);
569             if(samples[sampleN] - pedestal > noise * noiseThreshold) {
570                 count++;
571             }
572         }
573         
574         // The cut is passed if enough samples are above the noise
575         // threshold to pass the minimum count threshold.
576         return count >= samplesAboveThreshold;
577     }

579     @Override
580     protected Collection<TriggeredLCIOData<?>> getOnTriggerData(double triggerTime) {
581         // No pile-up events are output on an event-by-event basis,
582         // and as such, do not output anything at this stage.
583         if(noPileup) { return null; }
584         
585         // Create a list to hold the analog data
586         List<RawTrackerHit> hits = new ArrayList<RawTrackerHit>();
587         List<SimTrackerHit> truthHits = new ArrayList<SimTrackerHit>();
588         List<LCRelation> trueHitRelations = new ArrayList<LCRelation>();
589         
590         // Calculate time of first sample
591         double firstSample = Math.floor(((triggerTime + 256) - readoutLatency - readoutOffset) / HPSSVTConstants.SAMPLING_INTERVAL)
592                 * HPSSVTConstants.SAMPLING_INTERVAL + readoutOffset;
593         
594         List<StripHit> processedHits = new ArrayList<StripHit>();
595         
596         for(SiSensor sensor : sensors) {
597             // Get the hit queues for the current sensor.
598             PriorityQueue<StripHit>[] hitQueues = hitMap.get(sensor);
599 
600             // Check if a channel offset needs to be applied.  This will be 
601             // needed to retrieve the proper conditions for queues associated
602             // with SiStriplets.
603             int channelOffset = 0; 
604             if (sensor.getReadoutElectrodes(ChargeCarrier.HOLE) instanceof SiStriplets)
605                 channelOffset = 1;  
606             
607             // Iterate over the hit queue channels.
608             for(int channel = 0; channel < hitQueues.length; channel++) { 
609                
610                 // Use this to retrieve the correct conditions from the 
611                 // conditions database.  This accounts for the shift in the 
612                 // channel map in the thin sensors.
613                 int conditionsChannel = channel + channelOffset; 
614                 
615                 // Unless noise should be added, there is nothing to
616                 // process on an empty hit queue. Skip it.
617                 if(!addNoise && (hitQueues[channel] == null || hitQueues[channel].isEmpty())) {
618                     continue;
619                 }
620                 
621                 // Create a buffer to hold the extracted signal for
622                 // the channel. Populate it with the appropriate
623                 // pedestal values.
624                 double[] signal = new double[6];
625                 for(int sampleN = 0; sampleN < 6; sampleN++) {
626                     signal[sampleN] = ((HpsSiSensor) sensor).getPedestal(conditionsChannel, sampleN);
627                 }
628                 
629                 // If noise should be added, do so.
630                 if(addNoise) {
631                     addNoise(sensor, conditionsChannel, signal);
632                 }
633                 
634                 // Create a list to store truth SVT hits.
635                 List<SimTrackerHit> simHits = new ArrayList<SimTrackerHit>();
636                 
637                 // If there is data in the hit queues, process it.
638                 if(hitQueues[channel] != null) {
639                     for(StripHit hit : hitQueues[channel]) {
640                         processedHits.add(hit);
641                         
642                         // Track the noise and contribution to the
643                         // signal from the current hit.
644                         double meanNoise = 0;
645                         double totalContrib = 0;
646                         
647                         // Emulate the pulse response for the hit
648                         // across all size samples.
649                         StringBuffer signalBuffer = new StringBuffer("\t\t\t\tSample Pulse       :: [");
650                         for(int sampleN = 0; sampleN < 6; sampleN++) {
651                             double sampleTime = firstSample + sampleN * HPSSVTConstants.SAMPLING_INTERVAL;
652                             shape.setParameters(channel, (HpsSiSensor) sensor);
653                             double signalAtTime = hit.amplitude * shape.getAmplitudePeakNorm(sampleTime - hit.time);
654                             totalContrib += signalAtTime;
655                             signal[sampleN] += signalAtTime;
656                             meanNoise += ((HpsSiSensor) sensor).getNoise(conditionsChannel, sampleN);
657                             
658                             signalBuffer.append(signalAtTime + " (" + sampleTime + ")");
659                             if(sampleN != 5) {
660                                 signalBuffer.append("   ");
661                             }
662                         }
663                         signalBuffer.append("]");
664                         
665                         // TODO: Move this to the noise comparison below.
666                         meanNoise /= 6;
667                         
668                         // Calculate the average noise across all
669                         // samples and compare it to the contribution
670                         // from the hit. If it exceeds a the noise
671                         // threshold, store it as a truth hit.
672                         //meanNoise /= 6;
673                         if(totalContrib > 4.0 * meanNoise) {
674                             simHits.addAll(hit.simHits);
675                         }
676                     }
677                 }
678                 
679                 // Convert the samples into a short array,
680                 short[] samples = new short[6];
681                 for(int sampleN = 0; sampleN < 6; sampleN++) {
682                     samples[sampleN] = (short) Math.round(signal[sampleN]);
683                 }
684                 
685                 // Get the proper channel ID.
686                 // For the thin sensors, the channel number needs to be offset
687                 // by 1. 
688                 long channel_id = ((HpsSiSensor) sensor).makeChannelID(conditionsChannel);
689                 
690                 // Create a new tracker hit.
691                 RawTrackerHit hit = new BaseRawTrackerHit(0, channel_id, samples, simHits, sensor);
692                 
693                 // Only tracker hits that pass the readout cuts may
694                 // be passed through to readout.
695                 if(readoutCuts(hit)) {
696                     // Add the hit to the readout hits collection.
697                     hits.add(hit);
698                     
699                     // Associate the truth hits with the raw hit and
700                     // add them to the truth hits collection.
701                     for(SimTrackerHit simHit : hit.getSimTrackerHits()) {
702                         LCRelation hitRelation = new BaseLCRelation(hit, simHit);
703                         trueHitRelations.add(hitRelation);
704                         truthHits.add(simHit);
705                     }
706                 }
707             }
708         }
709         
710         // Create the collection data objects for output to the
711         // readout event.
712         TriggeredLCIOData<RawTrackerHit> hitCollection = new TriggeredLCIOData<RawTrackerHit>(trackerHitCollectionParams);
713         hitCollection.getData().addAll(hits);
714         TriggeredLCIOData<SimTrackerHit> truthHitCollection = new TriggeredLCIOData<SimTrackerHit>(truthHitsCollectionParams);
715         truthHitCollection.getData().addAll(truthHits);
716         TriggeredLCIOData<LCRelation> truthRelationCollection = new TriggeredLCIOData<LCRelation>(truthRelationsCollectionParams);
717         truthRelationCollection.getData().addAll(trueHitRelations);
718         
719         // MC particles need to be extracted from the truth hits
720         // and included in the readout data to ensure that the
721         // full truth chain is available.
722         Set<MCParticle> truthParticles = new java.util.HashSet<MCParticle>();
723         for(SimTrackerHit simHit : truthHits) {
724             ReadoutDataManager.addParticleParents(simHit.getMCParticle(), truthParticles);
725         }
726         
727         // Create the truth MC particle collection.
728         LCIOCollectionFactory.setCollectionName("MCParticle");
729         LCIOCollectionFactory.setProductionDriver(this);
730         LCIOCollection<MCParticle> truthParticleCollection = LCIOCollectionFactory.produceLCIOCollection(MCParticle.class);
731         TriggeredLCIOData<MCParticle> truthParticleData = new TriggeredLCIOData<MCParticle>(truthParticleCollection);
732         truthParticleData.getData().addAll(truthParticles);
733         
734         // A trigger timestamp needs to be produced as well.
735         ReadoutTimestamp timestamp = new ReadoutTimestamp(ReadoutTimestamp.SYSTEM_TRACKER, firstSample);
736         LCIOCollectionFactory.setCollectionName(ReadoutTimestamp.collectionName);
737         LCIOCollection<ReadoutTimestamp> timestampCollection = LCIOCollectionFactory.produceLCIOCollection(ReadoutTimestamp.class);
738         TriggeredLCIOData<ReadoutTimestamp> timestampData = new TriggeredLCIOData<ReadoutTimestamp>(timestampCollection);
739         timestampData.getData().add(timestamp);
740         
741         // Store them in a single collection.
742         Collection<TriggeredLCIOData<?>> eventOutput = new ArrayList<TriggeredLCIOData<?>>(5);
743         eventOutput.add(hitCollection);
744         eventOutput.add(truthParticleData);
745         eventOutput.add(truthHitCollection);
746         eventOutput.add(truthRelationCollection);
747         eventOutput.add(timestampData);
748         
749         // Return the event output.
750         return eventOutput;
751     }

753     /**
754      * Class <code>StripHit</code> is responsible for storing several
755      * parameters defining a simulated hit object.
756      */
757     private class StripHit implements Comparable<Object> {
758         SiSensor sensor;
759         int channel;
760         double amplitude;
761         double time;
762         Set<SimTrackerHit> simHits;
763 
764         public StripHit(SiSensor sensor, int channel, double amplitude, double time, Set<SimTrackerHit> simHits) {
765             this.sensor = sensor;
766             this.channel = channel;
767             this.amplitude = amplitude;
768             this.time = time;
769             this.simHits = simHits;
770         }
771 
772         @Override
773         public int compareTo(Object o) {
774             double deltaT = time - ((StripHit) o).time;
775             if(deltaT > 0) {
776                 return 1;
777             } else if(deltaT < 0) {
778                 return -1;
779             } else {
780                 return 0;
781             }
782         }
783     }

787         return 0;
788     }
789     
790     @Override
791     protected double getTimeNeededForLocalOutput() {
792         // TODO: Probably should have some defined value - buffer seems to be filled enough from the ecal delay alone, though.
793         return 100;
