21  /**
22   * Generic object to contain hybrid temperatures and data tail value. Converts and ADC value to a
23   * temperature in celsius
24   */
25  public class FpgaData implements GenericObject {
26  
27      int fpgaID;
28      List<Double> temperatures = new ArrayList<Double>();
29      int tail;
30      private static double[] temperatureTable = null;
31  
32      /**
33       * 
34       * @param fpgaID FPGA ID
35       * @param data : array containing hybrid temperatures
36       * @param tail : word present at the end of a FPGA data set     
37       */
38      public FpgaData(int fpgaID, int[] data, int tail) {
39          this.fpgaID = fpgaID;
40  
41          int[] temperature = extractTemperature(data);
42  
43          // Fill the temperature lookup table
44          fillTemperatureTable();
45  
46          this.tail = tail;
47  
48          // Fill the temperature list
49          for (int index = 0; index < temperature.length; index++) {
50              temperatures.add(intToTemperature(temperature[index]));
51          }
52      }
53  
54      public FpgaData(int fpgaID, double[] temperatures, int tail) {
55          this.fpgaID = fpgaID;
56  
57          this.tail = tail;
58  
59          this.temperatures.clear();
60          // Fill the temperature list
61          for (int index = 0; index < temperatures.length; index++) {
62              this.temperatures.add(temperatures[index]);
63          }
64      }
65  
66      /**
67       * Extract temperatures from the SVT data stream
68       * 
69       * @param data : array containing temperature data
70       * @return temperatures
71       * 
72       */
73      public static int[] extractTemperature(int[] data) {
74          int[] temperatures = new int[(data.length) * 2];
75  
76          int tempIndex = 0;
77          for (int index = 0; index < data.length; index++) {
78              temperatures[tempIndex] = data[index] & TEMP_MASK;
79              temperatures[tempIndex + 1] = (data[index] >>> 16) & TEMP_MASK;
80              tempIndex += 2;
81          }
82          return temperatures;
83      }
84  
85      /**
86       * Temperature lookup table. Takes an ADC value and returns a temperature in Celsius
87       */
88      private static void fillTemperatureTable() {
89  
90          if (temperatureTable == null) {
91              temperatureTable = new double[ADC_TEMP_COUNT];
92  
93              double tempK, res, volt;
94              int idx;
95              double temp = MIN_TEMP;
96  
97              while (temp < MAX_TEMP) {
98  
99                  tempK = TEMP_K0 + temp;
100                 res = CONST_A * Math.exp(BETA / tempK);
101                 volt = (res * V_MAX) / (R_DIV + res);
102                 idx = (int) ((volt / V_REF) * (double) (ADC_TEMP_COUNT - 1));
103                 if (idx < ADC_TEMP_COUNT) {
104                     temperatureTable[idx] = temp;
105                 }
106                 temp += TEMP_INC;
107             }
108         }
109     }
110 
111     public static double intToTemperature(int tempIndex) {
112         fillTemperatureTable();
113         return temperatureTable[tempIndex];
114     }
115 
116     public static int temperatureToInt(double temperature) {
117         fillTemperatureTable();
118         return Math.abs(Arrays.binarySearch(temperatureTable, temperature));
119     }
120 
121     public int[] extractData() {
122         fillTemperatureTable();
123 
124         int[] header = new int[(temperatures.size() + 1) / 2];
125         for (int i = 0; i < temperatures.size(); i++) {
126             if (i % 2 == 0) {
127                 header[i / 2] = (header[i / 2] &= ~TEMP_MASK) | (temperatureToInt(temperatures.get(i)) & TEMP_MASK);
128             } else {
129                 header[i / 2] = (header[i / 2] &= ~(TEMP_MASK << 16)) | ((temperatureToInt(temperatures.get(i)) & TEMP_MASK) << 16);
130             }
131         }
132         return header;
133     }
