22  /**
23   * Overlay signal event collections every N events, applying optional
24   * filters to reject unwanted events.
25   */
26  public class SignalOverlayDriver extends Driver {
27  
28      private static Logger LOGGER =
29              Logger.getLogger(SignalOverlayDriver.class.getPackage().getName());
30      private Queue<String> signalFileNames = new LinkedList<String>();
31      private int eventSpacing = 250;
32      private int sequence = 0;
33      private int nRejected = 0;
34      private int nSignal = 0;
35      private String ecalHitCollection = "EcalHits";
36      private String hodoHitCollection = "HodoscopeHits";
37      private LCIOReader reader = null;
38      private Set<EventFilter> eventFilters = new HashSet<EventFilter>();
39  
40      interface EventFilter {
41          public boolean accept(EventHeader event);
42      }
43  
44      class EcalEnergyFilter implements EventFilter {
45  
46          double eCut = 0;
47  
48          public EcalEnergyFilter(double eCut) {
49              this.eCut = eCut;
50          }
51  
52          public boolean accept(EventHeader event) {
53              List<SimCalorimeterHit> ecalHits = event.get(SimCalorimeterHit.class,
54                      SignalOverlayDriver.this.ecalHitCollection);
55              double totalE = 0;
56              for (SimCalorimeterHit hit : ecalHits) {
57                  totalE += hit.getRawEnergy();
58              }
59              LOGGER.fine("ECAL totalE = " + totalE);
60              return totalE > eCut;
61          }
62      }
63  
64      class HodoHitFilter implements EventFilter {
65  
66          int nHits = 0;
67  
68          public HodoHitFilter(int nHits) {
69              this.nHits = nHits;
70          }
71  
72          public boolean accept(EventHeader event) {
73              List<SimTrackerHit> hodoHits = event.get(SimTrackerHit.class,
74                      SignalOverlayDriver.this.hodoHitCollection);
75              return hodoHits.size() >= nHits;
76          }
77      }
78  
79      class EcalPairFilter implements EventFilter {
80  
81          double eCut;
82  
83          public EcalPairFilter(double eCut) {
84              this.eCut = eCut;
85          }
86  
87          @Override
88          public boolean accept(EventHeader event)  {
89              List<SimCalorimeterHit> ecalHits = event.get(SimCalorimeterHit.class,
90                      SignalOverlayDriver.this.ecalHitCollection);
91              double topE = 0, botE = 0;
92              for (SimCalorimeterHit hit : ecalHits) {
93                  if (hit.getIdentifierFieldValue("iy") > 0) {
94                      topE += hit.getRawEnergy();
95                  } else {
96                      botE += hit.getRawEnergy();
97                  }
98              }
99              return (topE > eCut && botE > eCut);
100         }
101     }

131     public void setSignalFileList(String signalFileList) {
132         BufferedReader reader;
133         try {
134             reader = new BufferedReader(new FileReader(signalFileList));
135             String line = reader.readLine();
136             while (line != null) {
137                 line = reader.readLine();
138                 signalFileNames.add(line);
139             }
140             reader.close();
141         } catch (IOException e) {
142             throw new RuntimeException(e);
143         }
144     }

146     public void setEventSpacing(int eventSpacing) {
147         if (eventSpacing <= 0) {
148             throw new IllegalArgumentException("Bad value for event spacing: " + eventSpacing);
149         }
150         this.eventSpacing = eventSpacing;
151     }
152 
153     @Override
154     protected void startOfData() {
155         if (this.signalFileNames.size() == 0) {
156             throw new RuntimeException("No signal files were provided!");
157         }
158         String filePath = this.signalFileNames.remove();
159         try {
160             LOGGER.info("Opening signal file: " + filePath);
161             File file = new File(filePath);
162             if (!file.exists()) {
163                 throw new RuntimeException("File does not exist: " + file.getName());
164             }
165             reader = new LCIOReader(file);
166         } catch (IOException e) {
167             LOGGER.severe("Failed to read from: " + filePath);
168             throw new RuntimeException(e);
169         }
170     }
171 
172     private void openNextFile() throws EndOfDataException {
173         try {
174             LOGGER.info("Closing previous reader...");
175             reader.close();
176         } catch (IOException e) {
177             e.printStackTrace();
178         }
179         if (signalFileNames.size() == 0) {
180             throw new EndOfDataException();
181         }
182         String path = signalFileNames.remove();
183         LOGGER.info("Opening next file: " + path);
184         try {
185             reader = new LCIOReader(new File(path));
186         } catch (IOException e) {
187             throw new RuntimeException(e);
188         }
189     }
190 
191     private EventHeader readNextEvent() throws EndOfDataException {
192         EventHeader event = null;
193         try {
194             LOGGER.fine("Reading next event...");
195             event = this.reader.read();
196         } catch (EOFException eof) {
197             openNextFile();
198             try {
199                 LOGGER.fine("Reading first event in new file...");
200                 event = this.reader.read();
201             } catch (IOException e) {
202                 e.printStackTrace();
203             }
204         } catch (IOException ioe) {
205             ioe.printStackTrace();
206         } finally {
207             if (event == null) {
208                 throw new RuntimeException("Failed to read next event.");
209             }
210         }
211         return event;
212     }
213 
214     private boolean accept(EventHeader event) {
215         for (EventFilter filter : eventFilters) {
216             if (!filter.accept(event)) {
217                 return false;
218             }
219         }
220         return true;
221     }
222 
223     @SuppressWarnings({ "rawtypes", "unchecked" })
224     @Override
225     protected void process(EventHeader event) {
226         LOGGER.fine("Processing background event: " + event.getEventNumber());
227 
228         // Overlay signal event
229         if (this.sequence % this.eventSpacing == 0) {
230 
231             // Find the next acceptable signal event
232             EventHeader signalEvent = null;
233             try {
234                 signalEvent = this.readNextEvent();
235                 while (true) {
236                     if (accept(signalEvent)) break;
237                     LOGGER.fine("Rejected signal event: " + signalEvent.getEventNumber());
238                     nRejected += 1;
239                     signalEvent = this.readNextEvent();
240                 }
241             } catch (EndOfDataException e) {
242                 throw new RuntimeException("Ran out of signal events!", e);
243             }
244             LOGGER.fine("Read signal event: " + signalEvent.getEventNumber());
245 
246             // Overlay signal collections onto background event
247             Set<List> lists = signalEvent.getLists();
248             for (List list : lists) {
249                 LCMetaData meta = signalEvent.getMetaData(list);
250                 String collName = meta.getName();
251                 if (!event.hasItem(collName)) {
252                     event.put(collName, list, meta.getType(), meta.getFlags());
253                 } else {
254                     event.get(meta.getClass(), collName).addAll(list);
255                 }
256                 LOGGER.finer("Added " + list.size() + " objects to collection: " + collName);
257             }
258 
259             ++nSignal;
260         }
261 
262         ++this.sequence;
263     }
264 
265     protected void endOfData() {
266         System.out.println("End event sequence: " + this.sequence);
267         System.out.println("Signal events rejected: " + nRejected);
268         System.out.println("Signal events overlaid: " + nSignal);
269     }
