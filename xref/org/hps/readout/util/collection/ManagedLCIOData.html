1   package org.hps.readout.util.collection;

3   import java.util.LinkedList;
4   
5   import org.hps.readout.util.TimedList;
6   
7   /**
8    * Class <code>ManagedLCIOData</code> represents the actual managed
9    * data associated with a managed LCIO collection. It stores both the
10   * collection parameters and also a list of data in the form of a
11   * {@link org.hps.readout.util.TimedList TimedList}, where each entry
12   * corresponds to the data present at a given time.
13   * 
14   * @param <T> - The object type of the data stored by the collection.
15   */
16  public class ManagedLCIOData<T> extends LCIOData<T, ManagedLCIOCollection<T>, LinkedList<TimedList<?>>> {
17      /**
18       * The collection data. Each entry in the data list represents a
19       * specific simulation time quantum, while the list itself holds
20       * the collection object data.
21       */
22      private final LinkedList<TimedList<?>> data;
23      
24      /**
25       * Creates a new <code>ManagedLCIOData</code> based on the
26       * collection parameters defined by the <code>params</code>
27       * object.
28       * @param params - The collection parameters.
29       */
30      public ManagedLCIOData(ManagedLCIOCollection<T> params) {
31          super(params);
32          this.data = new LinkedList<TimedList<?>>();
33      }
34      
35      @Override
36      public LinkedList<TimedList<?>> getData() {
37          return data;
38      }
39  }
