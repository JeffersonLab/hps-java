5   /**
6    * Class <code>LCIOCollectionFactory</code> is responsible for
7    * instantiating LCIO collection parameter objects. It allows all of
8    * the relevant parameters to be set, at which point one the two
9    * methods {@link
10   * org.hps.readout.util.collection.LCIOCollectionFactory#produceLCIOCollection(Class)
11   * produceLCIOCollection(Class)} or {@link
12   * org.hps.readout.util.collection.LCIOCollectionFactory#produceManagedLCIOCollection(Class)
13   * produceManagedLCIOCollection(Class)} may be called to generate the
14   * actual object.<br/><br/>
15   * Note that {@link
16   * org.hps.readout.util.collection.ManagedLCIOCollection
17   * ManagedLCIOCollection} objects are used by the {@link
18   * org.hps.readout.ReadoutDataManager ReadoutDataManager} and
19   * generally should not be instantiated by users. {@link
20   * org.hps.readout.util.collection.LCIOCollection LCIOCollection}
21   * collection class is intended to be defined by {@link
22   * org.hps.readout.ReadoutDriver ReadoutDriver} objects for use in
23   * either outputting on-triggered or defining their output to the
24   * readout manager.
25   * 
26   * @see org.hps.readout.util.collection.LCIOCollection
27   * @see org.hps.readout.util.collection.ManagedLCIOCollection
28   */
29  public class LCIOCollectionFactory {
30      private static Integer flags = null;
31      private static Boolean persistent = null;
32      private static double windowAfter = Double.NaN;
33      private static double windowBefore = Double.NaN;
34      private static double timeDisplacement = Double.NaN;
35      private static String readoutName = null;
36      private static String collectionName = null;
37      private static ReadoutDriver productionDriver = null;
38      
39      /**
40       * Clones an LCIO collection parameters object. If a parameter is
41       * defined in the factory, that parameter will be used for the
42       * new cloned object. Otherwise, the parameter value from the
43       * argument object will be used instead.<br/><br/>
44       * This method is intended to be used to create a new instance of
45       * an LCIO collection parameters object with a new parameter
46       * value.
47       * @param baseParams - The set of default parameters.
48       * @return Returns a new LCIO collection parameters object with
49       * the same parameter values as the original, unless specifically
50       * defined in the factory beforehand.
51       */
52      public static final <T> LCIOCollection<T> cloneCollection(LCIOCollection<T> baseParams) {
53          // Set any undefined parameters to match the base object.
54          if(collectionName == null) { setCollectionName(baseParams.getCollectionName()); }
55          if(flags == null) { setFlags(baseParams.getFlags()); }
56          if(productionDriver == null) { setProductionDriver(baseParams.getProductionDriver()); }
57          if(readoutName == null) { setReadoutName(baseParams.getReadoutName()); }
58          
59          // Produce a new collection parameters object.
60          return produceLCIOCollection(baseParams.getObjectType());
61      }
62      
63      /**
64       * Creates an {@link
65       * org.hps.readout.util.collection.LCIOCollection LCIOCollection}
66       * from the parameters set in the factory. The following
67       * parameters are used for this object type:
68       * <ul>
69       * <li>Collection Name</li>
70       * <li>Production Driver</li>
71       * <li>Flags</li>
72       * <li>Readout Name</li>
73       * </ul>
74       * Of these, the collection name and the production driver are
75       * required and, if not set, will generate an exception.
76       * <br/><br/>
77       * The collection parameters object will be parameterized as per
78       * the input class type argument.
79       * <br/><br/>
80       * Note that all parameters are reset automatically after a
81       * collection parameters object is generated.
82       * @param objectType - The class type of the objects represented
83       * by this collection.
84       * @return Returns an LCIO collection parameters object with the
85       * specified parameters.
86       * @param <T> - The type of object stored in the collection.
87       */
88      public static final <T> LCIOCollection<T> produceLCIOCollection(Class<T> objectType) {
89          // Verify that the necessary parameters have been set.
90          String errorMessage = verifyBaseCollectionParameters();
91          if(errorMessage != null) {
92              throw new IllegalArgumentException(errorMessage);
93          }
94          
95          // Create the new object.
96          LCIOCollection<T> collection  = new ManagedLCIOCollection<T>(collectionName, productionDriver, objectType, timeDisplacement,
97                  (flags == null ? 0 : flags.intValue()), readoutName);
98          
99          // Reset the factory.
100         reset();
101         
102         // Return the result.
103         return collection;
104     }

106     /**
107      * Creates an {@link
108      * org.hps.readout.util.collection.ManagedLCIOCollection
109      * ManagedLCIOCollection} from the parameters set in the factory.
110      * The following parameters are used for this object type:
111      * <ul>
112      * <li>Collection Name</li>
113      * <li>Production Driver</li>
114      * <li>Flags</li>
115      * <li>Readout Name</li>
116      * <li>Persistent</li>
117      * <li>Time Displacement</li>
118      * <li>Window Before</li>
119      * <li>Window After</li>
120      * </ul>
121      * Of these, the collection name, the production driver, and the
122      * time displacement are required and, if not set, will generate
123      * an exception. <br/><br/>
124      * The collection parameters object will be parameterized as per
125      * the input class type argument.<br/><br/>
126      * Note that all parameters are reset automatically after a
127      * collection parameters object is generated.
128      * @param objectType - The class type of the objects represented
129      * by this collection.
130      * @return Returns an LCIO collection parameters object with the
131      * specified parameters.
132      * @param <T> - The type of object stored in the collection.
133      */
134     public static final <T> ManagedLCIOCollection<T> produceManagedLCIOCollection(Class<T> objectType) {
135         // Verify that the necessary parameters have been set.
136         String errorMessage = verifyManagedCollectionParameters();
137         if(errorMessage != null) {
138             throw new IllegalArgumentException(errorMessage);
139         }
140         
141         // Create the new object.
142         ManagedLCIOCollection<T> collection  = new ManagedLCIOCollection<T>(collectionName, productionDriver, objectType, timeDisplacement,
143                 (flags == null ? 0 : flags.intValue()), readoutName);
144         if(!Double.isNaN(windowAfter)) { collection.setWindowAfter(windowAfter); }
145         if(!Double.isNaN(windowBefore)) { collection.setWindowBefore(windowBefore); }
146         if(persistent != null) { collection.setPersistent(persistent.booleanValue()); }
147         
148         // Reset the factory.
149         reset();
150         
151         // Return the result.
152         return collection;
153     }

276     public static final void reset() {
277         flags = null;
278         persistent = null;
279         windowAfter = Double.NaN;

287     /**
288      * Checks that all of the parameters needed to define a basic
289      * LCIO collection are in fact defined.
290      * @return If all parameters are defined, <code>null</code> is
291      * returned. Otherwise, a {@link java.lang.String String} is
292      * returned containing an error message.
293      */
294     private static final String verifyBaseCollectionParameters() {
295         boolean setCollectionName = (collectionName != null);
296         //boolean setProductionDriver = (productionDriver != null);
297         
298         if(setCollectionName) {// && setProductionDriver) {
299             return null;
300         } else {
301             StringBuffer errorMessage = new StringBuffer("Error: The following parameters have not been set:");
302             if(!setCollectionName) { errorMessage.append("\n\tCollection Name"); }
303             //if(!setProductionDriver) { errorMessage.append("\n\tProduction Driver"); }
304             return errorMessage.toString();
305         }
306     }
307     
308     /**
309      * Checks that all of the parameters needed to define a managed
310      * LCIO collection are in fact defined.
311      * @return If all parameters are defined, <code>null</code> is
312      * returned. Otherwise, a {@link java.lang.String String} is
313      * returned containing an error message.
314      */
315     private static final String verifyManagedCollectionParameters() {
316         String baseError = verifyBaseCollectionParameters();
317         boolean setTimeDisplacement = (!Double.isNaN(timeDisplacement));
318         
319         if(baseError == null && !setTimeDisplacement) {
320             return "Error: The following parameters have not been set:\n\tTime Displacement";
321         } else if(baseError != null && !setTimeDisplacement) {
322             return baseError + "\n\tTime Displacement";
323         } else if(baseError != null && setTimeDisplacement) {
324             return baseError;
325         } else { return null; }
326     }
