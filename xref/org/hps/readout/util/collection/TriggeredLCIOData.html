1   package org.hps.readout.util.collection;
2   
3   import java.util.HashSet;
4   import java.util.Set;
5   
6   /**
7    * Class <code>TriggeredLCIOData</code> represents data output from
8    * a {@link org.hps.readout.ReadoutDriver ReadoutDriver}. It contains
9    * both a list of the output data, corresponding to a single event,
10   * and also the LCIO parameters needed to write the event.
11   * 
12   * @param <T> - The object type of the data stored by the collection.
13   */
14  public class TriggeredLCIOData<T> extends LCIOData<T, LCIOCollection<T>, Set<T>> {
15      /**
16       * The collection data.
17       */
18      private final Set<T> data;
19      
20      /**
21       * Creates a new <code>TriggeredLCIOData</code based on the
22       * collection parameters defined by the <code>params</code>
23       * object.
24       * @param params - The collection parameters.
25       */
26      public TriggeredLCIOData(LCIOCollection<T> params) {
27          super(params);
28          this.data = new HashSet<T>();
29      }
30      
31      /**
32       * Creates a copy of this object with no data stored.
33       * @return Returns a duplicate of this object with the same
34       * parameters, but no stored data.
35       */
36      public TriggeredLCIOData<T> cloneEmpty() {
37          return new TriggeredLCIOData<T>(getCollectionParameters());
38      }
39      
40      @Override
41      public Set<T> getData() {
42          return data;
43      }
44  }
