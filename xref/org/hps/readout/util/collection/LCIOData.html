3   import java.util.Collection;

5   /**
6    * Class <code>LCIOData</code> functions as a superclass to all the
7    * readout framework LCIO collection data set objects. It requires
8    * that they define LCIO collection parameters, and provide access to
9    * the data sets.
10   * 
11   * @param <T> - The object type of the data stored by the collection.
12   * @param <E> - The type of LCIO collection parameters object used by
13   * the data set.
14   * @param <V> - The collection that actually stores the data.
15   */
16  public abstract class LCIOData<T, E extends LCIOCollection<T>, V extends Collection<?>> {
17      private final E params;
18      
19      /**
20       * Instantiates the LCIO data collection.
21       * @param collectionParameters
22       */
23      LCIOData(E collectionParameters) {
24          params = collectionParameters;
25      }
26      
27      /**
28       * Gets the collection parameters for the data.
29       * @return Returns the collection parameters as a {@link
30       * org.hps.readout.util.collection.LCIOCollection
31       * LCSimCollection} object or subclass.
32       */
33      public E getCollectionParameters() {
34          return params;
35      }
36      
37      /**
38       * Gets the collection data.
39       * @return Returns the collection data.
40       */
41      public abstract V getData();
42  }
