1   package org.hps.readout.util;
2   
3   import java.util.ArrayList;
4   import java.util.Collection;
5   import java.util.Comparator;
6   
7   /**
8    * Class <code>TimedList</code> is an extension of {@link
9    * java.util.ArrayList ArrayList} that additionally stores a single
10   * time. This is used to store a collection of LCIO data associated
11   * with a specific simulation time. It can be compared to other
12   * timed lists, but only compares the list time, not the contents of
13   * the lists.
14   * 
15   * @param <E> - The object type of the stored data.
16   */
17  public class TimedList<E> extends ArrayList<E> implements Comparable<TimedList<?>> {
18      private static final long serialVersionUID = -4261502557924453657L;
19      /**
20       * The simulation time of the data stored in the list.
21       */
22      private final double listTime;
23      
24      /**
25       * A default comparator for comparing the times of two lists.
26       */
27      private static final Comparator<TimedList<?>> COMPARATOR = new Comparator<TimedList<?>>() {
28          @Override
29          public int compare(TimedList<?> t0, TimedList<?> t1) {
30              return t0.compareTo(t1);
31          }
32      };
33      
34      /**
35       * Instantiates a <code>TimedList</code> object with a simulation
36       * time <code>time</code>.
37       * @param time - The time in nanoseconds at which the data stored
38       * in the list occurred.
39       */
40      public TimedList(double time) {
41          super();
42          listTime = time;
43      }
44      
45      /**
46       * Instantiates a <code>TimedList</code> object with a simulation
47       * time <code>time</code> which contains the objects stored in
48       * <code>data</code>.
49       * @param time - The time in nanoseconds at which the data stored
50       * in the list occurred.
51       * @param data - A collection of data to be stored in the list.
52       */
53      public TimedList(double time, Collection<? extends E> data) {
54          super(data);
55          listTime = time;
56      }
57      
58      /**
59       * Instantiates a <code>TimedList</code> object with a simulation
60       * time <code>time</code> with a given initial capacity.
61       * @param time - The time in nanoseconds at which the data stored
62       * in the list occurred.
63       * @param initialCapacity - The initial capacity of the list.
64       */
65      public TimedList(double time, int initialCapacity) {
66          super(initialCapacity);
67          listTime = time;
68      }
69      
70      @Override
71      public int compareTo(TimedList<?> arg0) {
72          return Double.compare(listTime, arg0.listTime);
73      }
74      
75      /**
76       * Gets the simulation time of the data stored in the list.
77       * @return Returns the simulation time of the list data in units
78       * of nanoseconds.
79       */
80      public double getTime() {
81          return listTime;
82      }
83      
84      /**
85       * Gets a comparator that compares the simulation times for any
86       * two <code>TimedList</code> objects.
87       * @return
88       */
89      public static final Comparator<TimedList<?>> getComparator() {
90          return COMPARATOR;
91      }
92  }
