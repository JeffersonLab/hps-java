18  public class PairTriggerReadoutDriver extends TriggerDriver {
19      // ==================================================================
20      // ==== Trigger General Default Parameters ==========================
21      // ==================================================================
22      private String inputCollectionName = "EcalClustersGTP";       // Name for the LCIO cluster collection.
23      private int pairCoincidence = 2;                              // Maximum allowed time difference between clusters. (4 ns clock-cycles)
24      private String ecalGeometryName = "Ecal";                     // Name of the calorimeter geometry object.
25      private TriggerModule triggerModule = new TriggerModule(1.0, 0.050,
26              6.600, 0.010, 6.600, 0.000, 13.200, 6.600, 0.0, 360, 0.0055);
27      
28      // ==================================================================
29      // ==== Driver Internal Variables ===================================
30      // ==================================================================
31      private Queue<List<Cluster>> topClusterQueue = null;           // Store clusters on the top half of the calorimeter.
32      private Queue<List<Cluster>> botClusterQueue = null;           // Store clusters on the bottom half of the calorimeter.
33      private double localTime = 0.0;                                // Stores the internal time clock for the driver.
34      private HPSEcal3 ecal = null;                                  // The calorimeter geometry object.
35      
36      @Override
37      public void detectorChanged(Detector detector) {
38          // Get the calorimeter sub-detector.
39          org.lcsim.geometry.compact.Subdetector ecalSub = detector.getSubdetector(ecalGeometryName);
40          if(ecalSub instanceof HPSEcal3) {
41              ecal = (HPSEcal3) ecalSub;
42          } else {
43              throw new IllegalStateException("Error: Unexpected calorimeter sub-detector of type \"" + ecalSub.getClass().getSimpleName() + "; expected HPSEcal3.");
44          }
45      }
46      
47      @Override
48      public void startOfData() {
49          // Define the driver collection dependencies.
50          addDependency(inputCollectionName);
51          
52          // Register the trigger.
53          ReadoutDataManager.registerTrigger(this);
54          
55          // Make sure that a valid cluster collection name has been
56          // defined. If it has not, throw an exception.
57          if(inputCollectionName == null) {
58              throw new RuntimeException("The parameter inputCollectionName was not set!");
59          }
60          
61          // Initialize the top and bottom cluster queues.
62          topClusterQueue = new LinkedList<List<Cluster>>();
63          botClusterQueue = new LinkedList<List<Cluster>>();
64          
65          // Populate the top cluster queue. It should be populated with
66          // a number of empty lists equal to (2 * pairCoincidence + 1).
67          for(int i = 0; i < 2 * pairCoincidence + 1; i++) {
68              topClusterQueue.add(new ArrayList<Cluster>());
69          }
70          
71          // Populate the bottom cluster queue. It should be populated with
72          // a number of empty lists equal to (2 * pairCoincidence + 1).
73          for(int i = 0; i < pairCoincidence + 1; i++) {
74              botClusterQueue.add(new ArrayList<Cluster>());
75          }
76          
77          // Run the superclass method.
78          super.startOfData();
79      }
80      
81      @Override
82      public void process(EventHeader event) {
83          // If there is no data ready, then nothing can be done/
84          if(!ReadoutDataManager.checkCollectionStatus(inputCollectionName, localTime)) {
85              return;
86          }
87          
88          // Otherwise, get the input clusters from the present time.
89          Collection<Cluster> clusters = ReadoutDataManager.getData(localTime, localTime + 4.0, inputCollectionName, Cluster.class);
90          
91          // Remove any clusters that do not pass the singles cuts. It
92          // is more efficient to eliminate these before forming pairs.
93          Collection<Cluster> goodClusters = getGoodClusters(clusters);
94              
95          // Put the good clusters into the cluster queue.
96          updateClusterQueues(goodClusters);
97          
98          // Check that if a trigger exists, if the trigger is not in
99          // dead time. If it is, no trigger may be issued, so this is
100         // not necessary.
101         if(!isInDeadTime() && testTrigger()) { sendTrigger(); }
102         
103         // Increment the local time.
104         localTime += 4.0;
105     }

232     /**
233      * Get a list of all unique cluster pairs in the event.
234      * @return A <code>List</code> collection of <code>Cluster</code> 
235      * objects containing all cluster pairs.
236      */
237     protected List<Cluster[]> getClusterPairsTopBot() {
238         // Create a list to store cluster pairs. 
239         List<Cluster[]> clusterPairs = new ArrayList<Cluster[]>();
240         
241         // Loop over all top-bottom pairs of clusters; higher-energy cluster goes first in the pair
242         // To apply pair coincidence time, use only bottom clusters from the 
243         // readout cycle pairCoincidence readout cycles ago, and top clusters 
244         // from all 2*pairCoincidence+1 previous readout cycles
245         for(Cluster botCluster : botClusterQueue.element()) {
246             for(List<Cluster> topClusters : topClusterQueue) {
247                 for(Cluster topCluster : topClusters) {
248                     // The first cluster in a pair should always be
249                     // the higher energy cluster. If the top cluster
250                     // is higher energy, it goes first.
251                     if(topCluster.getEnergy() > botCluster.getEnergy()) {
252                         Cluster[] clusterPair = { topCluster, botCluster };
253                         clusterPairs.add(clusterPair);
254                     }
255                     
256                     // Otherwise, the bottom cluster goes first.
257                     else {
258                         Cluster[] clusterPair = { botCluster, topCluster };
259                         clusterPairs.add(clusterPair);
260                     }
261                 }
262             }
263         }
264         
265         // Return the cluster pair lists.
266         return clusterPairs;
267     }

279     private Collection<Cluster> getGoodClusters(Collection<Cluster> clusters) {
280         // Create a list to hold clusters which pass the single
281         // cluster cuts.
282         Collection<Cluster> goodClusterList = new ArrayList<Cluster>();
283         
284         // Sort through the cluster list and add clusters that pass
285         // the single cluster cuts to the good list.
286         clusterLoop:
287         for(Cluster cluster : clusters) {
288             // ==== Seed Hit Energy Cut ====================================
289             // =============================================================
290             // If the cluster fails the cut, skip to the next cluster.
291             if(!triggerModule.clusterSeedEnergyCut(cluster)) {
292                 continue clusterLoop;
293             }
294             
295             // ==== Cluster Hit Count Cut ==================================
296             // =============================================================
297             // If the cluster fails the cut, skip to the next cluster.
298             if(!triggerModule.clusterHitCountCut(cluster)) {
299                 continue clusterLoop;
300             }
301             
302             // ==== Cluster Total Energy Cut ===============================
303             // =============================================================
304             // If the cluster fails the cut, skip to the next cluster.
305             if(!triggerModule.clusterTotalEnergyCut(cluster)) {
306                 continue clusterLoop;
307             }
308             
309             // A cluster that passes all of the single-cluster cuts
310             // can be used in cluster pairs.
311             goodClusterList.add(cluster);
312         }
313         
314         // Return the good clusters.
315         return goodClusterList;
316     }

318     /**
319      * Tests all of the current cluster pairs for triggers.
320      * @return Returns <code>true</code> if one of the cluster pairs
321      * passes all of the cluster cuts and <code>false</code> otherwise.
322      */
323     private boolean testTrigger() {
324         // Track whether a trigger has occurred.
325         boolean triggered = false;
326         
327         // Get the list of cluster pairs.
328         List<Cluster[]> clusterPairs = getClusterPairsTopBot();
329         
330         // Iterate over the cluster pairs and perform each of the cluster
331         // pair cuts on them. A cluster pair that passes all of the
332         // cuts registers as a trigger.
333         pairLoop:
334         for(Cluster[] clusterPair : clusterPairs) {
335             // Get the x and y indices. Note that LCSim meta data is
336             // not available during readout, so crystal indices must
337             // be obtained directly from the calorimeter geometry.
338             java.awt.Point ixy0 = ecal.getCellIndices(clusterPair[0].getCalorimeterHits().get(0).getCellID());
339             java.awt.Point ixy1 = ecal.getCellIndices(clusterPair[1].getCalorimeterHits().get(0).getCellID());
340             
341             
342             // ==== Pair Energy Sum Cut ====================================
343             // =============================================================
344             // If the cluster fails the cut, skip to the next pair.
345             if(!triggerModule.pairEnergySumCut(clusterPair)) {
346                 continue pairLoop;
347             }
348             
349             // ==== Pair Energy Difference Cut =============================
350             // =============================================================
351             // If the cluster fails the cut, skip to the next pair.
352             if(!triggerModule.pairEnergyDifferenceCut(clusterPair)) {
353                 continue pairLoop;
354             }
355             
356             // ==== Pair Energy Slope Cut ==================================
357             // =============================================================
358             // If the cluster fails the cut, skip to the next pair.
359             if(!triggerModule.pairEnergySlopeCut(clusterPair, ixy0, ixy1)) {
360                 continue pairLoop;
361             }
362             
363             // ==== Pair Coplanarity Cut ===================================
364             // =============================================================
365             // If the cluster fails the cut, skip to the next pair.
366             if(!triggerModule.pairCoplanarityCut(clusterPair, ixy0, ixy1)) {
367                 continue pairLoop;
368             }
369             
370             // Clusters that pass all of the pair cuts produce a trigger.
371             triggered = true;
372         }
373         
374         // Return whether or not a trigger was observed.
375         return triggered;
376     }

378     /**
379      * Adds clusters from a new event into the top and bottom cluster
380      * queues so that they may be formed into pairs.
381      * @param clusterList - The clusters to add to the queues.
382      */
383     private void updateClusterQueues(Collection<Cluster> clusterList) {
384         // Create lists to store the top and bottom clusters.
385         ArrayList<Cluster> topClusterList = new ArrayList<Cluster>();
386         ArrayList<Cluster> botClusterList = new ArrayList<Cluster>();
387         
388         // Loop over the clusters in the cluster list.
389         for(Cluster cluster : clusterList) {
390             // Get the x and y indices. Note that LCSim meta data is
391             // not available during readout, so crystal indices must
392             // be obtained directly from the calorimeter geometry.
393             java.awt.Point ixy = ecal.getCellIndices(cluster.getCalorimeterHits().get(0).getCellID());
394             
395             // If the cluster is on the top of the calorimeter, it
396             // goes into the top cluster list.
397             if(ixy.y > 0) { topClusterList.add(cluster); }
398             
399             // Otherwise, it goes into the bottom cluster list.
400             else { botClusterList.add(cluster); }
401         }
402         
403         // Add the new cluster lists to the cluster queues.
404         topClusterQueue.add(topClusterList);
405         botClusterQueue.add(botClusterList);
406         
407         // Remove the oldest cluster lists from the queues.
408         topClusterQueue.remove();
409         botClusterQueue.remove();
410     }
