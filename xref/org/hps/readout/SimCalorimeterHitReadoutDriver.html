3   import java.util.Collection;
4   import java.util.List;
5   import java.util.Set;

7   import org.hps.readout.util.collection.LCIOCollection;
8   import org.hps.readout.util.collection.TriggeredLCIOData;
9   import org.lcsim.event.MCParticle;
10  import org.lcsim.event.SimCalorimeterHit;
11  
12  /**
13   * <code>SimCalorimeterHitReadoutDriver</code> handles SLIC objects
14   * in input Monte Carlo files of type {@link
15   * org.lcsim.event.SimCalorimeterHit SimCalorimeterHit}.
16   * 
17   * @see org.hps.readout.SLICDataReadoutDriver
18   */
19  public class SimCalorimeterHitReadoutDriver extends SLICDataReadoutDriver<SimCalorimeterHit> {
20      /**
21       * Instantiate an instance of {@link
22       * org.hps.readout.SLICDataReadoutDriver SLICDataReadoutDriver}
23       * for objects of type {@link
24       * org.lcsim.event.SimCalorimeterHit SimCalorimeterHit} and set
25       * the appropriate LCIO flags.
26       */
27      public SimCalorimeterHitReadoutDriver() {
28          super(SimCalorimeterHit.class, 0xe0000000);
29      }
30      
31      @Override
32      protected Collection<TriggeredLCIOData<?>> getOnTriggerData(double triggerTime) {
33          // If hodoscope hits are not persisted, truth data doesn't
34          // need to be written out.
35          if(!isPersistent()) { return null; }
36          
37          // Get the truth hits in the indicated time range.
38          Collection<SimCalorimeterHit> truthHits = ReadoutDataManager.getData(triggerTime - getReadoutWindowBefore(),
39                  triggerTime + getReadoutWindowAfter(), collectionName, SimCalorimeterHit.class);
40          
41          // MC particles need to be extracted from the truth hits
42          // and included in the readout data to ensure that the
43          // full truth chain is available.
44          Set<MCParticle> truthParticles = new java.util.HashSet<MCParticle>();
45          for(SimCalorimeterHit simHit : truthHits) {
46              for(int i = 0; i < simHit.getMCParticleCount(); i++) {
47                  ReadoutDataManager.addParticleParents(simHit.getMCParticle(i), truthParticles);
48              }
49          }
50          
51          // Create the truth MC particle collection.
52          LCIOCollection<MCParticle> truthParticleCollection = ReadoutDataManager.getCollectionParameters("MCParticle", MCParticle.class);
53          TriggeredLCIOData<MCParticle> truthParticleData = new TriggeredLCIOData<MCParticle>(truthParticleCollection);
54          truthParticleData.getData().addAll(truthParticles);
55          
56          // Create a list to store the output data.
57          List<TriggeredLCIOData<?>> output = new java.util.ArrayList<TriggeredLCIOData<?>>(2);
58          output.add(truthParticleData);
59          
60          // Return the result.
61          return output;
62      }
63      
64      @Override
65      protected double getTimeNeededForLocalOutput() {
66          return isPersistent() ? getReadoutWindowAfter() : 0;
67      }
68  }
