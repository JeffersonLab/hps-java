25  /**
26   * <code>SinglesTrigger2019ReadoutDriver</code> simulates an HPS singles trigger
27   * for 2019 MC. It takes in clusters produced by the
28   * {@link org.hps.readout.ecal.updated.GTPClusterReadoutDriver
29   * GTPClusterReadoutDriver} and hodoscope patterns produced by the
30   * {@link HodoscopePatternReadoutDriver}, and perform the necessary trigger
31   * logic on them. If a trigger is detected, it is sent to the readout data
32   * manager so that a triggered readout event may be written.
33   */
34  public class SinglesTrigger2019ReadoutDriver extends TriggerDriver {     
35      // ==============================================================
36      // ==== LCIO Collections ========================================
37      // ==============================================================
38      /**
39       * Indicates singles trigger type. Corresponding DAQ configuration is accessed by DAQ
40       * configuration system, and applied into readout.
41       */
42      private String triggerType = "singles3";   
43      
44      /**
45       * Indicates the name of the calorimeter geometry object. This is
46       * needed to allow access to the calorimeter channel listings.
47       */
48      private String ecalGeometryName = "Ecal";
49      /**
50       * Specifies the name of the LCIO collection containing the input
51       * GTP clusters that are used for triggering.
52       */
53      private String inputCollectionNameEcal = "EcalClustersGTP";
54      
55      private String inputCollectionNameHodo = "HodoscopePatterns";
56      
57      // ==============================================================
58      // ==== Driver Options ==========================================
59      // ==============================================================
60      
61      /**
62       * Specifies the beam energy for the input data. This defines the
63       * limits of the energy trigger plots and has no further effect.
64       */
65      private double beamEnergy = 4.55;
66      /**
67       * Stores the trigger settings and performs trigger logic.
68       */
69      private TriggerModule2019 triggerModule = new TriggerModule2019();
70      
71      // ==============================================================
72      // ==== Driver Parameters =======================================
73      // ==============================================================
74      
75      /**
76       * Tracks the current local time in nanoseconds for this driver.
77       */
78      private double localTime = 0.0;
79      /**
80       * Stores a reference to the calorimeter subdetector model. This
81       * is needed to extract the crystal indices from the cell ID.
82       */
83      private HPSEcal3 ecal = null;
84      /**
85       * Defines the size of an energy bin for trigger output plots.
86       */
87      private static final double BIN_SIZE = 0.025;      
88      
89      
90      // ==============================================================
91      // ==== AIDA Plots ==============================================
92      // ==============================================================
93      
94      private AIDA aida = AIDA.defaultInstance();
95      private static final int NO_CUTS = 0;
96      private static final int WITH_CUTS = 1;
97      private IHistogram1D[] clusterSeedEnergy = new IHistogram1D[2];
98      private IHistogram1D[] clusterHitCount = new IHistogram1D[2];
99      private IHistogram1D[] clusterTotalEnergy = new IHistogram1D[2];
100     private IHistogram2D[] clusterDistribution = new IHistogram2D[2];
101   
102     /**
103      * Sets whether or not the DAQ configuration is applied into the driver
104      * the EvIO data stream or whether to read the configuration from data files.
105      * 
106      * @param state - <code>true</code> indicates that the DAQ configuration is
107      * applied into the readout system, and <code>false</code> that it
108      * is not applied into the readout system.
109      */
110     public void setDaqConfigurationAppliedintoReadout(boolean state) {
111         // If the DAQ configuration should be read, attach a listener
112         // to track when it updates.               
113         if (state) {
114             ConfigurationManager2019.addActionListener(new ActionListener() {
115                 @Override
116                 public void actionPerformed(ActionEvent e) {
117                     // Get the DAQ configuration.
118                     DAQConfig2019 daq = ConfigurationManager2019.getInstance();  
119                     if(triggerType.contentEquals(SINGLES3)) triggerModule.loadDAQConfiguration(daq.getVTPConfig().getSingles3Config());
120                     else if(triggerType.equals(SINGLES2)) triggerModule.loadDAQConfiguration(daq.getVTPConfig().getSingles2Config());
121                     else if(triggerType.equals(SINGLES1)) triggerModule.loadDAQConfiguration(daq.getVTPConfig().getSingles1Config());
122                     else if(triggerType.equals(SINGLES0)) triggerModule.loadDAQConfiguration(daq.getVTPConfig().getSingles0Config());
123                 }
124             });
125         }                               
126     }

128     @Override
129     public void detectorChanged(Detector detector) {
130         // Get the calorimeter sub-detector.
131         org.lcsim.geometry.compact.Subdetector ecalSub = detector.getSubdetector(ecalGeometryName);
132         if(ecalSub instanceof HPSEcal3) {
133             ecal = (HPSEcal3) ecalSub;
134         } else {
135             throw new IllegalStateException("Error: Unexpected calorimeter sub-detector of type \"" + ecalSub.getClass().getSimpleName() + "; expected HPSEcal3.");
136         }        
137     }

139     @Override
140     public void process(EventHeader event) {                
141         // Check that clusters are available for the trigger.
142         Collection<Cluster> clusters = null;
143         Collection<HodoscopePattern> hodoPatterns = null;
144         ArrayList<HodoscopePattern> hodoPatternList = null;
145         if(ReadoutDataManager.checkCollectionStatus(inputCollectionNameEcal, localTime) && ReadoutDataManager.checkCollectionStatus(inputCollectionNameHodo, localTime)) {
146             clusters = ReadoutDataManager.getData(localTime, localTime + 4.0, inputCollectionNameEcal, Cluster.class);
147             hodoPatterns = ReadoutDataManager.getData(localTime, localTime + 4.0, inputCollectionNameHodo, HodoscopePattern.class);
148             
149             localTime += 4.0;
150             
151             if(clusters.size() == 0 || hodoPatterns.size() == 0) return;
152                        
153             hodoPatternList = new ArrayList<>(hodoPatterns);
154             
155         } else { return; }
156         
157         // Track whether or not a trigger was seen.
158         boolean triggered = false;
159         
160         // There is no need to perform the trigger cuts if the
161         // trigger is in dead time, as no trigger may be issued
162         // regardless of the outcome.
163         if(isInDeadTime()) { return; }
164         
165         // Record top/bot status for singles triggers
166         List<String> topBot = new ArrayList();
167         
168         // Plot the trigger distributions before trigger cuts are
169         // performed.
170         for(Cluster cluster : clusters) {
171             // Get the x and y indices. Note that LCSim meta data is
172             // not available during readout, so crystal indices must
173             // be obtained directly from the calorimeter geometry.
174             java.awt.Point ixy = ecal.getCellIndices(cluster.getCalorimeterHits().get(0).getCellID());
175             
176             // Populate the uncut plots.
177             clusterSeedEnergy[NO_CUTS].fill(TriggerModule2019.getValueClusterSeedEnergy(cluster));
178             clusterTotalEnergy[NO_CUTS].fill(TriggerModule2019.getValueClusterTotalEnergy(cluster));
179             clusterHitCount[NO_CUTS].fill(TriggerModule2019.getClusterHitCount(cluster));
180             clusterDistribution[NO_CUTS].fill(ixy.x, ixy.y);           
181             
182             // Perform the hit count cut.
183             if(triggerModule.getCutEn(TriggerModule2019.CLUSTER_HIT_COUNT_LOW_EN) && !triggerModule.clusterHitCountCut(cluster)) {
184                 continue;
185             }            
186             
187             // Perform the cluster energy cut.
188             if(triggerModule.getCutEn(TriggerModule2019.CLUSTER_TOTAL_ENERGY_LOW_EN) && !triggerModule.clusterTotalEnergyCutLow(cluster)) {
189                 continue;
190             }
191             
192             if(triggerModule.getCutEn(TriggerModule2019.CLUSTER_TOTAL_ENERGY_HIGH_EN) && !triggerModule.clusterTotalEnergyCutHigh(cluster)) {
193                 continue;
194             }
195             
196             // In the setup calorimeter geometry, range of X coordinates is [-23, -1] and [1, 23]. 
197             // The hardware uses cluster X coordinates [-22,0] and [1,23].
198             int clusterX = ixy.x;
199             if(clusterX < 0) clusterX++;
200             
201             int clusterY = ixy.y;
202             
203             // XMin is at least 0.
204             if(!triggerModule.getCutEn(TriggerModule2019.SINGLES_MOLLERMODE_EN)) {
205                 if(triggerModule.getCutEn(TriggerModule2019.CLUSTER_XMIN_EN) && !triggerModule.clusterXMinCut(clusterX)) {
206                     continue;
207                 }
208                 
209                 // XMin cut has been applied.
210                 if(triggerModule.getCutEn(TriggerModule2019.CLUSTER_PDE_EN) && !triggerModule.clusterPDECut(cluster, clusterX)) {
211                     continue;                
212                 } 
213             }
214             
215             if(triggerModule.getCutEn(TriggerModule2019.SINGLES_L1L2ECAL_MATCHING_EN) && !triggerModule.geometryMatchingCut(clusterX, ixy.y, hodoPatternList)) {
216                 continue;
217             }
218             
219             //For 2021 update, Moller triggers
220             if(triggerModule.getCutEn(TriggerModule2019.SINGLES_MOLLERMODE_EN)) {
221                 if(triggerModule.getCutEn(TriggerModule2019.SINGLES_XYMINMAX_EN) && !triggerModule.clusterXMinCut(clusterX)) {
222                     continue;
223                 }
224                 if(triggerModule.getCutEn(TriggerModule2019.SINGLES_XYMINMAX_EN) && !triggerModule.clusterXMaxCut(clusterX)) {
225                     continue;
226                 }            
227                 if(triggerModule.getCutEn(TriggerModule2019.SINGLES_XYMINMAX_EN) && !triggerModule.clusterYMinCut(clusterY)) {
228                     continue;
229                 }
230                 if(triggerModule.getCutEn(TriggerModule2019.SINGLES_XYMINMAX_EN) && !triggerModule.clusterYMaxCut(clusterY)) {
231                     continue;
232                 }
233                 if(triggerModule.getCutEn(TriggerModule2019.CLUSTER_PDE_EN) && !triggerModule.clusterMollerPDECut(cluster, clusterX)) {
234                     continue;                
235                 }  
236             }
237             // Note that a trigger occurred.
238             triggered = true;
239             
240             if(ixy.y > 0) topBot.add(TOP);
241             else topBot.add(BOT);
242             
243             // Populate the cut plots.
244             clusterSeedEnergy[WITH_CUTS].fill(TriggerModule2019.getValueClusterSeedEnergy(cluster));
245             clusterTotalEnergy[WITH_CUTS].fill(TriggerModule2019.getValueClusterTotalEnergy(cluster));
246             clusterHitCount[WITH_CUTS].fill(TriggerModule2019.getClusterHitCount(cluster));
247             clusterDistribution[WITH_CUTS].fill(ixy.x, ixy.y);
248         }
249         
250         if(triggered) {
251             boolean topStat = false;
252             boolean botStat = false;            
253             if(topBot.contains(TOP)) topStat = true;
254             if(topBot.contains(BOT)) botStat = true;
255             
256             if(topStat && botStat) sendTrigger(triggerType, TOPBOT);
257             else if(topStat) sendTrigger(triggerType, TOP);
258             else sendTrigger(triggerType, BOT);            
259         }
260     }

262     @Override
263     public void startOfData() {         
264         // Define the driver collection dependencies.
265         addDependency(inputCollectionNameEcal);
266         
267         addDependency(inputCollectionNameHodo);
268         
269         // Register the trigger.
270         ReadoutDataManager.registerTrigger(this);
271         
272         // Set the plot range based on the beam energy.
273         int bins = (int) Math.ceil((beamEnergy * 1.1) / BIN_SIZE);
274         double xMax = bins * BIN_SIZE;
275         
276         // Instantiate the trigger plots.
277         String[] postscripts = { " (No Cuts)", " (With Cuts)" };
278         for(int i = NO_CUTS; i <= WITH_CUTS; i++) {
279             clusterSeedEnergy[i] = aida.histogram1D("Trigger Plots\\Cluster Seed Energy Distribution" + postscripts[i], bins, 0.0, xMax);
280             clusterHitCount[i] = aida.histogram1D("Trigger Plots\\Cluster Hit Count Distribution" + postscripts[i], 10, -0.5, 9.5);
281             clusterTotalEnergy[i] = aida.histogram1D("Trigger Plots\\Cluster Total Energy Distribution" + postscripts[i], bins, 0.0, xMax);
282             clusterDistribution[i] = aida.histogram2D("Trigger Plots\\Cluster Seed Distribution" + postscripts[i], 46, -23, 23, 11, -5.5, 5.5);
283         }
284         
285         // Run the superclass method.
286         super.startOfData();
287     }

321     public void setTriggerType(String trigger) {
322         if(!trigger.equals(SINGLES0) && !trigger.equals(SINGLES1) && !trigger.equals(SINGLES2) && !trigger.equals(SINGLES3))
323             throw new IllegalArgumentException("Error: wrong trigger type name \"" + trigger + "\".");
324         triggerType = trigger;
325     }
