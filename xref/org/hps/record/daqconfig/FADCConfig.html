12  /**
13   * Class <code>FADCConfig</code> stores FADC configuration settings
14   * parsed from the an EVIO file. This class manages the following
15   * properties:
16   * <ul>
17   * <li>FADC Mode</li>
18   * <li>NSA</li>
19   * <li>NSB</li>
20   * <li>FADC Window Width</li>
21   * <li>FADC Window Time Offset</li>
22   * <li>Max Pulses per Channel Window</li>
23   * <li>Crystal Gains</li>
24   * <li>Crystal Thresholds</li>
25   * <li>Crystal Pedestals</li>
26   * </ul>
27   */
28  public class FADCConfig extends IDAQConfig {
29      // Store basic FADC information.
30      private int mode        = -1;
31      private int nsa         = -1;
32      private int nsb         = -1;
33      private int windowWidth = -1;
34      private int offset      = -1;
35      private int maxPulses   = -1;
36      
37      // Store a map of calorimeter channel number to crystal indices.
38      private Map<Point, Integer> indexChannelMap = new HashMap<Point, Integer>();
39      
40      // Store crystal calibrations and energy conversion factors.
41      private float[] gains = new float[443];
42      private int[] thresholds = new int[443];
43      private float[] pedestals = new float[443];
44      
45      // Store the calorimeter conditions table for converting between
46      // geometric IDs and channel objects.
47      private EcalChannelCollection geoMap = null;
48      
49      @Override
50      void loadConfig(EvioDAQParser parser) {
51          // Store the basic FADC information.
52          mode = parser.fadcMODE;
53          nsa = parser.fadcNSA;
54          nsb = parser.fadcNSB;
55          windowWidth = parser.fadcWIDTH;
56          offset = parser.fadcOFFSET;
57          maxPulses = parser.fadcNPEAK;
58          
59          // Get the channel collection from the database.
60          DatabaseConditionsManager database = DatabaseConditionsManager.getInstance();
61          geoMap = database.getCachedConditions(EcalChannelCollection.class, "ecal_channels").getCachedData();
62          
63          // Create a mapping of calorimeter crystal positions to their
64          // corresponding channels. Also place the mapped values into
65          // an array by their channel ID.
66          indexChannelMap.clear();
67          for(EcalChannel ecalChannel : geoMap) {
68              // Map the channel IDs to the crystal position.
69              int channel = ecalChannel.getChannelId();
70              int ix = ecalChannel.getX();
71              int iy = ecalChannel.getY();
72              indexChannelMap.put(new Point(ix, iy), new Integer(channel));
73              
74              // Place the mapped values into the arrays.
75              gains[ecalChannel.getChannelId()]      = parser.GAIN.get(ecalChannel);
76              pedestals[ecalChannel.getChannelId()]  = parser.PEDESTAL.get(ecalChannel);
77              thresholds[ecalChannel.getChannelId()] = parser.THRESHOLD.get(ecalChannel);
78          }
79      }

306     @Override
307     public void printConfig(PrintStream ps) {
308         // Print the basic configuration information.
309         ps.println("FADC Configuration:");
310         ps.printf("\tMode          :: %d%n", mode);
311         ps.printf("\tNSA           :: %d%n", nsa);
312         ps.printf("\tNSB           :: %d%n", nsb);
313         ps.printf("\tWindow Width  :: %d%n", windowWidth);
314         ps.printf("\tWindow Offset :: %d%n", offset);
315         ps.printf("\tMax Peaks     :: %d%n", maxPulses);
316         
317         // Output the pedestal/gain write-out header.
318         ps.println("\tix\tiy\tPedestal (ADC)\tGain (MeV/ADC)\tThreshold (ADC)");
319         
320         // Iterate over each crystal y-index.
321         yLoop:
322         for(int iy = -5; iy <= 5; iy++) {
323             // iy = 0 does not exists; skip it!
324             if(iy == 0) { continue yLoop; }
325             
326             // Iterate over each crystal x-index.
327             xLoop:
328             for(int ix = -23; ix <= 23; ix++) {
329                 // ix = 0 and the beam hole do not exist; skip these!
330                 if(ix == 0) { continue xLoop; }
331                 if((ix >= -10 && ix <= -2) && (iy == -1 || iy == 1)) {
332                     continue xLoop;
333                 }
334                 
335                 // Output the crystal indices, pedestal, and gain.
336                 int channelID = indexChannelMap.get(new Point(ix, iy));
337                 ps.printf("\t%3d\t%3d\t%8.3f\t%8.3f\t%4d%n", ix, iy,
338                         getPedestal(channelID), getGain(channelID), getThreshold(channelID));
339             }
340         }
341     }

343     /**
344      * Throws an exception if the argument channel ID is not within
345      * the allowed range.
346      * @param channelID - The channel ID to validate.
347      */
348     private static final void validateChannelID(int channelID) {
349         if(channelID < 1 || channelID > 442) {
350             throw new IndexOutOfBoundsException(String.format("Channel ID \"%d\" is invalid. Channel IDs must be between 1 and 442.", channelID));
351         }
352     }
