1   package org.hps.record.daqconfig;
2   
3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import java.util.ArrayList;
6   import java.util.List;
7   
8   /**
9    * Class <code>ConfigurationManager</code> provides static access to
10   * the DAQ configuration that can be parsed from EVIO files. It works
11   * in conjunction with the <code>DAQConfigDriver</code>, which obtains
12   * the configuration parser object when available and passes it to this
13   * manager, and <code>EvioDAQParser</code>, which parses the EVIO data.
14   * 
15   * @see DAQConfigDriver
16   * @see EvioDAQParser
17   */
18  public class ConfigurationManager {
19      // Store the configuration object.
20      private static final DAQConfig DAQ_CONFIG = new DAQConfig();
21      
22      // Track whether a DAQ configuration has been read yet.
23      private static boolean INITIALIZED = false;
24      
25      // Store listeners to alert other classes when an update occurs.
26      private static final List<ActionListener> AL_LIST = new ArrayList<ActionListener>();
27      private static final ActionEvent EVENT = new ActionEvent(DAQ_CONFIG, ActionEvent.RESERVED_ID_MAX + 12, "update");
28      
29      /**
30       * Gets an instance of the DAQ configuration settings object if it
31       * exists. If no configuration has been read, this will return
32       * <code>null</code> instead.
33       * @return Returns the DAQ settings as a <code>DAQConfig</code>
34       * object or <code>null</code>.
35       */
36      public synchronized static final DAQConfig getInstance() {
37          if(INITIALIZED) { return DAQ_CONFIG; }
38          else { return null; }
39      }
40      
41      /**
42       * Adds a listener to track when updates occur in the DAQ settings.
43       * @param listener - The listener.
44       */
45      public synchronized static final void addActionListener(ActionListener listener) {
46          if(listener != null) { AL_LIST.add(listener); }
47      }
48      
49      /**
50       * Gets the list of all listeners attached to the manager.
51       * @return Returns a <code>List</code> collection containing the
52       * <code>ActionListener</code> objects attached to the manager.
53       */
54      public static final List<ActionListener> getActionListeners() {
55          return AL_LIST;
56      }
57      
58      /**
59       * Indicates whether a DAQ configuration has been received yet.
60       * If <code>false</code>, then calls to <code>getInstance</code>
61       * will return <code>null</code>.
62       * @return Returns <code>true</code> if a DAQ configuration has
63       * been read and <code>false</code> otherwise.
64       */
65      public static final boolean isInitialized() {
66          return INITIALIZED;
67      }
68      
69      /**
70       * Removes an listener so that it will no longer receive updates
71       * when the DAQ configuration changes.
72       * @param listener - The listener to remove.
73       */
74      public synchronized static final void removeActionListener(ActionListener listener) {
75          if(listener != null) { AL_LIST.remove(listener); }
76      }
77      
78      /**
79       * Updates the DAQ configuration with the given EVIO parser. The
80       * manager will also note that it is initialized and inform any
81       * associated listeners that an update has occurred.
82       * @param parser - The updated DAQ information.
83       */
84      public synchronized static final void updateConfiguration(EvioDAQParser parser) {
85          INITIALIZED = false;
86          DAQ_CONFIG.loadConfig(parser);
87          INITIALIZED = true;
88          updateListeners();
89      }
90      
91      /**
92       * Sends an update event to all associated listeners.
93       */
94      private static final void updateListeners() {
95          for(ActionListener al : AL_LIST) {
96              al.actionPerformed(EVENT);
97          }
98      }
99  }
