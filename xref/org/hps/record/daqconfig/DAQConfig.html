1   package org.hps.record.daqconfig;
2   
3   import java.io.ByteArrayOutputStream;
4   import java.io.PrintStream;
5   
6   
7   /**
8    * Class <code>DAQConfig</code> holds all of the supported parameters
9    * from the DAQ configuration that exists in EVIO files. These values
10   * are stored in various subclasses appropriate to the parameter that
11   * are accessed through this primary interface.
12   */
13  public class DAQConfig extends IDAQConfig {
14      // Store the configuration objects.
15      private SSPConfig sspConfig = new SSPConfig();
16      private GTPConfig gtpConfig = new GTPConfig();
17      private FADCConfig fadcConfig = new FADCConfig();
18      
19      /**
20       * Gets the configuration parameters for the FADC.
21       * @return Returns the FADC configuration.
22       */
23      public FADCConfig getFADCConfig() {
24          return fadcConfig;
25      }
26      
27      /**
28       * Gets the configuration parameters for the GTP.
29       * @return Returns the GTP configuration.
30       */
31      public GTPConfig getGTPConfig() {
32          return gtpConfig;
33      }
34      
35      /**
36       * Gets the configuration parameters for the SSP.
37       * @return Returns the SSP configuration.
38       */
39      public SSPConfig getSSPConfig() {
40          return sspConfig;
41      }
42      
43      @Override
44      void loadConfig(EvioDAQParser parser) {
45          // Pass the configuration parser to the system-specific objects.
46          sspConfig.loadConfig(parser);
47          gtpConfig.loadConfig(parser);
48          fadcConfig.loadConfig(parser);
49          
50          // Print the loaded configuration to the terminal.
51          printConfig(System.out);
52      }
53  
54      @Override
55      public void printConfig(PrintStream ps) {
56          // Print the system-specific objects.
57          fadcConfig.printConfig(ps);
58          ps.println();
59          gtpConfig.printConfig(ps);
60          ps.println();
61          sspConfig.printConfig(ps);
62      }
63  
64      public String toString() {
65          ByteArrayOutputStream os = new ByteArrayOutputStream();
66          PrintStream ps = new PrintStream(os);
67          printConfig(ps);
68          try {
69              return os.toString("UTF8");
70          } catch (Exception e) {
71              throw new RuntimeException(e);
72          }
73      }
74  }
