1   package org.hps.record.daqconfig;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * Abstract class <code>AbstractCutConfig</code> holds a given number
8    * of values and allows implementing classes to access them. It also
9    * stores whether the configuration object is enabled.
10   */
11  abstract class AbstractConfig<E> {
12      // Store the cut values.
13      private final List<E> values;
14      private boolean enabled = false;
15      
16      /**
17       * Instantiates an <code>AbstractConfig</code> with the indicated
18       * number of values.
19       * @param count - The number of values that the object should store.
20       */
21      AbstractConfig(int count) {
22          // A configuration object must have at least one value.
23          if(count <= 0) {
24              throw new IllegalArgumentException("There must be at least one value.");
25          }
26          
27          // Instantiate the value array.
28          values = new ArrayList<E>(count);
29          for(int i = 0; i < count; i++) { values.add(null); }
30      }
31      
32      /**
33       * Gets the value of the cut with the associated value ID.
34       * @param valueIndex - The ID corresponding to the desired value.
35       * @return Returns the value as a parameterized <code>E</code>
36       * object.
37       */
38      protected E getValue(int valueIndex) {
39          validateValueIndex(valueIndex);
40          return values.get(valueIndex);
41      }
42      
43      /**
44       * Indicates whether the object is enabled or not.
45       * @return Returns <code>true</code> if the object is enabled and
46       * <code>false</code> otherwise.
47       */
48      public boolean isEnabled() {
49          return enabled;
50      }
51      
52      /**
53       * Sets the value corresponding to the argument value ID to a new
54       * object.
55       * @param valueIndex - The ID corresponding to the desired value.
56       * @param value - The new value.
57       */
58      protected void setValue(int valueIndex, E value) {
59          validateValueIndex(valueIndex);
60          values.set(valueIndex, value);
61      }
62      
63      /**
64       * Sets whether the configuration object is enabled.
65       * @param state <code>true</code> means that the object is enabled
66       * and <code>false</code> that it is disabled.
67       */
68      protected void setIsEnabled(boolean state) {
69          enabled = state;
70      }
71      
72      /**
73       * Throws an exception if the argument index does not correspond to
74       * any value.
75       * @param index - The index to check.
76       */
77      private final void validateValueIndex(int index) {
78          if(index < 0 || index >= values.size()) {
79              throw new IndexOutOfBoundsException("Value index \"" + index + "\" is invalid.");
80          }
81      }
82  }
