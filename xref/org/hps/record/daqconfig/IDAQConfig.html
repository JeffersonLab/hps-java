6   /**
7    * Interface <code>DAQConfig</code> represents a configuration bank
8    * generated from the DAQ configuration bank. This requires that all
9    * implementing classes have the ability to load settings from the DAQ
10   * bank parser and also print their contents to the terminal.
11   */
12  abstract class IDAQConfig {
13      /**
14       * Updates the stored settings based on the argument parser.
15       * @param parser - The EVIO DAQ bank parser.
16       */
17      abstract void loadConfig(EvioDAQParser parser);
18      
19      /**
20       * Prints a textual representation of the configuration bank to the
21       * terminal.
22       */
23      public abstract void printConfig(PrintStream ps);
24  }
