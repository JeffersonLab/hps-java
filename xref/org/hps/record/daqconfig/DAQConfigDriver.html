13  /**
14   * Class <code>DAQConfigDriver</code> is responsible for accessing the
15   * DAQ configuration settings, and then passing them to the associated
16   * class <code>ConfigurationManager</code> so that they can be accessed
17   * by other classes.<br/>
18   * <br/>
19   * The driver may accomplish this by two means. By default, it will
20   * check each event for an <code>EvioDAQParser</code> object which
21   * contains all of the DAQ configuration information and pass this to
22   * the <code>ConfigurationManager</code>. It will continue to update
23   * the <code>ConfigurationManager</code> during the run if new parser
24   * objects appear, and can thusly account for changing DAQ conditions.
25   * <br/><br/>
26   * The driver may also be set to read a DAQ configuration from text
27   * files containing the DAQ configuration bank information. To enable
28   * this mode, the parameter <code>readDataFiles</code> must be set to
29   * <code>true</code> and the parameters <code>runNumber</code> and also
30   * <code>filepath</code> must be defined. <code>runNumber</code> defines
31   * the run number of the configuration to be loaded and the parameter
32   * <code>filepath</code> defines the location of the data file repository.
33   * <br/><br/>
34   * This driver must be included in the driver chain if any other drivers
35   * in the chain rely on <code>ConfigurationManager</code>, as it can
36   * not be initialized otherwise.
37   * 
38   * @see ConfigurationManager
39   */
40  public class DAQConfigDriver extends Driver {
41      private int runNumber = -1;
42      private String filepath = null;
43      private boolean firstEvent = true;
44      private boolean readDataFiles = false;
45      private File[] dataFiles = new File[3];
46      private int[] crateNumber = { 46, 37, 39 };
47      
48      /**
49       * Verifies the parameter <code>filepath</code> for the data file
50       * repository and checks that appropriate data files exist for the
51       * requested run number if the driver is set to read from data files.
52       * Otherwise, this does nothing.
53       */
54      @Override
55      public void startOfData() {
56          // Check whether to use stored data files or the EvIO data stream
57          // as the source of the DAQ settings. Nothing needs to be done
58          // in the latter case.
59          if(readDataFiles) {
60              // The user must define a data file prefix and repository
61              // location for this option to be used.
62              if(filepath == null) {
63                  throw new NullPointerException("DAQ settings repository filepath must be defined.");
64              } 
65              if(runNumber == -1) {
66                  throw new NullPointerException("Run number must be defined.");
67              }
68              
69              // Verify that the repository actually exist.
70              File repository = new File(filepath);
71              if(!repository.exists() || !repository.isDirectory()) {
72                  throw new IllegalArgumentException("Repository location \"" + filepath + "\" must be an existing directory.");
73              }
74              
75              // Define the data file objects.
76              for(int i = 0; i < dataFiles.length; i++) {
77                  try {
78                      dataFiles[i] = new File(repository.getCanonicalPath() + "/" + runNumber + "_" + crateNumber[i] + ".txt");
79                  } catch(IOException e) {
80                      throw new RuntimeException("Error resolving absolute repository filepath.");
81                  }
82              }
83              
84              // Verify that the data files actually exist.
85              for(File dataFile : dataFiles) {
86                  if(!dataFile.exists() || !dataFile.canRead()) {
87                      throw new IllegalArgumentException("Data file \"" + dataFile.getName() + "\" does not exist or can not be read.");
88                  }
89              }
90          }
91      }
92      
93      /**
94       * Checks an event for the DAQ configuration banks and passes them
95       * to the <code>ConfigurationManager</code> if the driver is set to
96       * read from the EvIO data stream. Otherwise, this will parse the
97       * data files on the first event and then do nothing.
98       * @param event - The current LCIO event.
99       */
100     @Override
101     public void process(EventHeader event) {
102         // If this is the first event and data files are to be read,
103         // import the data files and generate the DAQ information.
104         if(firstEvent && readDataFiles) {
105             // Get the data files in the form of a data array.
106             String[][] data;
107             try { data = getDataFileArrays(dataFiles); }
108             catch(IOException e) {
109                 throw new RuntimeException("An error occurred when processing the data files.");
110             }
111             
112             // Instantiate an EvIO DAQ parser and feed it the data.
113             EvioDAQParser daqConfig = new EvioDAQParser();
114             for(int i = 0; i < dataFiles.length; i++) {
115                 daqConfig.parse(crateNumber[i], runNumber, data[i]);
116             }
117             
118             // Update the configuration manager.
119             ConfigurationManager.updateConfiguration(daqConfig);
120         }
121         
122         // Check if a trigger configuration bank exists.
123         if(!readDataFiles && event.hasCollection(EvioDAQParser.class, "TriggerConfig")) {
124             // Get the trigger configuration bank. There should only be
125             // one in the list.
126             List<EvioDAQParser> configList = event.get(EvioDAQParser.class, "TriggerConfig");
127             EvioDAQParser daqConfig = configList.get(0);
128             
129             // Get the DAQ configuration and update it with the new
130             // configuration object.
131             ConfigurationManager.updateConfiguration(daqConfig);
132         }
133         
134         // Note that it is no longer the first event.
135         firstEvent = false;
136     }

138     /**
139      * Converts DAQ configuration data files into an array of strings
140      * where each array entry represents a line in the configuration
141      * file. The first array index of the returned object corresponds
142      * to the file, and the second array index corresponds to the line.
143      * @param dataFiles - An array of <code>File</code> objects pointing
144      * to the data files that are to be converted. These are expected
145      * to be plain text files.
146      * @return Returns a two-dimensional array of <code>String</code>
147      * objects where the first array index corresponds to the object
148      * of the same index in the <code>File</code> array and the second
149      * array index corresponds to the lines in the file referenced by
150      * the <code>File</code> object.
151      * @throws IOException Occurs if there is an issue with accessing
152      * or reading the objects in the objects referred to by the files
153      * pointed to in the <code>dataFiles</code> array.
154      */
155     private static final String[][] getDataFileArrays(File[] dataFiles) throws IOException {
156         // Create file readers to process the data files.
157         FileReader[] fr = new FileReader[dataFiles.length];
158         BufferedReader[] reader = new BufferedReader[dataFiles.length];
159         for(int i = 0; i < dataFiles.length; i++) {
160             fr[i] = new FileReader(dataFiles[i]);
161             reader[i] = new BufferedReader(fr[i]);
162         }
163         
164         // Convert the reader streams into line-delimited strings.
165         String[][] data = getDataFileArrays(reader);
166         
167         // Close the readers.
168         for(int i = 0; i < dataFiles.length; i++) {
169             reader[i].close();
170             fr[i].close();
171         }
172         
173         // Return the data array.
174         return data;
175     }

177     /**
178      * Converts DAQ configuration data streams into an array of strings
179      * where each array entry represents a line in the configuration
180      * file. The first array index of the returned object corresponds
181      * to the stream, and the second array index corresponds to the line.
182      * @param reader - An array of <code>BufferedReader</code> objects
183      * containing DAQ trigger configuration crate data.
184      * @return Returns a two-dimensional array of <code>String</code>
185      * objects where the first array index corresponds to the object
186      * of the same index in the <code>BufferedReader</code> array and
187      * the second array index corresponds to the lines in the stream
188      * referenced by the <code>BufferedReader</code> object.
189      * @throws IOException Occurs if there is an issue with reading
190      * data stream.
191      */
192     protected static final String[][] getDataFileArrays(BufferedReader[] reader) throws IOException {
193         // Generate String arrays where each entry in the array is
194         // a line from the data file.
195         String[][] data = new String[reader.length][0];
196         for(int i = 0; i < reader.length; i++) {
197             // Create a list to hold the raw strings.
198             List<String> rawData = new ArrayList<String>();
199             
200             // Add each line from the current data file to the list
201             // as a single entry.
202             String curLine = null;
203             while((curLine = reader[i].readLine()) != null) {
204                 rawData.add(curLine);
205             }
206             
207             // Convert the list into a String array.
208             data[i] = rawData.toArray(new String[rawData.size()]);
209         }
210         
211         // Return the data array.
212         return data;
213     }

215     /**
216      * Gets the run number that the DAQConfigDriver is set to use. This
217      * will be <code>-1</code> in the event that the driver reads from
218      * an EvIO file.
219      * @return Returns the run number as an <code>int</code> primitive.

243     /**
244      * Sets whether or not to read the DAQ configuration directly from
245      * the EvIO data stream or whether to read the configuration from
246      * data files. Parameters <code>runNumber</code> and <code>filepath</code>
247      * must also be defined if this is set to <code>true</code>.
248      * @param state - <code>true</code> indicates that the configuration
249      * should be read from data files, and <code>false</code> that it
250      * should be read from the EvIO stream.

252     public void setReadDataFiles(boolean state) {
253         readDataFiles = state;
