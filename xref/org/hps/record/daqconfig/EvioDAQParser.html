212     /**
213      * Parses a set of configuration tables to obtain DAQ configuration
214      * parameters.
215      * @param crate - The crate associated with the configuration tables.
216      * @param runNumber - The run number for the current data set.
217      * @param configurationTables - Tables containing DAQ configuration
218      * parameters.
219      */
220     public void parse(int crate, int runNumber, String[] configurationTables) {
221         // Track the number of banks that have been parsed. If the
222         // parameter values have not been populated after a certain
223         // number of banks, there is missing information.
224         nBanks++;
225         
226         // Create a map that maps an identifier for each configuration
227         // parameter (its parameter key) to any values associated
228         // with it (its parameter values).
229         loadConfigMap(crate, configurationTables);
230         
231         // If debugging text is enabled, print the map to the terminal.
232         if(debug) { printMap(); }
233         
234         // If this run is known to be missing configuration values,
235         // handle the missing values.
236         fixConfigMap2014Run(runNumber);
237         
238         // Parse the previously generated configuration map and extract
239         // the DAQ configuration from it.
240         parseConfigMap();
241         
242         // If the expected number of banks have been parsed and debugging
243         // text is enabled, print out all of the parsed variables.
244         if(nBanks > 2 && debug) { printVars(); }
245     }

247     /**
248      * Converts the textual configuration information into a map, where
249      * the first column value becomes the map entry key and the remainder
250      * becomes the map entry value.
251      * @param crate - The calorimeter crate associated with the textual
252      * configuration data.
253      * @param configTables - An array of textual configuration tables that
254      * contain the DAQ configuration parameters.
255      */
256     private void loadConfigMap(int crate, String[] configTables) {
257         // Iterate over each configuration table.
258         for(String configTable : configTables) {
259             // Split each table into rows and iterate over the rows.
260             rowLoop:
261             for(String line : configTable.trim().split("\n")) {
262                 // Split the first column from the row.
263                 String[] cols = line.trim().split(" +", 2);
264                 
265                 // If there are fewer than two segments after the split,
266                 // then this is not a valid parameter entry.
267                 if(cols.length < 2) continue rowLoop;
268                 
269                 // The row name is the value of the first column. The
270                 // rest are typically values.
271                 String key = cols[0];
272                 List<String> vals = new ArrayList<String>(Arrays.asList(cols[1].trim().split(" +")));
273                 
274                 // If no values are present, this is not a valid entry.
275                 if (vals.size() < 1) {
276                     continue rowLoop;
277                 }
278                 
279                 // SPECIAL CASE:: Key "FADC250"
280                 // This entry marks parameter values for FADC channels,
281                 // such as pedestals, gains, and thresholds. These are
282                 // stored in separate lists from the other parameters.
283                 if (key.startsWith("FADC250")) {
284                     parseFADC(crate, key.trim(), vals);
285                 }
286                 
287                 // SPECIAL CASE: Key "SSP_HPS_SET_IO_SRC"
288                 // This entry indicates which triggers are enabled and
289                 // needs to be parsed differently than normal.
290                 else if(key.startsWith("SSP_HPS_SET_IO_SRC")) {
291                     // The first "parameter value" is a hardware code
292                     // that identifies the trigger. Obtain it.
293                     int trig = Integer.valueOf(vals.get(1));
294                     
295                     // There are two trigger of each type, singles and
296                     // pairs. Compare the hardware code to the codes
297                     // for each of these triggers to determine which
298                     // it this parameter entry represents and then set
299                     // its value appropriately.
300                     for (int ii = 0; ii < pairsIOsrc.length; ii++) {
301                         if(trig == singlesIOsrc[ii]) {
302                             singlesEn[ii] = true;
303                         }
304                         else if(trig == pairsIOsrc[ii]) {
305                             pairsEn[ii] = true;
306                         }
307                     }
308                 }
309                 
310                 // GENERAL CASE: Basic Parameter
311                 // This indicates a regular parameter that does not
312                 // require any special parsing.
313                 if(vals.size() > 1 && key.startsWith("SSP")) {
314                     // List the parameter by "[ROW NAME]_[KEY]" and
315                     // remove the key so that only the values remain.
316                     key += "_" + vals.remove(0);
317                 }
318                 
319                 // Add the parameter key and its values to the map.
320                 configMap.put(key, vals);
321             }
322         }
323     }

325     /**
326      * Parses the configuration parameter map entries and extracts the
327      * parameter values for those parameters which have the standard
328      * format of <code>[PARAMETER KEY] --> { [PARAMETER VALUES] }</code>.
329      */
330     public void parseConfigMap() {
331         // Parse simple FADC data.
332         fadcNSA    = Integer.valueOf(getConfigParameter("FADC250_NSA",      0));
333         fadcNSB    = Integer.valueOf(getConfigParameter("FADC250_NSB",      0));
334         fadcNPEAK  = Integer.valueOf(getConfigParameter("FADC250_NPEAK",    0));
335         fadcMODE   = Integer.valueOf(getConfigParameter("FADC250_MODE",     0));
336         fadcWIDTH  = Integer.valueOf(getConfigParameter("FADC250_W_WIDTH",  0));
337         fadcOFFSET = Integer.valueOf(getConfigParameter("FADC250_W_OFFSET", 0));
338         
339         // Parse GTP data.
340         gtpMinSeedEnergy = Integer.valueOf(getConfigParameter("GTP_CLUSTER_PULSE_THRESHOLD", 0));
341         gtpWindowBefore  = Integer.valueOf(getConfigParameter("GTP_CLUSTER_PULSE_COIN",      0));
342         gtpWindowAfter   = Integer.valueOf(getConfigParameter("GTP_CLUSTER_PULSE_COIN",      1));
343         
344         // Parse trigger data.
345         for(int ii = 0; ii < 2; ii++) {
346             // Check singles trigger cuts enabled status.
347             singlesNhitsEn[ii]         = getBoolConfigSSP(ii,  "SINGLES_NMIN",          1);
348             singlesEnergyMinEn[ii]     = getBoolConfigSSP(ii,  "SINGLES_EMIN",          1);
349             singlesEnergyMaxEn[ii]     = getBoolConfigSSP(ii,  "SINGLES_EMAX",          1);
350             
351             // Check pair trigger cuts enabled status.
352             pairsEnergySumMaxMinEn[ii] = getBoolConfigSSP(ii,  "PAIRS_SUMMAX_MIN",      2);
353             pairsEnergyDiffEn[ii]      = getBoolConfigSSP(ii,  "PAIRS_DIFFMAX",         1);
354             pairsCoplanarityEn[ii]     = getBoolConfigSSP(ii,  "PAIRS_COPLANARITY",     1);
355             pairsEnergyDistEn[ii]      = getBoolConfigSSP(ii,  "PAIRS_ENERGYDIST",      2);
356             
357             // Get the singles trigger cuts.
358             singlesNhits[ii]           = getIntConfigSSP(ii,   "SINGLES_NMIN",          0);
359             singlesEnergyMin[ii]       = getIntConfigSSP(ii,   "SINGLES_EMIN",          0);
360             singlesEnergyMax[ii]       = getIntConfigSSP(ii,   "SINGLES_EMAX",          0);
361             
362             // Get the pair trigger cuts.
363             pairsNhitsMin[ii]          = getIntConfigSSP(ii,   "PAIRS_NMIN",            0);
364             pairsEnergyMin[ii]         = getIntConfigSSP(ii,   "PAIRS_EMIN",            0);
365             pairsEnergyMax[ii]         = getIntConfigSSP(ii,   "PAIRS_EMAX",            0);
366             pairsEnergySumMin[ii]      = getIntConfigSSP(ii,   "PAIRS_SUMMAX_MIN",      1);
367             pairsEnergySumMax[ii]      = getIntConfigSSP(ii,   "PAIRS_SUMMAX_MIN",      0);
368             pairsEnergyDiffMax[ii]     = getIntConfigSSP(ii,   "PAIRS_DIFFMAX",         0);
369             pairsCoplanarityMax[ii]    = getIntConfigSSP(ii,   "PAIRS_COPLANARITY",     0);
370             pairsTimeDiffMax[ii]       = getIntConfigSSP(ii,   "PAIRS_TIMECOINCIDENCE", 0);
371             pairsEnergyDistSlope[ii]   = getFloatConfigSSP(ii, "PAIRS_ENERGYDIST",      0);
372             pairsEnergyDistMin[ii]     = getIntConfigSSP(ii,   "PAIRS_ENERGYDIST",      1);
373         }
374     }

401     /**
402      * Parses FADC configuration parameter entries. These all have 17
403      * lines, the first of which is the parameter key and the subsequent
404      * being parameter values corresponding to the 16 FADC channels for
405      * the indicated FADC slot. These entries contain thresholds, gains,
406      * and pedestals.
407      * @param crate - The crate associated with this parameter entry.
408      * @param key - The parameter key.
409      * @param vals - A list of 16 values with indices corresponding to
410      * the FADC channel with which they are associated.
411      */
412     private void parseFADC(int crate, String key, List<String> vals) {
413         // The FADC slot is not stored on the same line as the other
414         // data and must be parsed and retained, as it is necessary
415         // for handling the subsequent lines. If this line is the
416         // FADC slot, store it.
417         if(key.equals("FADC250_SLOT")) {
418             thisFadcSlot = Integer.valueOf(vals.get(0));
419         }
420         
421         // Parse the channel thresholds.
422         else if(key.equals("FADC250_ALLCH_TET")) {
423             setChannelParsInt(crate, thisFadcSlot, THRESHOLD, vals);
424         }
425         
426         // Parse the channel pedestals.
427         else if(key.equals("FADC250_ALLCH_PED")) {
428             setChannelParsFloat(crate, thisFadcSlot, PEDESTAL, vals);    
429         }
430         
431         // Parse the channel gains.
432         else if(key.equals("FADC250_ALLCH_GAIN")) {
433             setChannelParsFloat(crate, thisFadcSlot, GAIN, vals);
434         }
435     }

437     /**
438      * Takes a list of 16 values (in argument <code>vals</code>) and
439      * maps the appropriate database calorimeter channel to the list
440      * value. This assumes that the <code>String</code> values should
441      * be parsed to <code>Float</code> objects.
442      * @param crate - The calorimeter crate associated with the values.
443      * @param slot - The FADC slot associated with the values.
444      * @param map - The map in which to place the values.
445      * @param vals - A <code>List</code> of 16 <code>String</code>
446      * objects representing the channel values. This should correspond
447      * to FADC channels 0 - 15.
448      */
449     private void setChannelParsFloat(int crate, int slot, Map<EcalChannel, Float> map, List<String> vals) {
450         // Iterate over each channel and map the database channel object
451         // to the corresponding list value.
452         for(int ii = 0; ii < 16; ii++) {
453             map.put(findChannel(crate, slot, ii), Float.valueOf(vals.get(ii)));
454         }
455     }

611     /**
612      * Gets a parameter value associated with a parameter key.
613      * @param key - The parameter key to which the value belongs.
614      * @param ival - The index of the desired parameter value.
615      * @return Returns the requested parameter value if it exists and
616      * returns <code>"0"</code> otherwise. In the event that a parameter
617      * can not be found, an error message is passed to the logger.
618      */
619     public String getConfigParameter(String key, int ival) {
620         // Check the parameter map for the requested parameter key.
621         if(configMap.containsKey(key)) {
622             // Get the list of values associated with this parameter key.
623             List<String> vals = configMap.get(key);
624             
625             // Check that the list of values contains a parameter for
626             // the requested parameter index. If it does, return it.
627             if (ival < vals.size()) { return configMap.get(key).get(ival); }
628             
629             // Otherwise, an error has occurred. Log this and return the
630             // default value of zero.
631             else {
632                 Logger.getLogger(this.getClass().getName()).log(Level.SEVERE, "ConfigMap TOO SHORT:   " + ival + " " + configMap.get(key));
633                 return "0";
634             }
635         }
636         
637         // If the key is not present...
638         else {
639             // If more than 2 banks have been read, the absence of a
640             // key represents an error. Log that this has occurred.
641             if(nBanks > 2) {
642                 Logger.getLogger(this.getClass().getName()).log(Level.SEVERE, "ConfigMap MISSING KEY:   " + key);
643             }
644             
645             // Return a default value of zero.
646             return "0";
647         }
648     }

650     /**
651      * Gets the database calorimeter channel for a channel defined by
652      * a crate number, FADC slot, and FADC channel.
653      * @param crate - The crate number.
654      * @param fadcSlot - The FADC slot.
655      * @param fadcChan - The FADC channel.
656      * @return Returns the database channel as a <code>EcalChannel</code>
657      * if it exists, and <code>null</code> if it does not.
658      */
659     public EcalChannel findChannel(int crate, int fadcSlot, int fadcChan) {
660         // Search through the database channels for a channel that
661         // matches the the argument parameters.
662         for (EcalChannel cc : channels) {
663             // A channel matches the argument if the slot and channel
664             // values are the same. Crate number must also match, but
665             // note that EcalChannel follows a different convention
666             // with respect to crate numbering.
667             if( ((cc.getCrate() - 1) * 2 == crate - 37) && (cc.getSlot() == fadcSlot) && (cc.getChannel() == fadcChan) ) {
668                 return cc;
669             }
670         }
671         
672         // If no matching channel is found, return null.
673         return null;
674     }
