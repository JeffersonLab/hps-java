1   package org.hps.record.composite;
2   
3   import org.hps.record.MaxRecordsException;
4   import org.hps.record.evio.EvioEventUtilities;
5   
6   /**
7    * A @{link CompositeProcessor} for throwing an error when the maximum number of records is reached or exceeded.
8    */
9   public class MaxRecordsProcessor extends CompositeRecordProcessor {
10  
11      /**
12       * The maximum number of records.
13       */
14      private final long maxRecords;
15  
16      /**
17       * The number of records received.
18       */
19      private long recordsReceived;
20  
21      /**
22       * Class constructor, with the maximum number of records.
23       *
24       * @param maxRecords the maximum number of records
25       */
26      public MaxRecordsProcessor(final long maxRecords) {
27          this.maxRecords = maxRecords;
28      }
29  
30      /**
31       * Process a record and check if max number of records was reached.
32       * <p>
33       * Only records with certain types are considered in this total, which are basically "physics" events when
34       * processing LCIO or EVIO files. For an ET system without any other record processing attached, all events count
35       * towards the total as it is not easy to tell generically which are physics data events.
36       *
37       * @param record the composite record to process
38       */
39      @Override
40      public void process(final CompositeRecord record) {
41          if (record.getLcioEvent() != null) {
42              // All LCSim events count as records.
43              ++this.recordsReceived;
44          } else if (record.getEvioEvent() != null) {
45              if (EvioEventUtilities.isPhysicsEvent(record.getEvioEvent())) {
46                  // Only EVIO physics events are counted.
47                  ++this.recordsReceived;
48              }
49          } else {
50              // Otherwise (ET only?) count all records.
51              ++this.recordsReceived;
52          }
53          if (this.recordsReceived >= this.maxRecords) {
54              // Throw exception if max records was reached or exceeded.
55              throw new MaxRecordsException("Maximum number of records received.", this.maxRecords);
56          }
57      }
58  }
