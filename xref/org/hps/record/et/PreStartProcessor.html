1   package org.hps.record.et;

3   import java.io.IOException;
4   
5   import org.hps.record.evio.EvioDetectorConditionsProcessor;
6   import org.jlab.coda.et.EtEvent;
7   import org.jlab.coda.jevio.EvioEvent;
8   import org.jlab.coda.jevio.EvioException;
9   import org.jlab.coda.jevio.EvioReader;
10  
11  /**
12   * An ET processor that will activate the conditions system from PRESTART events.
13   */
14  // FIXME: This class is currently unused in HPS Java.
15  public class PreStartProcessor extends EtEventProcessor {
16  
17      /**
18       * The EVIO processor that will activate the conditions system.
19       */
20      private final EvioDetectorConditionsProcessor conditionsProcessor;
21  
22      /**
23       * Class constructor.
24       *
25       * @param detectorName the name of the detector model
26       */
27      public PreStartProcessor(final String detectorName) {
28          this.conditionsProcessor = new EvioDetectorConditionsProcessor(detectorName);
29      }
30  
31      /**
32       * Process an ET event and activate the conditions system if applicable.
33       *
34       * @param event the <code>EtEvent</code> to process
35       */
36      @Override
37      public void process(final EtEvent event) {
38          EvioEvent evioEvent = null;
39          try {
40              evioEvent = new EvioReader(event.getDataBuffer()).parseNextEvent();
41              this.conditionsProcessor.startRun(evioEvent);
42          } catch (IOException | EvioException e) {
43              throw new RuntimeException(e);
44          }
45      }
46  }
