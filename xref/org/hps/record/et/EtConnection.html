24  /**
25   * A class for encapsulating the connection information for an ET client including the EtSystem and EtAttachment
26   * objects.
27   */
28  public class EtConnection {
29  
30      protected EtConnection() {
31      }
32  
33      /**
34       * Create an EtConnection with full list of configuration parameters.
35       *
36       * @param name the name of the ET system e.g. the buffer file on disk
37       * @param host the name of the network host
38       * @param port the port of the network host
39       * @param blocking <code>true</code> for blocking behavior
40       * @param queueSize the queue size
41       * @param prescale the event prescale factor or 0 for none
42       * @param stationName the name of the ET station
43       * @param stationPosition he position of the ET station
44       * @param waitMode the wait mode
45       * @param waitTime the wait time if using timed wait mode
46       * @param chunkSize the number of ET events to return at once
47       * @return the <code>EtConnection</code> created from the parameters
48       */
49      public static EtConnection createConnection(final String name, final String host, final int port,
50              final boolean blocking, final int queueSize, final int prescale, final String stationName,
51              final int stationPosition, final Mode waitMode, final int waitTime, final int chunkSize) {
52          try {
53  
54              // make a direct connection to ET system's tcp server
55              final EtSystemOpenConfig etConfig = new EtSystemOpenConfig(name, host, port);
56  
57              // create ET system object with verbose debugging output
58              final EtSystem sys = new EtSystem(etConfig, EtConstants.debugInfo);
59              sys.open();
60  
61              // configuration of a new station
62              final EtStationConfig stationConfig = new EtStationConfig();
63              // statConfig.setFlowMode(cn.flowMode);
64              // FIXME: Flow mode hard-coded.
65              stationConfig.setFlowMode(EtConstants.stationSerial);
66              if (!blocking) {
67                  stationConfig.setBlockMode(EtConstants.stationNonBlocking);
68                  if (queueSize > 0) {
69                      stationConfig.setCue(queueSize);
70                  }
71              }
72              // Set prescale.
73              if (prescale > 0) {
74                  // System.out.println("setting prescale to " + cn.prescale);
75                  stationConfig.setPrescale(prescale);
76              }
77  
78              // Create the station.
79              final EtStation stat = sys.createStation(stationConfig, stationName, stationPosition);
80  
81              // attach to new station
82              final EtAttachment att = sys.attach(stat);
83  
84              // Return new connection.
85              final EtConnection connection = new EtConnection(sys, att, stat, waitMode, waitTime, chunkSize);
86  
87              return connection;
88  
89          } catch (IOException | EtException | EtExistsException | EtClosedException | EtDeadException
90                  | EtTooManyException e) {
91              throw new RuntimeException("Failed to create ET connection.", e);
92          }
93      }
94  
95      /**
96       * Create an <code>EtConnection</code> with a set of default parameters.
97       *
98       * @return an <code>EtConnection</code> with default parameters
99       */
100     public static EtConnection createDefaultConnection() {
101         return createConnection("ETBuffer", "localhost", 11111, false, 0, 0, "MY_STATION", 1, Mode.TIMED, 5000000, 1);
102     }

134     /**
135      * A class constructor for internal convenience.
136      *
137      * @param param The connection parameters.
138      * @param sys The ET system.
139      * @param att The ET attachment.
140      * @param stat The ET station.
141      */
142     private EtConnection(final EtSystem sys, final EtAttachment att, final EtStation stat, final Mode waitMode,
143             final int waitTime, final int chunkSize) {
144         this.sys = sys;
145         this.att = att;
146         this.stat = stat;
147         this.waitMode = waitMode;
148         this.waitTime = waitTime;
149         this.chunkSize = chunkSize;
150     }
151 
152     /**
153      * Cleanup the ET connection.
154      */
155     public void cleanup() {
156         try {
157             if (!this.sys.alive()) {
158                 throw new RuntimeException("EtSystem is not alive!");
159             }
160             this.sys.detach(this.att);
161             this.sys.removeStation(this.stat);
162             this.sys.close();
163         } catch (final Exception e) {
164             e.printStackTrace();
165         }
166     }
167

195     /**
196      * Read an array of <code>EtEvent</code> objects from the ET server.
197      * <p>
198      * Method signature preserves all specific exception types in the throws clause so that the caller may easily
199      * implement their own error and state handling depending on the kind of error that was thrown.
200      *
201      * @return The array of EtEvents.
202      * @throws IOException if <code>getEvents</code> throws this exception type
203      * @throws EtException if <code>getEvents</code> throws this exception type
204      * @throws EtDeadException if <code>getEvents</code> throws this exception type
205      * @throws EtEmptyException if <code>getEvents</code> throws this exception type
206      * @throws EtBusyException if <code>getEvents</code> throws this exception type
207      * @throws EtTimeoutException if <code>getEvents</code> throws this exception type
208      * @throws EtWakeUpException if <code>getEvents</code> throws this exception type
209      * @throws EtClosedException if <code>getEvents</code> throws this exception type
210      */
211     public EtEvent[] readEtEvents() throws IOException, EtException, EtDeadException, EtEmptyException, EtBusyException,
212             EtTimeoutException, EtWakeUpException, EtClosedException {
213         return getEtSystem().getEvents(getEtAttachment(), this.waitMode, Modify.NOTHING, this.waitTime, this.chunkSize);
214     }
