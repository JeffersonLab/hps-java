3   /**
4    * When set this can be used to limit the number of processing stages that are executed by the
5    * {@link org.hps.record.composite.CompositeLoop}.
6    * <p>
7    * For example, if the <code>ProcessingStage</code> is set to <code>EVIO</code> then the <code>ET</code> and
8    * <code>EVIO</code> adapters will be activated but LCIO events will not be created or processed.
9    */
10  public enum ProcessingStage {
11      /**
12       * Execute only the reading of ET events from the server.
13       */
14      ET,
15      /**
16       * Execute reading of ET events and conversion to EVIO.
17       */
18      EVIO,
19      /**
20       * Execute full processing chain of ET to EVIO to LCIO.
21       */
22      LCIO
23  }
