1   package org.hps.record.evio;
2   
3   import java.io.File;
4   import java.io.FileFilter;
5   
6   /**
7    * This is a simple file filter that will accept EVIO files with a certain pattern to their file names:<br/>
8    * <i>FILENAME.evio.SEQUENCE</i>.
9    * <p>
10   * This matches the convention used by the CODA DAQ software.
11   */
12  public final class EvioFileFilter implements FileFilter {
13  
14      /**
15       * Return <code>true</code> if file is an EVIO file with the correct file naming convention.
16       *
17       * @return <code>true</code> if file is an EVIO file with correct file naming convention
18       */
19      @Override
20      public boolean accept(final File pathname) {
21          final boolean isEvio = pathname.getName().contains(".evio");
22          boolean hasSeqNum = false;
23          try {
24              EvioFileUtilities.getSequenceFromName(pathname);
25              hasSeqNum = true;
26          } catch (final Exception e) {
27          }
28          return isEvio && hasSeqNum;
29      }
30  }
