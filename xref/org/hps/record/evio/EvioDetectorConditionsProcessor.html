1   package org.hps.record.evio;
2   
3   import org.jlab.coda.jevio.BaseStructure;
4   import org.jlab.coda.jevio.EvioEvent;
5   import org.lcsim.conditions.ConditionsManager;
6   import org.lcsim.conditions.ConditionsManager.ConditionsNotFoundException;
7   
8   /**
9    * <p>
10   * This is an {@link EvioEventProcessor} for initializing the conditions system from EVIO events. The
11   * {@link #startRun(EvioEvent)} method will setup conditions from PRESTART events. The {@link #process(EvioEvent)}
12   * method will setup conditions from a head bank, if it is present in the event.
13   */
14  public class EvioDetectorConditionsProcessor extends EvioEventProcessor {
15  
16      /**
17       * The name of the detector model.
18       */
19      private final String detectorName;
20  
21      /**
22       * Class constructor.
23       *
24       * @param detectorName the name of the detector model
25       */
26      public EvioDetectorConditionsProcessor(final String detectorName) {
27          if (detectorName == null) {
28              throw new IllegalArgumentException("The detectorName argument is null.");
29          }
30          this.detectorName = detectorName;
31      }
32  
33      /**
34       * Process an <code>EvioEvent</code> and activate the conditions system if applicable.
35       *
36       * @param evioEvent the <code>EvioEvent</code> to process
37       */
38      @Override
39      public void process(final EvioEvent evioEvent) throws Exception {
40          
41          // Get the head head bank from event.
42          final BaseStructure headBank = EvioEventUtilities.getHeadBank(evioEvent);
43  
44          // Initialize from head bank.
45          if (headBank != null) {            
46              initializeConditions(headBank.getIntData()[1]);
47          }
48          
49          // Initialize from PRESTART.
50          if (EventTagConstant.PRESTART.matches(evioEvent)) {
51              int runNumber = EvioEventUtilities.getControlEventData(evioEvent)[1];
52              initializeConditions(runNumber);
53          }
54      }
55  
56      private void initializeConditions(final int runNumber) {
57          // Initialize the conditions system from the detector name and run number.
58          try {
59              ConditionsManager.defaultInstance().setDetector(this.detectorName, runNumber);
60          } catch (final ConditionsNotFoundException e) {
61              throw new RuntimeException("Error setting up conditions from EVIO head bank.", e);
62          }
63      }
64  
65      /**
66       * Start of run action.
67       * <p>
68       * This will only activate if the evioEvent is a PRESTART event.
69       *
70       * @param evioEvent the <code>EvioEvent</code> to process
71       */
72      @Override
73      // FIXME: not activated by EvioLoop
74      public void startRun(final EvioEvent evioEvent) {
75          // System.out.println("EvioDetectorConditionsProcessor.startRun");
76          if (EvioEventUtilities.isPreStartEvent(evioEvent)) {
77              // Get the pre start event's data bank.
78              final int[] data = EvioEventUtilities.getControlEventData(evioEvent);
79  
80              // Get the run number from the bank.
81              final int runNumber = data[1];
82  
83              // Initialize the conditions system from the detector name and run number.
84              try {
85                  // System.out.println("  setting up conditions from pre start: " + detectorName + " #" + runNumber);
86                  ConditionsManager.defaultInstance().setDetector(this.detectorName, runNumber);
87              } catch (final ConditionsNotFoundException e) {
88                  throw new RuntimeException("Error setting up conditions from EVIO pre start event.", e);
89              }
90          }
91      }
92  }
