106     /**
107      * The externally accessible main method.
108      *
109      * @param args the command line arguments
110      */
111     public static void main(final String[] args) {
112         EvioFileProducer producer = new EvioFileProducer();
113         try {
114             producer.parse(args);
115         } catch (Exception e) {
116             throw new RuntimeException("Error parsing command line", e);
117         }
118         try {
119             producer.run();
120         } catch (Exception e) {
121             throw new RuntimeException(e);
122         }
123     }

130         help.printHelp(80, "EvioFileProducer [options] file.evio.0 file.evio.1 [...]", "EVIO File Producer", OPTIONS,
131                 "Provide EVIO files as extra arguments or using the -l option to provide a file containing a list of them.");
132         System.exit(1);

181     /**
182      * Print the command line job configuration to the log.
183      */
184     private void logConfig() {
185         final StringBuffer sb = new StringBuffer();
186         sb.append("count: " + this.eventPrintInterval + '\n');
187         sb.append("delay: " + this.delay + '\n');
188         sb.append("etName: " + this.etName + '\n');
189         sb.append("group: " + this.group + '\n');
190         sb.append("host: " + this.host + '\n');
191         sb.append("port: " + this.port + '\n');
192         sb.append("size: " + this.size + '\n');
193         sb.append("EVIO files: " + '\n');
194         for (final File evioFile : this.evioFiles) {
195             sb.append(evioFile.getPath() + '\n');
196         }
197         LOG.config(sb.toString());
198     }

200     /**
201      * Run the job by streaming all the EVIO files to the ET server using the command line arguments.
202      *
203      * @param args the command line arguments
204      */
205     private void parse(final String[] args) throws ParseException {
206 
207         // Command line parser.
208         final DefaultParser parser = new DefaultParser();
209 
210         // Parse the command line arguments.
211         final CommandLine cl = parser.parse(OPTIONS, args);
212 
213         // Print usage and exit.
214         if (cl.hasOption("help")) {
215             printUsage();
216             System.exit(0);
217         }
218 
219         // Add EVIO files to the job from extra args.
220         List<String> extraArgs = cl.getArgList();
221         for (final String extraArg : extraArgs) {
222             final File evioFile = new File(extraArg);
223             LOG.config("Adding EVIO file: " + evioFile.getPath());
224             this.evioFiles.add(evioFile);
225         }
226 
227         // Set ET name which is the buffer file.
228         if (cl.hasOption("f")) {
229             this.etName = cl.getOptionValue("f");
230         }
231 
232         // Add EVIO files from a text file list, assuming one file path per line.
233         // Option cannot be used when providing a list of EVIO files on the command line
234         // or an exception will be thrown.
235         if (cl.hasOption("l")) {
236             if (cl.getArgList().size() > 0) {
237                 throw new RuntimeException("The -l option is not allowed with EVIO files on the command line.");
238             }
239             final String filePath = cl.getOptionValue("l");
240             final File listFile = new File(filePath);
241             if (!listFile.exists()) {
242                 throw new IllegalArgumentException("The file " + listFile.getPath() + " does not exist.");
243             }
244             BufferedReader br = null;
245             try {
246                 br = new BufferedReader(new InputStreamReader(new FileInputStream(listFile)));
247                 String line;
248                 while ((line = br.readLine()) != null) {
249                     this.evioFiles.add(new File(line.trim()));
250                 }
251             } catch (IOException e) {
252                 throw new RuntimeException("Error reading EVIO file list", e);
253             } finally {
254                 if (br != null) {
255                     try {
256                         br.close();
257                     } catch (IOException e) {
258                         e.printStackTrace();
259                     }
260                 }
261             }
262         }
263 
264         // Set host name.
265         if (cl.hasOption("h")) {
266             this.host = cl.getOptionValue("h");
267         }
268         // if (this.host == null) {
269         // host = EtConstants.hostAnywhere;
270         // this.host = InetAddress.getLocalHost().getHostName();
271         // }
272 
273         // Set the port number.
274         if (cl.hasOption("p")) {
275             this.port = Integer.parseInt(cl.getOptionValue("p"));
276             if (this.port < ET_PORT_MIN || this.port > ET_PORT_MAX) {
277                 throw new IllegalArgumentException("Port number must be between 1024 and 65535.");
278             }
279         }
280         // Set the size of the event buffer in bytes.
281         if (cl.hasOption("s")) {
282             this.size = Integer.parseInt(cl.getOptionValue("s"));
283             if (this.size < 1) {
284                 throw new IllegalArgumentException("Size needs to be a positive int.");
285             }
286         }
287 
288         // Set the group number.
289         if (cl.hasOption("g")) {
290             this.group = Integer.parseInt(cl.getOptionValue("g"));
291             if (this.group < 1 || this.group > 10) {
292                 throw new IllegalArgumentException("Group number must be between 0 and 10.");
293             }
294         }
295 
296         // Set the delay in milliseconds between putting events.
297         if (cl.hasOption("d")) {
298             this.delay = Integer.parseInt(cl.getOptionValue("d"));
299             if (this.delay < 1) {
300                 throw new IllegalArgumentException("The delay must be > 0.");
301             }
302         }
303 
304         if (cl.hasOption("e")) {
305             this.eventPrintInterval = Integer.parseInt(cl.getOptionValue("e"));
306             if (this.eventPrintInterval < 1) {
307                 throw new IllegalArgumentException("The count must be > 0.");
308             }
309         }
310 
311         // At least one EVIO file must be present from parsing command line arguments.
312         if (this.evioFiles.size() == 0) {
313             throw new RuntimeException("At least one EVIO file is required.");
314         }
315 
316         // Check existence of EVIO files.
317         LOG.info("Checking EVIO files");
318         if (this.evioFiles.size() == 0) {
319             throw new RuntimeException("No EVIO files were given by command line options!");
320         }
321         for (final File evioFile : this.evioFiles) {
322             if (!evioFile.exists()) {
323                 throw new IllegalArgumentException("EVIO file does not exist: " + evioFile.getPath());
324             }
325         }
326 
327         // Print out the configuration for the job to the log.
328         logConfig();
329     }

331     private void run() throws Exception {
332 
333         EtSystem sys = null;
334         EvioReader reader = null;
335 
336         try {
337 
338             // Setup ET system from the command line options.
339             LOG.info("Opening ET system on host " + this.host + ", port " + this.port + " and file " + this.etName);
340             final EtSystemOpenConfig config = new EtSystemOpenConfig(this.etName, this.host, this.port);
341             sys = new EtSystem(config, EtConstants.debugInfo);
342             // TODO: Add multiple connection attempts as the initial handshake tends to fail a lot
343             sys.open();
344             final EtStation gc = sys.stationNameToObject(GRAND_CENTRAL);
345             final EtAttachment att = sys.attach(gc);
346 
347             // Array of ET events.
348             EtEvent[] mevs;
349 
350             // Loop over input EVIO file list.
351             for (final File evioFile : this.evioFiles) {
352 
353                 // Open a new EVIO reader in sequential read mode so events are immediately streamed to server.
354                 LOG.info("Opening next EVIO file " + evioFile.getPath());
355                 reader = new EvioReader(evioFile.getPath(), false, true);
356 
357                 // Reference to the current EVIO event.
358                 EvioEvent event;
359 
360                 // Event sequence number.
361                 int eventCount = 0;
362 
363                 // Loop until event source is exhausted.
364                 while ((event = reader.nextEvent()) != null) {
365 
366                     try {
367                         // Parse the next EVIO event.
368                         reader.parseEvent(event);
369 
370                         // Increment event count.
371                         ++eventCount;
372 
373                         // Print event sequence.
374                         if (eventCount % this.eventPrintInterval == 0) {
375                             LOG.info("EVIO event: " + eventCount + ", bytes: " + event.getTotalBytes());
376                         }
377                     } catch (final Exception e) { /* Catches parse errors reading the EVIO events. */
378                         e.printStackTrace();
379                         LOG.severe("Error making EVIO event with seq number " + eventCount + " in file "
380                                 + evioFile.getPath());
381                         // Attempt to recover from errors by skipping to next event if there are exceptions.
382                         continue;
383                     }
384 
385                     final int eventTag = EvioEventUtilities.getEventTag(event);
386 
387                     LOG.finest("Creating new ET events...");
388                     // Create a new array of ET events. This always has one event.
389                     mevs = sys.newEvents(att, // attachment
390                             Mode.SLEEP, // wait mode
391                             false, // create a buffer
392                             0, // delay
393                             1, // number of events
394                             this.size, // size of event but overwritten later
395                             this.group); // group number; default value is arbitrary
396                     LOG.finest("Done creating ET events!");
397 
398                     // Create control data array for event selection.
399                     final int[] control = new int[EtConstants.stationSelectInts];
400                     Arrays.fill(control, eventTag);
401                     mevs[0].setControl(control);
402 
403                     // Apply delay in milliseconds.
404                     if (this.delay > 0) {
405                         Thread.sleep(this.delay);
406                     }
407 
408                     // Write the next EVIO event to the EtEvent's buffer.
409                     try {
410                         LOG.finest("Writing to buffer...");
411                         final ByteBuffer buf = mevs[0].getDataBuffer();
412                         buf.order(ByteOrder.nativeOrder());
413                         final EventWriter writer = new EventWriter(buf, 100000, 100, null, null);
414                         writer.writeEvent(event);
415                         LOG.finest("Done writing to buffer!");
416                         try {
417                             writer.close();
418                         } catch (final Exception e) {
419                             LOG.log(Level.WARNING, "Error while closing writer.", e);
420                         }
421                         mevs[0].setLength(buf.position());
422                         mevs[0].setByteOrder(ByteOrder.nativeOrder());
423 
424                         for (final EtEvent mev : mevs) {
425                             LOG.finest("event length = " + mev.getLength() + ", remaining bytes: "
426                                     + mev.getDataBuffer().remaining());
427                         }
428 
429                         // Put events onto the ET ring.
430                         LOG.finest("Putting events onto ET ring...");
431                         sys.putEvents(att, mevs);
432                         LOG.finest("Successfully wrote events to ET ring!");
433                     } catch (Exception e) {
434                         e.printStackTrace();
435                     }
436                 }
437                 LOG.info(eventCount + " events were read from " + evioFile.getPath());
438                 reader.close();
439             }
440         } finally {
441             // Cleanup the EVIO reader if needed.
442             if (reader != null && !reader.isClosed()) {
443                 try {
444                     reader.close();
445                 } catch (final IOException e) {
446                     LOG.log(Level.WARNING, e.getMessage(), e);
447                 }
448             }
449             // Cleanup the ET system.
450             if (sys != null && sys.alive()) {
451                 sys.close();
452             }
453         }
454 
455         LOG.info("Done!");
456     }
