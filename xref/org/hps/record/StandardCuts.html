8   public class StandardCuts {
9       // max number of hits a track can share with other tracks
10      private int maxSharedHitsPerTrack;
11      // max GBL chisq prob for all tracks 
12      private double maxTrackChisqProb;
13      // max GBL raw chisq for various numbers of hits
14      private Map<Integer, Double> maxTrackChisq;
15      // max (absolute) chisq for track-cluster match for recon particle 
16      private double maxMatchChisq;
17      // max time diff [ns] between track and cluster time for recon particle
18      private double maxMatchDt;
19      //max x diff [mm] between track and cluster for recon particle
20      private double maxMatchDx;
21      //max y diff [mm] between track and cluster for recon particle
22      private double maxMatchDy;
23      // max vertex momentum (magnitude)
24      private double maxVertexP;
25      // max chisq prob for V0 vertex fit
26      private double minVertexChisqProb;
27      // max chisq prob for vertex fit
28      private double minMollerChisqProb;
29      // max time diff [ns] between the two recon particle clusters in vertex
30      private double maxVertexClusterDt;
31      // max momentum for good electron recon particle
32      private double maxElectronP;
33      // min momentum sum of tracks in good Moller vertex
34      private double minMollerP;
35      // max momentum sum of tracks in good Moller vertex
36      private double maxMollerP;
37      // this should be set in data according to beam energy, but may be different in MC
38      private double trackClusterTimeOffset;

42      private boolean maxElectronPset = false;
43      private boolean minMollerPset = false;

46      private boolean OffsetSet = false;

48      private boolean maxMatchDtSet = false;
49      private boolean maxMatchDxSet = false;

74      public void setMaxTrackChisq(int nhits, double input) {
75          int dof = nhits*2-5;
76          maxTrackChisq.put(dof, input);
77      }
78      
79      public double getMaxTrackChisq(int nhits) {
80          int dof = nhits*2-5;
81          if (!maxTrackChisq.containsKey(dof)) {
82              maxTrackChisq.put(dof, new ChiSquaredDistribution(dof).inverseCumulativeProbability(1.0-maxTrackChisqProb));
83          }
84          return maxTrackChisq.get(dof);
85      }

188     public StandardCuts(double ebeam) {
189         maxSharedHitsPerTrack = 5;
190         maxMatchChisq = 10.0;
191         maxMatchDt = 6.0;
192         maxMatchDx = 20.0;
193         maxMatchDy = 20.0;
194         maxVertexClusterDt = 2.0;
195         minVertexChisqProb = 0.00001;
196         minMollerChisqProb = 0.00001;
197         maxTrackChisqProb = 0.00001;
198         
199         maxElectronPset = false;
200         minMollerPset = false;
201         maxMollerPset = false;
202         maxVertexPset = false;
203         OffsetSet = false;
204         
205         maxTrackChisq = new HashMap<Integer, Double>();
206         maxTrackChisq.put(5, new ChiSquaredDistribution(5).inverseCumulativeProbability(1.0-maxTrackChisqProb));
207         maxTrackChisq.put(7, new ChiSquaredDistribution(7).inverseCumulativeProbability(1.0-maxTrackChisqProb));
208         
209         maxElectronP = 0.75*ebeam;
210         minMollerP = 0.8*ebeam;
211         maxMollerP = 1.2*ebeam;
212         maxVertexP = 1.2*ebeam;
213         if (ebeam < 2)
214             trackClusterTimeOffset=43;
215         else
216             trackClusterTimeOffset=55;
217     }
218     
219     public void changeChisqTrackProb(double prob) {        
220         for (int dof : maxTrackChisq.keySet()) {
221             maxTrackChisq.put(dof, new ChiSquaredDistribution(dof).inverseCumulativeProbability(1.0-prob));
222         }
223     }
224     
225     public void changeBeamEnergy(double ebeam) {
226         setMaxElectronP(0.75*ebeam);
227         setMinMollerP(0.8*ebeam);
228         setMaxMollerP(1.2*ebeam);
229         setMaxVertexP(1.2*ebeam);
230         if (ebeam < 2)
231             setTrackClusterTimeOffset(43);
232         else
233             setTrackClusterTimeOffset(55);
234     }
