3   import java.nio.charset.StandardCharsets;
4   import java.util.ArrayList;
5   import java.util.List;
6   import java.util.logging.Logger;
7   
8   import org.hps.record.evio.EvioEventProcessor;
9   import org.hps.record.evio.EvioEventUtilities;
10  import org.hps.record.svt.SvtConfigData.RocTag;
11  import org.jlab.coda.jevio.BaseStructure;
12  import org.jlab.coda.jevio.EvioEvent;
13  
14  /**
15   * Get a list of SVT config data from an EVIO data stream.
16   * 
17   * @see SvtConfigData
18   */
19  public class SvtConfigEvioProcessor extends EvioEventProcessor {
20  
21      private static Logger LOGGER = Logger.getLogger(SvtConfigEvioProcessor.class.getPackage().getName());
22  
23      private static final int DATA_TAG = 51;
24      private static final int CONTROL_TAG = 66;
25      private static final int CONFIG_TAG = 57614;
26  
27      private List<SvtConfigData> configs = new ArrayList<SvtConfigData>();
28  
29      private int timestamp = 0;
30      
31      public void process(EvioEvent evioEvent) {
32          SvtConfigData config = null;
33          BaseStructure headBank = EvioEventUtilities.getHeadBank(evioEvent);
34          int configBanks = 0;
35          if (headBank != null) {
36              if (headBank.getIntData()[3] != 0) {
37                  timestamp = headBank.getIntData()[3];
38                  //LOGGER.finest("set timestamp " + timestamp + " from head bank");
39              }
40          }
41          for (BaseStructure bank : evioEvent.getChildrenList()) {
42              if (bank.getHeader().getTag() == DATA_TAG || bank.getHeader().getTag() == CONTROL_TAG) {
43                  if (bank.getChildrenList() != null) {
44                      for (BaseStructure subBank : bank.getChildrenList()) {
45                          if (subBank.getHeader().getTag() == CONFIG_TAG) {
46                              String[] stringData = subBank.getStringData();
47                              if (stringData == null) {
48                                  LOGGER.warning("string data is null");
49                                  if (subBank.getRawBytes() != null) {
50                                      LOGGER.info("raw byte array len " + subBank.getRawBytes().length);
51                                      LOGGER.info("cnv to string data" + '\n' + new String(subBank.getRawBytes(), StandardCharsets.UTF_8));
52                                  } else {
53                                      LOGGER.warning("Raw byte array is null.");
54                                  }
55                              } else {
56                                  if (config == null) {
57                                      config = new SvtConfigData(timestamp);
58                                  }
59                                  if (stringData.length > 0) {
60                                      System.out.println("found string data with length " + stringData.length);
61                                      for (int i = 0; i < stringData.length; i++) {
62                                          System.out.println("Printing raw string data " + i + " ...");
63                                          System.out.println(stringData[i]);
64                                          System.out.println("End print raw string data");
65                                      }                                                                       
66                                      if (!stringData[0].trim().isEmpty()) {
67                                          LOGGER.info("Adding SVT config data with len " + stringData[0].length() + " ..." + '\n' + stringData[0]);                                        
68                                          config.setData(RocTag.fromTag(bank.getHeader().getTag()), stringData[0]);
69                                          ++configBanks;
70                                      } else {
71                                          LOGGER.warning("String data has no content.");
72                                      }
73                                  } else {
74                                      LOGGER.warning("String data has zero len.");
75                                  }
76                              }
77                          }
78                      }
79                  }
80              }
81          }
82          if (config != null) {
83              LOGGER.info("Adding SVT config " + evioEvent.getEventNumber() + " with " + configBanks
84                      + " banks and timestamp " + timestamp + " from event " + evioEvent.getEventNumber());
85              this.configs.add(config);
86          }
87      }
88  
89      public List<SvtConfigData> getSvtConfigs() {
90          return configs;
91      }
92  }
