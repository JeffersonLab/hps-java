1   package org.hps.record.svt;
2   
3   import java.util.HashSet;
4   import java.util.List;
5   import java.util.Set;
6   
7   /**
8    * Represents information about errors found in SVT EVIO headers.
9    * 
10   * Replaces direct usage of {@link SvtEvioExceptions.SvtEvioHeaderException} when
11   * checking event header information using {@link SvtEventHeaderCheckerNew}.
12   */
13  public class EvioHeaderError {
14     
15      /**
16       * The basic error types.
17       */
18      public enum ErrorType {
19          ApvBufferAddress,
20          ApvFrameCount,
21          ApvRead,
22          MultisampleErrorBit,
23          Overflow,
24          SkipCount,
25          Sync
26      }
27      
28      /*
29       * A set of static class instances are provided here with default error messages to avoid creating new objects when
30       * it isn't necessary.
31       */
32      
33      public static final EvioHeaderError APV_BUFFER_ADDRESS_COUNT = new EvioHeaderError(ErrorType.ApvBufferAddress, 
34              "Invalid number of APV buffer addresses.");
35          
36      public static final EvioHeaderError APV_BUFFER_ADDRESS_MISMATCH = new EvioHeaderError(ErrorType.ApvBufferAddress,
37              "The APV buffer addresses in this event do not match");
38      
39      public static final EvioHeaderError APV_FRAME_COUNT = new EvioHeaderError(ErrorType.ApvFrameCount, 
40              "Invalid number of APV frame counts.");
41      
42      public static final EvioHeaderError APV_FRAME_COUNT_MISMATCH = new EvioHeaderError(ErrorType.ApvFrameCount, 
43              "The APV frame counts in this event do not match.");
44     
45      public static final EvioHeaderError APV_READ_ERRORS = new EvioHeaderError(ErrorType.ApvRead, 
46              "Invalid number of read errors.");
47      
48      public static final EvioHeaderError MULTISAMPLE_ERROR_BIT = new EvioHeaderError(ErrorType.MultisampleErrorBit, 
49              "A multisample header error bit was set.");
50      
51      public static final EvioHeaderError SYNC = new EvioHeaderError(ErrorType.Sync,
52              "The header had a sync error.");
53      
54      public static final EvioHeaderError OVERFLOW = new EvioHeaderError(ErrorType.Overflow,
55              "The header had an overflow error.");
56      
57      public static final EvioHeaderError SKIP_COUNT = new EvioHeaderError(ErrorType.SkipCount,
58              "The header had a skip count.");
59      
60      private final String debugString;
61      
62      private final ErrorType errorType;
63      
64      private final String message;
65      
66      /**
67       * Get a list of unique error names from a list.
68       * @param errors The list of errors
69       * @return The set of unique error names
70       */
71      public static Set<String> getUniqueNames(List<EvioHeaderError> errors) {
72          Set<String> uniqueErrorNames = new HashSet<String>();
73          for (EvioHeaderError error : errors) {
74              uniqueErrorNames.add(error.getType().name());
75          }
76          return uniqueErrorNames;
77      }
78      
79      /**
80       * Create with type and message and blank debug string.
81       * @param errorType The type of header error
82       * @param message The specific error message
83       */
84      public EvioHeaderError(ErrorType errorType, String message) {
85          this.errorType = errorType;
86          this.message = message;
87          this.debugString = "";
88      }
89      
90      /**
91       * Create with type and message and a populated debug string.
92       * @param errorType The type of header error
93       * @param message The specific error message
94       * @param debugString The detailed debug string 
95       */
96      public EvioHeaderError(ErrorType errorType, String message, String debugString) {
97          this.errorType = errorType;
98          this.message = message;
99          this.debugString = debugString;
100     }    
101     
102     /**
103      * Get the detailed debug string 
104      * @return The detailed debug string
105      */
106     String getDebugString() {
107         return debugString;
108     }    
109     
110     /**
111      * Get the error message
112      * @return The error message
113      */
114     String getMessage() {
115         return message;
116     }
117     
118     /**
119      * Get the type of the error
120      * @return The type of the error
121      */
122     ErrorType getType() {
123         return errorType;
124     }        
125 }
