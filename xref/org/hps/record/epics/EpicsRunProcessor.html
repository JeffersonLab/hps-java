1   package org.hps.record.epics;
2   
3   import java.util.ArrayList;
4   import java.util.HashSet;
5   import java.util.List;
6   import java.util.Set;
7   import java.util.logging.Logger;
8   
9   import org.hps.record.evio.EvioEventProcessor;
10  import org.jlab.coda.jevio.EvioEvent;
11  
12  /**
13   * Creates a list of EPICS data found in EVIO events across an entire job.
14   */
15  public final class EpicsRunProcessor extends EvioEventProcessor {
16  
17      /**
18       * Initialize the logger.
19       */
20      private static final Logger LOGGER = Logger.getLogger(EpicsRunProcessor.class.getPackage().getName());
21      
22      /**
23       * The current EPICS data block from the EVIO events (last one that was found).
24       */
25      private EpicsData currentEpicsData;
26  
27      /**
28       * Collection of the EPICS data accumulated during the job.
29       * <p>
30       * A set is used here to avoid adding duplicate objects.
31       */
32      private Set<EpicsData> epicsDataSet;
33  
34      /**
35       * The processor for extracting the EPICS information from EVIO events.
36       */
37      private final EpicsEvioProcessor processor = new EpicsEvioProcessor();
38  
39      /**
40       * Create a processor that will make a list of EPICS data.
41       */
42      public EpicsRunProcessor() {
43      }
44  
45      /**
46       * Get the EPICS data from the job.
47       *
48       * @return the EPICS data from the job
49       */
50      public List<EpicsData> getEpicsData() {
51          return new ArrayList<EpicsData>(this.epicsDataSet);
52      }
53  
54      /**
55       * Process a single EVIO event, setting the current EPICS data and updating the variable counts.
56       */
57      @Override
58      public void process(final EvioEvent evioEvent) {
59  
60          // Call the processor that will load EPICS data if it exists in the event.
61          this.processor.reset();
62          this.processor.process(evioEvent);
63          this.currentEpicsData = this.processor.getEpicsData();
64  
65          // Add EPICS data to the collection.
66          if (this.currentEpicsData != null) {
67              LOGGER.fine("Adding EPICS data with run " + this.currentEpicsData.getEpicsHeader().getRun() + "; timestamp " 
68                      + this.currentEpicsData.getEpicsHeader().getTimestamp() + "; seq "
69                      + this.currentEpicsData.getEpicsHeader().getSequence() + ".");
70              this.epicsDataSet.add(this.currentEpicsData);
71          }
72      }
73  
74      /**
75       * Start of job hook (reset the list of EPICS data).
76       */
77      @Override
78      public void startJob() {
79          epicsDataSet = new HashSet<EpicsData>();
80      }
81  }
