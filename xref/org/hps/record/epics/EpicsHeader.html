1   package org.hps.record.epics;
2   
3   /**
4    * Representation of EPICs header data from the EVIO bank (run, sequence, time stamp).
5    */
6   public final class EpicsHeader {
7   
8       /**
9        * Create an {@link EpicsHeader} from an int array in the EVIO bank.
10       * <p>
11       * This reads in indices 1 to 3 as 0 and 5 are unused.
12       *
13       * @param headerBank the header bank data
14       * @return the {@link EpicsHeader} object
15       */
16      static EpicsHeader fromEvio(final int[] headerBank) {
17          final int[] headerData = new int[] {headerBank[1], headerBank[2], headerBank[3]};
18          return new EpicsHeader(headerData);
19      }
20  
21      /**
22       * The run number.
23       */
24      private final int run;
25  
26      /**
27       * The sequence number.
28       */
29      private final int sequence;
30  
31      /**
32       * The unix time in seconds.
33       */
34      private final int timestamp;
35  
36      /**
37       * Class constructor.
38       * <p>
39       * The data array should be length 3 and usually will come from the int data of a <code>GenericObject</code>. In
40       * order, it should contain the run, sequence and timestamp values.
41       *
42       * @param data the header data with length 3
43       */
44      public EpicsHeader(final int[] data) {
45          if (data.length != 3) {
46              throw new IllegalArgumentException("Bad array length: " + data.length);
47          }
48          run = data[0];
49          sequence = data[1];
50          timestamp = data[2];
51      }
52  
53      /**
54       * Get the run number.
55       *
56       * @return the run number
57       */
58      public int getRun() {
59          return run;
60      }
61  
62      /**
63       * Get the sequence number.
64       *
65       * @return the sequence number
66       */
67      public int getSequence() {
68          return sequence;
69      }
70  
71      /**
72       * Get the time stamp.
73       *
74       * @return the time stamp
75       */
76      public int getTimestamp() {
77          return timestamp;
78      }
79  }
