3   import java.io.IOException;
4   
5   import org.hps.record.et.EtEventProcessor;
6   import org.jlab.coda.et.EtEvent;
7   import org.jlab.coda.jevio.EvioEvent;
8   import org.jlab.coda.jevio.EvioException;
9   import org.jlab.coda.jevio.EvioReader;
10  
11  /**
12   * An ET event processor that builds EPICS events.
13   */
14  // FIXME: Class is currently unused.
15  public class EpicsEtProcessor extends EtEventProcessor {
16  
17      /**
18       * The processor for creating the EPICS data class from EVIO data.
19       */
20      private final EpicsEvioProcessor evioProcessor = new EpicsEvioProcessor();
21  
22      /**
23       * Process an <code>EtEvent</code> and create an EPICS data object from it.
24       *
25       * @param event the <code>EtEvent</code> to process
26       */
27      @Override
28      public void process(final EtEvent event) {
29          EvioEvent evio;
30          try {
31              evio = new EvioReader(event.getDataBuffer()).parseNextEvent();
32          } catch (IOException | EvioException e) {
33              throw new RuntimeException(e);
34          }
35          this.evioProcessor.process(evio);
36          if (this.evioProcessor.getEpicsData() != null) {
37              System.out.println("EpicsEtProcessor created EpicsData ...");
38              System.out.println(this.evioProcessor.getEpicsData());
39          }
40      }
41  }
