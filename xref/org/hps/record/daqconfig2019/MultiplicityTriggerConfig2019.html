1   package org.hps.record.daqconfig2019;
2   
3   /**
4    * Class <clode>MultiplicityTriggerConfig2019</code> holds the configuration data
5    * for a multiplicity trigger
6    */
7   public class MultiplicityTriggerConfig2019 extends AbstractConfig2019<AbstractConfig2019<Double>> {
8       private static final int CUT_ENERGY_MIN = 0;
9       private static final int CUT_ENERGY_MAX = 1;
10      private static final int CUT_HIT_COUNT  = 2;
11      private static final int CUT_TOP_NCLUSTERS  = 3;
12      private static final int CUT_BOT_NCLUSTERS  = 4;
13      private static final int CUT_TOT_NCLUSTERS  = 5;
14      private static final int CUT_TIME_DIFF  = 6;
15      
16      /**
17       * Creates a new <code>MultiplicityTrigger2019</code> object.
18       */
19      public MultiplicityTriggerConfig2019() {
20          // Instantiate the base object.
21          super(7);
22          
23          // Define the singles cuts.
24          setValue(CUT_ENERGY_MIN, new LBOCutConfig2019());
25          setValue(CUT_ENERGY_MAX, new UBOCutConfig2019());
26          setValue(CUT_HIT_COUNT,  new LBOCutConfig2019());
27          setValue(CUT_TOP_NCLUSTERS,  new LBOCutConfig2019());
28          setValue(CUT_BOT_NCLUSTERS,  new LBOCutConfig2019());
29          setValue(CUT_TOT_NCLUSTERS,  new LBOCutConfig2019());
30          setValue(CUT_TIME_DIFF,  new UBOCutConfig2019());
31      }
32      
33      /**
34       * Gets the configuration object for the cluster energy lower bound
35       * cut. Note that cuts are in units of GeV.
36       * @return Returns the configuration object for the cut.
37       */
38      public LBOCutConfig2019 getEnergyMinCutConfig() {
39          return (LBOCutConfig2019) getValue(CUT_ENERGY_MIN);
40      }
41      
42      /**
43       * Gets the configuration object for the cluster energy upper bound
44       * cut. Note that cuts are in units of GeV.
45       * @return Returns the configuration object for the cut.
46       */
47      public UBOCutConfig2019 getEnergyMaxCutConfig() {
48          return (UBOCutConfig2019) getValue(CUT_ENERGY_MAX);
49      }
50      
51      /**
52       * Gets the configuration object for the cluster hit count cut.
53       * Note that cuts are in units of calorimeter hits.
54       * @return Returns the configuration object for the cut.
55       */
56      public LBOCutConfig2019 getHitCountCutConfig() {
57          return (LBOCutConfig2019) getValue(CUT_HIT_COUNT);
58      }
59      
60      /**
61       * Gets the configuration object for minimum of the number of clusters at top.
62       * Note that cuts are in units of calorimeter clusters.
63       * @return Returns the configuration object for the cut.
64       */
65      public LBOCutConfig2019 getNClusterTopCutConfig() {
66          return (LBOCutConfig2019) getValue(CUT_TOP_NCLUSTERS);
67      }
68      
69      /**
70       * Gets the configuration object for minimum of the number of clusters at bot.
71       * Note that cuts are in units of calorimeter clusters.
72       * @return Returns the configuration object for the cut.
73       */
74      public LBOCutConfig2019 getNClusterBotCutConfig() {
75          return (LBOCutConfig2019) getValue(CUT_BOT_NCLUSTERS);
76      }
77      
78      /**
79       * Gets the configuration object for minimum of the total number of clusters.
80       * Note that cuts are in units of calorimeter clusters.
81       * @return Returns the configuration object for the cut.
82       */
83      public LBOCutConfig2019 getNClusterTotCutConfig() {
84          return (LBOCutConfig2019) getValue(CUT_TOT_NCLUSTERS);
85      }
86              
87      /**
88       * Gets the configuration object for the pair time coincidence cut.
89       * Note that cuts are in units of nanoseconds.
90       * @return Returns the configuration object for the cut.
91       */
92      public UBOCutConfig2019 getTimeDifferenceCutConfig() {
93          return (UBOCutConfig2019) getValue(CUT_TIME_DIFF);
94      }
95  
96  }
