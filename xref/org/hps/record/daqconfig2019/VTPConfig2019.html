5   /**
6    * Class <code>VTPConfig2019</code> stores VTP configuration settings
7    * parsed from the an EVIO file. This class manages the following
8    * properties:
9    * <ul>
10   * <li>Singles 1 Trigger</li>
11   * <li>Singles 2 Trigger</li>
12   * <li>Singles 3 Trigger</li>
13   * <li>Singles 4 Trigger</li>
14   * <li>Pair 1 Trigger</li>
15   * <li>Pair 2 Trigger</li>
16   * <li>Pair 3 Trigger</li>
17   * <li>Pair 4 Trigger</li>
18   * </ul>
19   */
20  public class VTPConfig2019 extends IDAQConfig2019 {
21      ////// Store cluster cut parameters
22      // Cluster hit timing coincidence: 0 to 16, units: +/-ns
23      private int ecalClusterHitDT = 0;
24      // Cluster seed threshold in: 1/1000 to 8191/1000, units GeV
25      private double ecalClusterSeedThr = 0;
26      // Hodoscope fadc hit cut: minimum acceptable FADC hit integral: 1 to 8191, units TBD
27      private int hodoFADCHitThr = 0;
28      // Hodoscope trigger hit cut: minimum acceptable integral (clustered or single tile): 1 to 8191, units TBD
29      private int hodoThr = 0;   
30      // Hodoscope hit coincidence between L1,L2, and also ECAL clusters (real with is specified value +4ns): 0 to 60, units: ns
31      private int hodoDT = 0;
32             
33      //////// Store trigger configuration parameters.
34      private PairTriggerConfig2019[] pairTrigger = { new PairTriggerConfig2019(), new PairTriggerConfig2019(), 
35              new PairTriggerConfig2019(), new PairTriggerConfig2019() };
36      private SinglesTriggerConfig2019[] singlesTrigger = { new SinglesTriggerConfig2019(), new SinglesTriggerConfig2019(), 
37              new SinglesTriggerConfig2019(), new SinglesTriggerConfig2019() };
38      private MultiplicityTriggerConfig2019[] multiplicityTrigger = { new MultiplicityTriggerConfig2019(), new MultiplicityTriggerConfig2019()};
39      private FEETriggerConfig2019 FEETrigger = new FEETriggerConfig2019();
40          
41      @Override
42      void loadConfig(EvioDAQParser2019 parser) {
43          // Set cluster cut parameters
44          ecalClusterHitDT = parser.ecalClusterHitDT;
45          ecalClusterSeedThr = parser.ecalClusterSeedThr / 1000.;
46          hodoFADCHitThr = parser.hodoFADCHitThr;
47          hodoThr = parser.hodoThr;
48          hodoDT = parser.hodoDT;
49          
50          // Set the trigger parameters.
51          for(int triggerNum = 0; triggerNum < 4; triggerNum++) {
52              // Set whether the triggers are enabled or not.
53              singlesTrigger[triggerNum].setIsEnabled(parser.singlesEn[triggerNum]);
54              pairTrigger[triggerNum].setIsEnabled(parser.pairsEn[triggerNum]);
55              
56              // Set the cut enabled statuses for the singles trigger.
57              singlesTrigger[triggerNum].getEnergyMinCutConfig().setIsEnabled(parser.singlesEnergyMinEn[triggerNum]);
58              singlesTrigger[triggerNum].getEnergyMaxCutConfig().setIsEnabled(parser.singlesEnergyMaxEn[triggerNum]);
59              singlesTrigger[triggerNum].getHitCountCutConfig().setIsEnabled(parser.singlesNhitsMinEn[triggerNum]);
60              singlesTrigger[triggerNum].getXMinCutConfig().setIsEnabled(parser.singlesXMinEn[triggerNum]);
61              
62              //For 2021 update
63              if(parser.singlesXYMinMaxEn[triggerNum]) {
64                  singlesTrigger[triggerNum].getXMinCutConfig().setIsEnabled(parser.singlesXYMinMaxEn[triggerNum]); // Default: false
65                  singlesTrigger[triggerNum].getXMaxCutConfig().setIsEnabled(parser.singlesXYMinMaxEn[triggerNum]); // Default: false
66                  singlesTrigger[triggerNum].getYMinCutConfig().setIsEnabled(parser.singlesXYMinMaxEn[triggerNum]); // Default: false
67                  singlesTrigger[triggerNum].getYMaxCutConfig().setIsEnabled(parser.singlesXYMinMaxEn[triggerNum]); // Default: false
68              }
69              singlesTrigger[triggerNum].setIsSinglesXYMinMaxEnabled(parser.singlesXYMinMaxEn[triggerNum]);
70              singlesTrigger[triggerNum].setIsSinglesMollerModeEnabled(parser.singlesMollerModeEn[triggerNum]);
71              
72                                    
73              singlesTrigger[triggerNum].getPDECutConfig().setIsEnabled(parser.singlesPDEEn[triggerNum]);
74              singlesTrigger[triggerNum].getL1MatchingConfig().setIsEnabled(parser.singlesL1MatchingEn[triggerNum]);
75              singlesTrigger[triggerNum].getL2MatchingConfig().setIsEnabled(parser.singlesL2MatchingEn[triggerNum]);
76              singlesTrigger[triggerNum].getL1L2GeoMatchingConfig().setIsEnabled(parser.singlesL1L2MatchingEn[triggerNum]);
77              singlesTrigger[triggerNum].getHodoEcalGeoMatchingConfig().setIsEnabled(parser.singlesL1L2EcalMatchingEn[triggerNum]);
78              
79              // Set the individual cut values.
80              singlesTrigger[triggerNum].getEnergyMinCutConfig().setLowerBound(parser.singlesEnergyMin[triggerNum] / 1000.0);
81              singlesTrigger[triggerNum].getEnergyMaxCutConfig().setUpperBound(parser.singlesEnergyMax[triggerNum] / 1000.0);
82              singlesTrigger[triggerNum].getHitCountCutConfig().setLowerBound(parser.singlesNhitsMin[triggerNum]);
83              singlesTrigger[triggerNum].getXMinCutConfig().setLowerBound(parser.singlesXMin[triggerNum]);
84              singlesTrigger[triggerNum].getPDECutConfig().setParC0(parser.singlesPDEC0[triggerNum]/ 1000.0);
85              singlesTrigger[triggerNum].getPDECutConfig().setParC1(parser.singlesPDEC1[triggerNum]/ 1000.0);
86              singlesTrigger[triggerNum].getPDECutConfig().setParC2(parser.singlesPDEC2[triggerNum]/ 1000.0);
87              singlesTrigger[triggerNum].getPDECutConfig().setParC3(parser.singlesPDEC3[triggerNum]/ 1000.0);
88              
89              //For 2021 update
90              singlesTrigger[triggerNum].getXMaxCutConfig().setUpperBound(parser.singlesXMax[triggerNum]);
91              singlesTrigger[triggerNum].getYMinCutConfig().setLowerBound(parser.singlesYMin[triggerNum]);
92              singlesTrigger[triggerNum].getYMaxCutConfig().setUpperBound(parser.singlesYMax[triggerNum]);              
93              
94              // The pair trigger singles cuts are always enabled.
95              pairTrigger[triggerNum].getEnergyMinCutConfig().setIsEnabled(true);
96              pairTrigger[triggerNum].getEnergyMaxCutConfig().setIsEnabled(true);
97              pairTrigger[triggerNum].getHitCountCutConfig().setIsEnabled(true);
98              
99              // The pair trigger time difference cut is always enabled.
100             pairTrigger[triggerNum].getTimeDifferenceCutConfig().setIsEnabled(true);
101 
102             // Set the pair cut enabled statuses for the pair trigger.
103             pairTrigger[triggerNum].getEnergySumCutConfig().setIsEnabled(parser.pairsEnergySumMaxMinEn[triggerNum]);
104             pairTrigger[triggerNum].getEnergyDifferenceCutConfig().setIsEnabled(parser.pairsEnergyDiffEn[triggerNum]);
105             pairTrigger[triggerNum].getEnergySlopeCutConfig().setIsEnabled(parser.pairsEnergyDistEn[triggerNum]);
106             pairTrigger[triggerNum].getCoplanarityCutConfig().setIsEnabled(parser.pairsCoplanarityEn[triggerNum]);
107                         
108             // Set the individual cut values.
109             pairTrigger[triggerNum].getEnergyMinCutConfig().setLowerBound(parser.pairsEnergyMin[triggerNum] / 1000.0);
110             pairTrigger[triggerNum].getEnergyMaxCutConfig().setUpperBound(parser.pairsEnergyMax[triggerNum] / 1000.0);
111             pairTrigger[triggerNum].getHitCountCutConfig().setLowerBound(parser.pairsNhitsMin[triggerNum]);
112             pairTrigger[triggerNum].getEnergySumCutConfig().setLowerBound(parser.pairsEnergySumMin[triggerNum] / 1000.0);
113             pairTrigger[triggerNum].getEnergySumCutConfig().setUpperBound(parser.pairsEnergySumMax[triggerNum] / 1000.0);
114             pairTrigger[triggerNum].getEnergyDifferenceCutConfig().setUpperBound(parser.pairsEnergyDiffMax[triggerNum] / 1000.0);
115             pairTrigger[triggerNum].getEnergySlopeCutConfig().setLowerBound(parser.pairsEnergyDistMin[triggerNum] / 1000.0);
116             pairTrigger[triggerNum].getEnergySlopeCutConfig().setParameterF(parser.pairsEnergyDistSlope[triggerNum] / 1000.0);
117             pairTrigger[triggerNum].getCoplanarityCutConfig().setUpperBound(parser.pairsCoplanarityMax[triggerNum]);
118             pairTrigger[triggerNum].getTimeDifferenceCutConfig().setUpperBound(parser.pairsTimeDiffMax[triggerNum]);
119         }
120         
121         pairTrigger[3].getL1MatchingConfig().setIsEnabled(parser.pairs3L1MatchingEn);
122         pairTrigger[3].getL2MatchingConfig().setIsEnabled(parser.pairs3L2MatchingEn);
123         pairTrigger[3].getL1L2GeoMatchingConfig().setIsEnabled(parser.pairs3L1L2MatchingEn);
124         pairTrigger[3].getHodoEcalGeoMatchingConfig().setIsEnabled(parser.pairs3L1L2EcalMatchingEn);
125         
126         for(int triggerNum = 0; triggerNum < 2; triggerNum++) {
127             // Set whether the triggers are enabled or not.
128             multiplicityTrigger[triggerNum].setIsEnabled(parser.multEn[triggerNum]);
129             
130             // Set the individual cut values.
131             multiplicityTrigger[triggerNum].getEnergyMinCutConfig().setLowerBound(parser.multEnergyMin[triggerNum] / 1000.0);
132             multiplicityTrigger[triggerNum].getEnergyMaxCutConfig().setUpperBound(parser.multEnergyMax[triggerNum] / 1000.0);
133             multiplicityTrigger[triggerNum].getHitCountCutConfig().setLowerBound(parser.multNhitsMin[triggerNum]);
134             multiplicityTrigger[triggerNum].getNClusterTopCutConfig().setLowerBound(parser.multTopMultMin[triggerNum]);
135             multiplicityTrigger[triggerNum].getNClusterBotCutConfig().setLowerBound(parser.multBotMultMin[triggerNum]);
136             multiplicityTrigger[triggerNum].getNClusterTotCutConfig().setLowerBound(parser.multTotMultMin[triggerNum]);
137             multiplicityTrigger[triggerNum].getTimeDifferenceCutConfig().setUpperBound(parser.multDT[triggerNum]);            
138         }
139         
140         FEETrigger.setIsEnabled(parser.FEEEn);
141         FEETrigger.getEnergyMinCutConfig().setLowerBound(parser.FEEEnergyMin / 1000.0);
142         FEETrigger.getEnergyMaxCutConfig().setUpperBound(parser.FEEEnergyMax / 1000.0);
143         FEETrigger.getHitCountCutConfig().setLowerBound(parser.FEENhitsMin);
144         FEETrigger.getPrescaleRegion0Config().setRegionMin(parser.FEERegionXMin[0]);
145         FEETrigger.getPrescaleRegion0Config().setRegionMax(parser.FEERegionXMax[0]);
146         FEETrigger.getPrescaleRegion0Config().setRegionPrescale(parser.FEERegionPrescale[0]);
147         FEETrigger.getPrescaleRegion1Config().setRegionMin(parser.FEERegionXMin[1]);
148         FEETrigger.getPrescaleRegion1Config().setRegionMax(parser.FEERegionXMax[1]);
149         FEETrigger.getPrescaleRegion1Config().setRegionPrescale(parser.FEERegionPrescale[1]);
150         FEETrigger.getPrescaleRegion2Config().setRegionMin(parser.FEERegionXMin[2]);
151         FEETrigger.getPrescaleRegion2Config().setRegionMax(parser.FEERegionXMax[2]);
152         FEETrigger.getPrescaleRegion2Config().setRegionPrescale(parser.FEERegionPrescale[2]);
153         FEETrigger.getPrescaleRegion3Config().setRegionMin(parser.FEERegionXMin[3]);
154         FEETrigger.getPrescaleRegion3Config().setRegionMax(parser.FEERegionXMax[3]);
155         FEETrigger.getPrescaleRegion3Config().setRegionPrescale(parser.FEERegionPrescale[3]);
156         FEETrigger.getPrescaleRegion4Config().setRegionMin(parser.FEERegionXMin[4]);
157         FEETrigger.getPrescaleRegion4Config().setRegionMax(parser.FEERegionXMax[4]);
158         FEETrigger.getPrescaleRegion4Config().setRegionPrescale(parser.FEERegionPrescale[4]);
159         FEETrigger.getPrescaleRegion5Config().setRegionMin(parser.FEERegionXMin[5]);
160         FEETrigger.getPrescaleRegion5Config().setRegionMax(parser.FEERegionXMax[5]);
161         FEETrigger.getPrescaleRegion5Config().setRegionPrescale(parser.FEERegionPrescale[5]);
162         FEETrigger.getPrescaleRegion6Config().setRegionMin(parser.FEERegionXMin[6]);
163         FEETrigger.getPrescaleRegion6Config().setRegionMax(parser.FEERegionXMax[6]);
164         FEETrigger.getPrescaleRegion6Config().setRegionPrescale(parser.FEERegionPrescale[6]);
165     }

282         ps.printf("\tThreshold for clusters or tiles of Hodoscope :: %d TBD%n", getHodoThr());
