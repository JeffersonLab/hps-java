598     /**
599      * Converts the textual configuration information into a map, where the first
600      * column value becomes the map entry key and the remainder becomes the map
601      * entry value.
602      * 
603      * @param crate        - The crate associated with the textual configuration
604      *                     data.
605      * @param configTables - An array of textual configuration tables that contain
606      *                     the DAQ configuration parameters.
607      */
608     private void loadConfigMap(int crate, String[] configTables) {
609         // Iterate over each configuration table.
610         for (String configTable : configTables) {
611             // Split each table into rows and iterate over the rows.
612             rowLoop: for (String line : configTable.trim().split("\n")) {
613                 // Split the first column from the row.
614                 String[] cols = line.trim().split(" +", 2);
615 
616                 // If there are fewer than two segments after the split,
617                 // then this is not a valid parameter entry.
618                 if (cols.length < 2)
619                     continue rowLoop;
620 
621                 // The row name is the value of the first column. The
622                 // rest are typically values.
623                 String key = cols[0];
624                 List<String> vals = new ArrayList<String>(Arrays.asList(cols[1].trim().split(" +")));
625 
626                 // If no values are present, this is not a valid entry.
627                 if (vals.size() < 1) {
628                     continue rowLoop;
629                 }
630 
631                 // SPECIAL CASE:: Key "FADC250"
632                 // This entry marks parameter values for FADC channels,
633                 // such as pedestals, gains, and thresholds. These are
634                 // stored in separate lists from the other parameters.
635                 if (key.startsWith("FADC250")) {
636                     parseFADC(crate, key.trim(), vals);
637                 }
638 
639                 // GENERAL CASE: Basic Parameter
640                 // This indicates a regular parameter that does not
641                 // require any special parsing.
642                 if (vals.size() > 1 && key.startsWith("VTP")) {
643                     // List the parameter by "[ROW NAME]_[KEY]" and
644                     // remove the key so that only the values remain.
645                     key += "_" + vals.remove(0);
646                 }
647 
648                 // Add the parameter key and its values to the map.
649                 configMap.put(key, vals);
650             }
651         }        
652     }

877      * the first of which is the parameter key and the subsequent being parameter
