13  /**
14   * Class <code>DAQConfig2019Driver</code> is responsible for accessing the 2019
15   * DAQ configuration settings, and then passing them to the associated class
16   * <code>ConfigurationManager2019</code> so that they can be accessed by other
17   * classes.<br/>
18   * <br/>
19   * The driver may accomplish this by two means. By default, it will check each
20   * event for an <code>EvioDAQParser2019</code> object which contains all of the
21   * DAQ configuration information and pass this to the
22   * <code>ConfigurationManager2019</code>. It will continue to update the
23   * <code>ConfigurationManager2019</code> during the run if new parser objects
24   * appear, and can thusly account for changing DAQ conditions. <br/>
25   * <br/>
26   * The driver may also be set to read a DAQ configuration from text files
27   * containing the DAQ configuration bank information. The text files are located
28   * in resources. To enable this mode, the parameter
29   * <code>daqConfigurationAppliedintoReadout</code> need to be set
30   * <code>true</code>, or the parameter <code>readDataFiles</code> need to be set
31   * to <code>true</code>. <code>runNumber</code> defines the run number of the
32   * configuration to be loaded. <br/>
33   * <br/>
34   * This driver must be included in the driver chain if any other drivers in the
35   * chain rely on <code>ConfigurationManager2019</code>, as it can not be
36   * initialized otherwise.
37   * 
38   * Code is developed referring to org.hps.record.daqconfig.DAQConfigDriver by
39   * Kyle McCarty
40   *
41   * @see ConfigurationManager2019
42   */
43  public class DAQConfig2019Driver extends Driver {
44      private int runNumber = -1;
45      private boolean readDataFiles = false;
46      private boolean firstEvent = true;
47      private InputStream[] dataFiles = new InputStream[3];
48      private int[] crateNumber = { 37, 39, 11};
49      private String daqVersion = null;
50      
51      /**
52       * Indicates whether the DAQ configuration is applied in the readout system.
53       */
54      private boolean daqConfigurationAppliedintoReadout = false;

56      /**
57       * Verifies the parameter <code>filepath</code> for the data file repository and
58       * checks that appropriate data files exist for the requested run number if the
59       * driver is set to read from data files. Otherwise, this does nothing.
60       */
61      @Override
62      public void startOfData() {
63          if(runNumber == -1) runNumber = this.getConditionsManager().getRun(); 
64          daqVersion = mapBetweenRunNumberDAQVersion(runNumber);
65          
66          // Check whether to apply the DAQ configuration into the readout system.
67          if (daqConfigurationAppliedintoReadout) {
68              // Define the data file objects.
69              for (int i = 0; i < dataFiles.length; i++) {
70                  dataFiles[i] = DAQConfig2019Driver.class.getResourceAsStream(daqVersion + "_" + crateNumber[i] + ".txt");
71                  if (dataFiles[i] == null) {
72                      if(runNumber == 1194550 || System.getProperties().containsKey("defaultDAQVersion2019") == true) 
73                          dataFiles[i] = DAQConfig2019Driver.class.getResourceAsStream("hps_v12_1" + "_" + crateNumber[i] + ".txt");
74                      else if(runNumber == 1193700 || System.getProperties().containsKey("defaultDAQVersion2021") == true) 
75                          dataFiles[i] = DAQConfig2019Driver.class.getResourceAsStream("hps2021_v2_3" + "_" + crateNumber[i] + ".txt");
76                      else if(runNumber == 1191920 || System.getProperties().containsKey("defaultDAQVersion20211920") == true) 
77                          dataFiles[i] = DAQConfig2019Driver.class.getResourceAsStream("hps_1.9_v2_6" + "_" + crateNumber[i] + ".txt");
78                      else throw new RuntimeException("No corresponding DAQ configuration file for run " + String.valueOf(runNumber) + " in hps-java/record-util/src/main/resources/org/hps/record/daqconfig2019.\n"
79                              + " Please change run number or set the system property -DdefaultDAQVersion2019 to apply hps_v12_1, or -DdefaultDAQVersion2021 to apply hps2021_v2_3, or -DdefaultDAQVersion20211920 to apply hps_1.9_v2_6.");
80                  }
81              }
82              
83              // If this is the first event and data files are to be read,
84              // import the data files and generate the DAQ information.
85              // Get the data files in the form of a data array.
86              String[][] data;
87              try {
88                  data = getDataFileArrays(dataFiles);
89              } catch (IOException e) {
90                  throw new RuntimeException("An error occurred when processing the data files.");
91              }
92  
93              // Instantiate an EvIO DAQ parser and feed it the data.
94              EvioDAQParser2019 daqConfig = new EvioDAQParser2019();
95              for (int i = 0; i < dataFiles.length; i++) {
96                  daqConfig.parse(crateNumber[i], runNumber, data[i]);
97              }
98  
99              // Update the configuration manager.
100             ConfigurationManager2019.updateConfiguration(daqConfig);
101         }
102         
103         // Check whether to use data files stored in resources and apply the DAQ configuration into the trigger diagnostics 
104         if(readDataFiles) {
105             // Define the data file objects.
106             for (int i = 0; i < dataFiles.length; i++) {
107                 dataFiles[i] = DAQConfig2019Driver.class.getResourceAsStream(daqVersion + "_" + crateNumber[i] + ".txt");                
108                 if (dataFiles[i] == null) {
109                     if(runNumber == 1194550 || System.getProperties().containsKey("defaultDAQVersion2019") == true) 
110                         dataFiles[i] = DAQConfig2019Driver.class.getResourceAsStream("hps_v12_1" + "_" + crateNumber[i] + ".txt");
111                     else if(runNumber == 1193700 || System.getProperties().containsKey("defaultDAQVersion2021") == true) 
112                         dataFiles[i] = DAQConfig2019Driver.class.getResourceAsStream("hps2021_v2_3" + "_" + crateNumber[i] + ".txt");
113                     else if(runNumber == 1191920 || System.getProperties().containsKey("defaultDAQVersion20211920") == true) 
114                         dataFiles[i] = DAQConfig2019Driver.class.getResourceAsStream("hps_1.9_v2_6" + "_" + crateNumber[i] + ".txt");
115                     else throw new RuntimeException("No corresponding DAQ configuration file for run " + String.valueOf(runNumber) + " in hps-java/record-util/src/main/resources/org/hps/record/daqconfig2019.\n"
116                             + " Please change run number or set the system property -DdefaultDAQVersion2019 to apply hps_v12_1, or -DdefaultDAQVersion2021 to apply hps2021_v2_3, or -DdefaultDAQVersion20211920 to apply hps_1.9_v2_6.");
117                 }
118             }
119         }
120     }

122     /**
123      * Checks an event for the DAQ configuration banks and passes them to the
124      * <code>ConfigurationManager2019</code> if the driver is set to read from the
125      * EvIO data stream. Otherwise, this will parse the data files on the first
126      * event and then do nothing.
127      * 
128      * @param event - The current LCIO event.
129      */
130     @Override
131     public void process(EventHeader event) {       
132         // If this is the first event and data files are to be read,
133         // import the data files and generate the DAQ information.
134         if(firstEvent && readDataFiles) {
135             // Get the data files in the form of a data array.
136             String[][] data;
137             try { data = getDataFileArrays(dataFiles); }
138             catch(IOException e) {
139                 throw new RuntimeException("An error occurred when processing the data files.");
140             }
141             
142             // Instantiate an EvIO DAQ parser and feed it the data.
143             EvioDAQParser2019 daqConfig = new EvioDAQParser2019();
144             for(int i = 0; i < dataFiles.length; i++) {
145                 daqConfig.parse(crateNumber[i], runNumber, data[i]);
146             }
147             
148             // Update the configuration manager.
149             ConfigurationManager2019.updateConfiguration(daqConfig);
150             
151             // Note that it is no longer the first event.
152             firstEvent = false;
153         }
154         
155         // Check if a trigger configuration bank exists.
156         if (!readDataFiles && event.hasCollection(EvioDAQParser2019.class, "TriggerConfig")) {
157             // Get the trigger configuration bank. There should only be
158             // one in the list.
159             List<EvioDAQParser2019> configList = event.get(EvioDAQParser2019.class, "TriggerConfig");
160             EvioDAQParser2019 daqConfig = configList.get(0);
161 
162             // Get the DAQ configuration and update it with the new
163             // configuration object.
164             ConfigurationManager2019.updateConfiguration(daqConfig);
165         }        
166     }

168     /**
169      * Converts DAQ configuration data files into an array of strings where each
170      * array entry represents a line in the configuration file. The first array
171      * index of the returned object corresponds to the file, and the second array
172      * index corresponds to the line.
173      * 
174      * @param dataFiles - An array of <code>File</code> objects pointing to the data
175      *                  files that are to be converted. These are expected to be
176      *                  plain text files.
177      * @return Returns a two-dimensional array of <code>String</code> objects where
178      *         the first array index corresponds to the object of the same index in
179      *         the <code>File</code> array and the second array index corresponds to
180      *         the lines in the file referenced by the <code>File</code> object.
181      * @throws IOException Occurs if there is an issue with accessing or reading the
182      *                     objects in the objects referred to by the files pointed
183      *                     to in the <code>dataFiles</code> array.
184      */
185     private static final String[][] getDataFileArrays(InputStream[] dataFiles) throws IOException {
186         // Create file readers to process the data files.
187         InputStreamReader[] fr = new InputStreamReader[dataFiles.length];
188         BufferedReader[] reader = new BufferedReader[dataFiles.length];
189         for (int i = 0; i < dataFiles.length; i++) {
190             fr[i] = new InputStreamReader(dataFiles[i]);
191             reader[i] = new BufferedReader(fr[i]);
192         }
193 
194         // Convert the reader streams into line-delimited strings.
195         String[][] data = getDataFileArrays(reader);
196 
197         // Close the readers.
198         for (int i = 0; i < dataFiles.length; i++) {
199             reader[i].close();
200             fr[i].close();
201         }
202 
203         // Return the data array.
204         return data;
205     }

207     /**
208      * Converts DAQ configuration data streams into an array of strings where each
209      * array entry represents a line in the configuration file. The first array
210      * index of the returned object corresponds to the stream, and the second array
211      * index corresponds to the line.
212      * 
213      * @param reader - An array of <code>BufferedReader</code> objects containing
214      *               DAQ trigger configuration crate data.
215      * @return Returns a two-dimensional array of <code>String</code> objects where
216      *         the first array index corresponds to the object of the same index in
217      *         the <code>BufferedReader</code> array and the second array index
218      *         corresponds to the lines in the stream referenced by the
219      *         <code>BufferedReader</code> object.
220      * @throws IOException Occurs if there is an issue with reading data stream.
221      */
222     protected static final String[][] getDataFileArrays(BufferedReader[] reader) throws IOException {
223         // Generate String arrays where each entry in the array is
224         // a line from the data file.
225         String[][] data = new String[reader.length][0];
226         for (int i = 0; i < reader.length; i++) {
227             // Create a list to hold the raw strings.
228             List<String> rawData = new ArrayList<String>();
229 
230             // Add each line from the current data file to the list
231             // as a single entry.
232             String curLine = null;
233             while ((curLine = reader[i].readLine()) != null) {
234                 rawData.add(curLine);
235             }
236 
237             // Convert the list into a String array.
238             data[i] = rawData.toArray(new String[rawData.size()]);
239         }
240 
241         // Return the data array.
242         return data;
243     }
