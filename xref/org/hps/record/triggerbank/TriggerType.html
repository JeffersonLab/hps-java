1   package org.hps.record.triggerbank;
2   
3   import java.util.HashSet;
4   import java.util.Set;
5   
6   import org.hps.record.triggerbank.AbstractIntData.IntBankDefinition;
7   import org.jlab.coda.jevio.BaseStructure;
8   import org.jlab.coda.jevio.EvioEvent;
9   
10  /**
11   * Provides bit mask checks for the trigger types defined in
12   * <a href="https://confluence.slac.stanford.edu/display/hpsg/EVIO+Data+Format#EVIODataFormat-EVIOEventtypes-2015DataSet">EVIO Event types</a>.
13   */
14  public enum TriggerType {
15      
16      /** LED or Cosmic trigger (sometimes called calibration also). */
17      LED_COSMIC(28),
18      /** Pair 0 trigger. */
19      PAIRS0(26),
20      /** Pair 1 trigger. */
21      PAIRS1(27),
22      /** Pulser triggered event. */
23      PULSER(29),
24      /** Single 0 trigger. */
25      SINGLES0(24),
26      /** Single 1 trigger. */
27      SINGLES1(25);
28      
29      /**
30       * The bit number.
31       */
32      private int bit;
33  
34      /**
35       * Constructor with bit number which is used to shift the input TI bits.    
36       * @param bit the bit number for shifting the TI bits
37       */
38      private TriggerType(final int bit) {
39          this.bit = bit;
40      }
41  
42      /**
43       * Get the bit number.    
44       * @return the bit number
45       */
46      public int getBit() {
47          return this.bit;
48      }
49      
50      /**
51       * Return <code>true</code> if the bits match this mask.    
52       * @param triggerBits the trigger bits from the TI bank
53       * @return <code>true</code> if the bits match this mask
54       */
55      public boolean matches(final int triggerBits) {
56          return ((triggerBits >> this.bit) & 1) == 1;
57      }
58      
59      /**
60       * Definition of the TI bank in the EVIO event.
61       */
62      private static IntBankDefinition TI_BANK = new IntBankDefinition(TIData.class, new int[] {0x2e, 0xe10a});
63      
64      /**
65       * Get the applicable trigger types from the TI data in an EVIO event.
66       * @param evioEvent the input EVIO event with the TI data
67       * @return the set of matching trigger types for the event
68       */
69      public static Set<TriggerType> getTriggerTypes(EvioEvent evioEvent) {
70          Set<TriggerType> matches = new HashSet<TriggerType>();
71          BaseStructure tiBank = TI_BANK.findBank(evioEvent);
72          if (tiBank != null) {
73              int[] triggerData = tiBank.getIntData();
74              if (triggerData != null) {
75                  for (TriggerType triggerType : TriggerType.values()) {
76                      if (triggerType.matches(triggerData[0])) {
77                          matches.add(triggerType);
78                      }
79                  }
80              }
81          }
82          return matches;
83      }      
84  }
