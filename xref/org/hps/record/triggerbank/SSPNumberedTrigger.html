1   package org.hps.record.triggerbank;
2   
3   /**
4    * Abstract class <code>SSPNumberedTrigger</code> is used as a base
5    * class for any trigger types where there is more than one trigger
6    * of that type in the SSP. It requires extending classes to support
7    * the ability to return which trigger created the object.
8    * 
9    * @see SSPTrigger
10   */
11  public abstract class SSPNumberedTrigger extends SSPTrigger {
12      /**
13       * Instantiates the <code>SSPNumberedTrigger</code>.
14       * @param type - The type of trigger.
15       * @param time - The time at which the trigger occurred in ns.
16       * @param data - The trigger bit data.
17       */
18      public SSPNumberedTrigger(int type, int time, int data) {
19          super(type, time, data);
20      }
21      
22      /**
23       * Gets the number of the trigger which generated this object.
24       * @return Returns either <code>0</code> or </code>1</code>.
25       */
26      public abstract int getTriggerNumber();
27      
28      /**
29       * Indicates whether the trigger was reported by the trigger number
30       * 0 trigger.
31       * @return <code>true</code> if the trigger was reported by the
32       * trigger number 0 trigger and <code>false</code> if by either
33       * the trigger number 1 or an unknown trigger.
34       */
35      public abstract boolean isTrigger0();
36      
37      /**
38       * Indicates whether the trigger was reported by the trigger number
39       * 1 trigger.
40       * @return <code>true</code> if the trigger was reported by the
41       * trigger number 1 trigger and <code>false</code> if by either
42       * the trigger number 0 or an unknown trigger.
43       */
44      public abstract boolean isTrigger1();
45      
46      
47      /**
48       * Indicates whether the trigger was reported by the first of the
49       * singles triggers.
50       * @return <code>true</code> if the trigger was reported by the
51       * first trigger and <code>false</code> if it was reported by the
52       * second trigger.
53       */
54      @Deprecated
55      public abstract boolean isFirstTrigger();
56      
57      /**
58       * Indicates whether the trigger was reported by the second of
59       * the singles triggers.
60       * @return <code>true</code> if the trigger was reported by the
61       * second trigger and <code>false</code> if it was reported by
62       * the first trigger.
63       */
64      @Deprecated
65      public abstract boolean isSecondTrigger();
66  }
