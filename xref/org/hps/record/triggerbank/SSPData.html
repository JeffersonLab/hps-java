3   import org.lcsim.event.GenericObject;
4   
5   import java.util.ArrayList;
6   import java.util.List;

8   /**
9    * Parses the trigger bank information in the SSP trigger bank and
10   * converts it into <code>SSPCluster</code> and <code>SSPTrigger</code>
11   * objects.
12   */
13  public class SSPData extends AbstractIntData {
14      // The EVIO header tag for SSP trigger banks.
15      public static final int BANK_TAG = 0xe10c;
16      
17      // EVIO 5-bit word identifiers for cluster parameters.
18      public static final int TRIG_HEADER = 0x12;
19      public static final int TRIG_TIME = 0x13;
20      public static final int TRIG_TYPE = 0x15;
21      public static final int CLUSTER_TYPE = 0x14;
22      
23      // EVIO trigger type identifiers.
24      public static final int TRIG_TYPE_COSMIC_TOP = 0x0;
25      public static final int TRIG_TYPE_COSMIC_BOT = 0x1;
26      public static final int TRIG_TYPE_SINGLES0_TOP = 0x2;
27      public static final int TRIG_TYPE_SINGLES0_BOT = 0x3;
28      public static final int TRIG_TYPE_SINGLES1_TOP = 0x4;
29      public static final int TRIG_TYPE_SINGLES1_BOT = 0x5;
30      public static final int TRIG_TYPE_PAIR0 = 0x6;
31      public static final int TRIG_TYPE_PAIR1 = 0x7;
32      
33      // Collections for storing the decoded SSP bank data.
34      private final List<SSPCluster> clusterList = new ArrayList<SSPCluster>();
35      private final List<SSPTrigger> triggerList = new ArrayList<SSPTrigger>();
36      private final List<SSPPairTrigger> pairList = new ArrayList<SSPPairTrigger>();
37      private final List<SSPCosmicTrigger> cosmicList = new ArrayList<SSPCosmicTrigger>();
38      private final List<SSPSinglesTrigger> singlesList = new ArrayList<SSPSinglesTrigger>();
39      
40      // Other SSP bank information.
41      private int eventNumber = 0;
42      private long triggerTime = 0;
43      
44      /**
45       * Instantiates an <code>SSPData</code> object from a bank of
46       * integer primitives defining an EVIO-encoded SSP trigger bank.
47       * @param bank - The EVIO bank from which to generate the object.
48       */
49      public SSPData(int[] bank) {
50          super(bank);
51          decodeData();
52      }
53      
54      /**
55       * Instantiates an <code>SSPData</code> object from an LCIO
56       * <code>GenericObject</code> object that contains an EVIO-encoded
57       * bank of integer primitives which represent an SSP trigger bank.
58       * @param sspData - The <code>GenericObject</code> containing the
59       * integer bank from which to generate the object.
60       */
61      public SSPData(GenericObject sspData) {
62          super(sspData, BANK_TAG);
63          decodeData();
64      }
65      
66      @Override
67      public int getTag() {
68          return BANK_TAG;
69      }
70      
71      @Override
72      protected final void decodeData() {
73          // Parse over the integer EVIO words and handle each type. Block
74          // headers and block trailers can be ignored because these are
75          // disentangled in the secondary CODA readout list.
76          for (int ii = 0; ii < bank.length; ii++) {
77              // Process the event number.
78              if (((bank[ii] >> 27) & (0x1f)) == TRIG_HEADER) {
79                  eventNumber = bank[ii] & 0x7FFFFFF;
80              }
81              
82              // Process the trigger time.
83              else if (((bank[ii] >> 27) & (0x1f)) == TRIG_TIME) {
84                  triggerTime = (bank[ii + 1] << 24) | (bank[ii] & 0xffffff);
85              }
86              
87              // Process SSP trigger data.
88              else if (((bank[ii] >> 27) & (0x1f)) == TRIG_TYPE) {
89                  // Parse the trigger information. Note that type is
90                  // the trigger identification bits and ranges from
91                  // zero to seven.
92                  int type = (bank[ii] >> 23) & 0xf;
93                  int data = (bank[ii] >> 16) & 0x7f;
94                  int time = (bank[ii]) & 0x3ff; 
95                  
96                  // Create an SSPTrigger and add it to the list.
97                  SSPTrigger trigger = SSPTriggerFactory.makeTrigger(type, time * 4, data);
98                  if(trigger instanceof SSPSinglesTrigger) { singlesList.add((SSPSinglesTrigger) trigger); }
99                  else if(trigger instanceof SSPPairTrigger) { pairList.add((SSPPairTrigger) trigger); }
100                 else if(trigger instanceof SSPCosmicTrigger) { cosmicList.add((SSPCosmicTrigger) trigger); }
101                 triggerList.add(trigger);
102             }
103             
104             // Process SSP clusters.
105             else if (((bank[ii] >> 27) & (0x1f)) == CLUSTER_TYPE) {
106                 // Get the number of hits in the cluster and add it
107                 // to the cluster hits list.
108                 int hits = (bank[ii] >> 23) & 0xf;
109                 
110                 // Get the cluster energy (which is in MeV) and add it
111                 // to the cluster energy list.
112                 int energy = (bank[ii] >> 10) & 0x1fff;
113                 
114                 // Get the cluster y-index.
115                 int iy = (bank[ii] >> 6) & 0xf;
116                 
117                 // If the first bit of the index is 1, then it is a
118                 // negative number and needs to be converted using
119                 // two's complement to get the proper value.
120                 if(((iy >> 3) & 0x1) == 0x1) {
121                     // Perform the two's complement. ('^' is the bit
122                     // wise inversion operator).
123                     iy = iy ^ (0xf);
124                     iy += 1;
125                     iy *=-1;
126                 }
127                 
128                 // Get the x-index of the cluster.
129                 int ix = (bank[ii]) & 0x3f;
130                 
131                 // If the first bit of the index is 1, then it is a
132                 // negative number and needs to be converted using
133                 // two's complement to get the proper value.
134                 if(((ix >> 5) & 0x1) == 0x1) {
135                     // Perform the two's complement. ('^' is the bit
136                     // wise inversion operator).
137                     ix = ix ^ (0x3f);
138                     ix += 1;
139                     ix *=-1;
140                 }
141                 
142                 // X-index values are encoded from -22 to 23; since
143                 // LCSIM defines these values in a range from -23 to
144                 // 23, zero and negative values need be shifted down
145                 // by an additional step to be accurate.
146                 if(ix <= 0) { ix -= 1; }
147                 
148                 // Get the cluster time. Time is 4 ns clock-cycles.
149                 int time = (bank[ii + 1]) & 0x3ff;
150                 
151                 // Create an SSPCluster from the parsed information
152                 // and add it to the cluster list.
153                 SSPCluster cluster = new SSPCluster(ix, iy, energy, hits, time * 4);
154                 clusterList.add(cluster);
155             }
156         }
157     }

159     /**
160      * Gets the list of clusters reported by the SSP.
161      * @return Returns the clusters as a <code>List</code> collection
162      * of <code>SSPCluster</code> objects.
163      */
164     public List<SSPCluster> getClusters() {
165         return clusterList;
166     }
167     
168     /**
169      * Gets the list of cosmic triggers reported by the SSP.
170      * @return Returns the triggers as a <code>List</code> collection
171      * of <code>SSPCosmicTrigger</code> objects.
172      */
173     public List<SSPCosmicTrigger> getCosmicTriggers() {
174         return cosmicList;
175     }
176     
177     /**
178      * Gets the list of pair triggers reported by the SSP.
179      * @return Returns the triggers as a <code>List</code> collection
180      * of <code>SSPPairTrigger</code> objects.
181      */
182     public List<SSPPairTrigger> getPairTriggers() {
183         return pairList;
184     }
185     
186     /**
187      * Gets the list of singles triggers reported by the SSP.
188      * @return Returns the triggers as a <code>List</code> collection
189      * of <code>SSPSinglesTrigger</code> objects.
190      */
191     public List<SSPSinglesTrigger> getSinglesTriggers() {
192         return singlesList;
193     }
194     
195     /**
196      * Gets the trigger time reported by the SSP.
197      * @return Returns the trigger time as a <code>long</code>.
198      */
199     public long getTime() { return triggerTime; }
200     
201     /**
202      * Gets the event number reported by the SSP.

204      */
205     public int getEventNumber() { return eventNumber; }
