3   import java.util.BitSet;
4   import java.util.logging.Logger;

6   /**
7    * Class <code>VTPSinglesTrigger</code> Parse HPS Singles Trigger of VTP and store information. 
8    * HPS Singles Trigger:
9    * <ul>
10   * <li>10bit [0 : 9] trigger time in 4 ns unit</li> 
11   * <li>9bit [10 : 18] pass bits</li>
12   * <li>1bit [19 : 19] TOP or BOT</li>
13   * <li>3bit [20 : 22] single cluster trigger bit instance: 0 to 3</li> 
14   * </ul>
15   */
16  public class VTPSinglesTrigger {   
17      // singles trigger's variables.
18      private int t; // in 4 ns units
19      private BitSet passBits = new BitSet(9);
20      private int tOrB; // 1 means top; 0 means bot 
21      private int inst;
22      
23      
24      /**
25       * Initialize the logger.
26       */
27      private static final Logger LOGGER = Logger.getLogger(VTPSinglesTrigger.class.getPackage().getName());
28  
29      /**
30       * Creates a new <code>VTPSinglesTrigger</code> object.
31       * 
32       * @param word - HPS Singles Trigger in the VTP bank.
33       */
34      
35      public VTPSinglesTrigger(int word) {
36          decodeData(word);   
37      }
38      
39      public final void decodeData(int word){
40          t = word & 0x3FF;
41          passBits = BitSet.valueOf(new long [] {word >> 10 & 0x01FF});
42          tOrB = word >> 19 & 0x01;
43          inst = word >> 20 & 0x07;
44          
45          // Make sure that the input values are valid.
46          if (inst > 3) {
47              LOGGER.warning("Received out-of-bounds singles trigger instance.");
48          }
49          if (t < 0) {
50              LOGGER.warning("Received singles trigger with negative time.");
51          }
52      }
53      
54      /**
55       * Get singles trigger time in ns referenced from the beginning of the readout window.
56       * 
57       * @return Return the singles trigger time as a <code>long</code>.
58       */
59      public long getTime() {
60          return (long)t * 4;
61      }
62      
63      /**
64       * Get singles trigger instance.
65       * 
66       * @return Return singles trigger instance as an <code>int</code>.
67       */
68      public int getTriggerInstance() {
69          return inst;
70      }
71      
72      /**
73       * Get singles trigger pass bits.
74       * 
75       * @return Returns singles trigger pass bits.
76      */
77      public BitSet getPassBits() {
78          return passBits;
79      }
80      
81      /**
82       * Check if a cut was passed.
83       * 
84       * @param bitIndex - index of a pass bit.
85       * 
86       * @return Returns <code>true</code> if the corresponding cut was passed, and
87       * <code>false</code> otherwise.
88       */
89      public boolean checkPass(int bitIndex) {
90          if(bitIndex > 8) throw new RuntimeException("Index " + bitIndex + " is out of range (0 : 8)");
91          else return passBits.get(bitIndex);
92      }
93      
94      /**
95       * Indicate if pass emin.
96       */
97      public boolean passEMin() {
98          return passBits.get(0);
99      }
