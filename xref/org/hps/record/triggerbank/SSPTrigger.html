1   package org.hps.record.triggerbank;
2   
3   import java.util.logging.Logger;
4   
5   import org.hps.record.scalers.ScalersEvioProcessor;
6   
7   /**
8    * Class <code>SSPTrigger</code> represents the data output by the SSP
9    * for a trigger. Individual implementing classes are expected to handle
10   * parsing the trigger bit data as appropriate for their type.
11   */
12  public class SSPTrigger {
13      // Trigger data.
14      protected final int type;
15      protected final int time;
16      protected final int data;
17      
18      /**
19       * Initialize the logger.
20       */
21      private static final Logger LOGGER = Logger.getLogger(ScalersEvioProcessor.class.getPackage().getName());
22      
23      /**
24       * Instantiates a new <code>SSPTrigger</code> with the indicated
25       * trigger data.
26       * @param type - The type of trigger.
27       * @param time - The time at which the trigger occurred in ns.
28       * @param data - The trigger bit data.
29       */
30      public SSPTrigger(int type, int time, int data) {
31          // Log any issues with processing the trigger.
32          if(!SSPTriggerFactory.isKnownTriggerType(type)) {
33              LOGGER.warning(String.format("Trigger type %d is not recognized.", type));
34          }
35          
36          // Store the trigger data.
37          this.type = type;
38          this.time = time;
39          this.data = data;
40          
41          // Note that a trigger was made.
42          LOGGER.fine(String.format("Constructed trigger of type %d occurred at time %3d with data %d.",
43                  type, time, data));
44      }
45      
46      /**
47       * Gets the raw, unparsed trigger data bank for this trigger.
48       * @return Returns the trigger data bank as an <code>int</code>.
49       */
50      public int getData() { return data; }
51      
52      /**
53       * Gets the type code for the trigger.
54       * @return Returns the trigger type as an <code>int</code>.
55       */
56      public int getType() { return type; }
57      
58      /**
59       * Gets the time at which the trigger occurred.
60       * @return Returns the trigger time as an <code>int</code>.
61       */
62      public int getTime() { return time; }
63  }
