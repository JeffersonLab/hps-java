3   import java.util.BitSet;
4   import java.util.logging.Level;
5   import java.util.logging.Logger;
6   
7   /**
8    * Class <code>VTPFEETrigger</code> Parse HPS FEE Trigger of VTP and store information. 
9    * HPS FEE Trigger:
10   * <ul>
11   * <li>10bit [0 : 9] trigger time in 4 ns unit</li> 
12   * <li>7bit [10 : 16] bit mask specify which FEE regions (ECal x-ranges) triggered the system</li>
13   * </ul>
14   */
15  public class VTPFEETrigger {
16      // FEE trigger's variables.
17      private int t; // in 4 ns units
18      private BitSet regionBits = new BitSet(7);
19  
20  
21      /**
22       * Initialize the logger.
23       */
24      private static final Logger LOGGER = Logger.getLogger(VTPFEETrigger.class.getPackage().getName());
25  
26      /**
27       * Creates a new <code>VTPFEETrigger</code> object.
28       * 
29       * @param word - HPS FEE Trigger in the VTP bank.
30       */
31  
32      public VTPFEETrigger(int word) {
33          decodeData(word);   
34      }
35  
36      public final void decodeData(int word){
37          t = word & 0x3FF;
38          regionBits = BitSet.valueOf(new long [] {word >> 10 & 0x07F});
39  
40          // Make sure that the input values are valid.
41          if (t < 0) {
42              LOGGER.warning("Received FEE trigger with negative time.");
43          }
44      }
45  
46      /**
47       * Get FEE trigger time in ns referenced from the beginning of the readout window.
48       * 
49       * @return Return the FEE trigger time as a <code>long</code>.
50       */
51      public long getTime() {
52          return (long)t * 4;
53      }
54  
55      /**
56       * Get singles trigger pass bits.
57       * 
58       * @return Returns singles trigger pass bits.
59       */
60      public BitSet getRegionBits() {
61          return regionBits;
62      }
63  
64      /**
65       * Check if a region triggered system.
66       * 
67       * @param bitIndex - index of a region bit.
68       * 
69       * @return Returns <code>true</code> if the corresponding region triggered system, and
70       * <code>false</code> otherwise.
71       */
72      public boolean checkRegion(int bitIndex) {
73          if(bitIndex > 6) throw new RuntimeException("Index " + bitIndex + " is out of range (0 : 6)");
74          else return regionBits.get(bitIndex);
75      }
76  
77      /**
78       * List indices of regions triggered system.
79       */
80      public String listIndicesofRegions() {
81          return regionBits.toString();
82      }
83  
84      /**
85       * Save indices of regions, which triggered the system, into an integer array.
86       */
87      public int[] getIndicesOfRegions() {
88          String str = regionBits.toString();
89          String[] items = str.replaceAll("\\{", "").replaceAll("\\}", "").replaceAll("\\s", "").split(",");
90          int[] arr = new int[items.length];
91          for (int i = 0; i < items.length; i++) {
92              try {
93                  arr[i] = Integer.parseInt(items[i]);
94              } catch (NumberFormatException e) {
95                  LOGGER.log(Level.SEVERE, "parsing a string as a signed integer is failed", e);
96              };
97          }
98          return arr;
99      }
100 
101     /**
102      * Get number of regions which triggered system.
103      */
104     public int getNumberOfRegions() {
105         return this.getIndicesOfRegions().length;
106     }
107 
108     /**
109      * Indicate if region 0 triggered the system.
110      */
111     public boolean isRegion0() {
112         return regionBits.get(0);
113     }

117      */

133         return regionBits.get(3);

136     /**
137      * Indicate if region 4 triggered the system.

139     public boolean isRegion4() {
140         return regionBits.get(4);

154         return regionBits.get(6);
