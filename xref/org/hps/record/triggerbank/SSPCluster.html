1   package org.hps.record.triggerbank;

3   import java.util.logging.Logger;
4   
5   import org.hps.record.scalers.ScalersEvioProcessor;
6   
7   /**
8    * Class <code>SSPCluster</code> stores all of the information on clusters that is reported by the SSP. SSP clusters store:
9    * <ul>
10   * <li>Cluster center x-index</li>
11   * <li>Cluster center y-index</li>
12   * <li>Cluster total energy</li>
13   * <li>Cluster hit count</li>
14   * <li>Cluster time</li>
15   * </ul>
16   * <code>SSPCluster</code> does not support the ability to track individual hits that are part of a cluster.
17   */
18  public final class SSPCluster {
19  
20      // Cluster definition variables.
21      private final int ix;
22      private final int iy;
23      private final int n;
24      private final int t;
25      private final double e;
26  
27      /**
28       * Initialize the logger.
29       */
30      private static final Logger LOGGER = Logger.getLogger(ScalersEvioProcessor.class.getPackage().getName());
31  
32      /**
33       * Creates a new <code>SSPCluster</code> object.
34       * 
35       * @param ix - The x-index of the cluster.
36       * @param iy - The y-index of the cluster.
37       * @param energy - The cluster energy in MeV.
38       * @param hits - The cluster hit count.
39       * @param time - The time at which the cluster occurred in ns.
40       */
41      public SSPCluster(int ix, int iy, int energy, int hits, int time) {
42          // Make sure that the input values are valid.
43          if (ix == 0 || ix < -23 || ix > 23) {
44              LOGGER.warning(String.format("Received out-of-bounds ix value of %d.", ix));
45          }
46          if (iy == 0 || iy < -5 || iy > 5) {
47              LOGGER.warning(String.format("Received out-of-bounds iy value of %d.", iy));
48          }
49          if (energy < 0) {
50              LOGGER.warning("Received negative energy for cluster.");
51          }
52          if (hits <= 0) {
53              LOGGER.warning("Received cluster with zero or fewer hits.");
54          }
55          if (time < 0) {
56              LOGGER.warning("Received cluster with negative time.");
57          }
58  
59          // Define the cluster parameters.
60          this.ix = ix;
61          this.iy = iy;
62          this.e = energy / 1000.0;
63          this.t = time;
64          this.n = hits;
65  
66          // Indicate that the cluster was made.
67          LOGGER.fine(String.format("Constructed cluster at (%3d, %3d) at time %3d ns with energy %4d MeV and %d hits.",
68                  ix, iy, time, energy, hits));
69      }
70  
71      /**
72       * Gets the x-index of the cluster.
73       * 
74       * @return Returns the cluster x-index as an <code>int</code>.
75       */
76      public int getXIndex() {
77          return ix;
78      }
79  
80      /**
81       * Gets the y-index of the cluster.
82       * 
83       * @return Returns the cluster y-index as an <code>int</code>.
84       */
85      public int getYIndex() {
86          return iy;
87      }
88  
89      /**
90       * Gets the number of hits in the cluster.
91       * 
92       * @return Returns the cluster hit count as an <code>int</code>.
93       */
94      public int getHitCount() {
95          return n;
96      }
97  
98      /**
99       * Gets the cluster time in nanoseconds.
100      * 
101      * @return Returns the cluster time as an <code>int</code>.
102      */
103     public int getTime() {
104         return t;
105     }
106 
107     /**
108      * Gets the energy of the cluster in GeV.
109      * 
110      * @return Returns the cluster energy as a <code>double</code>.
111      */
112     public double getEnergy() {
113         return e;
114     }
115 }
