1   package org.hps.record.triggerbank;
2   
3   import java.util.logging.Logger;
4   
5   /**
6    * Class <code>VTPCluster</code> Parse HPS Cluster of VTP and store information. 
7    * HPS Cluster Trigger:
8    * <ul>
9    * <li>6bit signed cluster X coordinate in the first word; Range: [-22, 0] and [1, 23]</li>
10   * <li>4bit signed Y coordinate in the first word; Range: [-5, -1] and [1, 5]</li>
11   * <li>13bit cluster energy in MeV in the first word</li>
12   * <li>10bit cluster time in 4 ns unit in the second word</li> 
13   * <li>4bit number of hits in the second word</li>
14   * </ul>
15   */
16  public class VTPCluster {
17      // cluster's variables.
18      private int ix;
19      private int iy;
20      private double e; // GeV
21      private int t; // in 4 ns units
22      private int n;
23  
24      /**
25       * Initialize the logger.
26       */
27      private static final Logger LOGGER = Logger.getLogger(VTPCluster.class.getPackage().getName());
28  
29      /**
30       * Creates a new <code>VTPCluster</code> object.
31       * 
32       * @param word0 - The first word for HPS cluster in the VTP bank.
33       * @param word1 - The second word for HPS cluster in the VTP bank.
34       */
35      
36      public VTPCluster(int word0, int word1) {
37          decodeData(word0, word1);   
38      }
39      
40      /**
41       * Parses words of HPS Cluster
42       */
43      public final void decodeData(int word0, int word1){
44          ix = word0 & 0x03F;
45          // If the first bit of the index is 1, then it is a negative number 
46          if((ix >> 5 & 0x1) == 0x1) ix = -((ix ^ 0x3F) + 1);
47          iy = word0 >> 6 & 0x0F;
48          // If the first bit of the index is 1, then it is a negative number 
49          if((iy >> 3 & 0x1) == 0x1) iy = -((iy ^ 0xF) + 1);
50          e = (double)(word0 >> 10 & 0x01FFF);
51          t = word1 & 0x03FF;
52          n = word1 >> 10 & 0x0F;    
53          
54          // Make sure that the input values are valid.
55          if (ix < -22 || ix > 23) {
56              LOGGER.warning(String.format("Received out-of-bounds ix value of %d.", ix));
57          }
58          if (iy == 0 || iy < -5 || iy > 5) {
59              LOGGER.warning(String.format("Received out-of-bounds iy value of %d.", iy));
60          }
61          if (e < 0) {
62              LOGGER.warning("Received negative energy for cluster.");
63          }
64          if (n <= 0) {
65              LOGGER.warning("Received cluster with zero or fewer hits.");
66          }
67          if (t < 0) {
68              LOGGER.warning("Received cluster with negative time.");
69          }
70      }
71      
72      /**
73       * Get the x-coordinate of the cluster.
74       * 
75       * @return Return the cluster x-coordinate as an <code>int</code>.
76       */
77      public int getXIndex() {
78          return ix;
79      }
80  
81      /**
82       * Get the y-coordinate of the cluster.
83       * 
84       * @return Return the cluster y-coordinate as an <code>int</code>.
85       */
86      public int getYIndex() {
87          return iy;
88      }
89  
90      /**
91       * Gets the number of hits in the cluster.
92       * 
93       * @return Returns the cluster hit count as an <code>int</code>.
94       */
95      public int getHitCount() {
96          return n;
97      }
98  
99      /**
100      * Gets the cluster time in ns.
101      * 
102      * @return Returns the cluster time as an <code>int</code>.
103      */
104     public long getTime() {
105         return (long)t * 4;
106     }
107 
108     /**
109      * Gets the energy of the cluster in GeV.
110      * 
111      * @return Returns the cluster energy as a <code>double</code>.
112      */
113     public double getEnergy() {
114         return e/1000.;
115     }
116 }
