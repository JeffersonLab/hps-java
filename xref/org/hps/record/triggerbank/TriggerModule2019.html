255      * Get the value of the cut specified by the identifier given in the argument to
256      * the specified value.
257      * 
258      * @param cut - The identifier of the cut to which the new value should be
259      *            assigned. These can be obtained as publicly-accessible static
260      *            variables from this class.
261      * @return the value of the cut specified by the identifier given in the
262      *         argument to the specified value.
263      * @throws IllegalArgumentException if the argument cut identifier is not valid.
264      */
265     public double getCutValue(String cut) throws IllegalArgumentException {
266         // Make sure that the cut exists. If it does, change it to the
267         // new cut value.
268         if (generalCuts.containsKey(cut)) {
269             return generalCuts.get(cut);
270         } else if (singlesTriggerCuts.containsKey(cut)) {
271             return singlesTriggerCuts.get(cut);
272         } else if (pairsTriggerCuts.containsKey(cut)) {
273             return pairsTriggerCuts.get(cut);
274         }
275 
276         // Otherwise, throw an exception.
277         else {
278             throw new IllegalArgumentException(String.format("Cut \"%s\" does not exist.", cut));
279         }
280     }

1154                 && (layer2.getHitStatus(HodoscopePattern.HODO_LX_2) || layer2.getHitStatus(HodoscopePattern.HODO_LX_3)))

1784     /**
1785      * Forms all possible pairs of top/bottom clusters from a given list of
1786      * clusters.
1787      * 
1788      * @param clusters - The clusters from which to form the pairs.
1789      * @return Returns a <code>List</code> collection of size two
1790      *         <code>Cluster</code> arrays representing top/bottom pairs. The first
1791      *         entry in the array is always the top cluster, with the bottom cluster
1792      *         in the next position.
1793      */
1794     public static <E> List<E[]> getTopBottomPairs(List<E> clusters, Class<E> clusterType)
1795             throws IllegalArgumentException {
1796         // Ensure that only valid cluster types are processed.
1797         if (!clusterType.equals(Cluster.class) && !clusterType.equals(VTPCluster.class)) {
1798             throw new IllegalArgumentException(
1799                     "Class \"" + clusterType.getSimpleName() + "\" is not a supported cluster type.");
1800         }
1801 
1802         // Create a list to store top clusters, bottom clusters, and
1803         // cluster pairs.
1804         List<E> topClusters = new ArrayList<E>();
1805         List<E> botClusters = new ArrayList<E>();
1806         List<E[]> pairClusters = new ArrayList<E[]>();
1807 
1808         // Separate the cluster list into top/bottom clusters.
1809         for (E cluster : clusters) {
1810             // Process LCIO clusters...
1811             if (clusterType.equals(Cluster.class)) {
1812                 if (getClusterYIndex((Cluster) cluster) > 0) {
1813                     topClusters.add(cluster);
1814                 } else {
1815                     botClusters.add(cluster);
1816                 }
1817             }
1818 
1819             // Process VTP clusters...
1820             else if (clusterType.equals(VTPCluster.class)) {
1821                 if (getClusterYIndex((VTPCluster) cluster) > 0) {
1822                     topClusters.add(cluster);
1823                 } else {
1824                     botClusters.add(cluster);
1825                 }
1826             }
1827         }
1828 
1829         // Form all top/bottom cluster pairs.
1830         for (E topCluster : topClusters) {
1831             for (E botCluster : botClusters) {
1832                 @SuppressWarnings("unchecked")
1833                 E[] pair = (E[]) Array.newInstance(clusterType, 2);
1834                 pair[0] = topCluster;
1835                 pair[1] = botCluster;
1836                 pairClusters.add(pair);
1837             }
1838         }
1839 
1840         // Return the cluster pairs.
1841         return pairClusters;
1842     }

1844     /**
1845      * Calculates the value used by the energy slope cut.
1846      * 
1847      * @param clusterPair       - The cluster pair from which the value should be
1848      *                          calculated.
1849      * @param energySlopeParamF - The value of the variable F in the energy slope
1850      *                          equation E_low + R_min * F.
1851      * @return Returns the cut value.
1852      */
1853     private static double getValueEnergySlope(double energy[], double x[], double y[], double energySlopeParamF) {
1854         // Determine which cluster is the lower-energy cluster.
1855         int lei = energy[0] < energy[1] ? 0 : 1;
1856 
1857         // E + R*F
1858         // Get the low energy cluster energy.
1859         double slopeParamE = energy[lei];
1860 
1861         // Get the low energy cluster radial distance.
1862         double slopeParamR = Math.sqrt((x[lei] * x[lei]) + (y[lei] * y[lei]));
1863 
1864         // Calculate the energy slope.
1865         return slopeParamE + slopeParamR * energySlopeParamF;
1866     }

1870      * This is the hardware position mapping, and does not always align perfectly
1871      * with the LCSim coordinate system. For instance, the electron side of the
1872      * detector is has positive x coordinates in the hardware system, and is
1873      * negative in LCSim.
1874      * 
1875      * @param ix - The crystal x-index.
1876      * @param iy - The crystal y-index.

1882      *                                   either of the cases where
1883      *                                   <code>iy == 0</code> or
1884      *                                   <code>|iy| > 5</code> for the y-index.
1885      */
1886     private static double[] getCrystalPosition(int ix, int iy) throws IndexOutOfBoundsException {

1888         if (ix == 0 || ix < -23 || ix > 23) {
1889             throw new IndexOutOfBoundsException(String.format("Value \"%d\" is invalid for field x-index.", ix));

1892             throw new IndexOutOfBoundsException(String.format("Value \"%d\" is invalid for field y-index.", iy));
1893         }
1894 
1895         // Get the position map.
1896         double posMap[];
1897         if (ix < 1) {
1898             posMap = position[5 - iy][22 - ix];
1899         } else {
1900             posMap = position[5 - iy][23 - ix];
1901         }
1902 
1903         // Return the corrected mapped position.
1904         return new double[] { posMap[0], posMap[2], posMap[1] };
