26  /**
27   * This class converts an LCIO ReconstructedParticle to the HepRep format for
28   * display by Wired4.  Clusters and tracks from a RP are given the same randomly
29   * generated color.  Charged Tracks, Neutral Tracks, and Calorimeter Clusters are
30   * given their own sub-types, so that they can be easily switched on and off.
31   */
32  class HPSParticleConverter implements HepRepCollectionConverter
33  {
34      
35      private static final double[] ORIGIN = {0, 0, 0.00001};
36      RandomColorMap rcolorMap;
37  
38      private static final double[] zero = {0, 0, 0};
39      
40      public HPSParticleConverter()
41      {
42          rcolorMap = new RandomColorMap(50);
43      }
44  
45      public boolean canHandle(Class k)
46      {
47          return ReconstructedParticle.class.isAssignableFrom(k);
48      }
49      
50      public void convert(EventHeader event, List collection, HepRepFactory factory, HepRepTypeTree typeTree, HepRepInstanceTree instanceTree)
51      {               
52          rcolorMap.reset(collection.size());
53          
54          String pfoName = event.getMetaData(collection).getName();        
55          
56          // RP top-level type.
57          HepRepType rpType = factory.createHepRepType(typeTree, pfoName);
58          rpType.addAttValue("layer", LCSimHepRepConverter.PARTICLES_LAYER);
59          
60          // Neutral Tracks type.
61          HepRepType neutralTracksType = factory.createHepRepType(rpType, "NeutralTracks");
62          neutralTracksType.addAttValue("layer", LCSimHepRepConverter.PARTICLES_LAYER);
63          neutralTracksType.addAttValue("drawAs","Line");
64          /*neutralTracksType.addAttValue("LineStyle", "Dashed"); // Doesn't work.
65          neutralTracksType.addAttValue("color",Color.GREEN);*/
66          
67          // Charged Tracks type.
68          HepRepType chargedTracksType = factory.createHepRepType(rpType, "ChargedTracks");
69          chargedTracksType.addAttValue("layer", LCSimHepRepConverter.PARTICLES_LAYER);
70          chargedTracksType.addAttValue("drawAs","Line");
71          
72          // Clusters type.
73          HepRepType clustersType = factory.createHepRepType(rpType, "CalClusters");
74          clustersType.addAttValue("color",Color.RED);
75          clustersType.addAttValue("fill",true);
76          clustersType.addAttValue("fillColor",Color.RED);
77          clustersType.addAttValue("layer", LCSimHepRepConverter.HITS_LAYER);
78          
79          /*
80          clustersType.addAttValue("layer", LCSimHepRepConverter.HITS_LAYER);
81          clustersType.addAttValue("drawAs", "Point");
82          clustersType.addAttValue("MarkName", "Box");
83          clustersType.addAttDef("energy", "Hit Energy", "physics", "");
84          clustersType.addAttDef("cluster", "Cluster Energy", "physics", "");
85          */
86          
87          int rpCnt = 0;
88          for (ReconstructedParticle rp : (List<ReconstructedParticle>) collection)
89          {                      
90              // Get the color for this RP.
91              Color rpColor = rcolorMap.getColor(rpCnt % rcolorMap.size());
92              
93              List<Cluster> clusters = rp.getClusters();
94              List<Track> tracks = rp.getTracks();
95              
96              convertClusters(event, clusters, factory, typeTree, instanceTree, rpColor, clustersType);                                                                    
97              convertTracks(event, tracks, factory, typeTree, instanceTree, rpColor, chargedTracksType);
98              
99              if (rp.getCharge() == 0)
100             {
101                 convertNeutralParticle(event.getDetector(), rp, instanceTree, factory, neutralTracksType, rpColor);
102             }                      
103             
104             ++rpCnt;
105         }
106     }
107     
108     private void convertClusters(EventHeader event, List<Cluster> collection, HepRepFactory factory, HepRepTypeTree typeTree, HepRepInstanceTree instanceTree, Color clusterColor, HepRepType type)
109     {       
110         List<CalorimeterHit> hits = new ArrayList<CalorimeterHit>();
111         
112         HepRepInstance instanceC = factory.createHepRepInstance(instanceTree, type);
113         
114         for (Cluster cluster : collection)
115         {
116             if (cluster.getCalorimeterHits().size() != 0) {
117                 hits.addAll(cluster.getCalorimeterHits());
118             }
119         }
120            
121         if (hits != null)
122         {
123             for (CalorimeterHit hit : hits)
124             {
125                 if (hit != null) {
126                     double[] pos = hit.getPosition();
127                     HepRepInstance instanceX = factory.createHepRepInstance(instanceC, type);
128                     instanceX.addAttValue("MarkSize", 5);
129                     instanceX.addAttValue("color", clusterColor);
130                     instanceX.addAttValue("showparentattributes", true);
131                     instanceX.addAttValue("pickparent", true);
132                     //HepRepPoint pp = 
133                     factory.createHepRepPoint(instanceX, pos[0], pos[1], pos[2]);
134                 }
135             }
136         }
137     }   
138 
139     private void convertTracks(EventHeader event, List<Track> collection, HepRepFactory factory, HepRepTypeTree typeTree, HepRepInstanceTree instanceTree, Color trackColor, HepRepType type)
140     {   
141         if (collection.size() == 0)
142             return;
143         
144         Detector detector = event.getDetector();
145 
146         double trackingRMax = detector.getConstants().get("tracking_region_radius").getValue();
147         double trackingZMax = detector.getConstants().get("tracking_region_zmax").getValue();
148 
149         double[] field = detector.getFieldMap().getField(ORIGIN);
150         HelixSwimmer helix = new HelixSwimmerYField(field[1]);
151              
152         for (Track t : (List<Track>) collection)
153         {                        
154             helix.setTrack(t);
155             double distanceToCylinder = helix.getDistanceToCylinder(trackingRMax, trackingZMax);
156 
157             HepRepInstance instanceX = factory.createHepRepInstance(instanceTree, type);
158             instanceX.addAttValue("color", trackColor);
159 
160             double dAlpha = 10; // 1cm
161             for (int k = 0; k < 200; k++)
162             {
163                 double d = k * dAlpha;
164                 if (d > distanceToCylinder)
165                     break;
166                 Hep3Vector point = helix.getPointAtDistance(d);
167                 factory.createHepRepPoint(instanceX, point.x(), point.y(), point.z());
168             }
169         }
170     }

172     // TODO: Connect to cluster center instead of swimming.
173     public void convertNeutralParticle(Detector detector, ReconstructedParticle p, HepRepInstanceTree tree, HepRepFactory factory, HepRepType type, Color rpColor)
174     {  
175         HepRepInstance instanceX = factory.createHepRepInstance(tree, type);
176         
177         double trackingRMax = detector.getConstants().get("tracking_region_radius").getValue();
178         double trackingZMax = detector.getConstants().get("tracking_region_zmax").getValue();
179 
180         double[] field = detector.getFieldMap().getField(zero);
181         HelixSwimmer helix = new HelixSwimmer(field[2]);
182         
183         double charge = p.getCharge();
184         Hep3Vector start =  p.getReferencePoint();
185         Hep3Vector momentum = p.getMomentum();
186         helix.setTrack(momentum, start, (int) charge);
187         double distanceToCylinder = helix.getDistanceToCylinder(trackingRMax,trackingZMax);
188         
189         Hep3Vector stop = helix.getPointAtDistance(distanceToCylinder);
190            
191         factory.createHepRepPoint(instanceX,start.x(),start.y(),start.z());
192         factory.createHepRepPoint(instanceX,stop.x(),stop.y(),stop.z());
193         instanceX.addAttValue("color", rpColor);
194         instanceX.addAttValue("momentum",p.getEnergy());
195         instanceX.addAttValue("type",p.getType());                   
196     }
