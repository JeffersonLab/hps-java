1   package org.hps.plugin;
2   
3   import org.freehep.application.studio.Plugin;
4   import org.freehep.application.studio.Studio;
5   import org.freehep.util.FreeHEPLookup;
6   import org.lcsim.event.ReconstructedParticle;
7   import org.lcsim.event.Track;
8   import org.lcsim.util.heprep.HepRepCollectionConverter;
9   import org.lcsim.util.heprep.LCSimHepRepConverter;
10  
11  /**
12   * JAS3 plugin for HPS.
13   * This install custom converter for the Track and ReconstructedParticle classes. 
14   */
15  public class HPSPlugin extends Plugin {
16      
17      /**
18       * Install a custom converter for HPS Track and ReconstructedParticle objects
19       * so that they are swum correctly in the B-field. 
20       */
21      protected void postInit() {
22          
23          Studio app = getApplication();
24          FreeHEPLookup lookup = app.getLookup();
25           
26          // Get the HepRep converter registered by lcsim.
27          Class<?> converterClass = null;
28          try {
29              converterClass = Class.forName("org.lcsim.util.heprep.LCSimHepRepConverter");
30          } catch (ClassNotFoundException e) {
31              throw new RuntimeException(e);
32          }                       
33          LCSimHepRepConverter converter = (LCSimHepRepConverter) lookup.lookup(converterClass);
34          if (converter == null) {
35              throw new RuntimeException("LCSimHepRepConverter was not found.");
36          }
37          
38          // Remove the existing track converter from the lcsim plugin.
39          HepRepCollectionConverter trackConverter = converter.findConverter(Track.class);
40          if (trackConverter != null) {
41              converter.deregister(trackConverter);            
42          } else {
43              throw new RuntimeException("The TrackConverter was not found.");
44          }
45          
46          // Register the HPS track converter in the heprep converter.
47          converter.register(new HPSTrackConverter());
48          
49          // Remove the existing particle converter.
50          HepRepCollectionConverter particleConverter = converter.findConverter(ReconstructedParticle.class);
51          if (particleConverter != null) {
52              converter.deregister(particleConverter);
53          } else {
54              throw new RuntimeException("The ReconstructedParticleConverter was not found.");
55          }
56          
57          // Register the HPS particle converter.
58          converter.register(new HPSParticleConverter());
59      }
60      
61  }
