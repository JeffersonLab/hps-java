23  /**
24   * Convert an HPS track to heprep for Wired display.
25   */
26  public class HPSTrackConverter implements HepRepCollectionConverter {
27  
28      private static final double[] ORIGIN = {0, 0, 0.00001};
29      
30      public boolean canHandle(Class k) {
31          return Track.class.isAssignableFrom(k);
32      }
33  
34      public void convert(EventHeader event, List collection, HepRepFactory factory, HepRepTypeTree typeTree, HepRepInstanceTree instanceTree) {
35          try {
36              event.getDetector();
37          } catch (Exception x) {
38              return;
39          }
40          
41          if (collection.size() == 0)
42              return;
43          
44          List<Track> trackList = (List<Track>) collection;
45          Track firstTrack = trackList.get(0);
46          
47          LCMetaData meta = event.getMetaData(collection);
48          String name = meta.getName();
49          Detector detector = event.getDetector();
50  
51          double trackingRMax = detector.getConstants().get("tracking_region_radius").getValue();
52          double trackingZMax = detector.getConstants().get("tracking_region_zmax").getValue();
53  
54          double[] field = detector.getFieldMap().getField(ORIGIN);
55          HelixSwimmer helix = new HelixSwimmerYField(field[1]);
56          
57          HepRepType typeX = factory.createHepRepType(typeTree, name);
58          typeX.addAttValue("layer", LCSimHepRepConverter.PARTICLES_LAYER);
59          typeX.addAttValue("drawAs", "Line");
60  
61          typeX.addAttDef("pT", "Transverse momentum", "physics", "");
62          typeX.addAttDef("dedX", "de/Dx", "physics", "GeV");
63          typeX.addAttDef("dedX error", "", "physics", "GeV");
64          typeX.addAttDef("Charge", "", "physics", "");
65          typeX.addAttDef("Chi2", "", "physics", "");
66          typeX.addAttDef("pX", "Momentum X", "physics", "GeV");
67          typeX.addAttDef("pY", "Momentum Y", "physics", "GeV");
68          typeX.addAttDef("pZ", "Momentum Z", "physics", "GeV");
69          typeX.addAttDef("NDF", "Number Degrees Freedom", "physics", "");
70          typeX.addAttDef("Reference Point X", "Reference Point X", "physics", "mm");
71          typeX.addAttDef("Reference Point Y", "Reference Point Y", "physics", "mm");
72          typeX.addAttDef("Reference Point Z", "Reference Point Z", "physics", "mm");
73          typeX.addAttDef("d0", "d0", "physics", "");
74          typeX.addAttDef("phi0", "phi0", "physics", "");
75          typeX.addAttDef("omega", "omega", "physics", "");
76          typeX.addAttDef("z0", "z0", "physics", "");
77          typeX.addAttDef("s", "s", "physics", "");
78  
79          for (Track track : trackList) {
80              
81              helix.setTrack(track);
82              double distanceToCylinder = helix.getDistanceToCylinder(trackingRMax, trackingZMax);
83  
84              TrackState ts = track.getTrackStates().get(0);
85              double[] referencePoint = ts.getReferencePoint();
86              double[] momentum = ts.getMomentum();
87  
88              HepRepInstance instanceX = factory.createHepRepInstance(instanceTree, typeX);
89              double pt = field[2] * Constants.fieldConversion / Math.abs(ts.getParameter(2));
90  
91              instanceX.addAttValue("pT", pt);
92              instanceX.addAttValue("dedX", track.getdEdx());
93              instanceX.addAttValue("dedX error", track.getdEdxError());
94              instanceX.addAttValue("Charge", track.getCharge());
95              instanceX.addAttValue("Chi2", track.getChi2());
96              instanceX.addAttValue("pX", momentum[0]);
97              instanceX.addAttValue("pY", momentum[1]);
98              instanceX.addAttValue("pZ", momentum[2]);
99              instanceX.addAttValue("NDF", track.getNDF());
100             instanceX.addAttValue("Reference Point X", referencePoint[0]);
101             instanceX.addAttValue("Reference Point Y", referencePoint[1]);
102             instanceX.addAttValue("Reference Point Z", referencePoint[2]);
103             instanceX.addAttValue("d0", ts.getParameter(0));
104             instanceX.addAttValue("phi0", ts.getParameter(1));
105             instanceX.addAttValue("omega", ts.getParameter(2));
106             instanceX.addAttValue("z0", ts.getParameter(3));
107             instanceX.addAttValue("s", ts.getParameter(4));
108 
109             double dAlpha = 10; // 1cm
110 
111             for (int k = 0; k < 200; k++) {
112                 double d = k * dAlpha;
113                 if (d > distanceToCylinder)
114                     break;
115                 Hep3Vector point = helix.getPointAtDistance(d);
116                 factory.createHepRepPoint(instanceX, point.x(), point.y(), point.z());
117             }
118         }
119     }       
120 }
