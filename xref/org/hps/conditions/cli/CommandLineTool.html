17  /**
18   * This class is a command-line tool for performing commands on the conditions database using sub-commands for
19   * operations such as 'add' and 'print'.
20   */
21  public final class CommandLineTool {
22  
23      /**
24       * Initialize the logger.
25       */
26      private static final Logger LOGGER = Logger.getLogger(CommandLineTool.class.getPackage().getName());
27  
28      private static Options OPTIONS = new Options();
29  
30      static {
31          OPTIONS.addOption(new Option("h", "help", false, "print help"));
32          OPTIONS.addOption(new Option("d", "detector", true, "detector name"));
33          OPTIONS.addOption(new Option("r", "run", true, "run number"));
34          OPTIONS.addOption(new Option("x", "xml", true, "conditions XML configuration file"));
35          OPTIONS.addOption(new Option("t", "tag", true, "conditions tag to use for filtering records"));
36      }
37  
38      /**
39       * Create a basic instance of this class.
40       *
41       * @return the instance of this class
42       */
43      private static CommandLineTool create() {
44          final CommandLineTool cli = new CommandLineTool();
45          cli.registerCommand(new LoadCommand());
46          cli.registerCommand(new PrintCommand());
47          cli.registerCommand(new AddCommand());
48          cli.registerCommand(new TagCommand());
49          cli.registerCommand(new RunSummaryCommand());
50          return cli;
51      }
52  
53      /**
54       * The main method for the class.
55       *
56       * @param arguments The command line arguments.
57       */
58      public static void main(final String[] arguments) {
59          CommandLineTool.create().run(arguments);
60      }
61  
62      /**
63       * The map of named command handlers.
64       */
65      private final Map<String, AbstractCommand> commands = new HashMap<String, AbstractCommand>();
66  
67      /**
68       * The database conditions system manager.
69       */
70      private DatabaseConditionsManager conditionsManager;
71  
72      /**
73       * The options parser.
74       */
75      private final PosixParser parser = new PosixParser();
76  
77      /**
78       * Exit with the given status.
79       *
80       * @param status the exit status
81       */
82      private void exit(final int status) {
83          System.exit(status);
84      }
85  
86      /**
87       * Print the usage statement for this tool to the console.
88       */
89      final void printUsage() {
90          final HelpFormatter help = new HelpFormatter();
91          final StringBuffer s = new StringBuffer();
92          for (final String command : this.commands.keySet()) {
93              s.append(command + '\n');
94          }
95          help.printHelp("CommandLineTool", "Commands:\n" + s.toString(), OPTIONS, "");
96      }
97  
98      /**
99       * Register a sub-command handler.
100      *
101      * @param command the sub-command handler
102      */
103     private void registerCommand(final AbstractCommand command) {
104         if (this.commands.containsKey(command.getName())) {
105             throw new IllegalArgumentException("There is already a command called " + command.getName());
106         }
107         this.commands.put(command.getName(), command);
108     }
109 
110     /**
111      * Run the command line tool, parsing the command line and sending arguments to sub-command handlers.
112      *
113      * @param arguments the command line arguments passed directly from {@link #main(String[])}
114      */
115     private void run(final String[] arguments) {
116         try {
117             if (arguments.length == 0) {
118                 this.printUsage();
119                 this.exit(0);
120             }
121 
122             CommandLine commandLine = null;
123             try {
124                 commandLine = this.parser.parse(OPTIONS, arguments, true);
125             } catch (final ParseException e) {
126                 LOGGER.log(Level.SEVERE, "Error parsing the options.", e);
127                 this.printUsage();
128                 this.exit(1);
129             }
130 
131             if (commandLine.hasOption("h") || commandLine.getArgs().length == 0) {
132                 this.printUsage();
133                 this.exit(0);
134             }
135 
136             // Setup conditions manager from command line options.
137             this.setupConditionsManager(commandLine);
138 
139             // Get the sub-command to use.
140             final String commandName = commandLine.getArgs()[0];
141             final AbstractCommand command = this.commands.get(commandName);
142             if (command == null) {
143                 throw new IllegalArgumentException("Unknown command " + commandName);
144             }
145 
146             // Copy remaining arguments for sub-command.
147             final String[] commandArguments = new String[commandLine.getArgs().length - 1];
148             System.arraycopy(commandLine.getArgs(), 1, commandArguments, 0, commandArguments.length);
149 
150             // Execute the sub-command.
151             command.execute(commandArguments);
152             command.cleanup();
153         } catch (final Exception e) {
154             e.printStackTrace();
155             System.exit(1);
156         } finally {
157             this.conditionsManager.closeConnection();
158         }
159     }
160 
161     /**
162      * Setup the conditions system based on command line arguments.
163      *
164      * @param commandLine the parsed command line arguments
165      */
166     private void setupConditionsManager(final CommandLine commandLine) {
167 
168         LOGGER.info("Setting up the conditions manager ...");
169 
170         // Create new manager.
171         this.conditionsManager = DatabaseConditionsManager.getInstance();
172 
173         // User specified tag of conditions records.
174         if (commandLine.hasOption("t")) {
175             final String tag = commandLine.getOptionValue("t");
176             this.conditionsManager.addTag(tag);
177             LOGGER.config("using tag " + tag);
178         }
179 
180         // If there is a run number or detector number then attempt to initialize the conditions system.
181         if (commandLine.hasOption("r") || commandLine.hasOption("d")) {
182 
183             if (!commandLine.hasOption("r")) {
184                 // Missing run number.
185                 throw new RuntimeException(
186                         "Missing -r option with run number which must be given when detector name is used.");
187             }
188 
189             if (!commandLine.hasOption("d")) {
190                 // Missing detector name.
191                 throw new RuntimeException(
192                         "Missing -d option with detector name which must be given when run number is used.");
193             }
194 
195             // Set detector name.
196             String detectorName = null;
197             if (commandLine.hasOption("d")) {
198                 detectorName = commandLine.getOptionValue("d");
199                 LOGGER.config("using detector " + detectorName);
200             } else {
201                 detectorName = "HPS-ECalCommissioning-v2";
202             }
203 
204             // Get run number.
205             Integer run = null;
206             if (commandLine.hasOption("r")) {
207                 run = Integer.parseInt(commandLine.getOptionValue("r"));
208                 LOGGER.config("using run " + run);
209             } else {
210                 run = 0;
211             }
212 
213             // Setup the conditions manager with user detector name and run number.
214             try {
215                 LOGGER.config("Initializing conditions manager with detector " + detectorName + " and run " + run
216                         + " ...");
217                 DatabaseConditionsManager.getInstance().setDetector(detectorName, run);
218                 LOGGER.config("Conditions manager was initialized successfully!");
219             } catch (final ConditionsNotFoundException e) {
220                 throw new RuntimeException("Error setting up the conditions manager.", e);
221             }
222         }
223         LOGGER.info("Done setting up the conditions manager!");
224     }
