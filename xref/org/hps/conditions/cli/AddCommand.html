1   package org.hps.conditions.cli;
2   
3   import java.sql.SQLException;
4   import java.util.Date;
5   import java.util.logging.Level;
6   import java.util.logging.Logger;
7   
8   import org.apache.commons.cli.CommandLine;
9   import org.apache.commons.cli.Option;
10  import org.apache.commons.cli.Options;
11  import org.hps.conditions.api.ConditionsRecord;
12  import org.hps.conditions.api.DatabaseObjectException;
13  import org.hps.conditions.api.FieldValuesMap;
14  import org.hps.conditions.api.TableRegistry;
15  import org.hps.conditions.database.DatabaseConditionsManager;
16  
17  /**
18   * This is a command for the conditions CLI that will add a conditions record, making a conditions set with a particular
19   * collection ID available by run number via the {@link org.hps.conditions.database.DatabaseConditionsManager}.
20   */
21  final class AddCommand extends AbstractCommand {
22  
23      /**
24       * Initialize the logger.
25       */
26      private static final Logger LOGGER = Logger.getLogger(AddCommand.class.getPackage().getName());
27  
28      /**
29       * Define command line options.
30       */
31      private static final Options OPTIONS = new Options();
32      static {
33          OPTIONS.addOption(new Option("h", "help", false, "print help for add command"));
34          OPTIONS.addOption("r", "run-start", true, "starting run number (required)");
35          OPTIONS.addOption("e", "run-end", true, "ending run number (default is starting run number)");
36          OPTIONS.addOption("t", "table", true, "table name (required)");
37          OPTIONS.addOption("c", "collection", true, "collection ID (required)");
38          OPTIONS.addOption("u", "user", true, "user name (optional)");
39          OPTIONS.addOption("m", "notes", true, "notes about this conditions set (optional)");
40      }
41  
42      /**
43       * Class constructor.
44       */
45      AddCommand() {
46          super("add", "Add a conditions record to associate a collection to a run range", OPTIONS);
47      }
48  
49      /**
50       * Create a conditions record.
51       *
52       * @param runStart the run start
53       * @param runEnd the run end
54       * @param tableName the table name
55       * @param name the key name
56       * @param collectionId the collection ID
57       * @param createdBy the user name
58       * @param tag the conditions tag
59       * @param notes the text notes about the collection
60       * @return the new conditions record
61       */
62      private ConditionsRecord createConditionsRecord(final int runStart, final int runEnd, final String tableName,
63              final String name, final int collectionId, final String createdBy, final String notes) {
64          final ConditionsRecord conditionsRecord = new ConditionsRecord();
65          final FieldValuesMap fieldValues = new FieldValuesMap();
66          fieldValues.setValue("run_start", runStart);
67          fieldValues.setValue("run_end", runEnd);
68          fieldValues.setValue("table_name", tableName);
69          fieldValues.setValue("name", name);
70          fieldValues.setValue("collection_id", collectionId);
71          fieldValues.setValue("created_by", createdBy);
72          if (notes != null) {
73              fieldValues.setValue("notes", notes);
74          }
75          conditionsRecord.setFieldValues(fieldValues);
76          fieldValues.setValue("created", new Date());
77          return conditionsRecord;
78      }
79  
80      /**
81       * Execute the command with the given arguments.
82       *
83       * @param arguments the command line arguments
84       */
85      @Override
86      final void execute(final String[] arguments) {
87  
88          final CommandLine commandLine = this.parse(arguments);
89  
90          // Run start (required).
91          final int runStart;
92          if (commandLine.hasOption("r")) {
93              runStart = Integer.parseInt(commandLine.getOptionValue("r"));
94          } else {
95              throw new RuntimeException("Missing required -r option with run number.");
96          }
97  
98          // Run end.
99          int runEnd = runStart;
100         if (commandLine.hasOption("e")) {
101             runEnd = Integer.parseInt(commandLine.getOptionValue("e"));
102         }
103 
104         // Name of table (required).
105         String tableName;
106         if (commandLine.hasOption("t")) {
107             tableName = commandLine.getOptionValue("t");
108         } else {
109             throw new RuntimeException("Missing required -t argument with table name");
110         }
111         final String name = tableName;
112 
113         // Collection ID (required).
114         int collectionId;
115         if (commandLine.hasOption("c")) {
116             collectionId = Integer.parseInt(commandLine.getOptionValue("c"));
117         } else {
118             throw new RuntimeException("Missing required -c argument with collection ID");
119         }
120 
121         // User name.
122         String createdBy = System.getProperty("user.name");
123         if (commandLine.hasOption("u")) {
124             createdBy = commandLine.getOptionValue("u");
125         }
126 
127         // Notes (optional).
128         String notes = null;
129         if (commandLine.hasOption("m")) {
130             notes = commandLine.getOptionValue("m");
131         }
132 
133         // Create the conditions record to insert.
134         final ConditionsRecord conditionsRecord = this.createConditionsRecord(runStart, runEnd, tableName, name,
135                 collectionId, createdBy, notes);
136         LOGGER.info("inserting conditions record ..." + '\n' + conditionsRecord);
137         try {
138             final DatabaseConditionsManager manager = this.getManager();
139             conditionsRecord.setConnection(manager.getConnection());
140             conditionsRecord.setTableMetaData(TableRegistry.getTableRegistry().findByTableName("conditions"));
141             conditionsRecord.insert();
142         } catch (final SQLException | DatabaseObjectException e) {
143             LOGGER.log(Level.SEVERE, "Error adding conditions record", e);
144             throw new RuntimeException("An error occurred while adding a conditions record.", e);
145         }
146         LOGGER.info("successfully added conditions record ..." + '\n' + conditionsRecord);       
147     }
148 }
