1   package org.hps.conditions.api;
2   
3   import java.sql.Connection;
4   import java.sql.SQLException;
5   
6   /**
7    * This is a simple ORM interface for mapping Java objects to a database.
8    */
9   public interface DatabaseObject {
10  
11      /**
12       * Delete the object from the database, using its table name from the meta data and row ID.
13       *
14       * @throws ConditionsObjectException if there was an error deleting the object
15       * @throws SQLException if there was a SQL query error
16       */
17      void delete() throws DatabaseObjectException, SQLException;
18  
19      /**
20       * Get the {@link TableMetaData} for this object.
21       *
22       * @return the {@link TableMetaData} for this object.
23       */
24      TableMetaData getTableMetaData();
25  
26      /**
27       * Insert the data of this object into the database.
28       * <p>
29       * This could be a single object or a collection of objects depending on the implementation.
30       *
31       * @throws ConditionsObjectException
32       * @throws SQLException
33       */
34      public void insert() throws DatabaseObjectException, SQLException;
35  
36      /**
37       * Return <code>true</code> if the record
38       *
39       * @return <code>true</code> if the object is new
40       */
41      boolean isNew();
42  
43      /**
44       * Select information from the database into this object using the row ID.
45       *
46       * @param id the row ID
47       * @return <code>true</code> if the select operation worked
48       * @throws DatabaseObjectException if there was an error selecting information into this object
49       * @throws SQLException if there was a query error
50       */
51      boolean select(final int id) throws DatabaseObjectException, SQLException;
52  
53      /**
54       * Set the database <code>Connection</code> for the object.
55       *
56       * @param connection the database <code>Connection</code> for the object
57       */
58      void setConnection(Connection connection);
59  
60      /**
61       * Set the table meta data of the object.
62       *
63       * @param tableMetaData the table meta data of the object
64       * @see TableMetaData
65       */
66      void setTableMetaData(TableMetaData tableMetaData);
67  
68      /**
69       * Update the information in the database from this object.
70       *
71       * @return <code>true</code> if an update occurred
72       * @throws ConditionsObjectException if there is an error performing the update from the object
73       * @throws SQLException if there is a SQL query error
74       */
75      boolean update() throws DatabaseObjectException, SQLException;
76  }
