1   package org.hps.conditions.svt;

3   import java.io.ByteArrayInputStream;
4   import java.io.FileWriter;
5   import java.io.IOException;
6   import java.io.InputStream;
7   
8   import org.hps.conditions.api.BaseConditionsObject;
9   import org.hps.conditions.api.BaseConditionsObjectCollection;
10  import org.hps.conditions.database.Converter;
11  import org.hps.conditions.database.Field;
12  import org.hps.conditions.database.MultipleCollectionsAction;
13  import org.hps.conditions.database.Table;
14  import org.jdom.Document;
15  import org.jdom.JDOMException;
16  import org.jdom.input.SAXBuilder;
17  import org.jdom.output.Format;
18  import org.jdom.output.XMLOutputter;
19  
20  /**
21   * This class is the conditions object model for an SVT configuration saved into the database.
22   */
23  @Table(names = {"svt_configurations"})
24  @Converter(multipleCollectionsAction = MultipleCollectionsAction.LAST_UPDATED)
25  public final class SvtConfiguration extends BaseConditionsObject {
26  
27      /**
28       * Collection implementation for {@link SvtConfiguration} objects.
29       */
30      @SuppressWarnings("serial")
31      public static class SvtConfigurationCollection extends BaseConditionsObjectCollection<SvtConfiguration> {
32      }
33  
34      /**
35       * Convert the raw database content for the configuration into an XML document.
36       *
37       * @return The Document created from the raw data.
38       * @throws IOException if there is an IO error
39       * @throws JDOMException is there is an XML parsing error
40       */
41      public Document createDocument() throws IOException, JDOMException {
42          final byte[] bytes = this.getFieldValue("content");
43          final InputStream inputStream = new ByteArrayInputStream(bytes);
44          final SAXBuilder builder = new SAXBuilder();
45          builder.setExpandEntities(false);
46  
47          builder.setValidation(false);
48          return builder.build(inputStream);
49      }
50  
51      /**
52       * Get the content of the XML file as a byte array.
53       *
54       * @return the content of the XML file as a byte array
55       */
56      @Field(names = {"content"})
57      public byte[] getContent() {
58          return this.getFieldValue("content");
59      }
60  
61      /**
62       * Get the filename associated with this configuration.
63       *
64       * @return The filename associated with the configuration.
65       */
66      @Field(names = {"filename"})
67      public String getFileName() {
68          return this.getFieldValue("filename");
69      }
70  
71      /**
72       * Save this configuration to a local file on disk using its name from the database.
73       */
74      public void writeToFile() {
75          final XMLOutputter out = new XMLOutputter();
76          out.setFormat(Format.getPrettyFormat());
77          try {
78              out.output(this.createDocument(), new FileWriter(this.getFileName()));
79          } catch (IOException | JDOMException e) {
80              throw new RuntimeException(e);
81          }
82      }
83  
84      /**
85       * Save this configuration to a local file on disk.
86       *
87       * @param filename the name of the local file
88       */
89      public void writeToFile(final String filename) {
90          final XMLOutputter out = new XMLOutputter();
91          out.setFormat(Format.getPrettyFormat());
92          try {
93              out.output(this.createDocument(), new FileWriter(filename));
94          } catch (IOException | JDOMException e) {
95              throw new RuntimeException(e);
96          }
97      }
98  }
