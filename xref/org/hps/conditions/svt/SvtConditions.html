1   package org.hps.conditions.svt;
2   
3   import static org.hps.conditions.svt.AbstractSvtChannel.MAX_NUMBER_OF_SAMPLES;
4   
5   import org.hps.conditions.svt.SvtChannel.SvtChannelCollection;
6   import org.hps.conditions.svt.SvtDaqMapping.SvtDaqMappingCollection;
7   import org.hps.conditions.svt.SvtT0Shift.SvtT0ShiftCollection;
8   
9   /**
10   * This class contains all test run SVT conditions data by readout channel. {@link SvtChannel} objects from the SVT
11   * channel map can be used to lookup the conditions for individual channels.
12   */
13  public final class SvtConditions extends AbstractSvtConditions {
14  
15      /**
16       * Get the {@link SvtChannelCollection} for this set of conditions.
17       *
18       * @return The SVT channel map.
19       */
20      @Override
21      public SvtChannelCollection getChannelMap() {
22          return (SvtChannelCollection) this.channelMap;
23      }
24  
25      /**
26       * Get the {@link SvtDaqMappingCollection} associated with these conditions.
27       *
28       * @return The SVT DAQ map.
29       */
30      @Override
31      public SvtDaqMappingCollection getDaqMap() {
32          return (SvtDaqMappingCollection) this.daqMap;
33      }
34  
35      /**
36       * Get the {@link SvtT0ShiftCollection} associated with these conditions.
37       *
38       * @return The {@link SvtT0ShiftCollection}
39       */
40      @Override
41      public SvtT0ShiftCollection getT0Shifts() {
42          return (SvtT0ShiftCollection) this.t0Shifts;
43      }
44  
45      /**
46       * Convert this object to a human readable string. This method prints a formatted table of channel data
47       * independently of how its member objects implement their string conversion method. For now, it does not print the
48       * time shifts by sensor as all other information is by channel.
49       *
50       * @return this object converted to a string, without the DAQ map
51       */
52      // FIXME: Make this look more human readable. At the moment, reading this requires a huge terminal window.
53      @Override
54      public String toString() {
55          final StringBuffer buff = new StringBuffer();
56  
57          buff.append('\n');
58          buff.append("Printing SVTConditions ...");
59          buff.append('\n');
60          buff.append('\n');
61  
62          // Table header:
63          buff.append("Channel ID");
64          buff.append("     ");
65          buff.append("FEB ID");
66          buff.append("  ");
67          buff.append("FEB Hybrid ID");
68          buff.append("   ");
69          buff.append("Channel");
70          buff.append("  ");
71          buff.append("Pedestal sample 0");
72          buff.append("     ");
73          buff.append("Pedestal sample 1");
74          buff.append("     ");
75          buff.append("Pedestal sample 2");
76          buff.append("     ");
77          buff.append("Pedestal sample 3");
78          buff.append("     ");
79          buff.append("Pedestal sample 4");
80          buff.append("     ");
81          buff.append("Pedestal sample 5");
82          buff.append("     ");
83          buff.append("Noise sample 0");
84          buff.append("     ");
85          buff.append("Noise sample 1");
86          buff.append("     ");
87          buff.append("Noise sample 2");
88          buff.append("     ");
89          buff.append("Noise sample 3");
90          buff.append("     ");
91          buff.append("Noise sample 4");
92          buff.append("     ");
93          buff.append("Noise sample 5");
94          buff.append("     ");
95          buff.append("Gain");
96          buff.append("   ");
97          buff.append("Offset");
98          buff.append("    ");
99          buff.append("Amplitude");
100         buff.append("  ");
101         buff.append("t0");
102         buff.append("       ");
103         buff.append("tp");
104         buff.append("    ");
105         buff.append("Bad Channels");
106         buff.append('\n');
107         for (int i = 0; i < 115; i++) {
108             buff.append("-");
109         }
110         buff.append('\n');
111         // Loop over channels.
112         for (final SvtChannel channel : this.getChannelMap()) {
113 
114             // Get the conditions for the channel.
115             final ChannelConstants constants = this.getChannelConstants(channel);
116             final SvtGain gain = constants.getGain();
117             final SvtShapeFitParameters shapeFit = constants.getShapeFitParameters();
118             final SvtCalibration calibration = constants.getCalibration();
119 
120             // Channel data.
121             buff.append(String.format("%-6d %-5d %-8d %-8d ", channel.getChannelID(), channel.getFebID(),
122                     channel.getFebHybridID(), channel.getChannel()));
123 
124             // Calibration.
125             for (int sample = 0; sample < MAX_NUMBER_OF_SAMPLES; sample++) {
126                 buff.append(calibration.getPedestal(sample));
127                 buff.append("      ");
128             }
129             for (int sample = 0; sample < MAX_NUMBER_OF_SAMPLES; sample++) {
130                 buff.append(calibration.getNoise(sample));
131                 buff.append("      ");
132             }
133 
134             // Gain.
135             buff.append(String.format("%-6.4f %-9.4f ", gain.getGain(), gain.getOffset()));
136 
137             // Pulse shape.
138             buff.append(String.format("%-10.4f %-8.4f %-8.4f", shapeFit.getAmplitude(), shapeFit.getT0(),
139                     shapeFit.getTp()));
140 
141             // Bad channel.
142             buff.append(constants.isBadChannel());
143 
144             buff.append('\n');
145         }
146 
147         buff.append('\n');
148         buff.append("Printing SVT DAQ Map...");
149         buff.append('\n');
150         buff.append('\n');
151         buff.append(this.getDaqMap());
152 
153         return buff.toString();
154     }
155 }
