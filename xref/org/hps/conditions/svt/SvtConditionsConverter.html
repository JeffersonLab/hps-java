1   package org.hps.conditions.svt;
2   
3   import org.hps.conditions.database.DatabaseConditionsManager;
4   import org.hps.conditions.svt.SvtChannel.SvtChannelCollection;
5   import org.hps.conditions.svt.SvtDaqMapping.SvtDaqMappingCollection;
6   import org.hps.conditions.svt.SvtT0Shift.SvtT0ShiftCollection;
7   import org.lcsim.conditions.ConditionsManager;
8   
9   /**
10   * This class creates an {@link SvtConditions} object from the database, based on the current run number known by the
11   * conditions manager.
12   */
13  public final class SvtConditionsConverter extends AbstractSvtConditionsConverter<SvtConditions> {
14  
15      /**
16       * Default constructor.
17       */
18      public SvtConditionsConverter() {
19          this.conditions = new SvtConditions();
20      }
21  
22      /**
23       * Create and return an {@link SvtConditions} object.
24       *
25       * @param manager the current conditions manager
26       * @param name the conditions key, which is ignored for now
27       */
28      @Override
29      public SvtConditions getData(final ConditionsManager manager, final String name) {
30  
31          final DatabaseConditionsManager dbConditionsManager = (DatabaseConditionsManager) manager;
32  
33          // Get the channel map from the conditions database
34          final SvtChannelCollection channels = dbConditionsManager.getCachedConditions(SvtChannelCollection.class,
35                  "svt_channels").getCachedData();
36  
37          // Create the SVT conditions object to use to encapsulate SVT condition
38          // collections
39          this.conditions.setChannelMap(channels);
40  
41          // Get the DAQ map from the conditions database
42          final SvtDaqMappingCollection daqMap = dbConditionsManager.getCachedConditions(SvtDaqMappingCollection.class,
43                  "svt_daq_map").getCachedData();
44          this.conditions.setDaqMap(daqMap);
45  
46          // Get the collection of T0 shifts from the conditions database
47          final SvtT0ShiftCollection t0Shifts = dbConditionsManager.getCachedConditions(SvtT0ShiftCollection.class,
48                  "svt_t0_shifts").getCachedData();
49          this.conditions.setT0Shifts(t0Shifts);
50  
51          this.conditions = super.getData(manager, name);
52  
53          return this.conditions;
54      }
55  
56      /**
57       * Get the type handled by this converter.
58       *
59       * @return The type handled by this converter.
60       */
61      @Override
62      public Class<SvtConditions> getType() {
63          return SvtConditions.class;
64      }
65  }
