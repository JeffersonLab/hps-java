70      private static void setupPlots(boolean show) {
71          IDataPointSetFactory dpsf = aida.analysisFactory().createDataPointSetFactory(aida.tree());
72          dpsRuns = dpsf.create("dpsRuns", "Run intervals", 2);
73          dpsBiasRuns = dpsf.create("dpsBiasRuns", "Bias ON intervals associated with runs", 2);
74          dpsPositionRuns = dpsf.create("dpsPositionRuns", "Position stable intervals associated with runs", 2);
75          IPlotter plotter = aida.analysisFactory().createPlotterFactory().create("Bias run ranges");
76          IPlotterStyle plotterStyle = aida.analysisFactory().createPlotterFactory().createPlotterStyle();
77          plotterStyle.xAxisStyle().setParameter("type", "date");
78          plotter.createRegions(1, 4);
79          plotter.region(0).plot(dpsRuns, plotterStyle);
80          plotter.region(1).plot(dpsBiasRuns, plotterStyle);
81          plotter.region(2).plot(dpsPositionRuns, plotterStyle);
82          plotter.region(3).plot(dpsRuns, plotterStyle);
83          plotter.region(3).plot(dpsBiasRuns, plotterStyle, "mode=overlay");
84          plotter.region(3).plot(dpsPositionRuns, plotterStyle, "mode=overlay");
85          if (show) {
86              plotter.show();
87          }

91      private static IDataPoint addPoint(IDataPointSet dps, long mstime, double val) {
92          IDataPoint dp = dps.addPoint();
93          dp.coordinate(0).setValue(mstime / 1000.);
94          dp.coordinate(1).setValue(val);

104     /**
105      * Check validity of @link RunData
106      *
107      * @param data the @link RunData to check
108      * @return <code>true</code> if valid, <code>false</code> otherwise.
109      */
110     private static boolean isValid(RunData data) {
111         if (data.getStartDate() == null || data.getEndDate() == null || data.getStartDate().before((new GregorianCalendar(1999, 1, 1)).getTime())) {
112             LOGGER.fine("This run data is not valid: " + data.toString());
113             return false;
114         }
115         if (data.getStartDate().after(data.getEndDate())) {
116             throw new RuntimeException("start date is after end date?!" + data.toString());
117         }
118         return true;
119     }

122     public static RunMap getRunMapFromSpreadSheet(String path) {
123         // Load in CSV records from the exported run spreadsheet.
124         LOGGER.info(path);
125         final RunSpreadsheet runSheet = new RunSpreadsheet(new File(path));
126 
127         // Find the run ranges that have the same fields values.
128         final List<RunRange> ranges = RunRange.findRunRanges(runSheet, FIELDS);
129         LOGGER.info("Found " + ranges.size() + " ranges.");
130         for (RunRange range : ranges) {
131             LOGGER.fine(range.toString());
132         }
133         // find the run records (has converted dates and stuff) for these ranges
134         RunMap runmap = runSheet.getRunMap(ranges);
135         LOGGER.info("Found " + runmap.size() + " runs in the run map.");
136         return runmap;
137     }

139     public static List<RunData> getRunListFromSpreadSheet(String path) {
140         // Load in CSV records from the exported run spreadsheet.
141         List<RunData> runList = new ArrayList<RunData>();
142 
143         // find the run records (has converted dates and stuff) for these ranges
144         RunMap runmap = getRunMapFromSpreadSheet(path);
145 
146         List<Integer> runNums = new ArrayList<Integer>(runmap.keySet());
147         Collections.sort(runNums);
148         for (Integer runNum : runNums) {
149             RunData data = runmap.get(runNum);
150             if (isValid(data)) {
151                 runList.add(data);
152             }
153         }
154         return runList;
155     }

163     public static void main(String[] args) {
164 
165         Options options = new Options();
166         options.addOption(new Option("c", true, "CSV run file"));
167         options.addOption(new Option("m", true, "MYA dump file for bias"));
168         options.addOption(new Option("p", true, "MYA dump file for motor positions"));
169         options.addOption(new Option("t", false, "use run table format (from crawler) for bias"));
170         options.addOption(new Option("d", false, "discard first line of MYA data (for myaData output)"));
171         options.addOption(new Option("g", false, "Actually load stuff into DB"));
172 //        options.addOption(new Option("b", true, "beam current file"));
173         options.addOption(new Option("s", false, "Show plots"));
174 
175         final CommandLineParser parser = new PosixParser();
176         CommandLine cl = null;
177         try {
178             cl = parser.parse(options, args);
179         } catch (ParseException e) {
180             throw new RuntimeException("Cannot parse.", e);
181         }
182 
183         if (!cl.hasOption("c") || (!cl.hasOption("m") && !cl.hasOption("p"))) {
184             printUsage(options);
185             return;
186         }
187 
188         // Setup plots
189         setupPlots(cl.hasOption("s"));
190 
191         // Load in CSV records from the exported run spreadsheet.
192         List<RunData> runList;
193         if (cl.hasOption("t")) {
194             runList = SvtBiasMyaDataReader.readRunTable(new File(cl.getOptionValue("c")));
195         } else {
196             runList = getRunListFromSpreadSheet(cl.getOptionValue("c"));
197         }
198 
199         List<SvtBiasRunRange> biasRunRanges = null;
200         List<SvtPositionRunRange> positionRunRanges = null;
201         // Load MYA dump
202         if (cl.hasOption("m")) {
203             List<SvtBiasMyaRange> biasRanges = SvtBiasMyaDataReader.readMyaData(new File(cl.getOptionValue("m")), 178.0, 2000, cl.hasOption("d"));
204             LOGGER.info("Got " + biasRanges.size() + " bias ranges");
205             biasRunRanges = SvtBiasMyaDataReader.findOverlappingRanges(runList, biasRanges);
206         }
207 
208         if (cl.hasOption("p")) {
209             List<SvtPositionMyaRange> positionRanges = SvtMotorMyaDataReader.readMyaData(new File(cl.getOptionValue("p")), 200, 10000);
210             LOGGER.info("Got " + positionRanges.size() + " position ranges");
211             positionRunRanges = SvtMotorMyaDataReader.findOverlappingRanges(runList, positionRanges);
212         }
213 
214         // Combine them to run ranges when bias was on        
215         // each run may have multiple bias ranges
216         // fill graphs
217         if (cl.hasOption("s")) {
218             if (cl.hasOption("m")) {
219                 for (SvtBiasRunRange r : biasRunRanges) {
220                     LOGGER.info(r.toString());
221                     if (r.getRun().getRun() > 5600) {//9999999999999.0) {
222                         //if(dpsRuns.size()/4.0<500) {//9999999999999.0) {
223                         addPoint(dpsRuns, r.getRun().getStartDate().getTime(), 0.0);
224                         addPoint(dpsRuns, r.getRun().getStartDate().getTime(), 1.0);
225                         addPoint(dpsRuns, r.getRun().getEndDate().getTime(), 1.0);
226                         addPoint(dpsRuns, r.getRun().getEndDate().getTime(), 0.0);
227 
228                         for (SvtBiasMyaRange br : r.getRanges()) {
229                             addPoint(dpsBiasRuns, br.getStartDate().getTime(), 0.0);
230                             addPoint(dpsBiasRuns, br.getStartDate().getTime(), 0.3);
231                             addPoint(dpsBiasRuns, br.getEndDate().getTime(), 0.3);
232                             addPoint(dpsBiasRuns, br.getEndDate().getTime(), 0.0);
233                         }
234                     }
235                 }
236             }
237             if (cl.hasOption("p")) {
238                 for (SvtPositionRunRange r : positionRunRanges) {
239                     LOGGER.info(r.toString());
240                     if (r.getRun().getRun() > 5600) {//9999999999999.0) {
241                         //if(dpsRuns.size()/4.0<500) {//9999999999999.0) {
242                         for (SvtPositionMyaRange br : r.getRanges()) {
243                             addPoint(dpsPositionRuns, br.getStartDate().getTime(), 0.0);
244                             addPoint(dpsPositionRuns, br.getStartDate().getTime(), 0.5 + 100 * Math.max(br.getTop(), br.getBottom()));
245                             addPoint(dpsPositionRuns, br.getEndDate().getTime(), 0.5 + 100 * Math.max(br.getTop(), br.getBottom()));
246                             addPoint(dpsPositionRuns, br.getEndDate().getTime(), 0.0);
247                         }
248                     }
249                 }
250             }
251         }
252 
253 //        if (cl.hasOption("b") && cl.hasOption("m") && cl.hasOption("p")) {
254 //            readBeamData(new File(cl.getOptionValue("b")), runList, positionRunRanges, biasRunRanges);
255 //        }
256 
257         // load to DB
258         if (cl.hasOption("g")) {
259             if (cl.hasOption("m")) {
260                 loadBiasesToConditionsDB(biasRunRanges);
261             }
262             if (cl.hasOption("p")) {
263                 loadPositionsToConditionsDB(positionRunRanges);
264             }
265         }

274     private static void loadBiasesToConditionsDB(List<SvtBiasRunRange> ranges) {
275         LOGGER.info("Load to DB...");
276 
277         // Create a new collection for each run
278         List<Integer> runsadded = new ArrayList<Integer>();
279 
280         for (SvtBiasRunRange range : ranges) {
281             LOGGER.info("Loading " + range.toString());
282             RunData rundata = range.getRun();
283             if (runsadded.contains(rundata.getRun())) {
284                 LOGGER.warning("Run " + Integer.toString(rundata.getRun()) + " was already added?");
285                 throw new RuntimeException("Run " + Integer.toString(rundata.getRun()) + " was already added?");
286             }
287             runsadded.add(rundata.getRun());
288 
289             if (range.getRanges().isEmpty()) {
290                 LOGGER.info("No bias range for run " + range.getRun().getRun());
291                 continue;
292             }
293 
294             // create a collection
295             SvtBiasConstantCollection collection = new SvtBiasConstantCollection();
296 
297             // Set the table meta data
298             collection.setTableMetaData(MANAGER.findTableMetaData("svt_bias_constants"));
299             collection.setConnection(MANAGER.getConnection());
300 
301             int collectionId = -1;
302             try {
303                 collectionId = MANAGER.getCollectionId(collection, "run ranges for SVT HV bias ON");
304             } catch (SQLException e1) {
305                 throw new RuntimeException(e1);
306             }
307 
308             collection.setCollectionId(collectionId);
309 
310             final ConditionsRecord condition = new ConditionsRecord();
311             condition.setFieldValue("run_start", rundata.getRun());
312             condition.setFieldValue("run_end", rundata.getRun());
313             condition.setFieldValue("name", "svt_bias_constants");
314             condition.setFieldValue("table_name", "svt_bias_constants");
315             condition.setFieldValue("notes", "constants from mya");
316             condition.setFieldValue("created", new Date());
317             condition.setFieldValue("created_by", System.getProperty("user.name"));
318             condition.setFieldValue("collection_id", collectionId);
319             condition.setTableMetaData(MANAGER.findTableMetaData("conditions"));
320             condition.setConnection(MANAGER.getConnection());
321             LOGGER.info(condition.toString());
322 
323             try {
324 
325                 for (SvtBiasMyaRange biasRange : range.getRanges()) {
326                     // create a constant and add to the collection
327                     final SvtBiasConstant constant = new SvtBiasConstant();
328                     constant.setFieldValue("start", biasRange.getStartDate().getTime());
329                     constant.setFieldValue("end", biasRange.getEndDate().getTime());
330                     constant.setFieldValue("value", biasRange.getValue());
331                     collection.add(constant);
332                 }
333 
334                 // Insert collection data.
335                 collection.insert();
336 
337                 // Insert conditions record.
338                 condition.insert();
339 
340             } catch (Exception e) {
341                 throw new RuntimeException(e);
342             }
343         }
344     }
