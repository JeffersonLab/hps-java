17  public class OpeningAngleLoader {
18  
19      private static final String BOT_FILE = "mya_svt_bot.txt";
20      private static final SimpleDateFormat DATE_FORMAT = new SimpleDateFormat("MMMM dd YYYY HH:mm z");
21      private static final String OUT_FILE = "svt_opening_angles.txt";
22      private static final String RUN_FILE = "runs.csv";
23  
24      private static final String TOP_FILE = "mya_svt_top.txt";
25  
26      /**
27       * Check if the run record looks good.
28       *
29       * @param data
30       * @return whether to accept the run or not
31       */
32      private static boolean acceptRun(final RunData data) {
33          return !data.getRecord().get("to_tape").equals("JUNK")
34                  && data.getRecord().get("trigger_config").trim().length() > 0
35                  && !data.getRecord().get("trigger_config").contains("cosmic") && data.getStartDate() != null;
36      }
37  
38      private static MotorPositionInterval findInterval(final List<MotorPositionInterval> intervals, final Date date) {
39          MotorPositionInterval interval = null;
40          final Iterator<MotorPositionInterval> it = intervals.listIterator();
41          while ((interval = it.next()) != null) {
42  
43              // Start and end dates in the interval.
44              final Date startDate = interval.getStartDate();
45              final Date endDate = interval.getEndDate();
46  
47              // Check if given date is within the interval.
48              if ((startDate.compareTo(date) == -1 || startDate.compareTo(date) == 0) && endDate.compareTo(date) == 1
49                      || endDate.compareTo(date) == 0) {
50                  break;
51              }
52  
53              // Didn't find it.
54              if (!it.hasNext()) {
55                  interval = null;
56                  break;
57              }
58          }
59          return interval;
60      }
61  
62      private static List<MotorPositionInterval> getMotorPositionIntervals(final String path, final Side side)
63              throws Exception {
64          final MotorPositionLoader loader = new MotorPositionLoader();
65          loader.setSide(side);
66          loader.load(path);
67          return loader.findIntervals();
68      }
69  
70      private static RunMap loadRunMap(final String path) {
71          final File runFile = new File(path);
72          final RunSpreadsheet runSpreadsheet = new RunSpreadsheet(runFile);
73          return runSpreadsheet.getRunMap();
74      }
75  
76      public static void main(final String[] args) throws Exception {
77  
78          // Load top and bottom intervals from MYA dump.
79          final List<MotorPositionInterval> topIntervals = getMotorPositionIntervals(TOP_FILE, Side.TOP);
80          final List<MotorPositionInterval> botIntervals = getMotorPositionIntervals(BOT_FILE, Side.BOT);
81  
82          // Load run map from spreadsheet.
83          final RunMap runMap = loadRunMap(RUN_FILE);
84  
85          // Write out run data combined with SVT opening angle intervals.
86          final PrintStream ps = new PrintStream(new FileOutputStream(OUT_FILE));
87          for (final RunData data : runMap.values()) {
88              if (acceptRun(data)) {
89                  final MotorPositionInterval topInterval = findInterval(topIntervals, data.getStartDate());
90                  final MotorPositionInterval botInterval = findInterval(botIntervals, data.getStartDate());
91                  printLine(ps, data, topInterval, botInterval);
92              }
93          }
94          ps.close();
95      }
96  
97      private static void printLine(final PrintStream ps, final RunData data, final MotorPositionInterval topInterval,
98              final MotorPositionInterval botInterval) {
99          ps.print("run: " + data.getRun() + " @ [" + data.getRecord().get("svt_y_position") + "], ");
100         if (topInterval != null) {
101             ps.print("start(top): " + DATE_FORMAT.format(topInterval.getStartDate()) + ", ");
102             ps.print("end(top): " + DATE_FORMAT.format(topInterval.getEndDate()) + ", ");
103         } else if (botInterval != null) {
104             ps.print("start(bot): " + DATE_FORMAT.format(botInterval.getStartDate()) + ", ");
105             ps.print("end(bot): " + DATE_FORMAT.format(botInterval.getEndDate()) + ", ");
106         } else {
107             ps.print("start: NONE, end: NONE, ");
108         }
109         ps.print("topAngle: ");
110         if (topInterval != null) {
111             ps.print(topInterval.getAngle());
112         } else {
113             ps.print("NONE");
114         }
115         ps.print(", botAngle: ");
116         if (botInterval != null) {
117             ps.print(botInterval.getAngle());
118         } else {
119             ps.print("NONE");
120         }
121         ps.print(", topYStage: ");
122         if (topInterval != null) {
123             ps.print(topInterval.getYStage());
124         } else {
125             ps.print("NONE");
126         }
127         ps.print(", botYStage: ");
128         if (botInterval != null) {
129             ps.print(botInterval.getYStage());
130         } else {
131             ps.print("NONE");
132         }
133         ps.println();
134         ps.flush();
135     }
