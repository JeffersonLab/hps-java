15  /**
16   * Abstract class providing some of the common methods used in creating SVT conditions objects from the database.
17   *
18   * @param <T> SVT conditions object type
19   */
20  public abstract class AbstractSvtConditionsConverter<T extends AbstractSvtConditions> implements ConditionsConverter<T> {
21  
22      /**
23       * Initialize the logger.
24       */
25      private static Logger LOGGER = Logger.getLogger(AbstractSvtConditionsConverter.class.getPackage().getName());
26  
27      /**
28       * The combined detector conditions object.
29       */
30      protected T conditions;
31  
32      /**
33       * Create and return the SVT conditions object.
34       *
35       * @param manager the current conditions manager
36       * @param name the conditions key, which is ignored for now
37       * @return the SVT conditions object
38       */
39      @Override
40      public T getData(final ConditionsManager manager, final String name) {
41  
42          final DatabaseConditionsManager dbConditionsManager = (DatabaseConditionsManager) manager;
43  
44          // Get the SVT calibrations (baseline, noise) from the conditions database
45          final SvtCalibrationCollection calibrations = this.getSvtCalibrationCollection(dbConditionsManager);
46          for (final SvtCalibration calibration : calibrations) {
47              final AbstractSvtChannel channel = this.conditions.getChannelMap().findChannel(calibration.getChannelID());
48              this.conditions.getChannelConstants(channel).setCalibration(calibration);
49          }
50  
51          // Get the Channel pulse fit parameters from the conditions database
52          final SvtShapeFitParametersCollection shapeFitParametersCollection = this
53                  .getSvtShapeFitParametersCollection(dbConditionsManager);
54          for (final SvtShapeFitParameters shapeFitParameters : shapeFitParametersCollection) {
55              final AbstractSvtChannel channel = this.conditions.getChannelMap().findChannel(
56                      shapeFitParameters.getChannelID());
57              this.conditions.getChannelConstants(channel).setShapeFitParameters(shapeFitParameters);
58          }
59  
60          // Get the bad channels from the conditions database.
61          // If there aren't any bad channels, notify the user and move on.
62          try {
63              final ConditionsSeries<SvtBadChannel, SvtBadChannelCollection> badChannelSeries = this
64                      .getSvtBadChannelSeries(dbConditionsManager);
65              for (final ConditionsObjectCollection<SvtBadChannel> badChannelCollection : badChannelSeries) {
66                  for (final SvtBadChannel badChannel : badChannelCollection) {
67                      final AbstractSvtChannel channel = this.conditions.getChannelMap().findChannel(
68                              badChannel.getChannelId());
69                      this.conditions.getChannelConstants(channel).setBadChannel(true);
70                  }
71              }
72          } catch (final RuntimeException e) {
73              LOGGER.warning("A set of SVT bad channels was not found.");
74          }
75  
76          // Get the gains and offsets from the conditions database
77          final SvtGainCollection channelGains = this.getSvtGainCollection(dbConditionsManager);
78          for (final SvtGain channelGain : channelGains) {
79              final int channelId = channelGain.getChannelID();
80              final AbstractSvtChannel channel = this.conditions.getChannelMap().findChannel(channelId);
81              this.conditions.getChannelConstants(channel).setGain(channelGain);
82          }
83  
84          return this.conditions;
85      }
86  
87      /**
88       * Get the default {@link SvtBadChannelCollection} collection from the manager.
89       *
90       * @param manager the current conditions manager
91       * @return the default {@link SvtBadChannelCollection}
92       */
93      protected SvtBadChannelCollection getSvtBadChannelCollection(final DatabaseConditionsManager manager) {
94          return manager.getCachedConditions(SvtBadChannelCollection.class, "svt_bad_channels").getCachedData();
95      }
96  
97      /**
98       * Get the default series of {@link SvtBadChannelCollection} collections from the manager.
99       *
100      * @param manager the current conditions manager
101      * @return the default series of {@link SvtBadChannelCollection}
102      */
103     protected ConditionsSeries<SvtBadChannel, SvtBadChannelCollection> getSvtBadChannelSeries(
104             final DatabaseConditionsManager manager) {
105         return manager.getConditionsSeries(SvtBadChannelCollection.class, "svt_bad_channels");
106     }
107 
108     /**
109      * Get the default {@link SvtCalibrationCollection} collection from the manager.
110      *
111      * @param manager the current conditions manager
112      * @return the default {@link SvtCalibrationCollection}
113      */
114     protected SvtCalibrationCollection getSvtCalibrationCollection(final DatabaseConditionsManager manager) {
115         return manager.getCachedConditions(SvtCalibrationCollection.class, "svt_calibrations").getCachedData();
116     }
117 
118     /**
119      * Get the default {@link SvtGainCollection} collection from the manager.
120      *
121      * @param manager the current conditions manager
122      * @return the default {@link SvtGainCollection}
123      */
124     protected SvtGainCollection getSvtGainCollection(final DatabaseConditionsManager manager) {
125         return manager.getCachedConditions(SvtGainCollection.class, "svt_gains").getCachedData();
126     }
127 
128     /**
129      * Get the default {@link SvtShapeFitParametersCollection} collection from the manager.
130      *
131      * @param manager the current conditions manager
132      * @return the default {@link SvtShapeFitParametersCollection}
133      */
134     protected SvtShapeFitParametersCollection getSvtShapeFitParametersCollection(final DatabaseConditionsManager manager) {
135         return manager.getCachedConditions(SvtShapeFitParametersCollection.class, "svt_shape_fit_parameters")
136                 .getCachedData();
137     }
