94      /**
95       * Read the data from a CSV export of the run spreadsheet.
96       *
97       * @param file the CSV file
98       */
99      public RunSpreadsheet(final File file) {
100         if (file == null) {
101             throw new IllegalArgumentException("The file argument is null.");
102         }
103         this.file = file;
104         try {
105             this.fromCsv(this.file);
106         } catch (final Exception e) {
107             throw new RuntimeException("Failed to parse run spreadsheet.", e);
108         }
109     }

111     /**
112      * Find the record for a run in the list of records.
113      *
114      * @param run the run number
115      * @return the <code>CSVRecord</code> or <code>null</code> if not found
116      */
117     public CSVRecord findRun(final int run) {
118         for (final CSVRecord record : records) {
119             try {
120                 if (run == Integer.parseInt(record.get("run"))) {
121                     return record;
122                 }
123             } catch (final NumberFormatException e) {
124                 // System.out.println("problem reading run" + run);
125                 // e.printStackTrace();
126             }
127         }
128         return null;
129     }

131     /**
132      * Read all the records from the CSV file.
133      *
134      * @param file the CSV file
135      * @throws FileNotFoundException if file does not exist
136      * @throws IOException if there is an IO error
137      */
138     private void fromCsv(final File file) throws FileNotFoundException, IOException {
139 
140         final FileReader reader = new FileReader(file);
141         final CSVFormat format = CSVFormat.DEFAULT.withHeader(HEADERS);
142 
143         final CSVParser parser = new CSVParser(reader, format);
144 
145         records = parser.getRecords();
146 
147         // Remove first three rows of headers.
148         records.remove(0);
149         records.remove(0);
150         records.remove(0);
151 
152         parser.close();
153     }

195     private static Date parseEndDate(CSVRecord record) throws ParseException {
196         // DATE_FORMAT.setTimeZone(TIME_ZONE);

204     public static class RunData {
205 
206         private int run;
207         private Date startDate;
208         private Date endDate;
209         private CSVRecord record;
210 
211         RunData(CSVRecord record) throws NumberFormatException {
212             this.record = record;
213             run = parseRunNumber(this.record);
214             try {
215                 startDate = RunSpreadsheet.parseStartDate(this.record);
216             } catch (ParseException e) {
217             }
218             try {
219                 endDate = RunSpreadsheet.parseEndDate(this.record);
220             } catch (ParseException e) {
221             }
222         }
223 
224         public RunData(Date startDate, Date endDate, int run) {
225             this.startDate = startDate;
226             this.endDate = endDate;
227             this.run = run;
228             record = null;
229         }
230 
231         public int getRun() {
232             return run;
233         }
234 
235         public Date getStartDate() {
236             return startDate;
237         }
238 
239         public Date getEndDate() {
240             return endDate;
241         }
242 
243         public String toString() {
244             return "RunData { run: " + run + ", startDate: " + startDate + ", endDate: " + endDate + " }";
245         }

283     public RunMap getRunMap(List<RunRange> ranges) {
284         List<CSVRecord> records = new ArrayList<CSVRecord>();
285         for (RunRange range : ranges) {
286             // System.out.println(range.toString());
287             if (range.getColumnNames().contains("run")) {
288                 if (!range.getValue("run").isEmpty()) {
289                     CSVRecord record = findRun(Integer.parseInt(range.getValue("run")));
290                     if (record != null) {
291                         records.add(record);
292                     } else {
293                         throw new RuntimeException("this RunRange object was not found. This shouldn't happen. "
294                                 + range.toString());
295                     }
296                 } else {
297                     throw new RuntimeException("this RunRange object has an empty run value ");
298                 }
299             } else {
300                 throw new RuntimeException("this RunRange object has no run column? " + range.toString());
301             }
302         }
303         return new RunMap(records);
304     }
