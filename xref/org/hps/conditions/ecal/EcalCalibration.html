1   package org.hps.conditions.ecal;
2   
3   import java.util.Comparator;
4   
5   import org.hps.conditions.api.BaseConditionsObject;
6   import org.hps.conditions.api.BaseConditionsObjectCollection;
7   import org.hps.conditions.api.ConditionsObjectCollection;
8   import org.hps.conditions.database.Field;
9   import org.hps.conditions.database.Table;
10  
11  /**
12   * This class represents ECal pedestal and noise values from the conditions database. The pedestal and noise are in
13   * units of ADC counts. They are the mean and the standard deviation of the digitized pre-amp output.
14   */
15  @Table(names = {"ecal_calibrations", "test_run_ecal_calibrations"})
16  public final class EcalCalibration extends BaseConditionsObject {
17  
18      /**
19       * The collection implementation for the object class.
20       */
21      public static class EcalCalibrationCollection extends BaseConditionsObjectCollection<EcalCalibration> {
22  
23          /**
24           * Comparison using channel ID.
25           */
26          class ChannelIdComparator implements Comparator<EcalCalibration> {
27              /**
28               * Compare two ECAL calibration objects.
29               *
30               * @param o1 the first object
31               * @param o2 the second object
32               * @return -1, 0, 1 if first channel ID is less than, equal to, or greater than the second
33               */
34              @Override
35              public int compare(final EcalCalibration o1, final EcalCalibration o2) {
36                  if (o1.getChannelId() < o2.getChannelId()) {
37                      return -1;
38                  } else if (o1.getChannelId() > o2.getChannelId()) {
39                      return 1;
40                  } else {
41                      return 0;
42                  }
43              }
44          }
45  
46          /**
47           * Sort and return the collection but do no modify in place.
48           *
49           * @return the sorted collection
50           */
51          public ConditionsObjectCollection<EcalCalibration> sorted() {
52              return this.sorted(new ChannelIdComparator());
53          }
54      }
55  
56      /**
57       * No argument constructor.
58       */
59      public EcalCalibration() {
60      }
61  
62      /**
63       * Full qualified constructor.
64       *
65       * @param channelId the channel ID
66       * @param pedestal the pedestal measurement (ADC counts)
67       * @param noise the noise measured as RMS
68       */
69      public EcalCalibration(final int channelId, final double pedestal, final double noise) {
70          this.setFieldValue("ecal_channel_id", channelId);
71          this.setFieldValue("pedestal", pedestal);
72          this.setFieldValue("noise", noise);
73      }
74  
75      /**
76       * Get the ECAL channel ID.
77       *
78       * @return the ECAL channel ID
79       */
80      @Field(names = {"ecal_channel_id"})
81      public Integer getChannelId() {
82          return this.getFieldValue("ecal_channel_id");
83      }
84  
85      /**
86       * Get the noise value in units of ADC counts, which is the standard deviation of the digitized preamplifier output.
87       *
88       * @return the noise value
89       */
90      @Field(names = {"noise"})
91      public Double getNoise() {
92          return this.getFieldValue("noise");
93      }
94  
95      /**
96       * Get the pedestal value in units of ADC counts, which is the mean of the digitized preamplifier output.
97       *
98       * @return the gain value
99       */
100     @Field(names = {"pedestal"})
101     public Double getPedestal() {
102         return this.getFieldValue("pedestal");
103     }
104 }
