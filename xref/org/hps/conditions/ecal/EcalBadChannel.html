5   import org.hps.conditions.api.BaseConditionsObject;
6   import org.hps.conditions.api.BaseConditionsObjectCollection;

11  /**
12   * This class represents an ECAL channel that is considered "bad" which means it should not be used in reconstruction.
13   */
14  @Table(names = {"ecal_bad_channels", "test_run_ecal_bad_channels"})
15  public final class EcalBadChannel extends BaseConditionsObject {
16  
17      /**
18       * The collection class for this object.
19       */
20      @SuppressWarnings("serial")
21      public static class EcalBadChannelCollection extends BaseConditionsObjectCollection<EcalBadChannel> {
22  
23          /**
24           * Comparison class for ECAL bad channels, which uses channel ID.
25           */
26          class ChannelIdComparator implements Comparator<EcalBadChannel> {
27              /**
28               * Compare two ECAL bad channel objects.
29               *
30               * @param o1 the first object
31               * @param o2 the second object
32               * @return -1, 0, 1 if first channel ID is less than, equal to, or greater than the second
33               */
34              @Override
35              public int compare(final EcalBadChannel o1, final EcalBadChannel o2) {
36                  if (o1.getChannelId() < o2.getChannelId()) {
37                      return -1;
38                  } else if (o1.getChannelId() > o2.getChannelId()) {
39                      return 1;
40                  } else {
41                      return 0;
42                  }
43              }
44          }
45  
46          /**
47           * Sort and return the collection without modifying in place.
48           *
49           * @return the sorted collection
50           */
51          public ConditionsObjectCollection<EcalBadChannel> sorted() {
52              return this.sorted(new ChannelIdComparator());
53          }
54      }
55  
56      /**
57       * Get the ECAL channel ID.
58       *
59       * @return the ECAL channel ID
60       */
61      @Field(names = {"ecal_channel_id"})
62      public Integer getChannelId() {
63          return this.getFieldValue("ecal_channel_id");
64      }
65  }
