5   import org.hps.conditions.api.BaseConditionsObject;

11  /**
12   * This class represents a time shift calibration value for an ECAL channel.
13   */
14  @Table(names = {"ecal_time_shifts", "test_run_ecal_time_shifts"})
15  public final class EcalTimeShift extends BaseConditionsObject {
16  
17      /**
18       * A collection of {@link EcalTimeShift} objects.
19       */
20      public static final class EcalTimeShiftCollection extends BaseConditionsObjectCollection<EcalTimeShift> {
21  
22          /**
23           * Compare two objects by their channel ID.
24           */
25          class ChannelIdComparator implements Comparator<EcalTimeShift> {
26              /**
27               * Compare two objects by channel ID.
28               *
29               * @param o1 the first object
30               * @param o2 the second object
31               * @return -1, 0 or 1 if first channel ID is less than, equal to, or greater than second
32               */
33              @Override
34              public int compare(final EcalTimeShift o1, final EcalTimeShift o2) {
35                  if (o1.getChannelId() < o2.getChannelId()) {
36                      return -1;
37                  } else if (o1.getChannelId() > o2.getChannelId()) {
38                      return 1;
39                  } else {
40                      return 0;
41                  }
42              }
43          }
44  
45          /**
46           * Sort and return a copy of the collection.
47           *
48           * @return The sorted copy of the collection.
49           */
50          public ConditionsObjectCollection<EcalTimeShift> sorted() {
51              return this.sorted(new ChannelIdComparator());
52          }
53      }
54  
55      /**
56       * Get the channel ID.
57       *
58       * @return the ECAL channel ID
59       */
60      @Field(names = {"ecal_channel_id"})
61      public Integer getChannelId() {
62          return this.getFieldValue("ecal_channel_id");
63      }
64  
65      /**
66       * Get the time shift in nanoseconds
67       *
68       * @return the time shift in nanoseconds
69       */
70      @Field(names = {"time_shift"})
71      public Double getTimeShift() {
72          return this.getFieldValue("time_shift");
73      }
74  }
