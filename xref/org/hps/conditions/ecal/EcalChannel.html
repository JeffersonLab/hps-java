26   * This class encapsulates all the information about a single ECal channel, corresponding to one physical crystal in the

35      /**
36       * The <code>ChannelId</code> is a unique number identifying the channel within its conditions collection. The
37       * channels in the database are given sequential channel IDs from 1 to N in semi-arbitrary order. The channel ID is
38       * generally the number used to connect other conditions objects such as {@link EcalGain} or {@link EcalCalibration}
39       * to the appropriate crystal in the calorimeter.
40       */
41      public static final class ChannelId extends AbstractIdentifier {
42  
43          /**
44           * The channel ID value.
45           */
46          private int id = -1;
47  
48          /**
49           * Create a channel ID.
50           *
51           * @param values the values (size 0 with single int value)
52           */
53          public ChannelId(final int[] values) {
54              this.id = values[0];
55          }
56  
57          /**
58           * Encode as long value (just returns the int value).
59           *
60           * @return the ID's value encoded as a <code>long</code>
61           */
62          @Override
63          public long encode() {
64              return this.id;
65          }
66  
67          /**
68           * Return <code>true</code> if ID is valid
69           *
70           * @return <code>true</code> if ID is valid
71           */
72          @Override
73          public boolean isValid() {
74              return this.id != -1;
75          }
76      }

78      /**
79       * The <code>DaqId</code> is the combination of crate, slot and channel that specify the channel's DAQ
80       * configuration.
81       */
82      public static final class DaqId extends AbstractIdentifier {
83  
84          /**
85           * The DAQ channel number.
86           */
87          private int channel = -1;
88  
89          /**
90           * The DAQ crate number.
91           */
92          private int crate = -1;
93  
94          /**
95           * The DAQ slot number.
96           */
97          private int slot = -1;
98  
99          /**
100          * Create a DAQ ID from an array of values.
101          *
102          * @param values The list of values (crate, slot, channel).
103          */
104         public DaqId(final int values[]) {
105             this.crate = values[0];
106             this.slot = values[1];
107             this.channel = values[2];
108         }
109 
110         /**
111          * Encode this ID into a long value.
112          *
113          * @return The encoded long value.
114          */
115         @Override
116         public long encode() {
117             // from ECAL readout sim code
118             return (long) this.crate << 32 | (long) this.slot << 16 | this.channel;
119         }
120 
121         /**
122          * Check if the values look valid.
123          *
124          * @return True if ID's values are valid.
125          */
126         @Override
127         public boolean isValid() {
128             return this.crate != -1 && this.slot != -1 && this.channel != -1;
129         }
130     }

132     /**
133      * A collection of {@link EcalChannel} objects.
134      */
135     public static class EcalChannelCollection extends BaseConditionsObjectCollection<EcalChannel> {
136 
137         /**
138          * Comparison of ECAL channel objects.
139          */
140         class ChannelIdComparator implements Comparator<EcalChannel> {
141             /**
142              * Compare two ECAL channel objects using their channel ID.
143              *
144              * @param c1 the first object
145              * @param c2 the second object
146              * @return -1, 0, or 1 if first channel is less than, equal to or greater than second
147              */
148             @Override
149             public int compare(final EcalChannel c1, final EcalChannel c2) {
150                 if (c1.getChannelId() < c2.getChannelId()) {
151                     return -1;
152                 } else if (c1.getChannelId() > c2.getChannelId()) {
153                     return 1;
154                 } else {
155                     return 0;
156                 }
157             }
158         }
159 
160         /**
161          * Map of {@link #ChannelId} to channel object.
162          */
163         private final Map<Long, EcalChannel> channelMap = new HashMap<Long, EcalChannel>();
164 
165         /**
166          * Map of {@link #DaqId} to channel object.
167          */
168         private final Map<Long, EcalChannel> daqMap = new HashMap<Long, EcalChannel>();
169 
170         /**
171          * Map of {@link #GeometryId} to channel object.
172          */
173         private final Map<Long, EcalChannel> geometryMap = new HashMap<Long, EcalChannel>();
174 
175         /**
176          * Add an <code>EcalChannel</code> to the collection and cache its ID information. The GeometryId must be
177          * created later as it requires access to the Detector API.
178          *
179          * @param channel the ECAL channel object
180          * @return <code>true</code> if object was added successfully
181          */
182         @Override
183         public boolean add(final EcalChannel channel) throws ConditionsObjectException {
184             super.add(channel);
185             final DaqId daqId = channel.createDaqId();
186             if (daqId.isValid()) {
187                 this.daqMap.put(daqId.encode(), channel);
188             }
189             final ChannelId channelId = channel.createChannelId();
190             if (channelId.isValid()) {
191                 this.channelMap.put(channelId.encode(), channel);
192             }
193             return true;
194         }
195 
196         /**
197          * Build the map of {@link #GeometryId} objects.
198          *
199          * @param helper the ID helper of the subdetector
200          * @param system the system ID of the subdetector
201          */
202         void buildGeometryMap(final IIdentifierHelper helper, final int system) {
203             for (final EcalChannel channel : this) {
204                 final GeometryId geometryId = channel.createGeometryId(helper, system);
205                 this.geometryMap.put(geometryId.encode(), channel);
206             }
207         }
208

279     /**
280      * Customized converter for this object.
281      */
282     public static final class EcalChannelConverter extends AbstractConditionsObjectConverter<EcalChannelCollection> {
283 
284         /**
285          * Create an {@link EcalChannel} collection.
286          *
287          * @param conditionsManager the conditions manager
288          * @param name the name of the conditions data table
289          * @return the collection of ECAL channel objects
290          */
291         @Override
292         public EcalChannelCollection getData(final ConditionsManager conditionsManager, final String name) {
293             final EcalChannelCollection collection = super.getData(conditionsManager, name);
294             DatabaseConditionsManager mgr = DatabaseConditionsManager.getInstance();
295             // FIXME: Ugly method call to get Ecal subdetector object!
296             final Subdetector ecal = 
297                     mgr.getCachedConditions(Detector.class, "compact.xml").getCachedData().getSubdetector("Ecal");
298             if (ecal != null) {
299                 if (ecal.getDetectorElement() != null) {
300                     collection.buildGeometryMap(ecal.getDetectorElement().getIdentifierHelper(), ecal.getSystemID());
301                 } else {
302                     // This can happen when not running with the detector-framework jar in the classpath.
303                     throw new IllegalStateException("The ECal subdetector's detector element is not setup.");
304                 }
305             } else {
306                 // FIXME: Probably this should be a fatal error.
307                 LOGGER.warning("ECal subdetector is not accessible so geometry map was not initialized.");
308             }            
309             return collection;
310         }

314          *
315          * @return the type this converter handles
316          */
317         @Override
318         public Class<EcalChannelCollection> getType() {
319             return EcalChannelCollection.class;
320         }

323     /**
324      * The <code>GeometryId</code> contains the x and y indices of the crystal in the LCSIM-based geometry
325      * representation.
326      */
327     public static final class GeometryId extends AbstractIdentifier {
328 
329         /**
330          * The helper for using identifiers.
331          */
332         private final IIdentifierHelper helper;
333 
334         /**
335          * The subdetector system ID.
336          */
337         private int system = -1;
338 
339         /**
340          * The crystal's X index.
341          */
342         private int x = Integer.MAX_VALUE;
343 
344         /**
345          * The crystal's Y index.
346          */
347         private int y = Integer.MAX_VALUE;
348 
349         /**
350          * Create a geometry ID.
351          *
352          * @param helper the ID helper
353          * @param values the list of values (order is system, x, y)
354          */
355         public GeometryId(final IIdentifierHelper helper, final int[] values) {
356             this.helper = helper;
357             this.system = values[0];
358             this.x = values[1];
359             this.y = values[2];
360         }
361 
362         /**
363          * Encode this ID as a long using the ID helper.
364          *
365          * @return The encoded long value.
366          */
367         @Override
368         public long encode() {
369             final IExpandedIdentifier expId = new ExpandedIdentifier(this.helper.getIdentifierDictionary()
370                     .getNumberOfFields());
371             expId.setValue(this.helper.getFieldIndex("system"), this.system);
372             expId.setValue(this.helper.getFieldIndex("ix"), this.x);
373             expId.setValue(this.helper.getFieldIndex("iy"), this.y);
374             return this.helper.pack(expId).getValue();
375         }
376 
377         /**
378          * Return <code>true</code> if ID is valid
379          *
380          * @return <code>true</code> if ID is valid
381          */
382         @Override
383         public boolean isValid() {
384             return this.system != -1 && this.x != Integer.MAX_VALUE && this.y != Integer.MAX_VALUE;
385         }
386     }

417     /**
418      * Implementation of equals.
419      *
420      * @param o the object to compare equality to
421      * @return <code>true</code> if objects are equal
422      */
423     @Override
424     public boolean equals(final Object o) {
425         if (o == null) {
426             return false;
427         }
428         if (!(o instanceof EcalChannel)) {
429             return false;
430         }
431         if (o == this) {
432             return true;
433         }
434         final EcalChannel c = (EcalChannel) o;
435         return c.getChannelId() == this.getChannelId() && c.getCrate() == this.getCrate()
436                 && c.getSlot() == this.getSlot() && c.getChannel() == this.getChannel() && c.getX() == this.getX()
437                 && c.getY() == this.getY();
438     }
