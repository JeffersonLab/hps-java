1   package org.hps.conditions.ecal;

3   import org.hps.conditions.database.DatabaseConditionsManager;
4   import org.hps.conditions.ecal.EcalChannel.GeometryId;
5   import org.lcsim.detector.identifier.IIdentifier;
6   import org.lcsim.detector.identifier.IIdentifierHelper;
7   import org.lcsim.detector.identifier.Identifier;
8   
9   /**
10   * This is a set of utility methods for the ECAL that use the database conditions system.
11   */
12  public final class EcalConditionsUtil {
13  
14      /**
15       * The combined ECAL conditions object.
16       */
17      private final EcalConditions conditions;
18  
19      /**
20       * Constructor which will find the ECAL conditions from the static conditions manager instance.
21       */
22      public EcalConditionsUtil() {
23          this.conditions = DatabaseConditionsManager.getInstance().getEcalConditions();
24      }
25  
26      /**
27       * Constructor which uses external reference to conditions object.
28       *
29       * @param conditions the ECAL conditions object
30       */
31      public EcalConditionsUtil(final EcalConditions conditions) {
32          this.conditions = conditions;
33      }
34  
35      /**
36       * Find a channel object from a cell ID, e.g. from a <code>CalorimeterHit</code>.
37       *
38       * @param helper the identifier helper of the hit
39       * @param cellId the cell ID of the hit
40       * @return the corresponding ECAL channel found from the physical (geometric) ID information
41       */
42      public EcalChannel findChannel(final IIdentifierHelper helper, final long cellId) {
43  
44          // Make an ID object from the hit ID.
45          final IIdentifier id = new Identifier(cellId);
46  
47          // Get the geometric field values which are subdetector ID and crystal indices.
48          final int system = helper.getValue(id, "system");
49          final int x = helper.getValue(id, "ix");
50          final int y = helper.getValue(id, "iy");
51  
52          // Create an ID to searching in the channel collection.
53          final GeometryId geometryId = new GeometryId(helper, new int[] {system, x, y});
54  
55          // Find and return the channel object.
56          return this.conditions.getChannelCollection().findChannel(geometryId);
57      }
58  
59      /**
60       * Get the DAQ crate number from a cell ID.
61       *
62       * @param helper the identifier helper of the hit
63       * @param cellId the cell ID of the hit
64       * @return the crate number of the channel
65       */
66      public int getCrate(final IIdentifierHelper helper, final long cellId) {
67          return this.findChannel(helper, cellId).getCrate();
68      }
69  
70      /**
71       * Get the DAQ slot number from a cell ID.
72       *
73       * @param helper the identifier helper of the hit
74       * @param cellId the cell ID of the hit
75       * @return the slot number of the channel
76       */
77      public int getSlot(final IIdentifierHelper helper, final long cellId) {
78          return this.findChannel(helper, cellId).getSlot();
79      }
80  
81  }
