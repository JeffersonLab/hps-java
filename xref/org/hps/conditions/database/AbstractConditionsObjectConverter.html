16  /**
17   * Implementation of default conversion from database tables records to a 
18   * {@link org.hps.conditions.api.ConditionsObjectCollection}.
19   *
20   * @param <T> The type of the returned data which should be a class extending 
21   * {@link org.hps.conditions.api.BaseConditionsObjectCollection}.
22   */
23  public abstract class AbstractConditionsObjectConverter<T> implements ConditionsConverter<T> {
24  
25      /**
26       * Initialize the logger.
27       */
28      private static final Logger LOGGER = Logger.getLogger(AbstractConditionsObjectConverter.class.getPackage().getName());
29      
30      /**
31       * Create a conditions object collection.
32       *
33       * @param manager the conditions manager
34       * @param conditionsRecord the conditions record
35       * @param tableMetaData the table data
36       * @return the conditions object collection
37       * @throws ConditionsObjectException if there is a problem creating the collection
38       */
39      private static ConditionsObjectCollection<?> createCollection(final DatabaseConditionsManager manager,
40              final ConditionsRecord conditionsRecord, final TableMetaData tableMetaData)
41              throws ConditionsObjectException {
42          ConditionsObjectCollection<?> collection;
43          try {
44              collection = tableMetaData.getCollectionClass().newInstance();
45              if (conditionsRecord != null) {
46                  collection.setConnection(manager.getConnection());
47                  collection.setTableMetaData(tableMetaData);
48                  collection.setCollectionId(conditionsRecord.getCollectionId());
49              }
50          } catch (InstantiationException | IllegalAccessException e) {
51              throw new ConditionsObjectException("Error creating conditions object collection.", e);
52          }
53          return collection;
54      }
55  
56      /**
57       * The action to take if multiple overlapping conditions sets are found. The default is using the most recently
58       * updated one.
59       */
60      private MultipleCollectionsAction multipleCollections = MultipleCollectionsAction.LAST_CREATED;
61  
62      /**
63       * Class constructor.
64       */
65      public AbstractConditionsObjectConverter() {
66      }
67  
68      /**
69       * Get the conditions data based on the name, e.g. "ecal_channels". The table information is found using the type
70       * handled by the Converter.
71       *
72       * @param conditionsManager the current conditions manager
73       * @param name the name of the conditions set (maps to table name)
74       * @return the conditions data
75       */
76      @Override
77      @SuppressWarnings({"unchecked", "rawtypes"})
78      public T getData(final ConditionsManager conditionsManager, final String name) {
79  
80          // Get the DatabaseConditionsManager which is required for using this converter.
81          final DatabaseConditionsManager databaseConditionsManager = (DatabaseConditionsManager) conditionsManager;
82  
83          // Get the TableMetaData from the table name.
84          final TableMetaData tableMetaData = TableRegistry.getTableRegistry().findByTableName(name);
85  
86          // Throw an exception if the table name does not map to a known type.
87          if (tableMetaData == null) {
88              throw new RuntimeException(new ConditionsObjectException("No table information found for name: " + name));
89          }
90  
91          // Get the ConditionsRecordCollection with the run number assignments.
92          final ConditionsRecordCollection conditionsRecords = databaseConditionsManager.findConditionsRecords(name);
93  
94          // The record with the collection information.
95          ConditionsRecord conditionsRecord = null;
96  
97          // Now we need to determine which ConditionsRecord object to use.
98          if (conditionsRecords.size() == 0) {
99              // No conditions records were found for the key.
100             throw new RuntimeException("No conditions were found with key: " + name);
101         } else if (conditionsRecords.size() == 1) {
102             // Use the single conditions set that was found.
103             conditionsRecord = conditionsRecords.get(0);
104         } else if (conditionsRecords.size() > 1) {
105             if (this.multipleCollections.equals(MultipleCollectionsAction.LAST_UPDATED)) {
106                 // Use the conditions set with the latest updated date.
107                 conditionsRecord = conditionsRecords.sortedByUpdated().get(conditionsRecords.size() - 1);
108             } else if (this.multipleCollections.equals(MultipleCollectionsAction.LAST_CREATED)) {
109                 // Use the conditions set with the latest created date.
110                 conditionsRecord = conditionsRecords.sortedByCreated().get(conditionsRecords.size() - 1);
111             } else if (this.multipleCollections.equals(MultipleCollectionsAction.LATEST_RUN_START)) {
112                 // Use the conditions set with the greatest run start value.
113                 conditionsRecord = conditionsRecords.sortedByRunStart().get(conditionsRecords.size() - 1);
114             } else if (this.multipleCollections.equals(MultipleCollectionsAction.ERROR)) {
115                 // The converter has been configured to throw an error.
116                 throw new RuntimeException("Multiple ConditionsRecord object found for conditions key " + name);
117             }
118         }
119 
120         // Create a collection of objects to return.
121         ConditionsObjectCollection collection = null;
122         try {
123             collection = createCollection(databaseConditionsManager, conditionsRecord, tableMetaData);
124         } catch (final ConditionsObjectException e) {
125             throw new RuntimeException(e);
126         }
127 
128         LOGGER.info("loading conditions set..." + '\n' + conditionsRecord);
129 
130         // Select the objects into the collection by the collection ID.
131         try {
132             collection.select(conditionsRecord.getCollectionId());
133         } catch (DatabaseObjectException | SQLException e) {
134             throw new RuntimeException("Error creating conditions collection from table " + name
135                     + " with collection ID " + conditionsRecord.getCollectionId(), e);
136         }
137 
138         return (T) collection;
139     }
140 
141     /**
142      * Get the multiple collections action.
143      *
144      * @return the multiple collections action
145      */
146     public final MultipleCollectionsAction getMultipleCollectionsAction() {
147         return this.multipleCollections;
148     }
149 
150     /**
151      * Get the specific type converted by this class.
152      *
153      * @return the class that this converter handles
154      */
155     @Override
156     public abstract Class<T> getType();
157 
158     /**
159      * Set the action that the converter will use to disambiguate when multiple conditions sets are found.
160      *
161      * @param multipleCollections the multiple collections action
162      */
163     public final void setMultipleCollectionsAction(final MultipleCollectionsAction multipleCollections) {
164         this.multipleCollections = multipleCollections;
165     }
166 
167     /**
168      * Convert object to string.
169      *
170      * @return the object converted to string
171      */
172     @Override
173     public String toString() {
174         return "ConditionsObjectConverter: type = " + this.getType() + ", multipleCollectionsAction = "
175                 + this.getMultipleCollectionsAction().toString();
176     }
