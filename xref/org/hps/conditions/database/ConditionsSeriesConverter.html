3   import java.sql.SQLException;
4   
5   import org.hps.conditions.api.ConditionsObject;
6   import org.hps.conditions.api.ConditionsObjectCollection;
7   import org.hps.conditions.api.ConditionsRecord;
8   import org.hps.conditions.api.ConditionsRecord.ConditionsRecordCollection;
9   import org.hps.conditions.api.ConditionsSeries;
10  import org.hps.conditions.api.DatabaseObjectException;
11  import org.hps.conditions.api.TableMetaData;
12  
13  /**
14   * This converter creates a {@link org.hps.conditions.api.ConditionsSeries} which is a list of
15   * {@link org.hps.conditions.api.ConditionsObjectCollection} objects having the same type. This can be used to retrieve
16   * sets of conditions that may overlap in time validity. The user may then use whichever collections are of interest to
17   * them.
18   *
19   * @see org.hps.conditions.api.ConditionsSeries
20   * @see org.hps.conditions.api.ConditionsObjectCollection
21   * @see org.hps.conditions.api.ConditionsObject
22   * @see DatabaseConditionsManager
23   * @param <ObjectType> The type of the ConditionsObject.
24   * @param <CollectionType> The type of the collection.
25   */
26  final class ConditionsSeriesConverter<ObjectType extends ConditionsObject, CollectionType extends ConditionsObjectCollection<ObjectType>> {
27  
28      /**
29       * The type of the collection.
30       */
31      final Class<CollectionType> collectionType;
32  
33      /**
34       * The type of the object.
35       */
36      final Class<ObjectType> objectType;
37  
38      /**
39       * Class constructor.
40       *
41       * @param objectType the type of the object
42       * @param collectionType the type of the collection
43       */
44      ConditionsSeriesConverter(final Class<ObjectType> objectType, final Class<CollectionType> collectionType) {
45          this.collectionType = collectionType;
46          this.objectType = objectType;
47      }
48  
49      /**
50       * Create a new conditions series.
51       *
52       * @param tableName the name of the data table
53       * @return the conditions series
54       */
55      @SuppressWarnings({"unchecked"})
56      final ConditionsSeries<ObjectType, CollectionType> createSeries(final String tableName) {
57  
58          if (tableName == null) {
59              throw new IllegalArgumentException("The tableName argument is null.");
60          }
61  
62          final DatabaseConditionsManager conditionsManager = DatabaseConditionsManager.getInstance();
63  
64          // Get the table meta data for the collection type.
65          final TableMetaData tableMetaData = conditionsManager.findTableMetaData(tableName);
66          if (tableMetaData == null) {
67              throw new RuntimeException("Table meta data for " + this.collectionType + " was not found.");
68          }
69  
70          // Create a new conditions series.
71          final ConditionsSeries<ObjectType, CollectionType> series = new ConditionsSeries<ObjectType, CollectionType>();
72  
73          // Get the ConditionsRecord with the meta-data, which will use the current run number from the manager.
74          final ConditionsRecordCollection conditionsRecords = conditionsManager.findConditionsRecords(tableName);
75  
76          for (final ConditionsRecord conditionsRecord : conditionsRecords) {
77  
78              ConditionsObjectCollection<?> collection = null;
79              try {
80                  collection = tableMetaData.getCollectionClass().newInstance();
81              } catch (InstantiationException | IllegalAccessException e1) {
82                  throw new RuntimeException(e1);
83              }
84              try {
85                  collection.setTableMetaData(tableMetaData);
86                  collection.setConnection(conditionsManager.getConnection());
87                  collection.select(conditionsRecord.getCollectionId());
88              } catch (final DatabaseObjectException | SQLException e) {
89                  throw new RuntimeException(e);
90              }
91              series.add((ConditionsObjectCollection<ObjectType>) collection);
92          }
93  
94          // Return new collection.
95          return series;
96      }
97  }
