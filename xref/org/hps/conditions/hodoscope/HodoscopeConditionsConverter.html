4   import org.hps.conditions.hodoscope.HodoscopeChannel.HodoscopeChannelCollection;

10  public class HodoscopeConditionsConverter implements ConditionsConverter<HodoscopeConditions> {
11                
12      public final HodoscopeConditions getData(final ConditionsManager manager, final String name) {
13         
14          final HodoscopeChannelCollection channels = 
15                  manager.getCachedConditions(HodoscopeChannelCollection.class, "hodo_channels").getCachedData();
16          
17          if (channels == null) {
18              throw new IllegalStateException("The Hodoscope channels collection is null.");
19          }
20                        
21          final HodoscopeConditions conditions = new HodoscopeConditions();
22  
23          conditions.setChannelCollection(channels);
24  
25          final HodoscopeGainCollection gains = 
26                  manager.getCachedConditions(HodoscopeGainCollection.class, "hodo_gains").getCachedData();
27          for (final HodoscopeGain gain : gains) {
28              final HodoscopeChannel channel = channels.findChannel(gain.getChannelId());
29              conditions.getChannelConstants(channel).setGain(gain);
30          }
31  
32          final HodoscopeCalibrationCollection calibrations = 
33                  manager.getCachedConditions(HodoscopeCalibrationCollection.class, "hodo_calibrations").getCachedData();
34          for (final HodoscopeCalibration calibration : calibrations) {
35              final HodoscopeChannel channel = channels.findChannel(calibration.getChannelId());
36              conditions.getChannelConstants(channel).setCalibration(calibration);
37          }
38  
39          final HodoscopeTimeShiftCollection timeShifts = 
40                  manager.getCachedConditions(HodoscopeTimeShiftCollection.class, "hodo_time_shifts").getCachedData();
41          for (final HodoscopeTimeShift timeShift : timeShifts) {
42              final HodoscopeChannel channel = channels.findChannel(timeShift.getChannelId());
43              conditions.getChannelConstants(channel).setTimeShift(timeShift);
44          }
45          
46          return conditions;
47      }
48  
49      public Class<HodoscopeConditions> getType() {
50          return HodoscopeConditions.class;
51      }
52  }
