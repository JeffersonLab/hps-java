100      * Convert station data to JSON.
101      * @return The converted JSON data
102      */
103     public JSONObject toJSON() {
104         JSONObject jo = new JSONObject();
105         jo.put("pid", pid);
106         jo.put("active", active);
107         jo.put("id", id);
108         jo.put("station", stationName);
109         jo.put("command", String.join(" ", command));
110         jo.put("dir", dir.getPath());
111         jo.put("log", log != null ? FilenameUtils.getBaseName(log.getPath()) : "");
112         jo.put("props", props.toJSON());
113         return jo;
114     }

116     /**
117      * Activate the reconstruction station, which includes starting its system process
118      * and connecting its remote AIDA tree
119      *
120      * @throws IOException If there is a problem activating the station's process
121      * @thows InterruptedException If station activation is interrupted
122      */
123     synchronized boolean activate(Server server) throws IOException, InterruptedException {
124 
125         LOG.info("Activating station: " + stationName);
126 
127         // Make sure station directory exists.
128         if (!dir.exists()) {
129             LOG.info("Recreating missing station dir: " + dir.getPath());
130             dir.mkdir();
131         }
132 
133         if (!configFile.exists()) {
134             // The configuration file disappeared, so recreate it
135             LOG.info("Rewriting station config file: " + configFile.getPath());
136             writeStationProperties(props, dir, stationName);
137         }
138 
139         // Setup the system process for the station
140         ProcessBuilder pb = new ProcessBuilder(command);
141         pb.directory(dir);
142         log = new File(dir.getPath() + File.separator + "out." + Integer.valueOf(id).toString() + ".log");
143         if (log.exists()) {
144             if (log.delete()) {
145                 LOG.info("Deleted old station log file: " + log.getPath());
146             } else {
147                 LOG.warning("Failed to delete old station log file " + log.getPath());
148             }
149         }
150         pb.redirectErrorStream(true);
151         pb.redirectOutput(Redirect.appendTo(log));
152         LOG.info("Station command: " + '\n' + String.join(" ", pb.command()) +'\n');
153 
154         // This can throw an exception which is fine
155         process = pb.start();
156         pid = getPid(process);
157         LOG.info("Started station process: " + pid);
158 
159         // Attempt to mount the station's remote tree which may fail with an exception
160         server.getAggregator().addRemoteTree(getRemoteTreeBind());
161 
162         // Set active flag
163         setActive(true);
164 
165         LOG.info("Activated station successfully: " + this.stationName);
166 
167         // This is returned for a result when using a Future
168         return this.active;
169     }

171     /**
172      * Deactivate a station by unmounting its remote AIDA tree and killing its system process.
173      *
174      * The ET station will be removed when the {@link Station} process actually exits.
175      *
176      * This is called by {@link StationManager#stopStation(StationProcess)}.
177      */
178     synchronized void deactivate(Server server) {
179 
180         LOG.log(Level.INFO, "Deactivating station: " + stationName);
181 
182         if (!active) {
183             LOG.warning("Station was already deactivated: " + stationName);
184             return;
185         }
186 
187         // Set active flag to false
188         setActive(false);
189 
190         // Unmount the remote AIDA tree
191         try {
192             server.getAggregator().unmount(getRemoteTreeBind());
193         } catch (Exception e) {
194             LOG.log(Level.SEVERE, "Error unmounting remote tree: " + getRemoteTreeBind(), e);
195         }
196 
197         // Wake up the ET station which signals the event loop to exit
198         server.wakeUp(this, 30000);
199 
200         // Destroy the station's system process
201         LOG.info("Destroying station's system process: " + this.getPid());
202         destroyProcess();
203 
204         LOG.info("Done deactivating station: " + stationName);
205     }

223     /**
224      * Destroy the station's system process
225      */
226     private void destroyProcess() {
227         try {
228             if (process != null) {
229                 // Wait a little bit, given that process is probably exiting already
230                 process.waitFor(10, TimeUnit.SECONDS);
231                 if (process.isAlive()) {
232                     LOG.fine("Destroying process: " + pid);
233                     // First try gentle exit
234                     process.destroy();
235                     try {
236                         process.waitFor(30, TimeUnit.SECONDS);
237                         if (process.isAlive()) {
238                             LOG.fine("Destroying process forcibly: " + pid);
239                             // Destroy forcibly e.g. using kill -9
240                             process.destroyForcibly();
241                             process.waitFor(10, TimeUnit.SECONDS);
242                         }
243                         if (!process.isAlive()) {
244                             exitValue = process.exitValue();
245                         } else {
246                             // It really shouldn't ever get here
247                             LOG.warning("Failed to stop process: " + pid);
248                         }
249                     } catch (InterruptedException e) {
250                         LOG.log(Level.WARNING, "Interrupted", e);
251                     }
252                 } else {
253                     exitValue = process.exitValue();
254                 }
255                 process = null;
256             }
257             pid = -1L;
258             LOG.fine("Exit value: " + exitValue);
259         } catch (Exception e) {
260             LOG.log(Level.WARNING, "Error destroying station system process", e);
261         }
262     }

267      * @param dir The target directory
268      * @param stationName The name of the station

271     private static File writeStationProperties(StationProperties props, File dir, String stationName) {
272         File scf = new File(dir.getPath() + File.separator + STATION_CONFIG_NAME);
273         try {
274             props.write(scf, "Configuration properties for online recon station: " + stationName);

276             throw new RuntimeException("Error writing station config file", e);
277         }
278         return scf;

281     /**
282      * Get the PID of a system process
283      * @param p The system process
284      * @return The process's PID
285      */
286     private static Long getPid(Process p) {
287         long pid = -1;
288         if (p != null) {
289             try {
290                 Field f = p.getClass().getDeclaredField("pid");
291                 f.setAccessible(true);
292                 pid = f.getLong(p);
293                 f.setAccessible(false);
294             } catch (Exception e) {
295                 pid = -1;
296             }
297         }
298         return pid;
299     }

398     /**
399      * Build the command for running the station
400      */
401     void buildCommand() {
402 
403         Property<String> logConfigFile = props.get("station.loggingConfig");
404 
405         command = new ArrayList<String>();
406 
407         command.add("java");
408 
409         // Set JVM arguments from the station properties
410         Property<String> jvmArgs = props.get("lcsim.jvm_args");
411         command.add(jvmArgs.value());
412 
413         // Logging configuration
414         if (logConfigFile.valid()) {
415             command.add("-Djava.util.logging.config.file=" + logConfigFile.value());
416         } else {
417             command.add("-Djava.util.logging.config.class=" + LoggingConfig.class.getCanonicalName());
418         }
419 
420         // Set the station name
421         command.add("-D" + Station.STAT_NAME_KEY + "=" + this.stationName);
422 
423         // Set the AIDA remote tree binding
424         command.add("-D" + Station.RTB_KEY + "=" + this.getRemoteTreeBind());
425 
426         command.add("-cp");
427         if (props.get("lcsim.classpath").valid()) {
428             // Set classpath from user setting
429             Property<String> cp = props.get("lcsim.classpath");
430             command.add(cp.value());
431         } else {
432             // Use default classpath from system
433             command.add(System.getProperty("java.class.path"));
434         }
435 
436         command.add(Station.class.getCanonicalName());
437         command.add(configFile.getPath());
438     }
