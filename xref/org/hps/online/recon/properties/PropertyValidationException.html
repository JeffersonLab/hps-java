1   package org.hps.online.recon.properties;

3   import java.util.List;
4   
5   /**
6    * Exception thrown by {@link PropertyStore#validate()} when one or more
7    * properties in a {@link PropertyStore} are invalid
8    */
9   @SuppressWarnings("serial")
10  public class PropertyValidationException extends Exception {
11  
12      PropertyValidationException(List<Property<?>> badProps) {
13          super("The following properties are not valid:" + propsToString(badProps));
14      }
15  
16      private static String propsToString(List<Property<?>> badProps) {
17          StringBuffer buff = new StringBuffer();
18          for (Property<?> p : badProps) {
19              buff.append(" " + p.name);
20          }
21          return buff.toString();
22      }
23  }
