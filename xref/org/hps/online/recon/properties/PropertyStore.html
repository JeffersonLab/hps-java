3   import java.io.File;
4   import java.io.FileInputStream;
5   import java.io.FileNotFoundException;
6   import java.io.FileOutputStream;
7   import java.io.IOException;
8   import java.util.ArrayList;
9   import java.util.HashMap;
10  import java.util.List;
11  import java.util.Map;
12  import java.util.Map.Entry;
13  import java.util.Properties;
14  
15  import org.json.JSONObject;
16  
17  /**
18   * A store of {@link Property} objects
19   */
20  public class PropertyStore {
21  
22      protected final Map<String, Property<?>> props = new HashMap<String, Property<?>>();
23  
24      public PropertyStore() {
25      }
26  
27      public final boolean has(String name) {
28          return props.containsKey(name);
29      }
30  
31      public final void add(Property<?> prop) {
32          if (props.containsKey(prop.name())) {
33              throw new IllegalArgumentException("Property already exists: " + prop.name());
34          }
35          props.put(prop.name(), prop);
36      }
37  
38      public final void add(Property<?> arr[]) {
39          for (Property<?> prop : arr) {
40              add(prop);
41          }
42      }
43  
44      @SuppressWarnings("unchecked")
45      public final <T> Property<T> get(String name) {
46          if (!has(name)) {
47              throw new IllegalArgumentException("Property does not exist: " + name);
48          }
49          return (Property<T>) props.get(name);
50      }
51  
52      public final void load(Properties propFile) {
53          for (Object keyObj : propFile.keySet()) {
54              String key = keyObj.toString();
55              if (has(key)) {
56                  Property<?> prop = props.get(key);
57                  Object rawVal = propFile.get(key);
58                  prop.from(rawVal);
59              }
60          }
61      }
62  
63      public final void load(File file) {
64          Properties p = new Properties();
65          try {
66              p.load(new FileInputStream(file));
67          } catch (IOException e) {
68              throw new RuntimeException(e);
69          }
70          load(p);
71      }
72  
73      public void fromJSON(JSONObject jo) {
74          for (String key : jo.keySet()) {
75              get(key).set(jo.get(key).toString());
76          }
77      }
78  
79      /**
80       * Convert properties to JSON.
81       * @return The converted JSON object
82       */
83      public JSONObject toJSON() {
84          JSONObject jo = new JSONObject();
85          for (Entry<String, Property<?>> entry : props.entrySet()) {
86              jo.put((String) entry.getKey(), entry.getValue().value);
87          }
88          return jo;
89      }
90  
91      /**
92       * Save properties to <code>Properties</code> for output
93       *
94       * Invalid properties with null values are skipped.
95       *
96       * @param propOut The output <code>Properties</code>
97       */
98      public void save(Properties propOut) {
99          for (Entry<String, Property<?>> entry : this.props.entrySet()) {
100             if (entry.getValue().valid()) {
101                 propOut.setProperty(entry.getKey(), entry.getValue().value().toString());
102             }
103         }
104     }
105 
106     public void write(File file, String comment) throws FileNotFoundException, IOException {
107         Properties props = new Properties();
108         save(props);
109         props.store(new FileOutputStream(file), comment);
110     }
111 
112     public void validate() throws PropertyValidationException {
113         List<Property<?>> badProps = new ArrayList<Property<?>>();
114         for (Entry<String, Property<?>> entry : this.props.entrySet()) {
115             Property<?> prop = entry.getValue();
116             if (prop.required() && !prop.valid()) {
117                 badProps.add(prop);
118             }
119         }
120         if (badProps.size() > 0) {
121             throw new PropertyValidationException(badProps);
122         }
123     }
124 
125     public String toString() {
126         return toJSON().toString();
127     }
128 
129     public int size() {
130         return this.props.size();
131     }
132 }
