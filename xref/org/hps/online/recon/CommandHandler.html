1   package org.hps.online.recon;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.logging.Logger;
6   
7   import org.json.JSONArray;
8   import org.json.JSONObject;
9   
10  /**
11   * Server-side handler for a {@link Command} sent by the {@link Client}
12   *
13   * Implementations of this class should not use any persistent state
14   * so that they are thread-safe, as the {@link #execute(JSONObject)}
15   * method may potentially be executed by multiple threads simultaneously.
16   */
17  public abstract class CommandHandler {
18  
19      protected Server server = null;
20      protected StationManager mgr = null;
21      protected Logger logger = null;
22  
23      /**
24       * Exception that handlers can throw if there is a command
25       * processing error
26       */
27      @SuppressWarnings("serial")
28      public static class CommandException extends Exception {
29  
30          public CommandException(String msg) {
31              super(msg);
32          }
33  
34          public CommandException(String msg, Exception e) {
35              super(msg, e);
36          }
37      }
38  
39      /**
40       * Create a new command handler
41       * @param server The instance of the server which is creating the handler
42       */
43      public CommandHandler(Server server) {
44          this.server = server;
45          this.mgr = this.server.getStationManager();
46          this.logger = this.server.getLogger();
47      }
48  
49      /**
50       * Get station IDs from JSON
51       * @param parameters The JSON data
52       * @return The list of station IDs (empty list if none specified)
53       */
54      public static List<Integer> getStationIDs(JSONObject parameters) {
55          List<Integer> ids = new ArrayList<Integer>();
56          if (parameters.has("ids")) {
57              JSONArray arr = parameters.getJSONArray("ids");
58              for (int i = 0; i < arr.length(); i++) {
59                  ids.add(arr.getInt(i));
60              }
61          }
62          return ids;
63      }
64  
65      /**
66       * Name of the command e.g. "create", "stop", etc.
67       * @return Name of the command
68       */
69      public abstract String getCommandName();
70  
71      /**
72       * Execute the command.
73       * @param jo The JSON input parameters
74       * @return The command result
75       */
76      public abstract CommandResult execute(JSONObject jo) throws CommandException;
77  
78  }
