1   package org.hps.online.recon.logging;
2   
3   import java.io.File;
4   import java.io.IOException;
5   import java.util.logging.FileHandler;
6   import java.util.logging.Handler;
7   import java.util.logging.Level;
8   import java.util.logging.LogManager;
9   import java.util.logging.Logger;
10  import java.util.logging.SimpleFormatter;
11  
12  import org.apache.log4j.BasicConfigurator;
13  import org.hps.online.recon.Server;
14  
15  /**
16   * Logging config for the online reconstruction {@link org.hps.online.recon.Server}
17   */
18  public class ServerLoggingConfig {
19  
20      private static final String LOG_FILE_PATH = System.getProperty("user.dir") + File.separator + "logs";
21  
22      /**
23       * Edit this and recompile the module to change the server log level
24       */
25      private static final Level LEVEL = Level.FINER;
26  
27      static {
28          System.setProperty(
29                  "java.util.logging.SimpleFormatter.format",
30                  "%1$tb %1$td, %1$tY %1$tl:%1$tM:%1$tS.%1$tL [%4$s] %5$s%6$s%n");
31      }
32  
33      public ServerLoggingConfig(String fileName) throws IOException {
34          setup(fileName);
35      }
36  
37      public ServerLoggingConfig() throws IOException {
38          setup("server.log");
39      }
40  
41      private final void setup(String fileName) throws IOException {
42  
43          // Fix stupid log4j warnings and turn all messages off
44          BasicConfigurator.configure();
45          org.apache.log4j.Logger.getRootLogger().setLevel(org.apache.log4j.Level.OFF);
46  
47          LogManager.getLogManager().reset();
48          Logger rootLogger = Logger.getLogger("");
49          File dir = new File(LOG_FILE_PATH);
50          if (!dir.exists()) {
51              dir.mkdir();
52          }
53          File logFile = new File(dir.getAbsolutePath() + File.separator + fileName);
54          Handler handler = new FileHandler(logFile.getAbsolutePath());
55          handler.setLevel(Level.ALL);
56          handler.setFormatter(new SimpleFormatter());
57          rootLogger.addHandler(handler);
58          Logger logger = Logger.getLogger(Server.class.getPackage().getName());
59          logger.setLevel(LEVEL);
60          rootLogger.setLevel(Level.WARNING);
61          System.out.println("Server log file: " + logFile.getAbsolutePath());
62      }
63  }
