1   package org.hps.online.recon.logging;
2   
3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.util.logging.LogManager;
6   
7   import org.apache.log4j.BasicConfigurator;
8   
9   /**
10   * Logging configuration for online reconstruction stations and other miscellaneous components
11   */
12  public class LoggingConfig {
13  
14      public LoggingConfig(String propName) {
15          setup(propName);
16      }
17  
18      public LoggingConfig() {
19          setup("logging.properties");
20      }
21  
22      private void setup(String propName) {
23  
24          // Fix stupid log4j warnings and turn all messages off
25          BasicConfigurator.configure();
26          org.apache.log4j.Logger.getRootLogger().setLevel(org.apache.log4j.Level.OFF);
27  
28          InputStream inputStream = LoggingConfig.class.getResourceAsStream(propName);
29          if (inputStream == null) {
30              throw new RuntimeException("Failed to read resource: " + propName);
31          }
32          try {
33              LogManager.getLogManager().readConfiguration(inputStream);
34          } catch (SecurityException | IOException e) {
35              throw new RuntimeException("Initialization of logging configuration failed", e);
36          }
37      }
38  }
