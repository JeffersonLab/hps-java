1   package org.hps.online.recon.commands;
2   
3   import java.io.File;
4   import java.io.FileInputStream;
5   import java.io.IOException;
6   import java.util.Properties;
7   
8   import org.hps.online.recon.Command;
9   
10  /**
11   * Set server configuration properties from a local file.
12   *
13   * Running this command with no arguments returns the current configuration.
14   */
15  public class LoadCommand extends Command {
16  
17      private Properties prop;
18  
19      public LoadCommand() {
20          super("config", "Set new server configuration properties", "[config.properties]",
21                  "Configuration will take effect for newly created stations."
22                  + " If no new config is provided the existing config will be printed.");
23      }
24  
25      /**
26       * Load properties file into command parameters.
27       * @param propFile The properties file
28       * @throws IOException If there is an error loading the properties
29       */
30      private void loadProperties(File propFile) throws IOException {
31          if (!propFile.exists()) {
32              throw new IllegalArgumentException("Prop file does not exist: " + propFile.getPath());
33          }
34          prop = new Properties();
35          prop.load(new FileInputStream(propFile));
36          for (Object ko : this.prop.keySet()) {
37              String key = (String) ko;
38              this.setParameter(key, this.prop.get(key).toString());
39          }
40      }
41  
42      @Override
43      protected void process() {
44          if (cl.getArgList().size() > 0) {
45              File propFile = new File(cl.getArgList().get(0));
46              try {
47                  loadProperties(propFile);
48              } catch (IOException e) {
49                  throw new RuntimeException("Error loading prop file: " + propFile.getPath(), e);
50              }
51          }
52      }
53  }
