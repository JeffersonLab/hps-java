1   package org.hps.online.recon.commands;
2   
3   import java.util.Arrays;
4   import java.util.List;
5   
6   import org.apache.commons.cli.ParseException;
7   import org.hps.online.recon.Command;
8   
9   /**
10   * Set a configuration property
11   */
12  public final class SetCommand extends Command {
13  
14      public SetCommand() {
15          super("set", "Set a configuration property", "[name] [value]", "");
16      }
17  
18      protected void process() {
19          List<String> argList = Arrays.asList(this.rawArgs);
20          if (argList.size() < 2) {
21              throw new RuntimeException("Name and value of property are both required!");
22          }
23          String name = argList.get(0);
24          String value = String.join(" ", argList.subList(1, argList.size()));
25          //System.out.println("value="+value);
26          setParameter("name", name);
27          setParameter("value", value);
28      }
29  
30      protected void parse(String[] args) throws ParseException {
31          this.rawArgs = args;
32  
33          // Don't parse options so we can easily allow characters values that would confuse the CL parser like dashes.
34      }
35  }
