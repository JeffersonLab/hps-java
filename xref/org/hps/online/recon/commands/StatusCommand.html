3   import org.apache.commons.cli.Option;
4   import org.apache.commons.cli.Options;
5   import org.hps.online.recon.Command;
6   
7   /**
8    * Show status summary of online reconstruction server and ET system.
9    */
10  public class StatusCommand extends Command {
11  
12      public StatusCommand() {
13          super("status", "Show server and station status", "[options]", "");
14      }
15  
16      protected Options getOptions() {
17          options.addOption(new Option("v", "verbose", false, "show verbose station info"));
18          return options;
19      }
20  
21      void setVerbose(boolean verbose) {
22          setParameter("verbose", verbose);
23      }
24  
25      public void process() {
26          if (cl.hasOption("v")) {
27              setVerbose(true);
28          } else {
29              setVerbose(false);
30          }
31      }
32  }
