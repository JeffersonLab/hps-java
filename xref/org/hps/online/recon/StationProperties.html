1   package org.hps.online.recon;
2   
3   import java.util.logging.Logger;
4   
5   import org.hps.online.recon.properties.IntegerProperty;
6   import org.hps.online.recon.properties.Property;
7   import org.hps.online.recon.properties.PropertyStore;
8   import org.hps.online.recon.properties.StringProperty;
9   import org.jlab.coda.et.EtConstants;
10  import org.jlab.coda.et.enums.Mode;
11  
12  /**
13   * Configuration properties for an online reconstruction
14   * {@link org.hps.online.recon.Station}
15   */
16  public class StationProperties extends PropertyStore {
17  
18      private final static String DIR = System.getProperty("user.dir");
19  
20      private final static Logger LOG = Logger.getLogger(StationProperties.class.getPackage().getName());
21  
22      /*
23      From EtConstants.java
24      debugNone           = 0;
25      debugSevere         = 1;
26      debugError          = 2;
27      debugWarn           = 3;
28      debugInfo           = 4;
29      */
30      private final static int LEVEL = EtConstants.debugInfo;
31  
32      /*
33      The wait mode for the station.
34      sleep = 0
35      timed = 1
36      async = 2
37      */
38      private final static Mode MODE = Mode.SLEEP;
39  
40      private final static String BUFFER = "/tmp/ETBuffer";
41  
42      private final static String BUILDER = org.hps.evio.LCSimPhys2019EventBuilder.class.getCanonicalName();
43  
44      /**
45       * Defines a set of properties specific to configuring and running an online reconstruction {@link Station}
46       */
47      public StationProperties() {
48          this.add(new Property<?>[] {
49                  new StringProperty ( "lcsim.detector",        "Name of detector",               null,           true),
50                  new StringProperty ( "lcsim.steering",        "Steering resource or file",      null,           true),
51                  new IntegerProperty( "lcsim.run",             "Run number for conditions",      null,           false),
52                  new StringProperty ( "lcsim.remoteTreeBind",  "Remote AIDA RMI binding",        null,           true),
53                  new StringProperty ( "lcsim.conditions",      "Conditions URL",                 null,           false),
54                  new StringProperty ( "lcsim.tag",             "Conditions tag",                 null,           false),
55                  new StringProperty ( "lcsim.builder",         "LCIO event builder",             BUILDER,        true),
56                  new StringProperty ( "lcsim.jvm_args",        "JVM args for the lcsim process", "-Xmx512m",     false),
57                  new StringProperty ( "lcsim.classpath",       "Classpath for running lcsim",    null,           false),
58                  new StringProperty ( "station.outputName",    "Base name for output files",     "output",       true),
59                  new StringProperty ( "station.outputDir",     "Directory for output files",     DIR,            true),
60                  new StringProperty ( "station.loggingConfig", "Logging config file",            null,           false),
61                  new IntegerProperty( "station.printInterval", "Event print interval",           1,              false),
62                  new StringProperty ( "et.buffer",             "Name of ET buffer file",         BUFFER,         true),
63                  new StringProperty ( "et.host",               "Host for ET connection",         "localhost",    true),
64                  new IntegerProperty( "et.port",               "Port for ET connection",         11111,          true),
65                  new StringProperty ( "et.stationName",        "Name of ET station",             null,           true),
66                  new IntegerProperty( "et.logLevel",           "ET log level",                   LEVEL,          false),
67                  new IntegerProperty( "et.waitTime",           "Wait time when getting events",  0,              false),
68                  new IntegerProperty( "et.mode",               "Mode when getting events",       MODE.ordinal(), false),
69                  new IntegerProperty( "et.chunk",              "Chunk size when getting events", 1,              false),
70                  new IntegerProperty( "et.prescale",           "Event prescale factor",          1,              false),
71                  new IntegerProperty( "et.connectionAttempts", "Max ET connection attempts",     10,             false)
72          });
73      }
74  
75      /**
76       * Copy constructor
77       * @param sp The source properties to copy into the new object
78       */
79      public StationProperties(StationProperties sp) {
80          for (Property<?> p : sp.props.values()) {
81              add((Property<?>) p.clone());
82          }
83      }
84  }
