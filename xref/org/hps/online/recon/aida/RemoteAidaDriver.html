27  /**
28   * Driver for providing remote AIDA functionality
29   *
30   * Some basic event performance plots are also implemented.
31   */
32  public class RemoteAidaDriver extends Driver {
33  
34      static final Logger LOG = Logger.getLogger(RemoteAidaDriver.class.getPackage().getName());
35  
36      static {
37          System.setProperty("hep.aida.IAnalysisFactory", BatchAnalysisFactory.class.getName());
38          System.setProperty("java.awt.headless", "true");
39      }
40  
41      protected RemoteServer treeServer;
42      protected RmiServer rmiTreeServer;
43      private boolean serverDuplex = true;
44  
45      protected final AIDA aida = AIDA.defaultInstance();
46      protected final IAnalysisFactory af = aida.analysisFactory();
47      protected final IDevTree tree = (IDevTree) aida.tree();
48      protected final IHistogramFactory hf = af.createHistogramFactory(tree);
49      protected final IDataPointSetFactory dpsf = af.createDataPointSetFactory(tree);
50  
51      private String remoteTreeBind = null;
52      private String stationName = null;
53      private Integer stationNum = null;
54  
55      /*
56       * Performance plots
57       */
58      private static final long UPDATE_INTERVAL = 1000L;
59      private IHistogram1D eventCountH1D;
60      private IDataPointSet eventRateDPS;
61      private IDataPointSet millisPerEventDPS;
62  
63      /*
64       * Event timing
65       */
66      private int eventsProcessed = 0;
67      private long start = -1L;
68      private Timer timer;
69      protected int eventCount = 0;
70      private int NPOINTS = 25;
71  
72      public RemoteAidaDriver() {
73  
74          // Set the station name from the system property
75          if (System.getProperties().containsKey(Station.STAT_NAME_KEY)) {
76              stationName = System.getProperty(Station.STAT_NAME_KEY);
77              LOG.info("Station name set from system prop: " + stationName);
78          } else {
79              throw new RuntimeException("Station name not set in system props: " + Station.STAT_NAME_KEY);
80          }
81  
82          // Set the station number
83          stationNum = Integer.valueOf(stationName.substring(stationName.lastIndexOf("_") + 1));
84          LOG.info("Station num set from name: " + stationNum);
85  
86          // Set the RMI URL for the remote AIDA connection
87          if (System.getProperties().containsKey(Station.RTB_KEY)) {
88              remoteTreeBind = System.getProperty(Station.RTB_KEY);
89              LOG.info("Station remote tree binding set from system prop: " + this.remoteTreeBind);
90          }
91  
92          // Do not allow object overwriting
93          tree.setOverwrite(false);
94      }
95  
96      public String getStationName() {
97          return stationName;
98      }
99  
100     public void setRemoteTreeBind(String remoteTreeBind) {
101         this.remoteTreeBind = remoteTreeBind;
102     }

104     @Override
105     protected void endOfData() {
106 
107         timer.cancel();
108 
109         disconnect();
110     }
111 
112     @Override
113     protected void startOfData() {
114         try {
115             connect();
116         } catch (IOException e) {
117             throw new RuntimeException("Failed to connect remote AIDA tree", e);
118         }
119 
120         /*
121          * Performance plots
122          */
123 
124         tree.mkdir("/EventsProcessed");
125         tree.cd("/EventsProcessed");
126 
127         eventCountH1D = aida.histogram1D("Event Count", 1, 0., 1.0);
128         eventCountH1D.annotation().addItem("yAxisLabel", "Total Events");
129 
130         eventRateDPS = dpsf.create("Event Rate", "Event Rate", 2);
131         eventRateDPS.annotation().addItem("yAxisLabel", "Events / Second");
132 
133         millisPerEventDPS = dpsf.create("Event Time", 2);
134         millisPerEventDPS.annotation().addItem("yAxisLabel", "Millis / Event");
135 
136         startEventTimer();
137     }
138 
139     private void startEventTimer() {
140         TimerTask task = new TimerTask() {
141             public void run() {
142                 if (eventsProcessed > 0 && start > 0) {
143                     //LOG.info("Event timer is updating -- start=" + start + "; eventsProcessed=" + eventsProcessed);
144                     long elapsed = System.currentTimeMillis() - start;
145                     double eps = (double) eventsProcessed / ((double) elapsed / 1000L);
146                     double mpe = (double) elapsed / (double) eventsProcessed;
147 
148                     long currTimeSec = System.currentTimeMillis() / 1000L;
149 
150                     IDataPoint dp = eventRateDPS.addPoint();
151                     dp.coordinate(0).setValue(currTimeSec);
152                     dp.coordinate(1).setValue(eps);
153                     while (eventRateDPS.size() > NPOINTS) {
154                         eventRateDPS.removePoint(0);
155                     }
156 
157                     dp = millisPerEventDPS.addPoint();
158                     dp.coordinate(0).setValue(currTimeSec);
159                     dp.coordinate(1).setValue(mpe);
160                     while (millisPerEventDPS.size() > NPOINTS) {
161                         millisPerEventDPS.removePoint(0);
162                     }
163                 }
164                 start = System.currentTimeMillis();
165                 eventsProcessed = 0;
166             }
167         };
168         timer = new Timer("Event Timer");
169         timer.scheduleAtFixedRate(task, 0, UPDATE_INTERVAL);
170     }
171 
172     synchronized final void disconnect() {
173         try {
174             if (rmiTreeServer != null) {
175                 ((RmiServerImpl) rmiTreeServer).disconnect();
176                 rmiTreeServer = null;
177             }
178             if (treeServer != null) {
179                 treeServer.close();
180                 treeServer = null;
181             }
182         } catch (Exception e) {
183             LOG.log(Level.WARNING, "Error disconnecting remote AIDA tree", e);
184         }
185     }
186 
187     synchronized final void connect() throws IOException {
188 
189         // HACK: Fixes exceptions from missing AIDA converters
190         final RmiStoreFactory rsf = new RmiStoreFactory();
191 
192         if (rmiTreeServer != null) {
193             LOG.warning("Already connected -- RMI tree server is not null");
194             return;
195         }
196         if (remoteTreeBind == null) {
197             throw new IllegalStateException("remoteTreeBind is not set");
198         }
199         LOG.info("Setting up remote AIDA tree: " + remoteTreeBind);
200 
201         treeServer = new RemoteServer(tree, serverDuplex);
202         rmiTreeServer = new RmiServerImpl(treeServer, remoteTreeBind);
203         LOG.info("Done setting up remote AIDA tree: " + remoteTreeBind);
204     }
205 
206     public void process(EventHeader event) {
207         eventCountH1D.fill(0.5);
208         eventCount += 1;
209         eventsProcessed += 1;
210     }
