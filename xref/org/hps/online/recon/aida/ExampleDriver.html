21  /**
22   * Create example histograms and data points in a remote AIDA tree
23   */
24  public class ExampleDriver extends RemoteAidaDriver {
25  
26      /*
27       * AIDA paths
28       */
29      private static final String ECAL_DIR = "/subdet/ecal";
30      private static final String TRACKER_DIR = "/subdet/tracker";
31      private static final String PERF_DIR = "/perf";
32  
33      /*
34       * Performance plots
35       */
36      private IHistogram1D eventCountH1D;
37      private IDataPointSet eventRateDPS;
38      private IDataPointSet millisPerEventDPS;
39  
40      /*
41       * Tracker plots
42       */
43      private IHistogram1D tracksPerEventH1D;
44      private IHistogram1D rawHitsPerTrackH1D;
45      private ICloud1D rawTrackerHitsPerEventC1D;
46      private ICloud1D chi2C1D;
47      private ICloud1D d0C1D;
48      private ICloud1D omegaC1D;
49      private ICloud1D phiC1D;
50      private ICloud1D tanLambdaC1D;
51      private ICloud1D z0C1D;
52      private ICloud1D hthLayerC1D;
53      private ICloud1D hthModuleC1D;
54      private ICloud1D fittedTrackerHitsPerEventC1D;
55  
56      /*
57       * Ecal plots
58       */
59      private IHistogram2D ecalReadoutHitsXYH2D;
60      private IHistogram1D ecalReadoutHitsPerEventH1D;
61      private IHistogram1D adcValuesH1D;
62      private IHistogram1D clustersPerEventH1D;
63      private ICloud1D clusEnergyC1D;
64      private ICloud1D clusTimeC1D;
65      private ICloud1D clusNHitsC1D;
66  
67      /*
68       * Collection names
69       */
70      private static final String CLUSTERS = "EcalClusters";
71      private static final String RAW_TRACKER_HITS = "SVTRawTrackerHits";
72      private static final String FITTED_HITS = "SVTFittedRawTrackerHits";
73      private static final String TRACKS = "MatchedTracks";
74      private static final String ECAL_READOUT_HITS = "EcalReadoutHits";
75      private static final String HELICAL_TRACK_HITS = "HelicalTrackHits";
76  
77      // Convert clouds to histograms after this many entries (-1 means do not convert automatically)
78      private static final int CLOUD_MAX = -1;
79  
80      /*
81       * Event timing
82       */
83      private int eventsProcessed = 0;
84      private long start = -1L;
85      private Timer timer;
86  
87      private ICloud1D createCloud1D(String name) {
88          return hf.createCloud1D(name, name, CLOUD_MAX, "autoconvert=false");
89      }
90  
91      public void startOfData() {
92  
93          /*
94           * Tracker plots
95           */
96  
97          tree.mkdirs(TRACKER_DIR);
98          tree.cd(TRACKER_DIR);
99  
100         tracksPerEventH1D = aida.histogram1D("Tracks Per Event", 20, 0., 20.);
101         tracksPerEventH1D.annotation().addItem("xAxisLabel", "Tracks / Event");
102         tracksPerEventH1D.annotation().addItem("yAxisLabel", "Count");
103 
104         rawHitsPerTrackH1D = aida.histogram1D("Raw Hits Per Track", 10, 0., 10);
105         //rawHitsPerTrackH1D.annotation().addItem("xAxisLabel", "Hits / Track");
106         //rawHitsPerTrackH1D.annotation().addItem("yAxisLabel", "Count");
107 
108         rawTrackerHitsPerEventC1D = createCloud1D("Raw Tracker Hits Per Event");
109         chi2C1D = createCloud1D("Chi2");
110         d0C1D = createCloud1D("D0");
111         omegaC1D = createCloud1D("Omega");
112         phiC1D = createCloud1D("Phi");
113         tanLambdaC1D = createCloud1D("Tan Lambda");
114         z0C1D = createCloud1D("Z0");
115         fittedTrackerHitsPerEventC1D = createCloud1D("Fitted Tracker Hits Per Event");
116         hthLayerC1D = createCloud1D("Helical Track Hit Layer");
117         hthModuleC1D = createCloud1D("Helical Track Hit Module");
118 
119         /*
120          * Ecal plots
121          */
122 
123         tree.mkdirs(ECAL_DIR);
124         tree.cd(ECAL_DIR);
125 
126         clusEnergyC1D = createCloud1D("Cluster Energy");
127         clusTimeC1D = createCloud1D("Cluster Time");
128         clusNHitsC1D = createCloud1D("Cluster N Hits");
129         ecalReadoutHitsXYH2D = aida.histogram2D("Readout Hits XY", 47, -23.5, 23.5, 11, -5.5, 5.5);
130         //ecalReadoutHitsXYH2D.annotation().addItem("xAxisLabel", "X Index");
131         //ecalReadoutHitsXYH2D.annotation().addItem("yAxisLabel", "Y Index");
132 
133         ecalReadoutHitsPerEventH1D = aida.histogram1D("Readout Hits Per Event ", 50, 0, 50.);
134         clustersPerEventH1D = aida.histogram1D("Clusters Per Event", 10, -0.5, 9.5);
135         adcValuesH1D = aida.histogram1D("Readout Hit ADC Values", 300, 0, 300.);
136 
137         tree.mkdir(PERF_DIR);
138         tree.cd(PERF_DIR);
139 
140         /*
141          * Performance plots
142          */
143 
144         eventCountH1D = aida.histogram1D("Event Count", 1, 0., 1.0);
145         eventRateDPS = dpsf.create("Event Rate", "Event Rate", 1);
146         millisPerEventDPS = dpsf.create("Millis Per Event", 1);
147 
148         TimerTask task = new TimerTask() {
149             public void run() {
150                 if (eventsProcessed > 0 && start > 0) {
151                     long elapsed = System.currentTimeMillis() - start;
152                     double eps = (double) eventsProcessed / ((double) elapsed / 1000L);
153                     double mpe = (double) elapsed / (double) eventsProcessed;
154                     LOG.fine("Event Timer: " + eps + " Hz");
155 
156                     IDataPoint dp = eventRateDPS.addPoint();
157                     dp.coordinate(0).setValue(eps);
158                     while (eventRateDPS.size() > 10) {
159                         eventRateDPS.removePoint(0);
160                     }
161 
162                     dp = millisPerEventDPS.addPoint();
163                     dp.coordinate(0).setValue(mpe);
164                     while (millisPerEventDPS.size() > 10) {
165                         millisPerEventDPS.removePoint(0);
166                     }
167                 }
168                 start = System.currentTimeMillis();
169                 eventsProcessed = 0;
170             }
171         };
172         Timer timer = new Timer("Event Timer");
173         timer.scheduleAtFixedRate(task, 0, 5000L);
174 
175         super.startOfData();
176     }
177 
178     public void endOfData() {
179 
180         timer.cancel();
181 
182         super.endOfData();
183     }
184 
185     public void process(EventHeader event) {
186 
187         eventCountH1D.fill(0.5);
188 
189         List<Track> tracks = event.get(Track.class, TRACKS);
190 
191         aida.tree().cd(TRACKER_DIR);
192 
193         tracksPerEventH1D.fill(tracks.size());
194 
195         for (Track track : tracks) {
196             chi2C1D.fill(track.getChi2());
197             List<TrackState> trackStates = track.getTrackStates();
198             TrackState ts = trackStates.get(0);
199             if (ts != null) {
200                 d0C1D.fill(ts.getD0());
201                 omegaC1D.fill(ts.getOmega());
202                 phiC1D.fill(ts.getPhi());
203                 tanLambdaC1D.fill(ts.getTanLambda());
204                 z0C1D.fill(ts.getZ0());
205             }
206             this.rawHitsPerTrackH1D.fill(track.getTrackerHits().size());
207         }
208 
209         List<TrackerHit> helicalTrackHits = event.get(TrackerHit.class, HELICAL_TRACK_HITS);
210         for (TrackerHit hit : helicalTrackHits) {
211             int layer = ((RawTrackerHit) hit.getRawHits().get(0)).getLayerNumber();
212             layer = (int) Math.ceil((double) layer / 2);
213             hthLayerC1D.fill(layer);
214             int module = layer / 2 + 1;
215             hthModuleC1D.fill(module);
216         }
217 
218         rawTrackerHitsPerEventC1D.fill(
219                 event.get(RawTrackerHit.class, RAW_TRACKER_HITS).size());
220 
221         fittedTrackerHitsPerEventC1D.fill(
222                 event.get(LCRelation.class, FITTED_HITS).size());
223 
224         List<RawTrackerHit> ecalHits = event.get(RawTrackerHit.class, ECAL_READOUT_HITS);
225         ecalReadoutHitsPerEventH1D.fill(ecalHits.size());
226         for (RawTrackerHit hit : ecalHits) {
227             int column = hit.getIdentifierFieldValue("ix");
228             int row = hit.getIdentifierFieldValue("iy");
229             ecalReadoutHitsXYH2D.fill(column, row);
230             for (short adcValue : hit.getADCValues()) {
231                 adcValuesH1D.fill(adcValue);
232             }
233         }
234 
235         List<Cluster> clusterList = event.get(Cluster.class, CLUSTERS);
236         for (Cluster clus : clusterList) {
237             clusEnergyC1D.fill(clus.getEnergy());
238             clusTimeC1D.fill(clus.getCalorimeterHits().get(0).getTime());
239             clusNHitsC1D.fill(clus.getCalorimeterHits().size());
240         }
241         clustersPerEventH1D.fill(clusterList.size());
242 
243         ++eventsProcessed;
244     }
