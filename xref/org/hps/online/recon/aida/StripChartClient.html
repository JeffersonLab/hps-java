111     @SuppressWarnings("deprecation")
112     private void parse(String[] args) {
113         Parser parser = new BasicParser();
114         CommandLine cl = null;
115         try {
116             cl = parser.parse(options, args);
117         } catch (ParseException e) {
118             throw new RuntimeException(e);
119         }
120         if (cl.hasOption("h")) {
121             HelpFormatter help = new HelpFormatter();
122             help.printHelp("r", "", options, "", true);
123             System.exit(0);
124         }
125 
126         if (cl.hasOption("H")) {
127             host = cl.getOptionValue("H");
128         } else {
129             try {
130                 host = InetAddress.getLocalHost().getHostName();
131             } catch (UnknownHostException e) {
132                 throw new RuntimeException(e);
133             }
134         }
135         if (cl.hasOption("p")) {
136             port = Integer.parseInt(cl.getOptionValue("p"));
137         }
138         if (cl.hasOption("s")) {
139             serverName = cl.getOptionValue("s");
140         }
141         if (cl.hasOption("x")) {
142             plotterWidth = Integer.parseInt(cl.getOptionValue("x"));
143         }
144         if (cl.hasOption("y")) {
145             plotterHeight = Integer.parseInt(cl.getOptionValue("y"));
146         }
147     }

149     private void connect() throws IOException {
150         boolean clientDuplex = true;
151         boolean hurry = false;
152         String treeBindName = "//"+host+":"+port+"/"+serverName;
153         System.out.println("Connecting to RMI server: " + treeBindName);
154         String options = "duplex=\""+clientDuplex+"\",RmiServerName=\"rmi:"+treeBindName+"\",hurry=\""+hurry+"\"";
155         clientTree = tf.create(host, RmiStoreFactory.storeType, true, false, options);
156         System.out.println("Successfully connected!");
157         System.out.println("Remote tree objects: ");
158         clientTree.ls("/", true);
159         System.out.println();
160     }

166     private void addPlots() {
167         String[] names = clientTree.listObjectNames("/", true);
168         String[] types = clientTree.listObjectTypes("/", true);
169         List<String> dpsNames = new ArrayList<String>();
170         for (int i = 0; i < names.length; i++) {
171             if (types[i].equals("IDataPointSet") && !names[i].contains("/avg")) {
172                 dpsNames.add(names[i]);
173             }
174         }
175         int n = dpsNames.size();
176         int rowsCols = (int) Math.ceil(Math.sqrt(n));
177         plotter.createRegions(rowsCols, rowsCols);
178         int regionInd = 0;
179         for (String dpsName : dpsNames) {
180             IDataPointSet dps = (IDataPointSet) clientTree.find(dpsName);
181             String label = "";
182             if (dps.annotation().hasKey("label")) {
183                 label = dps.annotation().value("label");
184             }
185             IPlotterRegion region = plotter.region(regionInd);
186             IPlotterStyle regStyle = pf.createPlotterStyle();
187             regStyle.yAxisStyle().setLabel(label);
188             region.setStyle(regStyle);
189             region.plot(dps);
190             plots.put(dps, region);
191 
192             String[] spl = dpsName.split("/");
193             String dirName = spl[1];
194             String plotName = spl[2];
195             String avgPath = "/" + dirName + "/avg/" + plotName;
196             try {
197                 IDataPointSet avg = (IDataPointSet) clientTree.find(avgPath);
198                 region.plot(avg, avgStyle);
199             } catch (IllegalArgumentException e) {
200             }
201             ++regionInd;
202         }
203     }

205     private void update() {
206         for (Entry<IDataPointSet, IPlotterRegion> entry : this.plots.entrySet()) {
207             update(entry.getKey(), entry.getValue());
208         }
209     }
210 
211     private void update(IDataPointSet dps, IPlotterRegion region) {
212         IDataPoint dp = dps.point(dps.size() - 1);
213         double sec = dp.coordinate(0).value();
214         double before = sec - NBEFORE;
215         double after = sec + NAFTER;
216         AxisStyle xStyle = (AxisStyle) region.style().xAxisStyle();
217         xStyle.setLowerLimit(Double.toString(before).toString());
218         xStyle.setUpperLimit(Double.toString(after).toString());
219         region.refresh();
220     }

222     private void setPanelEnabled(JPanel panel, Boolean isEnabled) {
223         panel.setEnabled(isEnabled);
224         Component[] components = panel.getComponents();
225         for (Component component : components) {
226             if (component instanceof JPanel) {
227                 setPanelEnabled((JPanel) component, isEnabled);
228             }
229             component.setEnabled(isEnabled);
230         }
231     }
232 
233     private void show() {
234         plotter.setParameter("plotterWidth", plotterWidth.toString());
235         plotter.setParameter("plotterHeight", plotterHeight.toString());
236         plotter.show();
237         setPanelEnabled(((Plotter) plotter).panel(), false);
238         JFrame frame = (JFrame) SwingUtilities.getWindowAncestor(((Plotter) plotter).panel());
239         frame.addWindowListener(new WindowAdapter() {
240             public void windowClosing(WindowEvent e) {
241                 StripChartClient.this.windowClosed = true;
242             }
243         });
244     }
245 
246     private void run() {
247         try {
248             connect();
249         } catch (IOException e) {
250             throw new RuntimeException(e);
251         }
252         addPlots();
253         show();
254         while(!this.windowClosed) {
255             update();
256             try {
257                 Thread.sleep(updateInterval);
258             } catch (InterruptedException e) {
259                 e.printStackTrace();
260             }
261         }
262         try {
263             close();
264         } catch (IOException e) {
265             e.printStackTrace();
266         }
267     }
