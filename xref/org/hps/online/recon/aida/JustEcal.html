205     public void process(EventHeader event) {
206 
207         super.process(event);
208 
209         int nhits = 0;
210         int chits[] = new int[11 * 47];
211 
212         int orTrigTime = 4097;
213         int topTrigTime = 4097;
214         int botTrigTime = 4097;
215 
216         if (event.hasCollection(GenericObject.class, "TriggerBank")) {
217             List<GenericObject> triggerList = event.get(GenericObject.class, "TriggerBank");
218             if (!triggerList.isEmpty()) {
219                 GenericObject triggerData = triggerList.get(0);
220 
221                 if (triggerData instanceof SSPData) {
222                     // TODO: TOP, BOTTOM, OR, and AND triggers are test
223                     // run specific parameters and are not supported
224                     // by SSPData.
225                     orTrigTime = 0; // ((SSPData)triggerData).getOrTrig();
226                     topTrigTime = 0; // ((SSPData)triggerData).getTopTrig();
227                     botTrigTime = 0; // ((SSPData)triggerData).getBotTrig();
228 
229                     orTrigTimePlot.fill(orTrigTime);
230 
231                 }
232 
233             }// end if triggerList isEmpty
234         }
235 
236         if (event.hasCollection(CalorimeterHit.class, RAW_ECAL_HITS)) {
237             List<CalorimeterHit> hits = event.get(CalorimeterHit.class, RAW_ECAL_HITS);
238             hitCountPlot.fill(hits.size());
239 
240             double maxEnergy = 0;
241             double topTime = Double.POSITIVE_INFINITY;
242             double botTime = Double.POSITIVE_INFINITY;
243             double orTime = Double.POSITIVE_INFINITY;
244             for (CalorimeterHit hit : hits) {
245 
246                 int column = hit.getIdentifierFieldValue("ix");
247                 int row = hit.getIdentifierFieldValue("iy");
248                 int id = EcalMonitoringUtilities.getHistoIDFromRowColumn(row, column);
249                 hitCountFillPlot.fill(column, row);
250                 {
251                     chits[id]++;
252                     nhits++;
253                 }
254                 hitEnergyPlot.fill(hit.getRawEnergy());
255                 hitTimePlot.fill(hit.getTime());
256 
257                 if (hit.getTime() < orTime) {
258                     orTime = hit.getTime();
259                 }
260                 if (hit.getIdentifierFieldValue("iy") > 0 && hit.getTime() < topTime) {
261                     topTime = hit.getTime();
262                 }
263                 if (hit.getIdentifierFieldValue("iy") < 0 && hit.getTime() < botTime) {
264                     botTime = hit.getTime();
265                 }
266                 if (hit.getRawEnergy() > maxEnergy) {
267                     maxEnergy = hit.getRawEnergy();
268                 }
269             }
270 
271             if (orTime != Double.POSITIVE_INFINITY) {
272                 orTimePlot.fill(orTime);
273                 orTimePlot2D.fill(orTime, orTrigTime);
274             }
275             if (topTime != Double.POSITIVE_INFINITY) {
276                 topTimePlot.fill(topTime);
277             }
278             if (botTime != Double.POSITIVE_INFINITY) {
279                 botTimePlot.fill(botTime);
280             }
281             hitMaxEnergyPlot.fill(maxEnergy);
282         }
283 
284         if (nhits > 0) {
285             for (int ii = 0;
286                     ii < (11 * 47); ii++) {
287                 occupancyFill[ii] += (1. * chits[ii]) / nhits;
288             }
289         }
290 
291         // calorimeter and the bottom.
292         List<Cluster> topList = new ArrayList<Cluster>();
293         List<Cluster> bottomList = new ArrayList<Cluster>();
294 
295         // Track the highest energy cluster in the event.
296         double maxEnergy = 0.0;
297         if (event.hasCollection(Cluster.class, ECAL_CLUSTERS)) {
298             List<Cluster> clusters = event.get(Cluster.class, ECAL_CLUSTERS);
299             for (Cluster cluster : clusters) {
300                 clusterCountFillPlot.fill(cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("ix"), cluster
301                         .getCalorimeterHits().get(0).getIdentifierFieldValue("iy"));
302 
303                 if (cluster.getEnergy() > maxEnergy) {
304                     maxEnergy = cluster.getEnergy();
305                 }
306 
307                 // Get the list of calorimeter hits and its size.
308                 List<CalorimeterHit> hitList = cluster.getCalorimeterHits();
309                 int hitCount = hitList.size();
310 
311                 // Track cluster statistics.
312                 double xEnergyWeight = 0.0;
313                 double yEnergyWeight = 0.0;
314                 double[] hitTimes = new double[hitCount];
315                 double totalHitEnergy = 0.0;
316 
317                 // Iterate over the hits and extract statistics from them.
318                 for (int hitIndex = 0; hitIndex < hitCount; hitIndex++) {
319                     hitTimes[hitIndex] = hitList.get(hitIndex).getTime();
320                     totalHitEnergy += hitList.get(hitIndex).getRawEnergy();
321                     xEnergyWeight += (hitList.get(hitIndex).getRawEnergy() * hitList.get(hitIndex)
322                             .getIdentifierFieldValue("ix"));
323                     yEnergyWeight += (hitList.get(hitIndex).getRawEnergy() * hitList.get(hitIndex)
324                             .getIdentifierFieldValue("iy"));
325                 }
326 
327                 // If the cluster energy exceeds zero, plot the cluster
328                 // statistics.
329                 if (cluster.getEnergy() > 0) {
330                     clusterSizePlot.fill(hitCount);
331                     clusterTimes.fill(StatUtils.mean(hitTimes, 0, hitCount));
332                     clusterTimeSigma.fill(Math.sqrt(StatUtils.variance(hitTimes, 0, hitCount)));
333                     edgePlot.fill(xEnergyWeight / totalHitEnergy, yEnergyWeight / totalHitEnergy);
334                 }
335 
336                 // Fill the single cluster plots.
337                 clusterEnergyPlot.fill(cluster.getEnergy());
338 
339                 // Cluster pairs are formed from all top/bottom cluster
340                 // combinations. To create these pairs, separate the
341                 // clusters into two lists based on their y-indices.
342                 if (cluster.getCalorimeterHits().get(0).getIdentifierFieldValue("ix") > 0) {
343                     topList.add(cluster);
344                 } else {
345                     bottomList.add(cluster);
346                 }
347             }
348             // Populate the event plots.
349             clusterCountPlot.fill(clusters.size());
350             if (maxEnergy > 0) {
351                 clusterMaxEnergyPlot.fill(maxEnergy);
352             }
353 
354             // Create a list to store cluster pairs.
355             List<Cluster[]> pairList = new ArrayList<Cluster[]>(topList.size() * bottomList.size());
356 
357             // Form pairs from all possible combinations of clusters
358             // from the top and bottom lists.
359             for (Cluster topCluster : topList) {
360                 for (Cluster bottomCluster : bottomList) {
361                     // Make a cluster pair array.
362                     Cluster[] pair = new Cluster[2];
363 
364                     // The lower energy cluster goes in the second slot.
365                     if (topCluster.getEnergy() > bottomCluster.getEnergy()) {
366                         pair[0] = topCluster;
367                         pair[1] = bottomCluster;
368                     } else {
369                         pair[0] = bottomCluster;
370                         pair[1] = topCluster;
371                     }
372 
373                     // Add the pair to the pair list.
374                     pairList.add(pair);
375                 }
376             }
377 
378             // Iterate over each pair and calculate the pair cut values.
379             for (Cluster[] pair : pairList) {
380                 // Get the energy slope value.
381                 double energySumValue = TriggerModule.getValueEnergySum(pair);
382                 double energyDifferenceValue = TriggerModule.getValueEnergyDifference(pair);
383                 double energySlopeValue = TriggerModule.getValueEnergySlope(pair, 0.005500);
384                 double coplanarityValue = TriggerModule.getValueCoplanarity(pair);
385                 double xMean = ((pair[0].getEnergy() * pair[0].getPosition()[0]) + (pair[1].getEnergy() * pair[1]
386                         .getPosition()[0])) / energySumValue;
387                 double yMean = ((pair[0].getEnergy() * pair[0].getPosition()[1]) + (pair[1].getEnergy() * pair[1]
388                         .getPosition()[1])) / energySumValue;
389 
390                 // Populate the cluster pair plots.
391                 pairEnergySum.fill(energySumValue, 1);
392                 ;
393                 pairEnergyDifference.fill(energyDifferenceValue, 1);
394 //                pairEnergySlope.fill(energySlopeValue, 1);
395                 pairCoplanarity.fill(coplanarityValue, 1);
396                 pairEnergyPositionMeanX.fill(xMean);
397                 pairEnergyPositionMeanY.fill(yMean);
398                 pairMassOppositeHalf.fill(getClusterPairMass(pair));
399 
400                 int idx0 = pair[0].getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
401                 int idx1 = pair[1].getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
402                 if (Math.abs(idx0) > 2 && Math.abs(idx1) > 2) {
403                     pairMassOppositeHalfFid.fill(getClusterPairMass(pair));
404                 }
405 
406             }
407             // Create a list to store cluster pairs.
408             List<Cluster[]> pairListSameHalf = new ArrayList<Cluster[]>();
409 
410             // Form pairs from all possible combinations of clusters
411             // from the top and bottom lists.
412             for (Cluster topCluster : topList) {
413                 for (Cluster topCluster2 : topList) {
414                     // Make a cluster pair array.
415                     Cluster[] pair = new Cluster[2];
416 
417                     // The lower energy cluster goes in the second slot.
418                     if (topCluster.getEnergy() > topCluster2.getEnergy()) {
419                         pair[0] = topCluster;
420                         pair[1] = topCluster2;
421                     } else {
422                         pair[0] = topCluster2;
423                         pair[1] = topCluster;
424                     }
425 
426                     // Add the pair to the pair list.
427                     pairListSameHalf.add(pair);
428                 }
429             }
430 
431             for (Cluster cluster : bottomList) {
432                 for (Cluster cluster2 : bottomList) {
433                     // Make a cluster pair array.
434                     Cluster[] pair = new Cluster[2];
435 
436                     // The lower energy cluster goes in the second slot.
437                     if (cluster.getEnergy() > cluster2.getEnergy()) {
438                         pair[0] = cluster;
439                         pair[1] = cluster2;
440                     } else {
441                         pair[0] = cluster2;
442                         pair[1] = cluster;
443                     }
444 
445                     // Add the pair to the pair list.
446                     pairListSameHalf.add(pair);
447                 }
448             }
449             // Iterate over each pair and calculate the pair cut values.
450             for (Cluster[] pair : pairListSameHalf) {
451                 // Get the energy slope value.
452                 pairMassSameHalf.fill(getClusterPairMass(pair));
453                 int idx0 = pair[0].getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
454                 int idx1 = pair[1].getCalorimeterHits().get(0).getIdentifierFieldValue("ix");
455                 if (Math.abs(idx0) > 2 && Math.abs(idx1) > 2) {
456                     pairMassSameHalfFid.fill(getClusterPairMass(pair));
457                 }
458             }
459 
460         } else {
461             clusterCountPlot.fill(0);
462         }
463 
464         thisTime = System.currentTimeMillis() / 1000;
465         thisEventN = event.getEventNumber();
466         thisEventTime = event.getTimeStamp() / 1E9;
467         if ((thisTime - prevTime) > eventRefreshRate) {
468             double scale = 1.;
469 
470             if (NoccupancyFill > 0) {
471                 scale = (thisEventN - prevEventN) / NoccupancyFill;
472                 scale = scale / (thisEventTime - prevEventTime);
473                 scale /= 1000.; // do KHz
474             }
475             // System.out.println("Event: "+thisEventN+" "+prevEventN);
476             // System.out.println("Time: "+thisEventTime+" "+prevEventTime);
477             // System.out.println("Monitor: "+thisTime+" "+prevTime+" "+NoccupancyFill);
478             if (scale > 0) {
479                 hitCountFillPlot.scale(scale);
480                 clusterCountFillPlot.scale(scale);
481                 redraw();
482             }
483             prevTime = thisTime;
484             prevEventN = thisEventN;
485             prevEventTime = thisEventTime;
486             NoccupancyFill = 0;
487         } else {
488             NoccupancyFill++;
489         }
490 
491     }

493     private static Map<Integer, Hep3Vector> createStripPositionMap(HpsSiSensor sensor) {
494         Map<Integer, Hep3Vector> positionMap = new HashMap<Integer, Hep3Vector>();
495         for (ChargeCarrier carrier : ChargeCarrier.values()) {
496             if (sensor.hasElectrodesOnSide(carrier)) {
497                 // SiSensorElectrodes electrodes = sensor.getReadoutElectrodes();
498                 SiSensorElectrodes strips = (SiSensorElectrodes) sensor.getReadoutElectrodes(carrier);
499                 ITransform3D parentToLocal = sensor.getReadoutElectrodes(carrier).getParentToLocal();
500                 ITransform3D localToGlobal = sensor.getReadoutElectrodes(carrier).getLocalToGlobal();
501                 for (int physicalChannel = 0; physicalChannel < 640; physicalChannel++) {
502                     Hep3Vector localStripPosition = strips.getCellPosition(physicalChannel);
503                     Hep3Vector stripPosition = parentToLocal.transformed(localStripPosition);
504                     Hep3Vector globalStripPosition = localToGlobal.transformed(stripPosition);
505                     positionMap.put(physicalChannel, globalStripPosition);
506                 }
507             }
508         }
509         return positionMap;
510     }

512     private double getClusterPairMass(Cluster clu1, Cluster clu2) {
513         double x0 = clu1.getPosition()[0];
514         double y0 = clu1.getPosition()[1];
515         double z0 = clu1.getPosition()[2];
516         double x1 = clu2.getPosition()[0];
517         double y1 = clu2.getPosition()[1];
518         double z1 = clu2.getPosition()[2];
519         double e0 = clu1.getEnergy();
520         double e1 = clu2.getEnergy();
521         double xlen0 = sqrt(x0 * x0 + y0 * y0 + z0 * z0);
522         double xlen1 = sqrt(x1 * x1 + y1 * y1 + z1 * z1);
523         Hep3Vector p0 = new BasicHep3Vector(x0 / xlen0 * e0, y0 / xlen0 * e0, z0 / xlen0 * e0);
524         Hep3Vector p1 = new BasicHep3Vector(x1 / xlen1 * e1, y1 / xlen1 * e1, z1 / xlen1 * e1);
525 
526         double esum = sqrt(p1.magnitudeSquared()) + sqrt(p0.magnitudeSquared());
527         double pxsum = p1.x() + p0.x();
528         double pysum = p1.y() + p0.y();
529         double pzsum = p1.z() + p0.z();
530 
531         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
532         double evtmass = esum * esum - psum * psum;
533 
534         if (evtmass > 0) {
535             return Math.sqrt(evtmass);
536         } else {
537             return -99;
538         }
539     }

541     private double getMomentum(Track trk) {
542 
543         double px = trk.getTrackStates().get(0).getMomentum()[0];
544         double py = trk.getTrackStates().get(0).getMomentum()[1];
545         double pz = trk.getTrackStates().get(0).getMomentum()[2];
546         return Math.sqrt(px * px + py * py + pz * pz);
547     }

554     void redraw() {
555         hitCountDrawPlot.reset();
556         hitCountDrawPlot.add(hitCountFillPlot);
557 
558         hitCountFillPlot.reset();
559 
560         clusterCountDrawPlot.reset();
561         clusterCountDrawPlot.add(clusterCountFillPlot);
562 
563         clusterCountFillPlot.reset();
564 
565         occupancyDrawPlot.reset();
566         for (int id = 0; id < (47 * 11); id++) {
567             int row = EcalMonitoringUtilities.getRowFromHistoID(id);
568             int column = EcalMonitoringUtilities.getColumnFromHistoID(id);
569             double mean = occupancyFill[id] / NoccupancyFill;
570 
571             occupancyFill[id] = 0;
572             if ((row != 0) && (column != 0) && (!EcalMonitoringUtilities.isInHole(row, column))) {
573                 occupancyDrawPlot.fill(column, row, mean);
574             }
575         }

579     public double getClusterPairMass(Cluster[] pair) {
580         double x0 = pair[0].getPosition()[0];
581         double y0 = pair[0].getPosition()[1];
582         double z0 = pair[0].getPosition()[2];
583         double x1 = pair[1].getPosition()[0];
584         double y1 = pair[1].getPosition()[1];
585         double z1 = pair[1].getPosition()[2];
586         double e0 = pair[0].getEnergy();
587         double e1 = pair[1].getEnergy();
588         double xlen0 = sqrt(x0 * x0 + y0 * y0 + z0 * z0);
589         double xlen1 = sqrt(x1 * x1 + y1 * y1 + z1 * z1);
590         Hep3Vector p0 = new BasicHep3Vector(x0 / xlen0 * e0, y0 / xlen0 * e0, z0 / xlen0 * e0);
591         Hep3Vector p1 = new BasicHep3Vector(x1 / xlen1 * e1, y1 / xlen1 * e1, z1 / xlen1 * e1);
592 
593         double esum = sqrt(p1.magnitudeSquared()) + sqrt(p0.magnitudeSquared());
594         double pxsum = p1.x() + p0.x();
595         double pysum = p1.y() + p0.y();
596         double pzsum = p1.z() + p0.z();
597 
598         double psum = Math.sqrt(pxsum * pxsum + pysum * pysum + pzsum * pzsum);
599         double evtmass = esum * esum - psum * psum;
600 
601         if (evtmass > 0) {
602             return Math.sqrt(evtmass);
603         } else {
604             return -99;
605         }
606     }
