4   import java.util.logging.Level;
5   import java.util.logging.Logger;
6   
7   import org.hps.online.recon.properties.Property;
8   import org.hps.record.et.EtConnection;
9   import org.jlab.coda.et.EtConstants;
10  import org.jlab.coda.et.EtStationConfig;
11  import org.jlab.coda.et.EtSystem;
12  import org.jlab.coda.et.EtSystemOpenConfig;
13  import org.jlab.coda.et.enums.Mode;
14  import org.jlab.coda.et.exception.EtClosedException;
15  import org.jlab.coda.et.exception.EtDeadException;
16  import org.jlab.coda.et.exception.EtException;
17  import org.jlab.coda.et.exception.EtExistsException;
18  import org.jlab.coda.et.exception.EtTooManyException;

20  /**
21   * Create an ET connection for a pool of stations to process
22   * reconstruction jobs in parallel.
23   */
24  class EtParallelStation extends EtConnection {
25  
26      private static Logger LOGGER = Logger.getLogger(EtParallelStation.class.getPackage().getName());
27  
28      private static final int STATION_POSITION = 1;
29  
30      /**
31       * Create an ET connection from station properties
32       *
33       * If there is an existing station with the same name, it will be woken
34       * up and removed before a new one is created.
35       *
36       * @param props The station properties
37       * @throws EtException
38       * @throws EtTooManyException
39       * @throws EtExistsException
40       * @throws EtClosedException
41       * @throws EtDeadException
42       * @throws IOException
43       * @throws Exception If there is an error opening the ET system
44       */
45      EtParallelStation(StationProperties props)
46              throws EtException, IOException, EtDeadException,
47              EtClosedException, EtExistsException, EtTooManyException {
48  
49          Property<String> host = props.get("et.host");
50          Property<String> buffer = props.get("et.buffer");
51          Property<Integer> port = props.get("et.port");
52          Property<Integer> connAttempts = props.get("et.connectionAttempts");
53          Property<Integer> logLevel = props.get("et.logLevel");
54          Property<Integer> prescale = props.get("et.prescale");
55          Property<String> stationName = props.get("et.stationName");
56          Property<Integer> waitMode = props.get("et.mode");
57          Property<Integer> waitTime = props.get("et.waitTime");
58          Property<Integer> chunk = props.get("et.chunk");
59  
60          LOGGER.config("Opening ET system: " + host.value() + ":" + port.value() + buffer.value());
61          EtSystemOpenConfig etConfig =
62                  new EtSystemOpenConfig(buffer.value(), host.value(), port.value());
63  
64          // Connection attempt loop
65          for (int i = 1; i <= connAttempts.value(); i++) {
66              LOGGER.config("Attempting ET connection: " + i);
67              try {
68                  sys = new EtSystem(etConfig, logLevel.value());
69                  sys.open();
70  
71              } catch (Exception e) {
72                  e.printStackTrace();
73              }
74              if (sys.alive()) {
75                  LOGGER.config("Connection successful!");
76                  break;
77              }
78  
79              try {
80                  Thread.sleep(1000L);
81              } catch (InterruptedException e) {
82                  LOGGER.info("Connection attempt was interrupted!");
83                  break;
84              }
85          }
86  
87          // Failed to connect to the ET system
88          if (!sys.alive()) {
89              RuntimeException rte = new RuntimeException(
90                      "Failed to connect to ET system after "
91                      + connAttempts.value() + " attempts");
92              LOGGER.log(Level.SEVERE, rte.getMessage(), rte);
93              throw rte;
94          }
95  
96          // See if there is an existing ET station
97          try {
98              stat = sys.stationNameToObject(stationName.value());
99          } catch (Exception e) {
100             LOGGER.log(Level.WARNING, "Error while attempting to find station", e);
101         }
102 
103         // Raise an error if the station exists already.
104         if (stat != null) {
105             throw new RuntimeException("ET station already exists: " + stationName.value());
106         }
107 
108         // Create the new ET station
109         final EtStationConfig stationConfig = new EtStationConfig();
110         stationConfig.setFlowMode(EtConstants.stationParallel);
111         stationConfig.setBlockMode(EtConstants.stationBlocking);
112         stationConfig.setSelectMode(EtConstants.stationSelectRRobin);
113         stationConfig.setRestoreMode(EtConstants.stationRestoreGC);
114         stat = sys.createStation(stationConfig, stationName.value(), STATION_POSITION, EtConstants.end);
115         if (prescale.value() > 0) {
116             stationConfig.setPrescale(prescale.value());
117         }
118 
119         att = sys.attach(stat);
120 
121         LOGGER.info("Created ET station: name=" + stat.getName() + "; pos=" + sys.getStationPosition(stat)
122                 + "; ppos: " + sys.getStationParallelPosition(stat));
123 
124         this.waitMode = Mode.getMode(waitMode.value());
125         this.waitTime = waitTime.value();
126         this.chunkSize = chunk.value();
127     }
128 
129     /**
130      * Cleanup the connection by detaching it.
131      */
132     public void cleanup() {
133         try {
134             if (sys!= null && sys.alive()) {
135                 sys.wakeUpAttachment(att);
136                 sys.detach(att);
137                 sys.removeStation(stat);
138                 sys.close();
139             }
140         } catch (final Exception e) {
141             e.printStackTrace();
142         }
143         // Can't use a logger when this is called in the shutdown hook
144         System.out.println("Cleaned up ET station");
145     }
146 }
