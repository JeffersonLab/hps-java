5   import org.hps.online.recon.CommandHandler;
6   import org.hps.online.recon.CommandResult;

8   import org.hps.online.recon.CommandResult.Error;
9   import org.hps.online.recon.Server;
10  import org.hps.online.recon.StationProcess;
11  import org.json.JSONArray;
12  import org.json.JSONObject;
13  
14  /**
15   * Handle the <i>list</i> command.
16   */
17  class ListCommandHandler extends CommandHandler {
18  
19      ListCommandHandler(Server server) {
20          super(server);
21      }
22  
23      public String getCommandName() {
24          return "list";
25      }
26  
27      public CommandResult execute(JSONObject parameters) {
28          CommandResult res = null;
29          List<Integer> ids = getStationIDs(parameters);
30          JSONArray arr = new JSONArray();
31          if (ids.size() == 0) {
32              // Return info on all stations.
33              for (StationProcess station : mgr.getStations()) {
34                  JSONObject jo = station.toJSON();
35                  arr.put(jo);
36              }
37          } else {
38              // Return info on selected station IDs.
39              for (Integer id : ids) {
40                  StationProcess station = mgr.find(id);
41                  if (station == null) {
42                      // One of the station IDs is invalid.  Just return message about the first bad one.
43                      res = new Error("Station with this ID does not exist: " + id);
44                      break;
45                  } else {
46                      JSONObject jo = station.toJSON();
47                      arr.put(jo);
48                  }
49              }
50          }
51          if (res == null) {
52              res = new GenericResult(arr);
53          }
54          return res;
55      }
56  }
