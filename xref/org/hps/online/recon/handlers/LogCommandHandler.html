3   import java.io.File;
4   import java.util.List;
5   
6   import org.apache.commons.io.input.Tailer;
7   import org.hps.online.recon.CommandHandler;
8   import org.hps.online.recon.CommandResult;
9   import org.hps.online.recon.CommandResult.LogStreamResult;
10  import org.hps.online.recon.Server;
11  import org.hps.online.recon.SimpleLogListener;
12  import org.json.JSONObject;
13  
14  /**
15   * Handle the <i>log</i> command for tailing a station's log file
16   */
17  class LogCommandHandler extends CommandHandler {
18  
19      LogCommandHandler(Server server) {
20          super(server);
21      }
22  
23      public CommandResult execute(JSONObject jo) {
24          CommandResult res = null;
25          List<Integer> ids = getStationIDs(jo);
26          if (ids.size() == 1) {
27              int id = ids.get(0);
28              Long delayMillis = 1000L;
29              if (jo.has("delayMillis")) {
30                  delayMillis = jo.getLong("delayMillis");
31              }
32              SimpleLogListener listener = new SimpleLogListener();
33              Tailer tailer = mgr.getLogTailer(id, listener, delayMillis);
34              File logFile = mgr.find(id).getLogFile();
35              res = new LogStreamResult(tailer, listener, logFile);
36          } else if (ids.size() > 1) {
37              res = new CommandResult.Error("Multiple station IDs not supported for log command.");
38          } else if (ids.size() == 0) {
39              res = new CommandResult.Error("No station IDs were given in parameters.");
40          }
41          return res;
42      }
43  
44      @Override
45      public String getCommandName() {
46          return "log";
47      }
48  }
