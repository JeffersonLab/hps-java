1   package org.hps.online.recon.handlers;
2   
3   import org.hps.online.recon.CommandHandler;
4   import org.hps.online.recon.CommandResult;
5   import org.hps.online.recon.CommandResult.JSONResult;
6   import org.hps.online.recon.CommandResult.Success;
7   import org.hps.online.recon.Server;
8   import org.json.JSONObject;
9   
10  /**
11   * Handle the <i>config</i> command
12   */
13  class ConfigCommandHandler extends CommandHandler {
14  
15      ConfigCommandHandler(Server server) {
16          super(server);
17      }
18  
19      public CommandResult execute(JSONObject parameters) {
20          CommandResult res = null;
21          if (parameters.length() == 0) {
22              logger.config("Returning existing station config.");
23              res = new JSONResult(server.getStationProperties().toJSON());
24          } else {
25              logger.config("Loading new station config: " + parameters.toString());
26              server.getStationProperties().fromJSON(parameters);
27              logger.config("New config loaded.");
28              res = new Success("Loaded new station config. Create a new station to use it.");
29          }
30          return res;
31      }
32  
33      @Override
34      public String getCommandName() {
35          return "config";
36      }
37  }
