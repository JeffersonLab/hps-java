3   import java.util.Arrays;
4   import java.util.List;

6   import org.hps.online.recon.CommandHandler;
7   import org.hps.online.recon.CommandResult;
8   import org.hps.online.recon.CommandResult.Error;
9   import org.hps.online.recon.CommandResult.Success;
10  import org.hps.online.recon.Server;
11  import org.hps.online.recon.StationProcess;
12  import org.json.JSONObject;
13  
14  /**
15   * Handle the <i>start</i> command.
16   */
17  class StartCommandHandler extends CommandHandler {
18  
19      StartCommandHandler(Server server) {
20          super(server);
21      }
22  
23      public CommandResult execute(JSONObject parameters) {
24          CommandResult res = null;
25          logger.info("Starting stations...");
26          List<Integer> ids = getStationIDs(parameters);
27          List<StationProcess> stations = null;
28          if (ids.size() == 0) {
29              logger.info("All inactive stations will be started");
30              stations = mgr.getInactiveStations();
31          } else {
32              stations = mgr.find(ids);
33              logger.info("Starting station IDs: " + Arrays.toString(ids.toArray()));
34          }
35          int count = stations.size();
36          int started = mgr.startStations(stations);
37  
38          if (started < count) {
39              res = new Error("Failed to start some stations or some were already active");
40          } else {
41              res = new Success("Started " + count + " stations successfully.");
42          }
43          logger.info("Done starting stations!");
44          return res;
45      }
46  
47      @Override
48      public String getCommandName() {
49          return "start";
50      }
51  }
