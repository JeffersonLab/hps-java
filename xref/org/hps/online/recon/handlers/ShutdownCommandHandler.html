1   package org.hps.online.recon.handlers;
2   
3   import org.hps.online.recon.CommandHandler;
4   import org.hps.online.recon.CommandResult;
5   import org.hps.online.recon.Server;
6   import org.json.JSONObject;
7   
8   /**
9    * Handle the <i>shutdown</i> command which shuts down a running server instance
10   */
11  class ShutdownCommandHandler extends CommandHandler {
12  
13      ShutdownCommandHandler(Server server) {
14          super(server);
15      }
16  
17      public CommandResult execute(JSONObject parameters) {
18          return new CommandResult.Shutdown(parameters.getInt("wait"));
19      }
20  
21      @Override
22      public String getCommandName() {
23          return "shutdown";
24      }
25  }
