1   package org.hps.online.recon.handlers;
2   
3   import java.io.File;
4   import java.io.IOException;
5   import java.util.logging.Level;
6   
7   import org.hps.online.recon.CommandHandler;
8   import org.hps.online.recon.CommandResult;
9   import org.hps.online.recon.CommandResult.Success;
10  import org.hps.online.recon.CommandResult.Error;
11  import org.hps.online.recon.Server;
12  import org.json.JSONObject;
13  
14  /**
15   * Handle the <i>save</i> command which writes the contents
16   * of the current AIDA tree to a local ROOT or AIDA file
17   */
18  class SaveCommandHandler extends CommandHandler {
19  
20      SaveCommandHandler(Server server) {
21          super(server);
22      }
23  
24      public String getCommandName() {
25          return "save";
26      }
27  
28      public CommandResult execute(JSONObject parameters) throws CommandException {
29          if (!parameters.has("filename")) {
30              throw new CommandException("Missing required parameter: filename");
31          }
32          String filename = parameters.getString("filename");
33          server.getLogger().info("Saving AIDA plots to: " + filename);
34          try {
35              server.getAggregator().save(new File(filename));
36              return new Success("Saved plots to: " + filename);
37          } catch (IOException ioe) {
38              server.getLogger().log(Level.SEVERE, "Failed to save plots to: " + filename, ioe);
39              return new Error("Failed to save plots to: " + filename);
40          }
41      }
42  }
