3   import java.util.Date;
4   
5   /**
6    * Object posted to the event bus indicating
7    * end of run occurred.
8    */
9   public class EndRun {
10  
11      Integer run = null;
12      Date date = null;
13  
14      EndRun(Integer run, Date date) {
15          if (run == null) {
16              throw new IllegalArgumentException("The run argument is null");
17          }
18          this.run = run;
19          this.date = date;
20          if (this.date == null) {
21              this.date = new Date();
22          }
23      }
24  
25      Integer getRun() {
26          return run;
27      }
28  
29      Date getDate() {
30          return date;
31      }
32  
33  }
