3   import java.util.logging.Level;
4   
5   import org.hps.job.JobManager;
6   import org.hps.record.LCSimEventBuilder;
7   import org.hps.record.evio.EvioEventUtilities;
8   import org.jlab.coda.jevio.EvioEvent;
9   import org.lcsim.event.EventHeader;
10  
11  import com.google.common.eventbus.Subscribe;
12  
13  /**
14   * Receive EVIO events, build raw events, and perform physics reconstruction.
15   */
16  public class EvioListener {
17  
18      private OnlineEventBus eventbus;
19      private LCSimEventBuilder builder;
20      private JobManager mgr;
21  
22      EvioListener(OnlineEventBus eventbus) {
23          this.eventbus = eventbus;
24          this.mgr = eventbus.getStation().getJobManager();
25          this.builder = eventbus.getStation().getEventBuilder();
26      }
27  
28      /**
29       * Receive an EVIO event, convert it to raw LCIO using the
30       * event builder, and then process it using the job manager
31       * to perform reconstruction. The LCIO event is then posted
32       * to the event bus.
33       *
34       * @param evioEvent The input EVIO event
35       */
36      @Subscribe
37      public void receiveEvioAndPostLcio(EvioEvent evioEvent) {
38          try {
39              /*
40              eventbus.getLogger().info("Station " + eventbus.getStation().getStationName()
41                      + " processing EVIO event to LCIO: "
42                      + EvioEventUtilities.getEventIdData(evioEvent)[0]);*/
43              if (EvioEventUtilities.isPhysicsEvent(evioEvent)) {
44                  builder.readEvioEvent(evioEvent);
45                  eventbus.getLogger().fine("Received EVIO event: "
46                          + EvioEventUtilities.getEventIdData(evioEvent)[0]);
47                  EventHeader lcioEvent = builder.makeLCSimEvent(evioEvent);
48                  eventbus.getLogger().fine("Built LCIO event: " + lcioEvent.getEventNumber());
49                  mgr.processEvent(lcioEvent);
50                  eventbus.getLogger().fine("Processed LCIO event: " + lcioEvent.getEventNumber());
51                  eventbus.post(lcioEvent);
52              }
53          } catch (Exception e) {
54              eventbus.getLogger().log(Level.SEVERE, "Error converting EVIO to LCIO", e);
55              eventbus.post(new EventProcessingError(e, false));
56          }
57      }
58  }
