1   package org.hps.online.recon.eventbus;

3   import java.util.Date;
4   
5   import org.hps.conditions.database.DatabaseConditionsManager;
6   import org.hps.online.recon.properties.Property;
7   import org.hps.record.evio.EventTagConstant;
8   import org.hps.record.evio.EvioEventUtilities;
9   import org.jlab.coda.jevio.BaseStructure;
10  import org.jlab.coda.jevio.EvioEvent;
11  import org.lcsim.conditions.ConditionsManager;
12  import org.lcsim.conditions.ConditionsManager.ConditionsNotFoundException;
13  
14  import com.google.common.eventbus.Subscribe;
15  
16  /**
17   * Activate lcsim conditions system from EVIO data
18   * on the event bus.
19   */
20  public class ConditionsListener {
21  
22      private OnlineEventBus eventbus;
23      private Property<String> detectorProp = null;
24      private Integer currentRun = null;
25      private DatabaseConditionsManager mgr;
26  
27      ConditionsListener(OnlineEventBus eventbus) {
28          this.eventbus = eventbus;
29          detectorProp = eventbus.getStation().getProperties().get("lcsim.detector");
30          if (!detectorProp.valid()) {
31              throw new IllegalArgumentException("The detector property is not valid");
32          }
33          mgr = (DatabaseConditionsManager) ConditionsManager.defaultInstance();
34      }
35  
36      @Subscribe
37      public void checkConditions(EvioEvent evioEvent) {
38  
39          if (EventTagConstant.PRESTART.matches(evioEvent)) {
40              currentRun = EvioEventUtilities.getControlEventData(evioEvent)[1];
41          } else {
42              final BaseStructure headBank = EvioEventUtilities.getHeadBank(evioEvent);
43              if (headBank != null) {
44                  currentRun = headBank.getIntData()[1];
45              }
46          }
47  
48          if (currentRun != null && !currentRun.equals(mgr.getRun()) && !mgr.isFrozen()) {
49              try {
50                  eventbus.getLogger().info("Setting conditions from EVIO: " + detectorProp.value() + ":" + currentRun);
51                  ConditionsManager.defaultInstance().setDetector(detectorProp.value(), currentRun);
52              } catch (ConditionsNotFoundException e) {
53                  // Post fatal error because conditions for the run are not found or detector name is bad
54                  eventbus.post(new EventProcessingError(e, true));
55              }
56          }
57  
58          try {
59              if (EvioEventUtilities.isEndEvent(evioEvent)) {
60                  eventbus.post(new EndRun(currentRun, new Date()));
61              }
62          } catch (Exception e) {
63              eventbus.post(new EventProcessingError(e, false));
64          }
65      }
66  }
