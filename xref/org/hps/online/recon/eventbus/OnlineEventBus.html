6   import org.hps.online.recon.Station;

8   import org.jlab.coda.et.EtEvent;

11  import com.google.common.eventbus.EventBus;
12  import com.google.common.eventbus.Subscribe;

14  /**
15   * An event bus to read ET events, convert to EVIO, and generate reconstructed
16   * LCIO events using an event builder and job manager.
17   *
18   * Special objects are used for error handling, signaling end of run, and stopping
19   * the ET event loop.
20   */
21  public class OnlineEventBus extends EventBus {
22  
23      private EtConnection conn;
24      private Station station;
25  
26      private boolean halt;
27  
28      private final Logger logger = Logger.getLogger(OnlineEventBus.class.getPackage().getName());
29  
30      public OnlineEventBus(Station station) {
31          logger.config("Initializing online event bus for station: " + station.getStationName());
32          this.station = station;
33          this.conn = this.station.getEtConnection();
34          register(this);
35          register(new ConditionsListener(this));
36          register(new EtListener(this));
37          register(new EvioListener(this));
38          register(new LcioListener(this));
39      }
40  
41      /**
42       * Continuously post ET events to drive the event bus
43       * @throws Exception If there is an error reading ET events
44       */
45      public void loop() {
46  
47          halt = false;
48  
49          post(new Start());
50  
51          logger.info("Event loop starting");
52          while (true) {
53              EtEvent[] events = null;
54              try {
55                  logger.finest("Reading ET events");
56                  events = conn.readEtEvents();
57                  for (EtEvent event : events) {
58                      logger.fine("Read ET event: " + event.getId());
59                      this.post(event);
60                  }
61                  conn.getEtSystem().dumpEvents(conn.getEtAttachment(), events);
62              } catch (EtWakeUpException e) {
63                  // This is used as an external signal by the server to stop processing
64                  post(new Stop("ET wake up received"));
65              } catch (Exception e) {
66                  // Errors when reading ET data are considered fatal
67                  post(new EventProcessingError(e, true));
68              }
69              if (halt) {
70                  // Halt flag will be set if a stop was received
71                  break;
72              }
73          }
74  
75          // Activate end of data hook
76          try {
77              station.getJobManager().getDriverAdapter().finish(null);
78          } catch (Exception e) {
79              e.printStackTrace();
80          }
81  
82          logger.info("Event loop exiting");
83      }
84  
85      @Subscribe
86      public void receiveStop(Stop stop) {
87          try {
88              logger.info("Stopping event processing: " + stop.getReason());
89              this.halt = true;
90          } catch (Exception e) {
91              logger.log(Level.WARNING, "Problem receiving stop command", e);
92          }
93      }
94  
95      @Subscribe
96      public void receiveError(EventProcessingError e) {
97          try {
98              if (e.fatal()) {
99                  logger.log(Level.SEVERE, "Fatal error occurred -- event processing will stop", e.getException());
100                 post(new Stop("Fatal error"));
101             } else {
102                 logger.log(Level.WARNING, "A non-fatal error occurred -- processing will continue", e.getException());
103             }
104         } catch (Exception ex) {
105             logger.log(Level.SEVERE, "Problem processing error", ex);
106         }
107     }
108 
109 
110     @Subscribe
111     public void receiveStart(Start start) {
112         try {
113             logger.info("Received start: " + start.getDate().toString());
114 
115             logger.info("Activating start on drivers");
116             this.station.getJobManager().getDriverAdapter().start(null);
117 
118         } catch (Exception e) {
119             logger.log(Level.WARNING, "Problem receiving the Start event", e);
120         }
121     }
122 
123     @Subscribe
124     public void receiveEndRun(EndRun end) {
125         try {
126             logger.info("Run " + end.getRun() + " ended at: " + end.getDate());
127             post(new Stop("Run ended"));
128         } catch (Exception e) {
129             logger.log(Level.WARNING, "Error receiving end run", e);
130         }
131     }
132 
133     /**
134      * Get the logger for the event bus
135      * @return The logger for the event bus
136      */
137     Logger getLogger() {
138         return this.logger;
139     }
140 
141     /**
142      * Get the {@link org.hps.online.recon.Station} associated with this event bus
143      * @return The {@link org.hps.online.recon.Station} associated with this event bus
144      */
145     Station getStation() {
146         return this.station;
147     }
148 }
