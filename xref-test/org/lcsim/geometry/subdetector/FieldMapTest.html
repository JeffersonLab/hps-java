48      public void testFieldMap() throws Exception {
49          GeometryReader geometryReader = new GeometryReader();
50          InputStream in = this.getClass().getResourceAsStream("/org/lcsim/geometry/subdetector/FieldMapTest.xml");
51          Detector det = geometryReader.read(in);
52          FieldMap fieldmap = det.getFieldMap();
53          double[] pos = new double[3];
54          double[] field = null;
55          for (double z = 0.; z < 1000.; z += 10.) {
56              pos[2] = z;
57              field = fieldmap.getField(pos);
58              System.out.println("B-field at z = " + z + " is [ " + field[0] + ", " + field[1] + ", " + field[2] + "] ");
59          }
60      }
