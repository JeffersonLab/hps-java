30      /**
31       * Convert EVIO from Engineering Run 2015 to LCIO with detector name specified from the command line,
32       * and also run the reconstruction within the job.  The run number from the EVIO data header will activate
33       * the conditions system.  Conditions are checked for validity against their expected collection IDs.
34       */
35      public void testEvioToLcioEngRun2015() {
36          File inputFile = TestUtil.downloadTestFile("run5772_integrationTest.evio");
37          File outputFile = new TestOutputFile(ConditionsIT.class, "EvioToLcioEngRun2015");
38          String args[] = {"-r", "-x", "/org/hps/steering/test/EngRun2015FullRecon_CondCheck.lcsim", "-d",
39                  "HPS-EngRun2015-Nominal-v3", "-D", "outputFile=" + outputFile.getPath(), "-n", "2",
40                  inputFile.getPath()};
41          EvioToLcio.main(args);
42      }

44      /**
45       * Run the physics reconstruction on LCIO data from Engineering Run 2015.  The run number and detector
46       * from the LCIO file will activate the conditions.  Condition sets are checked for validity against their
47       * expected collection IDs.
48       */
49      public void testLcioReconEngRun2015() {
50          File inputFile = TestUtil.downloadTestFile("run_5772_data_only.slcio");
51          File outputFile = new TestOutputFile(ConditionsIT.class, "LcioRecon");
52          JobManager job = new JobManager();
53          job.addVariableDefinition("outputFile", outputFile.getPath());
54          job.addInputFile(inputFile);
55          job.setup("/org/hps/steering/test/EngRun2015FullRecon_CondCheck.lcsim");
56          job.setNumberOfEvents(2);
57          job.run();
58      }

106     /**
107      * Run MC reconstruction on LCIO output from readout simulation.  The detector and run number
108      * are specified to the job manager in the conditions setup.  Conditions are checked for
109      * validity against their expected collection IDs.
110      */
111     public void testMCReconUserConditions() {
112         File inputFile = TestUtil.downloadTestFile("tritrig_EngRun2015_readout.slcio");
113         File outputFile = new TestOutputFile(ConditionsIT.class, "MCReconDefaultConditions");
114         JobManager job = new JobManager();
115         DatabaseConditionsManagerSetup cond = new DatabaseConditionsManagerSetup();
116         cond.setDetectorName("HPS-EngRun2015-Nominal-v5-0-fieldmap");
117         cond.setRun(5772);
118         cond.setFreeze(true);
119         job.setConditionsSetup(cond);
120         job.addVariableDefinition("outputFile", outputFile.getPath());
121         job.addInputFile(inputFile);
122         job.setup("/org/hps/steering/test/EngineeringRun2015FullReconMC_CondCheck.lcsim");
123         job.setNumberOfEvents(2);
124         job.run();
125     }
