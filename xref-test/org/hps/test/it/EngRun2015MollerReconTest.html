123         protected void process(EventHeader event) {
124             super.process(event);
125             for (String vertexCollectionName : vertexCollectionNames) {
126 
127                 List<Vertex> vertices = event.get(Vertex.class, vertexCollectionName);
128                 for (Vertex v : vertices) {
129                     aida.tree().cd("/");
130                     aida.tree().mkdirs(vertexCollectionName);
131                     aida.tree().cd(vertexCollectionName);
132                     ReconstructedParticle rp = v.getAssociatedParticle();
133                     int type = rp.getType();
134                     boolean isGbl = TrackType.isGBL(type);
135                     // require GBL tracks in vertex
136                     if (isGbl) {
137                         List<ReconstructedParticle> parts = rp.getParticles();
138                         ReconstructedParticle rp1 = parts.get(0);
139                         ReconstructedParticle rp2 = parts.get(1);
140                         // basic sanity check here, remove full energy electrons (fee)
141                         if (rp1.getMomentum().magnitude() > 1.5 * beamEnergy || rp2.getMomentum().magnitude() > 1.5 * beamEnergy) {
142                             continue;
143                         }
144                         // require both reconstructed particles to have a track and a cluster
145                         if (rp1.getClusters().size() != 1) {
146                             continue;
147                         }
148                         if (rp2.getClusters().size() != 1) {
149                             continue;
150                         }
151                         if (rp1.getTracks().size() != 1) {
152                             continue;
153                         }
154                         if (rp2.getTracks().size() != 1) {
155                             continue;
156                         }
157                         Track t1 = rp1.getTracks().get(0);
158                         Track t2 = rp2.getTracks().get(0);
159                         Cluster c1 = rp1.getClusters().get(0);
160                         Cluster c2 = rp2.getClusters().get(0);
161                         double deltaT = ClusterUtilities.getSeedHitTime(c1) - ClusterUtilities.getSeedHitTime(c2);
162                         // require cluster times to be coincident within 2 ns
163                         if (abs(deltaT) > 2.0) {
164                             continue;
165                         }
166                         // require momentum sum to equal beam energy +-
167                         double psum = rp1.getMomentum().magnitude() + rp2.getMomentum().magnitude();
168                         if (psum < psumMin || psum > psumMax) {
169                             continue;
170                         }
171                         //rotate into physiscs frame of reference
172                         Hep3Vector rprot = VecOp.mult(BEAM_AXIS_ROTATION, rp.getMomentum());
173                         Hep3Vector p1rot = VecOp.mult(BEAM_AXIS_ROTATION, rp1.getMomentum());
174                         Hep3Vector p2rot = VecOp.mult(BEAM_AXIS_ROTATION, rp2.getMomentum());
175                         double theta1 = Math.acos(p1rot.z() / p1rot.magnitude());
176                         double theta2 = Math.acos(p2rot.z() / p2rot.magnitude());
177                         double thetasum = theta1 + theta2;
178                         // cut on thetasum
179                         if (thetasum > thetaSumCut) {
180                             continue;
181                         }
182                         // cut on Moller pX
183                         if (abs(rprot.x()) > 0.01) {
184                             continue;
185                         }
186                         // cut on Moller pY
187                         if (abs(rp.getMomentum().y()) > .01) {
188                             continue;
189                         }
190                         double t1ChisqNdf = t1.getChi2() / t1.getNDF();
191                         double t2ChisqNdf = t2.getChi2() / t2.getNDF();
192 
193                         double t1ChisqProb = ChisqProb.gammp(t1.getNDF(), t1.getChi2());
194                         double t2ChisqProb = ChisqProb.gammp(t2.getNDF(), t2.getChi2());
195                         // used to cut on prob < 0.995, which corresponds to roughly 3.4
196                         // change this to a cut on chi-squared/dof which people are more familiar with.
197                         // Omar currently cuts on chi-squared <40(!), irrespective of 5 or 6 hit tracks
198                         // let's start at chisq/dof of 8
199                         if (t1ChisqNdf > trackChi2NdfCut) {//(t1ChisqProb > 0.995) {
200                             continue;
201                         }
202                         if (t2ChisqNdf > trackChi2NdfCut) {//(t2ChisqProb > 0.995) {
203                             continue;
204                         }
205                         // all cuts passed, let's fill some histograms
206                         Hep3Vector pos = v.getPosition();
207                         double p1 = rp1.getMomentum().magnitude();
208                         double p2 = rp2.getMomentum().magnitude();
209                         if (vertexCollectionName.equals("UnconstrainedMollerVertices")) {
210                             invMassHist_UnconstrainedMollerVertices.fill(rp.getMass());
211                             pHist_UnconstrainedMollerVertices.fill(rp.getMomentum().magnitude());
212                             pxHist_UnconstrainedMollerVertices.fill(rprot.x());
213                             pyHist_UnconstrainedMollerVertices.fill(rprot.y());
214                             pzHist_UnconstrainedMollerVertices.fill(rprot.z());
215                             trkpHist_UnconstrainedMollerVertices.fill(p1);
216                             trkpHist_UnconstrainedMollerVertices.fill(p2);
217                             if (isTopTrack(t1)) {
218                                 trkptopHist_UnconstrainedMollerVertices.fill(p1);
219                                 ptopvspbotHist_UnconstrainedMollerVertices.fill(p1, p2);
220                             } else {
221                                 trkpbotHist_UnconstrainedMollerVertices.fill(p1);
222                             }
223                             if (isTopTrack(t2)) {
224                                 trkptopHist_UnconstrainedMollerVertices.fill(p2);
225                             } else {
226                                 trkpbotHist_UnconstrainedMollerVertices.fill(p2);
227                             }

229                             trkNhitsHist_UnconstrainedMollerVertices.fill(t2.getTrackerHits().size());
230                             trkChisqHist_UnconstrainedMollerVertices.fill(t1.getChi2() / t1.getNDF());
231                             trkChisqHist_UnconstrainedMollerVertices.fill(t2.getChi2() / t2.getNDF());
232                             trkChisqProbHist_UnconstrainedMollerVertices.fill(t1ChisqProb);
233                             trkChisqProbHist_UnconstrainedMollerVertices.fill(t2ChisqProb);
234                             vtxXHist_UnconstrainedMollerVertices.fill(pos.x());
235                             vtxYHist_UnconstrainedMollerVertices.fill(pos.y());
236                             vtxZHist_UnconstrainedMollerVertices.fill(pos.z());
237                             vtxChisqHist_UnconstrainedMollerVertices.fill(v.getChi2());
238 
239                             p1vsp2Hist_UnconstrainedMollerVertices.fill(p1, p2);
240                             theta1vstheta2Hist_UnconstrainedMollerVertices.fill(theta1, theta2);
241                             pvsthetaHist_UnconstrainedMollerVertices.fill(p1, theta1);
242                             pvsthetaHist_UnconstrainedMollerVertices.fill(p2, theta2);
243                             xvsyHist_UnconstrainedMollerVertices.fill(pos.x(), pos.y());
244                         }
