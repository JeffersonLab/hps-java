38           * Count of clusters per event.

43           * Individual cluster energies.
44           */
45          private final IHistogram1D clusEnergyH1D = aida.histogram1D("Cluster Energy", 100, -0.5, 9.5);

47          /**
48           * Highest cluster energy in event.
49           */
50          private final IHistogram1D clusHighEnergyH1D = aida.histogram1D("Cluster Highest Energy", 100, -0.5, 9.5);
51  
52          /**
53           * Cluster X position.
54           */
55          private final IHistogram1D clusPosX = aida.histogram1D("Pos X", 750, -300, 350);
56  
57          /**
58           * Cluster Y position.
59           */
60          private final IHistogram1D clusPosY = aida.histogram1D("Pos Y", 180, -90, 90);
61  
62          /**
63           * Cluster Z position.
64           */
65          private final IHistogram1D clusPosZ = aida.histogram1D("Pos Z", 100, 1393, 1397);
66  
67          /**
68           * Number of clusters found.
69           */
70          private int clusterCount;
71  
72          /**
73           * First hit time in highest energy cluster.
74           */
75          private final IHistogram1D clusTimeH1D = aida.histogram1D("Cluster Time", 500, -0.5, 499.5);
76  
77          /**
78           * Save histograms and perform checks on statistics.
79           */
80          @Override
81          public void endOfData() {
82  
83              // Print some statistics.
84              System.out.println("clus count: " + clusterCount);
85              System.out.println("mean clus per event: " + clusCountH1D.mean());
86              System.out.println("mean clus energy: " + clusEnergyH1D.mean());
87              System.out.println("high energy clus mean: " + clusHighEnergyH1D.mean());
88              System.out.println("mean clus time: " + clusTimeH1D.mean());
89  
90              // Save plots to AIDA file.
91              try {
92                  aida.saveAs(new TestOutputFile(EcalSimReconTest.class, "plots.aida"));
93              } catch (final IOException e) {
94                  throw new RuntimeException(e);
95              }
96  
97              // Check high cluster energy mean.
98              //TestCase.assertEquals("High cluster energy does not match.", CLUS_HIGH_MEAN_E, clusHighEnergyH1D.mean(),
99              //        0.0005);
100 
101             // Check high cluster time mean.
102             TestCase.assertEquals("High cluster mean time does not match.", CLUS_MEAN_T, clusTimeH1D.mean(), 1.0);
103 
104             // Check mean number of clusters per event.
105             TestCase.assertEquals("Mean number of clusters per event does not match.", CLUS_COUNT_MEAN,
106                     clusCountH1D.mean(), 0.002);
107 
108             // Check total number of clusters.
109             TestCase.assertEquals("Number of clusters does not match.", CLUS_COUNT, clusterCount, 1);
110         }
111 
112         /**
113          * Process events and fill histograms from cluster collection.
114          */
115         @Override
116         public void process(final EventHeader event) {
117             final List<Cluster> clusters = event.get(Cluster.class, "EcalClustersCorr");
118             clusterCount += clusters.size();
119             clusCountH1D.fill(clusters.size());
120             for (final Cluster cluster : clusters) {
121                 clusEnergyH1D.fill(cluster.getEnergy());
122                 if (cluster.getEnergy() == 0) {
123                     throw new RuntimeException("Cluster has 0. energy.");
124                 }
125                 if (cluster.getCalorimeterHits().size() == 0) {
126                     throw new RuntimeException("Cluster has no hits.");
127                 }
128 
129                 final double[] position = cluster.getPosition();
130                 final double x = position[0];
131                 final double y = position[1];
132                 final double z = position[2];
133 
134                 clusPosX.fill(x);
135                 clusPosY.fill(y);
136                 clusPosZ.fill(z);
137 
138                 // Rough checks that cluster position looks reasonable.
139                 TestCase.assertTrue("Pos X " + x + " is out of range.", x > -280. && x < 350.);
140                 TestCase.assertTrue("Pos Y " + y + " is out of range.", y > -84. && y < 83.);
141                 TestCase.assertTrue("Pos Y " + y + " is in beam gap.", !(y > -25. && y < 25.));
142                 TestCase.assertTrue("Pos Z" + z + " is out of range", z > 1393 && z < 1396.2);
143             }
144 
145             final Cluster highClus = ClusterUtilities.findHighestEnergyCluster(clusters);
146             clusHighEnergyH1D.fill(highClus.getEnergy());
147             clusTimeH1D.fill(highClus.getCalorimeterHits().get(0).getTime());
148         }
149     }

181     /**
182      * Run the test.
183      *
184      * @throws Exception if there is an uncaught exception thrown
185      */
186     public void testEcalSimRecon() throws Exception {
187 
188         // Get the input events file.
189         final File readoutFile = TestUtil.downloadTestFile("EcalSimReconTest.slcio");
190 
191         // Run the recon on the readout output.
192         final File reconFile = new TestOutputFile(EcalSimReconTest.class, "recon.slcio");
193         final JobManager job = new JobManager();
194         job.addInputFile(readoutFile);
195         job.addVariableDefinition("detector", "HPS-EngRun2015-Nominal-v1");
196         job.addVariableDefinition("outputFile", reconFile.getPath().replace(".slcio", ""));
197         job.addVariableDefinition("run", RUN.toString());
198         job.setup(RECON_STEERING);
199         job.run();
200         System.out.println("recon ran on " + job.getLCSimLoop().getTotalCountableConsumed() + " events");
201 
202         // Check the recon output.
203         final LCSimLoop loop = new LCSimLoop();
204         loop.setLCIORecordSource(reconFile);
205         final EcalSimReconCheckDriver checkDriver = new EcalSimReconCheckDriver();
206         loop.add(checkDriver);
207         loop.loop(-1);
208         loop.dispose();
209         System.out.println("EcalSimReconCheckDriver ran on " + loop.getTotalCountableConsumed() + " events");
210     }
