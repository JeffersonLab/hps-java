52      public class HodoscopeTestDriver extends Driver {
53  
54          private HodoscopeDetectorElement hodoDetElem;
55          private String hitCollName = "HodoscopeHits";
56  
57          private boolean debug = true;
58  
59          protected void detectorChanged(Detector det) {
60              hodoDetElem = (HodoscopeDetectorElement) det.getSubdetector("Hodoscope").getDetectorElement();
61          }
62  
63          protected void process(EventHeader event) {
64              List<SimTrackerHit> simHits = event.get(SimTrackerHit.class, hitCollName);
65              for (SimTrackerHit simHit : simHits) {
66                  Hep3Vector posVec = simHit.getPositionVec();
67  
68                  IIdentifier hitId = simHit.getIdentifier();
69                  IDetectorElement idDetElem = hodoDetElem.findDetectorElement(hitId).get(0);
70                  if (debug)
71                      System.out.println("found DE from ID: " + idDetElem.getName());
72  
73                  IDetectorElement posDetElem = hodoDetElem.findDetectorElement(posVec);
74                  if (debug) {
75                      System.out.println("found DE from pos: " + posDetElem.getName());
76                  }
77  
78                  TestCase.assertEquals("Detector elements different from ID and position!", idDetElem, posDetElem);
79  
80                  //IGeometryInfo geom = idDetElem.getGeometry();
81  
82                  double[] scintDims = hodoDetElem.getScintillatorHalfDimensions(simHit);
83                  if (debug)
84                      System.out.println("scintDims: " + scintDims[0] + " " + scintDims[1] + " " + scintDims[2]);
85  
86                  if (debug)
87                      System.out.println();
88              }
89          }
90  
91          void setDebug(boolean debug) {
92              this.debug = debug;
93          }
94      }
