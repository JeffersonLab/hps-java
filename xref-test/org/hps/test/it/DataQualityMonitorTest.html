5   import org.hps.job.JobManager;
6   import org.hps.util.test.TestUtil;

8   import junit.framework.TestCase;

10  /**
11   * Test package for data quality monitoring of reconstructed data
12   */
13  public class DataQualityMonitorTest extends TestCase {
14  
15      private static final String CLASS_NAME = DataQualityMonitorTest.class.getSimpleName();
16      private static final File OUTPUT_DIR = new File("./target/test-output/" + CLASS_NAME);
17      private static final File OUTPUT_FILE = new File(OUTPUT_DIR.getAbsolutePath() + File.separator + CLASS_NAME + ".aida");
18      private static final String STEERING_RESOURCE = "/org/hps/steering/test/DataQualityTest.lcsim";
19  
20      public void setUp() {
21  
22          // Delete files if they already exist.
23          if (OUTPUT_FILE.exists())
24              OUTPUT_FILE.delete();
25  
26          // Create output dir.
27          OUTPUT_DIR.mkdirs();
28          if (!OUTPUT_DIR.exists())
29              throw new RuntimeException("Failed to create test output dir.");
30      }
31  
32      public void testQualityMonitor() {
33          File dataFile = TestUtil.downloadTestFile("DataQualityMonitorTest.slcio");
34          System.out.println("running data quality job with steering resource " + STEERING_RESOURCE + " ...");
35          JobManager jobManager = new JobManager();
36          jobManager.addVariableDefinition("outputFile", OUTPUT_FILE.getPath());
37          jobManager.addInputFile(dataFile);
38          jobManager.setup(STEERING_RESOURCE);
39          jobManager.run();
40          System.out.println("Done!");
41      }
42  }
