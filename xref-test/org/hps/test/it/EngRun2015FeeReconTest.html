46      private static class PlotDriver extends RefDriver {
47  
48          IHistogram1D trkChisqNdfTop = aida.histogram1D("Top Track Chisq per DoF", 100, 0., 100.);
49          IHistogram1D trkChisqProbTop = aida.histogram1D("Top Track Chisq Prob", 100, 0., 1.);
50          IHistogram1D trkNhitsTop = aida.histogram1D("Top Track Number of Hits", 7, -0.5, 6.5);
51          IHistogram1D trkMomentumTop = aida.histogram1D("Top Track Momentum", 200, 0., 3.);
52          IHistogram1D trkMomentumTop5 = aida.histogram1D("Top 5 Hit Track Momentum", 200, 0., 3.);
53          IHistogram1D trkMomentumTop6 = aida.histogram1D("Top 6 Hit Track Momentum", 200, 0., 3.);
54          IHistogram1D trkdEdXTop5 = aida.histogram1D("Top 5 Track dEdx", 100, 0., .00015);
55          IHistogram1D trkdEdXTop6 = aida.histogram1D("Top 6 Track dEdx", 100, 0., .00015);
56          IHistogram1D trkthetaTop = aida.histogram1D("Top Track theta", 100, 0.01, 0.05);
57          IHistogram1D trkX0Top = aida.histogram1D("Top Track X0", 100, -0.5, 0.5);
58          IHistogram1D trkY0Top = aida.histogram1D("Top Track Y0", 100, -5.0, 5.0);
59          IHistogram1D trkZ0Top = aida.histogram1D("Top Track Z0", 100, -1.0, 1.0);
60  
61          IHistogram1D trkChisqNdfBottom = aida.histogram1D("Bottom Track Chisq per DoF", 100, 0., 100.);
62          IHistogram1D trkChisqProbBottom = aida.histogram1D("Bottom Track Chisq Prob", 100, 0., 1.);
63          IHistogram1D trkNhitsBottom = aida.histogram1D("Bottom Track Number of Hits", 7, -0.5, 6.5);
64          IHistogram1D trkMomentumBottom = aida.histogram1D("Bottom Track Momentum", 200, 0., 3.);
65          IHistogram1D trkMomentumBottom5 = aida.histogram1D("Bottom 5 Hit Track Momentum", 200, 0., 3.);
66          IHistogram1D trkMomentumBottom6 = aida.histogram1D("Bottom 6 Hit Track Momentum", 200, 0., 3.);
67          IHistogram1D trkdEdXBottom5 = aida.histogram1D("Bottom 5 Track dEdx", 100, 0., .00015);
68          IHistogram1D trkdEdXBottom6 = aida.histogram1D("Bottom 6 Track dEdx", 100, 0., .00015);
69          IHistogram1D trkthetaBottom = aida.histogram1D("Bottom Track theta", 100, 0.01, 0.05);
70          IHistogram1D trkX0Bottom = aida.histogram1D("Bottom Track X0", 100, -0.5, 0.5);
71          IHistogram1D trkY0Bottom = aida.histogram1D("Bottom Track Y0", 100, -5.0, 5.0);
72          IHistogram1D trkZ0Bottom = aida.histogram1D("Bottom Track Z0", 100, -1.0, 1.0);
73  
74          final String finalStateParticlesColName = "OtherElectrons";
75  
76          Double beamEnergy = 1.056;
77  
78          //Set min seed energy value, default to 2015 run
79          double seedCut = 0.4; //= 0.4
80  
81          //set min cluster energy value, default to 2015 run
82          double clusterCut = 0.6;
83  
84          //minimum number of hits per cluster
85          int minHits = 3; // = 3;
86  
87          double ctMin = 40.;
88          double ctMax = 49.;
89  
90          protected void process(EventHeader event) {
91              // only keep singles triggers:
92              if (!event.hasCollection(GenericObject.class, "TriggerBank")) {
93                  return;
94              }
95              boolean isSingles = false;
96              for (GenericObject gob : event.get(GenericObject.class, "TriggerBank")) {
97                  if (!(AbstractIntData.getTag(gob) == TIData.BANK_TAG)) {
98                      continue;
99                  }
100                 TIData tid = new TIData(gob);
101                 if (tid.isSingle0Trigger() || tid.isSingle1Trigger()) {
102                     isSingles = true;
103                     break;
104                 }
105             }
106             if (!isSingles) {
107                 return;
108             }
109             if (!event.hasCollection(ReconstructedParticle.class, finalStateParticlesColName)) {
110                 return;
111             }
112             super.process(event);
113             List<ReconstructedParticle> rpList = event.get(ReconstructedParticle.class, finalStateParticlesColName);
114             for (ReconstructedParticle rp : rpList) {
115                 if (!TrackType.isGBL(rp.getType())) {
116                     continue;
117                 }
118                 if (rp.getMomentum().magnitude() > 1.5 * beamEnergy) {
119                     continue;
120                 }
121                 // require both track and cluster
122                 if (rp.getClusters().size() != 1) {
123                     continue;
124                 }
125 
126                 if (rp.getTracks().size() != 1) {
127                     continue;
128                 }
129 
130                 Track t = rp.getTracks().get(0);
131                 double p = rp.getMomentum().magnitude();
132                 Cluster c = rp.getClusters().get(0);
133                 // debug diagnostics to set cuts
134                 if (debug) {
135                     aida.cloud1D("clusterSeedHit energy").fill(ClusterUtilities.findSeedHit(c).getCorrectedEnergy());
136                     aida.cloud1D("cluster nHits").fill(c.getCalorimeterHits().size());
137                     aida.cloud2D("clusterSeedHit energy vs p").fill(p, ClusterUtilities.findSeedHit(c).getCorrectedEnergy());
138                     aida.cloud2D("cluster nHits vs p").fill(p, c.getCalorimeterHits().size());
139                     aida.cloud2D("cluster time vs p").fill(p, ClusterUtilities.getSeedHitTime(c));
140                 }
141                 double ct = ClusterUtilities.getSeedHitTime(c);
142 
143                 if (c.getEnergy() > clusterCut
144                         && ClusterUtilities.findSeedHit(c).getCorrectedEnergy() > seedCut
145                         && c.getCalorimeterHits().size() >= minHits
146                         && ct > ctMin
147                         && ct < ctMax) {
148                     double chiSquared = t.getChi2();
149                     int ndf = t.getNDF();
150                     double chi2Ndf = t.getChi2() / t.getNDF();
151                     double chisqProb = ChisqProb.gammp(ndf, chiSquared);
152                     int nHits = t.getTrackerHits().size();
153                     double dEdx = t.getdEdx();
154                     //rotate into physiscs frame of reference
155                     Hep3Vector rprot = VecOp.mult(BEAM_AXIS_ROTATION, rp.getMomentum());
156                     double theta = Math.acos(rprot.z() / rprot.magnitude());
157 
158                     // debug diagnostics to set cuts
159                     if (debug) {
160                         aida.cloud1D("Track chisq per df").fill(chiSquared / ndf);
161                         aida.cloud1D("Track chisq prob").fill(chisqProb);
162                         //aida.cloud1D("Track nHits").fill(t.getTrackerHits().size());
163                         aida.cloud1D("Track momentum").fill(p);
164                         aida.cloud1D("Track deDx").fill(t.getdEdx());
165                         aida.cloud1D("Track theta").fill(theta);
166                         aida.cloud2D("Track theta vs p").fill(theta, p);
167                         aida.cloud1D("rp x0").fill(TrackUtils.getX0(t));
168                         aida.cloud1D("rp y0").fill(TrackUtils.getY0(t));
169                         aida.cloud1D("rp z0").fill(TrackUtils.getZ0(t));
170                     }
171 
172                     double trackDataTime = TrackData.getTrackTime(TrackData.getTrackData(event, t));
173                     aida.cloud1D("track data time").fill(trackDataTime);
174                     if (isTopTrack(t)) {
175                         trkChisqNdfTop.fill(chi2Ndf);
176                         trkChisqProbTop.fill(chisqProb);
177                         trkNhitsTop.fill(nHits);
178                         trkMomentumTop.fill(p);
179                         trkthetaTop.fill(theta);
180                         trkX0Top.fill(TrackUtils.getX0(t));
181                         trkY0Top.fill(TrackUtils.getY0(t));
182                         trkZ0Top.fill(TrackUtils.getZ0(t));
183 
184                         if (nHits == 5) {
185                             trkMomentumTop5.fill(p);
186                             trkdEdXTop5.fill(dEdx);
187                         } else if (nHits == 6) {
188                             trkMomentumTop6.fill(p);
189                             trkdEdXTop6.fill(dEdx);
190                         }
191                     } else {
192                         trkChisqNdfBottom.fill(chi2Ndf);
193                         trkChisqProbBottom.fill(chisqProb);
194                         trkNhitsBottom.fill(nHits);
195                         trkMomentumBottom.fill(p);
196                         trkthetaBottom.fill(theta);
197                         trkX0Bottom.fill(TrackUtils.getX0(t));
198                         trkY0Bottom.fill(TrackUtils.getY0(t));
199                         trkZ0Bottom.fill(TrackUtils.getZ0(t));
200 
201                         if (nHits == 5) {
202                             trkMomentumBottom5.fill(p);
203                             trkdEdXBottom5.fill(dEdx);
204                         } else if (nHits == 6) {
205                             trkMomentumBottom6.fill(p);
206                             trkdEdXBottom6.fill(dEdx);
207                         }
208                     }
209                 } // End of cluster cuts
210             }
211         }
212     }
