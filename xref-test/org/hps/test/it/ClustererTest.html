231     /**
232      * Run the standard test for a Clusterer.
233      * @param clustererName The name of the Clusterer.
234      * @param cuts The cut values.
235      * @param writeLcioFile Whether or not to write an LCIO output file.
236      */
237     private void runClustererTest(ClustererTestConfig config) throws Exception {
238 
239         /**********************************************************
240          * Run the job to create clusters and write to LCIO file. *
241          **********************************************************/
242         System.out.println("testing Clusterer " + config.clustererName + " ...");
243 
244         // Configure the loop.
245         LCSimLoop loop = new LCSimLoop();
246         try {
247             // Set the input LCIO file.
248             loop.setLCIORecordSource(inputFile);
249         } catch (IOException e) {
250             throw new RuntimeException(e);
251         }
252         DatabaseConditionsManager.reset();
253 
254         // Configure the ClusterDriver and add it to the loop.
255         ClusterDriver clusterDriver = new ClusterDriver();
256         clusterDriver.setClustererName(config.clustererName);
257         if (config.cuts != null) {
258             clusterDriver.setCuts(config.cuts);
259         }
260         clusterDriver.setInputHitCollectionName("EcalHits");
261         clusterDriver.setOutputClusterCollectionName(config.getClusterCollectionName());
262         clusterDriver.setRaiseErrorNoHitCollection(true);
263         loop.add(clusterDriver);
264 
265         // This Driver checks the Clusters for validity.
266         loop.add(new ClusterCheckDriver(config));
267 
268         // This Driver makes plots from the Cluster data.
269         loop.add(new ClusterPlotsDriver(config));
270 
271         // Setup writing of LCIO file with the output Clusters from this clustering algorithm.
272         String outputFilePath = testOutputDir.getPath() + File.separator + config.clustererName + ".slcio";
273         if (config.writeLcioFile) {
274             loop.add(new LCIODriver(outputFilePath));
275         }
276 
277         // Run job over the input events to generate clusters, check their validity, and fill plots.
278         //long startNanos = System.nanoTime();
279         loop.loop(nEvents, null);
280         //long elapsedMillis = (System.nanoTime() - startNanos) / 1000000;
281         //System.out.println(config.clustererName + " took " + elapsedMillis + "ms for " + loop.getTotalSupplied()
282         //        + " events which is " + (double) loop.getTotalSupplied() / (((double) elapsedMillis) / 1000.) + " events/s");
283         loop.dispose();
284 
285         /***************************************************
286          * Reread the file and run the check Driver again. *
287          ***************************************************/
288         File clusterFile = new File(testOutputDir.getPath() + File.separator + config.clustererName + ".slcio");
289         System.out.println("rereading " + clusterFile.getPath() + "...");
290         loop = new LCSimLoop();
291         loop.setLCIORecordSource(clusterFile);
292         loop.add(new ClusterCheckDriver(config));
293         loop.loop(nEvents);
294         loop.dispose();
295 
296     }

298     /**
299      * Driver that performs test assertions on Cluster collections.
300      */
301     static class ClusterCheckDriver extends Driver {
302 
303         ClustererTestConfig config;
304 
305         ClusterCheckDriver(ClustererTestConfig config) {
306             this.config = config;
307         }
308 
309         public void process(EventHeader event) {
310             List<Cluster> clusters = event.get(Cluster.class, config.getClusterCollectionName());
311             for (Cluster cluster : clusters) {
312                 // Test assertions.
313                 checkCluster(cluster);
314             }
315         }
316 
317         /**
318          * @param cluster
319          */
320         private void checkCluster(Cluster cluster) {
321             assertTrue("The cluster energy is invalid.", cluster.getEnergy() > 0.);
322             assertTrue("The cluster has no hits.", !cluster.getCalorimeterHits().isEmpty());
323             if (config.checkSeedHit) {
324                 assertEquals("First hit is not seed.", cluster.getCalorimeterHits().get(0), ClusterUtilities.findHighestEnergyHit(cluster));
325             }
326             if (config.checkPropCalc) {
327                 assertTrue("Cluster properties not calculated.", !((BaseCluster) cluster).needsPropertyCalculation());
328             }
329             if (config.clusterType != null) {
330                 assertEquals("Cluster type is not correct.", config.clusterType, ClusterType.getClusterType(cluster.getType()));
331             }
332             if (config.checkClusterPosition) {
333                 double[] position = cluster.getPosition();
334                 assertTrue("Position X is invalid.", Math.abs(position[0]) < 400. && position[0] != 0.);
335                 assertTrue("Position Y is invalid.", Math.abs(position[1]) > 25. && Math.abs(position[1]) < 90.);
336                 assertTrue("Position Z is invalid.", position[2] > 1385. && position[2] < 1480.);
337             }
338             for (CalorimeterHit hit : cluster.getCalorimeterHits()) {
339                 if (config.checkNullHits) {
340                     TestCase.assertNotNull("A hit from the cluster points to null.", hit);
341                 }
342                 if (hit != null) {
343                     if (config.checkHitPositions) {
344                         double[] hitPosition = hit.getPosition();
345                         assertTrue("Hit Position X is invalid.", hitPosition[0] != 0.);
346                         assertTrue("Hit Position Y is invalid.", hitPosition[1] != 0.);
347                         assertTrue("Hit Position Z is invalid.", hitPosition[2] != 0.);
348                         EcalCrystal crystal = CalorimeterHitUtilities.findCrystal(hit);
349                         assertTrue("Hit does not link correctly to geometry.", crystal != null);
350                     }
351                     if (config.checkHitEnergy) {
352                         assertTrue("Hit energy " + hit.getCorrectedEnergy() + " is <= 0.", hit.getCorrectedEnergy() > 0.);
353                     }
354                     if (config.checkHitTime) {
355                         double time = hit.getTime();
356                         assertTrue("Hit time is invalid.", time != 0.);
357                     }
358                 }
359             }
360         }
361     }

427             latestHitTimeH1D = aida.histogram1D(clusterCollectionName + "/Latest Hit Time", 500, 0., 500.);

431         public void process(EventHeader event) {
432             List<Cluster> clusters = event.get(Cluster.class, this.clusterCollectionName);
433             for (Cluster cluster : clusters) {
434 
435                 // Fill plots.
436                 fillClusterPlots(cluster);
437             }
438             countH1D.fill(clusters.size());
439             clusterVsHitCountC2d.fill(clusters.size(), ClusterUtilities.getHits(clusters).size());
440         }
441 
442         /**
443          * @param cluster
444          */
445         private void fillClusterPlots(Cluster cluster) {
446             energyH1D.fill(cluster.getEnergy());
447             double rawEnergy = ClusterUtilities.computeRawEnergy(cluster);
448             uncorrectedEnergyH1D.fill(rawEnergy);
449             sizeH1D.fill(cluster.getCalorimeterHits().size());
450             rawVsCorrectedH2D.fill(rawEnergy, cluster.getEnergy());
451             for (CalorimeterHit hit : cluster.getCalorimeterHits()) {
452                 hitEnergyH1D.fill(hit.getCorrectedEnergy());
453             }
454             highestHitEnergyH1D.fill(ClusterUtilities.findHighestEnergyHit(cluster).getCorrectedEnergy());
455             positionXH1D.fill(Math.abs(cluster.getPosition()[0]));
456             positionYH1D.fill(Math.abs(cluster.getPosition()[1]));
457             positionZH1D.fill(Math.abs(cluster.getPosition()[2]));
458             if (cluster.getShape() != null) {
459                 shapeParam1H1D.fill(cluster.getShape()[0]);
460                 shapeParam2H1D.fill(cluster.getShape()[1]);
461                 shapeParam3H1D.fill(cluster.getShape()[2]);
462             }
463             iphiC1D.fill(Math.toDegrees(cluster.getIPhi()));
464             ithetaC1D.fill(Math.toDegrees(cluster.getITheta()));
465             clusterPositionC2D.fill(cluster.getPosition()[0], cluster.getPosition()[1]);
466 
467             // Map<MCParticle, List<SimCalorimeterHit>> particleHitMap = ClusterUtilities.createParticleHitMap(cluster);
468 
469             Set<MCParticle> particles = ClusterUtilities.findMCParticles(cluster);
470             double particleEnergy = 0;
471             double highestParticleEnergy = Double.MIN_VALUE;
472             for (MCParticle particle : particles) {
473                 particleEnergy += particle.getEnergy();
474                 if (particle.getEnergy() > highestParticleEnergy) {
475                     highestParticleEnergy = particle.getEnergy();
476                 }
477             }
478             particleVsClusterEnergyH2D.fill(particleEnergy, cluster.getEnergy());
479             particleMinusClusterEnergyH1D.fill(particleEnergy - cluster.getEnergy());
480             nParticlesC1D.fill(particles.size());
481             particleEnergyC1D.fill(particleEnergy);
482             highestParticleEnergyVsClusterEnergyC2D.fill(highestParticleEnergy, cluster.getEnergy());
483 
484             List<CalorimeterHit> hitsForTime = new ArrayList<CalorimeterHit>();
485             hitsForTime.addAll(cluster.getCalorimeterHits());
486             Collections.sort(hitsForTime, new CalorimeterHit.TimeComparator());
487             earliestHitTimeH1D.fill(hitsForTime.get(0).getTime());
488             latestHitTimeH1D.fill(hitsForTime.get(hitsForTime.size() - 1).getTime());
489         }
