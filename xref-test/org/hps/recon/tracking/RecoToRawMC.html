21      protected String testOutputFileName = "raw_" + testInputFileName;
22      protected String testURLBase = "http://www.lcsim.org/test/hps-java";
23      protected long nEvents = -1;
24  
25      public void testClearMC() throws Exception {
26  
27          File inputFile = null;
28          if (testURLBase == null) {
29              inputFile = new File(testInputFileName);
30          } else {
31              URL testURL = new URL(testURLBase + "/" + testInputFileName);
32              FileCache cache = new FileCache();
33              inputFile = cache.getCachedFile(testURL);
34          }
35  
36          File outputFile = new TestOutputFile(testOutputFileName);
37          outputFile.getParentFile().mkdirs();
38  
39          LCSimLoop loop = new LCSimLoop();
40          loop.setLCIORecordSource(inputFile);
41  
42          loop.add(new RecoToRawMCDriver());
43          loop.add(new LCIODriver(outputFile));
44          try {
45              loop.loop(nEvents);
46          } catch (Exception e) {
47              System.out.println(e.toString());
48          }
49          loop.dispose();
50      }
