20  /**
21   * Test class to check truth particle and propagated position.
22   */
23  public class TruthResidualTest extends ReconTestSkeleton {
24  
25      static {
26          System.setProperty("hep.aida.IAnalysisFactory", "hep.aida.ref.BatchAnalysisFactory");
27      }
28  
29      static final String inputFileName = "ap_prompt_raw.slcio";
30      private AIDA aida;
31      private static final double maxResMean = 0.05; //in mm 
32      private static final double maxResRMS = 0.05; //in mm 
33  
34      public void testRecon() throws Exception {
35  
36          testInputFileName = inputFileName;
37          aida = AIDA.defaultInstance();
38  
39          String aidaOutputName = "target/test-output/TestResiduals_" + inputFileName.replaceAll("slcio", "aida");
40          nEvents = 500;
41          testTrackingDriver = new TestResiduals();
42          ((TestResiduals) testTrackingDriver).setOutputPlots(aidaOutputName);
43          ((TestResiduals) testTrackingDriver).aida = aida;
44          super.testRecon();
45  
46          IHistogram1D hx1 = aida.histogram1D("dres_truthsimhit_layer1_x");
47          if (hx1 != null && hx1.entries() > 10) {
48              System.out.printf("Layer 1 x truth hit residual has mean %f , RMS %f \n", hx1.mean(), hx1.rms());
49              assertTrue("Mean of layer 1 x truth hit residual is not zero " + hx1.mean(), Math.abs(hx1.mean()) < maxResMean);
50              assertTrue("RMS of layer 1 x truth hit residual is not zero" + hx1.rms(), Math.abs(hx1.rms()) < maxResRMS);
51          }
52  
53          IHistogram1D hy1 = aida.histogram1D("dres_truthsimhit_layer1_y");
54          if (hy1 != null && hy1.entries() > 10) {
55              System.out.printf("Layer 1 y truth hit residual has mean %f , RMS %f \n", hy1.mean(), hy1.rms());
56              assertTrue("Mean of layer 1 y truth hit residual is not zero " + hy1.mean(), Math.abs(hy1.mean()) < maxResMean);
57              assertTrue("RMS of layer 1 y truth hit residual is not zero" + hy1.rms(), Math.abs(hy1.rms()) < maxResRMS);
58          }
59  
60      }
61  
62      static class TestResiduals extends Driver {
63  
64          private TruthResiduals truthRes;
65          public AIDA aida = null;
66          private String outputPlots = null;
67  
68          public void setOutputPlots(String name) {
69              outputPlots = name;
70          }
71  
72          @Override
73          public void detectorChanged(Detector detector) {
74              if (aida == null)
75                  aida = AIDA.defaultInstance();
76              Hep3Vector bfield = detector.getFieldMap().getField(new BasicHep3Vector(0., 0., 1.));
77              truthRes = new TruthResiduals(bfield);
78              truthRes.setHideFrame(true);
79          }
80  
81          @Override
82          protected void process(EventHeader event) {
83  
84              super.process(event);
85  
86              List<MCParticle> mcParticles = null;
87              if (event.hasCollection(MCParticle.class, "MCParticle")) {
88                  mcParticles = event.get(MCParticle.class, "MCParticle");
89              }
90  
91              List<SimTrackerHit> simTrackerHits = event.get(SimTrackerHit.class, "TrackerHits");
92  
93              if (simTrackerHits != null && mcParticles != null) {
94                  truthRes.processSim(mcParticles, simTrackerHits);
95              }
96  
97          }
98  
99          public void endOfData() {
100             if (outputPlots != null) {
101                 try {
102                     aida.saveAs(outputPlots);
103                 } catch (IOException ex) {
104                     Logger.getLogger(TestResiduals.class.getName()).log(Level.SEVERE, null, ex);
105                 }
106             }
107         }
108 
109     }
