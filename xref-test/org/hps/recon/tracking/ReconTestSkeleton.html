35      public void testRecon() throws Exception {
36  
37          File inputFile = TestUtil.downloadTestFile(testInputFileName);
38  
39          testOutputFileName = "RecoTest_" + testInputFileName;
40          File outputFile = new TestOutputFile(testOutputFileName);
41          outputFile.getParentFile().mkdirs();
42          boolean loop1Success = true;
43  
44          LCSimLoop loop = new LCSimLoop();
45          loop.setLCIORecordSource(inputFile);
46  
47          final DatabaseConditionsManager manager = DatabaseConditionsManager.getInstance();
48          manager.addConditionsListener(new SvtDetectorSetup());
49  
50          loop.add(new MainTrackingDriver());
51          loop.add(new LCIODriver(outputFile));
52          try {
53              loop.loop(nEvents);
54          } catch (Exception e) {
55              System.out.println("Failure of main tracking sequence with following exception");
56              System.out.println(e.toString());
57              loop1Success = false;
58          }
59          loop.dispose();
60  
61          if (loop1Success && (testTrackingDriver != null)) {
62              LCSimLoop loop2 = new LCSimLoop();
63              loop2.add(testTrackingDriver);
64              loop2.setLCIORecordSource(outputFile);
65              try {
66                  loop2.loop(nEvents);
67              } catch (Exception e) {
68                  System.out.println("Failure of testing sequence with following exception");
69                  System.out.println(e.toString());
70              }
71              loop2.dispose();
72          }
73      }
