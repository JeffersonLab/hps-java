23  /**
24   * Test of reading EPICs scalar data from EVIO and writing it out to LCIO.
25   */
26  public class EpicsDataTest extends TestCase {
27  
28      public void test() throws IOException {
29  
30          // Cache input data file.
31          final File inputFile = TestUtil.downloadTestFile("hpsecal_004469_1000_events.evio.0");
32  
33          // Setup conditions and event building.
34          final DatabaseConditionsManager manager = DatabaseConditionsManager.getInstance();
35          final LCSimEventBuilder builder = new LCSimEngRunEventBuilder();
36          manager.addConditionsListener(builder);
37  
38          // Setup and run the loop to write out LCIO events with the EPICS scalar data.
39          final CompositeLoopConfiguration configuration = new CompositeLoopConfiguration();
40          configuration.setDataSourceType(DataSourceType.EVIO_FILE);
41          configuration.add(new EvioDetectorConditionsProcessor("HPS-ECalCommissioning-v2"));
42          configuration.setLCSimEventBuilder(builder);
43          configuration.setFilePath(inputFile.getPath());
44          configuration.setProcessingStage(ProcessingStage.LCIO);
45          configuration.setStopOnEndRun(false);
46          configuration.setStopOnErrors(false);
47          final File outputFile = new TestOutputFile("EpicsDataTest.slcio");
48          configuration.add(new LCIODriver(outputFile.getPath()));
49          final CompositeLoop loop = new CompositeLoop();
50          loop.setConfiguration(configuration);
51          loop.loop(100);
52  
53          // Read back and print out the scalar data.
54          final LCSimLoop readLoop = new LCSimLoop();
55          readLoop.setLCIORecordSource(outputFile);
56          readLoop.add(new Driver() {
57              @Override
58              public void process(final EventHeader event) {
59                  final EpicsData data = EpicsData.read(event);
60                  if (data != null) {
61                      System.out.println("read back EPICS data ...");
62                      System.out.println(data.toString());
63                  }
64              }
65          });
66          readLoop.loop(-1);
67      }
