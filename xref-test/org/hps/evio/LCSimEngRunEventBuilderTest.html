21   * Test conversion of EVIO to LCIO for Engineering Run EVIO data. This is ECAL
22   * data only for now.
23   */
24  public class LCSimEngRunEventBuilderTest extends TestCase {
25  
26      public void testLCSimEngRunEventBuilder() throws Exception {
27  
28          // Setup database conditions.
29          DatabaseConditionsManager conditionsManager = DatabaseConditionsManager.getInstance();
30          conditionsManager.setDetector("HPS-Proposal2014-v8-6pt6", 2000);
31  
32          // Configure LCIO writer.
33          new TestOutputFile(getClass().getSimpleName()).mkdirs();
34          File lcioFile = new TestOutputFile(getClass().getSimpleName() + File.separator + getClass().getSimpleName() + "_output.slcio");
35          LCIOWriter writer;
36          try {
37              writer = new LCIOWriter(lcioFile);
38          } catch (IOException e) {
39              throw new RuntimeException(e);
40          }
41  
42          // Create event builder.
43          LCSimEventBuilder builder = new LCSimEngRunEventBuilder();
44          conditionsManager.addConditionsListener(builder);
45          //builder.setDetectorName("HPS-Proposal2014-v8-6pt6");
46          conditionsManager.setDetector("HPS-Proposal2014-v8-6pt6", 2744);
47  
48          // Get remote test file.
49          final File evioFile = TestUtil.downloadTestFile("hps_002744.evio.0");
50  
51          // Open the EVIO reader.
52          System.out.println("Opening file " + evioFile);
53          EvioReader reader = null;
54          try {
55              reader = new EvioReader(evioFile);
56          } catch (Exception e) {
57              throw new RuntimeException(e);
58          }
59  
60          // Run the event builder on the EVIO.
61          EvioEvent evioEvent = null;
62          while ((evioEvent = reader.nextEvent()) != null) {
63              reader.parseEvent(evioEvent);
64              builder.readEvioEvent(evioEvent);
65              if (EvioEventUtilities.isPhysicsEvent(evioEvent)) {
66                  try {
67                      EventHeader lcsimEvent = builder.makeLCSimEvent(evioEvent);
68                      System.out.println("created LCSim event #" + lcsimEvent.getEventNumber());
69                      writer.write(lcsimEvent);
70                  } catch (Exception e) {
71                      e.printStackTrace();
72                  }
73              }
74          }
75  
76          // Close the LCIO writer.
77          writer.flush();
78          writer.close();
79      }
80  }
