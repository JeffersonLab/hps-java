12  public class HodoscopeConditionsTest extends TestCase {
13      
14      private static final String DETECTOR = "HPS-HodoscopeTest-v1";
15      private static final int RUN = 1000000;
16      
17      public void testHodoscopeConditions() {
18          
19          final DatabaseConditionsManager mgr = DatabaseConditionsManager.getInstance();
20          try {
21              mgr.setDetector(DETECTOR, RUN);
22          } catch (ConditionsNotFoundException e) {
23              throw new RuntimeException(e);
24          }
25          
26          final HodoscopeChannelCollection channels = 
27                  mgr.getCachedConditions(HodoscopeChannelCollection.class, "hodo_channels").getCachedData();
28          System.out.println("Printing Hodoscope channels ...");
29          for (HodoscopeChannel channel : channels) {
30              System.out.println(channel);
31          }
32       
33          final HodoscopeCalibrationCollection calibrations =
34                  mgr.getCachedConditions(HodoscopeCalibrationCollection.class, "hodo_calibrations").getCachedData();
35          System.out.println("Printing Hodoscope calibrations ...");
36          for (HodoscopeCalibration calibration : calibrations) {
37              System.out.println(calibration);
38          }
39          
40          final HodoscopeGainCollection gains = 
41                  mgr.getCachedConditions(HodoscopeGainCollection.class, "hodo_gains").getCachedData();
42          System.out.println("Printing Hodoscope gains ...");
43          for (HodoscopeGain gain : gains) {
44              System.out.println(gain);
45          }
46          
47          final HodoscopeTimeShiftCollection timeShifts = 
48                  mgr.getCachedConditions(HodoscopeTimeShiftCollection.class, "hodo_time_shifts").getCachedData();
49          for (HodoscopeTimeShift timeShift : timeShifts) {
50              System.out.println(timeShift);
51          }
52          
53          System.out.println("Printing Hodoscope channel constants ...");
54          final HodoscopeConditions conditions =
55                  mgr.getCachedConditions(HodoscopeConditions.class, "hodo_conditions").getCachedData();
56          for (HodoscopeChannel channel : conditions.getChannels()) {
57              HodoscopeChannelConstants channelData = conditions.getChannelConstants(channel);
58              System.out.println("Channel constants for ID " + channel.getChannelId());
59              System.out.println(channelData.getCalibration());
60              System.out.println(channelData.getGain());
61              System.out.println(channelData.getTimeShift());
62              System.out.println();
63          }
64      }
