110     /**
111      * Perform a run-by-run comparison between the conditions values and the data map.
112      * @throws ConditionsNotFoundException If there is a problem initializing the conditions system
113      */
114     public void testBeamPositions() throws ConditionsNotFoundException {
115         final DatabaseConditionsManager mgr = DatabaseConditionsManager.getInstance();
116         for (Entry<Integer, double[]> entry : this.beamPositionMap.entrySet()) {
117             final Integer runNumber = entry.getKey();
118             final double[] beamPositionArr = entry.getValue();
119             mgr.setDetector("HPS-dummy-detector", runNumber);
120             final BeamPositionCollection beamPositionCond = 
121                     mgr.getCachedConditions(BeamPositionCollection.class, "beam_positions").getCachedData();
122             final BeamPosition beamPosition = beamPositionCond.get(0);
123             System.out.println("Testing run " + runNumber);
124             assertEquals("Beam Z position does not match.", beamPositionArr[0], beamPosition.getPositionZ());
125             assertEquals("Beam X position does not match.", beamPositionArr[1], beamPosition.getPositionX());
126             assertEquals("Beam Y position does not match.", beamPositionArr[2], beamPosition.getPositionY());
127         }
128     }
