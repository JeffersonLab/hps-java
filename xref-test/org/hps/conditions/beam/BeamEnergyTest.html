3   import junit.framework.TestCase;

5   import org.hps.conditions.beam.BeamEnergy.BeamEnergyCollection;
6   import org.hps.conditions.database.DatabaseConditionsManager;

8   /**
9    * Basic test of {@link BeamEnergy} condition.
10   */
11  public class BeamEnergyTest extends TestCase {
12      
13      private static final double[] BEAM_ENERGIES = {1.920, 1.056, 2.306};
14      private static final int[][] RUNS = {{3000, 3500, 3999}, {4000, 6000, 6999}, {7000, 7500, 9999}};
15      private static final String DETECTOR = "HPS-dummy-detector";
16      
17      public void testBeamEnergy() throws Exception {
18          DatabaseConditionsManager manager = DatabaseConditionsManager.getInstance();        
19          for (int i = 0; i < BEAM_ENERGIES.length; i++) {
20              double expectedBeamEnergy = BEAM_ENERGIES[i];
21              for (int j = 0; j < 3; j++) {
22                  int run = RUNS[i][j];
23                  manager.setDetector(DETECTOR, run);
24                  BeamEnergyCollection beamEnergyCollection = 
25                          manager.getCachedConditions(BeamEnergyCollection.class, "beam_energies").getCachedData();        
26                  double beamEnergy = beamEnergyCollection.get(0).getBeamEnergy();
27                  System.out.println("read beam energy " + beamEnergy + " for run " + run);
28                  assertEquals("Beam energy has wrong value.", expectedBeamEnergy, beamEnergy);
29              }
30          }
31      }   
32  }
