17       * Just use a dummy detector.

20  
21      /**
22       * This is a list of run start values to check.
23       */
24      private static final int[] RUNS = new int[] {4871, 5038, 5076, 5139, 5174, 5218, 5236, 5251, 5263, 5299, 5310,
25              5375, 5400, 5533, 5558, 5575, 5596, 5601, 5603, 5610, 4871, 5038, 5076, 5139, 5174, 5218, 5236, 5251, 5263,
26              5299, 5310, 5375, 5400, 5533, 5558, 5575, 5596, 5601, 5603, 5610, 5640, 5641, 5642, 5686, 5722, 5779};
27  
28      /**
29       * Load SVT timing constants and print them out by run range.
30       *
31       * @throws Exception if any error occurs
32       */
33      public void testSvtTimingConstants() throws Exception {
34          final DatabaseConditionsManager manager = DatabaseConditionsManager.getInstance();
35          // manager.setConnectionResource("/org/hps/conditions/config/jeremym_dev_connection.prop");
36          for (final int run : RUNS) {
37              manager.setDetector(DETECTOR, run);
38              final SvtTimingConstantsCollection collection = manager.getCachedConditions(
39                      SvtTimingConstantsCollection.class, "svt_timing_constants").getCachedData();
40              final ConditionsRecord condi = manager.findConditionsRecords("svt_timing_constants").get(0);
41              System.out.println("run_start: " + condi.getRunStart() + ", run_end: " + condi.getRunEnd()
42                      + ", offset_phase: " + collection.get(0).getOffsetPhase() + ", offset_time: "
43                      + collection.get(0).getOffsetTime());
44          }
45      }
46  }
