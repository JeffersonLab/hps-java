10  import org.hps.conditions.database.DatabaseConditionsManager;
11  import org.hps.util.test.TestUtil;
12  import org.lcsim.geometry.Detector;
13  import org.lcsim.util.Driver;
14  import org.lcsim.util.loop.LCSimLoop;
15  
16  import junit.framework.TestCase;
17  
18  /**
19   * This class checks that event processing works correctly for files that have multiple runs in them.
20   */
21  public final class RunNumberTest extends TestCase {
22  
23      /**
24       * Simple Driver to store information about runs processed.
25       */
26      static class RunNumberDriver extends Driver {
27  
28          /**
29           * Reference to conditions manager.
30           */
31          private static DatabaseConditionsManager conditionsManager = DatabaseConditionsManager.getInstance();
32  
33          /**
34           * Number of times {@link #detectorChanged(Detector)} was called.
35           */
36          private int nDetectorChanged = 0;
37  
38          /**
39           * Number of runs processed.
40           */
41          private int nRuns = 0;
42  
43          /**
44           * List of run numbers processed.
45           */
46          private final List<Integer> runsProcessed = new ArrayList<Integer>();
47  
48          /**
49           * Set of unique detector objects.
50           */
51          private final Set<Detector> uniqueDetectorObjects = new HashSet<Detector>();
52  
53          /**
54           * Set of unique run numbers processed.
55           */
56          private final Set<Integer> uniqueRuns = new LinkedHashSet<Integer>();
57  
58          /**
59           * Hook for conditions changed used to test multiple run processing.
60           */
61          @Override
62          public void detectorChanged(final Detector detector) {
63              final int run = conditionsManager.getRun();
64              this.uniqueRuns.add(run);
65              this.runsProcessed.add(run);
66              ++this.nRuns;
67              ++this.nDetectorChanged;
68              this.uniqueDetectorObjects.add(detector);
69          }
70      }
71  
72      /**
73       * Number of runs that should be processed in the job.
74       */
75      private static final int RUN_COUNT = 9;
76  
77      /**
78       * Run the test.
79       *
80       * @throws Exception if there is a test error
81       */
82      public void test() throws Exception {
83  
84          final File testFile = TestUtil.downloadTestFile("ConditionsTest.slcio");
85  
86          // Create the record loop.
87          final LCSimLoop loop = new LCSimLoop();
88  
89          // Configure the loop.
90          loop.setLCIORecordSource(testFile);
91          final RunNumberDriver runNumberDriver = new RunNumberDriver();
92          loop.add(runNumberDriver);
93  
94          // Run over all events.
95          loop.loop(-1, null);
96  
97          System.out.println("Done processing events!");
98  
99          // Print out unique runs.
100         System.out.println("Unique run numbers in this job ...");
101         for (final int runNumber : runNumberDriver.uniqueRuns) {
102             System.out.println(runNumber);
103         }
104 
105         System.out.println();
106 
107         // Print out runs processed.
108         System.out.println("Processed runs in order ...");
109         for (final int runNumber : runNumberDriver.runsProcessed) {
110             System.out.println(runNumber);
111         }
112 
113         // Check that correct number of runs was processed.
114         assertEquals("Number of runs processed was incorrect.", RUN_COUNT, runNumberDriver.nRuns);
115 
116         // Check that the number of unique runs was correct.
117         assertEquals("Number of unique runs was incorrect.", RUN_COUNT, runNumberDriver.uniqueRuns.size());
118 
119         // Check that detectorChanged was called the correct number of times.
120         assertEquals("The detectorChanged method was called an incorrect number of times.", RUN_COUNT,
121                 runNumberDriver.nDetectorChanged);
122 
123         // Check that there was a unique Detector created for each run.
124         assertEquals("The number of Detector objects created was not correct.", RUN_COUNT,
125                 runNumberDriver.uniqueDetectorObjects.size());
126     }
127 }
