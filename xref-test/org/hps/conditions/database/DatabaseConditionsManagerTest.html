17      
18      /**
19       * Perform basic tests of the database conditions manager.
20       * 
21       * @throws Exception if any error is thrown
22       */
23      public void testDatabaseConditionsManager() throws Exception {
24          
25          DatabaseConditionsManager manager = DatabaseConditionsManager.getInstance();
26          
27          // Check initial state.
28          TestCase.assertTrue("The conditions manager instance is null.", manager != null);
29          TestCase.assertFalse("The manager should not be initialized.", manager.isInitialized());
30          TestCase.assertFalse("The manager should not be frozen.", manager.isFrozen());        
31          TestCase.assertTrue("The manager should be setup.", ConditionsManager.isSetup());
32          
33          // Open database connection.
34          manager.openConnection();
35                  
36          // Check that a new collection can be created.
37          EcalCalibrationCollection newCollection = manager.newCollection(EcalCalibrationCollection.class, "ecal_calibrations");
38          TestCase.assertNotNull("New collection should have metadata.", newCollection.getTableMetaData());
39                  
40          // Check connection state.
41          TestCase.assertTrue("The connection should not be closed.", !manager.getConnection().isClosed());
42          TestCase.assertNotNull("The connection is null.", manager.getConnection());
43                  
44          // Initialize the conditions system.
45          manager.setDetector("HPS-EngRun2015-Nominal-v2", 5772);
46          
47          // Check basic state after initialization.
48          TestCase.assertTrue("The manager should be initialized.", manager.isInitialized());
49          
50          // Make sure that freezing the conditions system works properly.
51          manager.freeze();        
52          TestCase.assertTrue("The manager should be frozen.", manager.isFrozen());
53          manager.setDetector("HPS-EngRun2015-Nominal-v2", 1234);
54          TestCase.assertEquals("The run number should be the same because system was frozen.", 5772, manager.getRun());
55                 
56          // Load a test collection.
57          EcalCalibrationCollection testCollection = manager.getCachedConditions(EcalCalibrationCollection.class, "ecal_calibrations").getCachedData();
58          TestCase.assertTrue("The test collection should not be empty.", testCollection.size() > 0);
59          
60          // Load a conditions series.
61          ConditionsSeries<?, ?> series = manager.getConditionsSeries(EcalCalibrationCollection.class, "ecal_calibrations");
62          TestCase.assertTrue("The conditions series should not be empty.", !series.isEmpty());
63                                                 
64          // Set a tag and check that it seems to work properly (not very detailed).
65          TestCase.assertFalse("There should be available tags.", manager.getAvailableTags().isEmpty());
66          TestCase.assertTrue("The pass1 tag should be available.", manager.getAvailableTags().contains("pass1"));        
67          manager.addTag("pass1");
68          TestCase.assertFalse("There should be an active tag.", manager.getActiveTags().isEmpty());
69          manager.setDetector("HPS-EngRun2015-Nominal-v2", 5772);
70          SvtTimingConstantsCollection testTagCollection = manager.getCachedConditions(SvtTimingConstantsCollection.class, "svt_timing_constants").getCachedData();
71          TestCase.assertTrue("The collection should not be empty.", testTagCollection.size() > 0);                        
72          manager.clearTags();
73          TestCase.assertTrue("The tags should be cleared.", manager.getActiveTags().isEmpty());
74          
75          // See that some conditions records are available.
76          ConditionsRecordCollection conditionsRecords = manager.getConditionsRecords();
77          TestCase.assertTrue("Conditions records should have been found.", conditionsRecords.size() > 0);               
78          conditionsRecords = manager.findConditionsRecords("ecal_calibrations");
79          TestCase.assertTrue("A conditions record should have been found.", conditionsRecords.size() > 0);        
80          TestCase.assertTrue("A conditions record should exist.", manager.hasConditionsRecord("ecal_calibrations"));
81          TestCase.assertNotNull("Table meta data should exist for collection.", manager.findTableMetaData("ecal_calibrations"));
82          
83          // Un-freeze the conditions system.
84          manager.unfreeze();
85          TestCase.assertFalse("Manager should not be frozen.", manager.isFrozen());
86          
87          // Check SLAC connection setup.
88          //manager.closeConnection();
89          //manager.setConnectionResource("/org/hps/conditions/config/slac_connection.prop");
90          //manager.openConnection();
91          //TestCase.assertTrue("Connection should be slac host.", manager.getConnection().getMetaData().getURL().contains("slac"));
92          
93          // Check JLAB connection setup.
94          //manager.closeConnection();
95          //manager.setConnectionResource("/org/hps/conditions/config/jlab_connection.prop");
96          //manager.openConnection();
97          //TestCase.assertTrue("Connection should be slac host.", manager.getConnection().getMetaData().getURL().contains("jlab"));
98          
99          manager.closeConnection();
100     }
