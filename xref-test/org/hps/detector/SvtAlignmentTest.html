1   package org.hps.detector;

3   import junit.framework.TestCase;
4   
5   import org.hps.conditions.database.DatabaseConditionsManager;
6   import org.hps.conditions.svt.SvtAlignmentConstant;
7   import org.hps.conditions.svt.SvtAlignmentConstant.SvtAlignmentConstantCollection;
8   
9   /**
10   * Test loading SVT alignment constants into the Java detector model.
11   */
12  public class SvtAlignmentTest extends TestCase {
13  
14      private static final String DETECTOR_NAME = "HPS-EngRun2015-Nominal-v1";
15      
16      private static final int[] RUNS = {
17          /* 4mm */
18          4847,
19          /* 3mm */
20          5037,
21          /* 2mm */
22          5066,
23          /* 1.5mm */
24          5259,
25          /* 0.5mm a.k.a. nominal */
26          5632
27          };
28          
29      private static double ANGLES[][] = {
30          /* 4mm */
31          {0.021401105301460045, -0.020891926880057254},
32          /* 3mm */
33          {0.00746162548005678, -0.007499693772411655},
34          /* 2mm */
35          {0.00439094334909705, -0.004185830909531962},
36          /* 1.5mm */
37          {0.003068520524453775, -0.0033041356336028917},
38          /* 0.5mm a.k.a. nominal */
39          {0, 0}
40      };
41  
42      public void testSvtAlignment() throws Exception {
43          final DatabaseConditionsManager manager = DatabaseConditionsManager.getInstance();
44          int runIndex = 0;
45          for (int run : RUNS) {
46              
47              System.out.println();            
48              System.out.println("loading run " + run + " SVT alignments for detector " + DETECTOR_NAME + " ...");
49              
50              manager.setDetector(DETECTOR_NAME, run);
51              
52              SvtAlignmentConstantCollection collection = 
53                      manager.getCachedConditions(SvtAlignmentConstantCollection.class, "svt_alignments").getCachedData();
54              System.out.println("got collection " + collection.getCollectionId() + " of size " + collection.size());
55              
56              SvtAlignmentConstant top = collection.find(13100);
57              SvtAlignmentConstant bottom = collection.find(23100);
58              
59              System.out.println("param " + top.getParameter() + " = " + top.getValue());
60              System.out.println("param " + bottom.getParameter() + " = " + bottom.getValue());
61              
62              assertEquals("Top angle wrong.", ANGLES[runIndex][0], top.getValue());
63              assertEquals("Bottom angle wrong.", ANGLES[runIndex][1], bottom.getValue());
64              ++runIndex;
65          }
66      }
67  }
