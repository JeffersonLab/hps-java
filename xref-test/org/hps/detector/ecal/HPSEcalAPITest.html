24      public void setUp() throws Exception {
25          final GeometryReader geometryReader = new GeometryReader();
26          final InputStream in = this.getClass().getResourceAsStream("/org/lcsim/geometry/subdetector/HPSEcal3Test.xml");
27          final Detector detector = geometryReader.read(in);
28          ecal = (HPSEcal3) detector.getSubdetector("Ecal");
29          api = (HPSEcalDetectorElement) ecal.getDetectorElement();
30      }
31      
32      public void testIdentifiers() {
33          IIdentifierHelper helper = api.getIdentifierHelper();
34          for (EcalCrystal crystal : api.getCrystals()) {
35              IExpandedIdentifier expandedId = helper.createExpandedIdentifier();
36              expandedId.setValue(helper.getFieldIndex("ix"), crystal.getX());
37              expandedId.setValue(helper.getFieldIndex("iy"), crystal.getY());
38              expandedId.setValue(helper.getFieldIndex("system"), api.getSystemID());
39              IIdentifier id = helper.pack(expandedId);
40              if (id.getValue() != crystal.getIdentifier().getValue()) {
41                  throw new RuntimeException("Reencoded ID " + id.getValue() + " does not match crystal ID " + crystal.getIdentifier().getValue());
42              }            
43              
44              IDetectorElementStore deStore = DetectorElementStore.getInstance();           
45              if (deStore.find(crystal.getIdentifier()).size() == 0) {
46                  throw new RuntimeException("Failed to find crystal ID in store.");
47              } 
48              
49              if (deStore.find(id).size() == 0) {
50                  throw new RuntimeException("Failed to find repacked ID in store.");
51              } 
52          }        
53      }
54      
55      public void testNeighborMap() {
56          NeighborMap neighborMap = ecal.getNeighborMap();
57          for (EcalCrystal crystal : api.getCrystals()) {
58              List<EcalCrystal> neighborCrystals = api.getNeighbors(crystal);
59              if (neighborCrystals.size() == 0) {
60                  throw new RuntimeException("Crystal has 0 neighbors.");
61              }
62              if (!neighborMap.containsKey(crystal.getIdentifier().getValue())) {
63                  throw new RuntimeException("Neighbor map does not contain ID.");
64              }
65              Set<Long> neighborIds = neighborMap.get(crystal.getIdentifier().getValue());
66              for (long id : neighborIds) {
67                  boolean foundId = false;
68                  for (EcalCrystal neighborCrystal : neighborCrystals) {
69                      if (neighborCrystal.getIdentifier().getValue() == id) {
70                          foundId = true;
71                          break;
72                      } 
73                  }
74                  if (!foundId) {
75                      throw new RuntimeException("Failed to find neighbor ID in map for " + crystal.getName());
76                  }
77              }
78          }
79          
80      }

82      public void testHPSEcalAPI() throws Exception {
83  
84          assertEquals("The max X index is wrong.", 23, api.getXIndexMax());
85          assertEquals("The min X index is wrong.", -23, api.getXIndexMin());
86          assertEquals("The max Y index is wrong.", 5, api.getYIndexMax());
87          assertEquals("The min Y index is wrong.", -5, api.getYIndexMin());
88  
89          for (final Integer yIndex : api.getYIndices()) {
90              if (yIndex == 0) {
91                  continue;
92              }
93              for (final Integer xIndex : api.getXIndices()) {
94                  if (xIndex == 0) {
95                      continue;
96                  }
97                  if ((yIndex == 1 || yIndex == -1) && xIndex <= -2 && xIndex >= -10) {
98                      assertTrue("Indices should be in gap: " + xIndex + ", " + yIndex, api.isInBeamGap(xIndex, yIndex));
99                      // Crystal is in the beam gap.
100                     continue;
101                 }
102                 final EcalCrystal crystal = api.getCrystal(xIndex, yIndex);
103                 assertNotNull("Failed to find crystal at ix = " + xIndex + ", iy = " + yIndex, crystal);
104             }
105         }
106 
107         for (final Integer yIndex : api.getYIndices()) {
108             final List<EcalCrystal> row = api.getRow(yIndex);
109             if (Math.abs(yIndex) != 1) {
110                 assertEquals("Wrong number of crystals in row.", 46, row.size());
111             } else {
112                 assertEquals("Wrong number of crystals in row.", 37, row.size());
113             }
114         }
115 
116         for (final Integer xIndex : api.getXIndices()) {
117             final List<EcalCrystal> column = api.getColumn(xIndex);
118             if (xIndex > -2 || xIndex < -10) {
119                 assertEquals("Wrong number of crystals in column.", 10, column.size());
120             } else {
121                 assertEquals("Wrong number of crystals in column.", 8, column.size());
122             }
123         }
