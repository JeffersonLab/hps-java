148         @Override
149         protected void process(EventHeader event) {
150 
151             List<SimTrackerHit> trackerHits = null;
152             if (event.hasCollection(SimTrackerHit.class, "TrackerHits"))
153                 trackerHits = event.get(SimTrackerHit.class, "TrackerHits");
154             else
155                 return;
156 
157             List<MCParticle> particles = null;
158             if (event.hasCollection(MCParticle.class, "MCParticle"))
159                 particles = event.get(MCParticle.class, "MCParticle");  
160             else
161                 return;
162             List<SimTrackerHit> scoringHits = null;
163             if (event.hasCollection(SimTrackerHit.class, "TrackerHitsECal"))
164                 scoringHits = event.get(SimTrackerHit.class, "TrackerHitsECal");
165             else
166                 return;
167             
168             Map<MCParticle, List<SimTrackerHit>> scoringHitMap = new HashMap<MCParticle, List<SimTrackerHit>>();
169             for (SimTrackerHit scoringHit : scoringHits) {
170                 MCParticle part = scoringHit.getMCParticle();
171                 if (scoringHitMap.get(part) == null) 
172                     scoringHitMap.put(part, new ArrayList<SimTrackerHit>());
173                 scoringHitMap.get(part).add(scoringHit);
174             }         
175             
176             Map<MCParticle, List<SimTrackerHit>> trackerHitMap = MCFullDetectorTruth.BuildTrackerHitMap(trackerHits);
177            
178             for (MCParticle part : particles) {
179                 if (part.getCharge() == 0)
180                     continue;
181                 if (part.getGeneratorStatus() != MCParticle.FINAL_STATE)
182                     continue;
183                 if (!part.getSimulatorStatus().isDecayedInCalorimeter())
184                     continue;
185                 if (!trackerHitMap.containsKey(part) || (!scoringHitMap.containsKey(part)))
186                     continue;
187                 List<SimTrackerHit> hits = trackerHitMap.get(part);
188                 //List<SimCalorimeterHit> caloHits = calHitMap.get(part);
189                 List<SimTrackerHit> caloHitList = scoringHitMap.get(part);
190                 if ((hits == null) || (caloHitList == null) || (caloHitList.isEmpty()))
191                     continue;
192                 
193                 SimTrackerHit lastHit = null;
194                 int lay=0;
195                 for (SimTrackerHit hit : hits) {
196                     if (hit.getLayer() > lay) {
197                         lay=hit.getLayer();
198                         lastHit = hit;
199                     }
200                 }
201                 if (lastHit == null || lastHit.getPosition()[2] < 890)
202                     continue;
203                     
204                 SimTrackerHit caloHit = caloHitList.get(0);
205 
206                 Hep3Vector hitPosition = lastHit.getPositionVec();
207                 Hep3Vector hitMomentum = new BasicHep3Vector(lastHit.getMomentum());   
208                 Hep3Vector extrapPos = doTrackExtrapRK(hitPosition, hitMomentum, part.getCharge(), false);
209                 //System.out.printf("   end pos %s \n", extrapPos.toString());
210                 caloHit = findClosestCaloHit(caloHitList, extrapPos);
211                 Hep3Vector caloHitPos = caloHit.getPositionVec();
212                 //System.out.printf("   calo hit pos %s \n", caloHitPos.toString());
213                 Hep3Vector residualVec = VecOp.sub(extrapPos, caloHitPos);
214                 
215 
216                 aida.histogram2D("Extrapolated SimTrackerHit - SimCalorimeterHit: Y vs X").fill(residualVec.x(), residualVec.y());
217                 aida.histogram2D("[Extrapolated SimTrackerHit - SimCalorimeterHit] X vs SimCalorimeterHit X").fill(caloHitPos.x(), residualVec.x());
218                 aida.histogram2D("[Extrapolated SimTrackerHit - SimCalorimeterHit] X vs SimCalorimeterHit Y").fill(caloHitPos.y(), residualVec.x());
219                 aida.histogram2D("[Extrapolated SimTrackerHit - SimCalorimeterHit] Y vs SimCalorimeterHit X").fill(caloHitPos.x(), residualVec.y());
220                 aida.histogram2D("[Extrapolated SimTrackerHit - SimCalorimeterHit] Y vs SimCalorimeterHit Y").fill(caloHitPos.y(), residualVec.y());
221 
222                 aida.histogram2D("Extrapolated SimTrackerHit - SimCalorimeterHit Y vs MCParticle Pz").fill(Math.abs(part.getPZ()), residualVec.y());
223                 aida.histogram2D("Extrapolated SimTrackerHit - SimCalorimeterHit X vs MCParticle Pz").fill(Math.abs(part.getPZ()), residualVec.x());
224 
225                 if (lastHit.getPosition()[1] > 0) {
226                     aida.histogram2D("Top: Extrapolated SimTrackerHit - SimCalorimeterHit: Y vs X").fill(residualVec.x(), residualVec.y());
227                     aida.histogram2D("Top: Extrapolated SimTrackerHit - SimCalorimeterHit Y vs MCParticle Pz").fill(Math.abs(part.getPZ()), residualVec.y());
228                     aida.histogram2D("Top: Extrapolated SimTrackerHit - SimCalorimeterHit X vs MCParticle Pz").fill(Math.abs(part.getPZ()), residualVec.x());
229                 }
230                 else {
231                     aida.histogram2D("Bot: Extrapolated SimTrackerHit - SimCalorimeterHit: Y vs X").fill(residualVec.x(), residualVec.y());
232                     aida.histogram2D("Bot: Extrapolated SimTrackerHit - SimCalorimeterHit Y vs MCParticle Pz").fill(Math.abs(part.getPZ()), residualVec.y());
233                     aida.histogram2D("Bot: Extrapolated SimTrackerHit - SimCalorimeterHit X vs MCParticle Pz").fill(Math.abs(part.getPZ()), residualVec.x());
234                 }
235 
236                 if (part.getCharge() > 0) {
237                     aida.histogram2D("e+: Extrapolated SimTrackerHit - SimCalorimeterHit: Y vs X").fill(residualVec.x(), residualVec.y());
238                     aida.histogram2D("e+: Extrapolated SimTrackerHit - SimCalorimeterHit Y vs MCParticle Pz").fill(Math.abs(part.getPZ()), residualVec.y());
239                     aida.histogram2D("e+: Extrapolated SimTrackerHit - SimCalorimeterHit X vs MCParticle Pz").fill(Math.abs(part.getPZ()), residualVec.x());
240                 }
241                 else {
242                     aida.histogram2D("e-: Extrapolated SimTrackerHit - SimCalorimeterHit: Y vs X").fill(residualVec.x(), residualVec.y());
243                     aida.histogram2D("e-: Extrapolated SimTrackerHit - SimCalorimeterHit Y vs MCParticle Pz").fill(Math.abs(part.getPZ()), residualVec.y());
244                     aida.histogram2D("e-: Extrapolated SimTrackerHit - SimCalorimeterHit X vs MCParticle Pz").fill(Math.abs(part.getPZ()), residualVec.x());
245                 }
246 
247             }
248         }

250         private SimTrackerHit findClosestCaloHit(List<SimTrackerHit> caloHitList, Hep3Vector extrapPos) {
251             SimTrackerHit closestHit = caloHitList.get(0);
252             double res = VecOp.sub(extrapPos, closestHit.getPositionVec()).magnitude();
253             for (SimTrackerHit caloHit : caloHitList) {
254                 Hep3Vector residualVec = VecOp.sub(extrapPos, caloHit.getPositionVec());
255                 if (residualVec.magnitude() < res) {
256                     res = residualVec.magnitude();
257                     closestHit = caloHit;
258                 }
259             }
260             return closestHit;
261         }

264         public Hep3Vector doTrackExtrapRK(Hep3Vector currentPosition, Hep3Vector currentMomentum, double q, boolean debug) {
265             double distanceZ = ecalPosition - currentPosition.z();
266             double distance = distanceZ / VecOp.cosTheta(currentMomentum);
267                     
268             RK4integrator RKint = new RK4integrator(q, epsilon, bFieldMap);
269             Pair<Hep3Vector,Hep3Vector> RKresults = RKint.integrate(currentPosition, currentMomentum, distance);
270            // System.out.printf("RKpos %s \n", RKresults.getFirstElement().toString());
271           
272             Hep3Vector mom = RKresults.getSecondElement();            
273             double dz = ecalPosition - RKresults.getFirstElement().z();
274             Hep3Vector delta = new BasicHep3Vector(dz * mom.x() / mom.z(), dz * mom.y() / mom.z(), dz);
275             Hep3Vector finalPos = VecOp.add(delta, RKresults.getFirstElement());
276             
277             return finalPos;
278         }
