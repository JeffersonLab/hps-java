80      private void fill(IHistogram1D h1) {
81          int nEntries = 10000;
82          Random r = new Random();
83          for (int i = 0; i < 10 * nEntries; i++) {
84              if (i % 10 == 0) {
85                  System.out.println("Filling entry: " + i);
86              }
87              try {
88                  Thread.sleep(500);
89              } catch (InterruptedException e) {
90                  e.printStackTrace();
91              }
92              double xVal = r.nextGaussian();
93              double w = r.nextDouble();
94              h1.fill(xVal, w);
95          }
96      }

98      public void fill(IDataPointSet dps2D) {
99          int entries = 9999;
100         long waitTime = 1000;
101         Random r = new Random();
102         for (int i = 0; i < entries; i++) {
103             long curr = System.currentTimeMillis() / 1000;
104             dps2D.addPoint();
105             dps2D.point(i).coordinate(0).setValue(curr);
106             dps2D.point(i).coordinate(0).setErrorPlus(0);
107             dps2D.point(i).coordinate(1).setValue(r.nextGaussian());
108             dps2D.point(i).coordinate(1).setErrorPlus(0);
109             dps2D.point(i).coordinate(1).setErrorMinus(0);
110             // Simulate one second update interval
111             try {
112                 Thread.sleep(waitTime);
113             } catch (InterruptedException e) {
114                 e.printStackTrace();
115             }
116         }
